67     void TemplateSimplifier::cleanupAfterSimplify(Token *tokens)
68     {
69         bool goback = false;
70         for (Token *tok = tokens; tok; tok = tok->next()) {
71             if (goback) {
72                 tok = tok->previous();
73                 goback = false;
74             }
75             if (tok->str() == "(")
76                 tok = tok->link();
77     
78             else if (Token::Match(tok, "template < > %name%")) {
79                 const Token *end = tok;
80                 while (end) {
81                     if (end->str() == ";")
82                         break;
83                     if (end->str() == "{") {
84                         end = end->link()->next();
85                         break;
86                     }
87                     if (!Token::Match(end, "%name%|::|<|>|,")) {
88                         end = nullptr;
89                         break;
90                     }
91                     end = end->next();
92                 }
93                 if (end) {
94                     Token::eraseTokens(tok,end);
95                     tok->deleteThis();
96                 }
97             }
98     
99             else if (Token::Match(tok, "%type% <") &&
100                      (!tok->previous() || tok->previous()->str() == ";")) {
101                 const Token *tok2 = tok->tokAt(2);
102                 std::string type;
103                 while (Token::Match(tok2, "%type% ,") || Token::Match(tok2, "%num% ,")) {
104                     type += tok2->str() + ",";
105                     tok2 = tok2->tokAt(2);
106                 }
107                 if (Token::Match(tok2, "%type% > (") || Token::Match(tok2, "%num% > (")) {
108                     type += tok2->str();
109                     tok->str(tok->str() + "<" + type + ">");
110                     Token::eraseTokens(tok, tok2->tokAt(2));
111                     if (tok == tokens)
112                         goback = true;
113                 }
114             }
115         }
116     }

119     void TemplateSimplifier::checkComplicatedSyntaxErrorsInTemplates(const Token *tokens)
120     {
121         // check for more complicated syntax errors when using templates..
122         for (const Token *tok = tokens; tok; tok = tok->next()) {
123             // skip executing scopes (ticket #3183)..
124             if (Token::simpleMatch(tok, "( {")) {
125                 tok = tok->link();
126                 if (!tok)
127                     syntaxError(nullptr);
128             }
129             // skip executing scopes..
130             const Token *start = Tokenizer::startOfExecutableScope(tok);
131             if (start) {
132                 tok = start->link();
133             }
134     
135             // skip executing scopes (ticket #1985)..
136             else if (Token::simpleMatch(tok, "try {")) {
137                 tok = tok->next()->link();
138                 while (Token::simpleMatch(tok, "} catch (")) {
139                     tok = tok->linkAt(2);
140                     if (Token::simpleMatch(tok, ") {"))
141                         tok = tok->next()->link();
142                 }
143             }
144     
145             if (!tok)
146                 syntaxError(nullptr);
147             // not start of statement?
148             if (tok->previous() && !Token::Match(tok, "[;{}]"))
149                 continue;
150     
151             // skip starting tokens.. ;;; typedef typename foo::bar::..
152             while (Token::Match(tok, "[;{}]"))
153                 tok = tok->next();
154             while (Token::Match(tok, "typedef|typename"))
155                 tok = tok->next();
156             while (Token::Match(tok, "%type% ::"))
157                 tok = tok->tokAt(2);
158             if (!tok)
159                 break;
160     
161             // template variable or type..
162             if (Token::Match(tok, "%type% <")) {
163                 // these are used types..
164                 std::set<std::string> usedtypes;
165     
166                 // parse this statement and see if the '<' and '>' are matching
167                 unsigned int level = 0;
168                 for (const Token *tok2 = tok; tok2 && !Token::Match(tok2, "[;{}]"); tok2 = tok2->next()) {
169                     if (tok2->str() == "(")
170                         tok2 = tok2->link();
171                     else if (tok2->str() == "<") {
172                         bool inclevel = false;
173                         if (Token::simpleMatch(tok2->previous(), "operator <"))
174                             ;
175                         else if (level == 0)
176                             inclevel = true;
177                         else if (tok2->next() && tok2->next()->isStandardType())
178                             inclevel = true;
179                         else if (Token::simpleMatch(tok2, "< typename"))
180                             inclevel = true;
181                         else if (Token::Match(tok2->tokAt(-2), "<|, %type% <") && usedtypes.find(tok2->previous()->str()) != usedtypes.end())
182                             inclevel = true;
183                         else if (Token::Match(tok2, "< %type%") && usedtypes.find(tok2->next()->str()) != usedtypes.end())
184                             inclevel = true;
185                         else if (Token::Match(tok2, "< %type%")) {
186                             // is the next token a type and not a variable/constant?
187                             // assume it's a type if there comes another "<"
188                             const Token *tok3 = tok2->next();
189                             while (Token::Match(tok3, "%type% ::"))
190                                 tok3 = tok3->tokAt(2);
191                             if (Token::Match(tok3, "%type% <"))
192                                 inclevel = true;
193                         }
194     
195                         if (inclevel) {
196                             ++level;
197                             if (Token::Match(tok2->tokAt(-2), "<|, %type% <"))
198                                 usedtypes.insert(tok2->previous()->str());
199                         }
200                     } else if (tok2->str() == ">") {
201                         if (level > 0)
202                             --level;
203                     } else if (tok2->str() == ">>") {
204                         if (level > 0)
205                             --level;
206                         if (level > 0)
207                             --level;
208                     }
209                 }
210                 if (level > 0)
211                     syntaxError(tok);
212             }
213         }
214     }

216     unsigned int TemplateSimplifier::templateParameters(const Token *tok)
217     {
218         unsigned int numberOfParameters = 1;
219     
220         if (!tok)
221             return 0;
222         if (tok->str() != "<")
223             return 0;
224         tok = tok->next();
225     
226         unsigned int level = 0;
227     
228         while (tok) {
229             // skip const/volatile
230             if (Token::Match(tok, "const|volatile"))
231                 tok = tok->next();
232     
233             // skip struct/union
234             if (Token::Match(tok, "struct|union"))
235                 tok = tok->next();
236     
237             // Skip '&'
238             if (Token::Match(tok, "& ::| %name%"))
239                 tok = tok->next();
240     
241             // Skip variadic types (Ticket #5774, #6059, #6172)
242             if (Token::Match(tok, "%type% . . .")) {
243                 tok = tok->tokAt(4);
244                 continue;
245             }
246     
247             // Skip '=', '?', ':'
248             if (tok && Token::Match(tok, "=|?|:"))
249                 tok = tok->next();
250             if (!tok)
251                 return 0;
252     
253             // Skip casts
254             if (tok->str() == "(") {
255                 tok = tok->link();
256                 if (tok)
257                     tok = tok->next();
258             }
259     
260             // skip std::
261             if (tok && tok->str() == "::")
262                 tok = tok->next();
263             while (Token::Match(tok, "%name% ::")) {
264                 tok = tok->tokAt(2);
265                 if (tok && tok->str() == "*") // Ticket #5759: Class member pointer as a template argument; skip '*'
266                     tok = tok->next();
267             }
268             if (!tok)
269                 return 0;
270     
271             // num/type ..
272             if (!tok->isNumber() && tok->tokType() != Token::eChar && !tok->isName())
273                 return 0;
274             tok = tok->next();
275             if (!tok)
276                 return 0;
277     
278             // * / const
279             while (Token::Match(tok, "*|&|const"))
280                 tok = tok->next();
281     
282             if (!tok)
283                 return 0;
284     
285             // Function pointer or prototype..
286             while (Token::Match(tok, "(|[")) {
287                 if (!tok->link())
288                     syntaxError(tok);
289     
290                 tok = tok->link()->next();
291                 while (Token::Match(tok, "const|volatile")) // Ticket #5786: Skip function cv-qualifiers
292                     tok = tok->next();
293             }
294             if (!tok)
295                 return 0;
296     
297             // inner template
298             if (tok->str() == "<") {
299                 ++level;
300                 tok = tok->next();
301             }
302     
303             if (!tok)
304                 return 0;
305     
306             // ,/>
307             while (Token::Match(tok, ">|>>")) {
308                 if (level == 0)
309                     return ((tok->str() == ">") ? numberOfParameters : 0);
310                 --level;
311                 if (tok->str() == ">>") {
312                     if (level == 0)
313                         return numberOfParameters;
314                     --level;
315                 }
316                 tok = tok->next();
317     
318                 // * / &
319                 while (Token::Match(tok, "[*&]"))
320                     tok = tok->next();
321     
322                 if (!tok)
323                     return 0;
324             }
325     
326             if (tok->str() != ",")
327                 continue;
328             if (level == 0)
329                 ++numberOfParameters;
330             tok = tok->next();
331         }
332         return 0;
333     }

335     bool TemplateSimplifier::removeTemplate(Token *tok)
336     {
337         if (!Token::simpleMatch(tok, "template <"))
338             return false;
339     
340         int indentlevel = 0;
341         unsigned int countgt = 0;   // Counter for ">"
342         for (const Token *tok2 = tok->next(); tok2; tok2 = tok2->next()) {
343     
344             if (tok2->str() == "(") {
345                 tok2 = tok2->link();
346             } else if (tok2->str() == ")") {  // garbage code! (#3504)
347                 Token::eraseTokens(tok,tok2);
348                 tok->deleteThis();
349                 return false;
350             }
351     
352             else if (tok2->str() == "{") {
353                 tok2 = tok2->link()->next();
354                 Token::eraseTokens(tok, tok2);
355                 if (tok2 && tok2->str() == ";" && tok2->next())
356                     tok->deleteNext();
357                 tok->deleteThis();
358                 return true;
359             } else if (tok2->str() == "}") {  // garbage code! (#3449)
360                 Token::eraseTokens(tok,tok2);
361                 tok->deleteThis();
362                 return false;
363             }
364     
365             // Count ">"
366             if (tok2->str() == ">")
367                 countgt++;
368     
369             // don't remove constructor
370             if (tok2->str() == "explicit" ||
371                 (countgt == 1 && Token::Match(tok2->previous(), "> %type% (") &&
372                  Tokenizer::startOfExecutableScope(tok2->linkAt(1)))) {
373                 Token::eraseTokens(tok, tok2);
374                 tok->deleteThis();
375                 return true;
376             }
377     
378             if (tok2->str() == ";") {
379                 tok2 = tok2->next();
380                 Token::eraseTokens(tok, tok2);
381                 tok->deleteThis();
382                 return true;
383             }
384     
385             if (tok2->str() == "<")
386                 ++indentlevel;
387     
388             else if (indentlevel >= 2 && tok2->str() == ">")
389                 --indentlevel;
390     
391             else if (Token::Match(tok2, "> class|struct %name% [,)]")) {
392                 tok2 = tok2->next();
393                 Token::eraseTokens(tok, tok2);
394                 tok->deleteThis();
395                 return true;
396             }
397         }
398     
399         return false;
400     }

402     std::set<std::string> TemplateSimplifier::expandSpecialized(Token *tokens)
403     {
404         std::set<std::string> expandedtemplates;
405     
406         // Locate specialized templates..
407         for (Token *tok = tokens; tok; tok = tok->next()) {
408             if (!Token::simpleMatch(tok, "template < >"))
409                 continue;
410     
411             // what kind of template is this?
412             Token *tok2 = tok->tokAt(3);
413             while (tok2 && (tok2->isName() || tok2->str() == "*"))
414                 tok2 = tok2->next();
415     
416             if (!TemplateSimplifier::templateParameters(tok2))
417                 continue;
418     
419             // unknown template.. bail out
420             if (!tok2->previous()->isName())
421                 continue;
422     
423             tok2 = tok2->previous();
424             std::string s;
425             {
426                 std::ostringstream ostr;
427                 const Token *tok3 = tok2;
428                 for (; tok3 && tok3->str() != ">"; tok3 = tok3->next()) {
429                     if (tok3 != tok2)
430                         ostr << " ";
431                     ostr << tok3->str();
432                 }
433                 if (!Token::Match(tok3, "> (|{"))
434                     continue;
435                 s = ostr.str();
436             }
437     
438             // save search pattern..
439             const std::string pattern(s + " >");
440     
441             // remove spaces to create new name
442             s.erase(std::remove(s.begin(), s.end(), ' '), s.end());
443             const std::string name(s + ">");
444             expandedtemplates.insert(name);
445     
446             // Rename template..
447             Token::eraseTokens(tok2, Token::findsimplematch(tok2, "<")->findClosingBracket()->next());
448             tok2->str(name);
449     
450             // delete the "template < >"
451             tok->deleteNext(2);
452             tok->deleteThis();
453     
454             // Use this special template in the code..
455             while (nullptr != (tok2 = const_cast<Token *>(Token::findsimplematch(tok2, pattern.c_str())))) {
456                 Token::eraseTokens(tok2, Token::findsimplematch(tok2, "<")->findClosingBracket()->next());
457                 tok2->str(name);
458             }
459         }
460     
461         return expandedtemplates;
462     }

464     std::list<Token *> TemplateSimplifier::getTemplateDeclarations(Token *tokens, bool &codeWithTemplates)
465     {
466         std::list<Token *> templates;
467         for (Token *tok = tokens; tok; tok = tok->next()) {
468             // TODO: handle namespaces. Right now we don't instantiate templates that are defined in namespaces.
469             if (Token::Match(tok, "namespace %type% {"))
470                 tok = tok->linkAt(2);
471     
472             if (Token::simpleMatch(tok, "template <")) {
473                 // Some syntax checks, see #6865
474                 if (!tok->tokAt(2))
475                     syntaxError(tok->next());
476                 if (tok->strAt(2)=="typename" &&
477                     (!tok->tokAt(3) || !Token::Match(tok->tokAt(3), "%name%|.|,|>")))
478                     syntaxError(tok->next());
479                 codeWithTemplates = true;
480                 Token *parmEnd = tok->next()->findClosingBracket();
481                 int indentlevel = 0;
482                 for (const Token *tok2 = parmEnd; tok2; tok2 = tok2->next()) {
483                     if (tok2->str() == "(")
484                         ++indentlevel;
485                     else if (tok2->str() == ")")
486                         --indentlevel;
487     
488                     if (indentlevel) // In an argument list; move to the next token
489                         continue;
490     
491                     // Just a declaration => ignore this
492                     if (tok2->str() == ";")
493                         break;
494                     // Implementation => add to "templates"
495                     if (tok2->str() == "{") {
496                         templates.push_back(tok);
497                         break;
498                     }
499                 }
500             }
501         }
502         return templates;
503     }

506     std::list<Token *> TemplateSimplifier::getTemplateInstantiations(Token *tokens)
507     {
508         std::list<Token *> used;
509     
510         for (Token *tok = tokens; tok; tok = tok->next()) {
511             // template definition.. skip it
512             if (Token::simpleMatch(tok, "template <")) {
513                 tok = tok->next()->findClosingBracket();
514                 if (!tok)
515                     break;
516             } else if (Token::Match(tok->previous(), "[({};=] %name% <") ||
517                        Token::Match(tok->previous(), "%type% %name% <") ||
518                        Token::Match(tok->tokAt(-2), "[,:] private|protected|public %name% <")) {
519     
520                 // Add inner template instantiations first => go to the ">"
521                 // and then parse backwards, adding all seen instantiations
522                 const Token *tok2 = tok->next()->findClosingBracket();
523     
524                 // parse backwards and add template instantiations
525                 for (; tok2 && tok2 != tok; tok2 = tok2->previous()) {
526                     if (Token::Match(tok2, ", %name% <") &&
527                         TemplateSimplifier::templateParameters(tok2->tokAt(2))) {
528                         used.push_back(tok2->next());
529                     }
530                 }
531     
532                 // Add outer template..
533                 if (TemplateSimplifier::templateParameters(tok->next()))
534                     used.push_back(tok);
535             }
536         }
537     
538         return used;
539     }

542     void TemplateSimplifier::useDefaultArgumentValues(const std::list<Token *> &templates,
543             std::list<Token *> * const templateInstantiations)
544     {
545         for (std::list<Token *>::const_iterator iter1 = templates.begin(); iter1 != templates.end(); ++iter1) {
546             // template parameters with default value has syntax such as:
547             //     x = y
548             // this list will contain all the '=' tokens for such arguments
549             std::list<Token *> eq;
550             // and this set the position of parameters with a default value
551             std::set<std::size_t> defaultedArgPos;
552     
553             // parameter number. 1,2,3,..
554             std::size_t templatepar = 1;
555     
556             // parameter depth
557             std::size_t templateParmDepth = 0;
558     
559             // the template classname. This will be empty for template functions
560             std::string classname;
561     
562             // Scan template declaration..
563             for (Token *tok = *iter1; tok; tok = tok->next()) {
564                 if (Token::simpleMatch(tok, "template < >")) { // Ticket #5762: Skip specialization tokens
565                     tok = tok->tokAt(2);
566                     if (0 == templateParmDepth)
567                         break;
568                     continue;
569                 }
570     
571                 if (tok->str() == "(") { // Ticket #6835
572                     tok = tok->link();
573                     continue;
574                 }
575     
576                 if (tok->str() == "<" && templateParameters(tok))
577                     ++templateParmDepth;
578     
579                 // end of template parameters?
580                 if (tok->str() == ">") {
581                     if (Token::Match(tok, "> class|struct %name%"))
582                         classname = tok->strAt(2);
583                     if (templateParmDepth<2)
584                         break;
585                     else
586                         --templateParmDepth;
587                 }
588     
589                 // next template parameter
590                 if (tok->str() == "," && (1 == templateParmDepth)) // Ticket #5823: Properly count parameters
591                     ++templatepar;
592     
593                 // default parameter value?
594                 else if (Token::Match(tok, "= !!>")) {
595                     if (defaultedArgPos.insert(templatepar).second) {
596                         eq.push_back(tok);
597                     } else {
598                         // Ticket #5605: Syntax error (two equal signs for the same parameter), bail out
599                         eq.clear();
600                         break;
601                     }
602                 }
603             }
604             if (eq.empty() || classname.empty())
605                 continue;
606     
607             // iterate through all template instantiations
608             for (std::list<Token *>::const_iterator iter2 = templateInstantiations->begin(); iter2 != templateInstantiations->end(); ++iter2) {
609                 Token *tok = *iter2;
610     
611                 if (!Token::simpleMatch(tok, (classname + " <").c_str()))
612                     continue;
613     
614                 // count the parameters..
615                 tok = tok->next();
616                 const unsigned int usedpar = TemplateSimplifier::templateParameters(tok);
617                 tok = tok->findClosingBracket();
618     
619                 if (tok && tok->str() == ">") {
620                     tok = tok->previous();
621                     std::list<Token *>::const_iterator it = eq.begin();
622                     for (std::size_t i = (templatepar - eq.size()); it != eq.end() && i < usedpar; ++i)
623                         ++it;
624                     while (it != eq.end()) {
625                         int indentlevel = 0;
626                         tok->insertToken(",");
627                         tok = tok->next();
628                         const Token *from = (*it)->next();
629                         std::stack<Token *> links;
630                         while (from && (!links.empty() || (from->str() != "," && (indentlevel || from->str() != ">")))) {
631                             if (from->str() == "<")
632                                 ++indentlevel;
633                             else if (from->str() == ">")
634                                 --indentlevel;
635                             tok->insertToken(from->str(), from->originalName());
636                             tok = tok->next();
637                             if (Token::Match(tok, "(|["))
638                                 links.push(tok);
639                             else if (!links.empty() && Token::Match(tok, ")|]")) {
640                                 Token::createMutualLinks(links.top(), tok);
641                                 links.pop();
642                             }
643                             from = from->next();
644                         }
645                         ++it;
646                     }
647                 }
648             }
649     
650             for (std::list<Token *>::iterator it = eq.begin(); it != eq.end(); ++it) {
651                 Token * const eqtok = *it;
652                 Token *tok2;
653                 int indentlevel = 0;
654                 for (tok2 = eqtok->next(); tok2; tok2 = tok2->next()) {
655                     if (Token::Match(tok2, ";|)|}|]")) { // bail out #6607
656                         tok2 = nullptr;
657                         break;
658                     }
659                     if (Token::Match(tok2, "(|{|["))
660                         tok2 = tok2->link();
661                     else if (Token::Match(tok2, "%type% <") && templateParameters(tok2->next())) {
662                         std::list<Token*>::iterator ti = std::find(templateInstantiations->begin(),
663                                                          templateInstantiations->end(),
664                                                          tok2);
665                         if (ti != templateInstantiations->end())
666                             templateInstantiations->erase(ti);
667                         ++indentlevel;
668                     } else if (indentlevel > 0 && tok2->str() == ">")
669                         --indentlevel;
670                     else if (indentlevel == 0 && Token::Match(tok2, ",|>"))
671                         break;
672                     if (indentlevel < 0)
673                         break;
674                 }
675                 // something went wrong, don't call eraseTokens()
676                 // with a nullptr "end" parameter (=all remaining tokens).
677                 if (!tok2)
678                     continue;
679     
680                 Token::eraseTokens(eqtok, tok2);
681                 eqtok->deleteThis();
682             }
683         }
684     }

686     bool TemplateSimplifier::instantiateMatch(const Token *instance, const std::string &name, std::size_t numberOfArguments, const char patternAfter[])
687     {
688         if (!Token::simpleMatch(instance, (name + " <").c_str()))
689             return false;
690     
691         if (numberOfArguments != TemplateSimplifier::templateParameters(instance->next()))
692             return false;
693     
694         if (patternAfter) {
695             const Token *tok = instance;
696             unsigned int indentlevel = 0;
697             for (tok = instance; tok && (tok->str() != ">" || indentlevel > 0); tok = tok->next()) {
698                 if (Token::Match(tok, "[<,] %name% <") && templateParameters(tok->tokAt(2)) > 0)
699                     ++indentlevel;
700                 if (indentlevel > 0 && tok->str() == ">")
701                     --indentlevel;
702             }
703             if (!tok || !Token::Match(tok->next(), patternAfter))
704                 return false;
705         }
706     
707         // nothing mismatching was found..
708         return true;
709     }

711     int TemplateSimplifier::getTemplateNamePosition(const Token *tok)
712     {
713         // get the position of the template name
714         int namepos = 0, starAmpPossiblePosition = 0;
715         if (Token::Match(tok, "> class|struct %type% {|:|<"))
716             namepos = 2;
717         else if (Token::Match(tok, "> %type% *|&| %type% ("))
718             namepos = 2;
719         else if (Token::Match(tok, "> %type% %type% *|&| %type% ("))
720             namepos = 3;
721         else if (Token::Match(tok, "> %type% *|&| %type% :: %type% (")) {
722             namepos = 4;
723             starAmpPossiblePosition = 2;
724         } else if (Token::Match(tok, "> %type% %type% *|&| %type% :: %type% (")) {
725             namepos = 5;
726             starAmpPossiblePosition = 3;
727         } else {
728             // Name not found
729             return -1;
730         }
731         if (Token::Match(tok->tokAt(starAmpPossiblePosition ? starAmpPossiblePosition : namepos), "*|&"))
732             ++namepos;
733     
734         return namepos;
735     }

738     void TemplateSimplifier::expandTemplate(
739         TokenList& tokenlist,
740         const Token *tok,
741         const std::string &name,
742         std::vector<const Token *> &typeParametersInDeclaration,
743         const std::string &newName,
744         std::vector<const Token *> &typesUsedInTemplateInstantiation,
745         std::list<Token *> &templateInstantiations)
746     {
747         bool inTemplateDefinition=false;
748         std::vector<const Token *> localTypeParametersInDeclaration;
749         for (const Token *tok3 = tokenlist.front(); tok3; tok3 = tok3 ? tok3->next() : nullptr) {
750             if (tok3->str()=="template") {
751                 if (tok3->next() && tok3->next()->str()=="<") {
752                     TemplateParametersInDeclaration(tok3->tokAt(2), localTypeParametersInDeclaration);
753                     if (localTypeParametersInDeclaration.size() != typeParametersInDeclaration.size())
754                         inTemplateDefinition = false; // Partial specialization
755                     else
756                         inTemplateDefinition = true;
757                 } else
758                     inTemplateDefinition = false; // Only template instantiation
759             }
760             if (Token::Match(tok3, "{|(|["))
761                 tok3 = tok3->link();
762     
763             // Start of template..
764             if (tok3 == tok) {
765                 tok3 = tok3->next();
766             }
767     
768             // member function implemented outside class definition
769             else if (inTemplateDefinition &&
770                      TemplateSimplifier::instantiateMatch(tok3, name, typeParametersInDeclaration.size(), ":: ~| %name% (")) {
771                 tokenlist.addtoken(newName, tok3->linenr(), tok3->fileIndex());
772                 while (tok3 && tok3->str() != "::")
773                     tok3 = tok3->next();
774             }
775     
776             // not part of template.. go on to next token
777             else
778                 continue;
779     
780             int indentlevel = 0;
781             std::stack<Token *> brackets; // holds "(", "[" and "{" tokens
782     
783             for (; tok3; tok3 = tok3->next()) {
784                 if (tok3->isName()) {
785                     // search for this token in the type vector
786                     unsigned int itype = 0;
787                     while (itype < typeParametersInDeclaration.size() && typeParametersInDeclaration[itype]->str() != tok3->str())
788                         ++itype;
789     
790                     // replace type with given type..
791                     if (itype < typeParametersInDeclaration.size()) {
792                         unsigned int typeindentlevel = 0;
793                         for (const Token *typetok = typesUsedInTemplateInstantiation[itype];
794                              typetok && (typeindentlevel>0 || !Token::Match(typetok, ",|>"));
795                              typetok = typetok->next()) {
796                             if (Token::simpleMatch(typetok, ". . .")) {
797                                 typetok = typetok->tokAt(2);
798                                 continue;
799                             }
800                             if (Token::Match(typetok, "%name% <") && templateParameters(typetok->next()) > 0)
801                                 ++typeindentlevel;
802                             else if (typeindentlevel > 0 && typetok->str() == ">")
803                                 --typeindentlevel;
804                             tokenlist.addtoken(typetok, tok3->linenr(), tok3->fileIndex());
805                         }
806                         continue;
807                     }
808                 }
809     
810                 // replace name..
811                 if (Token::Match(tok3, (name + " !!<").c_str())) {
812                     tokenlist.addtoken(newName, tok3->linenr(), tok3->fileIndex());
813                     continue;
814                 }
815     
816                 // copy
817                 tokenlist.addtoken(tok3, tok3->linenr(), tok3->fileIndex());
818                 if (Token::Match(tok3, "%type% <")) {
819                     //if (!Token::simpleMatch(tok3, (name + " <").c_str()))
820                     //done = false;
821                     templateInstantiations.push_back(tokenlist.back());
822                 }
823     
824                 // link() newly tokens manually
825                 else if (tok3->str() == "{") {
826                     brackets.push(tokenlist.back());
827                     indentlevel++;
828                 } else if (tok3->str() == "(") {
829                     brackets.push(tokenlist.back());
830                 } else if (tok3->str() == "[") {
831                     brackets.push(tokenlist.back());
832                 } else if (tok3->str() == "}") {
833                     assert(brackets.empty() == false && brackets.top()->str() == "{");
834                     Token::createMutualLinks(brackets.top(), tokenlist.back());
835                     if (tok3->strAt(1) == ";") {
836                         const Token * tokSemicolon = tok3->next();
837                         tokenlist.addtoken(tokSemicolon, tokSemicolon->linenr(), tokSemicolon->fileIndex());
838                     }
839                     brackets.pop();
840                     if (indentlevel <= 1 && brackets.empty()) {
841                         // there is a bug if indentlevel is 0
842                         // the "}" token should only be added if indentlevel is 1 but I add it always intentionally
843                         // if indentlevel ever becomes 0, cppcheck will write:
844                         // ### Error: Invalid number of character {
845                         break;
846                     }
847                     --indentlevel;
848                 } else if (tok3->str() == ")") {
849                     assert(brackets.empty() == false && brackets.top()->str() == "(");
850                     Token::createMutualLinks(brackets.top(), tokenlist.back());
851                     brackets.pop();
852                 } else if (tok3->str() == "]") {
853                     assert(brackets.empty() == false && brackets.top()->str() == "[");
854                     Token::createMutualLinks(brackets.top(), tokenlist.back());
855                     brackets.pop();
856                 }
857             }
858     
859             assert(brackets.empty());
860         }
861     }

863     static bool isLowerThanLogicalAnd(const Token *lower)
864     {
865         return lower->isAssignmentOp() || Token::Match(lower, "}|;|(|[|]|)|,|?|:|%oror%|return|throw|case");
866     }
867     static bool isLowerThanOr(const Token* lower)
868     {
869         return isLowerThanLogicalAnd(lower) || lower->str() == "&&";
870     }
871     static bool isLowerThanXor(const Token* lower)
872     {
873         return isLowerThanOr(lower) || lower->str() == "|";
874     }
875     static bool isLowerThanAnd(const Token* lower)
876     {
877         return isLowerThanXor(lower) || lower->str() == "^";
878     }
879     static bool isLowerThanShift(const Token* lower)
880     {
881         return isLowerThanAnd(lower) || lower->str() == "&";
882     }
883     static bool isLowerThanPlusMinus(const Token* lower)
884     {
885         return isLowerThanShift(lower) || Token::Match(lower, "%comp%|<<|>>");
886     }
887     static bool isLowerThanMulDiv(const Token* lower)
888     {
889         return isLowerThanPlusMinus(lower) || Token::Match(lower, "+|-");
890     }
891     static bool isLowerEqualThanMulDiv(const Token* lower)
892     {
893         return isLowerThanMulDiv(lower) || Token::Match(lower, "[*/%]");
894     }

896     static std::string ShiftInt(const char cop, const Token* left, const Token* right)
897     {
898         if (cop == '&' || cop == '|' || cop == '^')
899             return MathLib::calculate(left->str(), right->str(), cop);
900     
901         const MathLib::bigint leftInt = MathLib::toLongNumber(left->str());
902         const MathLib::bigint rightInt = MathLib::toLongNumber(right->str());
903         const bool rightIntIsPositive = rightInt >= 0;
904     
905         if (cop == '<') {
906             const bool leftOperationIsNotLeftShift = left->previous()->str() != "<<";
907             const bool operandIsLeftShift = right->previous()->str() == "<<";
908     
909             // Ensure that its not a shift operator as used for streams
910             if (leftOperationIsNotLeftShift && operandIsLeftShift && rightIntIsPositive) {
911                 const bool leftIntIsPositive = leftInt >= 0;
912                 if (!leftIntIsPositive) { // In case the left integer is negative, e.g. -1000 << 16. Do not simplify.
913                     return left->str() + " << " + right->str();
914                 }
915                 return MathLib::toString(leftInt << rightInt);
916             }
917         } else if (rightIntIsPositive) {
918             return MathLib::toString(leftInt >> rightInt);
919         }
920         return "";
921     }
922     
923     static std::string ShiftUInt(const char cop, const Token* left, const Token* right)
924     {
925         if (cop == '&' || cop == '|' || cop == '^')
926             return MathLib::calculate(left->str(), right->str(), cop);
927     
928         const MathLib::biguint leftInt=MathLib::toULongNumber(left->str());
929         const MathLib::biguint rightInt=MathLib::toULongNumber(right->str());
930         if (cop == '<') {
931             if (left->previous()->str() != "<<") // Ensure that its not a shift operator as used for streams
932                 return MathLib::toString(leftInt << rightInt);
933         } else {
934             return MathLib::toString(leftInt >> rightInt);
935         }
936         return "";
937     }

939     bool TemplateSimplifier::simplifyNumericCalculations(Token *tok)
940     {
941         bool ret = false;
942         // (1-2)
943         while (tok->tokAt(4) && tok->next()->isNumber() && tok->tokAt(3)->isNumber()) { // %any% %num% %any% %num% %any%
944             const Token* op = tok->tokAt(2);
945             const Token* after = tok->tokAt(4);
946             if (Token::Match(tok, "* %num% /") && (op->strAt(1) != "0") && tok->next()->str() == MathLib::multiply(op->strAt(1), MathLib::divide(tok->next()->str(), op->strAt(1)))) {
947                 // Division where result is a whole number
948             } else if (!((op->str() == "*" && (isLowerThanMulDiv(tok) || tok->str() == "*") && isLowerEqualThanMulDiv(after)) || // associative
949                          (Token::Match(op, "[/%]") && isLowerThanMulDiv(tok) && isLowerEqualThanMulDiv(after)) || // NOT associative
950                          (Token::Match(op, "[+-]") && isLowerThanMulDiv(tok) && isLowerThanMulDiv(after)) || // Only partially (+) associative, but handled later
951                          (Token::Match(op, ">>|<<") && isLowerThanShift(tok) && isLowerThanPlusMinus(after)) || // NOT associative
952                          (op->str() == "&" && isLowerThanShift(tok) && isLowerThanShift(after)) || // associative
953                          (op->str() == "^" && isLowerThanAnd(tok) && isLowerThanAnd(after)) || // associative
954                          (op->str() == "|" && isLowerThanXor(tok) && isLowerThanXor(after)) || // associative
955                          (op->str() == "&&" && isLowerThanOr(tok) && isLowerThanOr(after)) ||
956                          (op->str() == "||" && isLowerThanLogicalAnd(tok) && isLowerThanLogicalAnd(after))))
957                 break;
958     
959             tok = tok->next();
960     
961             // Don't simplify "%num% / 0"
962             if (Token::Match(op, "[/%] 0"))
963                 continue;
964     
965             // Integer operations
966             if (Token::Match(op, ">>|<<|&|^|%or%")) {
967                 // Don't simplify if operand is negative, shifting with negative
968                 // operand is UB. Bitmasking with negative operand is implementation
969                 // defined behaviour.
970                 if (MathLib::isNegative(tok->str()) || MathLib::isNegative(tok->strAt(2)))
971                     continue;
972     
973                 const char cop = op->str()[0];
974                 std::string result;
975                 if (tok->str().find_first_of("uU") != std::string::npos)
976                     result = ShiftUInt(cop, tok, tok->tokAt(2));
977                 else
978                     result = ShiftInt(cop, tok, tok->tokAt(2));
979                 if (result.empty())
980                     break;
981                 tok->str(result);
982             }
983     
984             // Logical operations
985             else if (Token::Match(op, "%oror%|&&")) {
986                 int op1 = !MathLib::isNullValue(tok->str());
987                 int op2 = !MathLib::isNullValue(tok->strAt(2));
988                 int result = (op->str() == "||") ? (op1 || op2) : (op1 && op2);
989                 tok->str(result ? "1" : "0");
990             }
991     
992             else if (Token::Match(tok->previous(), "- %num% - %num%"))
993                 tok->str(MathLib::add(tok->str(), tok->strAt(2)));
994             else if (Token::Match(tok->previous(), "- %num% + %num%"))
995                 tok->str(MathLib::subtract(tok->str(), tok->strAt(2)));
996             else {
997                 try {
998                     tok->str(MathLib::calculate(tok->str(), tok->strAt(2), op->str()[0]));
999                 } catch (InternalError &e) {
1000                     e.token = tok;
1001                     throw;
1002                 }
1003             }
1004     
1005             tok->deleteNext(2);
1006             tok = tok->previous();
1007     
1008             ret = true;
1009         }
1010     
1011         if (Token::Match(tok, "%oror%|&& %num% %oror%|&&|,|)") || Token::Match(tok, "[(,] %num% %oror%|&&")) {
1012             tok->next()->str(MathLib::isNullValue(tok->next()->str()) ? "0" : "1");
1013         }
1014     
1015         return ret;
1016     }

1018     // TODO: This is not the correct class for simplifyCalculations(), so it
1019     // should be moved away.
1020     bool TemplateSimplifier::simplifyCalculations(Token *_tokens)
1021     {
1022         bool ret = false, goback = false;
1023         for (Token *tok = _tokens; tok; tok = tok->next()) {
1024             if (goback) {
1025                 tok = tok->previous();
1026                 goback = false;
1027             }
1028             // Remove parentheses around variable..
1029             // keep parentheses here: dynamic_cast<Fred *>(p);
1030             // keep parentheses here: A operator * (int);
1031             // keep parentheses here: int ( * ( * f ) ( ... ) ) (int) ;
1032             // keep parentheses here: int ( * * ( * compilerHookVector ) (void) ) ( ) ;
1033             // keep parentheses here: operator new [] (size_t);
1034             // keep parentheses here: Functor()(a ... )
1035             // keep parentheses here: ) ( var ) ;
1036             if ((Token::Match(tok->next(), "( %name% ) ;|)|,|]") ||
1037                  (Token::Match(tok->next(), "( %name% ) %cop%") && (tok->tokAt(2)->varId()>0 || !Token::Match(tok->tokAt(4), "[*&+-~]")))) &&
1038                 !tok->isName() &&
1039                 tok->str() != ">" &&
1040                 tok->str() != ")" &&
1041                 tok->str() != "]") {
1042                 tok->deleteNext();
1043                 tok = tok->next();
1044                 tok->deleteNext();
1045                 ret = true;
1046             }
1047     
1048             if (Token::Match(tok->previous(), "(|&&|%oror% %char% %comp% %num% &&|%oror%|)")) {
1049                 tok->str(MathLib::toString(MathLib::toLongNumber(tok->str())));
1050             }
1051     
1052             if (tok->isNumber()) {
1053                 // Remove redundant conditions (0&&x) (1||x)
1054                 if (Token::Match(tok->previous(), "[(=,] 0 &&") ||
1055                     Token::Match(tok->previous(), "[(=,] 1 %oror%")) {
1056                     unsigned int par = 0;
1057                     const Token *tok2 = tok;
1058                     bool andAnd = (tok->next()->str() == "&&");
1059                     for (; tok2; tok2 = tok2->next()) {
1060                         if (tok2->str() == "(")
1061                             ++par;
1062                         else if (tok2->str() == ")") {
1063                             if (par == 0)
1064                                 break;
1065                             --par;
1066                         } else if (par == 0 && isLowerThanLogicalAnd(tok2) && (andAnd || tok2->str() != "||"))
1067                             break;
1068                     }
1069                     if (tok2) {
1070                         Token::eraseTokens(tok, tok2);
1071                         ret = true;
1072                     }
1073                     continue;
1074                 }
1075     
1076                 if (tok->str() == "0") {
1077                     if ((Token::Match(tok->previous(), "[+-] 0 %cop%|;") && isLowerThanMulDiv(tok->next())) ||
1078                         (Token::Match(tok->previous(), "%or% 0 %cop%|;") && isLowerThanXor(tok->next()))) {
1079                         tok = tok->previous();
1080                         if (Token::Match(tok->tokAt(-4), "[;{}] %name% = %name% [+-|] 0 ;") &&
1081                             tok->strAt(-3) == tok->previous()->str()) {
1082                             tok = tok->tokAt(-3);
1083                             tok->deleteNext(2);
1084                             tok->deleteThis();
1085                         }
1086                         tok->deleteNext();
1087                         tok->deleteThis();
1088                         ret = true;
1089                     } else if (Token::Match(tok->previous(), "[=([,] 0 [+|]") ||
1090                                Token::Match(tok->previous(), "return|case 0 [+|]")) {
1091                         tok = tok->previous();
1092                         tok->deleteNext(2);
1093                         ret = true;
1094                     } else if (Token::Match(tok->previous(), "[=[(,] 0 * %name% ,|]|)|;|=|%cop%") ||
1095                                Token::Match(tok->previous(), "[=[(,] 0 * %num% ,|]|)|;|%op%") ||
1096                                Token::Match(tok->previous(), "[=[(,] 0 * (") ||
1097                                Token::Match(tok->previous(), "return|case 0 *|&& %name% ,|:|;|=|%cop%") ||
1098                                Token::Match(tok->previous(), "return|case 0 *|&& %num% ,|:|;|%op%") ||
1099                                Token::Match(tok->previous(), "return|case 0 *|&& (")) {
1100                         tok->deleteNext();
1101                         if (tok->next()->str() == "(")
1102                             Token::eraseTokens(tok, tok->next()->link());
1103                         tok->deleteNext();
1104                         ret = true;
1105                     } else if (Token::Match(tok->previous(), "[=[(,] 0 && *|& %any% ,|]|)|;|=|%cop%") ||
1106                                Token::Match(tok->previous(), "return|case 0 && *|& %any% ,|:|;|=|%cop%")) {
1107                         tok->deleteNext();
1108                         tok->deleteNext();
1109                         if (tok->next()->str() == "(")
1110                             Token::eraseTokens(tok, tok->next()->link());
1111                         tok->deleteNext();
1112                         ret = true;
1113                     }
1114                 }
1115     
1116                 if (tok->str() == "1") {
1117                     if (Token::Match(tok->previous(), "[=[(,] 1 %oror% %any% ,|]|)|;|=|%cop%") ||
1118                         Token::Match(tok->previous(), "return|case 1 %oror% %any% ,|:|;|=|%cop%")) {
1119                         tok->deleteNext();
1120                         if (tok->next()->str() == "(")
1121                             Token::eraseTokens(tok, tok->next()->link());
1122                         tok->deleteNext();
1123                         ret = true;
1124                     } else if (Token::Match(tok->previous(), "[=[(,] 1 %oror% *|& %any% ,|]|)|;|=|%cop%") ||
1125                                Token::Match(tok->previous(), "return|case 1 %oror% *|& %any% ,|:|;|=|%cop%")) {
1126                         tok->deleteNext();
1127                         tok->deleteNext();
1128                         if (tok->next()->str() == "(")
1129                             Token::eraseTokens(tok, tok->next()->link());
1130                         tok->deleteNext();
1131                         ret = true;
1132                     }
1133                 }
1134     
1135                 if (Token::simpleMatch(tok->previous(), "* 1") || Token::simpleMatch(tok, "1 *")) {
1136                     if (tok->previous() && tok->previous()->isConstOp())
1137                         tok = tok->previous();
1138                     tok->deleteNext();
1139                     tok->deleteThis();
1140                     ret = true;
1141                 }
1142     
1143                 // Remove parentheses around number..
1144                 if (Token::Match(tok->tokAt(-2), "%op%|< ( %num% )") && !tok->tokAt(-2)->isName() && tok->strAt(-2) != ">") {
1145                     tok = tok->previous();
1146                     tok->deleteThis();
1147                     tok->deleteNext();
1148                     ret = true;
1149                 }
1150     
1151                 if (Token::simpleMatch(tok->previous(), "( 0 |") ||
1152                     Token::simpleMatch(tok->previous(), "| 0 )")) {
1153                     if (tok->previous()->isConstOp())
1154                         tok = tok->previous();
1155                     tok->deleteNext();
1156                     tok->deleteThis();
1157                     ret = true;
1158                 }
1159     
1160                 if (Token::Match(tok, "%num% %comp% %num%") &&
1161                     MathLib::isInt(tok->str()) &&
1162                     MathLib::isInt(tok->strAt(2))) {
1163                     if (Token::Match(tok->previous(), "(|&&|%oror%") && Token::Match(tok->tokAt(3), ")|&&|%oror%|?")) {
1164                         const MathLib::bigint op1(MathLib::toLongNumber(tok->str()));
1165                         const std::string &cmp(tok->next()->str());
1166                         const MathLib::bigint op2(MathLib::toLongNumber(tok->strAt(2)));
1167     
1168                         std::string result;
1169     
1170                         if (cmp == "==")
1171                             result = (op1 == op2) ? "1" : "0";
1172                         else if (cmp == "!=")
1173                             result = (op1 != op2) ? "1" : "0";
1174                         else if (cmp == "<=")
1175                             result = (op1 <= op2) ? "1" : "0";
1176                         else if (cmp == ">=")
1177                             result = (op1 >= op2) ? "1" : "0";
1178                         else if (cmp == "<")
1179                             result = (op1 < op2) ? "1" : "0";
1180                         else
1181                             result = (op1 > op2) ? "1" : "0";
1182     
1183                         tok->str(result);
1184                         tok->deleteNext(2);
1185                         ret = true;
1186                         goback = true;
1187                     }
1188                 }
1189             }
1190             // Division where result is a whole number
1191             else if (Token::Match(tok->previous(), "* %num% /") &&
1192                      tok->str() == MathLib::multiply(tok->strAt(2), MathLib::divide(tok->str(), tok->strAt(2)))) {
1193                 tok->deleteNext(2);
1194             }
1195     
1196             else if (simplifyNumericCalculations(tok)) {
1197                 ret = true;
1198                 while (Token::Match(tok->tokAt(-2), "%cop%|,|( %num% %cop% %num% %cop%|,|)")) {
1199                     Token *before = tok->tokAt(-2);
1200                     if (simplifyNumericCalculations(before))
1201                         tok = before;
1202                     else
1203                         break;
1204                 }
1205             }
1206         }
1207         return ret;
1208     }

1222     bool TemplateSimplifier::simplifyTemplateInstantiations(
1223         TokenList& tokenlist,
1224         ErrorLogger* errorlogger,
1225         const Settings *_settings,
1226         const Token *tok,
1227         std::list<Token *> &templateInstantiations,
1228         std::set<std::string> &expandedtemplates)
1229     {
1230         // this variable is not used at the moment. The intention was to
1231         // allow continuous instantiations until all templates has been expanded
1232         //bool done = false;
1233     
1234         // Contains tokens such as "T"
1235         std::vector<const Token *> typeParametersInDeclaration;
1236         tok = TemplateParametersInDeclaration(tok->tokAt(2), typeParametersInDeclaration);
1237     
1238         // bail out if the end of the file was reached
1239         if (!tok)
1240             return false;
1241     
1242         const bool printDebug = _settings->debugwarnings;
1243     
1244         // get the position of the template name
1245         int namepos = TemplateSimplifier::getTemplateNamePosition(tok);
1246         if (namepos == -1) {
1247             // debug message that we bail out..
1248             if (printDebug && errorlogger) {
1249                 std::list<const Token *> callstack(1, tok);
1250                 errorlogger->reportErr(ErrorLogger::ErrorMessage(callstack, &tokenlist, Severity::debug, "debug", "simplifyTemplates: bailing out", false));
1251             }
1252             return false;
1253         }
1254     
1255         // name of template function/class..
1256         const std::string name(tok->strAt(namepos));
1257     
1258         const bool isfunc(tok->strAt(namepos + 1) == "(");
1259     
1260         // locate template usage..
1261         std::string::size_type amountOftemplateInstantiations = templateInstantiations.size();
1262         unsigned int recursiveCount = 0;
1263     
1264         bool instantiated = false;
1265     
1266         for (std::list<Token *>::const_iterator iter2 = templateInstantiations.begin(); iter2 != templateInstantiations.end(); ++iter2) {
1267             if (amountOftemplateInstantiations != templateInstantiations.size()) {
1268                 amountOftemplateInstantiations = templateInstantiations.size();
1269                 simplifyCalculations(tokenlist.front());
1270                 ++recursiveCount;
1271                 if (recursiveCount > 100) {
1272                     // bail out..
1273                     break;
1274                 }
1275             }
1276             Token * const tok2 = *iter2;
1277             assert(tokenlist.validateToken(tok2)); // that assertion fails on examples from #6021
1278             if (tok2->str() != name)
1279                 continue;
1280     
1281             if (Token::Match(tok2->previous(), "[;{}=]") &&
1282                 !TemplateSimplifier::instantiateMatch(*iter2, name, typeParametersInDeclaration.size(), isfunc ? "(" : "*| %name%"))
1283                 continue;
1284     
1285             // New type..
1286             std::vector<const Token *> typesUsedInTemplateInstantiation;
1287             std::string typeForNewName;
1288             std::string templateMatchPattern(name + " < ");
1289             unsigned int indentlevel = 0;
1290             for (const Token *tok3 = tok2->tokAt(2); tok3 && (indentlevel > 0 || tok3->str() != ">"); tok3 = tok3->next()) {
1291                 // #2648 - unhandled parentheses => bail out
1292                 // #2721 - unhandled [ => bail out
1293                 if (Token::Match(tok3, "(|[")) {
1294                     typeForNewName.clear();
1295                     break;
1296                 }
1297                 if (!tok3->next()) {
1298                     typeForNewName.clear();
1299                     break;
1300                 }
1301                 if (Token::Match(tok3->tokAt(-2), "[<,] %name% <") && templateParameters(tok3) > 0)
1302                     ++indentlevel;
1303                 else if (indentlevel > 0 && Token::Match(tok3, "> [,>]"))
1304                     --indentlevel;
1305                 bool constconst = tok3->str() == "const" && tok3->strAt(1) == "const";
1306                 if (!constconst) {
1307                     templateMatchPattern += tok3->str();
1308                     templateMatchPattern += ' ';
1309                 }
1310                 if (indentlevel == 0 && Token::Match(tok3->previous(), "[<,]"))
1311                     typesUsedInTemplateInstantiation.push_back(tok3);
1312                 // add additional type information
1313                 if (!constconst && tok3->str() != "class") {
1314                     if (tok3->isUnsigned())
1315                         typeForNewName += "unsigned";
1316                     else if (tok3->isSigned())
1317                         typeForNewName += "signed";
1318                     if (tok3->isLong())
1319                         typeForNewName += "long";
1320                     typeForNewName += tok3->str();
1321                 }
1322             }
1323             templateMatchPattern += ">";
1324     
1325             if (typeForNewName.empty() || typeParametersInDeclaration.size() != typesUsedInTemplateInstantiation.size()) {
1326                 if (printDebug && errorlogger) {
1327                     std::list<const Token *> callstack(1, tok2);
1328                     errorlogger->reportErr(ErrorLogger::ErrorMessage(callstack, &tokenlist, Severity::debug, "debug",
1329                                            "Failed to instantiate template \"" + name + "\". The checking continues anyway.", false));
1330                 }
1331                 if (typeForNewName.empty())
1332                     continue;
1333                 break;
1334             }
1335     
1336             // New classname/funcname..
1337             const std::string newName(name + "<" + typeForNewName + ">");
1338     
1339             if (expandedtemplates.find(newName) == expandedtemplates.end()) {
1340                 expandedtemplates.insert(newName);
1341                 TemplateSimplifier::expandTemplate(tokenlist, tok,name,typeParametersInDeclaration,newName,typesUsedInTemplateInstantiation,templateInstantiations);
1342                 instantiated = true;
1343             }
1344     
1345             // Replace all these template usages..
1346             std::list< std::pair<Token *, Token *> > removeTokens;
1347             for (Token *tok4 = tok2; tok4; tok4 = tok4->next()) {
1348                 if (Token::simpleMatch(tok4, templateMatchPattern.c_str())) {
1349                     Token * tok5 = tok4->tokAt(2);
1350                     unsigned int typeCountInInstantiation = 1U; // There is always at least one type
1351                     const Token *typetok = (!typesUsedInTemplateInstantiation.empty()) ? typesUsedInTemplateInstantiation[0] : 0;
1352                     unsigned int indentlevel5 = 0;  // indentlevel for tok5
1353                     while (tok5 && (indentlevel5 > 0 || tok5->str() != ">")) {
1354                         if (tok5->str() == "<" && templateParameters(tok5) > 0)
1355                             ++indentlevel5;
1356                         else if (indentlevel5 > 0 && Token::Match(tok5, "> [,>]"))
1357                             --indentlevel5;
1358                         else if (indentlevel5 == 0) {
1359                             if (tok5->str() != ",") {
1360                                 if (!typetok ||
1361                                     tok5->isUnsigned() != typetok->isUnsigned() ||
1362                                     tok5->isSigned() != typetok->isSigned() ||
1363                                     tok5->isLong() != typetok->isLong()) {
1364                                     break;
1365                                 }
1366     
1367                                 typetok = typetok ? typetok->next() : 0;
1368                             } else {
1369                                 typetok = (typeCountInInstantiation < typesUsedInTemplateInstantiation.size()) ? typesUsedInTemplateInstantiation[typeCountInInstantiation] : 0;
1370                                 ++typeCountInInstantiation;
1371                             }
1372                         }
1373                         tok5 = tok5->next();
1374                     }
1375     
1376                     // matching template usage => replace tokens..
1377                     // Foo < int >  =>  Foo<int>
1378                     if (tok5 && tok5->str() == ">" && typeCountInInstantiation == typesUsedInTemplateInstantiation.size()) {
1379                         tok4->str(newName);
1380                         for (Token *tok6 = tok4->next(); tok6 != tok5; tok6 = tok6->next()) {
1381                             if (tok6->isName())
1382                                 templateInstantiations.remove(tok6);
1383                         }
1384                         removeTokens.push_back(std::pair<Token*,Token*>(tok4, tok5->next()));
1385                     }
1386     
1387                     tok4 = tok5;
1388                     if (!tok4)
1389                         break;
1390                 }
1391             }
1392             while (!removeTokens.empty()) {
1393                 Token::eraseTokens(removeTokens.back().first, removeTokens.back().second);
1394                 removeTokens.pop_back();
1395             }
1396         }
1397     
1398         // Template has been instantiated .. then remove the template declaration
1399         return instantiated;
1400     }

1403     void TemplateSimplifier::simplifyTemplates(
1404         TokenList& tokenlist,
1405         ErrorLogger* errorlogger,
1406         const Settings *_settings,
1407         bool &_codeWithTemplates
1408     )
1409     {
1410     
1411         std::set<std::string> expandedtemplates(TemplateSimplifier::expandSpecialized(tokenlist.front()));
1412     
1413         if (TemplateSimplifier::getTemplateDeclarations(tokenlist.front(), _codeWithTemplates).empty())
1414             return;
1415     
1416         // There are templates..
1417         // Remove "typename" unless used in template arguments..
1418         for (Token *tok = tokenlist.front(); tok; tok = tok->next()) {
1419             if (tok->str() == "typename")
1420                 tok->deleteThis();
1421     
1422             if (Token::simpleMatch(tok, "template <")) {
1423                 while (tok && tok->str() != ">")
1424                     tok = tok->next();
1425                 if (!tok)
1426                     break;
1427             }
1428         }
1429     
1430         std::list<Token *> templates(TemplateSimplifier::getTemplateDeclarations(tokenlist.front(), _codeWithTemplates));
1431     
1432         // Locate possible instantiations of templates..
1433         std::list<Token *> templateInstantiations(TemplateSimplifier::getTemplateInstantiations(tokenlist.front()));
1434     
1435         // No template instantiations? Then return.
1436         if (templateInstantiations.empty())
1437             return;
1438     
1439         // Template arguments with default values
1440         TemplateSimplifier::useDefaultArgumentValues(templates, &templateInstantiations);
1441     
1442         // expand templates
1443         //bool done = false;
1444         //while (!done)
1445         {
1446             //done = true;
1447             std::list<Token *> templates2;
1448             for (std::list<Token *>::reverse_iterator iter1 = templates.rbegin(); iter1 != templates.rend(); ++iter1) {
1449                 bool instantiated = TemplateSimplifier::simplifyTemplateInstantiations(tokenlist,
1450                                     errorlogger,
1451                                     _settings,
1452                                     *iter1,
1453                                     templateInstantiations,
1454                                     expandedtemplates);
1455                 if (instantiated)
1456                     templates2.push_back(*iter1);
1457             }
1458     
1459             for (std::list<Token *>::const_iterator it = templates2.begin(); it != templates2.end(); ++it) {
1460                 std::list<Token *>::iterator it1 = std::find(templates.begin(), templates.end(), *it);
1461                 if (it1 != templates.end()) {
1462                     templates.erase(it1);
1463                     removeTemplate(*it);
1464                 }
1465             }
1466         }
1467     }
