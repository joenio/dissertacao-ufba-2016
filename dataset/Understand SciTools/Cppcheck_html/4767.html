19     #ifndef FILELIST_H
20     #define FILELIST_H
21     
22     #include <QFileInfoList>
23     #include <QStringList>
24     
25     /**
26      * @brief A class for listing files and directories to check.
27      * This class creates a list of files to check. If directory name is given then
28      * all files in the directory matching the filter will be added. The directory
29      * can be also added recursively when all files in subdirectories are added too.
30      * The filenames are matched against the filter and only those files whose
31      * filename extension is included in the filter list are added.
32      *
33      * This class also handles filtering of paths against ignore filters given. If
34      * there is ignore filters then only paths not matching those filters are
35      * returned.
36      */
37     class FileList {
38     public:
39     
40         /**
41         * @brief Add filename to the list.
42         * @param filepath Full path to the file.
43         */
44         void AddFile(const QString &filepath);
45     
46         /**
47         * @brief Add files in the directory to the list.
48         * @param directory Full pathname to directory to add.
49         * @param recursive If true also files in subdirectories are added.
50         */
51         void AddDirectory(const QString &directory, bool recursive = false);
52     
53         /**
54         * @brief Add list of filenames and directories to the list.
55         * @param paths List of paths to add.
56         */
57         void AddPathList(const QStringList &paths);
58     
59         /**
60         * @brief Return list of filenames (to check).
61         * @return list of filenames to check.
62         */
63         QStringList GetFileList() const;
64     
65         /**
66         * @brief Add list of paths to exclusion list.
67         * @param paths Paths to exclude.
68         */
69         void AddExcludeList(const QStringList &paths);
70     
71         /**
72         * @brief Return list of default filename extensions included.
73         * @return list of default filename extensions included.
74         */
75         static QStringList GetDefaultFilters();
76     
77     protected:
78     
79         /**
80         * @brief Test if filename matches the filename extensions filtering.
81         * @return true if filename matches filtering.
82         */
83         static bool FilterMatches(const QFileInfo &inf);
84     
85         /**
86         * @brief Get filtered list of paths.
87         * This method takes the list of paths and applies the exclude lists to
88         * it. And then returns the list of paths that did not match the
89         * exclude filters.
90         * @return Filtered list of paths.
91         */
92         QStringList ApplyExcludeList() const;
93     
94         /**
95         * @brief Test if path matches any of the exclude filters.
96         * @param path Path to test against exclude filters.
97         * @return true if any of the filters matches, false otherwise.
98         */
99         bool Match(const QString &path) const;
100     
101     private:
102         QFileInfoList mFileList;
103         QStringList mExcludedPaths;
104     };
105     
106     #endif // FILELIST_H
