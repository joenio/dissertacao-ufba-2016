19     #ifndef filelisterH
20     #define filelisterH
21     
22     #include <string>
23     #include <set>
24     #include <map>
25     
26     class PathMatch;
27     
28     /// @addtogroup CLI
29     /// @{
30     
31     /** @brief Cross-platform FileLister */
32     class FileLister {
33     public:
34         /**
35          * @brief Recursively add source files to a map.
36          * Add source files from given directory and all subdirectries to the
37          * given map. Only files with accepted extensions
38          * (*.c;*.cpp;*.cxx;*.c++;*.cc;*.txx) are added.
39          * @param files output map that associates the size of each file with its name
40          * @param path root path
41          * @param ignored ignored paths
42          */
43         static void recursiveAddFiles(std::map<std::string, std::size_t> &files, const std::string &path, const PathMatch& ignored) {
44             const std::set<std::string> extra;
45             recursiveAddFiles(files, path, extra, ignored);
46         }
47     
48         /**
49          * @brief Recursively add source files to a map.
50          * Add source files from given directory and all subdirectries to the
51          * given map. Only files with accepted extensions
52          * (*.c;*.cpp;*.cxx;*.c++;*.cc;*.txx) are added.
53          * @param files output map that associates the size of each file with its name
54          * @param path root path
55          * @param extra Extra file extensions
56          * @param ignored ignored paths
57          */
58         static void recursiveAddFiles(std::map<std::string, std::size_t> &files, const std::string &path, const std::set<std::string> &extra, const PathMatch& ignored);
59     
60         /**
61          * @brief (Recursively) add source files to a map.
62          * Add source files from given directory and all subdirectries to the
63          * given map. Only files with accepted extensions
64          * (*.c;*.cpp;*.cxx;*.c++;*.cc;*.txx) are added.
65          * @param files output map that associates the size of each file with its name
66          * @param path root path
67          * @param extra Extra file extensions
68          * @param recursive Enable recursion
69          * @param ignored ignored paths
70          */
71         static void addFiles(std::map<std::string, std::size_t> &files, const std::string &path, const std::set<std::string> &extra, bool recursive, const PathMatch& ignored);
72     
73         /**
74          * @brief Is given path a directory?
75          * @return returns true if the path is a directory
76          */
77         static bool isDirectory(const std::string &path);
78     
79         /**
80           * @brief Check if the given path is a file and if it exists?
81           * @return true if path points to file and the file exists.
82           */
83         static bool fileExists(const std::string &path);
84     };
85     
86     /// @}
87     
88     #endif // #ifndef filelisterH
