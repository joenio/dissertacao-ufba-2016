61     class CPPCHECKLIB Type {
62     public:
63         const Token* classDef; // Points to "class" token
64         const Scope* classScope;
65         const Scope* enclosingScope;
66         enum NeedInitialization {
67             Unknown, True, False

70         class BaseInfo {
71         public:
72             BaseInfo() :
73                 type(nullptr), nameTok(nullptr), access(Public), isVirtual(false) {
74             }
75     
76             std::string name;
77             const Type* type;
78             const Token* nameTok;
79             AccessControl access;  // public/protected/private
80             bool isVirtual;
81             // allow ordering within containers
82             bool operator<(const BaseInfo& rhs) const {
83                 return this->type < rhs.type;
84             }
85         };
86     
87         struct FriendInfo {
88             FriendInfo() :
89                 nameStart(nullptr), nameEnd(nullptr), type(nullptr) {
90             }
91     
92             const Token* nameStart;
93             const Token* nameEnd;
94             std::string name;
95             const Type* type;
96         };
97     
98         std::vector<BaseInfo> derivedFrom;
99         std::list<FriendInfo> friendList;
100     
101         Type(const Token* classDef_ = 0, const Scope* classScope_ = 0, const Scope* enclosingScope_ = 0) :
102             classDef(classDef_),
103             classScope(classScope_),
104             enclosingScope(enclosingScope_),
105             needInitialization(Unknown) {
106         }
107     
108         const std::string& name() const {
109             const Token* next = classDef->next();
110             if (next->isName())
111                 return next->str();
112             return emptyString;
113         }
114     
115         const Token *initBaseInfo(const Token *tok, const Token *tok1);
116     
117         const Function* getFunction(const std::string& funcName) const;
118     
119         /**
120         * Check for circulare dependencies, i.e. loops within the class hierarchie
121         * @param ancestors list of ancestors. For internal usage only, clients should not supply this argument.
122         * @return true if there is a circular dependency
123         */
124         bool hasCircularDependencies(std::set<BaseInfo>* ancestors = nullptr) const;
125     
126         /**
127         * Check for dependency
128         * @param ancestor potential ancestor
129         * @return true if there is a dependency
130         */
131         bool findDependency(const Type* ancestor) const;

134     /** @brief Information about a member variable. */
135     class CPPCHECKLIB Variable {
136         /** @brief flags mask used to access specific bit. */
137         enum {
138             fIsMutable   = (1 << 0), /** @brief mutable variable */
139             fIsStatic    = (1 << 1), /** @brief static variable */
140             fIsConst     = (1 << 2), /** @brief const variable */
141             fIsExtern    = (1 << 3), /** @brief extern variable */
142             fIsClass     = (1 << 4), /** @brief user defined type */
143             fIsArray     = (1 << 5), /** @brief array variable */
144             fIsPointer   = (1 << 6), /** @brief pointer variable */
145             fIsReference = (1 << 7), /** @brief reference variable */
146             fIsRValueRef = (1 << 8), /** @brief rvalue reference variable */
147             fHasDefault  = (1 << 9), /** @brief function argument with default value */
148             fIsStlType   = (1 << 10), /** @brief STL type ('std::') */
149             fIsStlString = (1 << 11), /** @brief std::string|wstring|basic_string&lt;T&gt;|u16string|u32string */
150             fIsIntType   = (1 << 12), /** @brief Integral type */
151             fIsFloatType = (1 << 13)  /** @brief Floating point type */
152         };
153     
154         /**
155          * Get specified flag state.
156          * @param flag_ flag to get state of
157          * @return true if flag set or false in flag not set
158          */
159         bool getFlag(unsigned int flag_) const {
160             return bool((_flags & flag_) != 0);
161         }
162     
163         /**
164          * Set specified flag state.
165          * @param flag_ flag to set state
166          * @param state_ new state of flag
167          */
168         void setFlag(unsigned int flag_, bool state_) {
169             _flags = state_ ? _flags | flag_ : _flags & ~flag_;
170         }
171     
172         /**
173          * @brief parse and save array dimension information
174          * @param lib Library instance
175          * @return true if array, false if not
176          */
177         bool arrayDimensions(const Library* lib);
178     
179     public:
180         Variable(const Token *name_, const Token *start_, const Token *end_,
181                  std::size_t index_, AccessControl access_, const Type *type_,
182                  const Scope *scope_, const Library* lib)
183             : _name(name_),
184               _start(start_),
185               _end(end_),
186               _index(index_),
187               _access(access_),
188               _flags(0),
189               _type(type_),
190               _scope(scope_) {
191             evaluate(lib);
192         }
193     
194         /**
195          * Get name token.
196          * @return name token
197          */
198         const Token *nameToken() const {
199             return _name;
200         }
201     
202         /**
203          * Get type start token.
204          * The type start token doesn't account 'static' and 'const' qualifiers
205          * E.g.:
206          *     static const int * const p = ...;
207          * type start token ^
208          * @return type start token
209          */
210         const Token *typeStartToken() const {
211             return _start;
212         }
213     
214         /**
215          * Get type end token.
216          * The type end token doesn't account the forward 'const' qualifier
217          * E.g.:
218          *     static const int * const p = ...;
219          *       type end token ^
220          * @return type end token
221          */
222         const Token *typeEndToken() const {
223             return _end;
224         }
225     
226         /**
227          * Get end token of variable declaration
228          * E.g.
229          * int i[2][3] = ...
230          *   end token ^
231          * @return variable declaration end token
232          */
233         const Token *declEndToken() const;
234     
235         /**
236          * Get name string.
237          * @return name string
238          */
239         const std::string &name() const {
240             // name may not exist for function arguments
241             if (_name)
242                 return _name->str();
243     
244             return emptyString;
245         }
246     
247         /**
248          * Get declaration ID (varId used for variable in its declaration).
249          * @return declaration ID
250          */
251         unsigned int declarationId() const {
252             // name may not exist for function arguments
253             if (_name)
254                 return _name->varId();
255     
256             return 0;
257         }
258     
259         /**
260          * Get index of variable in declared order.
261          * @return variable index
262          */
263         std::size_t index() const {
264             return _index;
265         }
266     
267         /**
268          * Is variable public.
269          * @return true if public, false if not
270          */
271         bool isPublic() const {
272             return _access == Public;
273         }
274     
275         /**
276          * Is variable protected.
277          * @return true if protected, false if not
278          */
279         bool isProtected() const {
280             return _access == Protected;
281         }
282     
283         /**
284          * Is variable private.
285          * @return true if private, false if not

288             return _access == Private;
289         }
290     
291         /**

295         bool isGlobal() const {
296             return _access == Global;
297         }
298     
299         /**
300          * Is variable in a namespace.
301          * @return true if in a namespace, false if not
302          */
303         bool isNamespace() const {
304             return _access == Namespace;
305         }
306     
307         /**
308          * Is variable a function argument.
309          * @return true if a function argument, false if not
310          */
311         bool isArgument() const {
312             return _access == Argument;

314     
315         /**
316          * Is variable local.
317          * @return true if local, false if not

319         bool isLocal() const {
320             return (_access == Local) && !isExtern();
321         }

324          * Is variable mutable.
325          * @return true if mutable, false if not

327         bool isMutable() const {

355         /**

359         bool isThrow() const {

361         }
362

367         bool isClass() const {
368             return getFlag(fIsClass);
369         }
370     
371         /**

375         bool isArray() const {

377         }
378     
379         /**
380          * Is pointer variable.
381          * @return true if pointer, false otherwise

383         bool isPointer() const {

385         }
386     
387         /**
388          * Is variable a pointer to an array
389          * @return true if pointer to array, false otherwise
390          */
391         bool isPointerToArray() const {

393         }
394     
395         /**
396          * Is variable an array of pointers
397          * @return true if array or pointers, false otherwise
398          */
399         bool isPointerArray() const;
400     
401         /**
402          * Is array or pointer variable.
403          * @return true if pointer or array, false otherwise
404          */
405         bool isArrayOrPointer() const {
406             return getFlag(fIsArray) || getFlag(fIsPointer);
407         }
408     
409         /**
410          * Is reference variable.
411          * @return true if reference, false otherwise
412          */
413         bool isReference() const {
414             return getFlag(fIsReference);
415         }
416     
417         /**
418          * Is reference variable.
419          * @return true if reference, false otherwise
420          */
421         bool isRValueReference() const {
422             return getFlag(fIsRValueRef);
423         }
424     
425         /**
426          * Does variable have a default value.

428          */
429         bool hasDefault() const {

431         }

439         }

447         }

449         /**

481         /**
482         * Checks if the variable is an STL type ('std::')
483         * E.g.:
484         *   std::string s;
485         *   ...
486         *   sVar->isStlType() == true
487         * @return true if it is an stl type and its type matches any of the types in 'stlTypes'
488         */
489         bool isStlType() const {
490             return getFlag(fIsStlType);
491         }
492     
493         /**
494          * Checks if the variable is an STL type ('std::')
495          * E.g.:
496          *   std::string s;
497          *   ...
498          *   sVar->isStlType() == true
499          * @return true if it is an stl type and its type matches any of the types in 'stlTypes'
500          */
501         bool isStlStringType() const {
502             return getFlag(fIsStlString);
503         }
504     
505         /**
506          * Checks if the variable is of any of the STL types passed as arguments ('std::')
507          * E.g.:
508          *   std::string s;
509          *   ...
510          *   const char *str[] = {"string", "wstring"};
511          *   sVar->isStlType(str) == true
512          * @param stlType stl type
513          * @return true if it is an stl type and its type matches any of the types in 'stlTypes'
514          */
515         bool isStlType(const std::string& stlType) const {
516             return isStlType() && stlType==_start->strAt(2);
517         }
518     
519         /**
520          * Checks if the variable is of any of the STL types passed as arguments ('std::')
521          * E.g.:
522          *   std::string s;
523          *   ...
524          *   const std::set<std::string> str = make_container< std::set<std::string> >() << "string" << "wstring";
525          *   sVar->isStlType(str) == true
526          * @param stlTypes set of stl types
527          * @return true if it is an stl type and its type matches any of the types in 'stlTypes'
528          */
529         bool isStlType(const std::set<std::string>& stlTypes) const {
530             return isStlType() && stlTypes.find(_start->strAt(2))!=stlTypes.end();
531         }
532     
533         /**
534         * Determine whether it's a floating number type
535         * @return true if the type is known and it's a floating type (float, double and long double) or a pointer/array to it
536         */
537         bool isFloatingType() const {
538             return getFlag(fIsFloatType);
539         }
540     
541         /**
542          * Determine whether it's an integral number type
543          * @return true if the type is known and it's an integral type (bool, char, short, int, long long and their unsigned counter parts) or a pointer/array to it
544          */
545         bool isIntegralType() const {
546             return getFlag(fIsIntType);
547         }

564
