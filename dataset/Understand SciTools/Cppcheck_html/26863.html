68     void CheckStl::iterators()
69     {
70         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
71     
72         for (unsigned int iteratorId = 1; iteratorId < symbolDatabase->getVariableListSize(); iteratorId++) {
73             const Variable* var = symbolDatabase->getVariableFromVarId(iteratorId);
74     
75             // Check that its an iterator
76             if (!var || !var->isLocal() || !Token::Match(var->typeEndToken(), "iterator|const_iterator|reverse_iterator|const_reverse_iterator|auto"))
77                 continue;
78     
79             if (var->typeEndToken()->str() == "auto") {
80                 if (Token::Match(var->typeEndToken(), "auto %name% ; %name% = %var% . %name% ( )")) {
81                     const Token* containertok = var->typeEndToken()->tokAt(5);
82                     if (!containertok->variable())
83                         continue;
84     
85                     const Library::Container* container = _settings->library.detectContainer(containertok->variable()->typeStartToken());
86                     if (!container)
87                         continue;
88     
89                     Library::Container::Yield yield = container->getYield(containertok->strAt(2));
90                     if (yield != Library::Container::END_ITERATOR && yield != Library::Container::START_ITERATOR && yield != Library::Container::ITERATOR)
91                         continue;
92                 } else
93                     continue;
94             }
95     
96             if (var->type()) { // If it is defined, ensure that it is defined like an iterator
97                 // look for operator* and operator++
98                 const Function* end = var->type()->getFunction("operator*");
99                 const Function* incOperator = var->type()->getFunction("operator++");
100                 if (!end || end->argCount() > 0 || !incOperator)
101                     continue;
102             }
103     
104             // the validIterator flag says if the iterator has a valid value or not
105             bool validIterator = Token::Match(var->nameToken()->next(), "[(=:]");
106             const Scope* invalidationScope = 0;
107     
108             // The container this iterator can be used with
109             const Variable* container = 0;
110             const Scope* containerAssignScope = 0;
111     
112             // When "validatingToken" is reached the validIterator is set to true
113             const Token* validatingToken = 0;
114     
115             const Token* eraseToken = 0;
116     
117             // Scan through the rest of the code and see if the iterator is
118             // used against other containers.
119             for (const Token *tok2 = var->nameToken(); tok2 && tok2 != var->scope()->classEnd; tok2 = tok2->next()) {
120                 if (invalidationScope && tok2 == invalidationScope->classEnd)
121                     validIterator = true; // Assume that the iterator becomes valid again
122                 if (containerAssignScope && tok2 == containerAssignScope->classEnd)
123                     container = 0; // We don't know which containers might be used with the iterator
124     
125                 if (tok2 == validatingToken)
126                     validIterator = true;
127     
128                 // Is iterator compared against different container?
129                 if (Token::Match(tok2, "%varid% !=|== %name% . end|rend|cend|crend ( )", iteratorId) && container && tok2->tokAt(2)->varId() != container->declarationId()) {
130                     iteratorsError(tok2, container->name(), tok2->strAt(2));
131                     tok2 = tok2->tokAt(6);
132                 }
133     
134                 // Is the iterator used in a insert/erase operation?
135                 else if (Token::Match(tok2, "%name% . insert|erase ( *| %varid% )|,", iteratorId)) {
136                     const Token* itTok = tok2->tokAt(4);
137                     if (itTok->str() == "*") {
138                         if (tok2->strAt(2) == "insert")
139                             continue;
140     
141                         itTok = itTok->next();
142                     }
143                     // It is bad to insert/erase an invalid iterator
144                     if (!validIterator)
145                         invalidIteratorError(tok2, itTok->str());
146     
147                     // If insert/erase is used on different container then
148                     // report an error
149                     if (container && tok2->varId() != container->declarationId()) {
150                         // skip error message if container is a set..
151                         const Variable *variableInfo = tok2->variable();
152                         const Token *decltok = variableInfo ? variableInfo->typeStartToken() : nullptr;
153     
154                         if (Token::simpleMatch(decltok, "std :: set"))
155                             continue; // No warning
156     
157                         // skip error message if the iterator is erased/inserted by value
158                         if (itTok->previous()->str() == "*")
159                             continue;
160     
161                         // Show error message, mismatching iterator is used.
162                         iteratorsError(tok2, container->name(), tok2->str());
163                     }
164     
165                     // invalidate the iterator if it is erased
166                     else if (tok2->strAt(2) == "erase" && (tok2->strAt(4) != "*" || (container && tok2->varId() == container->declarationId()))) {
167                         validIterator = false;
168                         eraseToken = tok2;
169                         invalidationScope = tok2->scope();
170                     }
171     
172                     // skip the operation
173                     tok2 = itTok->next();
174                 }
175     
176                 // it = foo.erase(..
177                 // taking the result of an erase is ok
178                 else if (Token::Match(tok2, "%varid% = %name% .", iteratorId) &&
179                          Token::simpleMatch(skipMembers(tok2->tokAt(2)), "erase (")) {
180                     // the returned iterator is valid
181                     validatingToken = tok2->linkAt(5);
182                     tok2 = tok2->tokAt(5);
183                 }
184     
185                 // Reassign the iterator
186                 else if (Token::Match(tok2, "%varid% = %name% . begin|rbegin|cbegin|crbegin|find (", iteratorId)) {
187                     validatingToken = tok2->linkAt(5);
188                     container = tok2->tokAt(2)->variable();
189                     containerAssignScope = tok2->scope();
190     
191                     // skip ahead
192                     tok2 = tok2->tokAt(5);
193                 }
194     
195                 // Reassign the iterator
196                 else if (Token::Match(tok2, "%varid% =", iteratorId)) {
197                     // Assume that the iterator becomes valid.
198                     // TODO: add checking that checks if the iterator becomes valid or not
199                     validatingToken = Token::findmatch(tok2->tokAt(2), "[;)]");
200     
201                     // skip ahead
202                     tok2 = tok2->tokAt(2);
203                 }
204     
205                 // Passing iterator to function. Iterator might be initialized
206                 else if (Token::Match(tok2, "%varid% ,|)", iteratorId)) {
207                     validIterator = true;
208                 }
209     
210                 // Dereferencing invalid iterator?
211                 else if (!validIterator && Token::Match(tok2, "* %varid%", iteratorId)) {
212                     dereferenceErasedError(eraseToken, tok2, tok2->strAt(1));
213                     tok2 = tok2->next();
214                 } else if (!validIterator && Token::Match(tok2, "%varid% . %name%", iteratorId)) {
215                     dereferenceErasedError(eraseToken, tok2, tok2->str());
216                     tok2 = tok2->tokAt(2);
217                 }
218     
219                 // bailout handling. Assume that the iterator becomes valid if we see return/break.
220                 // TODO: better handling
221                 else if (Token::Match(tok2, "return|break")) {
222                     validatingToken = Token::findsimplematch(tok2->next(), ";");
223                 }
224     
225                 // bailout handling. Assume that the iterator becomes valid if we see else.
226                 // TODO: better handling
227                 else if (tok2 && tok2->str() == "else") {
228                     validIterator = true;
229                 }
230             }
231         }
232     }

265     void CheckStl::mismatchingContainers()
266     {
267         // Check if different containers are used in various calls of standard functions
268         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
269         const std::size_t functions = symbolDatabase->functionScopes.size();
270         for (std::size_t ii = 0; ii < functions; ++ii) {
271             const Scope * scope = symbolDatabase->functionScopes[ii];
272             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
273                 if (!Token::Match(tok, "std :: %type% ( !!)"))
274                     continue;
275                 const Token* arg1 = tok->tokAt(4);
276     
277                 // TODO: If iterator variables are used instead then there are false negatives.
278                 if (Token::Match(arg1, pattern2.c_str()) && algorithm2.find(tok->strAt(2)) != algorithm2.end()) {
279                     if (arg1->str() != arg1->strAt(6)) {
280                         mismatchingContainersError(arg1);
281                     }
282                 } else if (algorithm22.find(tok->strAt(2)) != algorithm22.end()) {
283                     if (Token::Match(arg1, pattern2.c_str()) && arg1->str() != arg1->strAt(6))
284                         mismatchingContainersError(arg1);
285                     // Find third parameter
286                     const Token* arg3 = arg1;
287                     for (unsigned int i = 0; i < 2 && arg3; i++)
288                         arg3 = arg3->nextArgument();
289                     if (Token::Match(arg3, pattern2.c_str()) && arg3->str() != arg3->strAt(6))
290                         mismatchingContainersError(arg3);
291                 } else if (Token::Match(arg1, pattern1x1_1.c_str()) && algorithm1x1.find(tok->strAt(2)) != algorithm1x1.end()) {
292                     // Find third parameter
293                     const Token *arg3 = arg1->tokAt(6)->nextArgument();
294                     if (Token::Match(arg3, pattern1x1_2.c_str())) {
295                         if (arg1->str() != arg3->str()) {
296                             mismatchingContainersError(arg1);
297                         }
298                     }
299                 }
300                 tok = arg1->linkAt(-1);
301             }
302         }
303         for (unsigned int varid = 0; varid < symbolDatabase->getVariableListSize(); varid++) {
304             const Variable* var = symbolDatabase->getVariableFromVarId(varid);
305             if (var && var->isStlStringType() && Token::Match(var->nameToken(), "%var% (") && Token::Match(var->nameToken()->tokAt(2), pattern2.c_str())) {
306                 if (var->nameToken()->strAt(2) != var->nameToken()->strAt(8)) {
307                     mismatchingContainersError(var->nameToken());
308                 }
309             }
310         }
311     }

314     void CheckStl::stlOutOfBounds()
315     {
316         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
317     
318         // Scan through all scopes..
319         for (std::list<Scope>::const_iterator i = symbolDatabase->scopeList.begin(); i != symbolDatabase->scopeList.end(); ++i) {
320             const Token* tok = i->classDef;
321             // only interested in conditions
322             if ((i->type != Scope::eFor && i->type != Scope::eWhile && i->type != Scope::eIf && i->type != Scope::eDo) || !tok)
323                 continue;
324     
325             if (i->type == Scope::eFor)
326                 tok = Token::findsimplematch(tok->tokAt(2), ";");
327             else if (i->type == Scope::eDo) {
328                 tok = tok->linkAt(1)->tokAt(2);
329             } else
330                 tok = tok->next();
331     
332             if (!tok)
333                 continue;
334             tok = tok->next();
335     
336             // check if the for loop condition is wrong
337             if (Token::Match(tok, "%var% <= %var% . %name% ( ) ;|)|%oror%")) {
338                 // Is it a vector?
339                 const Variable *var = tok->tokAt(2)->variable();
340                 if (!var)
341                     continue;
342     
343                 const Library::Container* container = _settings->library.detectContainer(var->typeStartToken());
344                 if (!container)
345                     continue;
346     
347                 if (container->getYield(tok->strAt(4)) != Library::Container::SIZE)
348                     continue;
349     
350                 // variable id for loop variable.
351                 const unsigned int numId = tok->varId();
352     
353                 // variable id for the container variable
354                 const unsigned int declarationId = var->declarationId();
355     
356                 for (const Token *tok3 = i->classStart; tok3 && tok3 != i->classEnd; tok3 = tok3->next()) {
357                     if (tok3->varId() == declarationId) {
358                         tok3 = tok3->next();
359                         if (Token::Match(tok3, ". %name% ( )")) {
360                             if (container->getYield(tok3->strAt(1)) == Library::Container::SIZE)
361                                 break;
362                         } else if (container->arrayLike_indexOp && Token::Match(tok3, "[ %varid% ]", numId))
363                             stlOutOfBoundsError(tok3, tok3->strAt(1), var->name(), false);
364                         else if (Token::Match(tok3, ". %name% ( %varid% )", numId)) {
365                             Library::Container::Yield yield = container->getYield(tok3->strAt(1));
366                             if (yield == Library::Container::AT_INDEX)
367                                 stlOutOfBoundsError(tok3, tok3->strAt(3), var->name(), true);
368                         }
369                     }
370                 }
371                 continue;
372             }
373         }
374     }

384     void CheckStl::erase()
385     {
386         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
387     
388         for (std::list<Scope>::const_iterator i = symbolDatabase->scopeList.begin(); i != symbolDatabase->scopeList.end(); ++i) {
389             if (i->type == Scope::eFor && Token::simpleMatch(i->classDef, "for (")) {
390                 const Token *tok = i->classDef->linkAt(1);
391                 if (!Token::Match(tok->tokAt(-3), "; ++| %var% ++| ) {"))
392                     continue;
393                 tok = tok->previous();
394                 if (!tok->isName())
395                     tok = tok->previous();
396                 eraseCheckLoopVar(*i, tok->variable());
397             } else if (i->type == Scope::eWhile && Token::Match(i->classDef, "while ( %var% !=")) {
398                 eraseCheckLoopVar(*i, i->classDef->tokAt(2)->variable());
399             }
400         }

403     void CheckStl::eraseCheckLoopVar(const Scope &scope, const Variable *var)
404     {
405         if (!var || !Token::simpleMatch(var->typeEndToken(), "iterator"))
406             return;
407         for (const Token *tok = scope.classStart; tok != scope.classEnd; tok = tok->next()) {
408             if (tok->str() != "(")
409                 continue;
410             if (!Token::Match(tok->tokAt(-2), ". erase ( ++| %varid% )", var->declarationId()))
411                 continue;
412             if (Token::simpleMatch(tok->astParent(), "="))
413                 continue;
414             // Iterator is invalid..
415             unsigned int indentlevel = 0U;
416             const Token *tok2 = tok->link();
417             for (; tok2 != scope.classEnd; tok2 = tok2->next()) {
418                 if (tok2->str() == "{") {
419                     ++indentlevel;
420                     continue;
421                 }
422                 if (tok2->str() == "}") {
423                     if (indentlevel > 0U)
424                         --indentlevel;
425                     else if (Token::simpleMatch(tok2, "} else {"))
426                         tok2 = tok2->linkAt(2);
427                     continue;
428                 }
429                 if (tok2->varId() == var->declarationId()) {
430                     if (Token::simpleMatch(tok2->next(), "="))
431                         break;
432                     dereferenceErasedError(tok, tok2, tok2->str());
433                     break;
434                 }
435                 if (indentlevel == 0U && Token::Match(tok2, "break|return|goto"))
436                     break;
437             }
438             if (tok2 == scope.classEnd)
439                 dereferenceErasedError(tok, scope.classDef, var->nameToken()->str());
440         }

443     void CheckStl::pushback()
444     {
445         // Pointer can become invalid after push_back, push_front, reserve or resize..
446         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
447         const std::size_t functions = symbolDatabase->functionScopes.size();
448         for (std::size_t i = 0; i < functions; ++i) {
449             const Scope * scope = symbolDatabase->functionScopes[i];
450             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
451                 if (Token::Match(tok, "%var% = & %var% [")) {
452                     // Skip it directly if it is a pointer or an array
453                     const Token* containerTok = tok->tokAt(3);
454                     if (containerTok->variable() && containerTok->variable()->isArrayOrPointer())
455                         continue;
456     
457                     // Variable id for pointer
458                     const unsigned int pointerId(tok->varId());
459     
460                     bool invalidPointer = false;
461                     const Token* function = nullptr;
462                     const Token* end2 = tok->scope()->classEnd;
463                     for (const Token *tok2 = tok; tok2 != end2; tok2 = tok2->next()) {
464                         // push_back on vector..
465                         if (Token::Match(tok2, "%varid% . push_front|push_back|insert|reserve|resize|clear", containerTok->varId())) {
466                             invalidPointer = true;
467                             function = tok2->tokAt(2);
468                         }
469     
470                         // Using invalid pointer..
471                         if (invalidPointer && tok2->varId() == pointerId) {
472                             bool unknown = false;
473                             if (CheckNullPointer::isPointerDeRef(tok2, unknown))
474                                 invalidPointerError(tok2, function->str(), tok2->str());
475                             break;
476                         }
477                     }
478                 }
479             }
480         }
481     
482         // Iterator becomes invalid after reserve, resize, insert, push_back or push_front..
483         for (unsigned int iteratorId = 1; iteratorId < symbolDatabase->getVariableListSize(); iteratorId++) {
484             const Variable* var = symbolDatabase->getVariableFromVarId(iteratorId);
485     
486             // Check that its an iterator
487             if (!var || !var->isLocal() || !Token::Match(var->typeEndToken(), "iterator|const_iterator|reverse_iterator|const_reverse_iterator"))
488                 continue;
489     
490             // ... on std::vector
491             if (!Token::Match(var->typeStartToken(), "std| ::| vector <"))
492                 continue;
493     
494             // the variable id for the vector
495             unsigned int vectorid = 0;
496     
497             const Token* validatingToken = 0;
498     
499             std::string invalidIterator;
500             const Token* end2 = var->scope()->classEnd;
501             for (const Token *tok2 = var->nameToken(); tok2 != end2; tok2 = tok2->next()) {
502     
503                 if (validatingToken == tok2) {
504                     invalidIterator.clear();
505                     validatingToken = 0;
506                 }
507     
508                 // Using push_back or push_front inside a loop..
509                 if (Token::simpleMatch(tok2, "for (")) {
510                     tok2 = tok2->tokAt(2);
511                 }
512     
513                 if (Token::Match(tok2, "%varid% = %var% . begin|rbegin|cbegin|crbegin ( ) ; %varid% != %var% . end|rend|cend|crend ( ) ; ++| %varid% ++| ) {", iteratorId)) {
514                     // variable id for the loop iterator
515                     const unsigned int varId(tok2->tokAt(2)->varId());
516     
517                     const Token *pushbackTok = nullptr;
518     
519                     // Count { and } for tok3
520                     const Token *tok3 = tok2->tokAt(20);
521                     for (const Token* const end3 = tok3->linkAt(-1); tok3 != end3; tok3 = tok3->next()) {
522                         if (tok3->str() == "break" || tok3->str() == "return") {
523                             pushbackTok = 0;
524                             break;
525                         } else if (Token::Match(tok3, "%varid% . push_front|push_back|insert|reserve|resize|clear|erase (", varId) && !tok3->previous()->isAssignmentOp()) {
526                             if (tok3->strAt(2) != "erase" || (tok3->tokAt(4)->varId() != iteratorId && tok3->tokAt(5)->varId() != iteratorId)) // This case is handled in: CheckStl::iterators()
527                                 pushbackTok = tok3->tokAt(2);
528                         }
529                     }
530     
531                     if (pushbackTok)
532                         invalidIteratorError(pushbackTok, pushbackTok->str(), tok2->str());
533                 }
534     
535                 // Assigning iterator..
536                 if (Token::Match(tok2, "%varid% =", iteratorId)) {
537                     if (Token::Match(tok2->tokAt(2), "%var% . begin|end|rbegin|rend|cbegin|cend|crbegin|crend|insert|erase|find (")) {
538                         if (!invalidIterator.empty() && Token::Match(tok2->tokAt(4), "insert|erase ( *| %varid% )|,", iteratorId)) {
539                             invalidIteratorError(tok2, invalidIterator, var->name());
540                             break;
541                         }
542                         vectorid = tok2->tokAt(2)->varId();
543                         tok2 = tok2->linkAt(5);
544                     } else {
545                         vectorid = 0;
546                     }
547                     invalidIterator = "";
548                 }
549     
550                 // push_back on vector..
551                 if (vectorid > 0 && Token::Match(tok2, "%varid% . push_front|push_back|insert|reserve|resize|clear|erase (", vectorid)) {
552                     if (!invalidIterator.empty() && Token::Match(tok2->tokAt(2), "insert|erase ( *| %varid% ,|)", iteratorId)) {
553                         invalidIteratorError(tok2, invalidIterator, var->name());
554                         break;
555                     }
556     
557                     if (tok2->strAt(2) != "erase" || (tok2->tokAt(4)->varId() != iteratorId && tok2->tokAt(5)->varId() != iteratorId)) // This case is handled in: CheckStl::iterators()
558                         invalidIterator = tok2->strAt(2);
559                     tok2 = tok2->linkAt(3);
560                 }
561     
562                 else if (tok2->str() == "return" || tok2->str() == "throw")
563                     validatingToken = Token::findsimplematch(tok2->next(), ";");
564     
565                 // TODO: instead of bail out for 'else' try to check all execution paths.
566                 else if (tok2->str() == "break" || tok2->str() == "else")
567                     invalidIterator.clear();
568     
569                 // Using invalid iterator..
570                 if (!invalidIterator.empty()) {
571                     if (Token::Match(tok2, "++|--|*|+|-|(|,|=|!= %varid%", iteratorId))
572                         invalidIteratorError(tok2, invalidIterator, tok2->strAt(1));
573                     if (Token::Match(tok2, "%varid% ++|--|+|-|.", iteratorId))
574                         invalidIteratorError(tok2, invalidIterator, tok2->str());
575                 }
576             }
577         }
578     }

627     static bool if_findCompare(const Token * const tokBack)
628     {
629         const Token *tok = tokBack->astParent();
630         if (!tok)
631             return true;
632         if (tok->isComparisonOp())
633             return (!tok->astOperand1()->isNumber() && !tok->astOperand2()->isNumber());
634         if (tok->isArithmeticalOp()) // result is used in some calculation
635             return true;  // TODO: check if there is a comparison of the result somewhere
636         if (tok->str() == ".")
637             return true; // Dereferencing is OK, the programmer might know that the element exists - TODO: An inconclusive warning might be appropriate
638         if (tok->isAssignmentOp())
639             return if_findCompare(tok); // Go one step upwards in the AST
640         return false;
641     }

643     void CheckStl::if_find()
644     {
645         const bool printWarning = _settings->isEnabled("warning");
646         const bool printPerformance = _settings->isEnabled("performance");
647         if (!printWarning && !printPerformance)
648             return;
649     
650         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
651     
652         for (std::list<Scope>::const_iterator i = symbolDatabase->scopeList.begin(); i != symbolDatabase->scopeList.end(); ++i) {
653             if ((i->type != Scope::eIf && i->type != Scope::eWhile) || !i->classDef)
654                 continue;
655     
656             for (const Token *tok = i->classDef->next(); tok->str() != "{"; tok = tok->next()) {
657                 const Token* funcTok = nullptr;
658                 const Library::Container* container = nullptr;
659     
660                 if (Token::Match(tok, "%name% ("))
661                     tok = tok->linkAt(1);
662     
663                 else if (tok->variable() && Token::Match(tok, "%var% . %name% (")) {
664                     container = _settings->library.detectContainer(tok->variable()->typeStartToken());
665                     funcTok = tok->tokAt(2);
666                 }
667     
668                 // check also for vector-like or pointer containers
669                 else if (tok->variable() && tok->astParent() && (tok->astParent()->str() == "*" || tok->astParent()->str() == "[")) {
670                     const Token *tok2 = tok->astParent();
671     
672                     if (!Token::Match(tok2->astParent(), ". %name% ("))
673                         continue;
674     
675                     funcTok = tok2->astParent()->next();
676     
677                     if (tok->variable()->isArrayOrPointer())
678                         container = _settings->library.detectContainer(tok->variable()->typeStartToken());
679                     else { // Container of container - find the inner container
680                         container = _settings->library.detectContainer(tok->variable()->typeStartToken()); // outer container
681                         tok2 = Token::findsimplematch(tok->variable()->typeStartToken(), "<", tok->variable()->typeEndToken());
682                         if (container && container->type_templateArgNo >= 0 && tok2) {
683                             tok2 = tok2->next();
684                             for (int j = 0; j < container->type_templateArgNo; j++)
685                                 tok2 = tok2->nextTemplateArgument();
686     
687                             container = _settings->library.detectContainer(tok2); // innner container
688                         } else
689                             container = nullptr;
690                     }
691                 }
692     
693                 if (container && container->getAction(funcTok->str()) == Library::Container::FIND) {
694                     if (if_findCompare(funcTok->next()))
695                         continue;
696     
697                     if (printWarning && container->getYield(funcTok->str()) == Library::Container::ITERATOR)
698                         if_findError(tok, false);
699                     else if (printPerformance && container->stdStringLike)
700                         if_findError(tok, true);
701                 } else if (printWarning && Token::Match(tok, "std :: find|find_if (")) {
702                     // check that result is checked properly
703                     if (!if_findCompare(tok->tokAt(3))) {
704                         if_findError(tok, false);
705                     }
706                 }
707             }
708         }
709     }

737     void CheckStl::size()
738     {
739         if (!_settings->isEnabled("performance"))
740             return;
741     
742         if (_settings->standards.cpp == Standards::CPP11)
743             return;
744     
745         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
746         const std::size_t functions = symbolDatabase->functionScopes.size();
747         for (std::size_t i = 0; i < functions; ++i) {
748             const Scope * scope = symbolDatabase->functionScopes[i];
749             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
750                 if (Token::Match(tok, "%var% . size ( )") ||
751                     Token::Match(tok, "%name% . %var% . size ( )")) {
752                     // get the variable
753                     const Token *varTok = tok;
754                     if (tok->strAt(2) != "size")
755                         varTok = varTok->tokAt(2);
756     
757                     const Token* const end = varTok->tokAt(5);
758     
759                     // check for comparison to zero
760                     if ((tok->previous() && !tok->previous()->isArithmeticalOp() && Token::Match(end, "==|<=|!=|> 0")) ||
761                         (end->next() && !end->next()->isArithmeticalOp() && Token::Match(tok->tokAt(-2), "0 ==|>=|!=|<"))) {
762                         if (isCpp03ContainerSizeSlow(varTok)) {
763                             sizeError(varTok);
764                             continue;
765                         }
766                     }
767     
768                     // check for comparison to one
769                     if ((tok->previous() && !tok->previous()->isArithmeticalOp() && Token::Match(end, ">=|< 1") && !end->tokAt(2)->isArithmeticalOp()) ||
770                         (end->next() && !end->next()->isArithmeticalOp() && Token::Match(tok->tokAt(-2), "1 <=|>") && !tok->tokAt(-3)->isArithmeticalOp())) {
771                         if (isCpp03ContainerSizeSlow(varTok))
772                             sizeError(varTok);
773                     }
774     
775                     // check for using as boolean expression
776                     else if ((Token::Match(tok->tokAt(-2), "if|while (") && end->str() == ")") ||
777                              (tok->previous()->tokType() == Token::eLogicalOp && Token::Match(end, "&&|)|,|;|%oror%"))) {
778                         if (isCpp03ContainerSizeSlow(varTok))
779                             sizeError(varTok);
780                     }
781                 }
782             }
783         }
784     }

836     void CheckStl::missingComparison()
837     {
838         if (!_settings->isEnabled("warning"))
839             return;
840     
841         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
842     
843         for (std::list<Scope>::const_iterator i = symbolDatabase->scopeList.begin(); i != symbolDatabase->scopeList.end(); ++i) {
844             if (i->type != Scope::eFor || !i->classDef)
845                 continue;
846     
847             for (const Token *tok2 = i->classDef->tokAt(2); tok2 != i->classStart; tok2 = tok2->next()) {
848                 if (tok2->str() == ";")
849                     break;
850     
851                 if (!Token::Match(tok2, "%var% = %name% . begin|rbegin|cbegin|crbegin ( ) ; %name% != %name% . end|rend|cend|crend ( ) ; ++| %name% ++| ) {"))
852                     continue;
853     
854                 // same container
855                 if (tok2->strAt(2) != tok2->strAt(10))
856                     break;
857     
858                 const unsigned int iteratorId(tok2->varId());
859     
860                 // same iterator
861                 if (iteratorId == tok2->tokAt(10)->varId())
862                     break;
863     
864                 // increment iterator
865                 if (!Token::Match(tok2->tokAt(16), "++ %varid% )", iteratorId) &&
866                     !Token::Match(tok2->tokAt(16), "%varid% ++ )", iteratorId)) {
867                     break;
868                 }
869     
870                 const Token *incrementToken = nullptr;
871     
872                 // Parse loop..
873                 for (const Token *tok3 = i->classStart; tok3 != i->classEnd; tok3 = tok3->next()) {
874                     if (Token::Match(tok3, "%varid% ++", iteratorId))
875                         incrementToken = tok3;
876                     else if (Token::Match(tok3->previous(), "++ %varid% !!.", iteratorId))
877                         incrementToken = tok3;
878                     else if (Token::Match(tok3, "%varid% !=|==", iteratorId))
879                         incrementToken = 0;
880                     else if (tok3->str() == "break" || tok3->str() == "return")
881                         incrementToken = 0;
882                     else if (Token::Match(tok3, "%varid% = %name% . insert ( ++| %varid% ++| ,", iteratorId)) {
883                         // skip insertion..
884                         tok3 = tok3->linkAt(6);
885                         if (!tok3)
886                             break;
887                     }
888                 }
889                 if (incrementToken)
890                     missingComparisonError(incrementToken, tok2->tokAt(16));
891             }
892         }
893     }

928     void CheckStl::string_c_str()
929     {
930         const bool printInconclusive = _settings->inconclusive;
931         const bool printPerformance = _settings->isEnabled("performance");
932     
933         const SymbolDatabase* symbolDatabase = _tokenizer->getSymbolDatabase();
934     
935         // Find all functions that take std::string as argument
936         std::multimap<std::string, unsigned int> c_strFuncParam;
937         if (printPerformance) {
938             for (std::list<Scope>::const_iterator scope = symbolDatabase->scopeList.begin(); scope != symbolDatabase->scopeList.end(); ++scope) {
939                 for (std::list<Function>::const_iterator func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
940                     if (c_strFuncParam.erase(func->tokenDef->str()) != 0) { // Check if function with this name was already found
941                         c_strFuncParam.insert(std::make_pair(func->tokenDef->str(), 0)); // Disable, because there are overloads. TODO: Handle overloads
942                         continue;
943                     }
944     
945                     unsigned int numpar = 0;
946                     c_strFuncParam.insert(std::make_pair(func->tokenDef->str(), numpar)); // Insert function as dummy, to indicate that there is at least one function with that name
947                     for (std::list<Variable>::const_iterator var = func->argumentList.cbegin(); var != func->argumentList.cend(); ++var) {
948                         numpar++;
949                         if (var->isStlStringType() && (!var->isReference() || var->isConst()))
950                             c_strFuncParam.insert(std::make_pair(func->tokenDef->str(), numpar));
951                     }
952                 }
953             }
954         }
955     
956         // Try to detect common problems when using string::c_str()
957         for (std::list<Scope>::const_iterator scope = symbolDatabase->scopeList.begin(); scope != symbolDatabase->scopeList.end(); ++scope) {
958             if (scope->type != Scope::eFunction || !scope->function)
959                 continue;
960     
961             enum {charPtr, stdString, stdStringConstRef, Other} returnType = Other;
962             if (Token::Match(scope->function->tokenDef->tokAt(-2), "char|wchar_t *"))
963                 returnType = charPtr;
964             else if (Token::Match(scope->function->tokenDef->tokAt(-5), "const std :: string|wstring &"))
965                 returnType = stdStringConstRef;
966             else if (Token::Match(scope->function->tokenDef->tokAt(-3), "std :: string|wstring !!&"))
967                 returnType = stdString;
968     
969             for (const Token *tok = scope->classStart; tok && tok != scope->classEnd; tok = tok->next()) {
970                 // Invalid usage..
971                 if (Token::Match(tok, "throw %var% . c_str|data ( ) ;") && isLocal(tok->next()) &&
972                     tok->next()->variable() && tok->next()->variable()->isStlStringType()) {
973                     string_c_strThrowError(tok);
974                 } else if (Token::Match(tok, "[;{}] %name% = %var% . str ( ) . c_str|data ( ) ;")) {
975                     const Variable* var = tok->next()->variable();
976                     const Variable* var2 = tok->tokAt(3)->variable();
977                     if (var && var->isPointer() && var2 && var2->isStlType(stl_string_stream))
978                         string_c_strError(tok);
979                 } else if (Token::Match(tok, "[;{}] %var% = %name% (") &&
980                            Token::Match(tok->linkAt(4), ") . c_str|data ( ) ;") &&
981                            tok->tokAt(3)->function() && Token::Match(tok->tokAt(3)->function()->retDef, "std :: string|wstring %name%")) {
982                     const Variable* var = tok->next()->variable();
983                     if (var && var->isPointer())
984                         string_c_strError(tok);
985                 } else if (printPerformance && Token::Match(tok, "%name% ( !!)") && c_strFuncParam.find(tok->str()) != c_strFuncParam.end() &&
986                            !Token::Match(tok->previous(), "::|.") && tok->varId() == 0 && tok->str() != scope->className) { // calling function. TODO: Add support for member functions
987                     std::pair<std::multimap<std::string, unsigned int>::const_iterator, std::multimap<std::string, unsigned int>::const_iterator> range = c_strFuncParam.equal_range(tok->str());
988                     for (std::multimap<std::string, unsigned int>::const_iterator i = range.first; i != range.second; ++i) {
989                         if (i->second == 0)
990                             continue;
991     
992                         const Token* tok2 = tok->tokAt(2);
993                         unsigned int j;
994                         for (j = 0; tok2 && j < i->second-1; j++)
995                             tok2 = tok2->nextArgument();
996                         if (tok2)
997                             tok2 = tok2->nextArgument();
998                         else
999                             break;
1000                         if (!tok2 && j == i->second-1)
1001                             tok2 = tok->next()->link();
1002                         else
1003                             tok2 = tok2->previous();
1004                         if (tok2 && Token::Match(tok2->tokAt(-4), ". c_str|data ( )")) {
1005                             const Variable* var = tok2->tokAt(-5)->variable();
1006                             if (var && var->isStlStringType()) {
1007                                 string_c_strParam(tok, i->second);
1008                             } else if (Token::Match(tok2->tokAt(-9), "%name% . str ( )")) { // Check ss.str().c_str() as parameter
1009                                 const Variable* ssVar = tok2->tokAt(-9)->variable();
1010                                 if (ssVar && ssVar->isStlType(stl_string_stream))
1011                                     string_c_strParam(tok, i->second);
1012                             }
1013     
1014                         }
1015                     }
1016                 }
1017     
1018                 // Using c_str() to get the return value is only dangerous if the function returns a char*
1019                 if (returnType == charPtr) {
1020                     if (Token::Match(tok, "return %var% . c_str|data ( ) ;") && isLocal(tok->next()) &&
1021                         tok->next()->variable() && tok->next()->variable()->isStlStringType()) {
1022                         string_c_strError(tok);
1023                     } else if (Token::Match(tok, "return %var% . str ( ) . c_str|data ( ) ;") && isLocal(tok->next()) &&
1024                                tok->next()->variable() && tok->next()->variable()->isStlType(stl_string_stream)) {
1025                         string_c_strError(tok);
1026                     } else if (Token::Match(tok, "return std :: string|wstring (") &&
1027                                Token::Match(tok->linkAt(4), ") . c_str|data ( ) ;")) {
1028                         string_c_strError(tok);
1029                     } else if (Token::Match(tok, "return %name% (") && Token::Match(tok->linkAt(2), ") . c_str|data ( ) ;")) {
1030                         const Function* func = tok->next()->function();
1031                         if (func && Token::Match(func->tokenDef->tokAt(-3), "std :: string|wstring"))
1032                             string_c_strError(tok);
1033                     } else if (Token::simpleMatch(tok, "return (") &&
1034                                Token::Match(tok->next()->link(), ") . c_str|data ( ) ;")) {
1035                         // Check for "+ localvar" or "+ std::string(" inside the bracket
1036                         bool is_implicit_std_string = printInconclusive;
1037                         const Token *search_end = tok->next()->link();
1038                         for (const Token *search_tok = tok->tokAt(2); search_tok != search_end; search_tok = search_tok->next()) {
1039                             if (Token::Match(search_tok, "+ %var%") && isLocal(search_tok->next()) &&
1040                                 search_tok->next()->variable() && search_tok->next()->variable()->isStlStringType()) {
1041                                 is_implicit_std_string = true;
1042                                 break;
1043                             } else if (Token::Match(search_tok, "+ std :: string|wstring (")) {
1044                                 is_implicit_std_string = true;
1045                                 break;
1046                             }
1047                         }
1048     
1049                         if (is_implicit_std_string)
1050                             string_c_strError(tok);
1051                     }
1052                 }
1053                 // Using c_str() to get the return value is redundant if the function returns std::string or const std::string&.

1111     void CheckStl::checkAutoPointer()
1112     {
1113         std::set<unsigned int> autoPtrVarId;
1114         std::map<unsigned int, const std::string> mallocVarId; // variables allocated by the malloc-like function
1115         const char STL_CONTAINER_LIST[] = "array|bitset|deque|list|forward_list|map|multimap|multiset|priority_queue|queue|set|stack|vector|hash_map|hash_multimap|hash_set|unordered_map|unordered_multimap|unordered_set|unordered_multiset|basic_string";
1116         const int malloc = _settings->library.alloc("malloc"); // allocation function, which are not compatible with auto_ptr
1117         const bool printStyle = _settings->isEnabled("style");
1118     
1119         for (const Token *tok = _tokenizer->tokens(); tok; tok = tok->next()) {
1120             if (Token::simpleMatch(tok, "auto_ptr <")) {
1121                 if ((tok->strAt(-1) == "<" && Token::Match(tok->tokAt(-2), STL_CONTAINER_LIST)) ||
1122                     (Token::simpleMatch(tok->tokAt(-3), "< std :: auto_ptr") && Token::Match(tok->tokAt(-4), STL_CONTAINER_LIST))) {
1123                     autoPointerContainerError(tok);
1124                 } else {
1125                     const Token *tok2 = tok->linkAt(1);
1126     
1127                     if (Token::Match(tok2, "> %name%")) {
1128                         const Token *tok3 = tok2->tokAt(2);
1129                         if (Token::Match(tok3, "( new %type%") && hasArrayEndParen(tok3)) {
1130                             autoPointerArrayError(tok2->next());
1131                         }
1132                         if (Token::Match(tok3, "( %name% (") && malloc && _settings->library.alloc(tok3->next()) == malloc) {
1133                             // malloc-like function allocated memory passed to the auto_ptr constructor -> error
1134                             autoPointerMallocError(tok2->next(), tok3->next()->str());
1135                         }
1136                         if (Token::Match(tok3, "( %var%")) {
1137                             std::map<unsigned int, const std::string>::const_iterator it = mallocVarId.find(tok3->next()->varId());
1138                             if (it != mallocVarId.cend()) {
1139                                 // pointer on the memory allocated by malloc used in the auto pointer constructor -> error
1140                                 autoPointerMallocError(tok2->next(), it->second);
1141                             }
1142                         }
1143                         while (tok3 && tok3->str() != ";") {
1144                             tok3 = tok3->next();
1145                         }
1146                         if (tok3) {
1147                             tok3 = tok3->tokAt(-2);
1148                             if (Token::simpleMatch(tok3->previous(), "[ ] )")) {
1149                                 autoPointerArrayError(tok2->next());
1150                             } else if (tok3->varId()) {
1151                                 const Token *decltok = Token::findmatch(_tokenizer->tokens(), "%varid% = new %type%", tok3->varId());
1152                                 if (decltok && hasArrayEnd(decltok)) {
1153                                     autoPointerArrayError(tok2->next());
1154                                 }
1155                             }
1156                             if (tok2->next()->varId()) {
1157                                 autoPtrVarId.insert(tok2->next()->varId());
1158                             }
1159                         }
1160                     }
1161                 }
1162             } else {
1163                 if (Token::Match(tok, "%name% = %var% ;")) {
1164                     if (printStyle) {
1165                         std::set<unsigned int>::const_iterator iter = autoPtrVarId.find(tok->tokAt(2)->varId());
1166                         if (iter != autoPtrVarId.end()) {
1167                             autoPointerError(tok->tokAt(2));
1168                         }
1169                     }
1170                 } else if ((Token::Match(tok, "%var% = new %type%") && hasArrayEnd(tok)) ||
1171                            (Token::Match(tok, "%var% . reset ( new %type%") && hasArrayEndParen(tok))) {
1172                     std::set<unsigned int>::const_iterator iter = autoPtrVarId.find(tok->varId());
1173                     if (iter != autoPtrVarId.end()) {
1174                         autoPointerArrayError(tok);
1175                     }
1176                 } else if (Token::Match(tok, "%var% = %name% (") && malloc && _settings->library.alloc(tok->tokAt(2)) == malloc) {
1177                     // C library function like 'malloc' used together with auto pointer -> error
1178                     std::set<unsigned int>::const_iterator iter = autoPtrVarId.find(tok->varId());
1179                     if (iter != autoPtrVarId.end()) {
1180                         autoPointerMallocError(tok, tok->strAt(2));
1181                     } else if (tok->varId()) {
1182                         // it is not an auto pointer variable and it is allocated by malloc like function.
1183                         mallocVarId.insert(std::make_pair(tok->varId(), tok->strAt(2)));
1184                     }
1185                 } else if (Token::Match(tok, "%var% . reset ( %name% (") && malloc && _settings->library.alloc(tok->tokAt(4)) == malloc) {
1186                     // C library function like 'malloc' used when resetting auto pointer -> error
1187                     std::set<unsigned int>::const_iterator iter = autoPtrVarId.find(tok->varId());
1188                     if (iter != autoPtrVarId.end()) {
1189                         autoPointerMallocError(tok, tok->strAt(4));
1190                     }
1191                 }
1192             }
1193         }
1194     }

1316     // Check for iterators being dereferenced before being checked for validity.
1317     // E.g.  if (*i && i != str.end()) { }
1318     void CheckStl::checkDereferenceInvalidIterator()
1319     {
1320         if (!_settings->isEnabled("warning"))
1321             return;
1322     
1323         // Iterate over "if", "while", and "for" conditions where there may
1324         // be an iterator that is dereferenced before being checked for validity.
1325         const std::list<Scope>& scopeList = _tokenizer->getSymbolDatabase()->scopeList;
1326         for (std::list<Scope>::const_iterator i = scopeList.begin(); i != scopeList.end(); ++i) {
1327             if (i->type == Scope::eIf || i->type == Scope::eDo || i->type == Scope::eWhile || i->type == Scope::eFor) {
1328     
1329                 const Token* const tok = i->classDef;
1330                 const Token* startOfCondition = tok->next();
1331                 if (i->type == Scope::eDo)
1332                     startOfCondition = startOfCondition->link()->tokAt(2);
1333                 if (!startOfCondition) // ticket #6626 invalid code
1334                     continue;
1335                 const Token* endOfCondition = startOfCondition->link();
1336                 if (!endOfCondition)
1337                     continue;
1338     
1339                 // For "for" loops, only search between the two semicolons
1340                 if (i->type == Scope::eFor) {
1341                     startOfCondition = Token::findsimplematch(tok->tokAt(2), ";", endOfCondition);
1342                     if (!startOfCondition)
1343                         continue;
1344                     endOfCondition = Token::findsimplematch(startOfCondition->next(), ";", endOfCondition);
1345                     if (!endOfCondition)
1346                         continue;
1347                 }
1348     
1349                 // Only consider conditions composed of all "&&" terms and
1350                 // conditions composed of all "||" terms
1351                 const bool isOrExpression =
1352                     Token::findsimplematch(startOfCondition, "||", endOfCondition) != 0;
1353                 const bool isAndExpression =
1354                     Token::findsimplematch(startOfCondition, "&&", endOfCondition) != 0;
1355     
1356                 // Look for a check of the validity of an iterator
1357                 const Token* validityCheckTok = 0;
1358                 if (!isOrExpression && isAndExpression) {
1359                     validityCheckTok =
1360                         Token::findmatch(startOfCondition, "&& %var% != %name% . end|rend|cend|crend ( )", endOfCondition);
1361                 } else if (isOrExpression && !isAndExpression) {
1362                     validityCheckTok =
1363                         Token::findmatch(startOfCondition, "%oror% %var% == %name% . end|rend|cend|crend ( )", endOfCondition);
1364                 }
1365     
1366                 if (!validityCheckTok)
1367                     continue;
1368                 const unsigned int iteratorVarId = validityCheckTok->next()->varId();
1369     
1370                 // If the iterator dereference is to the left of the check for
1371                 // the iterator's validity, report an error.
1372                 const Token* const dereferenceTok =
1373                     Token::findmatch(startOfCondition, "* %varid%", validityCheckTok, iteratorVarId);
1374                 if (dereferenceTok)
1375                     dereferenceInvalidIteratorError(dereferenceTok, dereferenceTok->strAt(1));
1376             }
1377         }
1378     }

1389     void CheckStl::readingEmptyStlContainer_parseUsage(const Token* tok, const Library::Container* container, std::map<unsigned int, const Library::Container*>& empty, bool noerror)
1390     {
1391         // Check for various conditions for the way stl containers and variables can be used
1392         if (tok->strAt(1) == "=" || (tok->strAt(1) == "[" && Token::simpleMatch(tok->linkAt(1), "] ="))) {
1393             // Assignment (LHS)
1394             empty.erase(tok->varId());
1395         } else if (Token::Match(tok, "%name% [")) {
1396             // Access through operator[]
1397             if (!container->arrayLike_indexOp) { // operator[] inserts an element if used on a std::map
1398                 if (!noerror && tok->strAt(-1) == "=")
1399                     readingEmptyStlContainerError(tok);
1400                 empty.erase(tok->varId());
1401             } else if (!noerror)
1402                 readingEmptyStlContainerError(tok);
1403         } else if (Token::Match(tok, "%name% . %type% (")) {
1404             Library::Container::Yield yield = container->getYield(tok->strAt(2));
1405             const Token* parent = tok->tokAt(3)->astParent();
1406             // Member function call
1407             if (yield != Library::Container::NO_YIELD &&
1408                 ((yield != Library::Container::ITERATOR &&
1409                   yield != Library::Container::START_ITERATOR &&
1410                   yield != Library::Container::END_ITERATOR) || !parent || Token::Match(parent, "%cop%|=|*"))) { // These functions read from the container
1411                 if (!noerror)
1412                     readingEmptyStlContainerError(tok);
1413             } else {
1414                 Library::Container::Action action = container->getAction(tok->strAt(2));
1415                 if (action == Library::Container::FIND || action == Library::Container::ERASE || action == Library::Container::POP || action == Library::Container::CLEAR) {
1416                     if (!noerror)
1417                         readingEmptyStlContainerError(tok);
1418                 } else
1419                     empty.erase(tok->varId());
1420             }
1421         } else if (tok->strAt(-1) == "=") {
1422             // Assignment (RHS)
1423             if (!noerror)
1424                 readingEmptyStlContainerError(tok);
1425         } else {
1426             // Unknown usage. Assume it is initialized.
1427             empty.erase(tok->varId());
1428         }
1429     }

1431     void CheckStl::readingEmptyStlContainer()
1432     {
1433         if (!_settings->isEnabled("style"))
1434             return;
1435     
1436         if (!_settings->inconclusive)
1437             return;
1438     
1439         std::map<unsigned int, const Library::Container*> emptyContainer;
1440     
1441         const std::list<Scope>& scopeList = _tokenizer->getSymbolDatabase()->scopeList;
1442     
1443         for (std::list<Scope>::const_iterator i = scopeList.begin(); i != scopeList.end(); ++i) {
1444             if (i->type != Scope::eFunction)
1445                 continue;
1446     
1447             for (const Token *tok = i->classStart->next(); tok != i->classEnd; tok = tok->next()) {
1448                 if (Token::Match(tok, "for|while")) { // Loops and end of scope clear the sets.
1449                     const Token* tok2 = tok->linkAt(1);
1450                     if (!tok2)
1451                         continue;
1452                     tok2 = tok2->next();
1453                     for (const Token* end2 = tok2->link(); tok2 && tok2 != end2; tok2 = tok2->next()) {
1454                         if (!tok2->varId())
1455                             continue;
1456     
1457                         std::map<unsigned int, const Library::Container*>::const_iterator container = emptyContainer.find(tok2->varId());
1458                         if (container == emptyContainer.end())
1459                             continue;
1460     
1461                         readingEmptyStlContainer_parseUsage(tok2, container->second, emptyContainer, true);
1462                     }
1463                 } else if (Token::Match(tok, "do|}|break|case")) {
1464                     emptyContainer.clear();
1465                 }
1466     
1467                 if (!tok->varId())
1468                     continue;
1469     
1470                 // Check whether a variable should be marked as "empty"
1471                 const Variable* var = tok->variable();
1472                 if (var && !var->isArrayOrPointer() && !var->typeStartToken()->isStandardType()) {
1473                     bool insert = false;
1474                     if (var->nameToken() == tok && var->isLocal() && !var->isStatic()) { // Local variable declared
1475                         insert = !Token::Match(tok->tokAt(1), "[(=]"); // Only if not initialized
1476                     } else if (Token::Match(tok, "%var% . clear ( ) ;")) {
1477                         insert = true;
1478                     }
1479     
1480                     if (insert) {
1481                         const Library::Container* container = _settings->library.detectContainer(var->typeStartToken());
1482                         if (container)
1483                             emptyContainer[var->declarationId()] = container;
1484                         continue;
1485                     }
1486                 }
1487     
1488                 std::map<unsigned int, const Library::Container*>::const_iterator container = emptyContainer.find(tok->varId());
1489                 if (container == emptyContainer.end())
1490                     continue;
1491     
1492                 readingEmptyStlContainer_parseUsage(tok, container->second, emptyContainer, false);
1493             }
1494             emptyContainer.clear();
1495         }
1496     }
