27     void CheckBoost::checkBoostForeachModification()
28     {
29         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
30         const std::size_t functions = symbolDatabase->functionScopes.size();
31         for (std::size_t i = 0; i < functions; ++i) {
32             const Scope * scope = symbolDatabase->functionScopes[i];
33             for (const Token *tok = scope->classStart->next(); tok && tok != scope->classEnd; tok = tok->next()) {
34                 if (!Token::simpleMatch(tok, "BOOST_FOREACH ("))
35                     continue;
36     
37                 const Token *containerTok = tok->next()->link()->previous();
38                 if (!Token::Match(containerTok, "%var% ) {"))
39                     continue;
40     
41                 const Token *tok2 = containerTok->tokAt(2);
42                 const Token *end = tok2->link();
43                 for (; tok2 != end; tok2 = tok2->next()) {
44                     if (Token::Match(tok2, "%varid% . insert|erase|push_back|push_front|pop_front|pop_back|clear|swap|resize|assign|merge|remove|remove_if|reverse|sort|splice|unique|pop|push", containerTok->varId())) {
45                         const Token* nextStatement = Token::findsimplematch(tok2->linkAt(3), ";", end);
46                         if (!Token::Match(nextStatement, "; break|return|throw"))
47                             boostForeachError(tok2);
48                         break;
49                     }
50                 }
51             }
52         }
53     }
54     
55     void CheckBoost::boostForeachError(const Token *tok)
56     {
57         reportError(tok, Severity::error, "boostForeachError",
58                     "BOOST_FOREACH caches the end() iterator. It's undefined behavior if you modify the container inside."
59                    );
60     }
