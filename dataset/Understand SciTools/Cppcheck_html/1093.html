20     //---------------------------------------------------------------------------
21     #ifndef astutilsH
22     #define astutilsH
23     //---------------------------------------------------------------------------
24     
25     #include <set>
26     #include <string>
27     
28     class Token;
29     
30     /** Is expression a 'signed char' if no promotion is used */
31     bool astIsSignedChar(const Token *tok);
32     /** Is expression of integral type? */
33     bool astIsIntegral(const Token *tok, bool unknown);
34     /** Is expression of floating point type? */
35     bool astIsFloat(const Token *tok, bool unknown);
36     
37     /**
38      * Get canonical type of expression. const/static/etc are not included and neither *&.
39      * For example:
40      * Expression type      Return
41      * std::string          std::string
42      * int *                int
43      * static const int     int
44      * std::vector<T>       std::vector
45      */
46     std::string astCanonicalType(const Token *expr);
47     
48     /** Is given syntax tree a variable comparison against value */
49     const Token * astIsVariableComparison(const Token *tok, const std::string &comp, const std::string &rhs, const Token **vartok=nullptr);
50     
51     bool isSameExpression(bool cpp, bool macro, const Token *tok1, const Token *tok2, const std::set<std::string> &constFunctions);
52     
53     /**
54      * Are two conditions opposite
55      * @param isNot  do you want to know if cond1 is !cond2 or if cond1 and cond2 are non-overlapping. true: cond1==!cond2  false: cond1==true => cond2==false
56      * @param cpp    c++ file
57      * @param cond1  condition1
58      * @param cond2  condition2
59      * @param constFunctions  constFunctions
60      */
61     bool isOppositeCond(bool isNot, bool cpp, const Token * const cond1, const Token * const cond2, const std::set<std::string> &constFunctions);
62     
63     bool isConstExpression(const Token *tok, const std::set<std::string> &constFunctions);
64     
65     bool isWithoutSideEffects(bool cpp, const Token* tok);
66     
67     /** Is variable changed in block of code? */
68     bool isVariableChanged(const Token *start, const Token *end, const unsigned int varid);
69     
70     /** Determines the number of arguments - if token is a function call or macro
71      * @param start token which is supposed to be the function/macro name.
72      * \return Number of arguments
73      */
74     int numberOfArguments(const Token *start);
75     
76     #endif // astutilsH
