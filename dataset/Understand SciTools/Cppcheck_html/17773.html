41     static BOOL MyIsDirectory(const std::string& path)
42     {
43     #ifdef __BORLANDC__
44         return (GetFileAttributes(path.c_str()) & FILE_ATTRIBUTE_DIRECTORY);
45     #else
46     // See http://msdn.microsoft.com/en-us/library/bb773621(VS.85).aspx
47         return PathIsDirectoryA(path.c_str());
48     #endif
49     }

51     static HANDLE MyFindFirstFile(const std::string& path, LPWIN32_FIND_DATAA findData)
52     {
53         HANDLE hFind = FindFirstFileA(path.c_str(), findData);
54         return hFind;
55     }

57     static BOOL MyFileExists(const std::string& path)
58     {
59     #ifdef __BORLANDC__
60         DWORD fa = GetFileAttributes(path.c_str());
61         BOOL result = FALSE;
62         if (fa != INVALID_FILE_ATTRIBUTES && !(fa & FILE_ATTRIBUTE_DIRECTORY))
63             result = TRUE;
64     #else
65         BOOL result = PathFileExistsA(path.c_str());
66     #endif
67         return result;
68     }

70     void FileLister::recursiveAddFiles(std::map<std::string, std::size_t> &files, const std::string &path, const std::set<std::string> &extra, const PathMatch& ignored)
71     {
72         const std::string cleanedPath = Path::toNativeSeparators(path);
73     
74         // basedir is the base directory which is used to form pathnames.
75         // It always has a trailing backslash available for concatenation.
76         std::string basedir;
77     
78         // searchPattern is the search string passed into FindFirst and FindNext.
79         std::string searchPattern = cleanedPath;
80     
81         // The user wants to check all files in a dir
82         const bool checkAllFilesInDir = (MyIsDirectory(cleanedPath) != FALSE);
83     
84         if (checkAllFilesInDir) {
85             char c = cleanedPath[ cleanedPath.size()-1 ];
86             switch (c) {
87             case '\\':
88                 searchPattern += '*';
89                 basedir = cleanedPath;
90                 break;
91             case '*':
92                 basedir = cleanedPath.substr(0, cleanedPath.length() - 1);
93                 break;
94             default:
95                 searchPattern += "\\*";
96                 if (cleanedPath != ".")
97                     basedir = cleanedPath + '\\';
98             }
99         } else {
100             std::string::size_type pos = cleanedPath.find_last_of('\\');
101             if (std::string::npos != pos) {
102                 basedir = cleanedPath.substr(0, pos + 1);
103             }
104         }
105     
106         WIN32_FIND_DATAA ffd;
107         HANDLE hFind = MyFindFirstFile(searchPattern, &ffd);
108         if (INVALID_HANDLE_VALUE == hFind)
109             return;
110     
111         do {
112             if (ffd.cFileName[0] == '.' || ffd.cFileName[0] == '\0')
113                 continue;
114     
115             const char* ansiFfd = ffd.cFileName;
116             if (std::strchr(ansiFfd,'?')) {
117                 ansiFfd = ffd.cAlternateFileName;
118             }
119     
120             const std::string fname(basedir + ansiFfd);
121     
122             if ((ffd.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) == 0) {
123                 // File
124                 if ((!checkAllFilesInDir || Path::acceptFile(fname, extra)) && !ignored.Match(fname)) {
125                     const std::string nativename = Path::fromNativeSeparators(fname);
126     
127                     // Limitation: file sizes are assumed to fit in a 'size_t'
128     #ifdef _WIN64
129                     files[nativename] = (static_cast<std::size_t>(ffd.nFileSizeHigh) << 32) | ffd.nFileSizeLow;
130     #else
131                     files[nativename] = ffd.nFileSizeLow;
132     #endif
133                 }
134             } else {
135                 // Directory
136                 if (!ignored.Match(fname))
137                     FileLister::recursiveAddFiles(files, fname, extra, ignored);
138             }
139         } while (FindNextFileA(hFind, &ffd) != FALSE);
140     
141         FindClose(hFind);
142     }

172     static void addFiles2(std::map<std::string, std::size_t> &files,
173                           const std::string &path,
174                           const std::set<std::string> &extra,
175                           bool recursive,
176                           const PathMatch& ignored
177                          )
178     {
179         struct stat file_stat;
180         if (stat(path.c_str(), &file_stat) != -1) {
181             if ((file_stat.st_mode & S_IFMT) == S_IFDIR) {
182                 DIR * dir = opendir(path.c_str());
183     
184                 if (!dir)
185                     return;
186     
187                 dirent entry;
188                 dirent * dir_result;
189     
190                 std::string new_path;
191                 new_path.reserve(path.length() + 100);// prealloc some memory to avoid constant new/deletes in loop
192     
193                 while ((readdir_r(dir, &entry, &dir_result) == 0) && (dir_result != NULL)) {
194     
195                     if ((std::strcmp(dir_result->d_name, ".") == 0) ||
196                         (std::strcmp(dir_result->d_name, "..") == 0))
197                         continue;
198     
199                     new_path = path + '/' + dir_result->d_name;
200     
201                     if (dir_result->d_type == DT_DIR) {
202                         if (recursive && !ignored.Match(new_path)) {
203                             addFiles2(files, new_path, extra, recursive, ignored);
204                         }
205                     } else {
206                         if (Path::acceptFile(new_path, extra) && !ignored.Match(new_path)) {
207                             files[new_path] = file_stat.st_size;
208                         }
209                     }
210                 }
211                 closedir(dir);
212             } else
213                 files[path] = file_stat.st_size;
214         }
215     }

217     void FileLister::recursiveAddFiles(std::map<std::string, std::size_t> &files, const std::string &path, const std::set<std::string> &extra, const PathMatch& ignored)
218     {
219         addFiles(files, path, extra, true, ignored);
220     }

222     void FileLister::addFiles(std::map<std::string, std::size_t> &files, const std::string &path, const std::set<std::string> &extra, bool recursive, const PathMatch& ignored)
223     {
224         if (!path.empty()) {
225             std::string corrected_path = path;
226             if (corrected_path.back() == '/')
227                 corrected_path.erase(corrected_path.end() - 1);
228     
229             addFiles2(files, corrected_path, extra, recursive, ignored);
230         }
231     }

233     bool FileLister::isDirectory(const std::string &path)
234     {
235         struct stat file_stat;
236         if (stat(path.c_str(), &file_stat) != -1)
237             return ((file_stat.st_mode & S_IFMT) == S_IFDIR);
238     
239         return false;
240     }

242     bool FileLister::fileExists(const std::string &path)
243     {
244         struct stat file_stat;
245         if (stat(path.c_str(), &file_stat) != -1)
246             return ((file_stat.st_mode & S_IFMT) == S_IFREG);
247     
248         return false;
249     }
