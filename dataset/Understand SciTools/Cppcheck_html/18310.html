30     class TestSamples : public TestFixture {
31     public:
32         TestSamples() : TestFixture("TestSamples") {
33         }
34     
35     private:
36     
37         void run() {
38             TEST_CASE(runSamples);
39         }
40     
41         void runSamples() const {
42             REDIRECT;
43     
44             std::map<std::string, std::size_t> files;
45             const std::vector<std::string> masks;
46             const PathMatch matcher(masks);
47     #ifdef _WIN32
48             FileLister::recursiveAddFiles(files, "..\\samples", matcher);
49     #else
50             FileLister::recursiveAddFiles(files, "samples", matcher);
51     #endif
52             for (std::map<std::string, std::size_t>::const_iterator i = files.begin(); i != files.end(); ++i) {
53                 CLEAR_REDIRECT_ERROUT;
54                 char* path = new char[i->first.size() + 1];
55                 strcpy(path, i->first.c_str());
56                 const char* argv[] = {
57     #ifdef _WIN32
58                     ".\\..\\testrunner",
59     #else
60                     "./testrunner",
61     #endif
62                     "--enable=style,warning,performance,portability", "--inconclusive", "-rp", "-f", "-q", path
63                 };
64                 std::string filename = i->first.substr(i->first.find_last_of("/\\")+1);
65                 if (filename == "good.cpp" || filename == "good.c") {
66                     CppCheckExecutor exec;
67                     exec.check(7, argv);
68                     ASSERT_EQUALS_MSG("", GET_REDIRECT_ERROUT, i->first);
69                 } else if (filename == "bad.cpp" || filename == "bad.c") {
70                     CppCheckExecutor exec;
71                     exec.check(7, argv);
72                     std::string expected_filename = Path::getPathFromFilename(i->first) + "out.txt";
73                     std::ifstream ifs(expected_filename.c_str());
74                     std::string expected((std::istreambuf_iterator<char>(ifs)), std::istreambuf_iterator<char>());
75                     std::string actual = GET_REDIRECT_ERROUT;
76                     // We need some uniformization to make this work on Unix and Windows
77                     std::replace(actual.begin(), actual.end(), '/', '\\'); // Uniformize slashes.
78                     while (actual.find("..\\") != std::string::npos)
79                         actual.erase(actual.find("..\\"), 3); // Remove '..\'
80                     ASSERT_EQUALS_MSG(expected, actual, i->first);
81                 }
82                 delete[] path;
83             }
84         }
85     };
