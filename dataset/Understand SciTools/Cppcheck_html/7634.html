32     namespace {
33         typedef std::pair<std::string, struct TimerResultsData> dataElementType;
34         bool more_second_sec(const dataElementType& lhs, const dataElementType& rhs)
35         {
36             return lhs.second.seconds() > rhs.second.seconds();
37         }
38     }
39     
40     void TimerResults::ShowResults(SHOWTIME_MODES mode) const
41     {
42         if (mode == SHOWTIME_NONE)
43             return;
44     
45         std::cout << std::endl;
46         TimerResultsData overallData;
47     
48         std::vector<dataElementType> data(_results.begin(), _results.end());
49         std::sort(data.begin(), data.end(), more_second_sec);
50     
51         size_t ordinal = 1; // maybe it would be nice to have an ordinal in output later!
52         for (std::vector<dataElementType>::const_iterator iter=data.begin() ; iter!=data.end(); ++iter) {
53             const double sec = iter->second.seconds();
54             const double secAverage = sec / (double)(iter->second._numberOfResults);
55             overallData._clocks += iter->second._clocks;
56             if ((mode != SHOWTIME_TOP5) || (ordinal<=5)) {
57                 std::cout << iter->first << ": " << sec << "s (avg. " << secAverage << "s - " << iter->second._numberOfResults  << " result(s))" << std::endl;
58             }
59             ++ordinal;
60         }
61     
62         const double secOverall = overallData.seconds();
63         std::cout << "Overall time: " << secOverall << "s" << std::endl;
64     }
65     
66     void TimerResults::AddResults(const std::string& str, std::clock_t clocks)
67     {
68         _results[str]._clocks += clocks;
69         _results[str]._numberOfResults++;
70     }
71     
72     Timer::Timer(const std::string& str, unsigned int showtimeMode, TimerResultsIntf* timerResults)
73         : _str(str)
74         , _timerResults(timerResults)
75         , _start(0)
76         , _showtimeMode(showtimeMode)
77         , _stopped(false)
78     {
79         if (showtimeMode != SHOWTIME_NONE)
80             _start = std::clock();
81     }
82     
83     Timer::~Timer()
84     {
85         Stop();
86     }
87     
88     void Timer::Stop()
89     {
90         if ((_showtimeMode != SHOWTIME_NONE) && !_stopped) {
91             const std::clock_t end = std::clock();
92             const std::clock_t diff = end - _start;
93     
94             if (_showtimeMode == SHOWTIME_FILE) {
95                 double sec = (double)diff / CLOCKS_PER_SEC;
96                 std::cout << _str << ": " << sec << "s" << std::endl;
97             } else {
98                 if (_timerResults)
99                     _timerResults->AddResults(_str, diff);
100             }
101         }
102     
103         _stopped = true;
104     }
