19     #include <QString>
20     #include <QStringList>
21     #include <QDir>
22     #include <QFileInfo>
23     #include "filelist.h"

25     QStringList FileList::GetDefaultFilters()
26     {
27         QStringList extensions;
28         extensions << "*.cpp" << "*.cxx" << "*.cc" << "*.c" << "*.c++" << "*.txx" << "*.tpp";
29         return extensions;
30     }
31     
32     bool FileList::FilterMatches(const QFileInfo &inf)
33     {
34         if (inf.isFile()) {
35             const QStringList filters = FileList::GetDefaultFilters();
36             QString ext("*.");
37             ext += inf.suffix();
38             if (filters.contains(ext, Qt::CaseInsensitive))
39                 return true;
40         }
41         return false;
42     }
43     
44     void FileList::AddFile(const QString &filepath)
45     {
46         QFileInfo inf(filepath);
47         if (FilterMatches(inf))
48             mFileList << inf;
49     }
50     
51     void FileList::AddDirectory(const QString &directory, bool recursive)
52     {
53         QDir dir(directory);
54         dir.setSorting(QDir::Name);
55         const QStringList filters = FileList::GetDefaultFilters();
56         const QStringList origNameFilters = dir.nameFilters();
57         dir.setNameFilters(filters);
58         if (!recursive) {
59             dir.setFilter(QDir::Files | QDir::NoDotAndDotDot);
60             QFileInfoList items = dir.entryInfoList();
61             mFileList += items;
62         } else {
63             dir.setFilter(QDir::Files | QDir::NoDotAndDotDot);
64             QFileInfoList items = dir.entryInfoList();
65             mFileList += items;
66     
67             dir.setNameFilters(origNameFilters);
68             dir.setFilter(QDir::Dirs | QDir::NoDotAndDotDot);
69             QFileInfoList list = dir.entryInfoList();
70             QFileInfo item;
71             foreach (item, list) {
72                 const QString path = item.canonicalFilePath();
73                 AddDirectory(path, recursive);
74             }
75         }
76     }
77     
78     void FileList::AddPathList(const QStringList &paths)
79     {
80         QString path;
81         foreach (path, paths) {
82             QFileInfo inf(path);
83             if (inf.isFile())
84                 AddFile(path);
85             else
86                 AddDirectory(path, true);
87         }
88     }

90     QStringList FileList::GetFileList() const
91     {
92         if (mExcludedPaths.empty()) {
93             QStringList names;
94             foreach (QFileInfo item, mFileList) {
95                 QString name = QDir::fromNativeSeparators(item.canonicalFilePath());
96                 names << name;
97             }
98             return names;
99         } else {
100             return ApplyExcludeList();
101         }
102     }
103     
104     void FileList::AddExcludeList(const QStringList &paths)
105     {
106         mExcludedPaths = paths;
107     }
108     
109     QStringList FileList::ApplyExcludeList() const
110     {
111         QStringList paths;
112         foreach (QFileInfo item, mFileList) {
113             QString name = QDir::fromNativeSeparators(item.canonicalFilePath());
114             if (!Match(name))
115                 paths << name;
116         }
117         return paths;
118     }
119     
120     bool FileList::Match(const QString &path) const
121     {
122         for (int i = 0; i < mExcludedPaths.size(); i++) {
123             if (mExcludedPaths[i].endsWith('/')) {
124                 const QString pathexclude("/" + mExcludedPaths[i]);
125                 if (path.indexOf(pathexclude) != -1)
126                     return true;
127             } else {
128                 if (path.endsWith(mExcludedPaths[i]))
129                     return true;
130             }
131         }
132         return false;
133     }
