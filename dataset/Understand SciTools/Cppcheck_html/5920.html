20     //---------------------------------------------------------------------------
21     #ifndef checkuninitvarH
22     #define checkuninitvarH
23     //---------------------------------------------------------------------------
24     
25     #include "config.h"
26     #include "check.h"
27     
28     class Scope;
29     class Variable;
30     
31     /// @addtogroup Checks
32     /// @{
33     
34     
35     /** @brief Checking for uninitialized variables */
36     
37     class CPPCHECKLIB CheckUninitVar : public Check {
38     public:
39         /** @brief This constructor is used when registering the CheckUninitVar */
40         CheckUninitVar() : Check(myName()) {
41         }
42     
43         /** @brief This constructor is used when running checks. */
44         CheckUninitVar(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
45             : Check(myName(), tokenizer, settings, errorLogger) {
46         }
47     
48         /** @brief Run checks against the simplified token list */
49         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
50             CheckUninitVar checkUninitVar(tokenizer, settings, errorLogger);
51             checkUninitVar.check();
52             checkUninitVar.deadPointer();
53         }
54     
55         /** Check for uninitialized variables */
56         void check();
57         void checkScope(const Scope* scope);
58         void checkStruct(const Token *tok, const Variable &structvar);
59         enum Alloc { NO_ALLOC, NO_CTOR_CALL, CTOR_CALL, ARRAY };
60         bool checkScopeForVariable(const Token *tok, const Variable& var, bool* const possibleInit, bool* const noreturn, Alloc* const alloc, const std::string &membervar);
61         bool checkIfForWhileHead(const Token *startparentheses, const Variable& var, bool suppressErrors, bool isuninit, Alloc alloc, const std::string &membervar);
62         bool checkLoopBody(const Token *tok, const Variable& var, const Alloc alloc, const std::string &membervar, const bool suppressErrors);
63         void checkRhs(const Token *tok, const Variable &var, Alloc alloc, unsigned int number_of_if, const std::string &membervar);
64         bool isVariableUsage(const Token *vartok, bool ispointer, Alloc alloc) const;
65         int isFunctionParUsage(const Token *vartok, bool ispointer, Alloc alloc) const;
66         bool isMemberVariableAssignment(const Token *tok, const std::string &membervar) const;
67         bool isMemberVariableUsage(const Token *tok, bool isPointer, Alloc alloc, const std::string &membervar) const;
68     
69         /** ValueFlow-based checking for dead pointer usage */
70         void deadPointer();
71         void deadPointerError(const Token *pointer, const Token *alias);
72     
73         /* data for multifile checking */
74         class MyFileInfo : public Check::FileInfo {
75         public:
76             /* functions that must have initialized data */
77             std::set<std::string>  uvarFunctions;
78     
79             /* functions calls with uninitialized data */
80             std::set<std::string>  functionCalls;
81         };
82     
83         void uninitstringError(const Token *tok, const std::string &varname, bool strncpy_);
84         void uninitdataError(const Token *tok, const std::string &varname);
85         void uninitvarError(const Token *tok, const std::string &varname);
86         void uninitvarError(const Token *tok, const std::string &varname, Alloc alloc) {
87             if (alloc == NO_CTOR_CALL || alloc == CTOR_CALL)
88                 uninitdataError(tok, varname);
89             else
90                 uninitvarError(tok, varname);
91         }
92         void uninitStructMemberError(const Token *tok, const std::string &membername);
93     
94     private:
95         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
96             CheckUninitVar c(0, settings, errorLogger);
97     
98             // error
99             c.uninitstringError(0, "varname", true);
100             c.uninitdataError(0, "varname");
101             c.uninitvarError(0, "varname");
102             c.uninitStructMemberError(0, "a.b");
103             c.deadPointerError(0,0);
104         }
105     
106         static std::string myName() {
107             return "Uninitialized variables";
108         }
109     
110         std::string classInfo() const {
111             return "Uninitialized variables\n"
112                    "- using uninitialized local variables\n"
113                    "- using allocated data before it has been initialized\n"
114                    "- using dead pointer\n";
115         }
116     };
117     /// @}
118     //---------------------------------------------------------------------------
119     #endif // checkuninitvarH
