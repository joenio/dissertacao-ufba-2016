61     ErrorLogger::ErrorMessage::ErrorMessage(const std::list<const Token*>& callstack, const TokenList* list, Severity::SeverityType severity, const std::string& id, const std::string& msg, bool inconclusive)
62         : _id(id), _severity(severity), _cwe(0U), _inconclusive(inconclusive)
63     {
64         // Format callstack
65         for (std::list<const Token *>::const_iterator it = callstack.begin(); it != callstack.end(); ++it) {
66             // --errorlist can provide null values here
67             if (!(*it))
68                 continue;
69     
70             _callStack.push_back(ErrorLogger::ErrorMessage::FileLocation(*it, list));
71         }
72     
73         if (list && !list->getFiles().empty())
74             file0 = list->getFiles()[0];
75     
76         setmsg(msg);
77     }

79     void ErrorLogger::ErrorMessage::setmsg(const std::string &msg)
80     {
81         // If a message ends to a '\n' and contains only a one '\n'
82         // it will cause the _verboseMessage to be empty which will show
83         // as an empty message to the user if --verbose is used.
84         // Even this doesn't cause problems with messages that have multiple
85         // lines, none of the the error messages should end into it.
86         assert(!(msg.back() =='\n'));
87     
88         // The summary and verbose message are separated by a newline
89         // If there is no newline then both the summary and verbose messages
90         // are the given message
91         const std::string::size_type pos = msg.find('\n');
92         if (pos == std::string::npos) {
93             _shortMessage = msg;
94             _verboseMessage = msg;
95         } else {
96             _shortMessage = msg.substr(0, pos);
97             _verboseMessage = msg.substr(pos + 1);
98         }
99     }

101     std::string ErrorLogger::ErrorMessage::serialize() const
102     {
103         // Serialize this message into a simple string
104         std::ostringstream oss;
105         oss << _id.length() << " " << _id;
106         oss << Severity::toString(_severity).length() << " " << Severity::toString(_severity);
107         oss << MathLib::toString(_cwe).length() << " " << MathLib::toString(_cwe);
108         if (_inconclusive) {
109             const std::string inconclusive("inconclusive");
110             oss << inconclusive.length() << " " << inconclusive;
111         }
112     
113         const std::string saneShortMessage = fixInvalidChars(_shortMessage);
114         const std::string saneVerboseMessage = fixInvalidChars(_verboseMessage);
115     
116         oss << saneShortMessage.length() << " " << saneShortMessage;
117         oss << saneVerboseMessage.length() << " " << saneVerboseMessage;
118         oss << _callStack.size() << " ";
119     
120         for (std::list<ErrorLogger::ErrorMessage::FileLocation>::const_iterator tok = _callStack.begin(); tok != _callStack.end(); ++tok) {
121             std::ostringstream smallStream;
122             smallStream << (*tok).line << ":" << (*tok).getfile();
123             oss << smallStream.str().length() << " " << smallStream.str();
124         }
125     
126         return oss.str();
127     }

129     bool ErrorLogger::ErrorMessage::deserialize(const std::string &data)
130     {
131         _inconclusive = false;
132         _callStack.clear();
133         std::istringstream iss(data);
134         std::array<std::string, 5> results;
135         std::size_t elem = 0;
136         while (iss.good()) {
137             unsigned int len = 0;
138             if (!(iss >> len))
139                 return false;
140     
141             iss.get();
142             std::string temp;
143             for (unsigned int i = 0; i < len && iss.good(); ++i) {
144                 char c = static_cast<char>(iss.get());
145                 temp.append(1, c);
146             }
147     
148             if (temp == "inconclusive") {
149                 _inconclusive = true;
150                 continue;
151             }
152     
153             results[elem++] = temp;
154             if (elem == 5)
155                 break;
156         }
157     
158         if (elem != 5)
159             throw InternalError(0, "Internal Error: Deserialization of error message failed");
160     
161         _id = results[0];
162         _severity = Severity::fromString(results[1]);
163         std::istringstream scwe(results[2]);
164         scwe >> _cwe;
165         _shortMessage = results[3];
166         _verboseMessage = results[4];
167     
168         unsigned int stackSize = 0;
169         if (!(iss >> stackSize))
170             return false;
171     
172         while (iss.good()) {
173             unsigned int len = 0;
174             if (!(iss >> len))
175                 return false;
176     
177             iss.get();
178             std::string temp;
179             for (unsigned int i = 0; i < len && iss.good(); ++i) {
180                 char c = static_cast<char>(iss.get());
181                 temp.append(1, c);
182             }
183     
184             ErrorLogger::ErrorMessage::FileLocation loc;
185             loc.setfile(temp.substr(temp.find(':') + 1));
186             temp = temp.substr(0, temp.find(':'));
187             std::istringstream fiss(temp);
188             fiss >> loc.line;
189     
190             _callStack.push_back(loc);
191     
192             if (_callStack.size() >= stackSize)
193                 break;
194         }
195     
196         return true;
197     }

199     std::string ErrorLogger::ErrorMessage::getXMLHeader(int xml_version)
200     {
201         // xml_version 1 is the default xml format
202     
203         tinyxml2::XMLPrinter printer;
204     
205         // standard xml header
206         printer.PushDeclaration("xml version=\"1.0\" encoding=\"UTF-8\"");
207     
208         // header
209         printer.OpenElement("results", false);
210         // version 2 header
211         if (xml_version == 2) {
212             printer.PushAttribute("version", xml_version);
213             printer.OpenElement("cppcheck", false);
214             printer.PushAttribute("version", CppCheck::version());
215             printer.CloseElement(false);
216             printer.OpenElement("errors", false);
217         }
218     
219         return std::string(printer.CStr()) + '>';
220     }

227     // There is no utf-8 support around but the strings should at least be safe for to tinyxml2.
228     // See #5300 "Invalid encoding in XML output" and  #6431 "Invalid XML created - Invalid encoding of string literal "
229     std::string ErrorLogger::ErrorMessage::fixInvalidChars(const std::string& raw)
230     {
231         std::string result;
232         result.reserve(raw.length());
233         std::string::const_iterator from=raw.begin();
234         while (from!=raw.end()) {
235             if (std::isprint(static_cast<unsigned char>(*from))) {
236                 result.push_back(*from);
237             } else {
238                 std::ostringstream es;
239                 // straight cast to (unsigned) doesn't work out.
240                 const unsigned uFrom = (unsigned char)*from;
241     #if 0
242                 if (uFrom<0x20)
243                     es << "\\XXX";
244                 else
245     #endif
246                     es << '\\' << std::setbase(8) << std::setw(3) << std::setfill('0') << uFrom;
247                 result += es.str();
248             }
249             ++from;
250         }
251         return result;
252     }

254     std::string ErrorLogger::ErrorMessage::toXML(bool verbose, int version) const
255     {
256         // The default xml format
257         if (version == 1) {
258             // No inconclusive messages in the xml version 1
259             if (_inconclusive)
260                 return "";
261     
262             tinyxml2::XMLPrinter printer(0, false, 1);
263             printer.OpenElement("error", false);
264             if (!_callStack.empty()) {
265                 printer.PushAttribute("file", _callStack.back().getfile().c_str());
266                 printer.PushAttribute("line", _callStack.back().line);
267             }
268             printer.PushAttribute("id", _id.c_str());
269             printer.PushAttribute("severity", (_severity == Severity::error ? "error" : "style"));
270             printer.PushAttribute("msg", (verbose ? _verboseMessage : _shortMessage).c_str());
271             printer.CloseElement(false);
272             return printer.CStr();
273         }
274     
275         // The xml format you get when you use --xml-version=2
276         else if (version == 2) {
277             tinyxml2::XMLPrinter printer(0, false, 2);
278             printer.OpenElement("error", false);
279             printer.PushAttribute("id", _id.c_str());
280             printer.PushAttribute("severity", Severity::toString(_severity).c_str());
281             printer.PushAttribute("msg", fixInvalidChars(_shortMessage).c_str());
282             printer.PushAttribute("verbose", fixInvalidChars(_verboseMessage).c_str());
283             if (_cwe)
284                 printer.PushAttribute("cwe", _cwe);
285             if (_inconclusive)
286                 printer.PushAttribute("inconclusive", "true");
287     
288             for (std::list<FileLocation>::const_reverse_iterator it = _callStack.rbegin(); it != _callStack.rend(); ++it) {
289                 printer.OpenElement("location", false);
290                 printer.PushAttribute("file", (*it).getfile().c_str());
291                 printer.PushAttribute("line", (*it).line);
292                 printer.CloseElement(false);
293             }
294             printer.CloseElement(false);
295             return printer.CStr();
296         }
297     
298         return "";
299     }

301     void ErrorLogger::ErrorMessage::findAndReplace(std::string &source, const std::string &searchFor, const std::string &replaceWith)
302     {
303         std::string::size_type index = 0;
304         while ((index = source.find(searchFor, index)) != std::string::npos) {
305             source.replace(index, searchFor.length(), replaceWith);
306             index += replaceWith.length();
307         }
308     }

310     std::string ErrorLogger::ErrorMessage::toString(bool verbose, const std::string &outputFormat) const
311     {
312         // Save this ErrorMessage in plain text.
313     
314         // No template is given
315         if (outputFormat.length() == 0) {
316             std::ostringstream text;
317             if (!_callStack.empty())
318                 text << callStackToString(_callStack) << ": ";
319             if (_severity != Severity::none) {
320                 text << '(' << Severity::toString(_severity);
321                 if (_inconclusive)
322                     text << ", inconclusive";
323                 text << ") ";
324             }
325             text << (verbose ? _verboseMessage : _shortMessage);
326             return text.str();
327         }
328     
329         // template is given. Reformat the output according to it
330         else {
331             std::string result = outputFormat;
332             // Support a few special characters to allow to specific formatting, see http://sourceforge.net/apps/phpbb/cppcheck/viewtopic.php?f=4&t=494&sid=21715d362c0dbafd3791da4d9522f814
333             // Substitution should be done first so messages from cppcheck never get translated.
334             findAndReplace(result, "\\b", "\b");
335             findAndReplace(result, "\\n", "\n");
336             findAndReplace(result, "\\r", "\r");
337             findAndReplace(result, "\\t", "\t");
338     
339             findAndReplace(result, "{id}", _id);
340             findAndReplace(result, "{severity}", Severity::toString(_severity));
341             findAndReplace(result, "{message}", verbose ? _verboseMessage : _shortMessage);
342             findAndReplace(result, "{callstack}", _callStack.empty() ? "" : callStackToString(_callStack));
343             if (!_callStack.empty()) {
344                 std::ostringstream oss;
345                 oss << _callStack.back().line;
346                 findAndReplace(result, "{line}", oss.str());
347                 findAndReplace(result, "{file}", _callStack.back().getfile());
348             } else {
349                 findAndReplace(result, "{file}", "");
350                 findAndReplace(result, "{line}", "");
351             }
352     
353             return result;
354         }
355     }

357     void ErrorLogger::reportUnmatchedSuppressions(const std::list<Suppressions::SuppressionEntry> &unmatched)
358     {
359         // Report unmatched suppressions
360         for (std::list<Suppressions::SuppressionEntry>::const_iterator i = unmatched.begin(); i != unmatched.end(); ++i) {
361             // don't report "unmatchedSuppression" as unmatched
362             if (i->id == "unmatchedSuppression")
363                 continue;
364     
365             // check if this unmatched suppression is suppressed
366             bool suppressed = false;
367             for (std::list<Suppressions::SuppressionEntry>::const_iterator i2 = unmatched.begin(); i2 != unmatched.end(); ++i2) {
368                 if (i2->id == "unmatchedSuppression") {
369                     if ((i2->file == "*" || i2->file == i->file) &&
370                         (i2->line == 0 || i2->line == i->line)) {
371                         suppressed = true;
372                         break;
373                     }
374                 }
375             }
376     
377             if (suppressed)
378                 continue;
379     
380             std::list<ErrorLogger::ErrorMessage::FileLocation> callStack;
381             callStack.push_back(ErrorLogger::ErrorMessage::FileLocation(i->file, i->line));
382             reportErr(ErrorLogger::ErrorMessage(callStack, Severity::information, "Unmatched suppression: " + i->id, "unmatchedSuppression", false));
383         }
384     }

386     std::string ErrorLogger::callStackToString(const std::list<ErrorLogger::ErrorMessage::FileLocation> &callStack)
387     {
388         std::ostringstream ostr;
389         for (std::list<ErrorLogger::ErrorMessage::FileLocation>::const_iterator tok = callStack.begin(); tok != callStack.end(); ++tok) {
390             ostr << (tok == callStack.begin() ? "" : " -> ") << tok->stringify();
391         }
392         return ostr.str();
393     }

425     std::string ErrorLogger::toxml(const std::string &str)
426     {
427         std::ostringstream xml;
428         const bool isstring(str[0] == '\"');
429         for (std::size_t i = 0U; i < str.length(); i++) {
430             char c = str[i];
431             switch (c) {
432             case '<':
433                 xml << "&lt;";
434                 break;
435             case '>':
436                 xml << "&gt;";
437                 break;
438             case '&':
439                 xml << "&amp;";
440                 break;
441             case '\"':
442                 xml << "&quot;";
443                 break;
444             case '\0':
445                 xml << "\\0";
446                 break;
447             default:
448                 if (!isstring || (c >= ' ' && c <= 'z'))
449                     xml << c;
450                 else
451                     xml << 'x';
452                 break;
453             }
454         }
455         return xml.str();
456     }
