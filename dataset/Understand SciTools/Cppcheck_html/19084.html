30     //---------------------------------------------------------------------------
31     // Ensure that correct parameter is passed to va_start()
32     //---------------------------------------------------------------------------
33     
34     void CheckVaarg::va_start_argument()
35     {
36         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
37         const std::size_t functions = symbolDatabase->functionScopes.size();
38         const bool printWarnings = _settings->isEnabled("warning");
39     
40         for (std::size_t i = 0; i < functions; ++i) {
41             const Scope* scope = symbolDatabase->functionScopes[i];
42             const Function* function = scope->function;
43             if (!function)
44                 continue;
45             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
46                 if (!tok->scope()->isExecutable())
47                     tok = tok->scope()->classEnd;
48                 else if (Token::simpleMatch(tok, "va_start (")) {
49                     const Token* param2 = tok->tokAt(2)->nextArgument();
50                     if (!param2)
51                         continue;
52                     const Variable* var = param2->variable();
53                     if (var && var->isReference())
54                         referenceAs_va_start_error(param2, var->name());
55                     if (var && var->index() + 2 < function->argCount() && printWarnings) {
56                         std::list<Variable>::const_reverse_iterator it = function->argumentList.rbegin();
57                         ++it;
58                         wrongParameterTo_va_start_error(tok, var->name(), it->name());
59                     }
60                     tok = tok->linkAt(1);
61                 }
62             }
63         }
64     }

66     void CheckVaarg::wrongParameterTo_va_start_error(const Token *tok, const std::string& paramIsName, const std::string& paramShouldName)
67     {
68         reportError(tok, Severity::warning,
69                     "va_start_wrongParameter", "'" + paramIsName + "' given to va_start() is not last named argument of the function. Did you intend to pass '" + paramShouldName + "'?");
70     }

72     void CheckVaarg::referenceAs_va_start_error(const Token *tok, const std::string& paramName)
73     {
74         reportError(tok, Severity::error,
75                     "va_start_referencePassed", "Using reference '" + paramName + "' as parameter for va_start() results in undefined behaviour.");
76     }

78     //---------------------------------------------------------------------------
79     // Detect missing va_end() if va_start() was used
80     // Detect va_list usage after va_end()
81     //---------------------------------------------------------------------------
82     
83     void CheckVaarg::va_list_usage()
84     {
85         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
86         for (unsigned int varid = 1; varid < symbolDatabase->getVariableListSize(); varid++) {
87             const Variable* var = symbolDatabase->getVariableFromVarId(varid);
88             if (!var || var->isPointer() || var->isReference() || var->isArray() || !var->scope() || var->typeStartToken()->str() != "va_list")
89                 continue;
90             if (!var->isLocal() && !var->isArgument()) // Check only local variables and arguments
91                 continue;
92     
93             bool open = var->isArgument(); // va_list passed as argument are opened
94             bool exitOnEndOfStatement = false;
95     
96             const Token* tok = var->nameToken()->next();
97             for (;  tok && tok != var->scope()->classEnd; tok = tok->next()) {
98                 if (Token::Match(tok, "va_start ( %varid%", var->declarationId())) {
99                     if (open)
100                         va_start_subsequentCallsError(tok, var->name());
101                     open = true;
102                     tok = tok->linkAt(1);
103                 } else if (Token::Match(tok, "va_end ( %varid%", var->declarationId())) {
104                     if (!open)
105                         va_list_usedBeforeStartedError(tok, var->name());
106                     open = false;
107                     tok = tok->linkAt(1);
108                 } else if (Token::simpleMatch(tok, "va_copy (")) {
109                     bool nopen = open;
110                     if (tok->linkAt(1)->previous()->varId() == var->declarationId()) { // Source
111                         if (!open)
112                             va_list_usedBeforeStartedError(tok, var->name());
113                     }
114                     if (tok->tokAt(2)->varId() == var->declarationId()) { // Destination
115                         if (open)
116                             va_start_subsequentCallsError(tok, var->name());
117                         nopen = true;
118                     }
119                     open = nopen;
120                     tok = tok->linkAt(1);
121                 } else if (Token::Match(tok, "throw|return"))
122                     exitOnEndOfStatement = true;
123                 else if (_tokenizer->isCPP() && tok->str() == "try") {
124                     open = false;
125                     break;
126                 } else if (!open && tok->varId() == var->declarationId())
127                     va_list_usedBeforeStartedError(tok, var->name());
128                 else if (exitOnEndOfStatement && tok->str() == ";")
129                     break;
130             }
131             if (open && !var->isArgument())
132                 va_end_missingError(tok, var->name());
133         }
134     }

136     void CheckVaarg::va_end_missingError(const Token *tok, const std::string& varname)
137     {
138         reportError(tok, Severity::error,
139                     "va_end_missing", "va_list '" + varname + "' was opened but not closed by va_end().");
140     }
141     
142     void CheckVaarg::va_list_usedBeforeStartedError(const Token *tok, const std::string& varname)
143     {
144         reportError(tok, Severity::error,
145                     "va_list_usedBeforeStarted", "va_list '" + varname + "' used before va_start() was called.");
146     }
147     
148     void CheckVaarg::va_start_subsequentCallsError(const Token *tok, const std::string& varname)
149     {
150         reportError(tok, Severity::error,
151                     "va_start_subsequentCalls", "va_start() or va_copy() called subsequently on '" + varname + "' without va_end() in between.");
152     }
