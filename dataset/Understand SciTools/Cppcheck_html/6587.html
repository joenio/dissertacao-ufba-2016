20     //---------------------------------------------------------------------------
21     #ifndef checkconditionH
22     #define checkconditionH
23     //---------------------------------------------------------------------------
24     
25     #include "config.h"
26     #include "check.h"
27     #include "mathlib.h"
28     
29     /// @addtogroup Checks
30     /// @{
31     
32     /**
33      * @brief Check for condition mismatches
34      */
35     
36     class CPPCHECKLIB CheckCondition : public Check {
37     public:
38         /** This constructor is used when registering the CheckAssignIf */
39         CheckCondition() : Check(myName()) {
40         }
41     
42         /** This constructor is used when running checks. */
43         CheckCondition(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
44             : Check(myName(), tokenizer, settings, errorLogger) {
45         }
46     
47         void runChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
48             CheckCondition checkCondition(tokenizer, settings, errorLogger);
49             checkCondition.multiCondition();
50             checkCondition.clarifyCondition();   // not simplified because ifAssign
51             checkCondition.oppositeInnerCondition();
52             checkCondition.checkIncorrectLogicOperator();
53             checkCondition.checkInvalidTestForOverflow();
54         }
55     
56         /** @brief Run checks against the simplified token list */
57         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
58             CheckCondition checkCondition(tokenizer, settings, errorLogger);
59             checkCondition.assignIf();
60             checkCondition.checkBadBitmaskCheck();
61             checkCondition.comparison();
62             checkCondition.checkModuloAlwaysTrueFalse();
63             checkCondition.alwaysTrueFalse();
64         }
65     
66         /** mismatching assignment / comparison */
67         void assignIf();
68     
69         /** parse scopes recursively */
70         bool assignIfParseScope(const Token * const assignTok,
71                                 const Token * const startTok,
72                                 const unsigned int varid,
73                                 const bool islocal,
74                                 const char bitop,
75                                 const MathLib::bigint num);
76     
77         /** check bitmask using | instead of & */
78         void checkBadBitmaskCheck();
79     
80         /** mismatching lhs and rhs in comparison */
81         void comparison();
82     
83         /** match 'if' and 'else if' conditions */
84         void multiCondition();
85     
86         /** To check the dead code in a program, which is inaccessible due to the counter-conditions check in nested-if statements **/
87         void oppositeInnerCondition();
88     
89         /** @brief %Check for testing for mutual exclusion over ||*/
90         void checkIncorrectLogicOperator();
91     
92         /** @brief %Check for suspicious usage of modulo (e.g. "if(var % 4 == 4)") */
93         void checkModuloAlwaysTrueFalse();
94     
95         /** @brief Suspicious condition (assignment+comparison) */
96         void clarifyCondition();
97     
98         /** @brief Condition is always true/false */
99         void alwaysTrueFalse();
100     
101         /** @brief %Check for invalid test for overflow 'x+100 < x' */
102         void checkInvalidTestForOverflow();
103     
104     private:
105     
106         bool isOverlappingCond(const Token * const cond1, const Token * const cond2, const std::set<std::string> &constFunctions) const;
107         void assignIfError(const Token *tok1, const Token *tok2, const std::string &condition, bool result);
108         void mismatchingBitAndError(const Token *tok1, const MathLib::bigint num1, const Token *tok2, const MathLib::bigint num2);
109         void badBitmaskCheckError(const Token *tok);
110         void comparisonError(const Token *tok,
111                              const std::string &bitop,
112                              MathLib::bigint value1,
113                              const std::string &op,
114                              MathLib::bigint value2,
115                              bool result);
116         void multiConditionError(const Token *tok, unsigned int line1);
117     
118         void oppositeInnerConditionError(const Token *tok1, const Token* tok2);
119     
120         void incorrectLogicOperatorError(const Token *tok, const std::string &condition, bool always);
121         void redundantConditionError(const Token *tok, const std::string &text);
122     
123         void moduloAlwaysTrueFalseError(const Token* tok, const std::string& maxVal);
124     
125         void clarifyConditionError(const Token *tok, bool assign, bool boolop);
126     
127         void alwaysTrueFalseError(const Token *tok, bool knownResult);
128     
129         void invalidTestForOverflow(const Token* tok, bool result);
130     
131         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
132             CheckCondition c(0, settings, errorLogger);
133     
134             c.assignIfError(0, 0, "", false);
135             c.badBitmaskCheckError(0);
136             c.comparisonError(0, "&", 6, "==", 1, false);
137             c.multiConditionError(0,1);
138             c.mismatchingBitAndError(0, 0xf0, 0, 1);
139             c.oppositeInnerConditionError(0, 0);
140             c.incorrectLogicOperatorError(0, "foo > 3 && foo < 4", true);
141             c.redundantConditionError(0, "If x > 11 the condition x > 10 is always true.");
142             c.moduloAlwaysTrueFalseError(0, "1");
143             c.clarifyConditionError(0, true, false);
144             c.alwaysTrueFalseError(0, true);
145             c.invalidTestForOverflow(0, false);
146         }
147     
148         static std::string myName() {
149             return "Condition";
150         }
151     
152         std::string classInfo() const {
153             return "Match conditions with assignments and other conditions:\n"
154                    "- Mismatching assignment and comparison => comparison is always true/false\n"
155                    "- Mismatching lhs and rhs in comparison => comparison is always true/false\n"
156                    "- Detect usage of | where & should be used\n"
157                    "- Detect matching 'if' and 'else if' conditions\n"
158                    "- Mismatching bitand (a &= 0xf0; a &= 1; => a = 0)\n"
159                    "- Find dead code which is inaccessible due to the counter-conditions check in nested if statements\n"
160                    "- Condition that is always true/false\n"
161                    "- Mutual exclusion over || always evaluating to true\n"
162                    "- Comparisons of modulo results that are always true/false.\n"
163                    "- Known variable values => condition is always true/false\n"
164                    "- Invalid test for overflow (for example 'ptr+u < ptr'). Condition is always false unless there is overflow, and overflow is UB.\n";
165         }
166     };
167     /// @}
168     //---------------------------------------------------------------------------
169     #endif // checkconditionH
