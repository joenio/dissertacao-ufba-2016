35             TEST_CASE(novardecl);

43         void check(const char code[]) {
44             // Clear the error buffer..
45             errout.str("");
46     
47             // Tokenize..
48             Tokenizer tokenizer(&settings, this);
49             std::istringstream istr(code);
50             tokenizer.tokenize(istr, "test.cpp");
51     
52             // Check char variable usage..
53             Check64BitPortability check64BitPortability(&tokenizer, &settings, this);
54             check64BitPortability.pointerassignment();
55         }
56     
57         void novardecl() {
58             // if the variable declarations can't be seen then skip the warning
59             check("void foo()\n"
60                   "{\n"
61                   "    a = p;\n"
62                   "}");
63             ASSERT_EQUALS("", errout.str());
64         }
65     
66         void functionpar() {
67             check("int foo(int *p)\n"
68                   "{\n"
69                   "    int a = p;\n"
70                   "    return a + 4;\n"
71                   "}");
72             ASSERT_EQUALS("[test.cpp:3]: (portability) Assigning a pointer to an integer is not portable.\n", errout.str());
73     
74             check("int foo(int p[])\n"
75                   "{\n"
76                   "    int a = p;\n"
77                   "    return a + 4;\n"
78                   "}");
79             ASSERT_EQUALS("[test.cpp:3]: (portability) Assigning a pointer to an integer is not portable.\n", errout.str());
80     
81             check("int foo(int p[])\n"
82                   "{\n"
83                   "    int *a = p;\n"
84                   "    return a;\n"
85                   "}");
86             ASSERT_EQUALS("[test.cpp:4]: (portability) Returning an address value in a function with integer return type is not portable.\n", errout.str());
87     
88             check("void foo(int x)\n"
89                   "{\n"
90                   "    int *p = x;\n"
91                   "    *p = 0;\n"
92                   "}");
93             ASSERT_EQUALS("[test.cpp:3]: (portability) Assigning an integer to a pointer is not portable.\n", errout.str());
94     
95             check("int f(const char *p) {\n" // #4659
96                   "    return 6 + p[2] * 256;\n"
97                   "}");
98             ASSERT_EQUALS("", errout.str());
99     
100             check("int foo(int *p) {\n" // #6096
101                   "    bool a = p;\n"
102                   "    return a;\n"
103                   "}");
104             ASSERT_EQUALS("", errout.str());
105         }
