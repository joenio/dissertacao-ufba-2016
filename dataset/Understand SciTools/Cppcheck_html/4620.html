30     static CppcheckLibraryData::Container loadContainer(QXmlStreamReader &xmlReader)
31     {
32         CppcheckLibraryData::Container container;
33         container.id           = xmlReader.attributes().value("id").toString();
34         container.inherits     = xmlReader.attributes().value("inherits").toString();
35         container.startPattern = xmlReader.attributes().value("startPattern").toString();
36         container.endPattern   = xmlReader.attributes().value("endPattern").toString();
37     
38         QXmlStreamReader::TokenType type;
39         while ((type = xmlReader.readNext()) != QXmlStreamReader::EndElement ||
40                xmlReader.name().toString() != "container") {
41             if (type != QXmlStreamReader::StartElement)
42                 continue;
43             const QString elementName = xmlReader.name().toString();
44             if (elementName == "type") {
45                 container.type.templateParameter = xmlReader.attributes().value("templateParameter").toString();
46                 container.type.string            = xmlReader.attributes().value("string").toString();
47             } else if (elementName == "size" || elementName == "access" || elementName == "other") {
48                 const QString indexOperator = xmlReader.attributes().value("indexOperator").toString();
49                 if (elementName == "access" && indexOperator == "array-like")
50                     container.access_arrayLike = true;
51                 const QString templateParameter = xmlReader.attributes().value("templateParameter").toString();
52                 if (elementName == "size" && !templateParameter.isEmpty())
53                     container.size_templateParameter = templateParameter.toInt();
54                 for (;;) {
55                     type = xmlReader.readNext();
56                     if (xmlReader.name().toString() == elementName)
57                         break;
58                     if (type != QXmlStreamReader::StartElement)
59                         continue;
60                     struct CppcheckLibraryData::Container::Function function;
61                     function.name   = xmlReader.attributes().value("name").toString();
62                     function.action = xmlReader.attributes().value("action").toString();
63                     function.yields = xmlReader.attributes().value("yields").toString();
64                     if (elementName == "size")
65                         container.sizeFunctions.append(function);
66                     else if (elementName == "access")
67                         container.accessFunctions.append(function);
68                     else
69                         container.otherFunctions.append(function);
70                 };
71             }
72         }
73         return container;
74     }

84     static CppcheckLibraryData::Function::Arg loadFunctionArg(QXmlStreamReader &xmlReader)
85     {
86         CppcheckLibraryData::Function::Arg arg;
87         QString argnr = xmlReader.attributes().value("nr").toString();
88         if (argnr == "any")
89             arg.nr = CppcheckLibraryData::Function::Arg::ANY;
90         else
91             arg.nr = argnr.toUInt();
92     
93         QXmlStreamReader::TokenType type;
94         while ((type = xmlReader.readNext()) != QXmlStreamReader::EndElement ||
95                xmlReader.name().toString() != "arg") {
96             if (type != QXmlStreamReader::StartElement)
97                 continue;
98             const QString elementName = xmlReader.name().toString();
99             if (elementName == "not-bool")
100                 arg.notbool = true;
101             else if (elementName == "not-null")
102                 arg.notnull = true;
103             else if (elementName == "not-uninit")
104                 arg.notuninit = true;
105             else if (elementName == "strz")
106                 arg.strz = true;
107             else if (elementName == "formatstr")
108                 arg.formatstr = true;
109             else if (elementName == "valid")
110                 arg.valid = xmlReader.readElementText();
111             else if (elementName == "minsize") {
112                 CppcheckLibraryData::Function::Arg::MinSize minsize;
113                 minsize.type = xmlReader.attributes().value("type").toString();
114                 minsize.arg  = xmlReader.attributes().value("arg").toString();
115                 minsize.arg2 = xmlReader.attributes().value("arg2").toString();
116                 arg.minsizes.append(minsize);
117             }
118         }
119         return arg;
120     }

122     static CppcheckLibraryData::Function loadFunction(QXmlStreamReader &xmlReader, const QString comments)
123     {
124         CppcheckLibraryData::Function function;
125         function.comments = comments;
126         function.name = xmlReader.attributes().value("name").toString();
127         QXmlStreamReader::TokenType type;
128         while ((type = xmlReader.readNext()) != QXmlStreamReader::EndElement ||
129                xmlReader.name().toString() != "function") {
130             if (type != QXmlStreamReader::StartElement)
131                 continue;
132             const QString elementName = xmlReader.name().toString();
133             if (elementName == "noreturn")
134                 function.noreturn = (xmlReader.readElementText() == "true") ? CppcheckLibraryData::Function::True : CppcheckLibraryData::Function::False;
135             else if (elementName == "pure")
136                 function.gccPure = true;
137             else if (elementName == "const")
138                 function.gccConst = true;
139             else if (elementName == "leak-ignore")
140                 function.leakignore = true;
141             else if (elementName == "use-retval")
142                 function.useretval = true;
143             else if (elementName == "formatstr") {
144                 function.formatstr.scan   = xmlReader.attributes().value("scan").toString();
145                 function.formatstr.secure = xmlReader.attributes().value("secure").toString();
146             } else if (elementName == "arg")
147                 function.args.append(loadFunctionArg(xmlReader));
148             else if (elementName == "warn") {
149                 function.warn.severity     = xmlReader.attributes().value("severity").toString();
150                 function.warn.reason       = xmlReader.attributes().value("reason").toString();
151                 function.warn.alternatives = xmlReader.attributes().value("alternatives").toString();
152                 function.warn.msg          = xmlReader.readElementText();
153             }
154         }
155         return function;
156     }

158     static CppcheckLibraryData::MemoryResource loadMemoryResource(QXmlStreamReader &xmlReader)
159     {
160         CppcheckLibraryData::MemoryResource memoryresource;
161         memoryresource.type = xmlReader.name().toString();
162         QXmlStreamReader::TokenType type;
163         while ((type = xmlReader.readNext()) != QXmlStreamReader::EndElement ||
164                xmlReader.name().toString() != memoryresource.type) {
165             if (type != QXmlStreamReader::StartElement)
166                 continue;
167             const QString elementName = xmlReader.name().toString();
168             if (elementName == "alloc") {
169                 CppcheckLibraryData::MemoryResource::Alloc alloc;
170                 alloc.init = (xmlReader.attributes().value("init").toString() == "true");
171                 alloc.name = xmlReader.readElementText();
172                 memoryresource.alloc.append(alloc);
173             } else if (elementName == "dealloc")
174                 memoryresource.dealloc.append(xmlReader.readElementText());
175             else if (elementName == "use")
176                 memoryresource.use.append(xmlReader.readElementText());
177         }
178         return memoryresource;
179     }

190     bool CppcheckLibraryData::open(QIODevice &file)
191     {
192         clear();
193         QString comments;
194         QXmlStreamReader xmlReader(&file);
195         while (!xmlReader.atEnd()) {
196             const QXmlStreamReader::TokenType t = xmlReader.readNext();
197             switch (t) {
198             case QXmlStreamReader::Comment:
199                 if (!comments.isEmpty())
200                     comments += "\n";
201                 comments += xmlReader.text().toString();
202                 break;
203             case QXmlStreamReader::StartElement:
204                 if (xmlReader.name() == "container")
205                     containers.append(loadContainer(xmlReader));
206                 if (xmlReader.name() == "define")
207                     defines.append(loadDefine(xmlReader));
208                 else if (xmlReader.name() == "function")
209                     functions.append(loadFunction(xmlReader, comments));
210                 else if (xmlReader.name() == "memory" || xmlReader.name() == "resource")
211                     memoryresource.append(loadMemoryResource(xmlReader));
212                 else if (xmlReader.name() == "podtype")
213                     podtypes.append(loadPodType(xmlReader));
214                 comments.clear();
215                 break;
216             default:
217                 break;
218             }
219         }
220     
221         return true;
222     }

224     static void writeContainerFunctions(QXmlStreamWriter &xmlWriter, const QString name, int extra, const QList<struct CppcheckLibraryData::Container::Function> &functions)
225     {
226         if (functions.isEmpty() && extra <= 0)
227             return;
228         xmlWriter.writeStartElement(name);
229         if (extra > 0) {
230             if (name == "access")
231                 xmlWriter.writeAttribute("indexOperator", "array-like");
232             else if (name == "size")
233                 xmlWriter.writeAttribute("templateParameter", QString::number(extra));
234         }
235         foreach (const CppcheckLibraryData::Container::Function &function, functions) {
236             xmlWriter.writeStartElement("function");
237             xmlWriter.writeAttribute("name", function.name);
238             if (!function.action.isEmpty())
239                 xmlWriter.writeAttribute("action", function.action);
240             if (!function.yields.isEmpty())
241                 xmlWriter.writeAttribute("yields", function.yields);
242             xmlWriter.writeEndElement();
243         }
244         xmlWriter.writeEndElement();
245     }

247     static void writeContainer(QXmlStreamWriter &xmlWriter, const CppcheckLibraryData::Container &container)
248     {
249         xmlWriter.writeStartElement("container");
250         xmlWriter.writeAttribute("id", container.id);
251         if (!container.startPattern.isEmpty())
252             xmlWriter.writeAttribute("startPattern", container.startPattern);
253         if (!container.endPattern.isNull())
254             xmlWriter.writeAttribute("endPattern", container.endPattern);
255         if (!container.inherits.isEmpty())
256             xmlWriter.writeAttribute("inherits", container.inherits);
257         if (!container.type.templateParameter.isEmpty() || !container.type.string.isEmpty()) {
258             xmlWriter.writeStartElement("type");
259             if (!container.type.templateParameter.isEmpty())
260                 xmlWriter.writeAttribute("templateParameter", container.type.templateParameter);
261             if (!container.type.string.isEmpty())
262                 xmlWriter.writeAttribute("string", container.type.string);
263             xmlWriter.writeEndElement();
264         }
265         writeContainerFunctions(xmlWriter, "size", container.size_templateParameter, container.sizeFunctions);
266         writeContainerFunctions(xmlWriter, "access", container.access_arrayLike, container.accessFunctions);
267         writeContainerFunctions(xmlWriter, "other", 0, container.otherFunctions);
268         xmlWriter.writeEndElement();
269     }

271     static void writeFunction(QXmlStreamWriter &xmlWriter, const CppcheckLibraryData::Function &function)
272     {
273         QString comments = function.comments;
274         while (comments.startsWith("\n"))
275             comments = comments.mid(1);
276         while (comments.endsWith("\n"))
277             comments.chop(1);
278         foreach (const QString &comment, comments.split('\n')) {
279             if (comment.length() >= 1)
280                 xmlWriter.writeComment(comment);
281         }
282     
283         xmlWriter.writeStartElement("function");
284         xmlWriter.writeAttribute("name", function.name);
285     
286         if (function.useretval)
287             xmlWriter.writeEmptyElement("use-retval");
288         if (function.gccConst)
289             xmlWriter.writeEmptyElement("const");
290         if (function.gccPure)
291             xmlWriter.writeEmptyElement("pure");
292         if (function.noreturn != CppcheckLibraryData::Function::Unknown)
293             xmlWriter.writeTextElement("noreturn", (function.noreturn == CppcheckLibraryData::Function::True) ? "true" : "false");
294         if (function.leakignore)
295             xmlWriter.writeEmptyElement("leak-ignore");
296         // Argument info..
297         foreach (const CppcheckLibraryData::Function::Arg &arg, function.args) {
298             if (arg.formatstr) {
299                 xmlWriter.writeStartElement("formatstr");
300                 if (!function.formatstr.scan.isNull())
301                     xmlWriter.writeAttribute("scan", function.formatstr.scan);
302                 if (!function.formatstr.secure.isNull())
303                     xmlWriter.writeAttribute("secure", function.formatstr.secure);
304                 xmlWriter.writeEndElement();
305             }
306     
307             xmlWriter.writeStartElement("arg");
308             if (arg.nr == CppcheckLibraryData::Function::Arg::ANY)
309                 xmlWriter.writeAttribute("nr", "any");
310             else
311                 xmlWriter.writeAttribute("nr", QString::number(arg.nr));
312             if (arg.formatstr)
313                 xmlWriter.writeEmptyElement("formatstr");
314             if (arg.notnull)
315                 xmlWriter.writeEmptyElement("not-null");
316             if (arg.notuninit)
317                 xmlWriter.writeEmptyElement("not-uninit");
318             if (arg.notbool)
319                 xmlWriter.writeEmptyElement("not-bool");
320             if (arg.strz)
321                 xmlWriter.writeEmptyElement("strz");
322     
323             if (!arg.valid.isEmpty())
324                 xmlWriter.writeTextElement("valid",arg.valid);
325     
326             foreach (const CppcheckLibraryData::Function::Arg::MinSize &minsize, arg.minsizes) {
327                 xmlWriter.writeStartElement("minsize");
328                 xmlWriter.writeAttribute("type", minsize.type);
329                 xmlWriter.writeAttribute("arg", minsize.arg);
330                 if (!minsize.arg2.isEmpty())
331                     xmlWriter.writeAttribute("arg2", minsize.arg2);
332                 xmlWriter.writeEndElement();
333             }
334     
335             xmlWriter.writeEndElement();
336         }
337     
338         if (!function.warn.isEmpty()) {
339             xmlWriter.writeStartElement("warn");
340     
341             if (!function.warn.severity.isEmpty())
342                 xmlWriter.writeAttribute("severity", function.warn.severity);
343     
344             if (!function.warn.reason.isEmpty())
345                 xmlWriter.writeAttribute("reason", function.warn.reason);
346     
347             if (!function.warn.alternatives.isEmpty())
348                 xmlWriter.writeAttribute("alternatives", function.warn.alternatives);
349     
350             if (!function.warn.msg.isEmpty())
351                 xmlWriter.writeCharacters(function.warn.msg);
352     
353             xmlWriter.writeEndElement();
354         }
355     
356         xmlWriter.writeEndElement();
357     }

359     static void writeMemoryResource(QXmlStreamWriter &xmlWriter, const CppcheckLibraryData::MemoryResource &mr)
360     {
361         xmlWriter.writeStartElement(mr.type);
362         foreach (const CppcheckLibraryData::MemoryResource::Alloc &alloc, mr.alloc) {
363             xmlWriter.writeStartElement("alloc");
364             xmlWriter.writeAttribute("init", alloc.init ? "true" : "false");
365             xmlWriter.writeCharacters(alloc.name);
366             xmlWriter.writeEndElement();
367         }
368         foreach (const QString &dealloc, mr.dealloc) {
369             xmlWriter.writeTextElement("dealloc", dealloc);
370         }
371         foreach (const QString &use, mr.use) {
372             xmlWriter.writeTextElement("use", use);
373         }
374         xmlWriter.writeEndElement();
375     }

377     QString CppcheckLibraryData::toString() const
378     {
379         QString outputString;
380         QXmlStreamWriter xmlWriter(&outputString);
381         xmlWriter.setAutoFormatting(true);
382         xmlWriter.setAutoFormattingIndent(2);
383         xmlWriter.writeStartDocument("1.0");
384         xmlWriter.writeStartElement("def");
385         xmlWriter.writeAttribute("format","2");
386     
387         foreach (const Define &define, defines) {
388             xmlWriter.writeStartElement("define");
389             xmlWriter.writeAttribute("name", define.name);
390             xmlWriter.writeAttribute("value", define.value);
391             xmlWriter.writeEndElement();
392         }
393     
394         foreach (const Function &function, functions) {
395             writeFunction(xmlWriter, function);
396         }
397     
398         foreach (const MemoryResource &mr, memoryresource) {
399             writeMemoryResource(xmlWriter, mr);
400         }
401     
402         foreach (const Container &container, containers) {
403             writeContainer(xmlWriter, container);
404         }
405     
406         foreach (const PodType &podtype, podtypes) {
407             xmlWriter.writeStartElement("podtype");
408             xmlWriter.writeAttribute("name", podtype.name);
409             if (!podtype.sign.isEmpty())
410                 xmlWriter.writeAttribute("sign", podtype.sign);
411             if (!podtype.size.isEmpty())
412                 xmlWriter.writeAttribute("size", podtype.size);
413             xmlWriter.writeEndElement();
414         }
415     
416         xmlWriter.writeEndElement();
417     
418         return outputString;
419     }
