78     bool TestFixture::prepareTest(const char testname[])
79     {
80         // Check if tests should be executed
81         if (testToRun.empty() || testToRun == testname) {
82             // Tests will be executed - prepare them
83             ++countTests;
84             if (quiet_tests) {
85                 std::putchar('.'); // Use putchar to write through redirection of std::cout/cerr
86                 std::fflush(stdout);
87             } else {
88                 std::cout << classname << "::" << testname << std::endl;
89             }
90             currentTest = classname + "::" + testname;
91             return true;
92         }
93         return false;
94     }

96     static std::string writestr(const std::string &str, bool gccStyle = false)
97     {
98         std::ostringstream ostr;
99         if (gccStyle)
100             ostr << '\"';
101         for (std::string::const_iterator i = str.begin(); i != str.end(); ++i) {
102             if (*i == '\n') {
103                 ostr << "\\n";
104                 if ((i+1) != str.end() && !gccStyle)
105                     ostr << std::endl;
106             } else if (*i == '\t')
107                 ostr << "\\t";
108             else if (*i == '\"')
109                 ostr << "\\\"";
110             else
111                 ostr << *i;
112         }
113         if (!str.empty() && !gccStyle)
114             ostr << std::endl;
115         else if (gccStyle)
116             ostr << '\"';
117         return ostr.str();
118     }

121     {

132     void TestFixture::assertEquals(const char *filename, unsigned int linenr, const std::string &expected, const std::string &actual, const std::string &msg) const
133     {
134         if (expected != actual) {
135             ++fails_counter;
136             if (gcc_style_errors) {
137                 errmsg << filename << ':' << linenr << ": Assertion failed. "
138                        << "Expected: "
139                        << writestr(expected, true)
140                        << ". Actual: "
141                        << writestr(actual, true)
142                        << '.'
143                        << std::endl;
144                 if (!msg.empty())
145                     errmsg << msg << std::endl;
146             } else {
147                 errmsg << "Assertion failed in " << filename << " at line " << linenr << std::endl
148                        << "Expected:" << std::endl
149                        << writestr(expected) << std::endl
150                        << "Actual:" << std::endl
151                        << writestr(actual) << std::endl;
152                 if (!msg.empty())
153                     errmsg << "Hint:" << std::endl << msg << std::endl;
154                 errmsg << "_____" << std::endl;
155             }
156         }
157     }

189     void TestFixture::todoAssertEquals(const char *filename, unsigned int linenr,
190                                        const std::string &wanted,
191                                        const std::string &current,
192                                        const std::string &actual) const
193     {
194         if (wanted == actual) {
195             if (gcc_style_errors) {
196                 errmsg << filename << ':' << linenr << ": Assertion succeeded unexpectedly. "
197                        << "Result: " << writestr(wanted, true) << "." << std::endl;
198             } else {
199                 errmsg << "Assertion succeeded unexpectedly in " << filename << " at line " << linenr << std::endl
200                        << "Result:" << std::endl << writestr(wanted) << std::endl << "_____" << std::endl;
201             }
202             ++succeeded_todos_counter;
203         } else {
204             assertEquals(filename, linenr, current, actual);
205             ++todos_counter;
206         }
207     }

235     void TestFixture::run(const std::string &str)
236     {
237         testToRun = str;
238         if (quiet_tests) {
239             std::cout << '\n' << classname << ':';
240         }
241         if (quiet_tests) {
242             REDIRECT;
243             run();
244         } else
245             run();
246     }

254     std::size_t TestFixture::runTests(const options& args)
255     {
256         std::string classname(args.which_test());
257         std::string testname("");
258         if (classname.find("::") != std::string::npos) {
259             testname = classname.substr(classname.find("::") + 2);
260             classname.erase(classname.find("::"));
261         }
262     
263         countTests = 0;
264         errmsg.str("");
265     
266         const std::list<TestFixture *> &tests = TestRegistry::theInstance().tests();
267     
268         for (std::list<TestFixture *>::const_iterator it = tests.begin(); it != tests.end(); ++it) {
269             if (classname.empty() || (*it)->classname == classname) {
270                 (*it)->processOptions(args);
271                 (*it)->run(testname);
272             }
273         }
274     
275         const std::string &w(warnings.str());
276         if (!w.empty()) {
277             std::cout << "\n\n" << w;
278         }
279         std::cout << "\n\nTesting Complete\nNumber of tests: " << countTests << std::endl;
280         std::cout << "Number of todos: " << todos_counter;
281         if (succeeded_todos_counter > 0)
282             std::cout << " (" << succeeded_todos_counter << " succeeded)";
283         std::cout << std::endl;
284         // calling flush here, to do all output before the error messages (in case the output is buffered)
285         std::cout.flush();
286     
287         std::cerr << "Tests failed: " << fails_counter << std::endl << std::endl;
288         std::cerr << errmsg.str();
289     
290         if (!missingLibs.empty()) {
291             std::cerr << "Missing libraries: ";
292             for (std::set<std::string>::const_iterator i = missingLibs.begin(); i != missingLibs.end(); ++i)
293                 std::cerr << *i << "  ";
294             std::cerr << std::endl << std::endl;
295         }
296         std::cerr.flush();
297         return fails_counter;
298     }
