22     
23     #include <list>
24     #include <string>
25     #include <istream>
26     #include <map>
27     #include "config.h"
28     
29     /// @addtogroup Core
30     /// @{
31     
32     /** @brief class for handling suppressions */
33     class CPPCHECKLIB Suppressions {
34     private:
35         class CPPCHECKLIB FileMatcher {
36             friend class Suppressions;
37         private:
38             /** @brief List of filenames suppressed, bool flag indicates whether suppression matched. */
39             std::map<std::string, std::map<unsigned int, bool> > _files;
40             /** @brief List of globs suppressed, bool flag indicates whether suppression matched. */
41             std::map<std::string, std::map<unsigned int, bool> > _globs;
42     
43             /**
44              * @brief Match a name against a glob pattern.
45              * @param pattern The glob pattern to match.
46              * @param name The filename to match against the glob pattern.
47              * @return match success
48              */
49             static bool match(const std::string &pattern, const std::string &name);
50     
51         public:
52             /**
53              * @brief Add a file or glob (and line number).
54              * @param name File name or glob pattern
55              * @param line Line number
56              * @return error message. empty upon success
57              */
58             std::string addFile(const std::string &name, unsigned int line);
59     
60             /**
61              * @brief Returns true if the file name matches a previously added file or glob pattern.
62              * @param file File name to check
63              * @param line Line number
64              * @return true if this filename/line matches
65              */
66             bool isSuppressed(const std::string &file, unsigned int line);
67     
68             /**
69              * @brief Returns true if the file name matches a previously added file (only, not glob pattern).
70              * @param file File name to check
71              * @param line Line number
72              * @return true if this filename/line matches
73              */
74             bool isSuppressedLocal(const std::string &file, unsigned int line);
75         };
76     
77         /** @brief List of error which the user doesn't want to see. */
78         std::map<std::string, FileMatcher> _suppressions;
79     public:
80         /**
81          * @brief Don't show errors listed in the file.
82          * @param istr Open file stream where errors can be read.
83          * @return error message. empty upon success
84          */
85         std::string parseFile(std::istream &istr);
86     
87         /**
88          * @brief Don't show the given error.
89          * @param line Description of error to suppress (in id:file:line format).
90          * @return error message. empty upon success
91          */
92         std::string addSuppressionLine(const std::string &line);
93     
94         /**
95          * @brief Don't show this error. File and/or line are optional. In which case
96          * the errorId alone is used for filtering.
97          * @param errorId the id for the error, e.g. "arrayIndexOutOfBounds"
98          * @param file File name with the path, e.g. "src/main.cpp"
99          * @param line number, e.g. "123"
100          * @return error message. empty upon success
101          */
102         std::string addSuppression(const std::string &errorId, const std::string &file = emptyString, unsigned int line = 0);
103     
104         /**
105          * @brief Returns true if this message should not be shown to the user.
106          * @param errorId the id for the error, e.g. "arrayIndexOutOfBounds"
107          * @param file File name with the path, e.g. "src/main.cpp"
108          * @param line number, e.g. "123"
109          * @return true if this error is suppressed.
110          */
111         bool isSuppressed(const std::string &errorId, const std::string &file, unsigned int line);
112     
113         /**
114          * @brief Returns true if this message should not be shown to the user (explicit files only, not glob patterns).
115          * @param errorId the id for the error, e.g. "arrayIndexOutOfBounds"
116          * @param file File name with the path, e.g. "src/main.cpp"
117          * @param line number, e.g. "123"
118          * @return true if this error is suppressed.
119          */
120         bool isSuppressedLocal(const std::string &errorId, const std::string &file, unsigned int line);
121     
122         struct SuppressionEntry {
123             SuppressionEntry(const std::string &aid, const std::string &afile, unsigned int aline)
124                 : id(aid), file(afile), line(aline) {
125             }
126     
127             std::string id;
128             std::string file;
129             unsigned int line;
130         };
131     
132         /**
133          * @brief Returns list of unmatched local (per-file) suppressions.
134          * @return list of unmatched suppressions
135          */
136         std::list<SuppressionEntry> getUnmatchedLocalSuppressions(const std::string &file, const bool unusedFunctionChecking) const;
137     
138         /**
139          * @brief Returns list of unmatched global (glob pattern) suppressions.
140          * @return list of unmatched suppressions
141          */
142         std::list<SuppressionEntry> getUnmatchedGlobalSuppressions(const bool unusedFunctionChecking) const;
143     };
144     
145     /// @}
146     //---------------------------------------------------------------------------
147     #endif // suppressionsH
