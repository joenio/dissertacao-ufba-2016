45     /** @brief enum class for severity. Used when reporting errors. */
46     class CPPCHECKLIB Severity {
47     public:
48         /**
49          * Message severities.
50          */
51         enum SeverityType {
52             /**
53              * No severity (default value).
54              */
55             none,
56             /**
57              * Programming error.
58              * This indicates severe error like memory leak etc.
59              * The error is certain.
60              */
61             error,
62             /**
63              * Warning.
64              * Used for dangerous coding style that can cause severe runtime errors.
65              * For example: forgetting to initialize a member variable in a constructor.
66              */
67             warning,
68             /**
69              * Style warning.
70              * Used for general code cleanup recommendations. Fixing these
71              * will not fix any bugs but will make the code easier to maintain.
72              * For example: redundant code, unreachable code, etc.
73              */
74             style,
75             /**
76              * Performance warning.
77              * Not an error as is but suboptimal code and fixing it probably leads
78              * to faster performance of the compiled code.
79              */
80             performance,
81             /**
82              * Portability warning.
83              * This warning indicates the code is not properly portable for
84              * different platforms and bitnesses (32/64 bit). If the code is meant
85              * to compile in different platforms and bitnesses these warnings
86              * should be fixed.
87              */
88             portability,
89             /**
90              * Checking information.
91              * Information message about the checking (process) itself. These
92              * messages inform about header files not found etc issues that are
93              * not errors in the code but something user needs to know.
94              */
95             information,
96             /**
97              * Debug message.
98              * Debug-mode message useful for the developers.
99              */
100             debug
101         };
102     
103         static std::string toString(SeverityType severity) {
104             switch (severity) {
105             case none:
106                 return "";
107             case error:
108                 return "error";
109             case warning:
110                 return "warning";
111             case style:
112                 return "style";
113             case performance:
114                 return "performance";
115             case portability:
116                 return "portability";
117             case information:
118                 return "information";
119             case debug:
120                 return "debug";
121             };
122             throw InternalError(NULL, "Unknown severity");
123         }
124         static SeverityType fromString(const std::string &severity) {
125             if (severity.empty())
126                 return none;
127             if (severity == "none")
128                 return none;
129             if (severity == "error")
130                 return error;
131             if (severity == "warning")
132                 return warning;
133             if (severity == "style")
134                 return style;
135             if (severity == "performance")
136                 return performance;
137             if (severity == "portability")
138                 return portability;
139             if (severity == "information")
140                 return information;
141             if (severity == "debug")
142                 return debug;
143             return none;
144         }
145     };

147     /**
148      * @brief This is an interface, which the class responsible of error logging
149      * should implement.
150      */
151     class CPPCHECKLIB ErrorLogger {
152     public:
153     
154         /**
155          * Wrapper for error messages, provided by reportErr()
156          */
157         class CPPCHECKLIB ErrorMessage {
158         public:
159             /**
160              * File name and line number.
161              * Internally paths are stored with / separator. When getting the filename
162              * it is by default converted to native separators.
163              */
164             class CPPCHECKLIB FileLocation {
165             public:
166                 FileLocation()
167                     : line(0) {
168                 }
169     
170                 FileLocation(const std::string &file, unsigned int aline)
171                     : line(aline), _file(file) {
172                 }
173     
174                 FileLocation(const Token* tok, const TokenList* list);
175     
176                 /**
177                  * Return the filename.
178                  * @param convert If true convert path to native separators.
179                  * @return filename.
180                  */
181                 std::string getfile(bool convert = true) const;
182     
183                 /**
184                  * Set the filename.
185                  * @param file Filename to set.
186                  */
187                 void setfile(const std::string &file);
188     
189                 /**
190                  * @return the location as a string. Format: [file:line]
191                  */
192                 std::string stringify() const;
193     
194                 unsigned int line;
195             private:
196                 std::string _file;
197     
198             };
199     
200             ErrorMessage(const std::list<FileLocation> &callStack, Severity::SeverityType severity, const std::string &msg, const std::string &id, bool inconclusive);
201             ErrorMessage(const std::list<const Token*>& callstack, const TokenList* list, Severity::SeverityType severity, const std::string& id, const std::string& msg, bool inconclusive);
202             ErrorMessage();
203     
204             /**
205              * Format the error message in XML format
206              * @param verbose use verbose message
207              * @param ver XML version
208              */
209             std::string toXML(bool verbose, int ver) const;
210     
211             static std::string getXMLHeader(int xml_version);
212             static std::string getXMLFooter(int xml_version);
213     
214             /**
215              * Format the error message into a string.
216              * @param verbose use verbose message
217              * @param outputFormat Empty string to use default output format
218              * or template to be used. E.g. "{file}:{line},{severity},{id},{message}"
219             * @return formatted string
220              */
221             std::string toString(bool verbose, const std::string &outputFormat = emptyString) const;
222     
223             std::string serialize() const;
224             bool deserialize(const std::string &data);
225     
226             std::list<FileLocation> _callStack;
227             std::string _id;
228     
229             /** source file (not header) */
230             std::string file0;
231     
232             Severity::SeverityType _severity;
233             unsigned int _cwe;
234             bool _inconclusive;
235     
236             /** set short and verbose messages */
237             void setmsg(const std::string &msg);
238     
239             /** Short message (single line short message) */
240             const std::string &shortMessage() const {
241                 return _shortMessage;
242             }
243     
244             /** Verbose message (may be the same as the short message) */
245             const std::string &verboseMessage() const {
246                 return _verboseMessage;
247             }
248     
249         private:
250             /**
251              * Replace all occurrences of searchFor with replaceWith in the
252              * given source.
253              * @param source The string to modify
254              * @param searchFor What should be searched for
255              * @param replaceWith What will replace the found item
256              */
257             static void findAndReplace(std::string &source, const std::string &searchFor, const std::string &replaceWith);
258     
259             static std::string fixInvalidChars(const std::string& raw);
260     
261             /** Short message */
262             std::string _shortMessage;
263     
264             /** Verbose message */
265             std::string _verboseMessage;
266         };
267     
268         ErrorLogger() { }
269         virtual ~ErrorLogger() { }
270     
271         /**
272          * Information about progress is directed here.
273          * Override this to receive the progress messages.
274          *
275          * @param outmsg Message to show e.g. "Checking main.cpp..."
276          */
277         virtual void reportOut(const std::string &outmsg) = 0;
278     
279         /**
280          * Information about found errors and warnings is directed
281          * here. Override this to receive the errormessages.
282          *
283          * @param msg Location and other information about the found error.
284          */
285         virtual void reportErr(const ErrorLogger::ErrorMessage &msg) = 0;
286     
287         /**
288          * Report progress to client
289          * @param filename main file that is checked
290          * @param stage for example preprocess / tokenize / simplify / check
291          * @param value progress value (0-100)
292          */
293         virtual void reportProgress(const std::string &filename, const char stage[], const std::size_t value) {
294             (void)filename;
295             (void)stage;
296             (void)value;
297         }
298     
299         /**
300          * Output information messages.
301          * @param msg Location and other information about the found error.
302          */
303         virtual void reportInfo(const ErrorLogger::ErrorMessage &msg) {
304             reportErr(msg);
305         }

308          * Report list of unmatched suppressions
309          * @param unmatched list of unmatched suppressions (from Settings::Suppressions::getUnmatched(Local|Global)Suppressions)
310          */
311         void reportUnmatchedSuppressions(const std::list<Suppressions::SuppressionEntry> &unmatched);
312     
313         static std::string callStackToString(const std::list<ErrorLogger::ErrorMessage::FileLocation> &callStack);
314     
315         /**
316          * Convert XML-sensitive characters into XML entities
317          * @param str The input string containing XML-sensitive characters
318          * @return The ouput string containing XML entities
319          */
320         static std::string toxml(const std::string &str);
321     };
