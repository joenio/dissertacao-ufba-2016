61     static unsigned char readChar(std::istream &istr, unsigned int bom)
62     {
63         unsigned char ch = (unsigned char)istr.get();
64     
65         // For UTF-16 encoded files the BOM is 0xfeff/0xfffe. If the
66         // character is non-ASCII character then replace it with 0xff
67         if (bom == 0xfeff || bom == 0xfffe) {
68             const unsigned char ch2 = (unsigned char)istr.get();
69             const int ch16 = (bom == 0xfeff) ? (ch<<8 | ch2) : (ch2<<8 | ch);
70             ch = (unsigned char)((ch16 >= 0x80) ? 0xff : ch16);
71         }
72     
73         // Handling of newlines..
74         if (ch == '\r') {
75             ch = '\n';
76             if (bom == 0 && (char)istr.peek() == '\n')
77                 (void)istr.get();
78             else if (bom == 0xfeff || bom == 0xfffe) {
79                 int c1 = istr.get();
80                 int c2 = istr.get();
81                 int ch16 = (bom == 0xfeff) ? (c1<<8 | c2) : (c2<<8 | c1);
82                 if (ch16 != '\n') {
83                     istr.unget();
84                     istr.unget();
85                 }
86             }
87         }
88     
89         return ch;
90     }

93     static std::string join(const std::set<std::string>& list, char separator)
94     {
95         std::string s;
96         for (std::set<std::string>::const_iterator it = list.begin(); it != list.end(); ++it) {
97             if (!s.empty())
98                 s += separator;
99     
100             s += *it;
101         }
102         return s;
103     }

105     // Removes duplicate string portions separated by the specified separator
106     static std::string unify(const std::string &s, char separator)
107     {
108         std::set<std::string> parts;
109     
110         std::string::size_type prevPos = 0;
111         for (std::string::size_type pos = 0; pos < s.length(); ++pos) {
112             if (s[pos] == separator) {
113                 if (pos > prevPos)
114                     parts.insert(s.substr(prevPos, pos - prevPos));
115                 prevPos = pos + 1;
116             }
117         }
118         if (prevPos < s.length())
119             parts.insert(s.substr(prevPos));
120     
121         return join(parts, separator);
122     }

134      * Get cfgmap - a map of macro names and values
135      */
136     static std::map<std::string,std::string> getcfgmap(const std::string &cfg, const Settings *settings, const std::string &filename)
137     {
138         std::map<std::string, std::string> cfgmap;
139     
140         if (!cfg.empty()) {
141             std::string::size_type pos = 0;
142             for (;;) {
143                 std::string::size_type pos2 = cfg.find_first_of(";=", pos);
144                 if (pos2 == std::string::npos) {
145                     cfgmap[cfg.substr(pos)] = "";
146                     break;
147                 }
148                 if (cfg[pos2] == ';') {
149                     cfgmap[cfg.substr(pos, pos2-pos)] = "";
150                 } else {
151                     std::string::size_type pos3 = pos2;
152                     pos2 = cfg.find(';', pos2);
153                     if (pos2 == std::string::npos) {
154                         cfgmap[cfg.substr(pos, pos3-pos)] = cfg.substr(pos3 + 1);
155                         break;
156                     } else {
157                         cfgmap[cfg.substr(pos, pos3-pos)] = cfg.substr(pos3 + 1, pos2 - pos3 - 1);
158                     }
159                 }
160                 pos = pos2 + 1;
161             }
162         }
163     
164         if (cfgmap.find("__cplusplus") == cfgmap.end() && Preprocessor::cplusplus(settings,filename))
165             cfgmap["__cplusplus"] = "1";
166     
167         return cfgmap;
168     }

171     /** Just read the code into a string. Perform simple cleanup of the code */
172     std::string Preprocessor::read(std::istream &istr, const std::string &filename)
173     {
174         // The UTF-16 BOM is 0xfffe or 0xfeff.
175         unsigned int bom = 0;
176         if (istr.peek() >= 0xfe) {
177             bom = ((unsigned char)istr.get() << 8);
178             if (istr.peek() >= 0xfe)
179                 bom |= (unsigned char)istr.get();
180             else
181                 bom = 0; // allowed boms are 0/0xfffe/0xfeff
182         }
183     
184         if (_settings.terminated())
185             return "";
186     
187         if (_settings.checkConfiguration)
188             return readpreprocessor(istr,bom);
189     
190         // ------------------------------------------------------------------------------------------
191         //
192         // handling <backslash><newline>
193         // when this is encountered the <backslash><newline> will be "skipped".
194         // on the next <newline>, extra newlines will be added
195         std::ostringstream code;
196         unsigned int newlines = 0;
197         for (unsigned char ch = readChar(istr,bom); istr.good(); ch = readChar(istr,bom)) {
198             // Replace assorted special chars with spaces..
199             if (((ch & 0x80) == 0) && (ch != '\n') && (std::isspace(ch) || std::iscntrl(ch)))
200                 ch = ' ';
201     
202             // <backslash><newline>..
203             // for gcc-compatibility the trailing spaces should be ignored
204             // for vs-compatibility the trailing spaces should be kept
205             // See tickets #640 and #1869
206             // The solution for now is to have a compiler-dependent behaviour.
207             if (ch == '\\') {
208                 unsigned char chNext;
209     
210                 std::string spaces;
211     
212     #ifdef __GNUC__
213                 // gcc-compatibility: ignore spaces
214                 for (;; spaces += ' ') {
215                     chNext = (unsigned char)istr.peek();
216                     if (chNext != '\n' && chNext != '\r' &&
217                         (std::isspace(chNext) || std::iscntrl(chNext))) {
218                         // Skip whitespace between <backslash> and <newline>
219                         (void)readChar(istr,bom);
220                         continue;
221                     }
222     
223                     break;
224                 }
225     #else
226                 // keep spaces
227                 chNext = (unsigned char)istr.peek();
228     #endif
229                 if (chNext == '\n' || chNext == '\r') {
230                     ++newlines;
231                     (void)readChar(istr,bom);   // Skip the "<backslash><newline>"
232                 } else {
233                     code << "\\" << spaces;
234                 }
235             } else {
236                 code << char(ch);
237     
238                 // if there has been <backslash><newline> sequences, add extra newlines..
239                 if (ch == '\n' && newlines > 0) {
240                     code << std::string(newlines, '\n');
241                     newlines = 0;
242                 }
243             }
244         }
245     
246         // ------------------------------------------------------------------------------------------
247         //
248         // Remove all comments..
249         std::string result = removeComments(code.str(), filename);
250         if (_settings.terminated())
251             return "";
252         code.str("");
253     
254         // ------------------------------------------------------------------------------------------
255         //
256         // Clean up all preprocessor statements
257         result = preprocessCleanupDirectives(result);
258         if (_settings.terminated())
259             return "";
260     
261         // ------------------------------------------------------------------------------------------
262         //
263         // Clean up preprocessor #if statements with Parentheses
264         result = removeParentheses(result);
265         if (_settings.terminated())
266             return "";
267     
268         // Remove '#if 0' blocks
269         if (result.find("#if 0\n") != std::string::npos)
270             result = removeIf0(result);
271         if (_settings.terminated())
272             return "";
273     
274         return result;
275     }

278     /** read preprocessor statements */
279     std::string Preprocessor::readpreprocessor(std::istream &istr, const unsigned int bom)
280     {
281         enum { NEWLINE, SPACE, PREPROCESSOR, BACKSLASH, OTHER } state = NEWLINE;
282         std::ostringstream code;
283         unsigned int newlines = 1;
284         unsigned char chPrev = ' ';
285         for (unsigned char ch = readChar(istr,bom); istr.good(); ch = readChar(istr,bom)) {
286             // Replace assorted special chars with spaces..
287             if (((ch & 0x80) == 0) && (ch != '\n') && (std::isspace(ch) || std::iscntrl(ch)))
288                 ch = ' ';
289     
290             if (ch == ' ' && chPrev == ' ')
291                 continue;
292             if (state == PREPROCESSOR && chPrev == '/' && (ch == '/' || ch == '*'))
293                 state = OTHER;
294             chPrev = ch;
295     
296             if (ch == '\n') {
297                 if (state != BACKSLASH) {
298                     state = NEWLINE;
299                     code << std::string(newlines, '\n');
300                     newlines = 1;
301                 } else {
302                     ++newlines;
303                     state = PREPROCESSOR;
304                 }
305                 continue;
306             }
307     
308             switch (state) {
309             case NEWLINE:
310                 if (ch==' ')
311                     state = SPACE;
312                 else if (ch == '#') {
313                     state = PREPROCESSOR;
314                     code << ch;
315                 } else
316                     state = OTHER;
317                 break;
318             case SPACE:
319                 if (ch == '#') {
320                     state = PREPROCESSOR;
321                     code << ch;
322                 } else if (ch != ' ')
323                     state = OTHER;
324                 break;
325             case PREPROCESSOR:
326                 code << ch;
327                 if (ch == '\\')
328                     state = BACKSLASH;
329                 break;
330             case BACKSLASH:
331                 code << ch;
332                 if (ch != ' ')
333                     state = PREPROCESSOR;
334                 break;
335             case OTHER:
336                 break;
337             };
338         }
339     
340         std::string result = preprocessCleanupDirectives(code.str());
341         result = removeParentheses(result);
342         return removeIf0(result);
343     }

345     std::string Preprocessor::preprocessCleanupDirectives(const std::string &processedFile)
346     {
347         std::ostringstream code;
348         std::istringstream sstr(processedFile);
349     
350         std::string line;
351         while (std::getline(sstr, line)) {
352             // Trim lines..
353             if (!line.empty() && line[0] == ' ')
354                 line.erase(0, line.find_first_not_of(" "));
355             if (!line.empty() && line.back() == ' ')
356                 line.erase(line.find_last_not_of(" ") + 1);
357     
358             // Preprocessor
359             if (!line.empty() && line[0] == '#') {
360                 enum {
361                     ESC_NONE,
362                     ESC_SINGLE,
363                     ESC_DOUBLE
364                 } escapeStatus = ESC_NONE;
365     
366                 char prev = ' '; // hack to make it skip spaces between # and the directive
367                 code << "#";
368                 std::string::const_iterator i = line.begin();
369                 ++i;
370     
371                 // need space.. #if( => #if (
372                 bool needSpace = true;
373                 while (i != line.end()) {
374                     // disable esc-mode
375                     if (escapeStatus != ESC_NONE) {
376                         if (prev != '\\' && escapeStatus == ESC_SINGLE && *i == '\'') {
377                             escapeStatus = ESC_NONE;
378                         }
379                         if (prev != '\\' && escapeStatus == ESC_DOUBLE && *i == '"') {
380                             escapeStatus = ESC_NONE;
381                         }
382                     } else {
383                         // enable esc-mode
384                         if (escapeStatus == ESC_NONE && *i == '"')
385                             escapeStatus = ESC_DOUBLE;
386                         if (escapeStatus == ESC_NONE && *i == '\'')
387                             escapeStatus = ESC_SINGLE;
388                     }
389                     // skip double whitespace between arguments
390                     if (escapeStatus == ESC_NONE && prev == ' ' && *i == ' ') {
391                         ++i;
392                         continue;
393                     }
394                     // Convert #if( to "#if ("
395                     if (escapeStatus == ESC_NONE) {
396                         if (needSpace) {
397                             if (*i == '(' || *i == '!')
398                                 code << " ";
399                             else if (!std::isalpha((unsigned char)*i))
400                                 needSpace = false;
401                         }
402                         if (*i == '#')
403                             needSpace = true;
404                     }
405                     code << *i;
406                     if (escapeStatus != ESC_NONE && prev == '\\' && *i == '\\') {
407                         prev = ' ';
408                     } else {
409                         prev = *i;
410                     }
411                     ++i;
412                 }
413                 if (escapeStatus != ESC_NONE) {
414                     // unmatched quotes.. compiler should probably complain about this..
415                 }
416             } else {
417                 // Do not mess with regular code..
418                 code << line;
419             }
420             code << (sstr.eof()?"":"\n");
421         }
422     
423         return code.str();
424     }

443     static bool isFallThroughComment(std::string comment)
444     {
445         // convert comment to lower case without whitespace
446         for (std::string::iterator i = comment.begin(); i != comment.end();) {
447             if (std::isspace(static_cast<unsigned char>(*i)))
448                 i = comment.erase(i);
449             else
450                 ++i;
451         }
452         std::transform(comment.begin(), comment.end(), comment.begin(), tolowerWrapper);
453     
454         return comment.find("fallthr") != std::string::npos ||
455                comment.find("fallsthr") != std::string::npos ||
456                comment.find("fall-thr") != std::string::npos ||
457                comment.find("dropthr") != std::string::npos ||
458                comment.find("passthr") != std::string::npos ||
459                comment.find("nobreak") != std::string::npos ||
460                comment == "fall";
461     }

463     std::string Preprocessor::removeComments(const std::string &str, const std::string &filename)
464     {
465         // For the error report
466         unsigned int lineno = 1;
467     
468         // handling <backslash><newline>
469         // when this is encountered the <backslash><newline> will be "skipped".
470         // on the next <newline>, extra newlines will be added
471         unsigned int newlines = 0;
472         std::ostringstream code;
473         unsigned char previous = 0;
474         bool inPreprocessorLine = false;
475         std::vector<std::string> suppressionIDs;
476         const bool detectFallThroughComments = _settings.experimental && _settings.isEnabled("style");
477         bool fallThroughComment = false;
478     
479         for (std::string::size_type i = hasbom(str) ? 3U : 0U; i < str.length(); ++i) {
480             unsigned char ch = static_cast<unsigned char>(str[i]);
481             if (ch & 0x80) {
482                 std::ostringstream errmsg;
483                 errmsg << "(character code = 0x" << std::hex << (int(ch) & 0xff) << ")";
484                 std::string info = errmsg.str();
485                 errmsg.str("");
486                 errmsg << "The code contains unhandled characters " << info << ". Checking continues, but do not expect valid results.\n"
487                        << "The code contains characters that are unhandled " << info << ". Neither unicode nor extended ASCII are supported. Checking continues, but do not expect valid results.";
488                 writeError(Path::simplifyPath(filename), lineno, _errorLogger, "unhandledCharacters", errmsg.str());
489             }
490     
491             if (_settings.terminated())
492                 return "";
493     
494             // First skip over any whitespace that may be present
495             if (std::isspace(ch)) {
496                 if (ch == ' ' && previous == ' ') {
497                     // Skip double white space
498                 } else {
499                     code << char(ch);
500                     previous = ch;
501                 }
502     
503                 // if there has been <backslash><newline> sequences, add extra newlines..
504                 if (ch == '\n') {
505                     if (previous != '\\')
506                         inPreprocessorLine = false;
507                     ++lineno;
508                     if (newlines > 0) {
509                         code << std::string(newlines, '\n');
510                         newlines = 0;
511                         previous = '\n';
512                     }
513                 }
514     
515                 continue;
516             }
517     
518             if ((ch == '#') && (str.compare(i, 7, "#error ") == 0 || str.compare(i, 9, "#warning ") == 0)) {
519                 if (str.compare(i, 6, "#error") == 0)
520                     code << "#error";
521     
522                 i = str.find('\n', i);
523                 if (i == std::string::npos)
524                     break;
525     
526                 --i;
527                 continue;
528             }
529     
530             // Remove comments..
531             if (str.compare(i, 2, "//") == 0) {
532                 const std::size_t commentStart = i + 2;
533                 i = str.find('\n', i);
534                 if (i == std::string::npos)
535                     break;
536                 std::string comment(str, commentStart, i - commentStart);
537     
538                 if (_settings.inlineSuppressions) {
539                     std::istringstream iss(comment);
540                     std::string word;
541                     iss >> word;
542                     if (word == "cppcheck-suppress") {
543                         iss >> word;
544                         if (iss)
545                             suppressionIDs.push_back(word);
546                     }
547                 }
548     
549                 if (detectFallThroughComments && isFallThroughComment(comment)) {
550                     fallThroughComment = true;
551                 }
552     
553                 code << "\n";
554                 previous = '\n';
555                 ++lineno;
556             } else if (str.compare(i, 2, "/*") == 0) {
557                 const std::size_t commentStart = i + 2;
558                 unsigned char chPrev = 0;
559                 ++i;
560                 while (i < str.length() && (chPrev != '*' || ch != '/')) {
561                     chPrev = ch;
562                     ++i;
563                     ch = static_cast<unsigned char>(str[i]);
564                     if (ch == '\n') {
565                         ++newlines;
566                         ++lineno;
567                     }
568                 }
569                 std::string comment(str, commentStart, i - commentStart - 1);
570     
571                 if (detectFallThroughComments && isFallThroughComment(comment)) {
572                     fallThroughComment = true;
573                 }
574     
575                 if (_settings.inlineSuppressions) {
576                     std::istringstream iss(comment);
577                     std::string word;
578                     iss >> word;
579                     if (word == "cppcheck-suppress") {
580                         iss >> word;
581                         if (iss)
582                             suppressionIDs.push_back(word);
583                     }
584                 }
585             } else if ((i == 0 || std::isspace((unsigned char)str[i-1])) && str.compare(i, 5, "__asm") == 0) {
586                 while (i < str.size() && (std::isalpha((unsigned char)str[i]) || str[i] == '_'))
587                     code << str[i++];
588                 while (i < str.size() && std::isspace((unsigned char)str[i])) {
589                     if (str[i] == '\n')
590                         lineno++;
591                     code << str[i++];
592                 }
593                 if (str[i] == '{') {
594                     // Ticket 4873: Extract comments from the __asm / __asm__'s content
595                     std::string asmBody;
596                     while (i < str.size() && str[i] != '}') {
597                         if (str[i] == ';') {
598                             std::string::size_type backslashN = str.find('\n', i);
599                             if (backslashN != std::string::npos) // Ticket #4922: Don't go in infinite loop or crash if there is no '\n'
600                                 i = backslashN;
601                         }
602                         if (str[i] == '\n')
603                             lineno++;
604                         asmBody += str[i++];
605                     }
606                     code << removeComments(asmBody, filename);
607                     code << '}';
608                 } else
609                     --i;
610             } else if (ch == '#' && previous == '\n') {
611                 code << ch;
612                 previous = ch;
613                 inPreprocessorLine = true;
614     
615                 // Add any pending inline suppressions that have accumulated.
616                 if (!suppressionIDs.empty()) {
617                     // Add the suppressions.
618                     for (std::size_t j = 0; j < suppressionIDs.size(); ++j) {
619                         const std::string errmsg(_settings.nomsg.addSuppression(suppressionIDs[j], filename, lineno));
620                         if (!errmsg.empty()) {
621                             writeError(filename, lineno, _errorLogger, "cppcheckError", errmsg);
622                         }
623                     }
624                     suppressionIDs.clear();
625                 }
626             } else {
627                 if (!inPreprocessorLine) {
628                     // Not whitespace, not a comment, and not preprocessor.
629                     // Must be code here!
630     
631                     // First check for a "fall through" comment match, but only
632                     // add a suppression if the next token is 'case' or 'default'
633                     if (detectFallThroughComments && fallThroughComment) {
634                         const std::string::size_type j = str.find_first_not_of("abcdefghijklmnopqrstuvwxyz", i);
635                         if (str.compare(i, j-i, "case") == 0 || str.compare(i, j-i, "default") == 0)
636                             suppressionIDs.push_back("switchCaseFallThrough");
637                         fallThroughComment = false;
638                     }
639     
640                     // Add any pending inline suppressions that have accumulated.
641                     if (!suppressionIDs.empty()) {
642                         // Relative filename
643                         std::string relativeFilename(filename);
644                         if (_settings.relativePaths) {
645                             for (std::size_t j = 0U; j < _settings.basePaths.size(); ++j) {
646                                 const std::string bp = _settings.basePaths[j] + "/";
647                                 if (relativeFilename.compare(0,bp.size(),bp)==0) {
648                                     relativeFilename = relativeFilename.substr(bp.size());
649                                 }
650                             }
651                         }
652     
653                         // Add the suppressions.
654                         for (std::size_t j = 0; j < suppressionIDs.size(); ++j) {
655                             const std::string errmsg(_settings.nomsg.addSuppression(suppressionIDs[j], relativeFilename, lineno));
656                             if (!errmsg.empty()) {
657                                 writeError(filename, lineno, _errorLogger, "cppcheckError", errmsg);
658                             }
659                         }
660                         suppressionIDs.clear();
661                     }
662                 }
663     
664                 // C++14 digit separators
665                 if (ch == '\'' && std::isxdigit(previous))
666                     ; // Just skip it.
667     
668                 // String or char constants..
669                 else if (ch == '\"' || ch == '\'') {
670                     code << char(ch);
671                     char chNext;
672                     do {
673                         ++i;
674                         chNext = str[i];
675                         if (chNext == '\\') {
676                             ++i;
677                             const char chSeq = str[i];
678                             if (chSeq == '\n')
679                                 ++newlines;
680                             else {
681                                 code << chNext;
682                                 code << chSeq;
683                                 previous = static_cast<unsigned char>(chSeq);
684                             }
685                         } else {
686                             code << chNext;
687                             previous = static_cast<unsigned char>(chNext);
688                         }
689                     } while (i < str.length() && chNext != ch && chNext != '\n');
690                 }
691     
692                 // Rawstring..
693                 else if (str.compare(i,2,"R\"")==0) {
694                     std::string delim;
695                     for (std::string::size_type i2 = i+2; i2 < str.length(); ++i2) {
696                         if (i2 > 16 + i ||
697                             std::isspace(str[i2]) ||
698                             std::iscntrl(str[i2]) ||
699                             str[i2] == ')' ||
700                             str[i2] == '\\') {
701                             delim = " ";
702                             break;
703                         } else if (str[i2] == '(')
704                             break;
705     
706                         delim += str[i2];
707                     }
708                     const std::string::size_type endpos = str.find(")" + delim + "\"", i);
709                     if (delim != " " && endpos != std::string::npos) {
710                         unsigned int rawstringnewlines = 0;
711                         code << '\"';
712                         for (std::string::size_type p = i + 3 + delim.size(); p < endpos; ++p) {
713                             if (str[p] == '\n') {
714                                 rawstringnewlines++;
715                                 code << "\\n";
716                             } else if (std::iscntrl((unsigned char)str[p]) ||
717                                        std::isspace((unsigned char)str[p])) {
718                                 code << " ";
719                             } else if (str[p] == '\\') {
720                                 code << "\\\\";
721                             } else if (str[p] == '\"') {
722                                 code << "\\" << (char)str[p];
723                             } else {
724                                 code << (char)str[p];
725                             }
726                         }
727                         code << "\"";
728                         if (rawstringnewlines > 0)
729                             code << std::string(rawstringnewlines, '\n');
730                         i = endpos + delim.size() + 1;
731                     } else {
732                         code << "R";
733                         previous = 'R';
734                     }
735                 } else {
736                     code << char(ch);
737                     previous = ch;
738                 }
739             }
740         }
741     
742         return code.str();
743     }

781     std::string Preprocessor::removeParentheses(const std::string &str)
782     {
783         if (str.find("\n#if") == std::string::npos && str.compare(0, 3, "#if") != 0)
784             return str;
785     
786         std::istringstream istr(str);
787         std::ostringstream ret;
788         std::string line;
789         while (std::getline(istr, line)) {
790             if (line.compare(0, 3, "#if") == 0 || line.compare(0, 5, "#elif") == 0) {
791                 std::string::size_type pos;
792                 pos = 0;
793                 while ((pos = line.find(" (", pos)) != std::string::npos)
794                     line.erase(pos, 1);
795                 pos = 0;
796                 while ((pos = line.find("( ", pos)) != std::string::npos)
797                     line.erase(pos + 1, 1);
798                 pos = 0;
799                 while ((pos = line.find(" )", pos)) != std::string::npos)
800                     line.erase(pos, 1);
801                 pos = 0;
802                 while ((pos = line.find(") ", pos)) != std::string::npos)
803                     line.erase(pos + 1, 1);
804     
805                 // Remove inner parentheses "((..))"..
806                 pos = 0;
807                 while ((pos = line.find("((", pos)) != std::string::npos) {
808                     ++pos;
809                     std::string::size_type pos2 = line.find_first_of("()", pos + 1);
810                     if (pos2 != std::string::npos && line[pos2] == ')') {
811                         line.erase(pos2, 1);
812                         line.erase(pos, 1);
813                     }
814                 }
815     
816                 // "#if(A) => #if A", but avoid "#if (defined A) || defined (B)"
817                 if ((line.compare(0, 4, "#if(") == 0 || line.compare(0, 6, "#elif(") == 0) &&
818                     line[line.length() - 1] == ')') {
819                     int ind = 0;
820                     for (std::string::size_type i = 0; i < line.length(); ++i) {
821                         if (line[i] == '(')
822                             ++ind;
823                         else if (line[i] == ')') {
824                             --ind;
825                             if (ind == 0) {
826                                 if (i == line.length() - 1) {
827                                     line[line.find('(')] = ' ';
828                                     line.erase(line.length() - 1);
829                                 }
830                                 break;
831                             }
832                         }
833                     }
834                 }
835     
836                 if (line.compare(0, 4, "#if(") == 0)
837                     line.insert(3, " ");
838                 else if (line.compare(0, 6, "#elif(") == 0)
839                     line.insert(5, " ");
840             }
841             ret << line << "\n";
842         }
843     
844         return ret.str();
845     }

901             if (pos2 > str.length() - 1)
902                 break;

904                 str.erase(pos2, 1);
905                 str.erase(pos + 3, 9);
906                 str.insert(pos + 3, "def ");

912         }
913     
914         pos = 0;
915         while ((pos = str.find("#if !defined(", pos)) != std::string::npos) {
916             std::string::size_type pos2 = str.find(')', pos + 9);
917             if (pos2 > str.length() - 1)

927                 return;

932             std::string::size_type pos2 = str.find(')', pos + 9);
933             if (pos2 > str.length() - 1)
934                 break;
935             if (str[pos2+1] == '\n') {
936                 str.erase(pos2, 1);

1087     // Get the DEF in this line: "#ifdef DEF"
1088     std::string Preprocessor::getdef(std::string line, bool def)
1089     {
1090         if (line.empty() || line[0] != '#')
1091             return "";
1092     
1093         // If def is true, the line must start with "#ifdef"
1094         if (def && line.compare(0, 7, "#ifdef ") != 0 && line.compare(0, 4, "#if ") != 0
1095             && (line.compare(0, 6, "#elif ") != 0 || line.compare(0, 7, "#elif !") == 0)) {
1096             return "";
1097         }
1098     
1099         // If def is false, the line must start with "#ifndef"
1100         if (!def && line.compare(0, 8, "#ifndef ") != 0 && line.compare(0, 7, "#elif !") != 0) {
1101             return "";
1102         }
1103     
1104         // Remove the "#ifdef" or "#ifndef"
1105         if (line.compare(0, 12, "#if defined ") == 0)
1106             line.erase(0, 11);
1107         else if (line.compare(0, 15, "#elif !defined(") == 0) {
1108             line.erase(0, 15);
1109             std::string::size_type pos = line.find(')');
1110             // if pos == ::npos then another part of the code will complain
1111             // about the mismatch
1112             if (pos != std::string::npos)
1113                 line.erase(pos, 1);
1114         } else
1115             line.erase(0, line.find(' '));
1116     
1117         // Remove all spaces.
1118         std::string::size_type pos = 0;
1119         while ((pos = line.find(' ', pos)) != std::string::npos) {
1120             const unsigned char chprev(static_cast<unsigned char>((pos > 0) ? line[pos-1] : 0));
1121             const unsigned char chnext(static_cast<unsigned char>((pos + 1 < line.length()) ? line[pos+1] : 0));
1122             if ((std::isalnum(chprev) || chprev == '_') && (std::isalnum(chnext) || chnext == '_'))
1123                 ++pos;
1124             else
1125                 line.erase(pos, 1);
1126         }
1127     
1128         // The remaining string is our result.
1129         return line;
1130     }

1133     static Token *simplifyVarMapExpandValue(Token *tok, const std::map<std::string, std::string> &variables, std::set<std::string> seenVariables, const Settings& settings)
1134     {
1135         // TODO: handle function-macros too.
1136     
1137         // Prevent infinite recursion..
1138         if (seenVariables.find(tok->str()) != seenVariables.end())
1139             return tok;
1140         seenVariables.insert(tok->str());
1141     
1142         const std::map<std::string, std::string>::const_iterator it = variables.find(tok->str());
1143         if (it != variables.end()) {
1144             TokenList tokenList(&settings);
1145             std::istringstream istr(it->second);
1146             if (tokenList.createTokens(istr)) {
1147                 // expand token list
1148                 for (Token *tok2 = tokenList.front(); tok2; tok2 = tok2->next()) {
1149                     if (tok2->isName()) {
1150                         tok2 = simplifyVarMapExpandValue(tok2, variables, seenVariables, settings);
1151                     }
1152                 }
1153     
1154                 // insert token list into "parent" token list
1155                 for (const Token *tok2 = tokenList.front(); tok2; tok2 = tok2->next()) {
1156                     if (tok2->previous()) {
1157                         tok->insertToken(tok2->str());
1158                         tok = tok->next();
1159                     } else
1160                         tok->str(tok2->str());
1161                 }
1162             }
1163         }
1164     
1165         return tok;
1166     }

1173     static void simplifyVarMap(std::map<std::string, std::string> &variables, const Settings& settings)

1177             std::istringstream istr(i->second);

1194     std::list<std::string> Preprocessor::getcfgs(const std::string &filedata, const std::string &filename, const std::map<std::string, std::string> &defs)
1195     {
1196         std::list<std::string> ret;
1197         ret.push_back("");
1198     
1199         std::list<std::string> deflist, ndeflist;
1200     
1201         // constants defined through "#define" in the code..
1202         std::set<std::string> defines;
1203         std::map<std::string, std::string> alldefinesmap(defs);
1204         std::stack<std::pair<std::string,bool> > includeStack;
1205         includeStack.push(std::pair<std::string,bool>(filename,false));
1206     
1207         // How deep into included files are we currently parsing?
1208         // 0=>Source file, 1=>Included by source file, 2=>included by header that was included by source file, etc
1209         int filelevel = 0;
1210     
1211         bool includeguard = false;
1212         unsigned int linenr = 0;
1213         std::istringstream istr(filedata);
1214         std::string line;
1215         const bool printDebug = _settings.debugwarnings;
1216         while (std::getline(istr, line)) {
1217             ++linenr;
1218     
1219             if (_settings.terminated())
1220                 return ret;
1221     
1222             if (_errorLogger)
1223                 _errorLogger->reportProgress(filename, "Preprocessing (get configurations 1)", 0);
1224     
1225             if (line.empty())
1226                 continue;
1227     
1228             if (line.compare(0, 6, "#file ") == 0) {
1229                 includeguard = true;
1230                 const std::string::size_type start=line.find('\"');
1231                 const std::string::size_type end=line.find('\"',start+1);
1232                 const std::string includeFile=line.substr(start+1,end-start-1);
1233                 ++filelevel;
1234                 bool fileExcluded = _settings.configurationExcluded(includeFile);
1235                 includeStack.push(std::pair<std::string,bool>(includeFile,fileExcluded));
1236                 continue;
1237             }
1238     
1239             else if (line == "#endfile") {
1240                 includeguard = false;
1241                 includeStack.pop();
1242                 if (filelevel > 0)
1243                     --filelevel;
1244                 continue;
1245             }
1246     
1247             if (line.compare(0, 8, "#define ") == 0) {
1248                 bool valid = false;
1249                 for (std::string::size_type pos = 8; pos < line.size(); ++pos) {
1250                     const char ch = line[pos];
1251                     if (ch=='_' || (ch>='a' && ch<='z') || (ch>='A' && ch<='Z') || (pos>8 && ch>='0' && ch<='9')) {
1252                         valid = true;
1253                         continue;
1254                     }
1255                     if (ch==' ' || ch=='(') {
1256                         if (valid)
1257                             break;
1258                     }
1259                     valid = false;
1260                     break;
1261                 }
1262                 if (!valid)
1263                     line.clear();
1264                 else {
1265                     std::string definestr = line.substr(8);
1266                     const std::string::size_type spacepos = definestr.find(' ');
1267                     if (spacepos != std::string::npos)
1268                         definestr[spacepos] = '=';
1269                     defines.insert(definestr);
1270     
1271                     const std::string::size_type separatorpos = definestr.find_first_of("=(");
1272                     if (separatorpos != std::string::npos && definestr[separatorpos] == '=') {
1273                         const std::string varname(definestr.substr(0, separatorpos));
1274                         const std::string value(definestr.substr(separatorpos + 1));
1275                         alldefinesmap[varname] = value;
1276                     }
1277                 }
1278             }
1279     
1280             if (!line.empty() && line.compare(0, 3, "#if") != 0)
1281                 includeguard = false;
1282     
1283             if (line.empty() || line[0] != '#')
1284                 continue;
1285     
1286             if (includeguard)
1287                 continue;
1288     
1289             if (line.compare(0, 5, "#line") == 0)
1290                 continue;
1291     
1292             bool from_negation = false;
1293     
1294             std::string def = getdef(line, true);
1295             if (def.empty()) {
1296                 def = getdef(line, false);
1297                 // sub conditionals of ndef blocks need to be
1298                 // constructed _without_ the negated define
1299                 if (!def.empty())
1300                     from_negation = true;
1301             }
1302             if (!def.empty()) {
1303                 int par = 0;
1304                 for (std::string::size_type pos = 0; pos < def.length(); ++pos) {
1305                     if (def[pos] == '(')
1306                         ++par;
1307                     else if (def[pos] == ')') {
1308                         --par;
1309                         if (par < 0)
1310                             break;
1311                     }
1312                 }
1313                 if (par != 0) {
1314                     std::ostringstream lineStream;
1315                     lineStream << __LINE__;
1316                     const std::string errorId = "preprocessor" + lineStream.str();
1317                     const std::string errorText = "mismatching number of '(' and ')' in this line: " + def;
1318                     writeError(filename, linenr, _errorLogger, errorId, errorText);
1319                     ret.clear();
1320                     return ret;
1321                 }
1322     
1323                 // Replace defined constants
1324                 simplifyCondition(alldefinesmap, def, false);
1325     
1326                 if (! deflist.empty() && line.compare(0, 6, "#elif ") == 0)
1327                     deflist.pop_back();
1328     
1329                 // translate A==1 condition to A=1 configuration
1330                 if (def.find("==") != std::string::npos) {
1331                     // Check if condition match pattern "%name% == %num%"
1332                     // %name%
1333                     std::string::size_type pos = 0;
1334                     if (std::isalpha((unsigned char)def[pos]) || def[pos] == '_') {
1335                         ++pos;
1336                         while (std::isalnum((unsigned char)def[pos]) || def[pos] == '_')
1337                             ++pos;
1338                     }
1339     
1340                     // ==
1341                     if (def.compare(pos,2,"==")==0)
1342                         pos += 2;
1343     
1344                     // %num%
1345                     if (pos<def.size() && std::isdigit(def[pos])) {
1346                         if (def.compare(pos,2,"0x")==0) {
1347                             pos += 2;
1348                             if (pos >= def.size())
1349                                 pos = 0;
1350                             while (pos < def.size() && std::isxdigit((unsigned char)def[pos]))
1351                                 ++pos;
1352                         } else {
1353                             while (pos < def.size() && std::isdigit((unsigned char)def[pos]))
1354                                 ++pos;
1355                         }
1356     
1357                         // Does the condition match the pattern "%name% == %num%"?
1358                         if (pos == def.size()) {
1359                             def.erase(def.find("=="),1);
1360                         }
1361                     }
1362                 }
1363     
1364                 deflist.push_back(def);
1365                 def = "";
1366     
1367                 for (std::list<std::string>::const_iterator it = deflist.begin(); it != deflist.end(); ++it) {
1368                     if (*it == "0")
1369                         break;
1370                     if (*it == "1" || *it == "!")
1371                         continue;
1372     
1373                     // don't add "T;T":
1374                     // treat two and more similar nested conditions as one
1375                     if (def != *it) {
1376                         if (! def.empty())
1377                             def += ";";
1378                         def += *it;
1379                     }
1380     
1381                     /* TODO: Fix TestPreprocessor::test7e (#2552)
1382                     else
1383                     {
1384                         std::ostringstream lineStream;
1385                         lineStream << __LINE__;
1386     
1387                         ErrorLogger::ErrorMessage errmsg;
1388                         ErrorLogger::ErrorMessage::FileLocation loc;
1389                         loc.setfile(filename);
1390                         loc.line = linenr;
1391                         errmsg._callStack.push_back(loc);
1392                         errmsg._severity = Severity::error;
1393                         errmsg.setmsg(*it+" is already guaranteed to be defined");
1394                         errmsg._id  = "preprocessor" + lineStream.str();
1395                         _errorLogger->reportErr(errmsg);
1396                     }
1397                     */
1398                 }
1399                 if (from_negation) {
1400                     ndeflist.push_back(deflist.back());
1401                     deflist.back() = "!";
1402                 }
1403     
1404                 if (std::find(ret.begin(), ret.end(), def) == ret.end()) {
1405                     if (!includeStack.top().second) {
1406                         ret.push_back(def);
1407                     } else {
1408                         if (_errorLogger && printDebug) {
1409                             std::list<ErrorLogger::ErrorMessage::FileLocation> locationList;
1410                             const ErrorLogger::ErrorMessage errmsg(locationList, Severity::debug,
1411                                                                    "Configuration not considered: " + def +" for file:"+includeStack.top().first, "debug", false);
1412                             _errorLogger->reportErr(errmsg);
1413                         }
1414                     }
1415                 }
1416             }
1417     
1418             else if (line.compare(0, 5, "#else") == 0 && ! deflist.empty()) {
1419                 if (deflist.back() == "!" && !ndeflist.empty()) {
1420                     deflist.back() = ndeflist.back();
1421                     ndeflist.pop_back();
1422                 } else {
1423                     std::string tempDef((deflist.back() == "1") ? "0" : "1");
1424                     deflist.back() = tempDef;
1425                 }
1426             }
1427     
1428             else if (line.compare(0, 6, "#endif") == 0 && ! deflist.empty()) {
1429                 if (deflist.back() == "!" && !ndeflist.empty())
1430                     ndeflist.pop_back();
1431                 deflist.pop_back();
1432             }
1433         }
1434     
1435         // Remove defined constants from ifdef configurations..
1436         std::size_t count = 0;
1437         for (std::list<std::string>::iterator it = ret.begin(); it != ret.end(); ++it) {
1438             if (_errorLogger)
1439                 _errorLogger->reportProgress(filename, "Preprocessing (get configurations 2)", (100 * count++) / ret.size());
1440     
1441             std::string cfg(*it);
1442             for (std::set<std::string>::const_iterator it2 = defines.begin(); it2 != defines.end(); ++it2) {
1443                 std::string::size_type pos = 0;
1444     
1445                 // Get name of define
1446                 std::string defineName(*it2);
1447                 const std::string::size_type end = defineName.find_first_of("=(");
1448                 if (end != std::string::npos)
1449                     defineName.erase(end);
1450     
1451                 // Remove ifdef configurations that match the defineName
1452                 while ((pos = cfg.find(defineName, pos)) != std::string::npos) {
1453                     const std::string::size_type pos1 = pos;
1454                     ++pos;
1455                     if (pos1 > 0 && cfg[pos1-1] != ';')
1456                         continue;
1457                     const std::string::size_type pos2 = pos1 + defineName.length();
1458                     if (pos2 < cfg.length() && cfg[pos2] != ';')
1459                         continue;
1460                     --pos;
1461                     cfg.erase(pos, defineName.length());
1462                 }
1463             }
1464             if (cfg.length() != it->length()) {
1465                 while (cfg.length() > 0 && cfg[0] == ';')
1466                     cfg.erase(0, 1);
1467     
1468                 while (cfg.length() > 0 && cfg.back() == ';')
1469                     cfg.erase(cfg.length() - 1);
1470     
1471                 std::string::size_type pos = 0;
1472                 while ((pos = cfg.find(";;", pos)) != std::string::npos)
1473                     cfg.erase(pos, 1);
1474     
1475                 *it = cfg;
1476             }
1477         }
1478     
1479         // convert configurations: "defined(A) && defined(B)" => "A;B"
1480         for (std::list<std::string>::iterator it = ret.begin(); it != ret.end(); ++it) {
1481             std::string s(*it);
1482     
1483             if (s.find("&&") != std::string::npos) {
1484                 Tokenizer tokenizer(&_settings, _errorLogger);
1485                 if (!tokenizer.tokenizeCondition(s)) {
1486                     std::ostringstream lineStream;
1487                     lineStream << __LINE__;
1488     
1489                     ErrorLogger::ErrorMessage errmsg;
1490                     ErrorLogger::ErrorMessage::FileLocation loc;
1491                     loc.setfile(filename);
1492                     loc.line = 1;
1493                     errmsg._callStack.push_back(loc);
1494                     errmsg._severity = Severity::error;
1495                     errmsg.setmsg("Error parsing this: " + s);
1496                     errmsg._id  = "preprocessor" + lineStream.str();
1497                     _errorLogger->reportErr(errmsg);
1498                 }
1499     
1500     
1501                 const Token *tok = tokenizer.tokens();
1502                 std::set<std::string> varList;
1503                 while (tok) {
1504                     if (Token::Match(tok, "defined ( %name% )")) {
1505                         varList.insert(tok->strAt(2));
1506                         tok = tok->tokAt(4);
1507                         if (tok && tok->str() == "&&") {
1508                             tok = tok->next();
1509                         }
1510                     } else if (Token::Match(tok, "%name% ;")) {
1511                         varList.insert(tok->str());
1512                         tok = tok->tokAt(2);
1513                     } else {
1514                         break;
1515                     }
1516                 }
1517     
1518                 s = join(varList, ';');
1519     
1520                 if (!s.empty())
1521                     *it = s;
1522             }
1523         }
1524     
1525         // Convert configurations into a canonical form: B;C;A or C;A;B => A;B;C
1526         for (std::list<std::string>::iterator it = ret.begin(); it != ret.end(); ++it)
1527             *it = unify(*it, ';');
1528     
1529         // Remove duplicates from the ret list..
1530         ret.sort();
1531         ret.unique();
1532     
1533         // cleanup unhandled configurations..
1534         for (std::list<std::string>::iterator it = ret.begin(); it != ret.end();) {
1535             const std::string s(*it + ";");
1536     
1537             bool unhandled = false;
1538     
1539             for (std::string::size_type pos = 0; pos < s.length(); ++pos) {
1540                 const unsigned char c = static_cast<unsigned char>(s[pos]);
1541     
1542                 // ok with ";"
1543                 if (c == ';')
1544                     continue;
1545     
1546                 // identifier..
1547                 if (std::isalpha(c) || c == '_') {
1548                     while (std::isalnum((unsigned char)s[pos]) || s[pos] == '_')
1549                         ++pos;
1550                     if (s[pos] == '=') {
1551                         ++pos;
1552                         while (std::isdigit((unsigned char)s[pos]))
1553                             ++pos;
1554                         if (s[pos] != ';') {
1555                             unhandled = true;
1556                             break;
1557                         }
1558                     }
1559     
1560                     --pos;
1561                     continue;
1562                 }
1563     
1564                 // not ok..
1565                 else {
1566                     unhandled = true;
1567                     break;
1568                 }
1569             }
1570     
1571             if (unhandled) {
1572                 // unhandled ifdef configuration..
1573                 if (_errorLogger && printDebug) {
1574                     std::list<ErrorLogger::ErrorMessage::FileLocation> locationList;
1575                     const ErrorLogger::ErrorMessage errmsg(locationList, Severity::debug, "unhandled configuration: " + *it, "debug", false);
1576                     _errorLogger->reportErr(errmsg);
1577                 }
1578     
1579                 ret.erase(it++);
1580             } else {
1581                 ++it;
1582             }
1583         }
1584     
1585         return ret;
1586     }

1589     void Preprocessor::simplifyCondition(const std::map<std::string, std::string> &cfg, std::string &condition, bool match)
1590     {
1591         Tokenizer tokenizer(&_settings, _errorLogger);
1592         if (!tokenizer.tokenizeCondition("(" + condition + ")")) {
1593             // If tokenize returns false, then there is syntax error in the
1594             // code which we can't handle. So stop here.
1595             return;
1596         }
1597     
1598         if (Token::Match(tokenizer.tokens(), "( %name% )")) {
1599             std::map<std::string,std::string>::const_iterator var = cfg.find(tokenizer.tokens()->strAt(1));
1600             if (var != cfg.end()) {
1601                 const std::string &value = var->second;
1602                 condition = (value == "0") ? "0" : "1";
1603             } else if (match)
1604                 condition = "0";
1605             return;
1606         }
1607     
1608         if (Token::Match(tokenizer.tokens(), "( ! %name% )")) {
1609             std::map<std::string,std::string>::const_iterator var = cfg.find(tokenizer.tokens()->strAt(2));
1610     
1611             if (var == cfg.end())
1612                 condition = "1";
1613             else if (var->second == "0")
1614                 condition = "1";
1615             else if (match)
1616                 condition = "0";
1617             return;
1618         }
1619     
1620         // replace variable names with values..
1621         for (Token *tok = const_cast<Token *>(tokenizer.tokens()); tok; tok = tok->next()) {
1622             if (!tok->isName())
1623                 continue;
1624     
1625             if (Token::Match(tok, "defined ( %name% )")) {
1626                 if (cfg.find(tok->strAt(2)) != cfg.end())
1627                     tok->str("1");
1628                 else if (match)
1629                     tok->str("0");
1630                 else
1631                     continue;
1632                 tok->deleteNext(3);
1633                 continue;
1634             }
1635     
1636             if (Token::Match(tok, "defined %name%")) {
1637                 if (cfg.find(tok->strAt(1)) != cfg.end())
1638                     tok->str("1");
1639                 else if (match)
1640                     tok->str("0");
1641                 else
1642                     continue;
1643                 tok->deleteNext();
1644                 continue;
1645             }
1646     
1647             const std::map<std::string, std::string>::const_iterator it = cfg.find(tok->str());
1648             if (it != cfg.end()) {
1649                 if (!it->second.empty()) {
1650                     // Tokenize the value
1651                     Tokenizer tokenizer2(&_settings, _errorLogger);
1652                     tokenizer2.tokenizeCondition(it->second);
1653     
1654                     // Copy the value tokens
1655                     std::stack<Token *> link;
1656                     for (const Token *tok2 = tokenizer2.tokens(); tok2; tok2 = tok2->next()) {
1657                         tok->str(tok2->str());
1658     
1659                         if (Token::Match(tok2,"[{([]"))
1660                             link.push(tok);
1661                         else if (!link.empty() && Token::Match(tok2,"[})]]")) {
1662                             Token::createMutualLinks(link.top(), tok);
1663                             link.pop();
1664                         }
1665     
1666                         if (tok2->next()) {
1667                             tok->insertToken("");
1668                             tok = tok->next();
1669                         }
1670                     }
1671                 } else if ((!tok->previous() || Token::Match(tok->previous(), "&&|%oror%|(")) &&
1672                            (!tok->next() || Token::Match(tok->next(), "&&|%oror%|)")))
1673                     tok->str("1");
1674                 else
1675                     tok->deleteThis();
1676             }
1677         }
1678     
1679         // simplify calculations..
1680         tokenizer.concatenateNegativeNumberAndAnyPositive();
1681         bool modified = true;
1682         while (modified) {
1683             modified = false;
1684             modified |= tokenizer.simplifySizeof();
1685             modified |= tokenizer.simplifyCalculations();
1686             modified |= tokenizer.simplifyConstTernaryOp();
1687             modified |= tokenizer.simplifyRedundantParentheses();
1688             for (Token *tok = const_cast<Token *>(tokenizer.tokens()); tok; tok = tok->next()) {
1689                 if (Token::Match(tok, "! %num%")) {
1690                     tok->deleteThis();
1691                     tok->str(tok->str() == "0" ? "1" : "0");
1692                     modified = true;
1693                 }
1694             }
1695         }
1696     
1697         for (Token *tok = const_cast<Token *>(tokenizer.tokens()); tok; tok = tok->next()) {
1698             if (Token::Match(tok, "(|%oror%|&& %num% &&|%oror%|)")) {
1699                 if (tok->next()->str() != "0") {
1700                     tok->next()->str("1");
1701                 }
1702             }
1703         }
1704     
1705         for (Token *tok = const_cast<Token *>(tokenizer.tokens()); tok; tok = tok->next()) {
1706             while (Token::Match(tok, "(|%oror% %any% %oror% 1")) {
1707                 tok->deleteNext(2);
1708                 if (tok->tokAt(-3))
1709                     tok = tok->tokAt(-3);
1710             }
1711         }
1712     
1713         if (Token::simpleMatch(tokenizer.tokens(), "( 1 )") ||
1714             Token::simpleMatch(tokenizer.tokens(), "( 1 ||"))
1715             condition = "1";
1716         else if (Token::simpleMatch(tokenizer.tokens(), "( 0 )"))
1717             condition = "0";
1718     }

1750     std::string Preprocessor::getcode(const std::string &filedata, const std::string &cfg, const std::string &filename)
1751     {
1752         // For the error report
1753         unsigned int lineno = 0;
1754     
1755         std::ostringstream ret;
1756     
1757         bool match = true;
1758         std::list<bool> matching_ifdef;
1759         std::list<bool> matched_ifdef;
1760     
1761         // Create a map for the cfg for faster access to defines
1762         std::map<std::string, std::string> cfgmap(getcfgmap(cfg, &_settings, filename));
1763     
1764         std::stack<std::string> filenames;
1765         filenames.push(filename);
1766         std::stack<unsigned int> lineNumbers;
1767         std::istringstream istr(filedata);
1768         std::string line;
1769         while (std::getline(istr, line)) {
1770             ++lineno;
1771     
1772             if (_settings.terminated())
1773                 return "";
1774     
1775             if (line.compare(0, 11, "#pragma asm") == 0) {
1776                 ret << "\n";
1777                 bool found_end = false;
1778                 while (getline(istr, line)) {
1779                     if (line.compare(0, 14, "#pragma endasm") == 0) {
1780                         found_end = true;
1781                         break;
1782                     }
1783     
1784                     ret << "\n";
1785                 }
1786                 if (!found_end)
1787                     break;
1788     
1789                 if (line.find('=') != std::string::npos) {
1790                     Tokenizer tokenizer(&_settings, _errorLogger);
1791                     line.erase(0, sizeof("#pragma endasm"));
1792                     std::istringstream tempIstr(line);
1793                     tokenizer.tokenize(tempIstr, "", "", true);
1794                     if (Token::Match(tokenizer.tokens(), "( %name% = %any% )")) {
1795                         ret << "asm(" << tokenizer.tokens()->strAt(1) << ");";
1796                     }
1797                 }
1798     
1799                 ret << "\n";
1800     
1801                 continue;
1802             }
1803     
1804             const std::string def = getdef(line, true);
1805             const std::string ndef = getdef(line, false);
1806     
1807             const bool emptymatch = matching_ifdef.empty() || matched_ifdef.empty();
1808     
1809             if (line.compare(0, 8, "#define ") == 0) {
1810                 match = true;
1811     
1812     
1813                 typedef std::set<std::string>::const_iterator It;
1814                 for (It it = _settings.userUndefs.begin(); it != _settings.userUndefs.end(); ++it) {
1815                     const std::string::size_type symbolPos = line.find_first_not_of(' ', 8);
1816                     if (symbolPos == std::string::npos)
1817                         continue;
1818                     const std::string::size_type undefMatchPos = line.find(*it, symbolPos);
1819                     if (undefMatchPos == std::string::npos)
1820                         continue;
1821                     const std::string::size_type behindUndefPos = undefMatchPos + (*it).size();
1822                     if ((line.size() == behindUndefPos) ||
1823                         (line[behindUndefPos] == ' ') ||
1824                         (line[behindUndefPos] == '(')) {
1825                         match = false;
1826                         break;
1827                     }
1828                 }
1829     
1830                 if (match) {
1831                     for (std::list<bool>::const_iterator it = matching_ifdef.begin(); it != matching_ifdef.end(); ++it) {
1832                         if (!bool(*it)) {
1833                             match = false;
1834                             break;
1835                         }
1836                     }
1837                 }
1838     
1839                 if (match) {
1840                     const std::string::size_type pos = line.find_first_of(" (", 8);
1841                     if (pos == std::string::npos)
1842                         cfgmap[line.substr(8)] = "";
1843                     else if (line[pos] == ' ') {
1844                         std::string value(line.substr(pos + 1));
1845                         std::map<std::string, std::string>::const_iterator cfgpos = cfgmap.find(value);
1846                         if (cfgpos != cfgmap.end())
1847                             value = cfgpos->second;
1848                         cfgmap[line.substr(8, pos - 8)] = value;
1849                     } else
1850                         cfgmap[line.substr(8, pos - 8)] = "";
1851                 }
1852             }
1853     
1854             else if (line.compare(0, 7, "#undef ") == 0) {
1855                 const std::string name(line.substr(7));
1856                 cfgmap.erase(name);
1857             }
1858     
1859             else if (!emptymatch && line.compare(0, 7, "#elif !") == 0) {
1860                 if (matched_ifdef.back()) {
1861                     matching_ifdef.back() = false;
1862                 } else {
1863                     if (!match_cfg_def(cfgmap, ndef)) {
1864                         matching_ifdef.back() = true;
1865                         matched_ifdef.back() = true;
1866                     }
1867                 }
1868             }
1869     
1870             else if (!emptymatch && line.compare(0, 6, "#elif ") == 0) {
1871                 if (matched_ifdef.back()) {
1872                     matching_ifdef.back() = false;
1873                 } else {
1874                     if (match_cfg_def(cfgmap, def)) {
1875                         matching_ifdef.back() = true;
1876                         matched_ifdef.back() = true;
1877                     }
1878                 }
1879             }
1880     
1881             else if (line.compare(0,4,"#if ") == 0) {
1882                 matching_ifdef.push_back(match_cfg_def(cfgmap, line.substr(4)));
1883                 matched_ifdef.push_back(matching_ifdef.back());
1884             }
1885     
1886             else if (! def.empty()) {
1887                 matching_ifdef.push_back(cfgmap.find(def) != cfgmap.end());
1888                 matched_ifdef.push_back(matching_ifdef.back());
1889             }
1890     
1891             else if (! ndef.empty()) {
1892                 matching_ifdef.push_back(cfgmap.find(ndef) == cfgmap.end());
1893                 matched_ifdef.push_back(matching_ifdef.back());
1894             }
1895     
1896             else if (!emptymatch && line == "#else") {
1897                 if (! matched_ifdef.empty())
1898                     matching_ifdef.back() = ! matched_ifdef.back();
1899             }
1900     
1901             else if (line.compare(0, 6, "#endif") == 0) {
1902                 if (! matched_ifdef.empty())
1903                     matched_ifdef.pop_back();
1904                 if (! matching_ifdef.empty())
1905                     matching_ifdef.pop_back();
1906             }
1907     
1908             if (!line.empty() && line[0] == '#') {
1909                 match = true;
1910                 for (std::list<bool>::const_iterator it = matching_ifdef.begin(); it != matching_ifdef.end(); ++it) {
1911                     if (!bool(*it)) {
1912                         match = false;
1913                         break;
1914                     }
1915                 }
1916             }
1917     
1918             // #error => return ""
1919             if (match && line.compare(0, 6, "#error") == 0) {
1920                 if (!_settings.userDefines.empty() && !_settings.force) {
1921                     error(Path::simplifyPath(filenames.top()), lineno, line);
1922                 }
1923                 return "";
1924             }
1925     
1926             if (!match && (line.compare(0, 8, "#define ") == 0 ||
1927                            line.compare(0, 6, "#undef") == 0)) {
1928                 // Remove define that is not part of this configuration
1929                 line = "";
1930             } else if (line.compare(0, 7, "#file \"") == 0 ||
1931                        line.compare(0, 8, "#endfile") == 0 ||
1932                        line.compare(0, 8, "#define ") == 0 ||
1933                        line.compare(0, 6, "#line ") == 0 ||
1934                        line.compare(0, 6, "#undef") == 0) {
1935                 // We must not remove #file tags or line numbers
1936                 // are corrupted. File tags are removed by the tokenizer.
1937     
1938                 // Keep location info updated
1939                 if (line.compare(0, 7, "#file \"") == 0) {
1940                     filenames.push(line.substr(7, line.size() - 8));
1941                     lineNumbers.push(lineno);
1942                     lineno = 0;
1943                 } else if (line.compare(0, 8, "#endfile") == 0) {
1944                     if (filenames.size() > 1U)
1945                         filenames.pop();
1946     
1947                     if (!lineNumbers.empty()) {
1948                         lineno = lineNumbers.top();
1949                         lineNumbers.pop();
1950                     }
1951                 }
1952             } else if (!match || line.compare(0, 1, "#") == 0) {
1953                 // Remove #if, #else, #pragma etc, leaving only
1954                 // #define, #undef, #file and #endfile. and also lines
1955                 // which are not part of this configuration.
1956                 line = "";
1957             }
1958     
1959             ret << line << "\n";
1960         }
1961     
1962         if (!validateCfg(ret.str(), cfg)) {
1963             return "";
1964         }
1965     
1966         return expandMacros(ret.str(), filename, cfg, _errorLogger);
1967     }

1983     Preprocessor::HeaderTypes Preprocessor::getHeaderFileName(std::string &str)
1984     {
1985         std::string::size_type i = str.find_first_of("<\"");
1986         if (i == std::string::npos) {
1987             str = "";
1988             return NoHeader;
1989         }
1990     
1991         char c = str[i];
1992         if (c == '<')
1993             c = '>';
1994     
1995         std::string result;
1996         for (i = i + 1; i < str.length(); ++i) {
1997             if (str[i] == c)
1998                 break;
1999     
2000             result.append(1, str[i]);
2001         }
2002     
2003         // Linux can't open include paths with \ separator, so fix them
2004         std::replace(result.begin(), result.end(), '\\', '/');
2005     
2006         str = result;
2007     
2008         return (c == '\"') ? UserHeader : SystemHeader;
2009     }

2045     std::string Preprocessor::handleIncludes(const std::string &code, const std::string &filePath, const std::list<std::string> &includePaths, std::map<std::string,std::string> &defs, std::set<std::string> &pragmaOnce, std::list<std::string> includes)
2046     {
2047         std::string path;
2048         std::string::size_type sep_pos = filePath.find_last_of("\\/");
2049         if (sep_pos != std::string::npos)
2050             path = filePath.substr(0, 1 + sep_pos);
2051     
2052         // current #if indent level.
2053         std::stack<bool>::size_type indent = 0;
2054     
2055         // how deep does the #if match? this can never be bigger than "indent".
2056         std::stack<bool>::size_type indentmatch = 0;
2057     
2058         // has there been a true #if condition at the current indentmatch level?
2059         // then no more #elif or #else can be true before the #endif is seen.
2060         std::stack<bool> elseIsTrueStack;
2061     
2062         unsigned int linenr = 0;
2063     
2064         const std::set<std::string> &undefs = _settings.userUndefs;
2065     
2066         if (_errorLogger)
2067             _errorLogger->reportProgress(filePath, "Preprocessor (handleIncludes)", 0);
2068     
2069         std::ostringstream ostr;
2070         std::istringstream istr(code);
2071         std::string line;
2072         bool suppressCurrentCodePath = false;
2073         while (std::getline(istr,line)) {
2074             ++linenr;
2075     
2076             if (_settings.terminated())
2077                 return "";
2078     
2079             // has there been a true #if condition at the current indentmatch level?
2080             // then no more #elif or #else can be true before the #endif is seen.
2081             while (elseIsTrueStack.size() != indentmatch + 1) {
2082                 if (elseIsTrueStack.size() < indentmatch + 1) {
2083                     elseIsTrueStack.push(true);
2084                 } else {
2085                     elseIsTrueStack.pop();
2086                 }
2087             }
2088     
2089             if (elseIsTrueStack.empty()) {
2090                 writeError(filePath, linenr, _errorLogger, "syntaxError", "Syntax error in preprocessor code");
2091                 return "";
2092             }
2093     
2094             std::stack<bool>::reference elseIsTrue = elseIsTrueStack.top();
2095     
2096             if (line == "#pragma once") {
2097                 pragmaOnce.insert(filePath);
2098             } else if (line.compare(0,7,"#ifdef ") == 0) {
2099                 if (indent == indentmatch) {
2100                     const std::string tag = getdef(line,true);
2101                     if (defs.find(tag) != defs.end()) {
2102                         elseIsTrue = false;
2103                         indentmatch++;
2104                     } else if (undefs.find(tag) != undefs.end()) {
2105                         elseIsTrue = true;
2106                         indentmatch++;
2107                         suppressCurrentCodePath = true;
2108                     }
2109                 }
2110                 ++indent;
2111     
2112                 if (indent == indentmatch + 1)
2113                     elseIsTrue = true;
2114             } else if (line.compare(0,8,"#ifndef ") == 0) {
2115                 if (indent == indentmatch) {
2116                     const std::string tag = getdef(line,false);
2117                     if (defs.find(tag) == defs.end()) {
2118                         elseIsTrue = false;
2119                         indentmatch++;
2120                     } else if (undefs.find(tag) != undefs.end()) {
2121                         elseIsTrue = false;
2122                         indentmatch++;
2123                         suppressCurrentCodePath = false;
2124                     }
2125                 }
2126                 ++indent;
2127     
2128                 if (indent == indentmatch + 1)
2129                     elseIsTrue = true;
2130     
2131             } else if (line.compare(0,4,"#if ") == 0) {
2132                 if (!suppressCurrentCodePath && indent == indentmatch && match_cfg_def(defs, line.substr(4))) {
2133                     elseIsTrue = false;
2134                     indentmatch++;
2135                 }
2136                 ++indent;
2137     
2138                 if (indent == indentmatch + 1)
2139                     elseIsTrue = true;  // this value doesn't matter when suppressCurrentCodePath is true
2140             } else if (line.compare(0,6,"#elif ") == 0 || line.compare(0,5,"#else") == 0) {
2141                 if (!elseIsTrue) {
2142                     if ((indentmatch > 0) && (indentmatch == indent)) {
2143                         indentmatch = indent - 1;
2144                     }
2145                 } else {
2146                     if ((indentmatch > 0) && (indentmatch == indent)) {
2147                         indentmatch = indent - 1;
2148                     } else if ((indent > 0) && indentmatch == indent - 1) {
2149                         if (line.compare(0,5,"#else")==0 || match_cfg_def(defs,line.substr(6))) {
2150                             indentmatch = indent;
2151                             elseIsTrue = false;
2152                         }
2153                     }
2154                 }
2155             } else if (line.compare(0, 6, "#endif") == 0) {
2156                 if (indent > 0)
2157                     --indent;
2158                 if (indentmatch > indent || indent == 0) {
2159                     indentmatch = indent;
2160                     elseIsTrue = false;
2161                     suppressCurrentCodePath = false;
2162                 }
2163             } else if (indentmatch == indent) {
2164                 if (!suppressCurrentCodePath && line.compare(0, 8, "#define ") == 0) {
2165                     const unsigned int endOfDefine = 8;
2166                     std::string::size_type endOfTag = line.find_first_of("( ", endOfDefine);
2167                     std::string tag;
2168     
2169                     // define a symbol
2170                     if (endOfTag == std::string::npos) {
2171                         tag = line.substr(endOfDefine);
2172                         defs[tag] = "";
2173                     } else {
2174                         tag = line.substr(endOfDefine, endOfTag-endOfDefine);
2175     
2176                         // define a function-macro
2177                         if (line[endOfTag] == '(') {
2178                             defs[tag] = "";
2179                         }
2180                         // define value
2181                         else {
2182                             ++endOfTag;
2183     
2184                             const std::string& value = line.substr(endOfTag, line.size()-endOfTag);
2185     
2186                             if (defs.find(value) != defs.end())
2187                                 defs[tag] = defs[value];
2188                             else
2189                                 defs[tag] = value;
2190                         }
2191                     }
2192     
2193                     if (undefs.find(tag) != undefs.end()) {
2194                         defs.erase(tag);
2195                     }
2196                 }
2197     
2198                 else if (!suppressCurrentCodePath && line.compare(0,7,"#undef ") == 0) {
2199                     defs.erase(line.substr(7));
2200                 }
2201     
2202                 else if (!suppressCurrentCodePath && line.compare(0,9,"#include ")==0) {
2203                     std::string filename(line.substr(9));
2204     
2205                     const HeaderTypes headerType = getHeaderFileName(filename);
2206                     if (headerType == NoHeader) {
2207                         ostr << std::endl;
2208                         continue;
2209                     }
2210     
2211                     // try to open file
2212                     std::string filepath;
2213                     if (headerType == UserHeader)
2214                         filepath = path;
2215                     std::ifstream fin;
2216                     if (!openHeader(filename, includePaths, filepath, fin)) {
2217                         missingInclude(Path::toNativeSeparators(filePath),
2218                                        linenr,
2219                                        filename,
2220                                        headerType
2221                                       );
2222                         ostr << std::endl;
2223                         continue;
2224                     }
2225     
2226                     // Prevent that files are recursively included
2227                     if (std::find(includes.begin(), includes.end(), filename) != includes.end()) {
2228                         ostr << std::endl;
2229                         continue;
2230                     }
2231     
2232                     includes.push_back(filename);
2233     
2234                     // Don't include header if it's already included and contains #pragma once
2235                     if (pragmaOnce.find(filename) != pragmaOnce.end()) {
2236                         ostr << std::endl;
2237                         continue;
2238                     }
2239     
2240                     ostr << "#file \"" << filename << "\"\n"
2241                          << handleIncludes(read(fin, filename), filename, includePaths, defs, pragmaOnce, includes) << std::endl
2242                          << "#endfile\n";
2243                     continue;
2244                 }
2245     
2246                 if (!suppressCurrentCodePath)
2247                     ostr << line;
2248             }
2249     
2250             // A line has been read..
2251             ostr << "\n";
2252         }
2253     
2254         return ostr.str();
2255     }

2258     void Preprocessor::handleIncludes(std::string &code, const std::string &filePath, const std::list<std::string> &includePaths)
2259     {
2260         std::list<std::string> paths;
2261         std::string path = filePath;
2262         const std::string::size_type sep_pos = path.find_last_of("\\/");
2263         if (sep_pos != std::string::npos)
2264             path.erase(1 + sep_pos);
2265         paths.push_back(path);
2266         std::string::size_type pos = 0;
2267         std::string::size_type endfilePos = 0;
2268         if (code.compare(0,7U,"#file \"")==0) {
2269             const std::string::size_type start = code.find("#file \"" + filePath, 7U);
2270             if (start != std::string::npos)
2271                 endfilePos = start;
2272         }
2273         std::set<std::string> handledFiles;
2274         while ((pos = code.find("#include", pos)) != std::string::npos) {
2275             if (_settings.terminated())
2276                 return;
2277     
2278             // Accept only includes that are at the start of a line
2279             if (pos > 0 && code[pos-1] != '\n') {
2280                 pos += 8; // length of "#include"
2281                 continue;
2282             }
2283     
2284             // If endfile is encountered, we have moved to a next file in our stack,
2285             // so remove last path in our list.
2286             while (!paths.empty() && (endfilePos = code.find("\n#endfile", endfilePos)) != std::string::npos && endfilePos < pos) {
2287                 paths.pop_back();
2288                 endfilePos += 9; // size of #endfile
2289             }
2290     
2291             endfilePos = pos;
2292             std::string::size_type end = code.find('\n', pos);
2293             std::string filename = code.substr(pos, end - pos);
2294     
2295             // Remove #include clause
2296             code.erase(pos, end - pos);
2297     
2298             HeaderTypes headerType = getHeaderFileName(filename);
2299             if (headerType == NoHeader)
2300                 continue;
2301     
2302             // filename contains now a file name e.g. "menu.h"
2303             std::string processedFile;
2304             std::string filepath;
2305             if (headerType == UserHeader && !paths.empty())
2306                 filepath = paths.back();
2307             std::ifstream fin;
2308             const bool fileOpened(openHeader(filename, includePaths, filepath, fin));
2309     
2310             if (fileOpened) {
2311                 filename = Path::simplifyPath(filename);
2312                 std::string tempFile = filename;
2313                 std::transform(tempFile.begin(), tempFile.end(), tempFile.begin(), tolowerWrapper);
2314                 if (handledFiles.find(tempFile) != handledFiles.end()) {
2315                     // We have processed this file already once, skip
2316                     // it this time to avoid eternal loop.
2317                     fin.close();
2318                     continue;
2319                 }
2320     
2321                 handledFiles.insert(tempFile);
2322                 processedFile = Preprocessor::read(fin, filename);
2323                 fin.close();
2324             }
2325     
2326             if (!processedFile.empty()) {
2327                 // Remove space characters that are after or before new line character
2328                 processedFile = "#file \"" + Path::fromNativeSeparators(filename) + "\"\n" + processedFile + "\n#endfile";
2329                 code.insert(pos, processedFile);
2330     
2331                 path = filename;
2332                 path.erase(1 + path.find_last_of("\\/"));
2333                 paths.push_back(path);
2334             } else if (!fileOpened) {
2335                 std::string f = filePath;
2336     
2337                 // Determine line number of include
2338                 unsigned int linenr = 1;
2339                 unsigned int level = 0;
2340                 for (std::string::size_type p = 1; p <= pos; ++p) {
2341                     if (level == 0 && code[pos-p] == '\n')
2342                         ++linenr;
2343                     else if (code.compare(pos-p, 9, "#endfile\n") == 0) {
2344                         ++level;
2345                     } else if (code.compare(pos-p, 6, "#file ") == 0) {
2346                         if (level == 0) {
2347                             linenr--;
2348                             const std::string::size_type pos1 = pos - p + 7;
2349                             const std::string::size_type pos2 = code.find_first_of("\"\n", pos1);
2350                             f = code.substr(pos1, (pos2 == std::string::npos) ? pos2 : (pos2 - pos1));
2351                             break;
2352                         }
2353                         --level;
2354                     }
2355                 }
2356     
2357                 missingInclude(Path::toNativeSeparators(f),
2358                                linenr,
2359                                filename,
2360                                headerType);
2361             }
2362         }
2363     }

2417      * If string is all spaces/tabs, return empty string.

2423         if (beg == std::string::npos)

2429     /**
2430      * @brief get parameters from code. For example 'foo(1,2)' => '1','2'
2431      * @param line in: The code
2432      * @param pos  in: Position to the '('. out: Position to the ')'
2433      * @param params out: The extracted parameters
2434      * @param numberOfNewlines out: number of newlines in the macro call
2435      * @param endFound out: was the end parentheses found?
2436      */
2437     static void getparams(const std::string &line,
2438                           std::string::size_type &pos,
2439                           std::vector<std::string> &params,
2440                           unsigned int &numberOfNewlines,
2441                           bool &endFound)
2442     {
2443         params.clear();
2444         numberOfNewlines = 0;
2445         endFound = false;
2446     
2447         if (line[pos] == ' ')
2448             pos++;
2449     
2450         if (line[pos] != '(')
2451             return;
2452     
2453         // parentheses level
2454         int parlevel = 0;
2455     
2456         // current parameter data
2457         std::string par;
2458     
2459         // scan for parameters..
2460         for (; pos < line.length(); ++pos) {
2461             // increase parentheses level
2462             if (line[pos] == '(') {
2463                 ++parlevel;
2464                 if (parlevel == 1)
2465                     continue;
2466             }
2467     
2468             // decrease parentheses level
2469             else if (line[pos] == ')') {
2470                 --parlevel;
2471                 if (parlevel <= 0) {
2472                     endFound = true;
2473                     params.push_back(trim(par));
2474                     break;
2475                 }
2476             }
2477     
2478             // string
2479             else if (line[pos] == '\"' || line[pos] == '\'') {
2480                 const std::string::size_type p = pos;
2481                 skipstring(line, pos);
2482                 if (pos == line.length())
2483                     break;
2484                 par += line.substr(p, pos + 1 - p);
2485                 continue;
2486             }
2487     
2488             // count newlines. the expanded macro must have the same number of newlines
2489             else if (line[pos] == '\n') {
2490                 ++numberOfNewlines;
2491                 continue;
2492             }
2493     
2494             // new parameter
2495             if (parlevel == 1 && line[pos] == ',') {
2496                 params.push_back(trim(par));
2497                 par = "";
2498             }
2499     
2500             // spaces are only added if needed
2501             else if (line[pos] == ' ') {
2502                 // Add space only if it is needed
2503                 if (par.size() && std::isalnum((unsigned char)par.back())) {
2504                     par += ' ';
2505                 }
2506             }
2507     
2508             // add character to current parameter
2509             else if (parlevel >= 1 && line[pos] != Preprocessor::macroChar) {
2510                 par.append(1, line[pos]);
2511             }
2512         }
2513     }

2538                 const std::map<std::string, PreprocessorMacro *> &macros) const {
2539             std::string innerMacroName;
2540     
2541             // Is there an inner macro..
2542             {
2543                 const Token *tok = Token::findsimplematch(tokens(), ")");

2550                     tok = tok->tokAt(2);
2551                     par++;
2552                 }
2553                 if (tok || par != params1.size())
2554                     return params1;
2555             }

2557             std::vector<std::string> params2(params1);
2558     
2559             for (std::size_t ipar = 0; ipar < params1.size(); ++ipar) {
2560                 const std::string s(innerMacroName + "(");
2561                 const std::string param(params1[ipar]);

2563                     std::vector<std::string> innerparams;
2564                     std::string::size_type pos = s.length() - 1;

2571                         if (it != macros.end()) {
2572                             // expand the inner macro
2573                             const PreprocessorMacro *innerMacro = it->second;
2574

2576                             std::map<std::string,PreprocessorMacro *> innermacros = macros;

2794     /**
2795      * Get data from a input string. This is an extended version of std::getline.
2796      * The std::getline only get a single line at a time. It can therefore happen that it
2797      * contains a partial statement. This function ensures that the returned data
2798      * doesn't end in the middle of a statement. The "getlines" name indicate that
2799      * this function will return multiple lines if needed.
2800      * @param istr input stream
2801      * @param line output data
2802      * @return success
2803      */
2804     static bool getlines(std::istream &istr, std::string &line)
2805     {
2806         if (!istr.good())
2807             return false;
2808         line = "";
2809         int parlevel = 0;
2810         bool directive = false;
2811         for (char ch = (char)istr.get(); istr.good(); ch = (char)istr.get()) {
2812             if (ch == '\'' || ch == '\"') {
2813                 line += ch;
2814                 char c = 0;
2815                 while (istr.good() && c != ch) {
2816                     if (c == '\\') {
2817                         c = (char)istr.get();
2818                         if (!istr.good())
2819                             return true;
2820                         line += c;
2821                     }
2822     
2823                     c = (char)istr.get();
2824                     if (!istr.good())
2825                         return true;
2826                     if (c == '\n' && directive)
2827                         return true;
2828                     line += c;
2829                 }
2830                 continue;
2831             }
2832             if (ch == '(')
2833                 ++parlevel;
2834             else if (ch == ')')
2835                 --parlevel;
2836             else if (ch == '\n') {
2837                 if (directive)
2838                     return true;
2839     
2840                 if (istr.peek() == '#') {
2841                     line += ch;
2842                     return true;
2843                 }
2844             } else if (!directive && parlevel <= 0 && ch == ';') {
2845                 line += ";";
2846                 return true;
2847             }
2848     
2849             if (ch == '#' && line.empty())
2850                 directive = true;
2851             line += ch;
2852         }
2853         return true;
2854     }
