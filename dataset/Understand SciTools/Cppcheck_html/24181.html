63     void StatsDialog::setScanDuration(double seconds)
64     {
65         // Factor the duration into units (days/hours/minutes/seconds)
66         int secs = seconds;
67         int days = secs / (24 * 60 * 60);
68         secs -= days * (24 * 60 * 60);
69         int hours = secs / (60 * 60);
70         secs -= hours * (60 * 60);
71         int mins = secs / 60;
72         secs -= mins * 60;
73     
74         // Concatenate the two most significant units (e.g. "1 day and 3 hours")
75         QStringList parts;
76         if (days)
77             parts << ((days == 1) ? tr("1 day") : tr("%1 days").arg(days));
78         if (hours)
79             parts << ((hours == 1) ? tr("1 hour") : tr("%1 hours").arg(hours));
80         if (mins && parts.size() < 2)
81             parts << ((mins == 1) ? tr("1 minute") : tr("%1 minutes").arg(mins));
82         if (secs && parts.size() < 2)
83             parts << ((secs == 1) ? tr("1 second") : tr("%1 seconds").arg(secs));
84     
85         // For durations < 1s, show the fraction of a second (e.g. "0.7 seconds")
86         if (parts.isEmpty())
87             parts << tr("0.%1 seconds").arg(int(10.0 *(seconds - secs)));
88     
89         mUI.mScanDuration->setText(parts.join(tr(" and ")));
90     }
