171         std::string tok(const char code[], bool simplify = true, Settings::PlatformType type = Settings::Native, bool debugwarnings = true) {
172             errout.str("");
173     
174             settings0.inconclusive = true;
175             settings0.debugwarnings = debugwarnings;   // show warnings about unhandled typedef
176             settings0.platform(type);
177             Tokenizer tokenizer(&settings0, this);
178     
179             std::istringstream istr(code);
180             tokenizer.tokenize(istr, "test.cpp");
181     
182             if (simplify)
183                 tokenizer.simplifyTokenList2();
184     
185             return tokenizer.tokens()->stringifyList(0, !simplify);
186         }
