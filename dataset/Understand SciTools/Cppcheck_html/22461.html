29     /// @addtogroup Checks
30     /// @{
31     
32     /**
33      * @brief Check for 64-bit portability issues
34      */
35     
36     class CPPCHECKLIB Check64BitPortability : public Check {
37     public:
38         /** This constructor is used when registering the Check64BitPortability */
39         Check64BitPortability() : Check(myName()) {
40         }
41     
42         /** This constructor is used when running checks. */
43         Check64BitPortability(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
44             : Check(myName(), tokenizer, settings, errorLogger) {
45         }
46     
47         /** @brief Run checks against the normal token list */
48         void runChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
49             Check64BitPortability check64BitPortability(tokenizer, settings, errorLogger);
50             check64BitPortability.pointerassignment();
51         }
52     
53         /** @brief Run checks against the simplified token list */
54         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
55             (void)tokenizer;
56             (void)settings;
57             (void)errorLogger;
58         }
59     
60         /** Check for pointer assignment */
61         void pointerassignment();
62     
63     private:
64     
65         void assignmentAddressToIntegerError(const Token *tok);
66         void assignmentIntegerToAddressError(const Token *tok);
67         void returnIntegerError(const Token *tok);
68         void returnPointerError(const Token *tok);
69     
70         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
71             Check64BitPortability c(0, settings, errorLogger);
72             c.assignmentAddressToIntegerError(0);
73             c.assignmentIntegerToAddressError(0);
74             c.returnIntegerError(0);
75             c.returnPointerError(0);
76         }
77     
78         static std::string myName() {
79             return "64-bit portability";
80         }
81     
82         std::string classInfo() const {
83             return "Check if there is 64-bit portability issues:\n"
84                    "- assign address to/from int/long\n"
85                    "- casting address from/to integer when returning from function\n";
86         }
87     };
88     /// @}
89     //---------------------------------------------------------------------------
90     #endif // check64bitH
