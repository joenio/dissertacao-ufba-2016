19     #ifndef CPPCHECKLIBRARYDATA_H
20     #define CPPCHECKLIBRARYDATA_H
21     
22     #include <QList>
23     #include <QString>
24     #include <QStringList>
25     #include <QIODevice>
26     
27     class CppcheckLibraryData {
28     public:
29         CppcheckLibraryData();
30     
31         struct Container {
32             Container() : access_arrayLike(false), size_templateParameter(-1) {}
33     
34             QString id;
35             QString inherits;
36             QString startPattern;
37             QString endPattern;
38     
39             bool access_arrayLike;
40             int  size_templateParameter;
41     
42             struct {
43                 QString templateParameter;
44                 QString string;
45             } type;
46     
47             struct Function {
48                 QString name;
49                 QString yields;
50                 QString action;
51             };
52             QList<struct Function> accessFunctions;
53             QList<struct Function> otherFunctions;
54             QList<struct Function> sizeFunctions;
55         };
56     
57         struct Define {
58             QString name;
59             QString value;
60         };
61     
62         struct Function {
63             Function() : noreturn(Unknown), gccPure(false), gccConst(false),
64                 leakignore(false), useretval(false) {
65             }
66     
67             QString comments;
68             QString name;
69             enum TrueFalseUnknown { False, True, Unknown } noreturn;
70             bool gccPure;
71             bool gccConst;
72             bool leakignore;
73             bool useretval;
74             struct {
75                 QString scan;
76                 QString secure;
77             } formatstr;
78             struct Arg {
79                 Arg() : nr(0), notbool(false), notnull(false), notuninit(false),
80                     formatstr(false), strz(false) {
81                 }
82     
83                 QString name;
84                 unsigned int nr;
85                 static const unsigned int ANY;
86                 bool notbool;
87                 bool notnull;
88                 bool notuninit;
89                 bool formatstr;
90                 bool strz;
91                 QString valid;
92                 struct MinSize {
93                     QString type;
94                     QString arg;
95                     QString arg2;
96                 };
97                 QList<struct MinSize> minsizes;
98             };
99             QList<struct Arg> args;
100     
101             struct {
102                 QString severity;
103                 QString reason;
104                 QString alternatives;
105                 QString msg;
106     
107                 bool isEmpty() const {
108                     return severity.isEmpty() &&
109                            reason.isEmpty() &&
110                            alternatives.isEmpty() &&
111                            msg.isEmpty();
112                 }
113             } warn;
114         };
115     
116         struct MemoryResource {
117             QString type; // "memory" or "resource"
118             struct Alloc {
119                 Alloc() : init(false) {}
120                 bool init;
121                 QString name;
122             };
123             QList<struct Alloc> alloc;
124             QStringList dealloc;
125             QStringList use;
126         };
127     
128         struct PodType {
129             QString name;
130             QString size;
131             QString sign;
132         };
133     
134         void clear() {
135             defines.clear();
136             functions.clear();
137             memoryresource.clear();
138             podtypes.clear();
139         }
140     
141         bool open(QIODevice &file);
142         QString toString() const;
143     
144         QList<struct Container> containers;
145         QList<struct Define> defines;
146         QList<struct Function> functions;
147         QList<struct MemoryResource> memoryresource;
148         QList<struct PodType> podtypes;
149     };
150     
151     #endif // LIBRARYDATA_H
