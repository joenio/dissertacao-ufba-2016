23     
24     #include <string>
25     #include "config.h"
26     
27     class Token;
28     class TokenList;
29     class SymbolDatabase;
30     class ErrorLogger;
31     class Settings;
32     
33     namespace ValueFlow {
34         class CPPCHECKLIB Value {
35         public:
36             explicit Value(long long val = 0) : intvalue(val), tokvalue(nullptr), varvalue(val), condition(0), varId(0U), conditional(false), inconclusive(false), defaultArg(false), valueKind(ValueKind::Possible) {}
37             Value(const Token *c, long long val) : intvalue(val), tokvalue(nullptr), varvalue(val), condition(c), varId(0U), conditional(false), inconclusive(false), defaultArg(false), valueKind(ValueKind::Possible) {}
38     
39             /** int value */
40             long long intvalue;
41     
42             /** token value - the token that has the value. this is used for pointer aliases, strings, etc. */
43             const Token *tokvalue;
44     
45             /** For calculated values - variable value that calculated value depends on */
46             long long varvalue;
47     
48             /** Condition that this value depends on (TODO: replace with a 'callstack') */
49             const Token *condition;
50     
51             /** For calculated values - varId that calculated value depends on */
52             unsigned int varId;
53     
54             /** Conditional value */
55             bool conditional;
56     
57             /** Is this value inconclusive? */
58             bool inconclusive;
59     
60             /** Is this value passed as default parameter to the function? */
61             bool defaultArg;
62     
63             /** How known is this value */
64             enum ValueKind {
65                 /** This value is possible, other unlisted values may also be possible */
66                 Possible,
67                 /** Only listed values are possible */
68                 Known,
69                 /** Max value. Greater values are impossible. */
70                 Max,
71                 /** Min value. Smaller values are impossible. */
72                 Min
73             } valueKind;
74     
75             void setKnown() {
76                 valueKind = ValueKind::Known;
77             }
78     
79             bool isKnown() const {
80                 return valueKind == ValueKind::Known;
81             }
82     
83             void setPossible() {
84                 valueKind = ValueKind::Possible;
85             }
86     
87             bool isPossible() const {
88                 return valueKind == ValueKind::Possible;
89             }
90     
91             void changeKnownToPossible() {
92                 if (isKnown())
93                     valueKind = ValueKind::Possible;
94             }
95         };
96     
97         void setValues(TokenList *tokenlist, SymbolDatabase* symboldatabase, ErrorLogger *errorLogger, const Settings *settings);
98     
99         std::string eitherTheConditionIsRedundant(const Token *condition);
100     }
101     
102     #endif // valueflowH
