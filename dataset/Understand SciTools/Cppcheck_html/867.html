39     /**
40      * @brief This is just a container for general settings so that we don't need
41      * to pass individual values to functions or constructors now or in the
42      * future when we might have even more detailed settings.
43      */
44     class CPPCHECKLIB Settings {
45     private:
46         /** @brief Code to append in the checks */
47         std::string _append;
48     
49         /** @brief enable extra checks by id */
50         std::set<std::string> _enabled;
51     
52         /** @brief terminate checking */
53         bool _terminated;
54     
55     public:
56         Settings();
57     
58         /** @brief Is --debug given? */
59         bool debug;
60     
61         /** @brief Is --debug-normal given? */
62         bool debugnormal;
63     
64         /** @brief Is --debug-warnings given? */
65         bool debugwarnings;
66     
67         /** @brief Is --dump given? */
68         bool dump;
69     
70         /** @brief Is --exception-handling given */
71         bool exceptionHandling;
72     
73         /** @brief Inconclusive checks */
74         bool inconclusive;
75     
76         /** @brief Collect unmatched suppressions in one run.
77           * This delays the reporting until all files are checked.
78           * It is needed by checks that analyse the whole code base. */
79         bool jointSuppressionReport;
80     
81         /**
82          * When this flag is false (default) then experimental
83          * heuristics and checks are disabled.
84          *
85          * It should not be possible to enable this from any client.
86          */
87         bool experimental;
88     
89         /** @brief Is --quiet given? */
90         bool quiet;
91     
92         /** @brief Is --inline-suppr given? */
93         bool inlineSuppressions;
94     
95         /** @brief Is --verbose given? */
96         bool verbose;
97     
98         /** @brief Request termination of checking */
99         void terminate() {
100             _terminated = true;
101         }
102     
103         /** @brief termination requested? */
104         bool terminated() const {
105             return _terminated;
106         }
107     
108         /** @brief Force checking the files with "too many" configurations (--force). */
109         bool force;
110     
111         /** @brief Use relative paths in output. */
112         bool relativePaths;
113     
114         /** @brief Paths used as base for conversion to relative paths. */
115         std::vector<std::string> basePaths;
116     
117         /** @brief write XML results (--xml) */
118         bool xml;
119     
120         /** @brief XML version (--xmlver=..) */
121         int xml_version;
122     
123         /** @brief How many processes/threads should do checking at the same
124             time. Default is 1. (-j N) */
125         unsigned int jobs;
126     
127         /** @brief Load average value */
128         unsigned int loadAverage;
129     
130         /** @brief If errors are found, this value is returned from main().
131             Default value is 0. */
132         int exitCode;
133     
134         /** @brief The output format in which the errors are printed in text mode,
135             e.g. "{severity} {file}:{line} {message} {id}" */
136         std::string outputFormat;
137     
138         /** @brief show timing information (--showtime=file|summary|top5) */
139         SHOWTIME_MODES showtime;
140     
141         /** @brief Using -E for debugging purposes */
142         bool preprocessOnly;
143     
144         /** @brief List of include paths, e.g. "my/includes/" which should be used
145             for finding include files inside source files. (-I) */
146         std::list<std::string> includePaths;
147     
148         /** @brief assign append code (--append) */
149         bool append(const std::string &filename);
150     
151         /** @brief get append code (--append) */
152         const std::string &append() const;
153     
154         /** @brief Maximum number of configurations to check before bailing.
155             Default is 12. (--max-configs=N) */
156         unsigned int maxConfigs;
157     
158         /**
159          * @brief Returns true if given id is in the list of
160          * enabled extra checks (--enable)
161          * @param str id for the extra check, e.g. "style"
162          * @return true if the check is enabled.
163          */
164         template<typename T>
165         bool isEnabled(T&& str) const {
166             return bool(_enabled.find(str) != _enabled.end());
167         }
168     
169         /**
170          * @brief Enable extra checks by id. See isEnabled()
171          * @param str single id or list of id values to be enabled
172          * or empty string to enable all. e.g. "style,possibleError"
173          * @return error message. empty upon success
174          */
175         std::string addEnabled(const std::string &str);
176     
177         /**
178          * @brief Disables all severities, except from error.
179          */
180         void clearEnabled() {
181             _enabled.clear();
182         }
183     
184         enum Language {
185             None, C, CPP
186         };
187     
188         /** @brief Name of the language that is enforced. Empty per default. */
189         Language enforcedLang;
190     
191         /** @brief suppress message (--suppressions) */
192         Suppressions nomsg;
193     
194         /** @brief suppress exitcode */
195         Suppressions nofail;
196     
197         /** @brief defines given by the user */
198         std::string userDefines;
199     
200         /** @brief undefines given by the user */
201         std::set<std::string> userUndefs;
202     
203         /** @brief forced includes given by the user */
204         std::list<std::string> userIncludes;
205     
206         /** @brief include paths excluded from checking the configuration */
207         std::set<std::string> configExcludePaths;
208     
209     
210         /** @brief --report-progress */
211         bool reportProgress;
212     
213         /** Library (--library) */
214         Library library;
215     
216         /** Rule */
217         class CPPCHECKLIB Rule {
218         public:
219             Rule()
220                 : tokenlist("simple")         // use simple tokenlist
221                 , id("rule")                  // default id
222                 , severity(Severity::style) { // default severity
223             }
224     
225             std::string tokenlist;
226             std::string pattern;
227             std::string id;
228             std::string summary;
229             Severity::SeverityType severity;
230         };
231     
232         /**
233          * @brief Extra rules
234          */
235         std::list<Rule> rules;
236     
237         /** Is the 'configuration checking' wanted? */
238         bool checkConfiguration;
239     
240         /** Check for incomplete info in library files? */
241         bool checkLibrary;
242     
243         /** Struct contains standards settings */
244         Standards standards;
245     
246         unsigned int char_bit;       /// bits in char
247         unsigned int short_bit;      /// bits in short
248         unsigned int int_bit;        /// bits in int
249         unsigned int long_bit;       /// bits in long
250         unsigned int long_long_bit;  /// bits in long long
251     
252         /** size of standard types */
253         unsigned int sizeof_bool;
254         unsigned int sizeof_short;
255         unsigned int sizeof_int;
256         unsigned int sizeof_long;
257         unsigned int sizeof_long_long;
258         unsigned int sizeof_float;
259         unsigned int sizeof_double;
260         unsigned int sizeof_long_double;
261         unsigned int sizeof_wchar_t;
262         unsigned int sizeof_size_t;
263         unsigned int sizeof_pointer;
264     
265         char defaultSign;  // unsigned:'u', signed:'s', unknown:'\0'
266     
267         enum PlatformType {
268             Unspecified, // No platform specified
269             Native, // whatever system this code was compiled on
270             Win32A,
271             Win32W,
272             Win64,
273             Unix32,
274             Unix64
275         };
276     
277         /** platform type */
278         PlatformType platformType;
279     
280         /** set the platform type for predefined platforms */
281         bool platform(PlatformType type);
282     
283         /** set the platform type for user specified platforms */
284         bool platformFile(const std::string &filename);
285     
286         /**
287          * @brief Returns true if platform type is Windows
288          * @return true if Windows platform type.
289          */
290         bool isWindowsPlatform() const {
291             return platformType == Win32A ||
292                    platformType == Win32W ||
293                    platformType == Win64;
294         }
295     
296         /**
297          * @brief return true if a file is to be excluded from configuration checking
298          * @return true for the file to be excluded.
299          */
300         bool configurationExcluded(const std::string &file) const {
301             for (std::set<std::string>::const_iterator i=configExcludePaths.begin(); i!=configExcludePaths.end(); ++i) {
302                 if (file.length()>=i->length() && file.compare(0,i->length(),*i)==0) {
303                     return true;
304                 }
305             }
306             return false;
307         }
308     };
309     
310     /// @}
311     //---------------------------------------------------------------------------
312     #endif // settingsH
