38     void CheckUninitVar::check()
39     {
40         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
41         std::list<Scope>::const_iterator scope;
42     
43         // check every executable scope
44         for (scope = symbolDatabase->scopeList.begin(); scope != symbolDatabase->scopeList.end(); ++scope) {
45             if (scope->isExecutable()) {
46                 checkScope(&*scope);
47             }
48         }
49     }

51     void CheckUninitVar::checkScope(const Scope* scope)
52     {
53         for (std::list<Variable>::const_iterator i = scope->varlist.begin(); i != scope->varlist.end(); ++i) {
54             if ((_tokenizer->isCPP() && i->type() && !i->isPointer() && i->type()->needInitialization != Type::True) ||
55                 i->isStatic() || i->isExtern() || i->isReference())
56                 continue;
57     
58             // don't warn for try/catch exception variable
59             if (i->isThrow())
60                 continue;
61     
62             if (i->nameToken()->strAt(1) == "(" || i->nameToken()->strAt(1) == "{"  || i->nameToken()->strAt(1) == ":")
63                 continue;
64     
65             if (Token::Match(i->nameToken(), "%name% =")) { // Variable is initialized, but Rhs might be not
66                 checkRhs(i->nameToken(), *i, NO_ALLOC, 0U, "");
67                 continue;
68             }
69             if (Token::Match(i->nameToken(), "%name% ) (") && Token::simpleMatch(i->nameToken()->linkAt(2), ") =")) { // Function pointer is initialized, but Rhs might be not
70                 checkRhs(i->nameToken()->linkAt(2)->next(), *i, NO_ALLOC, 0U, "");
71                 continue;
72             }
73     
74             if (i->isArray() || i->isPointerToArray()) {
75                 const Token *tok = i->nameToken()->next();
76                 if (i->isPointerToArray())
77                     tok = tok->next();
78                 while (Token::simpleMatch(tok->link(), "] ["))
79                     tok = tok->link()->next();
80                 if (Token::Match(tok->link(), "] =|{"))
81                     continue;
82             }
83     
84             bool stdtype = _tokenizer->isC();
85             const Token* tok = i->typeStartToken();
86             for (; tok != i->nameToken() && tok->str() != "<"; tok = tok->next()) {
87                 if (tok->isStandardType())
88                     stdtype = true;
89             }
90             if (i->isArray() && !stdtype)
91                 continue;
92     
93             while (tok && tok->str() != ";")
94                 tok = tok->next();
95             if (!tok)
96                 continue;
97     
98             if (i->isArray()) {
99                 Alloc alloc = ARRAY;
100                 checkScopeForVariable(tok, *i, nullptr, nullptr, &alloc, "");
101                 continue;
102             }
103             if (stdtype || i->isPointer()) {
104                 Alloc alloc = NO_ALLOC;
105                 checkScopeForVariable(tok, *i, nullptr, nullptr, &alloc, "");
106             }
107             if (i->type())
108                 checkStruct(tok, *i);
109         }
110     
111         if (scope->function) {
112             for (unsigned int i = 0; i < scope->function->argCount(); i++) {
113                 const Variable *arg = scope->function->getArgumentVar(i);
114                 if (arg && arg->declarationId() && Token::Match(arg->typeStartToken(), "struct| %type% * %name% [,)]")) {
115                     // Treat the pointer as initialized until it is assigned by malloc
116                     for (const Token *tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
117                         if (Token::Match(tok, "[;{}] %varid% = %name% (", arg->declarationId()) &&
118                             _settings->library.returnuninitdata.count(tok->strAt(3)) == 1U) {
119                             if (arg->typeStartToken()->str() == "struct")
120                                 checkStruct(tok, *arg);
121                             else if (arg->typeStartToken()->isStandardType()) {
122                                 Alloc alloc = NO_ALLOC;
123                                 checkScopeForVariable(tok->next(), *arg, nullptr, nullptr, &alloc, "");
124                             }
125                         }
126                     }
127                 }
128             }
129         }
130     }

132     void CheckUninitVar::checkStruct(const Token *tok, const Variable &structvar)
133     {
134         const Token *typeToken = structvar.typeStartToken();
135         if (typeToken->str() == "struct")
136             typeToken = typeToken->next();
137         const SymbolDatabase * symbolDatabase = _tokenizer->getSymbolDatabase();
138         for (std::size_t j = 0U; j < symbolDatabase->classAndStructScopes.size(); ++j) {
139             const Scope *scope2 = symbolDatabase->classAndStructScopes[j];
140             if (scope2->className == typeToken->str() && scope2->numConstructors == 0U) {
141                 for (std::list<Variable>::const_iterator it = scope2->varlist.begin(); it != scope2->varlist.end(); ++it) {
142                     const Variable &var = *it;
143     
144                     if (var.hasDefault() || var.isArray() || (!_tokenizer->isC() && var.isClass() && (!var.type() || var.type()->needInitialization != Type::True)))
145                         continue;
146     
147                     // is the variable declared in a inner union?
148                     bool innerunion = false;
149                     for (std::list<Scope>::const_iterator it2 = symbolDatabase->scopeList.begin(); it2 != symbolDatabase->scopeList.end(); ++it2) {
150                         const Scope &innerScope = *it2;
151                         if (innerScope.type == Scope::eUnion && innerScope.nestedIn == scope2) {
152                             if (var.typeStartToken()->linenr() >= innerScope.classStart->linenr() &&
153                                 var.typeStartToken()->linenr() <= innerScope.classEnd->linenr()) {
154                                 innerunion = true;
155                                 break;
156                             }
157     
158                         }
159                     }
160     
161                     if (!innerunion) {
162                         Alloc alloc = NO_ALLOC;
163                         const Token *tok2 = tok;
164                         if (tok->str() == "}")
165                             tok2 = tok2->next();
166                         checkScopeForVariable(tok2, structvar, nullptr, nullptr, &alloc, var.name());
167                     }
168                 }
169             }
170         }
171     }

173     struct VariableValue {
174         explicit VariableValue(MathLib::bigint val = 0) : value(val), notEqual(false) {}
175         MathLib::bigint value;
176         bool notEqual;
177     };
178     static VariableValue operator!(VariableValue v)
179     {
180         v.notEqual = !v.notEqual;
181         return v;
182     }
183     static bool operator==(const VariableValue & v, MathLib::bigint i)
184     {
185         return v.notEqual ? (i != v.value) : (i == v.value);
186     }
187     static bool operator!=(const VariableValue & v, MathLib::bigint i)
188     {
189         return v.notEqual ? (i == v.value) : (i != v.value);
190     }

192     static void conditionAlwaysTrueOrFalse(const Token *tok, const std::map<unsigned int, VariableValue> &variableValue, bool *alwaysTrue, bool *alwaysFalse)
193     {
194         if (!tok)
195             return;
196     
197         if (tok->isName() || tok->str() == ".") {
198             while (tok && tok->str() == ".")
199                 tok = tok->astOperand2();
200             const std::map<unsigned int, VariableValue>::const_iterator it = variableValue.find(tok ? tok->varId() : ~0U);
201             if (it != variableValue.end()) {
202                 *alwaysTrue = (it->second != 0LL);
203                 *alwaysFalse = (it->second == 0LL);
204             }
205         }
206     
207         else if (tok->isComparisonOp()) {
208             const Token *vartok, *numtok;
209             if (tok->astOperand2() && tok->astOperand2()->isNumber()) {
210                 vartok = tok->astOperand1();
211                 numtok = tok->astOperand2();
212             } else if (tok->astOperand1() && tok->astOperand1()->isNumber()) {
213                 vartok = tok->astOperand2();
214                 numtok = tok->astOperand1();
215             } else {
216                 return;
217             }
218     
219             while (vartok && vartok->str() == ".")
220                 vartok = vartok->astOperand2();
221     
222             const std::map<unsigned int, VariableValue>::const_iterator it = variableValue.find(vartok ? vartok->varId() : ~0U);
223             if (it == variableValue.end())
224                 return;
225     
226             if (tok->str() == "==")
227                 *alwaysTrue  = (it->second == MathLib::toLongNumber(numtok->str()));
228             else if (tok->str() == "!=")
229                 *alwaysTrue  = (it->second != MathLib::toLongNumber(numtok->str()));
230             else
231                 return;
232             *alwaysFalse = !(*alwaysTrue);
233         }
234     
235         else if (tok->str() == "!") {
236             bool t=false,f=false;
237             conditionAlwaysTrueOrFalse(tok->astOperand1(), variableValue, &t, &f);
238             if (t||f) {
239                 *alwaysTrue = !t;
240                 *alwaysFalse = !f;
241             }
242         }
243     
244         else if (tok->str() == "||") {
245             bool t1=false, f1=false;
246             conditionAlwaysTrueOrFalse(tok->astOperand1(), variableValue, &t1, &f1);
247             bool t2=false, f2=false;
248             if (!t1)
249                 conditionAlwaysTrueOrFalse(tok->astOperand2(), variableValue, &t2, &f2);
250             *alwaysTrue = (t1 || t2);
251             *alwaysFalse = (f1 && f2);
252         }
253     
254         else if (tok->str() == "&&") {
255             bool t1=false, f1=false;
256             conditionAlwaysTrueOrFalse(tok->astOperand1(), variableValue, &t1, &f1);
257             bool t2=false, f2=false;
258             if (!f1)
259                 conditionAlwaysTrueOrFalse(tok->astOperand2(), variableValue, &t2, &f2);
260             *alwaysTrue = (t1 && t2);
261             *alwaysFalse = (f1 || f2);
262         }
263     }

265     static bool isVariableUsed(const Token *tok, const Variable& var)
266     {
267         if (!tok)
268             return false;
269         if (tok->str() == "&" && !tok->astOperand2())
270             return false;
271         if (tok->isConstOp())
272             return isVariableUsed(tok->astOperand1(),var) || isVariableUsed(tok->astOperand2(),var);
273         if (tok->varId() != var.declarationId())
274             return false;
275         if (!var.isArray())
276             return true;
277     
278         const Token *parent = tok->astParent();
279         while (Token::Match(parent, "[?:]"))
280             parent = parent->astParent();
281         // no dereference, then array is not "used"
282         if (!Token::Match(parent, "*|["))
283             return false;
284         const Token *parent2 = parent->astParent();
285         // TODO: handle function calls. There is a TODO assertion in TestUninitVar::uninitvar_arrays
286         return !parent2 || parent2->isConstOp() || (parent2->str() == "=" && parent2->astOperand2() == parent);
287     }

289     bool CheckUninitVar::checkScopeForVariable(const Token *tok, const Variable& var, bool * const possibleInit, bool * const noreturn, Alloc* const alloc, const std::string &membervar)
290     {
291         const bool suppressErrors(possibleInit && *possibleInit);
292         const bool printDebug = _settings->debugwarnings;
293     
294         if (possibleInit)
295             *possibleInit = false;
296     
297         unsigned int number_of_if = 0;
298     
299         if (var.declarationId() == 0U)
300             return true;
301     
302         // variable values
303         std::map<unsigned int, VariableValue> variableValue;
304     
305         for (; tok; tok = tok->next()) {
306             // End of scope..
307             if (tok->str() == "}") {
308                 if (number_of_if && possibleInit)
309                     *possibleInit = true;
310     
311                 // might be a noreturn function..
312                 if (_tokenizer->IsScopeNoReturn(tok)) {
313                     if (noreturn)
314                         *noreturn = true;
315                     return false;
316                 }
317     
318                 break;
319             }
320     
321             // Unconditional inner scope or try..
322             if (tok->str() == "{" && Token::Match(tok->previous(), ";|{|}|try")) {
323                 if (checkScopeForVariable(tok->next(), var, possibleInit, noreturn, alloc, membervar))
324                     return true;
325                 tok = tok->link();
326                 continue;
327             }
328     
329             // assignment with nonzero constant..
330             if (Token::Match(tok->previous(), "[;{}] %var% = - %name% ;"))
331                 variableValue[tok->varId()] = !VariableValue(0);
332     
333             // Inner scope..
334             else if (Token::simpleMatch(tok, "if (")) {
335                 bool alwaysTrue = false;
336                 bool alwaysFalse = false;
337     
338                 conditionAlwaysTrueOrFalse(tok->next()->astOperand2(), variableValue, &alwaysTrue, &alwaysFalse);
339     
340                 // initialization / usage in condition..
341                 if (!alwaysTrue && checkIfForWhileHead(tok->next(), var, suppressErrors, bool(number_of_if == 0), *alloc, membervar))
342                     return true;
343     
344                 // checking if a not-zero variable is zero => bail out
345                 unsigned int condVarId = 0;
346                 VariableValue condVarValue(0);
347                 const Token *condVarTok = nullptr;
348                 if (alwaysFalse)
349                     ;
350                 else if (Token::simpleMatch(tok, "if (") &&
351                          astIsVariableComparison(tok->next()->astOperand2(), "!=", "0", &condVarTok)) {
352                     std::map<unsigned int,VariableValue>::const_iterator it = variableValue.find(condVarTok->varId());
353                     if (it != variableValue.end() && it->second != 0)
354                         return true;   // this scope is not fully analysed => return true
355                     else {
356                         condVarId = condVarTok->varId();
357                         condVarValue = !VariableValue(0);
358                     }
359                 } else if (Token::simpleMatch(tok, "if (") && Token::Match(tok->next()->astOperand2(), "==|!=")) {
360                     const Token *condition = tok->next()->astOperand2();
361                     const Token *lhs = condition->astOperand1();
362                     const Token *rhs = condition->astOperand2();
363                     const Token *vartok = rhs && rhs->isNumber() ? lhs : rhs;
364                     const Token *numtok = rhs && rhs->isNumber() ? rhs : lhs;
365                     while (Token::simpleMatch(vartok, "."))
366                         vartok = vartok->astOperand2();
367                     if (vartok && vartok->varId() && numtok) {
368                         std::map<unsigned int,VariableValue>::const_iterator it = variableValue.find(vartok->varId());
369                         if (it != variableValue.end() && it->second != MathLib::toLongNumber(numtok->str()))
370                             return true;   // this scope is not fully analysed => return true
371                         else {
372                             condVarId = vartok->varId();
373                             condVarValue = VariableValue(MathLib::toLongNumber(numtok->str()));
374                             if (condition->str() == "!=")
375                                 condVarValue = !condVarValue;
376                         }
377                     }
378                 }
379     
380                 // goto the {
381                 tok = tok->next()->link()->next();
382     
383                 if (!tok)
384                     break;
385                 if (tok->str() == "{") {
386                     bool possibleInitIf((!alwaysTrue && number_of_if > 0) || suppressErrors);
387                     bool noreturnIf = false;
388                     const bool initif = !alwaysFalse && checkScopeForVariable(tok->next(), var, &possibleInitIf, &noreturnIf, alloc, membervar);
389     
390                     // bail out for such code:
391                     //    if (a) x=0;    // conditional initialization
392                     //    if (b) return; // cppcheck doesn't know if b can be false when a is false.
393                     //    x++;           // it's possible x is always initialized
394                     if (!alwaysTrue && noreturnIf && number_of_if > 0) {
395                         if (printDebug) {
396                             std::string condition;
397                             for (const Token *tok2 = tok->linkAt(-1); tok2 != tok; tok2 = tok2->next()) {
398                                 condition += tok2->str();
399                                 if (tok2->isName() && tok2->next()->isName())
400                                     condition += ' ';
401                             }
402                             reportError(tok, Severity::debug, "debug", "bailout uninitialized variable checking for '" + var.name() + "'. can't determine if this condition can be false when previous condition is false: " + condition);
403                         }
404                         return true;
405                     }
406     
407                     if (alwaysTrue && noreturnIf)
408                         return true;
409     
410                     std::map<unsigned int, VariableValue> varValueIf;
411                     if (!alwaysFalse && !initif && !noreturnIf) {
412                         for (const Token *tok2 = tok; tok2 && tok2 != tok->link(); tok2 = tok2->next()) {
413                             if (Token::Match(tok2, "[;{}.] %name% = - %name% ;"))
414                                 varValueIf[tok2->next()->varId()] = !VariableValue(0);
415                             else if (Token::Match(tok2, "[;{}.] %name% = %num% ;"))
416                                 varValueIf[tok2->next()->varId()] = VariableValue(MathLib::toLongNumber(tok2->strAt(3)));
417                         }
418                     }
419     
420                     if (initif && condVarId > 0U)
421                         variableValue[condVarId] = !condVarValue;
422     
423                     // goto the }
424                     tok = tok->link();
425     
426                     if (!Token::simpleMatch(tok, "} else {")) {
427                         if (initif || possibleInitIf) {
428                             ++number_of_if;
429                             if (number_of_if >= 2)
430                                 return true;
431                         }
432                     } else {
433                         // goto the {
434                         tok = tok->tokAt(2);
435     
436                         bool possibleInitElse((!alwaysFalse && number_of_if > 0) || suppressErrors);
437                         bool noreturnElse = false;
438                         const bool initelse = !alwaysTrue && checkScopeForVariable(tok->next(), var, &possibleInitElse, &noreturnElse, alloc, membervar);
439     
440                         std::map<unsigned int, VariableValue> varValueElse;
441                         if (!alwaysTrue && !initelse && !noreturnElse) {
442                             for (const Token *tok2 = tok; tok2 && tok2 != tok->link(); tok2 = tok2->next()) {
443                                 if (Token::Match(tok2, "[;{}.] %var% = - %name% ;"))
444                                     varValueElse[tok2->next()->varId()] = !VariableValue(0);
445                                 else if (Token::Match(tok2, "[;{}.] %var% = %num% ;"))
446                                     varValueElse[tok2->next()->varId()] = VariableValue(MathLib::toLongNumber(tok2->strAt(3)));
447                             }
448                         }
449     
450                         if (initelse && condVarId > 0U && !noreturnIf && !noreturnElse)
451                             variableValue[condVarId] = condVarValue;
452     
453                         // goto the }
454                         tok = tok->link();
455     
456                         if ((alwaysFalse || initif || noreturnIf) &&
457                             (alwaysTrue || initelse || noreturnElse))
458                             return true;
459     
460                         if (initif || initelse || possibleInitElse)
461                             ++number_of_if;
462                         if (!initif && !noreturnIf)
463                             variableValue.insert(varValueIf.begin(), varValueIf.end());
464                         if (!initelse && !noreturnElse)
465                             variableValue.insert(varValueElse.begin(), varValueElse.end());
466                     }
467                 }
468             }
469     
470             // = { .. }
471             else if (Token::simpleMatch(tok, "= {")) {
472                 // end token
473                 const Token *end = tok->next()->link();
474     
475                 // If address of variable is taken in the block then bail out
476                 if (Token::findmatch(tok->tokAt(2), "& %varid%", end, var.declarationId()))
477                     return true;
478     
479                 // Skip block
480                 tok = end;
481                 continue;
482             }
483     
484             // skip sizeof / offsetof
485             if (Token::Match(tok, "sizeof|typeof|offsetof|decltype ("))
486                 tok = tok->next()->link();
487     
488             // for/while..
489             else if (Token::Match(tok, "for|while (") || Token::simpleMatch(tok, "do {")) {
490                 const bool forwhile = Token::Match(tok, "for|while (");
491     
492                 // is variable initialized in for-head (don't report errors yet)?
493                 if (forwhile && checkIfForWhileHead(tok->next(), var, true, false, *alloc, membervar))
494                     return true;
495     
496                 // goto the {
497                 const Token *tok2 = forwhile ? tok->next()->link()->next() : tok->next();
498     
499                 if (tok2 && tok2->str() == "{") {
500                     bool init = checkLoopBody(tok2, var, *alloc, membervar, (number_of_if > 0) || suppressErrors);
501     
502                     // variable is initialized in the loop..
503                     if (init)
504                         return true;
505     
506                     // is variable used in for-head?
507                     bool initcond = false;
508                     if (!suppressErrors) {
509                         const Token *startCond = forwhile ? tok->next() : tok->next()->link()->tokAt(2);
510                         initcond = checkIfForWhileHead(startCond, var, false, bool(number_of_if == 0), *alloc, membervar);
511                     }
512     
513                     // goto "}"
514                     tok = tok2->link();
515     
516                     // do-while => goto ")"
517                     if (!forwhile) {
518                         // Assert that the tokens are '} while ('
519                         if (!Token::simpleMatch(tok, "} while (")) {
520                             if (printDebug)
521                                 reportError(tok,Severity::debug,"","assertion failed '} while ('");
522                             break;
523                         }
524     
525                         // Goto ')'
526                         tok = tok->linkAt(2);
527     
528                         if (!tok)
529                             // bailout : invalid code / bad tokenizer
530                             break;
531     
532                         if (initcond)
533                             // variable is initialized in while-condition
534                             return true;
535                     }
536                 }
537             }
538     
539             // Unknown or unhandled inner scope
540             else if (Token::simpleMatch(tok, ") {") || (Token::Match(tok, "%name% {") && tok->str() != "try")) {
541                 if (tok->str() == "struct" || tok->str() == "union") {
542                     tok = tok->linkAt(1);
543                     continue;
544                 }
545                 return true;
546             }
547     
548             // bailout if there is ({
549             if (Token::simpleMatch(tok, "( {")) {
550                 return true;
551             }
552     
553             // bailout if there is assembler code or setjmp
554             if (Token::Match(tok, "asm|setjmp (")) {
555                 return true;
556             }
557     
558             // bailout if there is a goto label
559             if (Token::Match(tok, "[;{}] %name% :")) {
560                 return true;
561             }
562     
563             if (tok->str() == "?") {
564                 if (!tok->astOperand2())
565                     return true;
566                 const bool used1 = isVariableUsed(tok->astOperand2()->astOperand1(), var);
567                 const bool used0 = isVariableUsed(tok->astOperand2()->astOperand2(), var);
568                 const bool err = (number_of_if == 0) ? (used1 || used0) : (used1 && used0);
569                 if (err)
570                     uninitvarError(tok, var.nameToken()->str(), *alloc);
571     
572                 // Todo: skip expression if there is no error
573                 return true;
574             }
575     
576             if (Token::Match(tok, "return|break|continue|throw|goto")) {
577                 if (noreturn)
578                     *noreturn = true;
579     
580                 tok = tok->next();
581                 while (tok && tok->str() != ";") {
582                     // variable is seen..
583                     if (tok->varId() == var.declarationId()) {
584                         if (!membervar.empty()) {
585                             if (Token::Match(tok, "%name% . %name% ;|%cop%") && tok->strAt(2) == membervar)
586                                 uninitStructMemberError(tok, tok->str() + "." + membervar);
587                             else
588                                 return true;
589                         }
590     
591                         // Use variable
592                         else if (!suppressErrors && isVariableUsage(tok, var.isPointer(), *alloc))
593                             uninitvarError(tok, tok->str(), *alloc);
594     
595                         else
596                             // assume that variable is assigned
597                             return true;
598                     }
599     
600                     else if (Token::Match(tok, "sizeof|typeof|offsetof|decltype ("))
601                         tok = tok->linkAt(1);
602     
603                     else if (tok->str() == "?") {
604                         if (!tok->astOperand2())
605                             return true;
606                         const bool used1 = isVariableUsed(tok->astOperand2()->astOperand1(), var);
607                         const bool used0 = isVariableUsed(tok->astOperand2()->astOperand2(), var);
608                         const bool err = (number_of_if == 0) ? (used1 || used0) : (used1 && used0);
609                         if (err)
610                             uninitvarError(tok, var.nameToken()->str(), *alloc);
611                         return true;
612                     }
613     
614                     tok = tok->next();
615                 }
616     
617                 return bool(noreturn==nullptr);
618             }
619     
620             // variable is seen..
621             if (tok->varId() == var.declarationId()) {
622                 // calling function that returns uninit data through pointer..
623                 if (var.isPointer() &&
624                     Token::Match(tok->next(), "= %name% (") &&
625                     Token::simpleMatch(tok->linkAt(3), ") ;") &&
626                     _settings->library.returnuninitdata.count(tok->strAt(2)) > 0U) {
627                     *alloc = NO_CTOR_CALL;
628                     continue;
629                 }
630                 if (var.isPointer() && (var.typeStartToken()->isStandardType() || (var.type() && var.type()->needInitialization == Type::True)) && Token::simpleMatch(tok->next(), "= new")) {
631                     *alloc = CTOR_CALL;
632                     if (var.typeScope() && var.typeScope()->numConstructors > 0)
633                         return true;
634                     continue;
635                 }
636     
637     
638                 if (!membervar.empty()) {
639                     if (isMemberVariableAssignment(tok, membervar)) {
640                         checkRhs(tok, var, *alloc, number_of_if, membervar);
641                         return true;
642                     }
643     
644                     if (isMemberVariableUsage(tok, var.isPointer(), *alloc, membervar))
645                         uninitStructMemberError(tok, tok->str() + "." + membervar);
646     
647                     else if (Token::Match(tok->previous(), "[(,] %name% [,)]"))
648                         return true;
649     
650                 } else {
651                     // Use variable
652                     if (!suppressErrors && isVariableUsage(tok, var.isPointer(), *alloc))
653                         uninitvarError(tok, tok->str(), *alloc);
654     
655                     else {
656                         if (tok->strAt(1) == "=")
657                             checkRhs(tok, var, *alloc, number_of_if, "");
658     
659                         // assume that variable is assigned
660                         return true;
661                     }
662                 }
663             }
664         }
665     
666         return false;
667     }

669     bool CheckUninitVar::checkIfForWhileHead(const Token *startparentheses, const Variable& var, bool suppressErrors, bool isuninit, Alloc alloc, const std::string &membervar)
670     {
671         const Token * const endpar = startparentheses->link();
672         if (Token::Match(startparentheses, "( ! %name% %oror%") && startparentheses->tokAt(2)->getValue(0))
673             suppressErrors = true;
674         for (const Token *tok = startparentheses->next(); tok && tok != endpar; tok = tok->next()) {
675             if (tok->varId() == var.declarationId()) {
676                 if (Token::Match(tok, "%name% . %name%")) {
677                     if (membervar.empty())
678                         return true;
679                     if (tok->strAt(2) == membervar) {
680                         if (isMemberVariableAssignment(tok, membervar))
681                             return true;
682     
683                         if (!suppressErrors && isMemberVariableUsage(tok, var.isPointer(), alloc, membervar))
684                             uninitStructMemberError(tok, tok->str() + "." + membervar);
685                     }
686                     continue;
687                 }
688     
689                 if (isVariableUsage(tok, var.isPointer(), alloc)) {
690                     if (suppressErrors)
691                         continue;
692                     uninitvarError(tok, tok->str(), alloc);
693                 }
694                 return true;
695             }
696             if (Token::Match(tok, "sizeof|decltype|offsetof ("))
697                 tok = tok->next()->link();
698             if ((!isuninit || !membervar.empty()) && tok->str() == "&&")
699                 suppressErrors = true;
700         }
701         return false;
702     }

704     bool CheckUninitVar::checkLoopBody(const Token *tok, const Variable& var, const Alloc alloc, const std::string &membervar, const bool suppressErrors)
705     {
706         const Token *usetok = nullptr;
707     
708         assert(tok->str() == "{");
709     
710         for (const Token * const end = tok->link(); tok != end; tok = tok->next()) {
711             if (tok->varId() == var.declarationId()) {
712                 if (!membervar.empty()) {
713                     if (isMemberVariableAssignment(tok, membervar)) {
714                         bool assign = true;
715                         bool rhs = false;
716                         for (const Token *tok2 = tok->next(); tok2; tok2 = tok2->next()) {
717                             if (tok2->str() == "=")
718                                 rhs = true;
719                             if (tok2->str() == ";")
720                                 break;
721                             if (rhs && tok2->varId() == var.declarationId() && isMemberVariableUsage(tok2, var.isPointer(), alloc, membervar)) {
722                                 assign = false;
723                                 break;
724                             }
725                         }
726                         if (assign)
727                             return true;
728                     }
729     
730                     if (Token::Match(tok, "%name% ="))
731                         return true;
732     
733                     if (isMemberVariableUsage(tok, var.isPointer(), alloc, membervar))
734                         usetok = tok;
735                     else if (Token::Match(tok->previous(), "[(,] %name% [,)]"))
736                         return true;
737                 } else {
738                     if (isVariableUsage(tok, var.isPointer(), alloc))
739                         usetok = tok;
740                     else if (tok->strAt(1) == "=") {
741                         // Is var used in rhs?
742                         bool rhs = false;
743                         std::stack<const Token *> tokens;
744                         tokens.push(tok->next()->astOperand2());
745                         while (!tokens.empty()) {
746                             const Token *t = tokens.top();
747                             tokens.pop();
748                             if (!t)
749                                 continue;
750                             if (t->varId() == var.declarationId()) {
751                                 // var is used in rhs
752                                 rhs = true;
753                                 break;
754                             }
755                             if (Token::simpleMatch(t->previous(),"sizeof ("))
756                                 continue;
757                             tokens.push(t->astOperand1());
758                             tokens.push(t->astOperand2());
759                         }
760                         if (!rhs)
761                             return true;
762                     } else {
763                         return true;
764                     }
765                 }
766             }
767     
768             if (Token::Match(tok, "sizeof|typeof ("))
769                 tok = tok->next()->link();
770     
771             if (Token::Match(tok, "asm ( %str% ) ;"))
772                 return true;
773         }
774     
775         if (!suppressErrors && usetok) {
776             if (membervar.empty())
777                 uninitvarError(usetok, usetok->str(), alloc);
778             else
779                 uninitStructMemberError(usetok, usetok->str() + "." + membervar);
780             return true;
781         }
782     
783         return false;
784     }

786     void CheckUninitVar::checkRhs(const Token *tok, const Variable &var, Alloc alloc, unsigned int number_of_if, const std::string &membervar)
787     {
788         bool rhs = false;
789         unsigned int indent = 0;
790         while (nullptr != (tok = tok->next())) {
791             if (tok->str() == "=")
792                 rhs = true;
793             else if (rhs && tok->varId() == var.declarationId()) {
794                 if (membervar.empty() && isVariableUsage(tok, var.isPointer(), alloc))
795                     uninitvarError(tok, tok->str(), alloc);
796                 else if (!membervar.empty() && isMemberVariableUsage(tok, var.isPointer(), alloc, membervar))
797                     uninitStructMemberError(tok, tok->str() + "." + membervar);
798     
799             } else if (tok->str() == ";" || (indent==0 && tok->str() == ","))
800                 break;
801             else if (tok->str() == "(")
802                 ++indent;
803             else if (tok->str() == ")") {
804                 if (indent == 0)
805                     break;
806                 --indent;
807             } else if (tok->str() == "?" && tok->astOperand2()) {
808                 const bool used1 = isVariableUsed(tok->astOperand2()->astOperand1(), var);
809                 const bool used0 = isVariableUsed(tok->astOperand2()->astOperand2(), var);
810                 const bool err = (number_of_if == 0) ? (used1 || used0) : (used1 && used0);
811                 if (err)
812                     uninitvarError(tok, var.nameToken()->str(), alloc);
813                 break;
814             } else if (Token::simpleMatch(tok, "sizeof ("))
815                 tok = tok->next()->link();
816         }
817     }

819     bool CheckUninitVar::isVariableUsage(const Token *vartok, bool pointer, Alloc alloc) const
820     {
821         if (alloc == NO_ALLOC && ((Token::Match(vartok->previous(), "return|delete") && vartok->strAt(1) != "=") || (vartok->strAt(-1) == "]" && vartok->linkAt(-1)->strAt(-1) == "delete")))
822             return true;
823     
824         // Passing variable to typeof/__alignof__
825         if (Token::Match(vartok->tokAt(-3), "typeof|__alignof__ ( * %name%"))
826             return false;
827     
828         // Accessing Rvalue member using "." or "->"
829         if (vartok->strAt(1) == "." && vartok->strAt(-1) != "&") {
830             // Is struct member passed to function?
831             if (!pointer && Token::Match(vartok->previous(), "[,(] %name% . %name%")) {
832                 // TODO: there are FN currently:
833                 // - should only return false if struct member is (or might be) array.
834                 // - should only return false if function argument is (or might be) non-const pointer or reference
835                 const Token *tok2 = vartok->next();
836                 do {
837                     tok2 = tok2->tokAt(2);
838                 } while (Token::Match(tok2, ". %name%"));
839                 if (Token::Match(tok2, "[,)]"))
840                     return false;
841             } else if (pointer && alloc != CTOR_CALL && Token::Match(vartok, "%name% . %name% (")) {
842                 return true;
843             }
844     
845             bool assignment = false;
846             const Token* parent = vartok->astParent();
847             while (parent) {
848                 if (parent->str() == "=") {
849                     assignment = true;
850                     break;
851                 }
852                 if (alloc != NO_ALLOC && parent->str() == "(") {
853                     if (_settings->library.functionpure.find(parent->strAt(-1)) == _settings->library.functionpure.end()) {
854                         assignment = true;
855                         break;
856                     }
857                 }
858                 parent = parent->astParent();
859             }
860             if (!assignment)
861                 return true;
862         }
863     
864         // Passing variable to function..
865         if (Token::Match(vartok->previous(), "[(,] %name% [,)]") || Token::Match(vartok->tokAt(-2), "[(,] & %name% [,)]")) {
866             const int use = isFunctionParUsage(vartok, pointer, alloc);
867             if (use >= 0)
868                 return (use>0);
869         }
870     
871         if (Token::Match(vartok->previous(), "++|--|%cop%")) {
872             if (_tokenizer->isCPP() && alloc == ARRAY && Token::Match(vartok->tokAt(-4), "& %var% =|( *"))
873                 return false;
874     
875             if (_tokenizer->isCPP() && Token::Match(vartok->previous(), ">>|<<")) {
876                 const Token* tok2 = vartok->previous();
877                 if (Token::simpleMatch(tok2->astOperand1(), ">>"))
878                     return false; // Looks like stream operator, initializes the variable
879                 if (Token::simpleMatch(tok2, "<<")) {
880                     // Looks like stream operator, but could also initialize the variable. Check lhs.
881                     do {
882                         tok2 = tok2->astOperand1();
883                     } while (Token::simpleMatch(tok2, "<<"));
884                     if (tok2 && tok2->strAt(-1) == "::")
885                         tok2 = tok2->previous();
886                     if (tok2 && (Token::simpleMatch(tok2->previous(), "std ::") || (tok2->variable() && tok2->variable()->isStlType()) || tok2->isStandardType()))
887                         return true;
888                 }
889                 const Variable *var = vartok->tokAt(-2)->variable();
890                 return (var && var->typeStartToken()->isStandardType());
891             }
892     
893             // is there something like: ; "*((&var ..expr.. ="  => the variable is assigned
894             if (vartok->previous()->str() == "&" && !vartok->previous()->astOperand2())
895                 return false;
896     
897             // bailout to avoid fp for 'int x = 2 + x();' where 'x()' is a unseen preprocessor macro (seen in linux)
898             if (!pointer && vartok->next() && vartok->next()->str() == "(")
899                 return false;
900     
901             if (vartok->previous()->str() != "&" || !Token::Match(vartok->tokAt(-2), "[(,=?:]")) {
902                 if (alloc != NO_ALLOC && vartok->previous()->str() == "*") {
903                     const Token *parent = vartok->previous()->astParent();
904                     if (parent && parent->str() == "=" && parent->astOperand1() == vartok->previous())
905                         return false;
906                     return true;
907                 }
908                 return alloc == NO_ALLOC;
909             }
910         }
911     
912         if (alloc == NO_ALLOC && Token::Match(vartok->previous(), "= %name% ;|%cop%")) {
913             // taking reference?
914             const Token *prev = vartok->tokAt(-2);
915             while (Token::Match(prev, "%name%|*"))
916                 prev = prev->previous();
917             if (!Token::simpleMatch(prev, "&"))
918                 return true;
919         }
920     
921         bool unknown = false;
922         if (pointer && alloc == NO_ALLOC && CheckNullPointer::isPointerDeRef(vartok, unknown)) {
923             // function parameter?
924             bool functionParameter = false;
925             if (Token::Match(vartok->tokAt(-2), "%name% (") || vartok->previous()->str() == ",")
926                 functionParameter = true;
927     
928             // if this is not a function parameter report this dereference as variable usage
929             if (!functionParameter)
930                 return true;
931         } else if (alloc != NO_ALLOC && Token::Match(vartok, "%var% [")) {
932             const Token *parent = vartok->next()->astParent();
933             while (Token::Match(parent, "[|."))
934                 parent = parent->astParent();
935             if (Token::simpleMatch(parent, "&") && !parent->astOperand2())
936                 return false;
937             if (parent && Token::Match(parent->previous(), "if|while|switch ("))
938                 return true;
939             if (Token::Match(parent, "[=,(]"))
940                 return false;
941             return true;
942         }
943     
944         if (_tokenizer->isCPP() && Token::Match(vartok->next(), "<<|>>")) {
945             // Is this calculation done in rhs?
946             const Token *tok = vartok;
947             while (tok && Token::Match(tok, "%name%|.|::"))
948                 tok = tok->previous();
949             if (Token::Match(tok, "[;{}]"))
950                 return false;
951     
952             // Is variable a known POD type then this is a variable usage,
953             // otherwise we assume it's not.
954             const Variable *var = vartok->variable();
955             return (var && var->typeStartToken()->isStandardType());
956         }
957     
958         if (alloc == NO_ALLOC && vartok->next() && vartok->next()->isOp() && !vartok->next()->isAssignmentOp())
959             return true;
960     
961         if (vartok->strAt(1) == "]")
962             return true;
963     
964         return false;
965     }

967     /***
968      * Is function parameter "used" so a "usage of uninitialized variable" can
969      * be written? If parameter is passed "by value" then it is "used". If it
970      * is passed "by reference" then it is not necessarily "used".
971      * @return  -1 => unknown   0 => not used   1 => used
972      */
973     int CheckUninitVar::isFunctionParUsage(const Token *vartok, bool pointer, Alloc alloc) const
974     {
975         if (!Token::Match(vartok->previous(), "[(,]") && !Token::Match(vartok->tokAt(-2), "[(,] &"))
976             return -1;
977     
978         // locate start parentheses in function call..
979         unsigned int argumentNumber = 0;
980         const Token *start = vartok;
981         while (start && !Token::Match(start, "[;{}(]")) {
982             if (start->str() == ")")
983                 start = start->link();
984             else if (start->str() == ",")
985                 ++argumentNumber;
986             start = start->previous();
987         }
988     
989         // is this a function call?
990         if (start && Token::Match(start->previous(), "%name% (")) {
991             const bool address(vartok->previous()->str() == "&");
992             const bool array(vartok->variable() && vartok->variable()->isArray());
993             // check how function handle uninitialized data arguments..
994             const Function *func = start->previous()->function();
995             if (func) {
996                 const Variable *arg = func->getArgumentVar(argumentNumber);
997                 if (arg) {
998                     const Token *argStart = arg->typeStartToken();
999                     if (!address && !array && Token::Match(argStart, "struct| %type% %name%| [,)]"))
1000                         return 1;
1001                     if (pointer && !address && alloc == NO_ALLOC && Token::Match(argStart, "struct| %type% * %name% [,)]"))
1002                         return 1;
1003                     while (argStart->previous() && argStart->previous()->isName())
1004                         argStart = argStart->previous();
1005                     if (Token::Match(argStart, "const %type% & %name% [,)]"))
1006                         return 1;
1007                     if ((pointer || address) && alloc == NO_ALLOC && Token::Match(argStart, "const struct| %type% * %name% [,)]"))
1008                         return 1;
1009                     if ((pointer || address) && Token::Match(argStart, "const %type% %name% [") && Token::Match(argStart->linkAt(3), "] [,)]"))
1010                         return 1;
1011                 }
1012     
1013             } else if (Token::Match(start->previous(), "if|while|for")) {
1014                 // control-flow statement reading the variable "by value"
1015                 return alloc == NO_ALLOC;
1016             } else {
1017                 const bool isnullbad = _settings->library.isnullargbad(start->previous(), argumentNumber + 1);
1018                 if (pointer && !address && isnullbad && alloc == NO_ALLOC)
1019                     return true;
1020                 const bool isuninitbad = _settings->library.isuninitargbad(start->previous(), argumentNumber + 1);
1021                 if (alloc != NO_ALLOC)
1022                     return isnullbad && isuninitbad;
1023                 return isuninitbad && (!address || isnullbad);
1024             }
1025         }
1026     
1027         // unknown
1028         return -1;
1029     }

1031     bool CheckUninitVar::isMemberVariableAssignment(const Token *tok, const std::string &membervar) const
1032     {
1033         if (Token::Match(tok, "%name% . %name%") && tok->strAt(2) == membervar) {
1034             if (Token::Match(tok->tokAt(3), "[=.[]"))
1035                 return true;
1036             else if (Token::Match(tok->tokAt(-2), "[(,=] &"))
1037                 return true;
1038             else if (Token::Match(tok->tokAt(-2), "%name% >>") && Token::Match(tok->tokAt(3), ";|>>")) // #6680
1039                 return true;
1040             else if ((tok->previous() && tok->previous()->isConstOp()) || Token::Match(tok->previous(), "[|="))
1041                 ; // member variable usage
1042             else if (tok->tokAt(3)->isConstOp())
1043                 ; // member variable usage
1044             else if (Token::Match(tok->previous(), "[(,] %name% . %name% [,)]") &&
1045                      1 == isFunctionParUsage(tok, false, NO_ALLOC)) {
1046                 return false;
1047             } else
1048                 return true;
1049         } else if (tok->strAt(1) == "=")
1050             return true;
1051         else if (Token::Match(tok, "%var% . %name% (")) {
1052             const Token *ftok = tok->tokAt(2);
1053             if (!ftok->function() || !ftok->function()->isConst())
1054                 // TODO: Try to determine if membervar is assigned in method
1055                 return true;
1056         } else if (tok->strAt(-1) == "&") {
1057             if (Token::Match(tok->tokAt(-2), "[(,] & %name%")) {
1058                 // locate start parentheses in function call..
1059                 unsigned int argumentNumber = 0;
1060                 const Token *ftok = tok;
1061                 while (ftok && !Token::Match(ftok, "[;{}(]")) {
1062                     if (ftok->str() == ")")
1063                         ftok = ftok->link();
1064                     else if (ftok->str() == ",")
1065                         ++argumentNumber;
1066                     ftok = ftok->previous();
1067                 }
1068     
1069                 // is this a function call?
1070                 ftok = ftok ? ftok->previous() : nullptr;
1071                 if (Token::Match(ftok, "%name% (")) {
1072                     // check how function handle uninitialized data arguments..
1073                     const Function *function = ftok->function();
1074                     const Variable *arg      = function ? function->getArgumentVar(argumentNumber) : nullptr;
1075                     const Token *argStart    = arg ? arg->typeStartToken() : nullptr;
1076                     while (argStart && argStart->previous() && argStart->previous()->isName())
1077                         argStart = argStart->previous();
1078                     if (Token::Match(argStart, "const struct| %type% * const| %name% [,)]"))
1079                         return false;
1080                 }
1081     
1082                 else if (ftok && Token::simpleMatch(ftok->previous(), "= * ("))
1083                     return false;
1084             }
1085             return true;
1086         }
1087         return false;
1088     }

1090     bool CheckUninitVar::isMemberVariableUsage(const Token *tok, bool isPointer, Alloc alloc, const std::string &membervar) const
1091     {
1092         if (Token::Match(tok->previous(), "[(,] %name% . %name% [,)]") &&
1093             tok->strAt(2) == membervar) {
1094             int use = isFunctionParUsage(tok, isPointer, alloc);
1095             if (use == 1)
1096                 return true;
1097         }
1098     
1099         if (isMemberVariableAssignment(tok, membervar))
1100             return false;
1101     
1102         if (Token::Match(tok, "%name% . %name%") && tok->strAt(2) == membervar && !(tok->tokAt(-2)->variable() && tok->tokAt(-2)->variable()->isReference()))
1103             return true;
1104         else if (!isPointer && Token::Match(tok->previous(), "[(,] %name% [,)]") && isVariableUsage(tok, isPointer, alloc))
1105             return true;
1106     
1107         else if (!isPointer && Token::Match(tok->previous(), "= %name% ;"))
1108             return true;
1109     
1110         // = *(&var);
1111         else if (!isPointer &&
1112                  Token::simpleMatch(tok->astParent(),"&") &&
1113                  Token::simpleMatch(tok->astParent()->astParent(),"*") &&
1114                  Token::Match(tok->astParent()->astParent()->astParent(), "= * (| &") &&
1115                  tok->astParent()->astParent()->astParent()->astOperand2() == tok->astParent()->astParent())
1116             return true;
1117     
1118         else if (_settings->experimental &&
1119                  !isPointer &&
1120                  Token::Match(tok->tokAt(-2), "[(,] & %name% [,)]") &&
1121                  isVariableUsage(tok, isPointer, alloc))
1122             return true;
1123     
1124         return false;
1125     }

1150     void CheckUninitVar::deadPointer()
1151     {
1152         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
1153         std::list<Scope>::const_iterator scope;
1154     
1155         // check every executable scope
1156         for (scope = symbolDatabase->scopeList.begin(); scope != symbolDatabase->scopeList.end(); ++scope) {
1157             if (!scope->isExecutable())
1158                 continue;
1159             // Dead pointers..
1160             for (const Token* tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
1161                 if (tok->variable() &&
1162                     tok->variable()->isPointer() &&
1163                     isVariableUsage(tok, true, NO_ALLOC)) {
1164                     const Token *alias = tok->getValueTokenDeadPointer();
1165                     if (alias) {
1166                         deadPointerError(tok,alias);
1167                     }
1168                 }
1169             }
1170         }
1171     }

1173     void CheckUninitVar::deadPointerError(const Token *pointer, const Token *alias)
1174     {
1175         const std::string strpointer(pointer ? pointer->str() : std::string("pointer"));
1176         const std::string stralias(alias ? alias->expressionString() : std::string("&x"));
1177     
1178         reportError(pointer,
