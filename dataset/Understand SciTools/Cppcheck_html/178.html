17     #include "options.h"
18     
19     options::options(int argc, const char* argv[])
20         :_options(&argv[1], &argv[0] + argc)
21         ,_which_test("")
22         ,_gcc_style_errors(_options.count("-g") != 0)
23         ,_quiet(_options.count("-q") != 0)
24     {
25         _options.erase("-g");
26         _options.erase("-q");
27         if (! _options.empty()) {
28             _which_test = *_options.rbegin();
29         }
30     }
31     
32     bool options::quiet() const
33     {
34         return _quiet;
35     }
36     
37     bool options::gcc_style_errors() const
38     {
39         return _gcc_style_errors;
40     }
41     
42     const std::string& options::which_test() const
43     {
44         return _which_test;
45     }
