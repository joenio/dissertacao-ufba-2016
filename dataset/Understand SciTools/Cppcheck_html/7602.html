28     /**
29      * @brief This is just a container for standards settings.
30      * This struct contains all possible standards that cppcheck recognize.
31      */
32     struct Standards {
33         /** C code C89/C99/C11 standard */
34         enum cstd_t { C89, C99, C11 } c;
35     
36         /** C++ code standard */
37         enum cppstd_t { CPP03, CPP11 } cpp;
38     
39         /** Code is posix */
40         bool posix;
41     
42         /** This constructor clear all the variables **/
43         Standards() : c(C11), cpp(CPP11), posix(false) {}
44     
45         bool setC(const std::string& str) {
46             if (str == "c89" || str == "C89") {
47                 c = C89;
48                 return true;
49             }
50             if (str == "c99" || str == "C99") {
51                 c = C99;
52                 return true;
53             }
54             if (str == "c11" || str == "C11") {
55                 c = C11;
56                 return true;
57             }
58             return false;
59         }
60         bool setCPP(const std::string& str) {
61             if (str == "c++03" || str == "C++03") {
62                 cpp = CPP03;
63                 return true;
64             }
65             if (str == "c++11" || str == "C++11") {
66                 cpp = CPP11;
67                 return true;
68             }
69             return false;
70         }
71     };
72     
73     /// @}
74     //---------------------------------------------------------------------------
75     #endif // standardsH
