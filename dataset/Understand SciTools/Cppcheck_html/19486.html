29     // Null pointer literal
30     // Source: SC22/WG21/N2431 = J16/07-0301
31     // http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2007/n2431.pdf
32     
33     const                        // this is a const object...
34     class {
35     public:
36         template<class T>          // convertible to any type
37         operator T*() const {    // of null non-member
38             return 0;    // pointer...
39         }
40         template<class C, class T> // or any type of null
41         operator T C::*() const { // member pointer...
42             return 0;
43         }
44     private:
45         void operator&() const;    // whose address can't be taken
46     } cppcheck_nullptr_impl = {};  // and whose name is nullptr
47     
48     // An evil workaround for the inability to disable -Wc++0x-compat using a #pragma.
49     // Once -std=c++0x is embraced, the above class can be renamed to nullptr and
50     // the define can be removed.
51     #define nullptr cppcheck_nullptr_impl
52     
53     // Static assertions
54     
55     #ifndef static_assert
56     #define XXJOIN(x, y) XXJOIN_AGAIN(x, y)
57     #define XXJOIN_AGAIN(x, y) x ## y
58     #define static_assert(e, msg) \
59      typedef char XXJOIN(assertion_failed_at_line_, __LINE__) [(e) ? 1 : -1]
60     #endif // static_assert
61     
62     #endif // __cplusplus < 201103L
63     #endif // __cplusplus
