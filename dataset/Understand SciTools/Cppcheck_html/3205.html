19     #ifndef CPPCHECKEXECUTOR_H
20     #define CPPCHECKEXECUTOR_H
21     
22     #include "errorlogger.h"
23     #include <cstdio>
24     #include <ctime>
25     #include <set>
26     #include <string>
27     
28     class CppCheck;
29     class Settings;
30     class Library;
31     
32     /**
33      * This class works as an example of how CppCheck can be used in external
34      * programs without very little knowledge of the internal parts of the
35      * program itself. If you wish to use cppcheck e.g. as a part of IDE,
36      * just rewrite this class for your needs and possibly use other methods
37      * from CppCheck class instead the ones used here.
38      */
39     class CppCheckExecutor : public ErrorLogger {
40     public:
41         /**
42          * Constructor
43          */
44         CppCheckExecutor();
45     
46         /**
47          * Destructor
48          */
49         virtual ~CppCheckExecutor();
50     
51         /**
52          * Starts the checking.
53          *
54          * @param argc from main()
55          * @param argv from main()
56          * @return EXIT_FAILURE if arguments are invalid or no input files
57          *         were found.
58          *         If errors are found and --error-exitcode is used,
59          *         given value is returned instead of default 0.
60          *         If no errors are found, 0 is returned.
61          */
62         int check(int argc, const char* const argv[]);
63     
64         /**
65          * Information about progress is directed here. This should be
66          * called by the CppCheck class only.
67          *
68          * @param outmsg Progress message e.g. "Checking main.cpp..."
69          */
70         virtual void reportOut(const std::string &outmsg);
71     
72         /** xml output of errors */
73         virtual void reportErr(const ErrorLogger::ErrorMessage &msg);
74     
75         void reportProgress(const std::string &filename, const char stage[], const std::size_t value);
76     
77         /**
78          * Output information messages.
79          */
80         virtual void reportInfo(const ErrorLogger::ErrorMessage &msg);
81     
82         /**
83          * Information about how many files have been checked
84          *
85          * @param fileindex This many files have been checked.
86          * @param filecount This many files there are in total.
87          * @param sizedone The sum of sizes of the files checked.
88          * @param sizetotal The total sizes of the files.
89          */
90         static void reportStatus(std::size_t fileindex, std::size_t filecount, std::size_t sizedone, std::size_t sizetotal);
91     
92         /**
93          * @param fp Output file
94          */
95         static void setExceptionOutput(FILE* fp);
96         /**
97         * @return file name to be used for output from exception handler. Has to be either "stdout" or "stderr".
98         */
99         static FILE* getExceptionOutput();
100     
101         /**
102         * Tries to load a library and prints warning/error messages
103         * @return false, if an error occured (except unknown XML elements)
104         */
105         static bool tryLoadLibrary(Library& destination, const char* basepath, const char* filename);
106     
107     protected:
108     
109         /**
110          * Helper function to print out errors. Appends a line change.
111          * @param errmsg String printed to error stream
112          */
113         void reportErr(const std::string &errmsg);
114     
115         /**
116          * @brief Parse command line args and get settings and file lists
117          * from there.
118          *
119          * @param cppcheck cppcheck instance
120          * @param argc argc from main()
121          * @param argv argv from main()
122          * @return false when errors are found in the input
123          */
124         bool parseFromArgs(CppCheck *cppcheck, int argc, const char* const argv[]);
125     
126     private:
127     
128         /**
129          * Wrapper around check_internal
130          *   - installs optional platform dependent signal handling
131          *
132          * * @param cppcheck cppcheck instance
133         * @param argc from main()
134         * @param argv from main()
135          **/
136         int check_wrapper(CppCheck& cppcheck, int argc, const char* const argv[]);
137     
138         /**
139         * Starts the checking.
140         *
141         * @param cppcheck cppcheck instance
142         * @param argc from main()
143         * @param argv from main()
144         * @return EXIT_FAILURE if arguments are invalid or no input files
145         *         were found.
146         *         If errors are found and --error-exitcode is used,
147         *         given value is returned instead of default 0.
148         *         If no errors are found, 0 is returned.
149         */
150         int check_internal(CppCheck& cppcheck, int argc, const char* const argv[]);
151     
152         /**
153          * Pointer to current settings; set while check() is running.
154          */
155         const Settings* _settings;
156     
157         /**
158          * Used to filter out duplicate error messages.
159          */
160         std::set<std::string> _errorList;
161     
162         /**
163          * Filename associated with size of file
164          */
165         std::map<std::string, std::size_t> _files;
166     
167         /**
168          * Report progress time
169          */
170         std::time_t time1;
171     
172         /**
173          * Output file name for exception handler
174          */
175         static FILE* exceptionOutput;
176     
177         /**
178          * Has --errorlist been given?
179          */
180         bool errorlist;
181     };
182     
183     #endif // CPPCHECKEXECUTOR_H
