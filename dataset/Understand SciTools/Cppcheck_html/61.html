31     static std::string builddir(std::string filename)
32     {
33         if (filename.compare(0,4,"lib/") == 0)
34             filename = "$(SRCDIR)" + filename.substr(3);
35         return filename;
36     }

38     static std::string objfile(std::string cppfile)
39     {
40         cppfile.erase(cppfile.rfind("."));
41         return builddir(cppfile + ".o");
42     }

44     static void getDeps(const std::string &filename, std::vector<std::string> &depfiles)
45     {
46         // Is the dependency already included?
47         if (std::find(depfiles.begin(), depfiles.end(), filename) != depfiles.end())
48             return;
49     
50         std::ifstream f(filename.c_str());
51         if (! f.is_open()) {
52             if (filename.compare(0, 4, "cli/") == 0 || filename.compare(0, 5, "test/") == 0)
53                 getDeps("lib" + filename.substr(filename.find("/")), depfiles);
54             return;
55         }
56         if (filename.find(".c") == std::string::npos)
57             depfiles.push_back(filename);
58     
59         std::string path(filename);
60         if (path.find("/") != std::string::npos)
61             path.erase(1 + path.rfind("/"));
62     
63         std::string line;
64         while (std::getline(f, line)) {
65             std::string::size_type pos1 = line.find("#include \"");
66             if (pos1 == std::string::npos)
67                 continue;
68             pos1 += 10;
69     
70             std::string::size_type pos2 = line.find("\"", pos1);
71             std::string hfile(path + line.substr(pos1, pos2 - pos1));
72             if (hfile.find("/../") != std::string::npos)    // TODO: Ugly fix
73                 hfile.erase(0, 4 + hfile.find("/../"));
74             getDeps(hfile, depfiles);
75         }
76     }

78     static void compilefiles(std::ostream &fout, const std::vector<std::string> &files, const std::string &args)
79     {
80         for (unsigned int i = 0; i < files.size(); ++i) {
81             fout << objfile(files[i]) << ": " << files[i];
82             std::vector<std::string> depfiles;
83             depfiles.push_back("lib/cxx11emu.h");
84             getDeps(files[i], depfiles);
85             for (unsigned int dep = 0; dep < depfiles.size(); ++dep)
86                 fout << " " << depfiles[dep];
87             fout << "\n\t$(CXX) " << args << " $(CPPFLAGS) $(CFG) $(CXXFLAGS) $(UNDEF_STRICT_ANSI) -c -o " << objfile(files[i]) << " " << builddir(files[i]) << "\n\n";
88         }
89     }

91     static void getCppFiles(std::vector<std::string> &files, const std::string &path, bool recursive)
92     {
93         std::map<std::string,size_t> filemap;
94         const std::set<std::string> extra;
95         const std::vector<std::string> masks;
96         const PathMatch matcher(masks);
97         FileLister::addFiles(filemap, path, extra, recursive, matcher);
98     
99         // add *.cpp files to the "files" vector..
100         for (std::map<std::string,size_t>::const_iterator it = filemap.begin(); it != filemap.end(); ++it) {
101             if (it->first.find(".cpp") != std::string::npos)
102                 files.push_back(it->first);
103         }
104     }

107     static void makeConditionalVariable(std::ostream &os, const std::string &variable, const std::string &defaultValue)
108     {
109         os << "ifndef " << variable << '\n'
110            << "    " << variable << '=' << defaultValue << '\n'
111            << "endif\n"
112            << "\n";
113     }

115     static std::string getLibName(const std::string &path)
116     {
117         // path can be e.g. "externals/foo/foo.cpp" then returned
118         // library name is "FOO".
119         std::string libName = path.substr(path.find('/')+1);
120         libName = libName.substr(0, libName.find('/'));
121         std::transform(libName.begin(), libName.end(),libName.begin(), ::toupper);
122         return libName;
123     }

125     static void makeExtObj(std::ostream &fout, const std::vector<std::string> &externalfiles)
126     {
127         bool start = true;
128         std::ostringstream libNames;
129         std::string libName;
130         for (std::size_t i = 0; i < externalfiles.size(); ++i) {
131             if (start) {
132                 libName = getLibName(externalfiles[i]);
133                 fout << "ifndef " << libName << "\n";
134                 fout << "    " << libName << " = " << objfile(externalfiles[i]);
135                 libNames << "EXTOBJ += $(" << libName << ")\n";
136                 start = false;
137             } else {
138                 fout << std::string(14, ' ') << objfile(externalfiles[i]);
139             }
140     
141             if (i+1 >= externalfiles.size() || libName != getLibName(externalfiles[i+1])) {
142                 // This was the last file for this library
143                 fout << "\nendif\n\n\n";
144                 start = true;
145             } else {
146                 // There are more files for this library
147                 fout << " \\\n";
148             }
149         }
150     
151         fout << libNames.str();
152     }

154     int main(int argc, char **argv)
155     {
156         const bool release(argc >= 2 && std::string(argv[1]) == "--release");
157     
158         // Get files..
159         std::vector<std::string> libfiles;
160         getCppFiles(libfiles, "lib/", false);
161     
162         std::vector<std::string> clifiles;
163         getCppFiles(clifiles, "cli/", false);
164     
165         std::vector<std::string> testfiles;
166         getCppFiles(testfiles, "test/", false);
167     
168         std::vector<std::string> toolsfiles;
169         getCppFiles(toolsfiles, "tools/", false);
170     
171         if (libfiles.empty() && clifiles.empty() && testfiles.empty()) {
172             std::cerr << "No files found. Are you in the correct directory?" << std::endl;
173             return EXIT_FAILURE;
174         }
175     
176         std::vector<std::string> externalfiles;
177         getCppFiles(externalfiles, "externals/", true);
178     
179     
180         // QMAKE - lib/lib.pri
181         {
182             std::ofstream fout1("lib/lib.pri");
183             if (fout1.is_open()) {
184                 fout1 << "# no manual edits - this file is autogenerated by dmake\n\n";
185                 fout1 << "include($$PWD/pcrerules.pri)\n";
186                 fout1 << "include($$PWD/../externals/tinyxml/tinyxml.pri)\n";
187                 fout1 << "INCLUDEPATH += $$PWD $$PWD/../externals/tinyxml\n";
188                 fout1 << "HEADERS += $${PWD}/check.h \\\n";
189                 for (unsigned int i = 0; i < libfiles.size(); ++i) {
190                     std::string fname(libfiles[i].substr(4));
191                     if (fname.find(".cpp") == std::string::npos)
192                         continue;   // shouldn't happen
193                     fname.erase(fname.find(".cpp"));
194                     fout1 << std::string(11, ' ') << "$${PWD}/" << fname << ".h";
195                     if (i + 1 < testfiles.size())
196                         fout1 << " \\\n";
197                 }
198                 fout1 << "\n\nSOURCES += ";
199                 for (unsigned int i = 0; i < libfiles.size(); ++i) {
200                     fout1 << "$${PWD}/" << libfiles[i].substr(4);
201                     if (i < libfiles.size() - 1)
202                         fout1 << " \\\n" << std::string(11, ' ');
203                 }
204                 fout1 << "\n";
205             }
206         }
207     
208         // QMAKE - test/testfiles.pri
209         {
210             std::ofstream fout1("test/testfiles.pri");
211             if (fout1.is_open()) {
212                 fout1 << "# no manual edits - this file is autogenerated by dmake\n\n";
213                 fout1 << "INCLUDEPATH += ../externals/tinyxml\n";
214                 fout1 << "\n\nSOURCES += ";
215                 for (unsigned int i = 0; i < testfiles.size(); ++i) {
216                     const std::string filename(testfiles[i].substr(5));
217                     // Include only files containing tests in this listing.
218                     // I.e. filenames beginning with "test".
219                     if (filename.compare(0, 4, "test") == 0) {
220                         fout1 << "$${BASEPATH}/" << filename;
221                         if (i + 1 < testfiles.size())
222                             fout1 << " \\\n" << std::string(11, ' ');
223                     }
224                 }
225                 fout1 << "\n";
226             }
227         }
228     
229         static const char makefile[] = "Makefile";
230         std::ofstream fout(makefile, std::ios_base::trunc);
231         if (!fout.is_open()) {
232             std::cerr << "An error occurred while trying to open "
233                       << makefile
234                       << ".\n";
235             return EXIT_FAILURE;
236         }
237     
238         fout << "# This file is generated by tools/dmake, do not edit.\n\n";
239         fout << "# To compile with rules, use 'make HAVE_RULES=yes'\n";
240         makeConditionalVariable(fout, "HAVE_RULES", "no");
241     
242         // compiled patterns..
243         fout << "# folder where lib/*.cpp files are located\n";
244         makeConditionalVariable(fout, "SRCDIR", "lib");
245         fout << "ifeq ($(SRCDIR),build)\n"
246              << "    ifdef VERIFY\n"
247              << "        matchcompiler_S := $(shell python tools/matchcompiler.py --verify)\n"
248              << "    else\n"
249              << "        matchcompiler_S := $(shell python tools/matchcompiler.py)\n"
250              << "    endif\n"
251              << "endif\n\n";
252     
253         // explicit cfg dir..
254         fout << "ifdef CFGDIR\n"
255              << "    CFG=-DCFGDIR=\\\"$(CFGDIR)\\\"\n"
256              << "else\n"
257              << "    CFG=\n"
258              << "endif\n\n";
259     
260         // enable backtrac
261         fout << "RDYNAMIC=-rdynamic\n";
262     
263         // The _GLIBCXX_DEBUG doesn't work in cygwin or other Win32 systems.
264         fout << "# Set the CPPCHK_GLIBCXX_DEBUG flag. This flag is not used in release Makefiles.\n"
265              << "# The _GLIBCXX_DEBUG define doesn't work in Cygwin or other Win32 systems.\n"
266              << "ifndef COMSPEC\n"
267              << "    ifdef ComSpec\n"
268              << "        #### ComSpec is defined on some WIN32's.\n"
269              << "        COMSPEC=$(ComSpec)\n"
270              << "    endif # ComSpec\n"
271              << "endif # COMSPEC\n"
272              << "\n"
273              << "ifdef COMSPEC\n"
274              << "    #### Maybe Windows\n"
275              << "    ifndef CPPCHK_GLIBCXX_DEBUG\n"
276              << "        CPPCHK_GLIBCXX_DEBUG=\n"
277              << "    endif # !CPPCHK_GLIBCXX_DEBUG\n"
278              << "\n"
279              << "    ifeq ($(MSYSTEM),MINGW32)\n"
280              << "        LDFLAGS=-lshlwapi\n"
281              << "    else\n"
282              << "        RDYNAMIC=-lshlwapi\n"
283              << "    endif\n"
284              << "else # !COMSPEC\n"
285              << "    uname_S := $(shell sh -c 'uname -s 2>/dev/null || echo not')\n"
286              << "\n"
287              << "    ifeq ($(uname_S),Linux)\n"
288              << "        ifndef CPPCHK_GLIBCXX_DEBUG\n"
289              << "            CPPCHK_GLIBCXX_DEBUG=-D_GLIBCXX_DEBUG\n"
290              << "        endif # !CPPCHK_GLIBCXX_DEBUG\n"
291              << "    endif # Linux\n"
292              << "\n"
293              << "    ifeq ($(uname_S),GNU/kFreeBSD)\n"
294              << "        ifndef CPPCHK_GLIBCXX_DEBUG\n"
295              << "            CPPCHK_GLIBCXX_DEBUG=-D_GLIBCXX_DEBUG\n"
296              << "        endif # !CPPCHK_GLIBCXX_DEBUG\n"
297              << "    endif # GNU/kFreeBSD\n"
298              << "\n"
299              << "endif # COMSPEC\n"
300              << "\n";
301     
302         // tinymxl2 requires __STRICT_ANSI__ to be undefined to compile under CYGWIN.
303         fout << "# Set the UNDEF_STRICT_ANSI flag to address compile time warnings\n"
304              << "# with tinyxml2 and Cygwin.\n"
305              << "ifdef COMSPEC\n"
306              << "    uname_S := $(shell uname -s)\n"
307              << "\n"
308              << "    ifneq (,$(findstring CYGWIN,$(uname_S)))\n"
309              << "        UNDEF_STRICT_ANSI=-U__STRICT_ANSI__\n"
310              << "    endif # CYGWIN\n"
311              << "endif # COMSPEC\n"
312              << "\n";
313     
314         // skip "-D_GLIBCXX_DEBUG" if clang, since it breaks the build
315         makeConditionalVariable(fout, "CXX", "g++");
316         fout << "ifeq ($(CXX), clang++)\n"
317              << "    CPPCHK_GLIBCXX_DEBUG=\n"
318              << "endif\n";
319     
320         // Makefile settings..
321         if (release) {
322             makeConditionalVariable(fout, "CXXFLAGS", "-std=c++0x -O2 -include lib/cxx11emu.h -DNDEBUG -Wall -Wno-sign-compare");
323         } else {
324             // TODO: add more compiler warnings.
325             // -Wlogical-op       : doesn't work on older GCC
326             // -Wsign-conversion  : too many warnings
327             // -Wunreachable-code : some GCC versions report lots of warnings
328             makeConditionalVariable(fout, "CXXFLAGS",
329                                     "-include lib/cxx11emu.h "
330                                     "-pedantic "
331                                     "-Wall "
332                                     "-Wextra "
333                                     "-Wabi "
334                                     "-Wcast-qual "
335     //                                "-Wconversion "  // danmar: gives fp. for instance: unsigned int sizeof_pointer = sizeof(void *);
336                                     "-Wfloat-equal "
337                                     "-Winline "
338     //                                "-Wlogical-op "
339                                     "-Wmissing-declarations "
340                                     "-Wmissing-format-attribute "
341                                     "-Wno-long-long "
342     //                                "-Woverloaded-virtual "  // danmar: we get fp when overloading analyseWholeProgram()
343                                     "-Wpacked "
344                                     "-Wredundant-decls "
345                                     "-Wshadow "
346     //                                "-Wsign-conversion "
347                                     "-Wsign-promo "
348                                     "-Wno-missing-field-initializers "
349                                     "-Wno-missing-braces "
350     //                                "-Wunreachable-code "
351                                     "-Wno-sign-compare "  // danmar: I don't like this warning, it's very rarelly a bug
352                                     "-Wno-multichar "
353                                     "$(CPPCHK_GLIBCXX_DEBUG) "
354                                     "-g");
355         }
356     
357         fout << "ifeq ($(CXX), g++)\n"
358              << "    override CXXFLAGS += -std=c++0x\n"
359              << "else ifeq ($(CXX), clang++)\n"
360              << "    override CXXFLAGS += -std=c++0x\n"
361              << "else ifeq ($(CXX), c++)\n"
362              << "    ifeq ($(shell uname -s), Darwin)\n"
363              << "        override CXXFLAGS += -std=c++0x\n"
364              << "    endif\n"
365              << "endif\n"
366              << "\n";
367     
368         fout << "ifeq ($(HAVE_RULES),yes)\n"
369              << "    override CXXFLAGS += -DHAVE_RULES -DTIXML_USE_STL $(shell pcre-config --cflags)\n"
370              << "    ifdef LIBS\n"
371              << "        LIBS += $(shell pcre-config --libs)\n"
372              << "    else\n"
373              << "        LIBS=$(shell pcre-config --libs)\n"
374              << "    endif\n"
375              << "endif\n\n";
376     
377         makeConditionalVariable(fout, "PREFIX", "/usr");
378         makeConditionalVariable(fout, "INCLUDE_FOR_LIB", "-Ilib -Iexternals/tinyxml");
379         makeConditionalVariable(fout, "INCLUDE_FOR_CLI", "-Ilib -Iexternals/tinyxml");
380         makeConditionalVariable(fout, "INCLUDE_FOR_TEST", "-Ilib -Icli -Iexternals/tinyxml");
381     
382         fout << "BIN=$(DESTDIR)$(PREFIX)/bin\n\n";
383         fout << "# For 'make man': sudo apt-get install xsltproc docbook-xsl docbook-xml on Linux\n";
384         fout << "DB2MAN?=/usr/share/sgml/docbook/stylesheet/xsl/nwalsh/manpages/docbook.xsl\n";
385         fout << "XP=xsltproc -''-nonet -''-param man.charmap.use.subset \"0\"\n";
386         fout << "MAN_SOURCE=man/cppcheck.1.xml\n\n";
387     
388         fout << "\n###### Object Files\n\n";
389         fout << "LIBOBJ =      " << objfile(libfiles[0]);
390         for (size_t i = 1; i < libfiles.size(); ++i)
391             fout << " \\\n" << std::string(14, ' ') << objfile(libfiles[i]);
392         fout << "\n\n";
393         fout << "CLIOBJ =      " << objfile(clifiles[0]);
394         for (size_t i = 1; i < clifiles.size(); ++i)
395             fout << " \\\n" << std::string(14, ' ') << objfile(clifiles[i]);
396         fout << "\n\n";
397         fout << "TESTOBJ =     " << objfile(testfiles[0]);
398         for (size_t i = 1; i < testfiles.size(); ++i)
399             fout << " \\\n" << std::string(14, ' ') << objfile(testfiles[i]);
400         fout << "\n\n";
401     
402         makeExtObj(fout, externalfiles);
403     
404         fout << ".PHONY: run-dmake\n\n";
405         fout << "\n###### Targets\n\n";
406         fout << "cppcheck: $(LIBOBJ) $(CLIOBJ) $(EXTOBJ)\n";
407         fout << "\t$(CXX) $(CPPFLAGS) $(CXXFLAGS) -o cppcheck $(CLIOBJ) $(LIBOBJ) $(EXTOBJ) $(LIBS) $(LDFLAGS) $(RDYNAMIC)\n\n";
408         fout << "all:\tcppcheck testrunner\n\n";
409         fout << "testrunner: $(TESTOBJ) $(LIBOBJ) $(EXTOBJ) cli/threadexecutor.o cli/cmdlineparser.o cli/cppcheckexecutor.o cli/filelister.o cli/pathmatch.o\n";

411         fout << "test:\tall\n";
412         fout << "\t./testrunner\n\n";
413         fout << "check:\tall\n";
414         fout << "\t./testrunner -g -q\n\n";
415         fout << "checkcfg:\tcppcheck\n";
416         fout << "\t./test/cfg/runtests.sh\n\n";

418         fout << "\t$(CXX) $(CXXFLAGS) -o dmake tools/dmake.o cli/filelister.o cli/pathmatch.o lib/path.o -Ilib $(LDFLAGS)\n\n";
419         fout << "run-dmake: dmake\n";
420         fout << "\t./dmake\n\n";
421         fout << "reduce:\ttools/reduce.o externals/tinyxml/tinyxml2.o $(LIBOBJ)\n";
422         fout << "\t$(CXX) $(CPPFLAGS) $(CXXFLAGS) -g -o reduce tools/reduce.o -Ilib -Iexternals/tinyxml $(LIBOBJ) $(LIBS) externals/tinyxml/tinyxml2.o $(LDFLAGS) $(RDYNAMIC)\n\n";
423         fout << "clean:\n";
424         fout << "\trm -f build/*.o lib/*.o cli/*.o test/*.o tools/*.o externals/tinyxml/*.o testrunner reduce dmake cppcheck cppcheck.1\n\n";
425         fout << "man:\tman/cppcheck.1\n\n";
426         fout << "man/cppcheck.1:\t$(MAN_SOURCE)\n\n";
427         fout << "\t$(XP) $(DB2MAN) $(MAN_SOURCE)\n\n";
428         fout << "tags:\n";
429         fout << "\tctags -R --exclude=doxyoutput --exclude=test/cfg cli externals gui lib test\n\n";
430         fout << "install: cppcheck\n";
431         fout << "\tinstall -d ${BIN}\n";
432         fout << "\tinstall cppcheck ${BIN}\n";
433         fout << "\tinstall htmlreport/cppcheck-htmlreport ${BIN}\n";
434         fout << "ifdef CFGDIR \n";
435         fout << "\tinstall -d ${DESTDIR}${CFGDIR}\n";
436         fout << "\tinstall -m 644 cfg/* ${DESTDIR}${CFGDIR}\n";
437         fout << "endif\n\n";
438     
439         fout << "\n###### Build\n\n";
440     
441         compilefiles(fout, libfiles, "${INCLUDE_FOR_LIB}");
442         compilefiles(fout, clifiles, "${INCLUDE_FOR_CLI}");
443         compilefiles(fout, testfiles, "${INCLUDE_FOR_TEST}");
444         compilefiles(fout, externalfiles, "${INCLUDE_FOR_LIB}");
445         compilefiles(fout, toolsfiles, "${INCLUDE_FOR_LIB}");
446     
447         return 0;
448     }
