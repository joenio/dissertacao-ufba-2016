24     #include "config.h"
25     #include "check.h"

27     /// @addtogroup Checks
28     /// @{

32     /**
33      * @brief %Check exception safety (exceptions shouldn't cause leaks nor corrupt data)
34      *
35      * The problem with these checks is that Cppcheck can't determine what the valid
36      * values are for variables. But in some cases (dead pointers) it can be determined
37      * that certain variable values are corrupt.
38      */
39     
40     class CPPCHECKLIB CheckExceptionSafety : public Check {
41     public:
42         /** This constructor is used when registering the CheckClass */
43         CheckExceptionSafety() : Check(myName()) {
44         }
45     
46         /** This constructor is used when running checks. */
47         CheckExceptionSafety(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
48             : Check(myName(), tokenizer, settings, errorLogger) {
49         }
50     
51         /** Checks that uses the simplified token list */
52         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
53             if (tokenizer->isC())
54                 return;
55     
56             CheckExceptionSafety checkExceptionSafety(tokenizer, settings, errorLogger);
57             checkExceptionSafety.destructors();
58             checkExceptionSafety.deallocThrow();
59             checkExceptionSafety.checkRethrowCopy();
60             checkExceptionSafety.checkCatchExceptionByValue();
61             checkExceptionSafety.nothrowThrows();
62             checkExceptionSafety.unhandledExceptionSpecification();
63         }
64     
65         /** Don't throw exceptions in destructors */
66         void destructors();
67     
68         /** deallocating memory and then throw (dead pointer) */
69         void deallocThrow();
70     
71         /** Don't rethrow a copy of the caught exception; use a bare throw instead */
72         void checkRethrowCopy();
73     
74         /** @brief %Check for exceptions that are caught by value instead of by reference */
75         void checkCatchExceptionByValue();
76     
77         /** @brief %Check for functions that throw that shouldn't */
78         void nothrowThrows();
79     
80         /** @brief %Check for unhandled exception specification */
81         void unhandledExceptionSpecification();
82     
83     private:
84         /** Don't throw exceptions in destructors */
85         void destructorsError(const Token * const tok, const std::string &className) {
86             reportError(tok, Severity::warning, "exceptThrowInDestructor",
87                         "Class " + className + " is not safe, destructor throws exception\n"
88                         "The class " + className + " is not safe because its destructor "
89                         "throws an exception. If " + className + " is used and an exception "
90                         "is thrown that is caught in an outer scope the program will terminate.");
91         }
92     
93         void deallocThrowError(const Token * const tok, const std::string &varname) {
94             reportError(tok, Severity::warning, "exceptDeallocThrow", "Exception thrown in invalid state, '" +
95                         varname + "' points at deallocated memory.");
96         }
97     
98         void rethrowCopyError(const Token * const tok, const std::string &varname) {
99             reportError(tok, Severity::style, "exceptRethrowCopy",
100                         "Throwing a copy of the caught exception instead of rethrowing the original exception.\n"
101                         "Rethrowing an exception with 'throw " + varname + ";' creates an unnecessary copy of '" + varname + "'. "
102                         "To rethrow the caught exception without unnecessary copying or slicing, use a bare 'throw;'.");
103         }
104     
105         void catchExceptionByValueError(const Token *tok) {
106             reportError(tok, Severity::style,
107                         "catchExceptionByValue", "Exception should be caught by reference.\n"
108                         "The exception is caught by value. It could be caught "
109                         "as a (const) reference which is usually recommended in C++.");
110         }
111     
112         void noexceptThrowError(const Token * const tok) {
113             reportError(tok, Severity::error, "throwInNoexceptFunction", "Exception thrown in function declared not to throw exceptions.");
114         }
115     
116         /** Missing exception specification */
117         void unhandledExceptionSpecificationError(const Token * const tok1, const Token * const tok2, const std::string & funcname) {
118             std::string str1(tok1 ? tok1->str() : "foo");
119             std::list<const Token*> locationList;
120             locationList.push_back(tok1);
121             locationList.push_back(tok2);
122             reportError(locationList, Severity::style, "unhandledExceptionSpecification",
123                         "Unhandled exception specification when calling function " + str1 + "().\n"
124                         "Unhandled exception specification when calling function " + str1 + "(). "
125                         "Either use a try/catch around the function call, or add a exception specification for " + funcname + "() also.", 0U, true);
126         }
127     
128         /** Generate all possible errors (for --errorlist) */
129         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
130             CheckExceptionSafety c(0, settings, errorLogger);
131             c.destructorsError(0, "Class");
132             c.deallocThrowError(0, "p");
133             c.rethrowCopyError(0, "varname");
134             c.catchExceptionByValueError(0);
135             c.noexceptThrowError(0);
136             c.unhandledExceptionSpecificationError(0, 0, "funcname");
137         }
138     
139         /** Short description of class (for --doc) */
140         static std::string myName() {
141             return "Exception Safety";
142         }
143     
144         /** wiki formatted description of the class (for --doc) */
145         std::string classInfo() const {
146             return "Checking exception safety\n"
147                    "- Throwing exceptions in destructors\n"
148                    "- Throwing exception during invalid state\n"
149                    "- Throwing a copy of a caught exception instead of rethrowing the original exception\n"
150                    "- Exception caught by value instead of by reference\n"
151                    "- Throwing exception in noexcept, nothrow(), __attribute__((nothrow)) or __declspec(nothrow) function\n"
152                    "- Unhandled exception specification when calling function foo()\n";
153         }
154     };
155     /// @}
156     //---------------------------------------------------------------------------
157     #endif // checkexceptionsafetyH
