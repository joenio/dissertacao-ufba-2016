19     #ifndef PROJECT_FILE_H
20     #define PROJECT_FILE_H
21     
22     #include <QObject>
23     #include <QString>
24     #include <QStringList>
25     #include <QXmlStreamReader>
26     
27     /// @addtogroup GUI
28     /// @{
29     
30     
31     /**
32     * @brief A class that reads and writes project files.
33     * The project files contain project-specific settings for checking. For
34     * example a list of include paths.
35     */
36     class ProjectFile : public QObject {
37         Q_OBJECT
38     
39     public:
40         explicit ProjectFile(QObject *parent = 0);
41         ProjectFile(const QString &filename, QObject *parent = 0);
42     
43         /**
44         * @brief Read the project file.
45         * @param filename Filename (can be also given to constructor).
46         */
47         bool Read(const QString &filename = QString());
48     
49         /**
50         * @brief Get project root path.
51         * @return project root path.
52         */
53         QString GetRootPath() const {
54             return mRootPath;
55         }
56     
57         /**
58         * @brief Get list of include directories.
59         * @return list of directories.
60         */
61         QStringList GetIncludeDirs() const;
62     
63         /**
64         * @brief Get list of defines.
65         * @return list of defines.
66         */
67         QStringList GetDefines() const;
68     
69         /**
70         * @brief Get list of paths to check.
71         * @return list of paths.
72         */
73         QStringList GetCheckPaths() const;
74     
75         /**
76         * @brief Get list of paths to exclude from the check.
77         * @return list of paths.
78         */
79         QStringList GetExcludedPaths() const;
80     
81         /**
82         * @brief Get list libraries.
83         * @return list of libraries.
84         */
85         QStringList GetLibraries() const;
86     
87         /**
88         * @brief Get list suppressions.
89         * @return list of suppressions.
90         */
91         QStringList GetSuppressions() const;
92     
93         /**
94         * @brief Get filename for the project file.
95         * @return file name.
96         */
97         QString GetFilename() const {
98             return mFilename;
99         }
100     
101         /**
102         * @brief Set project root path.
103         * @param rootpath new project root path.
104         */
105         void SetRootPath(const QString &rootpath) {
106             mRootPath = rootpath;
107         }
108     
109         /**
110         * @brief Set list of includes.
111         * @param includes List of defines.
112         */
113         void SetIncludes(const QStringList &includes);
114     
115         /**
116         * @brief Set list of defines.
117         * @param defines List of defines.
118         */
119         void SetDefines(const QStringList &defines);
120     
121         /**
122         * @brief Set list of paths to check.
123         * @param paths List of paths.
124         */
125         void SetCheckPaths(const QStringList &paths);
126     
127         /**
128         * @brief Set list of paths to exclude from the check.
129         * @param paths List of paths.
130         */
131         void SetExcludedPaths(const QStringList &paths);
132     
133         /**
134         * @brief Set list of libraries.
135         * @param libraries List of libraries.
136         */
137         void SetLibraries(const QStringList &libraries);
138     
139         /**
140         * @brief Set list of suppressions.
141         * @param suppressions List of suppressions.
142         */
143         void SetSuppressions(const QStringList &suppressions);
144     
145         /**
146         * @brief Write project file (to disk).
147         * @param filename Filename to use.
148         */
149         bool Write(const QString &filename = QString());
150     
151         /**
152          * @brief Write string list
153          * @param xmlWriter xml writer
154          * @param stringlist string list to write
155          * @param startelementname name of start element
156          * @param stringelementname name of each string element
157          */
158         static void WriteStringList(QXmlStreamWriter &xmlWriter, const QStringList &stringlist, const char startelementname[], const char stringelementname[]);
159     
160         /**
161         * @brief Set filename for the project file.
162         * @param filename Filename to use.
163         */
164         void SetFilename(const QString &filename) {
165             mFilename = filename;
166         }
167     
168     protected:
169     
170         /**
171         * @brief Read optional root path from XML.
172         * @param reader XML stream reader.
173         */
174         void ReadRootPath(QXmlStreamReader &reader);
175     
176         /**
177         * @brief Read list of include directories from XML.
178         * @param reader XML stream reader.
179         */
180         void ReadIncludeDirs(QXmlStreamReader &reader);
181     
182         /**
183         * @brief Read list of defines from XML.
184         * @param reader XML stream reader.
185         */
186         void ReadDefines(QXmlStreamReader &reader);
187     
188         /**
189         * @brief Read list paths to check.
190         * @param reader XML stream reader.
191         */
192         void ReadCheckPaths(QXmlStreamReader &reader);
193     
194         /**
195         * @brief Read lists of excluded paths.
196         * @param reader XML stream reader.
197         */
198         void ReadExcludes(QXmlStreamReader &reader);
199     
200         /**
201          * @brief Read string list
202          * @param stringlist   destination string list
203          * @param reader       XML stream reader
204          * @param elementname  elementname for each string
205          */
206         void ReadStringList(QStringList &stringlist, QXmlStreamReader &reader, const char elementname[]);
207     
208     private:
209     
210         /**
211         * @brief Filename (+path) of the project file.
212         */
213         QString mFilename;
214     
215         /**
216         * @brief Root path (optional) for the project.
217         * This is the project root path. If it is present then all relative paths in
218         * the project file are relative to this path. Otherwise paths are relative
219         * to project file's path.
220         */
221         QString mRootPath;
222     
223         /**
224         * @brief List of include directories used to search include files.
225         */
226         QStringList mIncludeDirs;
227     
228         /**
229         * @brief List of defines.
230         */
231         QStringList mDefines;
232     
233         /**
234         * @brief List of paths to check.
235         */
236         QStringList mPaths;
237     
238         /**
239         * @brief Paths excluded from the check.
240         */
241         QStringList mExcludedPaths;
242     
243         /**
244         * @brief List of libraries.
245         */
246         QStringList mLibraries;
247     
248         /**
249          * @brief List of suppressions.
250          */
251         QStringList mSuppressions;
252     };
253     /// @}
254     #endif  // PROJECT_FILE_H
