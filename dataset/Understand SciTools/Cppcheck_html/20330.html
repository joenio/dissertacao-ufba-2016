76     void CheckClass::constructors()
77     {
78         const bool printStyle = _settings->isEnabled("style");
79         const bool printWarnings = _settings->isEnabled("warning");
80         if (!printStyle && !printWarnings)
81             return;
82     
83         const bool printInconclusive = _settings->inconclusive;
84         const std::size_t classes = symbolDatabase->classAndStructScopes.size();
85         for (std::size_t i = 0; i < classes; ++i) {
86             const Scope * scope = symbolDatabase->classAndStructScopes[i];
87     
88             // There are no constructors.
89             if (scope->numConstructors == 0 && printStyle) {
90                 // If there is a private variable, there should be a constructor..
91                 std::list<Variable>::const_iterator var;
92                 for (var = scope->varlist.begin(); var != scope->varlist.end(); ++var) {
93                     if (var->isPrivate() && !var->isStatic() && !Token::Match(var->nameToken(), "%varid% ; %varid% =", var->declarationId()) &&
94                         (!var->isClass() || (var->type() && var->type()->needInitialization == Type::True))) {
95                         noConstructorError(scope->classDef, scope->className, scope->classDef->str() == "struct");
96                         break;
97                     }
98                 }
99             }
100     
101             if (!printWarnings)
102                 continue;
103     
104             // #3196 => bailout if there are nested unions
105             // TODO: handle union variables better
106             {
107                 bool bailout = false;
108                 for (std::list<Scope *>::const_iterator it = scope->nestedList.begin(); it != scope->nestedList.end(); ++it) {
109                     const Scope * const nestedScope = *it;
110                     if (nestedScope->type == Scope::eUnion) {
111                         bailout = true;
112                         break;
113                     }
114                 }
115                 if (bailout)
116                     continue;
117             }
118     
119     
120             std::list<Function>::const_iterator func;
121             std::vector<Usage> usage(scope->varlist.size());
122     
123             for (func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
124                 if (!func->hasBody() || !(func->isConstructor() ||
125                                           func->type == Function::eOperatorEqual))
126                     continue;
127     
128                 // Mark all variables not used
129                 clearAllVar(usage);
130     
131                 std::list<const Function *> callstack;
132                 initializeVarList(*func, callstack, scope, usage);
133     
134                 // Check if any variables are uninitialized
135                 std::list<Variable>::const_iterator var;
136                 unsigned int count = 0;
137                 for (var = scope->varlist.begin(); var != scope->varlist.end(); ++var, ++count) {
138                     // check for C++11 initializer
139                     if (var->hasDefault()) {
140                         usage[count].init = true;
141                         continue;
142                     }
143     
144                     if (usage[count].assign || usage[count].init || var->isStatic())
145                         continue;
146     
147                     if (var->isConst() && func->isOperator()) // We can't set const members in assignment operator
148                         continue;
149     
150                     // Check if this is a class constructor
151                     if (!var->isPointer() && !var->isPointerArray() && var->isClass() && func->type == Function::eConstructor) {
152                         // Unknown type so assume it is initialized
153                         if (!var->type())
154                             continue;
155     
156                         // Known type that doesn't need initialization or
157                         // known type that has member variables of an unknown type
158                         else if (var->type()->needInitialization != Type::True)
159                             continue;
160                     }
161     
162                     // Check if type can't be copied
163                     if (!var->isPointer() && !var->isPointerArray() && var->typeScope()) {
164                         if (func->type == Function::eMoveConstructor) {
165                             if (canNotMove(var->typeScope()))
166                                 continue;
167                         } else {
168                             if (canNotCopy(var->typeScope()))
169                                 continue;
170                         }
171                     }
172     
173                     bool inconclusive = false;
174                     // Don't warn about unknown types in copy constructors since we
175                     // don't know if they can be copied or not..
176                     if (!var->isPointer() &&
177                         !(var->type() && var->type()->needInitialization != Type::True) &&
178                         (func->type == Function::eCopyConstructor || func->type == Function::eOperatorEqual)) {
179                         if (!var->typeStartToken()->isStandardType()) {
180                             if (printInconclusive)
181                                 inconclusive = true;
182                             else
183                                 continue;
184                         }
185                     }
186     
187                     // It's non-static and it's not initialized => error
188                     if (func->type == Function::eOperatorEqual) {
189                         const Token *operStart = func->arg;
190     
191                         bool classNameUsed = false;
192                         for (const Token *operTok = operStart; operTok != operStart->link(); operTok = operTok->next()) {
193                             if (operTok->str() == scope->className) {
194                                 classNameUsed = true;
195                                 break;
196                             }
197                         }
198     
199                         if (classNameUsed)
200                             operatorEqVarError(func->token, scope->className, var->name(), inconclusive);
201                     } else if (func->access != Private) {
202                         const Scope *varType = var->typeScope();
203                         if (!varType || varType->type != Scope::eUnion) {
204                             if (func->type == Function::eConstructor &&
205                                 func->nestedIn && (func->nestedIn->numConstructors - func->nestedIn->numCopyOrMoveConstructors) > 1 &&
206                                 func->argCount() == 0 && func->functionScope &&
207                                 func->arg && func->arg->link()->next() == func->functionScope->classStart &&
208                                 func->functionScope->classStart->link() == func->functionScope->classStart->next()) {
209                                 // don't warn about user defined default constructor when there are other constructors
210                                 if (printInconclusive)
211                                     uninitVarError(func->token, scope->className, var->name(), true);
212                             } else
213                                 uninitVarError(func->token, scope->className, var->name(), inconclusive);
214                         }
215                     }
216                 }
217             }
218         }
219     }

221     void CheckClass::checkExplicitConstructors()
222     {
223         if (!_settings->isEnabled("style"))
224             return;
225     
226         const std::size_t classes = symbolDatabase->classAndStructScopes.size();
227         for (std::size_t i = 0; i < classes; ++i) {
228             const Scope * scope = symbolDatabase->classAndStructScopes[i];
229     
230             // Do not perform check, if the class/struct has not any constructors
231             if (scope->numConstructors == 0)
232                 continue;
233     
234             // Is class abstract? Maybe this test is over-simplification, but it will suffice for simple cases,
235             // and it will avoid false positives.
236             bool isAbstractClass = false;
237             for (std::list<Function>::const_iterator func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
238                 if (func->isPure()) {
239                     isAbstractClass = true;
240                     break;
241                 }
242             }
243     
244             // Abstract classes can't be instantiated. But if there is C++11
245             // "misuse" by derived classes then these constructors must be explicit.
246             if (isAbstractClass && _settings->standards.cpp != Standards::CPP11)
247                 continue;
248     
249             for (std::list<Function>::const_iterator func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
250     
251                 // We are looking for constructors, which are meeting following criteria:
252                 //  1) Constructor is declared with a single parameter
253                 //  2) Constructor is not declared as explicit
254                 //  3) It is not a copy/move constructor of non-abstract class
255                 //  4) Constructor is not marked as delete (programmer can mark the default constructor as deleted, which is ok)
256                 if (!func->isConstructor() || func->isDelete() || (!func->hasBody() && func->access == Private))
257                     continue;
258     
259                 if (!func->isExplicit() &&
260                     func->argCount() == 1 &&
261                     func->type != Function::eCopyConstructor &&
262                     func->type != Function::eMoveConstructor) {
263                     noExplicitConstructorError(func->tokenDef, scope->className, scope->type == Scope::eStruct);
264                 }
265             }
266         }

269     void CheckClass::copyconstructors()
270     {
271         if (!_settings->isEnabled("style"))
272             return;
273     
274         const std::size_t classes = symbolDatabase->classAndStructScopes.size();
275         for (std::size_t i = 0; i < classes; ++i) {
276             const Scope * scope = symbolDatabase->classAndStructScopes[i];
277             std::map<unsigned int, const Token*> allocatedVars;
278     
279             for (std::list<Function>::const_iterator func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
280                 if (func->type == Function::eConstructor && func->functionScope) {
281                     const Token* tok = func->functionScope->classDef->linkAt(1);
282                     for (const Token* const end = func->functionScope->classStart; tok != end; tok = tok->next()) {
283                         if (Token::Match(tok, "%var% ( new|malloc|g_malloc|g_try_malloc|realloc|g_realloc|g_try_realloc")) {
284                             const Variable* var = tok->variable();
285                             if (var && var->isPointer() && var->scope() == scope)
286                                 allocatedVars[tok->varId()] = tok;
287                         }
288                     }
289                     for (const Token* const end = func->functionScope->classEnd; tok != end; tok = tok->next()) {
290                         if (Token::Match(tok, "%var% = new|malloc|g_malloc|g_try_malloc|realloc|g_realloc|g_try_realloc")) {
291                             const Variable* var = tok->variable();
292                             if (var && var->isPointer() && var->scope() == scope && !var->isStatic())
293                                 allocatedVars[tok->varId()] = tok;
294                         }
295                     }
296                 }
297             }
298     
299             std::set<const Token*> copiedVars;
300             const Token* copyCtor = 0;
301             for (std::list<Function>::const_iterator func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
302                 if (func->type == Function::eCopyConstructor) {
303                     copyCtor = func->tokenDef;
304                     if (func->functionScope) {
305                         const Token* tok = func->tokenDef->linkAt(1)->next();
306                         if (tok->str()==":") {
307                             tok=tok->next();
308                             while (Token::Match(tok, "%name% (")) {
309                                 if (allocatedVars.find(tok->varId()) != allocatedVars.end()) {
310                                     if (tok->varId() && Token::Match(tok->tokAt(2), "%name% . %name% )"))
311                                         copiedVars.insert(tok);
312                                     else if (!Token::Match(tok->tokAt(2), "%any% )"))
313                                         allocatedVars.erase(tok->varId()); // Assume memory is allocated
314                                 }
315                                 tok = tok->linkAt(1)->tokAt(2);
316                             }
317                         }
318                         for (tok=func->functionScope->classStart; tok!=func->functionScope->classEnd; tok=tok->next()) {
319                             if (Token::Match(tok, "%var% = new|malloc|g_malloc|g_try_malloc|realloc|g_realloc|g_try_realloc")) {
320                                 allocatedVars.erase(tok->varId());
321                             } else if (Token::Match(tok, "%var% = %name% . %name% ;") && allocatedVars.find(tok->varId()) != allocatedVars.end()) {
322                                 copiedVars.insert(tok);
323                             }
324                         }
325                     } else // non-copyable or implementation not seen
326                         allocatedVars.clear();
327                     break;
328                 }
329             }
330             if (!copyCtor) {
331                 if (!allocatedVars.empty() && scope->definedType->derivedFrom.empty()) // TODO: Check if base class is non-copyable
332                     noCopyConstructorError(scope->classDef, scope->className, scope->type == Scope::eStruct);
333             } else {
334                 if (!copiedVars.empty()) {
335                     for (std::set<const Token*>::const_iterator it = copiedVars.begin(); it != copiedVars.end(); ++it) {
336                         copyConstructorShallowCopyError(*it, (*it)->str());
337                     }
338                 }
339                 // throw error if count mismatch
340                 /* FIXME: This doesn't work. See #4154
341                 for (std::map<unsigned int, const Token*>::const_iterator i = allocatedVars.begin(); i != allocatedVars.end(); ++i) {
342                     copyConstructorMallocError(copyCtor, i->second, i->second->str());
343                 }
344                 */
345             }
346         }
347     }

372     bool CheckClass::canNotCopy(const Scope *scope)
373     {
374         std::list<Function>::const_iterator func;
375         bool constructor = false;
376         bool publicAssign = false;
377         bool publicCopy = false;
378     
379         for (func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
380             if (func->isConstructor())
381                 constructor = true;
382             if ((func->type == Function::eCopyConstructor) &&
383                 func->access == Public)
384                 publicCopy = true;
385             else if (func->type == Function::eOperatorEqual && func->access == Public)
386                 publicAssign = true;
387         }
388     
389         return constructor && !(publicAssign || publicCopy);
390     }

481     void CheckClass::initializeVarList(const Function &func, std::list<const Function *> &callstack, const Scope *scope, std::vector<Usage> &usage)
482     {
483         if (!func.functionScope)
484             throw InternalError(0, "Internal Error: Invalid syntax"); // #5702
485         bool initList = func.isConstructor();
486         const Token *ftok = func.arg->link()->next();
487         int level = 0;
488         for (; ftok && ftok != func.functionScope->classEnd; ftok = ftok->next()) {
489             // Class constructor.. initializing variables like this
490             // clKalle::clKalle() : var(value) { }
491             if (initList) {
492                 if (level == 0 && Token::Match(ftok, "%name% {|(") && Token::Match(ftok->linkAt(1), "}|) ,|{")) {
493                     if (ftok->str() != func.name()) {
494                         initVar(ftok->varId(), scope, usage);
495                     } else { // c++11 delegate constructor
496                         const Function *member = ftok->function();
497                         // member function found
498                         if (member) {
499                             // recursive call
500                             // assume that all variables are initialized
501                             if (std::find(callstack.begin(), callstack.end(), member) != callstack.end()) {
502                                 /** @todo false negative: just bail */
503                                 assignAllVar(usage);
504                                 return;
505                             }
506     
507                             // member function has implementation
508                             if (member->hasBody()) {
509                                 // initialize variable use list using member function
510                                 callstack.push_back(member);
511                                 initializeVarList(*member, callstack, scope, usage);
512                                 callstack.pop_back();
513                             }
514     
515                             // there is a called member function, but it has no implementation, so we assume it initializes everything
516                             else {
517                                 assignAllVar(usage);
518                             }
519                         }
520                     }
521                 } else if (level != 0 && Token::Match(ftok, "%name% =")) // assignment in the initializer: var(value = x)
522                     assignVar(ftok->varId(), scope, usage);
523     
524                 // Level handling
525                 if (ftok->link() && Token::Match(ftok, "(|<"))
526                     level++;
527                 else if (ftok->str() == "{") {
528                     if (level != 0 ||
529                         (Token::Match(ftok->previous(), "%name%|>") && Token::Match(ftok->link(), "} ,|{")))
530                         level++;
531                     else
532                         initList = false;
533                 } else if (ftok->link() && Token::Match(ftok, ")|>|}"))
534                     level--;
535             }
536     
537             if (initList)
538                 continue;
539     
540             // Variable getting value from stream?
541             if (Token::Match(ftok, ">> %name%")) {
542                 assignVar(ftok->next()->varId(), scope, usage);
543             }
544     
545             // Before a new statement there is "[{};()=[]"
546             if (! Token::Match(ftok, "[{};()=[]"))
547                 continue;
548     
549             if (Token::simpleMatch(ftok, "( !"))
550                 ftok = ftok->next();
551     
552             // Using the operator= function to initialize all variables..
553             if (Token::Match(ftok->next(), "return| (| * this )| =")) {
554                 assignAllVar(usage);
555                 break;
556             }
557     
558             // Using swap to assign all variables..
559             if (func.type == Function::eOperatorEqual && Token::Match(ftok, "[;{}] %name% (") && Token::Match(ftok->linkAt(2), ") . %name% ( *| this ) ;")) {
560                 assignAllVar(usage);
561                 break;
562             }
563     
564             // Calling member variable function?
565             if (Token::Match(ftok->next(), "%var% . %name% (")) {
566                 std::list<Variable>::const_iterator var;
567                 for (var = scope->varlist.begin(); var != scope->varlist.end(); ++var) {
568                     if (var->declarationId() == ftok->next()->varId()) {
569                         /** @todo false negative: we assume function changes variable state */
570                         assignVar(ftok->next()->varId(), scope, usage);
571                         break;
572                     }
573                 }
574     
575                 ftok = ftok->tokAt(2);
576             }
577     
578             if (!Token::Match(ftok->next(), "::| %name%") &&
579                 !Token::Match(ftok->next(), "*| this . %name%") &&
580                 !Token::Match(ftok->next(), "* %name% =") &&
581                 !Token::Match(ftok->next(), "( * this ) . %name%"))
582                 continue;
583     
584             // Goto the first token in this statement..
585             ftok = ftok->next();
586     
587             // skip "return"
588             if (ftok->str() == "return")
589                 ftok = ftok->next();
590     
591             // Skip "( * this )"
592             if (Token::simpleMatch(ftok, "( * this ) .")) {
593                 ftok = ftok->tokAt(5);
594             }
595     
596             // Skip "this->"
597             if (Token::simpleMatch(ftok, "this ."))
598                 ftok = ftok->tokAt(2);
599     
600             // Skip "classname :: "
601             if (Token::Match(ftok, ":: %name%"))
602                 ftok = ftok->next();
603             while (Token::Match(ftok, "%name% ::"))
604                 ftok = ftok->tokAt(2);
605     
606             // Clearing all variables..
607             if (Token::Match(ftok, "::| memset ( this ,")) {
608                 assignAllVar(usage);
609                 return;
610             }
611     
612             // Clearing array..
613             else if (Token::Match(ftok, "::| memset ( %name% ,")) {
614                 if (ftok->str() == "::")
615                     ftok = ftok->next();
616                 assignVar(ftok->tokAt(2)->varId(), scope, usage);
617                 ftok = ftok->linkAt(1);
618                 continue;
619             }
620     
621             // Calling member function?
622             else if (Token::simpleMatch(ftok, "operator= (") &&
623                      ftok->previous()->str() != "::") {
624                 if (ftok->function() && ftok->function()->nestedIn == scope) {
625                     const Function *member = ftok->function();
626                     // recursive call
627                     // assume that all variables are initialized
628                     if (std::find(callstack.begin(), callstack.end(), member) != callstack.end()) {
629                         /** @todo false negative: just bail */
630                         assignAllVar(usage);
631                         return;
632                     }
633     
634                     // member function has implementation
635                     if (member->hasBody()) {
636                         // initialize variable use list using member function
637                         callstack.push_back(member);
638                         initializeVarList(*member, callstack, scope, usage);
639                         callstack.pop_back();
640                     }
641     
642                     // there is a called member function, but it has no implementation, so we assume it initializes everything
643                     else {
644                         assignAllVar(usage);
645                     }
646                 }
647     
648                 // using default operator =, assume everything initialized
649                 else {
650                     assignAllVar(usage);
651                 }
652             } else if (Token::Match(ftok, "::| %name% (") && ftok->str() != "if") {
653                 if (ftok->str() == "::")
654                     ftok = ftok->next();
655     
656                 // Passing "this" => assume that everything is initialized
657                 for (const Token *tok2 = ftok->next()->link(); tok2 && tok2 != ftok; tok2 = tok2->previous()) {
658                     if (tok2->str() == "this") {
659                         assignAllVar(usage);
660                         return;
661                     }
662                 }
663     
664                 // check if member function
665                 if (ftok->function() && ftok->function()->nestedIn == scope &&
666                     !ftok->function()->isConstructor()) {
667                     const Function *member = ftok->function();
668     
669                     // recursive call
670                     // assume that all variables are initialized
671                     if (std::find(callstack.begin(), callstack.end(), member) != callstack.end()) {
672                         assignAllVar(usage);
673                         return;
674                     }
675     
676                     // member function has implementation
677                     if (member->hasBody()) {
678                         // initialize variable use list using member function
679                         callstack.push_back(member);
680                         initializeVarList(*member, callstack, scope, usage);
681                         callstack.pop_back();
682     
683                         // Assume that variables that are passed to it are initialized..
684                         for (const Token *tok2 = ftok; tok2; tok2 = tok2->next()) {
685                             if (Token::Match(tok2, "[;{}]"))
686                                 break;
687                             if (Token::Match(tok2, "[(,] &| %name% [,)]")) {
688                                 tok2 = tok2->next();
689                                 if (tok2->str() == "&")
690                                     tok2 = tok2->next();
691                                 assignVar(tok2->varId(), scope, usage);
692                             }
693                         }
694                     }
695     
696                     // there is a called member function, but it has no implementation, so we assume it initializes everything
697                     else {
698                         assignAllVar(usage);
699                     }
700                 }
701     
702                 // not member function
703                 else {
704                     // could be a base class virtual function, so we assume it initializes everything
705                     if (!func.isConstructor() && isBaseClassFunc(ftok, scope)) {
706                         /** @todo False Negative: we should look at the base class functions to see if they
707                          *  call any derived class virtual functions that change the derived class state
708                          */
709                         assignAllVar(usage);
710                     }
711     
712                     // has friends, so we assume it initializes everything
713                     if (!scope->definedType->friendList.empty())
714                         assignAllVar(usage);
715     
716                     // the function is external and it's neither friend nor inherited virtual function.
717                     // assume all variables that are passed to it are initialized..
718                     else {
719                         for (const Token *tok = ftok->tokAt(2); tok && tok != ftok->next()->link(); tok = tok->next()) {
720                             if (tok->isName()) {
721                                 assignVar(tok->varId(), scope, usage);
722                             }
723                         }
724                     }
725                 }
726             }
727     
728             // Assignment of member variable?
729             else if (Token::Match(ftok, "%name% =")) {
730                 assignVar(ftok->varId(), scope, usage);
731                 bool bailout = ftok->variable() && ftok->variable()->isReference();
732                 const Token* tok2 = ftok->tokAt(2);
733                 if (tok2->str() == "&") {
734                     tok2 = tok2->next();
735                     bailout = true;
736                 }
737                 if (tok2->variable() && (bailout || tok2->variable()->isArray()) && tok2->strAt(1) != "[")
738                     assignVar(tok2->varId(), scope, usage);
739             }
740     
741             // Assignment of array item of member variable?
742             else if (Token::Match(ftok, "%name% [|.")) {
743                 const Token *tok2 = ftok;
744                 while (tok2) {
745                     if (tok2->strAt(1) == "[")
746                         tok2 = tok2->next()->link();
747                     else if (Token::Match(tok2->next(), ". %name%"))
748                         tok2 = tok2->tokAt(2);
749                     else
750                         break;
751                 }
752                 if (tok2 && tok2->strAt(1) == "=")
753                     assignVar(ftok->varId(), scope, usage);
754             }
755     
756             // Assignment of array item of member variable?
757             else if (Token::Match(ftok, "* %name% =")) {
758                 assignVar(ftok->next()->varId(), scope, usage);
759             } else if (Token::Match(ftok, "* this . %name% =")) {
760                 assignVar(ftok->tokAt(3)->varId(), scope, usage);
761             }
762     
763             // The functions 'clear' and 'Clear' are supposed to initialize variable.
764             if (Token::Match(ftok, "%name% . clear|Clear (")) {
765                 assignVar(ftok->varId(), scope, usage);
766             }
767         }
768     }

800     // ClassCheck: Use initialization list instead of assignment
801     //---------------------------------------------------------------------------
802     
803     void CheckClass::initializationListUsage()
804     {
805         if (!_settings->isEnabled("performance"))
806             return;
807     
808         const std::size_t functions = symbolDatabase->functionScopes.size();
809         for (std::size_t i = 0; i < functions; ++i) {
810             const Scope * scope = symbolDatabase->functionScopes[i];
811     
812             // Check every constructor
813             if (!scope->function || (!scope->function->isConstructor()))
814                 continue;
815     
816             const Scope* owner = scope->functionOf;
817             for (const Token* tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
818                 if (Token::Match(tok, "%name% (")) // Assignments might depend on this function call or if/for/while/switch statement from now on.
819                     break;
820                 if (Token::Match(tok, "try|do {"))
821                     break;
822                 if (Token::Match(tok, "%var% =") && tok->strAt(-1) != "*") {
823                     const Variable* var = tok->variable();
824                     if (var && var->scope() == owner && !var->isStatic()) {
825                         if (var->isPointer() || var->isReference() || (!var->type() && !var->isStlStringType() && !(Token::Match(var->typeStartToken(), "std :: %type% <") && !Token::simpleMatch(var->typeStartToken()->linkAt(3), "> ::"))))
826                             continue;
827     
828                         bool allowed = true;
829                         for (const Token* tok2 = tok->tokAt(2); tok2 && tok2->str() != ";"; tok2 = tok2->next()) {
830                             const Variable* var2 = tok2->variable();
831                             if (var2) {
832                                 if (var2->scope() == owner && tok2->strAt(-1)!=".") { // Is there a dependency between two member variables?
833                                     allowed = false;
834                                     break;
835                                 } else if (var2->isArray() && var2->isLocal()) { // Can't initialize with a local array
836                                     allowed = false;
837                                     break;
838                                 }
839                             } else if (tok2->str() == "this") { // 'this' instance is not completely constructed in initialization list
840                                 allowed = false;
841                                 break;
842                             } else if (Token::Match(tok2, "%name% (") && tok2->strAt(-1) != "." && isMemberFunc(owner, tok2)) { // Member function called?
843                                 allowed = false;
844                                 break;
845                             }
846                         }
847                         if (!allowed)
848                             continue;
849     
850                         suggestInitializationList(tok, tok->str());
851                     }
852                 }
853             }
854         }
855     }

866     // ClassCheck: Unused private functions
867     //---------------------------------------------------------------------------
868     
869     static bool checkFunctionUsage(const std::string& name, const Scope* scope)
870     {
871         if (!scope)
872             return true; // Assume it is used, if scope is not seen
873     
874         for (std::list<Function>::const_iterator func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
875             if (func->functionScope) {
876                 if (Token::Match(func->tokenDef, "%name% (")) {
877                     for (const Token *ftok = func->tokenDef->tokAt(2); ftok && ftok->str() != ")"; ftok = ftok->next()) {
878                         if (Token::Match(ftok, "= %name% [(,)]") && ftok->strAt(1) == name)
879                             return true;
880                         if (ftok->str() == "(")
881                             ftok = ftok->link();
882                     }
883                 }
884                 for (const Token *ftok = func->functionScope->classDef->linkAt(1); ftok != func->functionScope->classEnd; ftok = ftok->next()) {
885                     if (ftok->str() == name) // Function used. TODO: Handle overloads
886                         return true;
887                 }
888             } else if ((func->type != Function::eCopyConstructor &&
889                         func->type != Function::eOperatorEqual) ||
890                        func->access != Private) // Assume it is used, if a function implementation isn't seen, but empty private copy constructors and assignment operators are OK
891                 return true;
892         }
893     
894         for (std::list<Scope*>::const_iterator i = scope->nestedList.begin(); i != scope->nestedList.end(); ++i) {
895             if ((*i)->isClassOrStruct())
896                 if (checkFunctionUsage(name, *i)) // Check nested classes, which can access private functions of their base
897                     return true;
898         }
899     
900         for (std::list<Variable>::const_iterator i = scope->varlist.begin(); i != scope->varlist.end(); ++i) {
901             if (i->isStatic()) {
902                 const Token* tok = Token::findmatch(scope->classEnd, "%varid% =|(|{", i->declarationId());
903                 if (tok)
904                     tok = tok->tokAt(2);
905                 while (tok && tok->str() != ";") {
906                     if (tok->str() == name && (tok->strAt(-1) == "." || tok->strAt(-2) == scope->className))
907                         return true;
908                     tok = tok->next();
909                 }
910             }
911         }
912     
913         return false; // Unused in this scope
914     }

916     void CheckClass::privateFunctions()
917     {
918         if (!_settings->isEnabled("style"))
919             return;
920     
921         const std::size_t classes = symbolDatabase->classAndStructScopes.size();
922         for (std::size_t i = 0; i < classes; ++i) {
923             const Scope * scope = symbolDatabase->classAndStructScopes[i];
924     
925             // do not check borland classes with properties..
926             if (Token::findsimplematch(scope->classStart, "; __property ;", scope->classEnd))
927                 continue;
928     
929             std::list<const Function*> privateFuncs;
930             for (std::list<Function>::const_iterator func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
931                 // Get private functions..
932                 if (func->type == Function::eFunction && func->access == Private && !func->isOperator()) // TODO: There are smarter ways to check private operator usage
933                     privateFuncs.push_back(&*func);
934             }
935     
936             // Bailout for overridden virtual functions of base classes
937             if (!scope->definedType->derivedFrom.empty()) {
938                 // Check virtual functions
939                 for (std::list<const Function*>::iterator it = privateFuncs.begin(); it != privateFuncs.end();) {
940                     if ((*it)->isImplicitlyVirtual(true)) // Give true as default value to be returned if we don't see all base classes
941                         privateFuncs.erase(it++);
942                     else
943                         ++it;
944                 }
945             }
946     
947             while (!privateFuncs.empty()) {
948                 const std::string& funcName = privateFuncs.front()->tokenDef->str();
949                 // Check that all private functions are used
950                 bool used = checkFunctionUsage(funcName, scope); // Usage in this class
951                 // Check in friend classes
952                 const std::list<Type::FriendInfo>& friendList = scope->definedType->friendList;
953                 for (std::list<Type::FriendInfo>::const_iterator it = friendList.begin(); !used && it != friendList.end(); ++it) {
954                     if (it->type)
955                         used = checkFunctionUsage(funcName, it->type->classScope);
956                     else
957                         used = true; // Assume, it is used if we do not see friend class
958                 }
959     
960                 if (!used)
961                     unusedPrivateFunctionError(privateFuncs.front()->tokenDef, scope->className, funcName);
962     
963                 privateFuncs.pop_front();
964             }
965         }
966     }

987     void CheckClass::checkMemset()
988     {
989         const bool printWarnings = _settings->isEnabled("warning");
990         const std::size_t functions = symbolDatabase->functionScopes.size();
991         for (std::size_t i = 0; i < functions; ++i) {
992             const Scope * scope = symbolDatabase->functionScopes[i];
993             for (const Token *tok = scope->classStart; tok && tok != scope->classEnd; tok = tok->next()) {
994                 if (Token::Match(tok, "memset|memcpy|memmove (")) {
995                     const Token* arg1 = tok->tokAt(2);
996                     const Token* arg3 = arg1->nextArgument();
997                     if (arg3)
998                         arg3 = arg3->nextArgument();
999                     if (!arg3)
1000                         // weird, shouldn't happen: memset etc should have
1001                         // 3 arguments.
1002                         continue;
1003     
1004     
1005                     const Token *typeTok = nullptr;
1006                     const Scope *type = nullptr;
1007                     if (Token::Match(arg3, "sizeof ( %type% ) )"))
1008                         typeTok = arg3->tokAt(2);
1009                     else if (Token::Match(arg3, "sizeof ( %type% :: %type% ) )"))
1010                         typeTok = arg3->tokAt(4);
1011                     else if (Token::Match(arg3, "sizeof ( struct %type% ) )"))
1012                         typeTok = arg3->tokAt(3);
1013                     else if (Token::simpleMatch(arg3, "sizeof ( * this ) )") || Token::simpleMatch(arg1, "this ,")) {
1014                         type = findFunctionOf(arg3->scope());
1015                     } else if (Token::Match(arg1, "&|*|%var%")) {
1016                         int numIndirToVariableType = 0; // Offset to the actual type in terms of dereference/addressof
1017                         for (;; arg1 = arg1->next()) {
1018                             if (arg1->str() == "&")
1019                                 ++numIndirToVariableType;
1020                             else if (arg1->str() == "*")
1021                                 --numIndirToVariableType;
1022                             else
1023                                 break;
1024                         }
1025     
1026                         const Variable * const var = arg1->variable();
1027                         if (var && arg1->strAt(1) == ",") {
1028                             if (var->isArrayOrPointer()) {
1029                                 const Token *endTok = var->typeEndToken();
1030                                 while (endTok && Token::simpleMatch(endTok, "*")) {
1031                                     ++numIndirToVariableType;
1032                                     endTok = endTok->previous();
1033                                 }
1034                             }
1035     
1036                             if (var->isArray())
1037                                 numIndirToVariableType += int(var->dimensions().size());
1038     
1039                             if (numIndirToVariableType == 1)
1040                                 type = var->typeScope();
1041                         }
1042                     }
1043     
1044                     // No type defined => The tokens didn't match
1045                     if (!typeTok && !type)
1046                         continue;
1047     
1048                     if (typeTok && typeTok->str() == "(")
1049                         typeTok = typeTok->next();
1050     
1051                     if (!type && typeTok->type())
1052                         type = typeTok->type()->classScope;
1053     
1054                     if (type) {
1055                         std::set<const Scope *> parsedTypes;
1056                         checkMemsetType(scope, tok, type, false, parsedTypes);
1057                     }
1058                 } else if (tok->variable() && tok->variable()->typeScope() && Token::Match(tok, "%var% = calloc|malloc|realloc|g_malloc|g_try_malloc|g_realloc|g_try_realloc (")) {
1059                     std::set<const Scope *> parsedTypes;
1060                     checkMemsetType(scope, tok->tokAt(2), tok->variable()->typeScope(), true, parsedTypes);
1061     
1062                     if (tok->variable()->typeScope()->numConstructors > 0 && printWarnings)
1063                         mallocOnClassWarning(tok, tok->strAt(2), tok->variable()->typeScope()->classDef);
1064                 }
1065             }
1066         }
1067     }

1069     void CheckClass::checkMemsetType(const Scope *start, const Token *tok, const Scope *type, bool allocation, std::set<const Scope *> parsedTypes)
1070     {
1071         // If type has been checked there is no need to check it again
1072         if (parsedTypes.find(type) != parsedTypes.end())
1073             return;
1074         parsedTypes.insert(type);
1075     
1076         const bool printPortability = _settings->isEnabled("portability");
1077     
1078         // recursively check all parent classes
1079         for (std::size_t i = 0; i < type->definedType->derivedFrom.size(); i++) {
1080             const Type* derivedFrom = type->definedType->derivedFrom[i].type;
1081             if (derivedFrom && derivedFrom->classScope)
1082                 checkMemsetType(start, tok, derivedFrom->classScope, allocation, parsedTypes);
1083         }
1084     
1085         // Warn if type is a class that contains any virtual functions
1086         std::list<Function>::const_iterator func;
1087     
1088         for (func = type->functionList.begin(); func != type->functionList.end(); ++func) {
1089             if (func->isVirtual()) {
1090                 if (allocation)
1091                     mallocOnClassError(tok, tok->str(), type->classDef, "virtual method");
1092                 else
1093                     memsetError(tok, tok->str(), "virtual method", type->classDef->str());
1094             }
1095         }
1096     
1097         // Warn if type is a class or struct that contains any std::* variables
1098         std::list<Variable>::const_iterator var;
1099     
1100         for (var = type->varlist.begin(); var != type->varlist.end(); ++var) {
1101             if (var->isReference() && !var->isStatic()) {
1102                 memsetErrorReference(tok, tok->str(), type->classDef->str());
1103                 continue;
1104             }
1105             // don't warn if variable static or const, pointer or reference
1106             if (!var->isStatic() && !var->isConst() && !var->isPointer()) {
1107                 const Token *tok1 = var->typeStartToken();
1108                 const Scope *typeScope = var->typeScope();
1109     
1110                 // check for std:: type
1111                 if (var->isStlType() && tok1->strAt(2) != "array" && !_settings->library.podtype(tok1->strAt(2)))
1112                     if (allocation)
1113                         mallocOnClassError(tok, tok->str(), type->classDef, "'std::" + tok1->strAt(2) + "'");
1114                     else
1115                         memsetError(tok, tok->str(), "'std::" + tok1->strAt(2) + "'", type->classDef->str());
1116     
1117                 // check for known type
1118                 else if (typeScope && typeScope != type)
1119                     checkMemsetType(start, tok, typeScope, allocation, parsedTypes);
1120     
1121                 // check for float
1122                 else if (tok->str() == "memset" && var->isFloatingType() && printPortability)
1123                     memsetErrorFloat(tok, type->classDef->str());
1124             }
1125         }
1126     }

1190                     // skip "deleted" functions - cannot be called anyway

1259     void CheckClass::checkReturnPtrThis(const Scope *scope, const Function *func, const Token *tok, const Token *last, std::set<const Function*>& analyzedFunctions)
1260     {
1261         bool foundReturn = false;
1262     
1263         const Token* const startTok = tok;
1264     
1265         for (; tok && tok != last; tok = tok->next()) {
1266             // check for return of reference to this
1267             if (tok->str() == "return") {
1268                 foundReturn = true;
1269                 std::string cast("( " + scope->className + " & )");
1270                 if (Token::simpleMatch(tok->next(), cast.c_str()))
1271                     tok = tok->tokAt(4);
1272     
1273                 // check if a function is called
1274                 if (tok->strAt(2) == "(" &&
1275                     tok->linkAt(2)->next()->str() == ";") {
1276                     std::list<Function>::const_iterator it;
1277     
1278                     // check if it is a member function
1279                     for (it = scope->functionList.begin(); it != scope->functionList.end(); ++it) {
1280                         // check for a regular function with the same name and a body
1281                         if (it->type == Function::eFunction && it->hasBody() &&
1282                             it->token->str() == tok->next()->str()) {
1283                             // check for the proper return type
1284                             if (it->tokenDef->previous()->str() == "&" &&
1285                                 it->tokenDef->strAt(-2) == scope->className) {
1286                                 // make sure it's not a const function
1287                                 if (!it->isConst()) {
1288                                     /** @todo make sure argument types match */
1289                                     // avoid endless recursions
1290                                     if (analyzedFunctions.find(&*it) == analyzedFunctions.end()) {
1291                                         analyzedFunctions.insert(&*it);
1292                                         checkReturnPtrThis(scope, &*it, it->arg->link()->next(), it->arg->link()->next()->link(),
1293                                                            analyzedFunctions);
1294                                     }
1295                                     // just bail for now
1296                                     else
1297                                         return;
1298                                 }
1299                             }
1300                         }
1301                     }
1302                 }
1303     
1304                 // check if *this is returned
1305                 else if (!(Token::Match(tok->next(), "(| * this ;|=") ||
1306                            Token::simpleMatch(tok->next(), "operator= (") ||
1307                            Token::simpleMatch(tok->next(), "this . operator= (") ||
1308                            (Token::Match(tok->next(), "%type% :: operator= (") &&
1309                             tok->next()->str() == scope->className)))
1310                     operatorEqRetRefThisError(func->token);
1311             }
1312         }
1313         if (foundReturn) {
1314             return;
1315         }
1316         if (startTok->next() == last) {
1317             if (Token::simpleMatch(func->argDef, std::string("( const " + scope->className + " &").c_str())) {
1318                 // Typical wrong way to suppress default assignment operator by declaring it and leaving empty
1319                 operatorEqMissingReturnStatementError(func->token, func->access == Public);
1320             } else {
1321                 operatorEqMissingReturnStatementError(func->token, true);
1322             }
1323             return;
1324         }
1325         if (_settings->library.isScopeNoReturn(last, 0)) {
1326             // Typical wrong way to prohibit default assignment operator
1327             // by always throwing an exception or calling a noreturn function
1328             operatorEqShouldBeLeftUnimplementedError(func->token);
1329             return;
1330         }
1331     
1332         operatorEqMissingReturnStatementError(func->token, func->access == Public);
1333     }

1354     //---------------------------------------------------------------------------
1355     // ClassCheck: "C& operator=(const C& rhs) { if (this == &rhs) ... }"
1356     // operator= should check for assignment to self
1357     //
1358     // For simple classes, an assignment to self check is only a potential optimization.
1359     //
1360     // For classes that allocate dynamic memory, assignment to self can be a real error
1361     // if it is deallocated and allocated again without being checked for.
1362     //
1363     // This check is not valid for classes with multiple inheritance because a
1364     // class can have multiple addresses so there is no trivial way to check for
1365     // assignment to self.
1366     //---------------------------------------------------------------------------
1367     
1368     void CheckClass::operatorEqToSelf()
1369     {
1370         if (!_settings->isEnabled("warning"))
1371             return;
1372     
1373         const std::size_t classes = symbolDatabase->classAndStructScopes.size();
1374         for (std::size_t i = 0; i < classes; ++i) {
1375             const Scope * scope = symbolDatabase->classAndStructScopes[i];
1376             // skip classes with multiple inheritance
1377             if (scope->definedType->derivedFrom.size() > 1)
1378                 continue;
1379     
1380             std::list<Function>::const_iterator func;
1381             for (func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
1382                 if (func->type == Function::eOperatorEqual && func->hasBody()) {
1383                     // make sure that the operator takes an object of the same type as *this, otherwise we can't detect self-assignment checks
1384                     if (func->argumentList.empty())
1385                         continue;
1386                     const Token* typeTok = func->argumentList.front().typeEndToken();
1387                     while (typeTok->str() == "const" || typeTok->str() == "&" || typeTok->str() == "*")
1388                         typeTok = typeTok->previous();
1389                     if (typeTok->str() != scope->className)
1390                         continue;
1391     
1392                     // make sure return signature is correct
1393                     if (Token::Match(func->retDef, "%type% &") && func->retDef->str() == scope->className) {
1394                         // find the parameter name
1395                         const Token *rhs = func->argumentList.begin()->nameToken();
1396     
1397                         if (!hasAssignSelf(&(*func), rhs)) {
1398                             if (hasAllocation(&(*func), scope))
1399                                 operatorEqToSelfError(func->token);
1400                         }
1401                     }
1402                 }
1403             }
1404         }
1405     }
1406     
1407     bool CheckClass::hasAllocation(const Function *func, const Scope* scope) const
1408     {
1409         // This function is called when no simple check was found for assignment
1410         // to self.  We are currently looking for:
1411         //    - deallocate member ; ... member =
1412         //    - alloc member
1413         // That is not ideal because it can cause false negatives but its currently
1414         // necessary to prevent false positives.
1415         const Token *last = func->functionScope->classEnd;
1416         for (const Token *tok = func->functionScope->classStart; tok && (tok != last); tok = tok->next()) {
1417             if (Token::Match(tok, "%var% = malloc|realloc|calloc|new") && isMemberVar(scope, tok))
1418                 return true;
1419     
1420             // check for deallocating memory
1421             const Token *var = nullptr;
1422             if (Token::Match(tok, "free ( %var%"))
1423                 var = tok->tokAt(2);
1424             else if (Token::Match(tok, "delete [ ] %var%"))
1425                 var = tok->tokAt(3);
1426             else if (Token::Match(tok, "delete %var%"))
1427                 var = tok->next();
1428             // Check for assignment to the deleted pointer (only if its a member of the class)
1429             if (var && isMemberVar(scope, var)) {
1430                 for (const Token *tok1 = var->next(); tok1 && (tok1 != last); tok1 = tok1->next()) {
1431                     if (Token::Match(tok1, "%varid% =", var->varId()))
1432                         return true;
1433                 }
1434             }
1435         }
1436     
1437         return false;
1438     }

1440     bool CheckClass::hasAssignSelf(const Function *func, const Token *rhs)
1441     {
1442         if (!rhs)
1443             return false;
1444         const Token *last = func->functionScope->classEnd;
1445         for (const Token *tok = func->functionScope->classStart; tok && tok != last; tok = tok->next()) {
1446             if (Token::simpleMatch(tok, "if (")) {
1447                 const Token *tok1 = tok->tokAt(2);
1448                 const Token *tok2 = tok->next()->link();
1449     
1450                 if (tok1 && tok2) {
1451                     for (; tok1 && tok1 != tok2; tok1 = tok1->next()) {
1452                         if (Token::Match(tok1, "this ==|!= & %name%")) {
1453                             if (tok1->strAt(3) == rhs->str())
1454                                 return true;
1455                         } else if (Token::Match(tok1, "& %name% ==|!= this")) {
1456                             if (tok1->strAt(1) == rhs->str())
1457                                 return true;
1458                         }
1459                     }
1460                 }
1461             }
1462         }
1463     
1464         return false;
1465     }

1475     //---------------------------------------------------------------------------
1476     // A destructor in a base class should be virtual
1477     //---------------------------------------------------------------------------
1478     
1479     void CheckClass::virtualDestructor()
1480     {
1481         // This error should only be given if:
1482         // * base class doesn't have virtual destructor
1483         // * derived class has non-empty destructor
1484         // * base class is deleted
1485         // unless inconclusive in which case:
1486         // * base class has virtual members but doesn't have virtual destructor
1487         const bool printInconclusive = _settings->inconclusive;
1488     
1489         std::list<const Function *> inconclusive_errors;
1490     
1491         const std::size_t classes = symbolDatabase->classAndStructScopes.size();
1492         for (std::size_t i = 0; i < classes; ++i) {
1493             const Scope * scope = symbolDatabase->classAndStructScopes[i];
1494     
1495             // Skip base classes (unless inconclusive)
1496             if (scope->definedType->derivedFrom.empty()) {
1497                 if (printInconclusive) {
1498                     const Function *destructor = scope->getDestructor();
1499                     if (destructor && !destructor->isVirtual()) {
1500                         std::list<Function>::const_iterator func;
1501                         for (func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
1502                             if (func->isVirtual()) {
1503                                 inconclusive_errors.push_back(destructor);
1504                                 break;
1505                             }
1506                         }
1507                     }
1508                 }
1509                 continue;
1510             }
1511     
1512             // Find the destructor
1513             const Function *destructor = scope->getDestructor();
1514     
1515             // Check for destructor with implementation
1516             if (!destructor || !destructor->hasBody())
1517                 continue;
1518     
1519             // Empty destructor
1520             if (destructor->token->linkAt(3) == destructor->token->tokAt(4))
1521                 continue;
1522     
1523             const Token *derived = scope->classDef;
1524             const Token *derivedClass = derived->next();
1525     
1526             // Iterate through each base class...
1527             for (std::size_t j = 0; j < scope->definedType->derivedFrom.size(); ++j) {
1528                 // Check if base class is public and exists in database
1529                 if (scope->definedType->derivedFrom[j].access != Private && scope->definedType->derivedFrom[j].type) {
1530                     const Type *derivedFrom = scope->definedType->derivedFrom[j].type;
1531                     const Scope *derivedFromScope = derivedFrom->classScope;
1532                     if (!derivedFromScope)
1533                         continue;
1534     
1535                     // Check for this pattern:
1536                     // 1. Base class pointer is given the address of derived class instance
1537                     // 2. Base class pointer is deleted
1538                     //
1539                     // If this pattern is not seen then bailout the checking of these base/derived classes
1540                     {
1541                         // pointer variables of type 'Base *'
1542                         std::set<unsigned int> basepointer;
1543     
1544                         for (std::size_t k = 1; k < symbolDatabase->getVariableListSize(); k++) {
1545                             const Variable* var = symbolDatabase->getVariableFromVarId(k);
1546                             if (var && var->isPointer() && var->type() == derivedFrom)
1547                                 basepointer.insert(var->declarationId());
1548                         }
1549     
1550                         // pointer variables of type 'Base *' that should not be deleted
1551                         std::set<unsigned int> dontDelete;
1552     
1553                         // No deletion of derived class instance through base class pointer found => the code is ok
1554                         bool ok = true;
1555     
1556                         for (const Token *tok = _tokenizer->tokens(); tok; tok = tok->next()) {
1557                             if (Token::Match(tok, "[;{}] %var% =") &&
1558                                 basepointer.find(tok->next()->varId()) != basepointer.end()) {
1559                                 // new derived class..
1560                                 if (Token::simpleMatch(tok->tokAt(3), ("new " + derivedClass->str()).c_str())) {
1561                                     dontDelete.insert(tok->next()->varId());
1562                                 }
1563                             }
1564     
1565                             // Delete base class pointer that might point at derived class
1566                             else if (Token::Match(tok, "delete %var% ;") &&
1567                                      dontDelete.find(tok->next()->varId()) != dontDelete.end()) {
1568                                 ok = false;
1569                                 break;
1570                             }
1571                         }
1572     
1573                         // No base class pointer that points at a derived class is deleted
1574                         if (ok)
1575                             continue;
1576                     }
1577     
1578                     // Find the destructor declaration for the base class.
1579                     const Function *base_destructor = derivedFromScope->getDestructor();
1580                     const Token *base = nullptr;
1581                     if (base_destructor)
1582                         base = base_destructor->token;
1583     
1584                     // Check that there is a destructor..
1585                     if (!base_destructor) {
1586                         if (derivedFrom->derivedFrom.empty()) {
1587                             virtualDestructorError(derivedFrom->classDef, derivedFrom->name(), derivedClass->str(), false);
1588                             // check for duplicate error and remove if if found
1589                             std::list<const Function *>::iterator found = find(inconclusive_errors.begin(), inconclusive_errors.end(), base_destructor);
1590                             if (found != inconclusive_errors.end())
1591                                 inconclusive_errors.erase(found);
1592                         }
1593                     } else if (!base_destructor->isVirtual()) {
1594                         // TODO: This is just a temporary fix, better solution is needed.
1595                         // Skip situations where base class has base classes of its own, because
1596                         // some of the base classes might have virtual destructor.
1597                         // Proper solution is to check all of the base classes. If base class is not
1598                         // found or if one of the base classes has virtual destructor, error should not
1599                         // be printed. See TODO test case "virtualDestructorInherited"
1600                         if (derivedFrom->derivedFrom.empty()) {
1601                             // Make sure that the destructor is public (protected or private
1602                             // would not compile if inheritance is used in a way that would
1603                             // cause the bug we are trying to find here.)
1604                             if (base_destructor->access == Public) {
1605                                 virtualDestructorError(base, derivedFrom->name(), derivedClass->str(), false);
1606                                 // check for duplicate error and remove if if found
1607                                 std::list<const Function *>::iterator found = find(inconclusive_errors.begin(), inconclusive_errors.end(), base_destructor);
1608                                 if (found != inconclusive_errors.end())
1609                                     inconclusive_errors.erase(found);
1610                             }
1611                         }
1612                     }
1613                 }
1614             }
1615         }
1616     
1617         for (std::list<const Function *>::const_iterator i = inconclusive_errors.begin(); i != inconclusive_errors.end(); ++i)
1618             virtualDestructorError((*i)->tokenDef, (*i)->name(), "", true);
1619     }

1661     // can member function be const?
1662     //---------------------------------------------------------------------------
1663     
1664     void CheckClass::checkConst()
1665     {
1666         // This is an inconclusive check. False positives: #3322.
1667         if (!_settings->inconclusive)
1668             return;
1669     
1670         if (!_settings->isEnabled("style"))
1671             return;
1672     
1673         const std::size_t classes = symbolDatabase->classAndStructScopes.size();
1674         for (std::size_t i = 0; i < classes; ++i) {
1675             const Scope * scope = symbolDatabase->classAndStructScopes[i];
1676             std::list<Function>::const_iterator func;
1677     
1678             for (func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
1679                 // does the function have a body?
1680                 if (func->type == Function::eFunction && func->hasBody() && !func->isFriend() && !func->isStatic() && !func->isVirtual()) {
1681                     // get last token of return type
1682                     const Token *previous = func->tokenDef->previous();
1683     
1684                     // does the function return a pointer or reference?
1685                     if (Token::Match(previous, "*|&")) {
1686                         if (func->retDef->str() != "const")
1687                             continue;
1688                     } else if (Token::Match(previous->previous(), "*|& >")) {
1689                         const Token *temp = previous->previous();
1690     
1691                         bool foundConst = false;
1692                         while (!Token::Match(temp->previous(), ";|}|{|public:|protected:|private:")) {
1693                             temp = temp->previous();
1694                             if (temp->str() == "const") {
1695                                 foundConst = true;
1696                                 break;
1697                             }
1698                         }
1699     
1700                         if (!foundConst)
1701                             continue;
1702                     } else if (func->isOperator() && Token::Match(previous, ";|{|}|public:|private:|protected:")) { // Operator without return type: conversion operator
1703                         const std::string& opName = func->tokenDef->str();
1704                         if (opName.compare(8, 5, "const") != 0 && opName.back() == '&')
1705                             continue;
1706                     } else {
1707                         // don't warn for unknown types..
1708                         // LPVOID, HDC, etc
1709                         if (previous->str().size() > 2 && !previous->type() && previous->isUpperCaseName())
1710                             continue;
1711                     }
1712     
1713                     // check if base class function is virtual
1714                     if (!scope->definedType->derivedFrom.empty()) {
1715                         if (func->isImplicitlyVirtual(true))
1716                             continue;
1717                     }
1718     
1719                     bool memberAccessed = false;
1720                     // if nothing non-const was found. write error..
1721                     if (checkConstFunc(scope, &*func, memberAccessed)) {
1722                         std::string classname = scope->className;
1723                         const Scope *nest = scope->nestedIn;
1724                         while (nest && nest->type != Scope::eGlobal) {
1725                             classname = std::string(nest->className + "::" + classname);
1726                             nest = nest->nestedIn;
1727                         }
1728     
1729                         // get function name
1730                         std::string functionName = (func->tokenDef->isName() ? "" : "operator") + func->tokenDef->str();
1731     
1732                         if (func->tokenDef->str() == "(")
1733                             functionName += ")";
1734                         else if (func->tokenDef->str() == "[")
1735                             functionName += "]";
1736     
1737                         if (!func->isConst() || (!memberAccessed && !func->isOperator())) {
1738                             if (func->isInline())
1739                                 checkConstError(func->token, classname, functionName, !memberAccessed && !func->isOperator());
1740                             else // not inline

1749     bool CheckClass::isMemberVar(const Scope *scope, const Token *tok) const
1750     {
1751         bool again = false;
1752     
1753         // try to find the member variable
1754         do {
1755             again = false;
1756     
1757             if (tok->str() == "this") {
1758                 return true;
1759             } else if (Token::simpleMatch(tok->tokAt(-3), "( * this )")) {
1760                 return true;
1761             } else if (Token::Match(tok->tokAt(-2), "%name% . %name%")) {
1762                 tok = tok->tokAt(-2);
1763                 again = true;
1764             } else if (Token::Match(tok->tokAt(-2), "] . %name%")) {
1765                 tok = tok->linkAt(-2)->previous();
1766                 again = true;
1767             } else if (tok->str() == "]") {
1768                 tok = tok->link()->previous();
1769                 again = true;
1770             }
1771         } while (again);
1772     
1773         std::list<Variable>::const_iterator var;
1774         for (var = scope->varlist.begin(); var != scope->varlist.end(); ++var) {
1775             if (var->name() == tok->str()) {
1776                 if (tok->varId() == 0)
1777                     symbolDatabase->debugMessage(tok, "CheckClass::isMemberVar found used member variable \'" + tok->str() + "\' with varid 0");
1778     
1779                 return !var->isStatic();
1780             }
1781         }
1782     
1783         // not found in this class
1784         if (!scope->definedType->derivedFrom.empty()) {
1785             // check each base class
1786             for (std::size_t i = 0; i < scope->definedType->derivedFrom.size(); ++i) {
1787                 // find the base class
1788                 const Type *derivedFrom = scope->definedType->derivedFrom[i].type;
1789     
1790                 // find the function in the base class
1791                 if (derivedFrom && derivedFrom->classScope) {
1792                     if (isMemberVar(derivedFrom->classScope, tok))
1793                         return true;
1794                 }
1795             }
1796         }
1797     
1798         return false;
1799     }

1852     bool CheckClass::checkConstFunc(const Scope *scope, const Function *func, bool& memberAccessed) const
1853     {
1854         // if the function doesn't have any assignment nor function call,
1855         // it can be a const function..
1856         for (const Token *tok1 = func->functionScope->classStart; tok1 && tok1 != func->functionScope->classEnd; tok1 = tok1->next()) {
1857             if (tok1->isName() && isMemberVar(scope, tok1)) {
1858                 memberAccessed = true;
1859                 const Variable* v = tok1->variable();
1860                 if (v && v->isMutable())
1861                     continue;
1862     
1863                 if (tok1->str() == "this" && tok1->previous()->isAssignmentOp())
1864                     return false;
1865     
1866     
1867                 const Token* lhs = tok1->previous();
1868                 if (lhs->str() == "&") {
1869                     lhs = lhs->previous();
1870                     if (lhs->tokType() == Token::eAssignmentOp && lhs->previous()->variable()) {
1871                         if (lhs->previous()->variable()->typeStartToken()->strAt(-1) != "const" && lhs->previous()->variable()->isPointer())
1872                             return false;
1873                     }
1874                 } else if (lhs->str() == ":" && lhs->astParent() && lhs->astParent()->str() == "(" && tok1->strAt(1) == ")") { // range-based for-loop (C++11)
1875                     // TODO: We could additionally check what is done with the elements to avoid false negatives. Here we just rely on "const" keyword being used.
1876                     if (lhs->astParent()->strAt(1) != "const")
1877                         return false;
1878                 } else {
1879                     if (lhs->tokType() == Token::eAssignmentOp) {
1880                         const Variable* lhsVar = lhs->previous()->variable();
1881                         if (lhsVar && !lhsVar->isConst() && lhsVar->isReference() && lhs == lhsVar->nameToken()->next())
1882                             return false;
1883                     }
1884                 }
1885     
1886                 const Token* jumpBackToken = nullptr;
1887                 const Token *lastVarTok = tok1;
1888                 const Token *end = tok1;
1889                 for (;;) {
1890                     if (Token::Match(end->next(), ". %name%")) {
1891                         end = end->tokAt(2);
1892                         if (end->varId())
1893                             lastVarTok = end;
1894                     } else if (end->strAt(1) == "[") {
1895                         if (end->varId()) {
1896                             const Variable *var = end->variable();
1897                             if (var && var->isStlType(stl_containers_not_const))
1898                                 return false;
1899                         }
1900                         if (!jumpBackToken)
1901                             jumpBackToken = end->next(); // Check inside the [] brackets
1902                         end = end->linkAt(1);
1903                     } else if (end->strAt(1) == ")")
1904                         end = end->next();
1905                     else
1906                         break;
1907                 }
1908     
1909                 if (end->strAt(1) == "(") {
1910                     const Variable *var = lastVarTok->variable();
1911                     if (!var)
1912                         return false;
1913                     if (var->isStlType() // assume all std::*::size() and std::*::empty() are const
1914                         && (Token::Match(end, "size|empty|cend|crend|cbegin|crbegin|max_size|length|count|capacity|get_allocator|c_str|str ( )") || Token::Match(end, "rfind|copy")))
1915                         ;
1916                     else if (!var->typeScope() || !isConstMemberFunc(var->typeScope(), end))
1917                         return false;
1918                 }
1919     
1920                 // Assignment
1921                 else if (end->next()->tokType() == Token::eAssignmentOp)
1922                     return false;
1923     
1924                 // Streaming
1925                 else if (end->strAt(1) == "<<" && tok1->strAt(-1) != "<<")
1926                     return false;
1927                 else if (tok1->strAt(-1) == ">>")
1928                     return false;
1929     
1930                 // ++/--
1931                 else if (end->next()->tokType() == Token::eIncDecOp || tok1->previous()->tokType() == Token::eIncDecOp)
1932                     return false;
1933     
1934     
1935                 const Token* start = tok1;
1936                 while (tok1->strAt(-1) == ")")
1937                     tok1 = tok1->linkAt(-1);
1938     
1939                 if (start->strAt(-1) == "delete")
1940                     return false;
1941     
1942                 tok1 = jumpBackToken?jumpBackToken:end; // Jump back to first [ to check inside, or jump to end of expression
1943             }
1944     
1945             // streaming: <<
1946             else if (Token::simpleMatch(tok1->previous(), ") <<") &&
1947                      isMemberVar(scope, tok1->tokAt(-2))) {
1948                 const Variable* var = tok1->tokAt(-2)->variable();
1949                 if (!var || !var->isMutable())
1950                     return false;
1951             }
1952     
1953     
1954             // function call..
1955             else if (Token::Match(tok1, "%name% (") && !tok1->isStandardType() &&
1956                      !Token::Match(tok1, "return|if|string|switch|while|catch|for")) {
1957                 if (isMemberFunc(scope, tok1) && tok1->strAt(-1) != ".") {
1958                     if (!isConstMemberFunc(scope, tok1))
1959                         return false;
1960                     memberAccessed = true;
1961                 }
1962                 // Member variable given as parameter
1963                 for (const Token* tok2 = tok1->tokAt(2); tok2 && tok2 != tok1->next()->link(); tok2 = tok2->next()) {
1964                     if (tok2->str() == "(")
1965                         tok2 = tok2->link();
1966                     else if (tok2->isName() && isMemberVar(scope, tok2)) {
1967                         const Variable* var = tok2->variable();
1968                         if (!var || !var->isMutable())
1969                             return false; // TODO: Only bailout if function takes argument as non-const reference
1970                     }
1971                 }
1972             } else if (Token::simpleMatch(tok1, "> (") && (!tok1->link() || !Token::Match(tok1->link()->previous(), "static_cast|const_cast|dynamic_cast|reinterpret_cast"))) {
1973                 return false;
1974             }
1975         }
1976     
1977         return true;
1978     }

2023     void CheckClass::initializerListOrder()
2024     {
2025         if (!_settings->isEnabled("style"))
2026             return;
2027     
2028         // This check is not inconclusive.  However it only determines if the initialization
2029         // order is incorrect.  It does not determine if being out of order causes
2030         // a real error.  Out of order is not necessarily an error but you can never
2031         // have an error if the list is in order so this enforces defensive programming.
2032         if (!_settings->inconclusive)
2033             return;
2034     
2035         const std::size_t classes = symbolDatabase->classAndStructScopes.size();
2036         for (std::size_t i = 0; i < classes; ++i) {
2037             const Scope * scope = symbolDatabase->classAndStructScopes[i];
2038             std::list<Function>::const_iterator func;
2039     
2040             // iterate through all member functions looking for constructors
2041             for (func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
2042                 if ((func->isConstructor()) && func->hasBody()) {
2043                     // check for initializer list
2044                     const Token *tok = func->arg->link()->next();
2045     
2046                     if (tok->str() == ":") {
2047                         std::vector<VarInfo> vars;
2048                         tok = tok->next();
2049     
2050                         // find all variable initializations in list
2051                         while (tok && tok != func->functionScope->classStart) {
2052                             if (Token::Match(tok, "%name% (|{")) {
2053                                 const Variable *var = scope->getVariable(tok->str());
2054                                 if (var)
2055                                     vars.push_back(VarInfo(var, tok));
2056     
2057                                 if (Token::Match(tok->tokAt(2), "%name% =")) {
2058                                     var = scope->getVariable(tok->strAt(2));
2059     
2060                                     if (var)
2061                                         vars.push_back(VarInfo(var, tok->tokAt(2)));
2062                                 }
2063                                 tok = tok->next()->link()->next();
2064                             } else
2065                                 tok = tok->next();
2066                         }
2067     
2068                         // need at least 2 members to have out of order initialization
2069                         for (std::size_t j = 1; j < vars.size(); j++) {
2070                             // check for out of order initialization
2071                             if (vars[j].var->index() < vars[j - 1].var->index())
2072                                 initializerListError(vars[j].tok,vars[j].var->nameToken(), scope->className, vars[j].var->name());
2073                         }
2074                     }
2075                 }
2076             }
2077         }
2078     }
