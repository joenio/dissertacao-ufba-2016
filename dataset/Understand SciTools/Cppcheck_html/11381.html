33     //---------------------------------------------------------------------------
34     // Checking for shift by too many bits
35     //---------------------------------------------------------------------------
36     
37     void CheckType::checkTooBigBitwiseShift()
38     {
39         const bool printWarnings = _settings->isEnabled("warning");
40         const bool printInconclusive = _settings->inconclusive;
41     
42         // unknown sizeof(int) => can't run this checker
43         if (_settings->platformType == Settings::Unspecified)
44             return;
45     
46         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
47         const std::size_t functions = symbolDatabase->functionScopes.size();
48         for (std::size_t i = 0; i < functions; ++i) {
49             const Scope * scope = symbolDatabase->functionScopes[i];
50             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
51                 if (!Token::Match(tok, "<<|>>|<<=|>>="))
52                     continue;
53     
54                 if (!tok->astOperand1() || !tok->astOperand2())
55                     continue;
56     
57                 // get number of bits of lhs
58                 const ValueType *lhstype = tok->astOperand1()->valueType();
59                 if (!lhstype || !lhstype->isIntegral() || lhstype->pointer >= 1U)
60                     continue;
61                 int lhsbits = 0;
62                 if (lhstype->type <= ValueType::Type::INT)
63                     lhsbits = _settings->int_bit;
64                 else if (lhstype->type == ValueType::Type::LONG)
65                     lhsbits = _settings->long_bit;
66                 else if (lhstype->type == ValueType::Type::LONGLONG)
67                     lhsbits = _settings->long_long_bit;
68                 else
69                     continue;
70     
71                 // Get biggest rhs value. preferably a value which doesn't have 'condition'.
72                 const ValueFlow::Value *value = tok->astOperand2()->getValueGE(lhsbits, _settings);
73                 if (!value)
74                     continue;
75                 if (value->condition && !printWarnings)
76                     continue;
77                 if (value->inconclusive && !printInconclusive)
78                     continue;
79                 tooBigBitwiseShiftError(tok, lhsbits, *value);
80             }
81         }
82     }

84     void CheckType::tooBigBitwiseShiftError(const Token *tok, int lhsbits, const ValueFlow::Value &rhsbits)
85     {
86         std::list<const Token*> callstack;
87         callstack.push_back(tok);
88         if (rhsbits.condition)
89             callstack.push_back(rhsbits.condition);
90         std::ostringstream errmsg;
91         errmsg << "Shifting " << lhsbits << "-bit value by " << rhsbits.intvalue << " bits is undefined behaviour";
92         if (rhsbits.condition)
93             errmsg << ". See condition at line " << rhsbits.condition->linenr() << ".";
94         reportError(callstack, rhsbits.condition ? Severity::warning : Severity::error, "shiftTooManyBits", errmsg.str(), 0U, rhsbits.inconclusive);
95     }

97     //---------------------------------------------------------------------------
98     // Checking for integer overflow
99     //---------------------------------------------------------------------------
100     
101     void CheckType::checkIntegerOverflow()
102     {
103         // unknown sizeof(int) => can't run this checker
104         if (_settings->platformType == Settings::Unspecified)
105             return;
106     
107         // max int value according to platform settings.
108         const MathLib::bigint maxint = (1LL << (_settings->int_bit - 1)) - 1;
109     
110         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
111         const std::size_t functions = symbolDatabase->functionScopes.size();
112         for (std::size_t i = 0; i < functions; ++i) {
113             const Scope * scope = symbolDatabase->functionScopes[i];
114             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
115                 if (!tok->isArithmeticalOp())
116                     continue;
117     
118                 // is there a overflow result value
119                 const ValueFlow::Value *value = tok->getValueGE(maxint + 1, _settings);
120                 if (!value)
121                     value = tok->getValueLE(-maxint - 2, _settings);
122                 if (!value)
123                     continue;
124     
125                 // is result signed integer?
126                 const ValueType *vt = tok->valueType();
127                 if (vt && vt->type == ValueType::Type::INT && vt->sign == ValueType::Sign::SIGNED)
128                     integerOverflowError(tok, *value);
129             }
130         }
131     }

133     void CheckType::integerOverflowError(const Token *tok, const ValueFlow::Value &value)
134     {
135         const std::string expr(tok ? tok->expressionString() : "");
136     
137         std::string msg;
138         if (value.condition)
139             msg = ValueFlow::eitherTheConditionIsRedundant(value.condition) +
140                   " or there is signed integer overflow for expression '" + expr + "'.";
141         else
142             msg = "Signed integer overflow for expression '" + expr + "'.";
143     
144         reportError(tok,
145                     value.condition ? Severity::warning : Severity::error,
146                     "integerOverflow",
147                     msg,
148                     0U,
149                     value.inconclusive);
150     }

152     //---------------------------------------------------------------------------
153     // Checking for sign conversion when operand can be negative
154     //---------------------------------------------------------------------------
155     
156     void CheckType::checkSignConversion()
157     {
158         if (!_settings->isEnabled("warning"))
159             return;
160     
161         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
162         const std::size_t functions = symbolDatabase->functionScopes.size();
163         for (std::size_t i = 0; i < functions; ++i) {
164             const Scope * scope = symbolDatabase->functionScopes[i];
165             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
166                 if (!tok->isArithmeticalOp() || Token::Match(tok,"+|-"))
167                     continue;
168     
169                 // Is result unsigned?
170                 if (!(tok->valueType() && tok->valueType()->sign == ValueType::Sign::UNSIGNED))
171                     continue;
172     
173                 // Check if an operand can be negative..
174                 std::stack<const Token *> tokens;
175                 tokens.push(tok->astOperand1());
176                 tokens.push(tok->astOperand2());
177                 while (!tokens.empty()) {
178                     const Token *tok1 = tokens.top();
179                     tokens.pop();
180                     if (!tok1)
181                         continue;
182                     if (!tok1->getValueLE(-1,_settings))
183                         continue;
184                     if (tok1->valueType() && tok1->valueType()->sign != ValueType::Sign::UNSIGNED)
185                         signConversionError(tok1, tok1->isNumber());
186                 }
187             }
188         }
189     }

191     void CheckType::signConversionError(const Token *tok, const bool constvalue)
192     {
193         const std::string varname(tok ? tok->str() : "var");
194     
195         reportError(tok,
196                     Severity::warning,
197                     "signConversion",
198                     (constvalue) ?
199                     "Suspicious code: sign conversion of " + varname + " in calculation because '" + varname + "' has a negative value" :
200                     "Suspicious code: sign conversion of " + varname + " in calculation, even though " + varname + " can have a negative value");
201     }

204     //---------------------------------------------------------------------------
205     // Checking for long cast of int result   const long x = var1 * var2;
206     //---------------------------------------------------------------------------
207     
208     void CheckType::checkLongCast()
209     {
210         if (!_settings->isEnabled("style"))
211             return;
212     
213         // Assignments..
214         for (const Token *tok = _tokenizer->tokens(); tok; tok = tok->next()) {
215             if (tok->str() != "=" || !Token::Match(tok->astOperand2(), "*|<<"))
216                 continue;
217     
218             const ValueType *lhstype = tok->astOperand1() ? tok->astOperand1()->valueType() : nullptr;
219             const ValueType *rhstype = tok->astOperand2()->valueType();
220     
221             if (!lhstype || !rhstype)
222                 continue;
223     
224             // assign int result to long/longlong const nonpointer?
225             if (rhstype->type == ValueType::Type::INT &&
226                 rhstype->pointer == 0U &&
227                 rhstype->originalTypeName.empty() &&
228                 (lhstype->type == ValueType::Type::LONG || lhstype->type == ValueType::Type::LONGLONG) &&
229                 lhstype->pointer == 0U &&
230                 lhstype->constness == 1U &&
231                 lhstype->originalTypeName.empty())
232                 longCastAssignError(tok);
233         }
234     
235         // Return..
236         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
237         const std::size_t functions = symbolDatabase->functionScopes.size();
238         for (std::size_t i = 0; i < functions; ++i) {
239             const Scope * scope = symbolDatabase->functionScopes[i];
240     
241             // function must return long data
242             const Token * def = scope->classDef;
243             bool islong = false;
244             while (Token::Match(def, "%type%|::")) {
245                 if (def->str() == "long" && def->originalName().empty()) {
246                     islong = true;
247                     break;
248                 }
249                 def = def->previous();
250             }
251             if (!islong)
252                 continue;
253     
254             // return statements
255             const Token *ret = nullptr;
256             for (const Token *tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
257                 if (tok->str() == "return") {
258                     if (Token::Match(tok->astOperand1(), "<<|*")) {
259                         const ValueType *type = tok->astOperand1()->valueType();
260                         if (type && type->type == ValueType::Type::INT && type->pointer == 0U && type->originalTypeName.empty())
261                             ret = tok;
262                     }
263                     // All return statements must have problem otherwise no warning
264                     if (ret != tok) {
265                         ret = nullptr;
266                         break;
267                     }
268                 }
269             }
270     
271             if (ret)
272                 longCastReturnError(ret);
273         }
274     }

276     void CheckType::longCastAssignError(const Token *tok)
277     {
278         reportError(tok,
279                     Severity::style,
280                     "truncLongCastAssignment",
281                     "int result is assigned to long variable. If the variable is long to avoid loss of information, then you have loss of information.\n"
282                     "int result is assigned to long variable. If the variable is long to avoid loss of information, then there is loss of information. To avoid loss of information you must cast a calculation operand to long, for example 'l = a * b;' => 'l = (long)a * b;'.");
283     }
