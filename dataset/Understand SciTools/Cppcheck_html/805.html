10     static void unencode(const char *src, char *dest)
11     {
12         for (; *src; src++, dest++) {
13             if (*src == '+')
14                 *dest = ' ';
15             else if (*src == '%') {
16                 int code;
17                 if (sscanf(src+1, "%2x", &code) != 1)
18                     code = '?';
19                 *dest = code;
20                 src += 2;
21             } else
22                 *dest = *src;
23         }
24         *dest = '\0';
25     }
