44         void check(const char code[], Settings* settings = 0) {
45             // Clear the error buffer..
46             errout.str("");
47     
48             if (!settings) {
49                 static Settings _settings;
50                 settings = &_settings;
51             }
52             settings->addEnabled("warning");
53     
54             // Tokenize..
55             Tokenizer tokenizer(settings, this);
56             std::istringstream istr(code);
57             tokenizer.tokenize(istr, "test.cpp");
58     
59             // Check..
60             CheckType checkType(&tokenizer, settings, this);
61             checkType.runChecks(&tokenizer, settings, this);
62         }

64         void checkTooBigShift() {
65             Settings settings;
66             settings.platform(Settings::Unix32);
67     
68             check("int foo(int x) {\n"
69                   "   return x << 32;\n"
70                   "}",&settings);
71             ASSERT_EQUALS("[test.cpp:2]: (error) Shifting 32-bit value by 32 bits is undefined behaviour\n", errout.str());
72     
73             check("int foo(int x) {\n"
74                   "   return x << 2;\n"
75                   "}",&settings);
76             ASSERT_EQUALS("", errout.str());
77     
78             check("int foo(int x) {\n"
79                   "   return (long long)x << 40;\n"
80                   "}",&settings);
81             ASSERT_EQUALS("", errout.str());
82     
83             check("void foo() {\n"
84                   "  QList<int> someList;\n"
85                   "  someList << 300;\n"
86                   "}", &settings);
87             ASSERT_EQUALS("", errout.str());
88     
89             // Ticket #6793
90             check("template<int I> int foo(int x) { return x << I; }\n"
91                   "const int f = foo<31>(1);\n"
92                   "const int g = foo<100>(1);\n"
93                   "template<int I> int hoo(int x) { return x << 32; }\n"
94                   "const int h = hoo<100>(1);", &settings);
95             ASSERT_EQUALS("[test.cpp:4]: (error) Shifting 32-bit value by 32 bits is undefined behaviour\n"
96                           "[test.cpp:1]: (error) Shifting 32-bit value by 100 bits is undefined behaviour\n", errout.str());
97         }

99         void checkIntegerOverflow() {
100             Settings settings;
101             settings.platform(Settings::Unix32);
102             settings.addEnabled("warning");
103     
104             check("int foo(signed int x) {\n"
105                   "   if (x==123456) {}\n"
106                   "   return x * x;\n"
107                   "}",&settings);
108             ASSERT_EQUALS("[test.cpp:3]: (warning) Either the condition 'x==123456' is redundant or there is signed integer overflow for expression 'x*x'.\n", errout.str());
109     
110             check("int foo(signed int x) {\n"
111                   "   if (x==123456) {}\n"
112                   "   return -123456 * x;\n"
113                   "}",&settings);
114             ASSERT_EQUALS("[test.cpp:3]: (warning) Either the condition 'x==123456' is redundant or there is signed integer overflow for expression '-123456*x'.\n", errout.str());
115     
116             check("int foo(signed int x) {\n"
117                   "   if (x==123456) {}\n"
118                   "   return 123456U * x;\n"
119                   "}",&settings);
120             ASSERT_EQUALS("", errout.str());
121         }

123         void signConversion() {
124             check("unsigned int f1(signed int x, unsigned int y) {" // x is signed
125                   "  return x * y;\n"
126                   "}\n"
127                   "void f2() { f1(-4,4); }");
128             ASSERT_EQUALS("[test.cpp:1]: (warning) Suspicious code: sign conversion of x in calculation, even though x can have a negative value\n", errout.str());
129     
130             check("unsigned int f1(int x) {" // x has no signedness, but it can have the value -1 so assume it's signed
131                   "  return x * 5U;\n"
132                   "}\n"
133                   "void f2() { f1(-4); }");
134             ASSERT_EQUALS("[test.cpp:1]: (warning) Suspicious code: sign conversion of x in calculation, even though x can have a negative value\n", errout.str());
135     
136             check("unsigned int f1(int x) {" // #6168: FP for inner calculation
137                   "  return 5U * (1234 - x);\n" // <- signed subtraction, x is not sign converted
138                   "}\n"
139                   "void f2() { f1(-4); }");
140             ASSERT_EQUALS("", errout.str());
141     
142             // Dont warn for + and -
143             check("void f1(int x) {"
144                   "  a = x + 5U;\n"
145                   "}\n"
146                   "void f2() { f1(-4); }");
147             ASSERT_EQUALS("", errout.str());
148     
149             check("size_t foo(size_t x) {\n"
150                   " return -2 * x;\n"
151                   "}");
152             ASSERT_EQUALS("[test.cpp:2]: (warning) Suspicious code: sign conversion of -2 in calculation because '-2' has a negative value\n", errout.str());
153         }

164             ASSERT_EQUALS("[test.cpp:2]: (style) int result is assigned to long variable. If the variable is long to avoid loss of information, then you have loss of information.\n", errout.str());
165     
166             // typedef
167             check("long f(int x, int y) {\n"
168                   "  const size_t ret = x * y;\n"
169                   "  return ret;\n"
170                   "}\n", &settings);
171             ASSERT_EQUALS("", errout.str());
172     
173             // astIsIntResult
174             check("long f(int x, int y) {\n"
175                   "  const long ret = (long)x * y;\n"
176                   "  return ret;\n"
177                   "}\n", &settings);

179         }
