20     #ifndef checkioH
21     #define checkioH

23     
24     #include "check.h"
25     #include "config.h"
26     
27     /// @addtogroup Checks
28     /// @{
29     
30     /** @brief %Check input output operations. */
31     class CPPCHECKLIB CheckIO : public Check {
32     public:
33         /** @brief This constructor is used when registering CheckIO */
34         CheckIO() : Check(myName()) {
35         }
36     
37         /** @brief This constructor is used when running checks. */
38         CheckIO(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
39             : Check(myName(), tokenizer, settings, errorLogger) {
40         }
41     
42         /** @brief Run checks on the normal token list */
43         void runChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
44             CheckIO checkIO(tokenizer, settings, errorLogger);
45     
46             checkIO.checkWrongPrintfScanfArguments();
47         }
48     
49         /** @brief Run checks on the simplified token list */
50         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
51             CheckIO checkIO(tokenizer, settings, errorLogger);
52     
53             checkIO.checkCoutCerrMisusage();
54             checkIO.checkFileUsage();
55             checkIO.invalidScanf();
56         }
57     
58         /** @brief %Check for missusage of std::cout */
59         void checkCoutCerrMisusage();
60     
61         /** @brief %Check usage of files*/
62         void checkFileUsage();
63     
64         /** @brief scanf can crash if width specifiers are not used */
65         void invalidScanf();
66     
67         /** @brief %Checks type and number of arguments given to functions like printf or scanf*/
68         void checkWrongPrintfScanfArguments();
69     
70     private:
71         class ArgumentInfo {
72         public:
73             ArgumentInfo(const Token *arg, const Settings *settings, bool isCPP);
74             ~ArgumentInfo();
75     
76             bool isArrayOrPointer() const;
77             bool isComplexType() const;
78             bool isKnownType() const;
79             bool isStdVectorOrString();
80             bool isStdContainer(const Token *tok);
81             bool isLibraryType(const Settings *settings) const;
82     
83             const Variable *variableInfo;
84             const Token *typeToken;
85             const Function *functionInfo;
86             Token *tempToken;
87             bool element;
88             bool _template;
89             bool address;
90             bool isCPP;
91     
92         private:
93             ArgumentInfo(const ArgumentInfo &); // not implemented
94             ArgumentInfo operator = (const ArgumentInfo &); // not implemented
95         };
96     
97         void checkFormatString(const Token * const tok,
98                                const Token * const formatStringTok,
99                                const Token *       argListTok,
100                                const bool scan,
101                                const bool scanf_s);
102     
103         // Reporting errors..
104         void coutCerrMisusageError(const Token* tok, const std::string& streamName);
105         void fflushOnInputStreamError(const Token *tok, const std::string &varname);
106         void ioWithoutPositioningError(const Token *tok);
107         void readWriteOnlyFileError(const Token *tok);
108         void writeReadOnlyFileError(const Token *tok);
109         void useClosedFileError(const Token *tok);
110         void seekOnAppendedFileError(const Token *tok);
111         void invalidScanfError(const Token *tok);
112         void wrongPrintfScanfArgumentsError(const Token* tok,
113                                             const std::string &function,
114                                             unsigned int numFormat,
115                                             unsigned int numFunction);
116         void wrongPrintfScanfPosixParameterPositionError(const Token* tok, const std::string& functionName,
117                 unsigned int index, unsigned int numFunction);
118         void invalidScanfArgTypeError_s(const Token* tok, unsigned int numFormat, const std::string& specifier, const ArgumentInfo* argInfo);
119         void invalidScanfArgTypeError_int(const Token* tok, unsigned int numFormat, const std::string& specifier, const ArgumentInfo* argInfo, bool isUnsigned);
120         void invalidScanfArgTypeError_float(const Token* tok, unsigned int numFormat, const std::string& specifier, const ArgumentInfo* argInfo);
121         void invalidPrintfArgTypeError_s(const Token* tok, unsigned int numFormat, const ArgumentInfo* argInfo);
122         void invalidPrintfArgTypeError_n(const Token* tok, unsigned int numFormat, const ArgumentInfo* argInfo);
123         void invalidPrintfArgTypeError_p(const Token* tok, unsigned int numFormat, const ArgumentInfo* argInfo);
124         void invalidPrintfArgTypeError_int(const Token* tok, unsigned int numFormat, const std::string& specifier, const ArgumentInfo* argInfo);
125         void invalidPrintfArgTypeError_uint(const Token* tok, unsigned int numFormat, const std::string& specifier, const ArgumentInfo* argInfo);
126         void invalidPrintfArgTypeError_sint(const Token* tok, unsigned int numFormat, const std::string& specifier, const ArgumentInfo* argInfo);
127         void invalidPrintfArgTypeError_float(const Token* tok, unsigned int numFormat, const std::string& specifier, const ArgumentInfo* argInfo);
128         void invalidLengthModifierError(const Token* tok, unsigned int numFormat, const std::string& modifier);
129         void invalidScanfFormatWidthError(const Token* tok, unsigned int numFormat, int width, const Variable *var);
130         static void argumentType(std::ostream & s, const ArgumentInfo * argInfo);
131     
132         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
133             CheckIO c(0, settings, errorLogger);
134     
135             c.coutCerrMisusageError(0, "cout");
136             c.fflushOnInputStreamError(0, "stdin");
137             c.ioWithoutPositioningError(0);
138             c.readWriteOnlyFileError(0);
139             c.writeReadOnlyFileError(0);
140             c.useClosedFileError(0);
141             c.seekOnAppendedFileError(0);
142             c.invalidScanfError(0);
143             c.wrongPrintfScanfArgumentsError(0,"printf",3,2);
144             c.invalidScanfArgTypeError_s(0, 1, "s", NULL);
145             c.invalidScanfArgTypeError_int(0, 1, "d", NULL, false);
146             c.invalidScanfArgTypeError_float(0, 1, "f", NULL);
147             c.invalidPrintfArgTypeError_s(0, 1, NULL);
148             c.invalidPrintfArgTypeError_n(0, 1, NULL);
149             c.invalidPrintfArgTypeError_p(0, 1, NULL);
150             c.invalidPrintfArgTypeError_int(0, 1, "X", NULL);
151             c.invalidPrintfArgTypeError_uint(0, 1, "u", NULL);
152             c.invalidPrintfArgTypeError_sint(0, 1, "i", NULL);
153             c.invalidPrintfArgTypeError_float(0, 1, "f", NULL);
154             c.invalidLengthModifierError(0, 1, "I");
155             c.invalidScanfFormatWidthError(0, 10, 5, NULL);
156             c.wrongPrintfScanfPosixParameterPositionError(0, "printf", 2, 1);
157         }
158     
159         static std::string myName() {
160             return "IO using format string";
161         }
162     
163         std::string classInfo() const {
164             return "Check format string input/output operations.\n"
165                    "- Bad usage of the function 'sprintf' (overlapping data)\n"
166                    "- Missing or wrong width specifiers in 'scanf' format string\n"
167                    "- Use a file that has been closed\n"
168                    "- File input/output without positioning results in undefined behaviour\n"
169                    "- Read to a file that has only been opened for writing (or vice versa)\n"
170                    "- Repositioning operation on a file opened in append mode\n"
171                    "- Using fflush() on an input stream\n"
172                    "- Invalid usage of output stream. For example: 'std::cout << std::cout;'\n"
173                    "- Wrong number of arguments given to 'printf' or 'scanf;'\n";
174         }
175     };
176     /// @}
177     //---------------------------------------------------------------------------
178     #endif // checkioH
