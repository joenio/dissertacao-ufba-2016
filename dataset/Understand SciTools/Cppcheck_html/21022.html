26     QString GetPath(const QString &type)
27     {
28         QSettings settings;
29         QString path = settings.value(type, "").toString();
30         if (path.isEmpty()) {
31             // if not set, fallback to last check path hoping that it will be close enough
32             path = settings.value(SETTINGS_LAST_CHECK_PATH, "").toString();
33             if (path.isEmpty())
34                 // if not set, return user's home directory as the best we can do for now
35                 return QDir::homePath();
36         }
37         return path;
38     }
39     
40     void SetPath(const QString &type, const QString &value)
41     {
42         QSettings settings;
43         settings.setValue(type, value);
44     }
