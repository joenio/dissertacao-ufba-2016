137     void ResultsView::Save(const QString &filename, Report::Type type) const
138     {
139         if (!HasResults()) {
140             QMessageBox msgBox;
141             msgBox.setText(tr("No errors found, nothing to save."));
142             msgBox.setIcon(QMessageBox::Critical);
143             msgBox.exec();
144         }
145     
146         Report *report = NULL;
147     
148         switch (type) {
149         case Report::CSV:
150             report = new CsvReport(filename);
151             break;
152         case Report::TXT:
153             report = new TxtReport(filename);
154             break;
155         case Report::XML:
156             report = new XmlReportV1(filename);
157             break;
158         case Report::XMLV2:
159             report = new XmlReportV2(filename);
160             break;
161         }
162     
163         if (report) {
164             if (report->Create())
165                 mUI.mTree->SaveResults(report);
166             else {
167                 QMessageBox msgBox;
168                 msgBox.setText(tr("Failed to save the report."));
169                 msgBox.setIcon(QMessageBox::Critical);
170                 msgBox.exec();
171             }
172             delete report;
173             report = NULL;
174         } else {
175             QMessageBox msgBox;
176             msgBox.setText(tr("Failed to save the report."));
177             msgBox.setIcon(QMessageBox::Critical);
178             msgBox.exec();
179         }
180     }

201     void ResultsView::Print(QPrinter* printer)
202     {
203         if (!HasResults()) {
204             QMessageBox msgBox;
205             msgBox.setText(tr("No errors found, nothing to print."));

241     void ResultsView::CheckingFinished()
242     {
243         mUI.mProgress->setVisible(false);
244         mUI.mProgress->setFormat("%p%");
245     
246         //Should we inform user of non visible/not found errors?
247         if (mShowNoErrorsMessage) {
248             //Tell user that we found no errors
249             if (!HasResults()) {
250                 QMessageBox msg(QMessageBox::Information,
251                                 tr("Cppcheck"),
252                                 tr("No errors found."),
253                                 QMessageBox::Ok,
254                                 this);
255     
256                 msg.exec();
257             } //If we have errors but they aren't visible, tell user about it
258             else if (!mUI.mTree->HasVisibleResults()) {
259                 QString text = tr("Errors were found, but they are configured to be hidden.\n"
260                                   "To toggle what kind of errors are shown, open view menu.");
261                 QMessageBox msg(QMessageBox::Information,
262                                 tr("Cppcheck"),
263                                 text,
264                                 QMessageBox::Ok,
265                                 this);
266     
267                 msg.exec();
268             }
269         }
270     }

300     void ResultsView::ReadErrorsXml(const QString &filename)
301     {
302         const int version = XmlReport::determineVersion(filename);
303         if (version == 0) {
304             QMessageBox msgBox;
305             msgBox.setText(tr("Failed to read the report."));
306             msgBox.setIcon(QMessageBox::Critical);
307             msgBox.exec();
308             return;
309         }
310     
311         XmlReport *report = NULL;
312         if (version == 1)
313             report = new XmlReportV1(filename);
314         else if (version == 2)
315             report = new XmlReportV2(filename);
316     
317         QList<ErrorItem> errors;
318         if (report) {
319             if (report->Open())
320                 errors = report->Read();
321             else {
322                 QMessageBox msgBox;
323                 msgBox.setText(tr("Failed to read the report."));
324                 msgBox.setIcon(QMessageBox::Critical);
325                 msgBox.exec();
326             }
327             delete report;
328             report = NULL;
329         } else {
330             QMessageBox msgBox;
331             msgBox.setText(tr("Failed to read the report."));
332             msgBox.setIcon(QMessageBox::Critical);
333             msgBox.exec();
334         }
335     
336         ErrorItem item;
337         foreach (item, errors) {
338             mUI.mTree->AddErrorItem(item);
339         }
340         mUI.mTree->SetCheckDirectory("");
341     }

343     void ResultsView::UpdateDetails(const QModelIndex &index)
344     {
345         QStandardItemModel *model = qobject_cast<QStandardItemModel*>(mUI.mTree->model());
346         QStandardItem *item = model->itemFromIndex(index);
347     
348         if (!item) {
349             mUI.mDetails->setText("");
350             return;
351         }
352     
353         // Make sure we are working with the first column
354         if (item->parent() && item->column() != 0)
355             item = item->parent()->child(item->row(), 0);
356     
357         QVariantMap data = item->data().toMap();
358     
359         // If there is no severity data then it is a parent item without summary and message
360         if (!data.contains("severity")) {
361             mUI.mDetails->setText("");
362             return;
363         }
364     
365         const QString summary = data["summary"].toString();
366         const QString message = data["message"].toString();
367         QString formattedMsg = QString("%1: %2\n%3: %4")
368                                .arg(tr("Summary")).arg(summary)
369                                .arg(tr("Message")).arg(message);
370     
371         const QString file0 = data["file0"].toString();
372         if (file0 != "" && file0 != data["file"].toString())
373             formattedMsg += QString("\n\n%1: %2").arg(tr("First included by")).arg(QDir::toNativeSeparators(file0));
374     
375         if (mUI.mTree->ShowIdColumn())
376             formattedMsg.prepend(tr("Id") + ": " + data["id"].toString() + "\n");
377         mUI.mDetails->setText(formattedMsg);
378     }
