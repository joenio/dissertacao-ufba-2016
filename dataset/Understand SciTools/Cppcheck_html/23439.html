19     #include <QtTest>
20     #include <QObject>
21     #include "testxmlreportv2.h"
22     #include "xmlreportv2.h"
23     #include "erroritem.h"
24     #include "errorlogger.h"
25     
26     void TestXmlReportV2::readXml()
27     {
28         const QString filepath(QString(SRCDIR) + "/../data/xmlfiles/xmlreport_v2.xml");
29         XmlReportV2 report(filepath);
30         QVERIFY(report.Open());
31         QList<ErrorItem> errors = report.Read();
32         QCOMPARE(errors.size(), 6);
33     
34         ErrorItem item = errors[0];
35         QCOMPARE(item.file, QString("test.cxx"));
36         QCOMPARE(item.files.size(), 1);
37         QCOMPARE(item.lines.size(), 1);
38         QCOMPARE(item.files[0], QString("test.cxx"));
39         QCOMPARE(item.lines[0], (unsigned int)11);
40         QCOMPARE(item.errorId, QString("unreadVariable"));
41         QCOMPARE(GuiSeverity::toString(item.severity), QString("style"));
42         QCOMPARE(item.summary, QString("Variable 'a' is assigned a value that is never used"));
43         QCOMPARE(item.message, QString("Variable 'a' is assigned a value that is never used"));
44     
45         ErrorItem item2 = errors[3];
46         QCOMPARE(item2.file, QString("test.cxx"));
47         QCOMPARE(item2.files.size(), 2);
48         QCOMPARE(item2.lines.size(), 2);
49         QCOMPARE(item2.files[0], QString("test.cxx"));
50         QCOMPARE(item2.lines[0], (unsigned int)32);
51         QCOMPARE(item2.files[1], QString("test.cxx"));
52         QCOMPARE(item2.lines[1], (unsigned int)16);
53         QCOMPARE(item2.errorId, QString("mismatchAllocDealloc"));
54         QCOMPARE(GuiSeverity::toString(item2.severity), QString("error"));
55         QCOMPARE(item2.summary, QString("Mismatching allocation and deallocation: k"));
56         QCOMPARE(item2.message, QString("Mismatching allocation and deallocation: k"));
57     }
58     
59     QTEST_MAIN(TestXmlReportV2)
