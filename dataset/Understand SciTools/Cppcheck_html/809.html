20     #ifndef cppcheckH
21     #define cppcheckH
22     //---------------------------------------------------------------------------
23     
24     #include "config.h"
25     #include "settings.h"
26     #include "errorlogger.h"
27     #include "check.h"
28     
29     #include <string>
30     #include <list>
31     #include <istream>
32     
33     class Tokenizer;
34     
35     /// @addtogroup Core
36     /// @{
37     
38     /**
39      * @brief This is the base class which will use other classes to do
40      * static code analysis for C and C++ code to find possible
41      * errors or places that could be improved.
42      * Usage: See check() for more info.
43      */
44     class CPPCHECKLIB CppCheck : ErrorLogger {
45     public:
46         /**
47          * @brief Constructor.
48          */
49         CppCheck(ErrorLogger &errorLogger, bool useGlobalSuppressions);
50     
51         /**
52          * @brief Destructor.
53          */
54         virtual ~CppCheck();
55     
56         /**
57          * @brief This starts the actual checking. Note that you must call
58          * parseFromArgs() or settings() and addFile() before calling this.
59          * @return amount of errors found or 0 if none were found.
60          */
61     
62         /**
63           * @brief Check the file.
64           * This function checks one given file for errors.
65           * @param path Path to the file to check.
66           * @return amount of errors found or 0 if none were found.
67           * @note You must set settings before calling this function (by calling
68           *  settings()).
69           */
70         unsigned int check(const std::string &path);
71     
72         /**
73           * @brief Check the file.
74           * This function checks one "virtual" file. The file is not read from
75           * the disk but the content is given in @p content. In errors the @p path
76           * is used as a filename.
77           * @param path Path to the file to check.
78           * @param content File content as a string.
79           * @return amount of errors found or 0 if none were found.
80           * @note You must set settings before calling this function (by calling
81           *  settings()).
82           */
83         unsigned int check(const std::string &path, const std::string &content);
84     
85         /**
86          * @brief Get reference to current settings.
87          * @return a reference to current settings
88          */
89         Settings &settings();
90     
91         /**
92          * @brief Returns current version number as a string.
93          * @return version, e.g. "1.38"
94          */
95         static const char * version();
96     
97         /**
98          * @brief Returns extra version info as a string.
99          * This is for returning extra version info, like Git commit id, build
100          * time/date etc.
101          * @return extra version info, e.g. "04d42151" (Git commit id).
102          */
103         static const char * extraVersion();
104     
105         virtual void reportStatus(unsigned int fileindex, unsigned int filecount, std::size_t sizedone, std::size_t sizetotal);
106     
107         /**
108          * @brief Terminate checking. The checking will be terminated as soon as possible.
109          */
110         void terminate() {
111             _settings.terminate();
112         }
113     
114         /**
115          * @brief Call all "getErrorMessages" in all registered Check classes.
116          * Also print out XML header and footer.
117          */
118         void getErrorMessages();
119     
120         void tooManyConfigsError(const std::string &file, const std::size_t numberOfConfigurations);
121         void purgedConfigurationMessage(const std::string &file, const std::string& configuration);
122     
123         void dontSimplify() {
124             _simplify = false;
125         }
126     
127         /** analyse whole program, run this after all TUs has been scanned. */
128         void analyseWholeProgram();
129     
130         /** Check if the user wants to check for unused functions
131          * and if it's possible at all */
132         bool unusedFunctionCheckIsEnabled() const;
133     
134     private:
135     
136         /** @brief There has been an internal error => Report information message */
137         void internalError(const std::string &filename, const std::string &msg);
138     
139         /**
140          * @brief Process one file.
141          * @param filename file name
142          * @param fileStream stream the file content can be read from
143          * @return amount of errors found
144          */
145         unsigned int processFile(const std::string& filename, std::istream& fileStream);
146     
147         /**
148          * @brief Check raw tokens
149          * @param tokenizer
150          */
151         void checkRawTokens(const Tokenizer &tokenizer);
152     
153         /**
154          * @brief Check normal tokens
155          * @param tokenizer
156          */
157         void checkNormalTokens(const Tokenizer &tokenizer);
158     
159         /**
160          * @brief Check simplified tokens
161          * @param tokenizer
162          */
163         void checkSimplifiedTokens(const Tokenizer &tokenizer);
164     
165         /**
166          * @brief Execute rules, if any
167          * @param tokenlist token list to use (normal / simple)
168          * @param tokenizer tokenizer
169          */
170         void executeRules(const std::string &tokenlist, const Tokenizer &tokenizer);
171     
172         /**
173          * @brief Errors and warnings are directed here.
174          *
175          * @param msg Errors messages are normally in format
176          * "[filepath:line number] Message", e.g.
177          * "[main.cpp:4] Uninitialized member variable"
178          */
179         virtual void reportErr(const ErrorLogger::ErrorMessage &msg);
180     
181         /**
182          * @brief Information about progress is directed here.
183          *
184          * @param outmsg Message to show, e.g. "Checking main.cpp..."
185          */
186         virtual void reportOut(const std::string &outmsg);
187     
188         std::list<std::string> _errorList;
189         Settings _settings;
190     
191         void reportProgress(const std::string &filename, const char stage[], const std::size_t value);
192     
193         /**
194          * Output information messages.
195          */
196         virtual void reportInfo(const ErrorLogger::ErrorMessage &msg);
197     
198         ErrorLogger &_errorLogger;
199     
200         /** @brief Current preprocessor configuration */
201         std::string cfg;
202     
203         unsigned int exitcode;
204     
205         bool _useGlobalSuppressions;
206     
207         /** Are there too many configs? */
208         bool tooManyConfigs;
209     
210         /** Simplify code? true by default */
211         bool _simplify;
212     
213         /** File info used for whole program analysis */
214         std::list<Check::FileInfo*> fileInfo;
215     };
216     
217     /// @}
218     //---------------------------------------------------------------------------
219     #endif // cppcheckH
