43         // inspired by #5895
44         void testaddtoken() {
45             const std::string code = "0x89504e470d0a1a0a";
46             TokenList tokenlist(&settings);
47             tokenlist.addtoken(code, 1, 1, false);
48             ASSERT_EQUALS("9894494448401390090", tokenlist.front()->str());
49             // that is supposed to break on 32bit
50             //unsigned long numberUL(0);
51             //std::istringstream(tokenlist.front()->str()) >> numberUL;
52             //ASSERT_EQUALS(9894494448401390090U, numberUL);
53             unsigned long long numberULL(0);
54             std::istringstream(tokenlist.front()->str()) >> numberULL;
55             ASSERT_EQUALS(9894494448401390090U, numberULL);
56         }

58         void line1() const {
59             // Test for Ticket #4408
60             const char code[] = "#file \"c:\\a.h\"\n"
61                                 "first\n"
62                                 "#line 5\n"
63                                 "second\n"
64                                 "#line not-a-number\n"
65                                 "third\n"
66                                 "#line 100 \"i.h\"\n"
67                                 "fourth\n"
68                                 "fifth\n"
69                                 "#endfile\n";
70     
71             errout.str("");
72     
73             TokenList tokenList(&settings);
74             std::istringstream istr(code);
75             bool res = tokenList.createTokens(istr, "a.cpp");
76             ASSERT_EQUALS(res, true);
77     
78             for (const Token *tok = tokenList.front(); tok; tok = tok->next()) {
79                 if (tok->str() == "first")
80                     ASSERT_EQUALS(1, tok->linenr());
81                 if (tok->str() == "second")
82                     ASSERT_EQUALS(5, tok->linenr());
83                 if (tok->str() == "third")
84                     ASSERT_EQUALS(7, tok->linenr());
85                 if (tok->str() == "fourth")
86                     ASSERT_EQUALS(100, tok->linenr());
87                 if (tok->str() == "fifth")
88                     ASSERT_EQUALS(101, tok->linenr());
89             }
90         }

92         void line2() const {
93             const char code[] = "#line 8 \"c:\\a.h\"\n"
94                                 "123\n";
95     
96             errout.str("");
97     
98             // tokenize..
99             TokenList tokenlist(&settings);
100             std::istringstream istr(code);
101             tokenlist.createTokens(istr, "a.cpp");
102     
103             ASSERT_EQUALS(Path::toNativeSeparators("[c:\\a.h:8]"), tokenlist.fileLine(tokenlist.front()));
104         }

106         void inc() const {
107             const char code[] = "a++1;1++b;";
108     
109             errout.str("");
110     
111             // tokenize..
112             TokenList tokenlist(&settings);
113             std::istringstream istr(code);
114             tokenlist.createTokens(istr, "a.cpp");
115     
116             ASSERT(Token::simpleMatch(tokenlist.front(), "a + + 1 ; 1 + + b ;"));
117         }
