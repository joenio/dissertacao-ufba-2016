21     #ifndef checkinternalH
22     #define checkinternalH
23     //---------------------------------------------------------------------------
24     
25     #include "check.h"
26     #include "config.h"
27     
28     /// @addtogroup Checks
29     /// @{
30     
31     
32     /** @brief %Check Internal cppcheck API usage */
33     class CPPCHECKLIB CheckInternal : public Check {
34     public:
35         /** This constructor is used when registering the CheckClass */
36         CheckInternal() : Check(myName()) {
37         }
38     
39         /** This constructor is used when running checks. */
40         CheckInternal(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
41             : Check(myName(), tokenizer, settings, errorLogger) {
42         }
43     
44         /** Simplified checks. The token list is simplified. */
45         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
46             if (!settings->isEnabled("internal"))
47                 return;
48     
49             CheckInternal checkInternal(tokenizer, settings, errorLogger);
50     
51             checkInternal.checkTokenMatchPatterns();
52             checkInternal.checkTokenSimpleMatchPatterns();
53             checkInternal.checkMissingPercentCharacter();
54             checkInternal.checkUnknownPattern();
55             checkInternal.checkRedundantNextPrevious();
56             checkInternal.checkExtraWhitespace();
57         }
58     
59         /** @brief %Check if a simple pattern is used inside Token::Match or Token::findmatch */
60         void checkTokenMatchPatterns();
61     
62         /** @brief %Check if a complex pattern is used inside Token::simpleMatch or Token::findsimplematch */
63         void checkTokenSimpleMatchPatterns();
64     
65         /** @brief %Check for missing % end character in Token::Match pattern */
66         void checkMissingPercentCharacter();
67     
68         /** @brief %Check for unknown (invalid) complex patterns like "%typ%" */
69         void checkUnknownPattern();
70     
71         /** @brief %Check for inefficient usage of Token::next(), Token::previous() and Token::tokAt() */
72         void checkRedundantNextPrevious();
73     
74         /** @brief %Check if there is whitespace at the beginning or at the end of a pattern */
75         void checkExtraWhitespace();
76     
77     private:
78         void multiComparePatternError(const Token *tok, const std::string &pattern, const std::string &funcname);
79         void simplePatternError(const Token *tok, const std::string &pattern, const std::string &funcname);
80         void complexPatternError(const Token *tok, const std::string &pattern, const std::string &funcname);
81         void missingPercentCharacterError(const Token *tok, const std::string &pattern, const std::string &funcname);
82         void unknownPatternError(const Token* tok, const std::string& pattern);
83         void redundantNextPreviousError(const Token* tok, const std::string& func1, const std::string& func2);
84         void orInComplexPattern(const Token *tok, const std::string &pattern, const std::string &funcname);
85         void extraWhitespaceError(const Token *tok, const std::string &pattern, const std::string &funcname);
86     
87         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
88             CheckInternal c(0, settings, errorLogger);
89             c.multiComparePatternError(0, ";|%type%", "Match");
90             c.simplePatternError(0, "class {", "Match");
91             c.complexPatternError(0, "%type% ( )", "Match");
92             c.missingPercentCharacterError(0, "%num", "Match");
93             c.unknownPatternError(0, "%typ");
94             c.redundantNextPreviousError(0, "previous", "next");
95             c.orInComplexPattern(0, "||", "Match");
96             c.extraWhitespaceError(0, "%str% ", "Match");
97         }
98     
99         static std::string myName() {
100             return "cppcheck internal API usage";
101         }
102     
103         std::string classInfo() const {
104             // Don't include these checks on the WIKI where people can read what
105             // checks there are. These checks are not intended for users.
106             return "";
107         }
108     };
109     /// @}
110     //---------------------------------------------------------------------------
111     #endif // checkinternalH
