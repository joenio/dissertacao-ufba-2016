230             // double the tests - run each example as C as well as C++

232             const char* alternatefilename = cpp ? "test.c" : "test.cpp";

243         std::string checkCodeInternal(const char code[], const char* filename) {
244             errout.str("");
245     
246             // tokenize..
247             Tokenizer tokenizer(&settings, this);
248             std::istringstream istr(code);
249             tokenizer.tokenize(istr, filename);
250     
251             // call all "runChecks" in all registered Check classes
252             for (std::list<Check *>::const_iterator it = Check::instances().begin(); it != Check::instances().end(); ++it) {
253                 (*it)->runChecks(&tokenizer, &settings, this);
254             }
255     
256             tokenizer.simplifyTokenList2();
257             // call all "runSimplifiedChecks" in all registered Check classes
258             for (std::list<Check *>::const_iterator it = Check::instances().begin(); it != Check::instances().end(); ++it) {
259                 (*it)->runSimplifiedChecks(&tokenizer, &settings, this);
260             }
261     
262             return tokenizer.tokens()->stringifyList(false, false, false, true, false, 0, 0);
263         }
