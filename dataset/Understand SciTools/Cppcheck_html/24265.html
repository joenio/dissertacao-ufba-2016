19     #include <QString>

25     CheckThread::CheckThread(ThreadResult &result) :
26         mState(Ready),
27         mResult(result),
28         mCppcheck(result, true)
29     {
30         //ctor
31     }
32     
33     CheckThread::~CheckThread()
34     {
35         //dtor
36     }
37     
38     void CheckThread::Check(const Settings &settings)
39     {
40         mCppcheck.settings() = settings;
41         start();
42     }
43     
44     void CheckThread::run()
45     {
46         mState = Running;
47         QString file;
48         file = mResult.GetNextFile();
49     
50         while (!file.isEmpty() && mState == Running) {
51             qDebug() << "Checking file" << file;
52             mCppcheck.check(file.toStdString());
53             emit FileChecked(file);
54     
55             if (mState == Running)
56                 file = mResult.GetNextFile();
57         }
58         if (mState == Running)
59             mState = Ready;
60         else
61             mState = Stopped;
62     
63         emit Done();
64     }
65     
66     void CheckThread::stop()
67     {
68         mState = Stopping;
69         mCppcheck.terminate();
70     }
