19     #ifndef THREADEXECUTOR_H
20     #define THREADEXECUTOR_H
21     
22     #include <map>
23     #include <string>
24     #include <list>
25     #include "errorlogger.h"
26     
27     #if (defined(__GNUC__) || defined(__sun)) && !defined(__MINGW32__)
28     #define THREADING_MODEL_FORK
29     #elif defined(_WIN32)
30     #define THREADING_MODEL_WIN
31     #include <windows.h>
32     #endif
33     
34     class Settings;
35     
36     /// @addtogroup CLI
37     /// @{
38     
39     /**
40      * This class will take a list of filenames and settings and check then
41      * all files using threads.
42      */
43     class ThreadExecutor : public ErrorLogger {
44     public:
45         ThreadExecutor(const std::map<std::string, std::size_t> &files, Settings &settings, ErrorLogger &_errorLogger);
46         virtual ~ThreadExecutor();
47         unsigned int check();
48     
49         virtual void reportOut(const std::string &outmsg);
50         virtual void reportErr(const ErrorLogger::ErrorMessage &msg);
51         virtual void reportInfo(const ErrorLogger::ErrorMessage &msg);
52     
53         /**
54          * @brief Add content to a file, to be used in unit testing.
55          *
56          * @param path File name (used as a key to link with real file).
57          * @param content If the file would be a real file, this should be
58          * the content of the file.
59          */
60         void addFileContent(const std::string &path, const std::string &content);
61     
62     private:
63         const std::map<std::string, std::size_t> &_files;
64         Settings &_settings;
65         ErrorLogger &_errorLogger;
66         unsigned int _fileCount;
67     
68     #if defined(THREADING_MODEL_FORK)
69     
70         /** @brief Key is file name, and value is the content of the file */
71         std::map<std::string, std::string> _fileContents;
72     private:
73         enum PipeSignal {REPORT_OUT='1',REPORT_ERROR='2', REPORT_INFO='3', CHILD_END='4'};
74     
75         /**
76          * Read from the pipe, parse and handle what ever is in there.
77          *@return -1 in case of error
78          *         0 if there is nothing in the pipe to be read
79          *         1 if we did read something
80          */
81         int handleRead(int rpipe, unsigned int &result);
82         void writeToPipe(PipeSignal type, const std::string &data);
83         /**
84          * Write end of status pipe, different for each child.
85          * Not used in master process.
86          */
87         std::list<std::string> _errorList;
88         int _wpipe;
89     
90         /**
91          * @brief Check load average condition
92          * @param nchildren - count of currently runned children
93          * @return true - if new process can be started
94          */
95         bool checkLoadAverage(size_t nchildren);
96     
97     public:
98         /**
99          * @return true if support for threads exist.
100          */
101         static bool isEnabled() {
102             return true;
103         }
104     
105     #elif defined(THREADING_MODEL_WIN)
106     
107     private:
108         enum MessageType {REPORT_ERROR, REPORT_INFO};
109     
110         std::map<std::string, std::string> _fileContents;
111         std::map<std::string, std::size_t>::const_iterator _itNextFile;
112         std::size_t _processedFiles;
113         std::size_t _totalFiles;
114         std::size_t _processedSize;
115         std::size_t _totalFileSize;
116         CRITICAL_SECTION _fileSync;
117     
118         std::list<std::string> _errorList;
119         CRITICAL_SECTION _errorSync;
120     
121         CRITICAL_SECTION _reportSync;
122     
123         void report(const ErrorLogger::ErrorMessage &msg, MessageType msgType);
124     
125         static unsigned __stdcall threadProc(void*);
126     
127     public:
128         /**
129          * @return true if support for threads exist.
130          */
131         static bool isEnabled() {
132             return true;
133         }
134     #else
135     public:
136         /**
137          * @return true if support for threads exist.
138          */
139         static bool isEnabled() {
140             return false;
141         }
142     #endif
143     
144     private:
145         /** disabled copy constructor */
146         ThreadExecutor(const ThreadExecutor &);
147     
148         /** disabled assignment operator */
149         void operator=(const ThreadExecutor &);
150     };
151     
152     /// @}
153     
154     #endif // THREADEXECUTOR_H
