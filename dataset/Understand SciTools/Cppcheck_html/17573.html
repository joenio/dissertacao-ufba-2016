34     void CheckAssert::assertWithSideEffects()
35     {
36         if (!_settings->isEnabled("warning"))
37             return;
38     
39         for (const Token* tok = _tokenizer->list.front(); tok; tok = tok->next()) {
40             if (!Token::simpleMatch(tok, "assert ("))
41                 continue;
42     
43             const Token *endTok = tok->next()->link();
44             for (const Token* tmp = tok->next(); tmp != endTok; tmp = tmp->next()) {
45                 checkVariableAssignment(tmp);
46     
47                 if (tmp->tokType() == Token::eFunction) {
48                     const Function* f = tmp->function();
49     
50                     if (f->nestedIn->isClassOrStruct() && !f->isStatic() && !f->isConst())
51                         sideEffectInAssertError(tmp, f->name()); // Non-const member function called
52                     else {
53                         const Scope* scope = f->functionScope;
54                         if (!scope) continue;
55     
56                         for (const Token *tok2 = scope->classStart; tok2 != scope->classEnd; tok2 = tok2->next()) {
57                             if (tok2->tokType() != Token::eAssignmentOp && tok2->tokType() != Token::eIncDecOp)
58                                 continue;
59     
60                             const Variable* var = tok2->previous()->variable();
61                             if (!var || var->isLocal() || (var->isArgument() && !var->isReference() && !var->isPointer()))
62                                 continue; // See ticket #4937. Assigning function arguments not passed by reference is ok.
63                             if (var->isArgument() && var->isPointer() && tok2->strAt(-2) != "*")
64                                 continue; // Pointers need to be dereferenced, otherwise there is no error
65     
66                             bool noReturnInScope = true;
67                             for (const Token *rt = scope->classStart; rt != scope->classEnd; rt = rt->next()) {
68                                 if (rt->str() != "return") continue; // find all return statements
69                                 if (inSameScope(rt, tok2)) {
70                                     noReturnInScope = false;
71                                     break;
72                                 }
73                             }
74                             if (noReturnInScope) continue;
75     
76                             sideEffectInAssertError(tmp, f->name());
77                             break;
78                         }
79                     }
80                 }
81             }
82             tok = endTok;
83         }
84     }

88     void CheckAssert::sideEffectInAssertError(const Token *tok, const std::string& functionName)
89     {
90         reportError(tok, Severity::warning,
91                     "assertWithSideEffect", "Assert statement calls a function which may have desired side effects: '" + functionName + "'.\n"
92                     "Non-pure function: '" + functionName + "' is called inside assert statement. "
93                     "Assert statements are removed from release builds so the code inside "
94                     "assert statement is not executed. If the code is needed also in release "
95                     "builds, this is a bug.");
96     }
97     
98     void CheckAssert::assignmentInAssertError(const Token *tok, const std::string& varname)
99     {
100         reportError(tok, Severity::warning,
101                     "assignmentInAssert", "Assert statement modifies '" + varname + "'.\n"
102                     "Variable '" + varname + "' is modified insert assert statement. "
103                     "Assert statements are removed from release builds so the code inside "
104                     "assert statement is not executed. If the code is needed also in release "
105                     "builds, this is a bug.");
106     }

108     // checks if side effects happen on the variable prior to tmp
109     void CheckAssert::checkVariableAssignment(const Token* assignTok)
110     {
111         const Variable* prevVar = assignTok->previous()->variable();
112         if (!prevVar)
113             return;
114     
115         // assignment
116         if (assignTok->isAssignmentOp() || assignTok->tokType() == Token::eIncDecOp) {
117             if (prevVar->isConst())
118                 return;
119     
120             assignmentInAssertError(assignTok, prevVar->name());
121         }
122         // TODO: function calls on prevVar
123     }
124     
125     bool CheckAssert::inSameScope(const Token* returnTok, const Token* assignTok)
126     {
127         // TODO: even if a return is in the same scope, the assignment might not affect it.
128         return returnTok->scope() == assignTok->scope();
129     }
