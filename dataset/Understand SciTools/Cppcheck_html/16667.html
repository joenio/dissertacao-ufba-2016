38     static void ShowUsage();
39     static void ShowVersion();
40     static bool CheckArgs(const QStringList &args);
41     
42     int main(int argc, char *argv[])
43     {
44         QApplication app(argc, argv);
45     
46     #if QT_VERSION < 0x050000
47         // Set codecs so that UTF-8 strings in sources are handled correctly.
48         // This is ONLY needed for Qt versions 4.x.
49         // Qt 5.x assumes UTF-8 by default.
50         QTextCodec::setCodecForTr(QTextCodec::codecForName("UTF-8"));
51         QTextCodec::setCodecForCStrings(QTextCodec::codecForName("UTF-8"));
52     #endif
53     
54         QCoreApplication::setOrganizationName("Cppcheck");
55         QCoreApplication::setApplicationName("Cppcheck-GUI");
56     
57         QSettings* settings = new QSettings("Cppcheck", "Cppcheck-GUI", &app);
58     
59         // Set data dir..
60         foreach (const QString arg, app.arguments()) {
61             if (arg.startsWith("--data-dir=")) {
62                 settings->setValue("DATADIR", arg.mid(11));
63                 return 0;
64             }
65         }
66     
67         TranslationHandler* th = new TranslationHandler(&app);
68         th->SetLanguage(settings->value(SETTINGS_LANGUAGE, th->SuggestLanguage()).toString());
69     
70         if (!CheckArgs(app.arguments()))
71             return 0;
72     
73         app.setWindowIcon(QIcon(":icon.png"));
74     
75         // Register this metatype that is used to transfer error info
76         qRegisterMetaType<ErrorItem>("ErrorItem");
77     
78         MainWindow window(th, settings);
79         window.show();
80         return app.exec();
81     }
82     
83     // Check only arguments needing action before GUI is shown.
84     // Rest of the arguments are handled in MainWindow::HandleCLIParams()
85     static bool CheckArgs(const QStringList &args)
86     {
87         if (args.contains("-h") || args.contains("--help")) {
88             ShowUsage();
89             return false;
90         }
91         if (args.contains("-v") || args.contains("--version")) {
92             ShowVersion();
93             return false;
94         }
95         return true;
96     }
97     
98     static void ShowUsage()
99     {
100         QString helpMessage = MainWindow::tr(
101                                   "Cppcheck GUI.\n\n"
102                                   "Syntax:\n"
103                                   "    cppcheck-gui [OPTIONS] [files or paths]\n\n"
104                                   "Options:\n"
105                                   "    -h, --help              Print this help\n"
106                                   "    -p <file>               Open given project file and start checking it\n"
107                                   "    -l <file>               Open given results xml file\n"
108                                   "    -d <directory>          Specify the directory that was checked to generate the results xml specified with -l\n"
109                                   "    -v, --version           Show program version\n"
110                                   "    --data-dir=<directory>  Specify directory where GUI datafiles are located (translations, cfg)");
111     #if defined(_WIN32)
112         QMessageBox msgBox(QMessageBox::Information,
113                            MainWindow::tr("Cppcheck GUI - Command line parameters"),
114                            helpMessage,
115                            QMessageBox::Ok
116                           );
117         (void)msgBox.exec();
118     #else
119         std::cout << helpMessage.toStdString() << std::endl;
120     #endif
121     }
122     
123     static void ShowVersion()
124     {
125     #if defined(_WIN32)
126         AboutDialog *dlg = new AboutDialog(CppCheck::version(), CppCheck::extraVersion(), 0);
127         dlg->exec();
128         delete dlg;
129     #else
130         std::string versionMessage("Cppcheck ");
131         versionMessage += CppCheck::version();
132         const char * extraVersion = CppCheck::extraVersion();
133         if (*extraVersion != 0)
134             versionMessage += std::string(" (") + extraVersion + ")";
135     
136         std::cout << versionMessage << std::endl;
137     #endif
138     }
