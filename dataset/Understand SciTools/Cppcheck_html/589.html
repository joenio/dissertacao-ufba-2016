33     /** @brief %Check STL usage (invalidation of iterators, mismatching containers, etc) */
34     class CPPCHECKLIB CheckStl : public Check {
35     public:
36         /** This constructor is used when registering the CheckClass */
37         CheckStl() : Check(myName()) {
38         }
39     
40         /** This constructor is used when running checks. */
41         CheckStl(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
42             : Check(myName(), tokenizer, settings, errorLogger) {
43         }
44     
45         /** Simplified checks. The token list is simplified. */
46         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
47             if (!tokenizer->isCPP())
48                 return;
49     
50             CheckStl checkStl(tokenizer, settings, errorLogger);
51     
52             checkStl.stlOutOfBounds();
53             checkStl.iterators();
54             checkStl.mismatchingContainers();
55             checkStl.erase();
56             checkStl.pushback();
57             checkStl.stlBoundaries();
58             checkStl.if_find();
59             checkStl.string_c_str();
60             checkStl.checkAutoPointer();
61             checkStl.uselessCalls();
62             checkStl.checkDereferenceInvalidIterator();
63     
64             // Style check
65             checkStl.size();
66             checkStl.redundantCondition();
67             checkStl.missingComparison();
68             checkStl.readingEmptyStlContainer();
69         }
70     
71     
72         /**
73          * Finds errors like this:
74          * for (unsigned ii = 0; ii <= foo.size(); ++ii)
75          */
76         void stlOutOfBounds();
77     
78         /**
79          * Finds errors like this:
80          * for (it = foo.begin(); it != bar.end(); ++it)
81          */
82         void iterators();
83     
84         /**
85          * Mismatching containers:
86          * std::find(foo.begin(), bar.end(), x)
87          */
88         void mismatchingContainers();
89     
90         /**
91          * Dangerous usage of erase. The iterator is invalidated by erase so
92          * it is bad to dereference it after the erase.
93          */
94         void erase();
95         void eraseCheckLoopVar(const Scope &scope, const Variable *var);
96     
97     
98         /**
99          * Dangerous usage of push_back and insert
100          */
101         void pushback();
102     
103         /**
104          * bad condition.. "it < alist.end()"
105          */
106         void stlBoundaries();
107     
108         /** if (a.find(x)) - possibly incorrect condition */
109         void if_find();
110     
111         /**
112          * Suggest using empty() instead of checking size() against zero for containers.
113          * Item 4 from Scott Meyers book "Effective STL".
114          */
115         void size();
116     
117         /**
118          * Check for redundant condition 'if (ints.find(1) != ints.end()) ints.remove(123);'
119          * */
120         void redundantCondition();
121     
122         /**
123          * @brief Missing inner comparison, when incrementing iterator inside loop
124          * Dangers:
125          *  - may increment iterator beyond end
126          *  - may unintentionally skip elements in list/set etc
127          */
128         void missingComparison();
129     
130         /** Check for common mistakes when using the function string::c_str() */
131         void string_c_str();
132     
133         /** @brief %Check for use and copy auto pointer */
134         void checkAutoPointer();
135     
136         /** @brief %Check calls that using them is useless */
137         void uselessCalls();
138     
139         /** @brief %Check for dereferencing an iterator that is invalid */
140         void checkDereferenceInvalidIterator();
141     
142         /**
143          * Dereferencing an erased iterator
144          * @param erased token where the erase occurs
145          * @param deref token where the dereference occurs
146          * @param itername iterator name
147          */
148         void dereferenceErasedError(const Token* erased, const Token* deref, const std::string &itername);
149     
150         /** @brief Reading from empty stl container */
151         void readingEmptyStlContainer();
152     
153     private:
154         void readingEmptyStlContainer_parseUsage(const Token* tok, const Library::Container* container, std::map<unsigned int, const Library::Container*>& empty, bool noerror);
155     
156         void missingComparisonError(const Token *incrementToken1, const Token *incrementToken2);
157         void string_c_strThrowError(const Token *tok);
158         void string_c_strError(const Token *tok);
159         void string_c_strReturn(const Token *tok);
160         void string_c_strParam(const Token *tok, unsigned int number);
161     
162         void stlOutOfBoundsError(const Token *tok, const std::string &num, const std::string &var, bool at);
163         void invalidIteratorError(const Token *tok, const std::string &iteratorName);
164         void iteratorsError(const Token *tok, const std::string &container1, const std::string &container2);
165         void mismatchingContainersError(const Token *tok);
166         void invalidIteratorError(const Token *tok, const std::string &func, const std::string &iterator_name);
167         void invalidPointerError(const Token *tok, const std::string &func, const std::string &pointer_name);
168         void stlBoundariesError(const Token *tok);
169         void if_findError(const Token *tok, bool str);
170         void sizeError(const Token *tok);
171         void redundantIfRemoveError(const Token *tok);
172     
173         void autoPointerError(const Token *tok);
174         void autoPointerContainerError(const Token *tok);
175         void autoPointerArrayError(const Token *tok);
176         void autoPointerMallocError(const Token *tok, const std::string& allocFunction);
177     
178         void uselessCallsReturnValueError(const Token *tok, const std::string &varname, const std::string &function);
179         void uselessCallsSwapError(const Token *tok, const std::string &varname);
180         void uselessCallsSubstrError(const Token *tok, bool empty);
181         void uselessCallsEmptyError(const Token *tok);
182         void uselessCallsRemoveError(const Token *tok, const std::string& function);
183     
184         void dereferenceInvalidIteratorError(const Token* deref, const std::string &itername);
185     
186         void readingEmptyStlContainerError(const Token *tok);
187     
188         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
189             CheckStl c(0, settings, errorLogger);
190             c.invalidIteratorError(0, "iterator");
191             c.iteratorsError(0, "container1", "container2");
192             c.mismatchingContainersError(0);
193             c.dereferenceErasedError(0, 0, "iter");
194             c.stlOutOfBoundsError(0, "i", "foo", false);
195             c.invalidIteratorError(0, "push_back|push_front|insert", "iterator");
196             c.invalidPointerError(0, "push_back", "pointer");
197             c.stlBoundariesError(0);
198             c.if_findError(0, false);
199             c.if_findError(0, true);
200             c.string_c_strError(0);
201             c.string_c_strReturn(0);
202             c.string_c_strParam(0, 0);
203             c.sizeError(0);
204             c.missingComparisonError(0, 0);
205             c.redundantIfRemoveError(0);
206             c.autoPointerError(0);
207             c.autoPointerContainerError(0);
208             c.autoPointerArrayError(0);
209             c.autoPointerMallocError(0, "malloc");
210             c.uselessCallsReturnValueError(0, "str", "find");
211             c.uselessCallsSwapError(0, "str");
212             c.uselessCallsSubstrError(0, false);
213             c.uselessCallsEmptyError(0);
214             c.uselessCallsRemoveError(0, "remove");
215             c.dereferenceInvalidIteratorError(0, "i");
216             c.readingEmptyStlContainerError(0);
217         }
218     
219         static std::string myName() {
220             return "STL usage";
221         }
222     
223         std::string classInfo() const {
224             return "Check for invalid usage of STL:\n"
225                    "- out of bounds errors\n"
226                    "- misuse of iterators when iterating through a container\n"
227                    "- mismatching containers in calls\n"
228                    "- dereferencing an erased iterator\n"
229                    "- for vectors: using iterator/pointer after push_back has been used\n"
230                    "- optimisation: use empty() instead of size() to guarantee fast code\n"
231                    "- suspicious condition when using find\n"
232                    "- redundant condition\n"
233                    "- common mistakes when using string::c_str()\n"
234                    "- using auto pointer (auto_ptr)\n"
235                    "- useless calls of string and STL functions\n"
236                    "- dereferencing an invalid iterator\n"
237                    "- reading from empty STL container\n";
238         }
239     };
240     /// @}
241     //---------------------------------------------------------------------------
242     #endif // checkstlH
