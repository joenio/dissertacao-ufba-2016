79     bool TranslationHandler::SetLanguage(const QString &code)
80     {
81         bool failure = false;
82         QString error;
83     
84         //If English is the language
85         if (code == "en") {
86             //Just remove all extra translators
87             if (mTranslator) {
88                 qApp->removeTranslator(mTranslator);
89                 delete mTranslator;
90                 mTranslator = NULL;
91             }
92     
93             mCurrentLanguage = code;
94             return true;
95         }
96     
97         //Make sure the translator is otherwise valid
98         int index = GetLanguageIndexByCode(code);
99         if (index == -1) {
100             error = QObject::tr("Unknown language specified!");
101             failure = true;
102         }
103     
104         // Make sure there is a translator
105         if (!mTranslator && !failure)
106             mTranslator = new QTranslator(this);
107     
108         //Load the new language
109         const QString appPath = QFileInfo(QCoreApplication::applicationFilePath()).canonicalPath();
110     
111         QSettings settings;
112         QString datadir = settings.value("DATADIR").toString();
113         if (datadir.isEmpty())
114             datadir = appPath;
115     
116         QString translationFile;
117         if (QFile::exists(datadir + "/lang/" + mTranslations[index].mFilename + ".qm"))
118             translationFile = datadir + "/lang/" + mTranslations[index].mFilename + ".qm";
119     
120         else if (QFile::exists(datadir + "/" + mTranslations[index].mFilename + ".qm"))
121             translationFile = datadir + "/" + mTranslations[index].mFilename + ".qm";
122     
123         else
124             translationFile = appPath + "/" + mTranslations[index].mFilename + ".qm";
125     
126         if (!mTranslator->load(translationFile) && !failure) {
127             translationFile += ".qm";
128             //If it failed, lets check if the default file exists
129             if (!QFile::exists(translationFile)) {
130                 error = QObject::tr("Language file %1 not found!");
131                 error = error.arg(translationFile);
132                 failure = true;
133             }
134     
135             //If file exists, there's something wrong with it
136             error = QObject::tr("Failed to load translation for language %1 from file %2");
137             error = error.arg(mTranslations[index].mName);
138             error = error.arg(translationFile);
139         }
140     
141         if (failure) {
142             const QString msg(tr("Failed to change the user interface language:"
143                                  "\n\n%1\n\n"
144                                  "The user interface language has been reset to English. Open "
145                                  "the Preferences-dialog to select any of the available "
146                                  "languages.").arg(error));
147             QMessageBox msgBox(QMessageBox::Warning,
148                                tr("Cppcheck"),
149                                msg,
150                                QMessageBox::Ok);
151             msgBox.exec();
152             return false;
153         }
154     
155         qApp->installTranslator(mTranslator);
156     
157         mCurrentLanguage = code;
158     
159         return true;
160     }

167     QString TranslationHandler::SuggestLanguage() const
168     {
169         //Get language from system locale's name ie sv_SE or zh_CN
170         QString language = QLocale::system().name();
171         //qDebug()<<"Your language is"<<language;
172     
173         //And see if we can find it from our list of language files
174         int index = GetLanguageIndexByCode(language);
175     
176         //If nothing found, return English
177         if (index < 0) {
178             return "en";
179         }
180     
181         return language;
182     }

184     void TranslationHandler::AddTranslation(const char *name, const char *filename)
185     {
186         TranslationInfo info;
187         info.mName = name;
188         info.mFilename = filename;
189         int codeLength = QString(filename).length() - QString(filename).indexOf('_') - 1;
190         info.mCode = QString(filename).right(codeLength);
191         mTranslations.append(info);
192     }
193     
194     int TranslationHandler::GetLanguageIndexByCode(const QString &code) const
195     {
196         int index = -1;
197         for (int i = 0; i < mTranslations.size(); i++) {
198             if (mTranslations[i].mCode == code) {
199                 index = i;
200                 break;
201             } else if (mTranslations[i].mCode == code.left(2)) {
202                 index = i;
203                 break;
204             }
205         }
206         return index;
207     }
