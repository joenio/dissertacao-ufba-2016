39         /** @brief This constructor is used when registering the CheckClass */
40         CheckOther() : Check(myName()) {

48         /** @brief Run checks against the normal token list */
49         void runChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
50             CheckOther checkOther(tokenizer, settings, errorLogger);
51     
52             // Checks
53             checkOther.warningOldStylePointerCast();
54             checkOther.invalidPointerCast();
55             checkOther.checkCharVariable();
56             checkOther.checkRedundantAssignment();
57             checkOther.checkRedundantAssignmentInSwitch();
58             checkOther.checkSuspiciousCaseInSwitch();
59             checkOther.checkDuplicateBranch();
60             checkOther.checkDuplicateExpression();
61             checkOther.checkUnreachableCode();
62             checkOther.checkSuspiciousSemicolon();
63             checkOther.checkVariableScope();
64             checkOther.checkSignOfUnsignedVariable();  // don't ignore casts (#3574)
65             checkOther.checkIncompleteArrayFill();
66             checkOther.checkVarFuncNullUB();
67             checkOther.checkNanInArithmeticExpression();
68             checkOther.checkCommaSeparatedReturn();
69             checkOther.checkRedundantPointerOp();
70             checkOther.checkZeroDivision();
71             checkOther.checkNegativeBitwiseShift();
72             checkOther.checkInterlockedDecrement();
73             checkOther.checkUnusedLabel();
74             checkOther.checkEvaluationOrder();
75         }

77         /** @brief Run checks against the simplified token list */
78         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
79             CheckOther checkOther(tokenizer, settings, errorLogger);
80     
81             // Checks
82             checkOther.clarifyCalculation();
83             checkOther.clarifyStatement();
84             checkOther.checkConstantFunctionParameter();
85             checkOther.checkIncompleteStatement();
86             checkOther.checkCastIntToCharAndBack();
87     
88             checkOther.checkMisusedScopedObject();
89             checkOther.checkMemsetZeroBytes();
90             checkOther.checkMemsetInvalid2ndParam();
91             checkOther.checkSwitchCaseFallThrough();
92             checkOther.checkPipeParameterSize();
93     
94             checkOther.checkInvalidFree();
95             checkOther.checkRedundantCopy();
96             checkOther.checkSuspiciousEqualityComparison();
97             checkOther.checkComparisonFunctionIsAlwaysTrueOrFalse();
98         }
