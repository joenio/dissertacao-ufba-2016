67         static const Scope *findFunctionScopeByToken(const SymbolDatabase * db, const Token *tok) {
68             std::list<Scope>::const_iterator scope;
69     
70             for (scope = db->scopeList.begin(); scope != db->scopeList.end(); ++scope) {
71                 if (scope->type == Scope::eFunction) {
72                     if (scope->classDef == tok)
73                         return &(*scope);
74                 }
75             }
76             return 0;
77         }

79         static const Function *findFunctionByName(const char str[], const Scope* startScope) {
80             const Scope* currScope = startScope;
81             while (currScope && currScope->isExecutable()) {
82                 if (currScope->functionOf)
83                     currScope = currScope->functionOf;
84                 else
85                     currScope = currScope->nestedIn;
86             }
87             while (currScope) {
88                 for (std::list<Function>::const_iterator i = currScope->functionList.begin(); i != currScope->functionList.end(); ++i) {
89                     if (i->tokenDef->str() == str)
90                         return &*i;
91                 }
92                 currScope = currScope->nestedIn;
93             }
94             return 0;
95         }
