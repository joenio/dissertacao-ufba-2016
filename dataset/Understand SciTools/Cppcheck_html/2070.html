19     #include <QDialog>
20     #include <QWidget>
21     #include <QString>
22     #include "aboutdialog.h"
23     
24     AboutDialog::AboutDialog(const QString &version, const QString &extraVersion, QWidget *parent)
25         : QDialog(parent)
26     {
27         mUI.setupUi(this);
28     
29         QString fmtVersion(version);
30         if (!extraVersion.isEmpty()) {
31             fmtVersion += " (" + extraVersion + ")";
32         }
33         mUI.mVersion->setText(mUI.mVersion->text().arg(fmtVersion));
34         QString url = "<a href=\"http://cppcheck.sourceforge.net/\">http://cppcheck.sourceforge.net/</a>";
35         mUI.mHomepage->setText(mUI.mHomepage->text().arg(url));
36         connect(mUI.mButtons, SIGNAL(accepted()), this, SLOT(accept()));
37     }
