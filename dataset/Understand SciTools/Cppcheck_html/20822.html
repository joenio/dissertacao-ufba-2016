29     ApplicationDialog::ApplicationDialog(const QString &title,
30                                          Application &app,
31                                          QWidget *parent) :
32         QDialog(parent),
33         mApplication(app)
34     {
35         mUI.setupUi(this);
36     
37         connect(mUI.mButtonBrowse, SIGNAL(clicked()), this, SLOT(Browse()));
38         connect(mUI.mButtons, SIGNAL(accepted()), this, SLOT(Ok()));
39         connect(mUI.mButtons, SIGNAL(rejected()), this, SLOT(reject()));
40         mUI.mPath->setText(app.getPath());
41         mUI.mName->setText(app.getName());
42         mUI.mParameters->setText(app.getParameters());
43         setWindowTitle(title);
44         adjustSize();
45     }

50         //dtor

53     void ApplicationDialog::Browse()
54     {
55         QString filter;
56     #ifdef Q_OS_WIN
57         // In Windows (almost) all executables have .exe extension
58         // so it does not make sense to show everything.
59         filter += tr("Executable files (*.exe);;All files(*.*)");
60     #endif // Q_OS_WIN
61         QString selectedFile = QFileDialog::getOpenFileName(this,
62                                tr("Select viewer application"),
63                                GetPath(SETTINGS_LAST_APP_PATH),
64                                filter);
65     
66         if (!selectedFile.isEmpty()) {
67             SetPath(SETTINGS_LAST_APP_PATH, selectedFile);
68             QString path(QDir::toNativeSeparators(selectedFile));
69             mUI.mPath->setText(path);
70         }
71     }
72     
73     void ApplicationDialog::Ok()
74     {
75         if (mUI.mName->text().isEmpty() || mUI.mPath->text().isEmpty()) {
76             QMessageBox msg(QMessageBox::Warning,
77                             tr("Cppcheck"),
78                             tr("You must specify a name, a path and optionally parameters for the application!"),
79                             QMessageBox::Ok,
80                             this);
81     
82             msg.exec();
83     
84             reject();
85         } else {
86             // Convert possible native (Windows) path to internal presentation format
87             mApplication.setName(mUI.mName->text());
88             mApplication.setPath(QDir::fromNativeSeparators(mUI.mPath->text()));
89             mApplication.setParameters(mUI.mParameters->text());
90     
91             accept();
92         }
93     }
