20     //---------------------------------------------------------------------------
21     #ifndef checkboostH
22     #define checkboostH
23     //---------------------------------------------------------------------------
24     
25     #include "config.h"
26     #include "check.h"
27     
28     class Token;
29     
30     /// @addtogroup Checks
31     /// @{
32     
33     
34     /** @brief %Check Boost usage */
35     class CPPCHECKLIB CheckBoost : public Check {
36     public:
37         /** This constructor is used when registering the CheckClass */
38         CheckBoost() : Check(myName()) {
39         }
40     
41         /** This constructor is used when running checks. */
42         CheckBoost(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
43             : Check(myName(), tokenizer, settings, errorLogger) {
44         }
45     
46         /** Simplified checks. The token list is simplified. */
47         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
48             if (!tokenizer->isCPP())
49                 return;
50     
51             CheckBoost checkBoost(tokenizer, settings, errorLogger);
52     
53             checkBoost.checkBoostForeachModification();
54         }
55     
56         /** @brief %Check for container modification while using the BOOST_FOREACH macro */
57         void checkBoostForeachModification();
58     
59     private:
60         void boostForeachError(const Token *tok);
61     
62         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
63             CheckBoost c(0, settings, errorLogger);
64             c.boostForeachError(0);
65         }
66     
67         static std::string myName() {
68             return "Boost usage";
69         }
70     
71         std::string classInfo() const {
72             return "Check for invalid usage of Boost:\n"
73                    "- container modification during BOOST_FOREACH\n";
74         }
75     };
76     /// @}
77     //---------------------------------------------------------------------------
78     #endif // checkboostH
