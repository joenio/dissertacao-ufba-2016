34     void CheckSizeof::checkSizeofForNumericParameter()
35     {
36         if (!_settings->isEnabled("warning"))
37             return;
38     
39         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
40         const std::size_t functions = symbolDatabase->functionScopes.size();
41         for (std::size_t i = 0; i < functions; ++i) {
42             const Scope * scope = symbolDatabase->functionScopes[i];
43             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
44                 if (Token::Match(tok, "sizeof ( %num% )") ||
45                     Token::Match(tok, "sizeof %num%")) {
46                     sizeofForNumericParameterError(tok);
47                 }
48             }
49         }
50     }
51     
52     void CheckSizeof::sizeofForNumericParameterError(const Token *tok)
53     {
54         reportError(tok, Severity::warning,
55                     "sizeofwithnumericparameter", "Suspicious usage of 'sizeof' with a numeric constant as parameter.\n"
56                     "It is unusual to use a constant value with sizeof. For example, 'sizeof(10)'"
57                     " returns 4 (in 32-bit systems) or 8 (in 64-bit systems) instead of 10. 'sizeof('A')'"
58                     " and 'sizeof(char)' can return different results.");
59     }

64     void CheckSizeof::checkSizeofForArrayParameter()
65     {
66         if (!_settings->isEnabled("warning"))
67             return;
68         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
69         const std::size_t functions = symbolDatabase->functionScopes.size();
70         for (std::size_t i = 0; i < functions; ++i) {
71             const Scope * scope = symbolDatabase->functionScopes[i];
72             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
73                 if (Token::Match(tok, "sizeof ( %var% )") ||
74                     Token::Match(tok, "sizeof %var% !![")) {
75                     const Token* varTok = tok->next();
76                     if (varTok->str() == "(") {
77                         varTok = varTok->next();
78                     }
79     
80                     const Variable *var = varTok->variable();
81                     if (var && var->isArray() && var->isArgument() && !var->isReference())
82                         sizeofForArrayParameterError(tok);
83                 }
84             }
85         }
86     }

88     void CheckSizeof::sizeofForArrayParameterError(const Token *tok)
89     {
90         reportError(tok, Severity::warning,
91                     "sizeofwithsilentarraypointer", "Using 'sizeof' on array given as function argument "
92                     "returns size of a pointer.\n"
93                     "Using 'sizeof' for array given as function argument returns the size of a pointer. "
94                     "It does not return the size of the whole array in bytes as might be "
95                     "expected. For example, this code:\n"
96                     "     int f(char a[100]) {\n"
97                     "         return sizeof(a);\n"
98                     "     }\n"
99                     "returns 4 (in 32-bit systems) or 8 (in 64-bit systems) instead of 100 (the "
100                     "size of the array in bytes)."
101                    );
102     }

104     void CheckSizeof::checkSizeofForPointerSize()
105     {
106         if (!_settings->isEnabled("warning"))
107             return;
108     
109         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
110         const std::size_t functions = symbolDatabase->functionScopes.size();
111         for (std::size_t i = 0; i < functions; ++i) {
112             const Scope * scope = symbolDatabase->functionScopes[i];
113             for (const Token* tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
114                 const Token* tokSize;
115                 const Token* tokFunc;
116                 const Token *variable = nullptr;
117                 const Token *variable2 = nullptr;
118     
119                 // Find any function that may use sizeof on a pointer
120                 // Once leaving those tests, it is mandatory to have:
121                 // - variable matching the used pointer
122                 // - tokVar pointing on the argument where sizeof may be used
123                 if (Token::Match(tok, "%var% = malloc|alloca (")) {
124                     variable = tok;
125                     tokSize = tok->tokAt(4);
126                     tokFunc = tok->tokAt(2);
127                 } else if (Token::Match(tok, "%var% = calloc (")) {
128                     variable = tok;
129                     tokSize = tok->tokAt(4)->nextArgument();
130                     tokFunc = tok->tokAt(2);
131                 } else if (Token::Match(tok, "return malloc|alloca (")) {
132                     tokSize = tok->tokAt(3);
133                     tokFunc = tok->next();
134                 } else if (Token::simpleMatch(tok, "return calloc (")) {
135                     tokSize = tok->tokAt(3)->nextArgument();
136                     tokFunc = tok->next();
137                 } else if (Token::simpleMatch(tok, "memset (") && tok->strAt(-1) != ".") {
138                     variable = tok->tokAt(2);
139                     tokSize = variable->nextArgument();
140                     if (tokSize)
141                         tokSize = tokSize->nextArgument();
142                     tokFunc = tok;
143                 } else if (Token::Match(tok, "memcpy|memcmp|memmove|strncpy|strncmp|strncat (") && tok->strAt(-1) != ".") {
144                     variable = tok->tokAt(2);
145                     variable2 = variable->nextArgument();
146                     if (!variable2)
147                         continue;
148                     tokSize = variable2->nextArgument();
149                     tokFunc = tok;
150                 } else {
151                     continue;
152                 }
153     
154                 if (tokFunc && tokSize) {
155                     for (const Token* tok2 = tokSize; tok2 != tokFunc->linkAt(1); tok2 = tok2->next()) {
156                         if (Token::simpleMatch(tok2, "/ sizeof"))
157                             divideBySizeofError(tok2, tokFunc->str());
158                     }
159                 }
160     
161                 if (!variable || !tokSize)
162                     continue;
163     
164                 while (Token::Match(variable, "%var% ::|."))
165                     variable = variable->tokAt(2);
166     
167                 // Ensure the variables are in the symbol database
168                 // Also ensure the variables are pointers
169                 // Only keep variables which are pointers
170                 const Variable *var = variable->variable();
171                 if (!var || !var->isPointer() || var->isArray()) {
172                     variable = 0;
173                 }
174     
175                 if (variable2) {
176                     var = variable2->variable();
177                     if (!var || !var->isPointer() || var->isArray()) {
178                         variable2 = 0;
179                     }
180                 }
181     
182                 // If there are no pointer variable at this point, there is
183                 // no need to continue
184                 if (variable == 0 && variable2 == 0) {
185                     continue;
186                 }
187     
188                 // Jump to the next sizeof token in the function and in the parameter
189                 // This is to allow generic operations with sizeof
190                 for (; tokSize && tokSize->str() != ")" && tokSize->str() != "," && tokSize->str() != "sizeof"; tokSize = tokSize->next()) {}
191     
192                 if (tokSize->str() != "sizeof")
193                     continue;
194     
195                 if (Token::simpleMatch(tokSize, "sizeof ( &"))
196                     tokSize = tokSize->tokAt(3);
197                 else if (Token::Match(tokSize, "sizeof (|&"))
198                     tokSize = tokSize->tokAt(2);
199                 else
200                     tokSize = tokSize->next();
201     
202                 while (Token::Match(tokSize, "%var% ::|."))
203                     tokSize = tokSize->tokAt(2);
204     
205                 if (Token::Match(tokSize, "%var% [|("))
206                     continue;
207     
208                 // Now check for the sizeof usage. Once here, everything using sizeof(varid) or sizeof(&varid)
209                 // looks suspicious
210                 // Do it for first variable
211                 if (variable && tokSize->varId() == variable->varId())
212                     sizeofForPointerError(variable, variable->str());
213                 if (variable2 && tokSize->varId() == variable2->varId())
214                     sizeofForPointerError(variable2, variable2->str());
215             }
216         }
217     }

220     {
221         reportError(tok, Severity::warning, "pointerSize",
222                     "Size of pointer '" + varname + "' used instead of size of its data.\n"
223                     "Size of pointer '" + varname + "' used instead of size of its data. "
224                     "This is likely to lead to a buffer overflow. You probably intend to "
225                     "write 'sizeof(*" + varname + ")'.");

236     void CheckSizeof::sizeofsizeof()
237     {
238         if (!_settings->isEnabled("warning"))
239             return;
240     
241         for (const Token *tok = _tokenizer->tokens(); tok; tok = tok->next()) {
242             if (Token::Match(tok, "sizeof (| sizeof")) {
243                 sizeofsizeofError(tok);
244                 tok = tok->next();
245             }
246         }
247     }

250     {

252                     "sizeofsizeof", "Calling 'sizeof' on 'sizeof'.\n"
253                     "Calling sizeof for 'sizeof looks like a suspicious code and "
254                     "most likely there should be just one 'sizeof'. The current "
255                     "code is equivalent to 'sizeof(size_t)'");

260     void CheckSizeof::sizeofCalculation()
261     {
262         if (!_settings->isEnabled("warning"))
263             return;
264     
265         const bool printInconclusive = _settings->inconclusive;
266     
267         for (const Token *tok = _tokenizer->tokens(); tok; tok = tok->next()) {
268             if (Token::simpleMatch(tok, "sizeof (")) {
269     
270                 // ignore if the `sizeof` result is cast to void inside a macro, i.e. the calculation is
271                 // expected to be parsed but skipped, such as in a disabled custom ASSERT() macro
272                 if (tok->isExpandedMacro() && tok->previous()) {
273                     const Token *cast_end = (tok->previous()->str() == "(") ? tok->previous() : tok;
274                     if (Token::simpleMatch(cast_end->tokAt(-3), "( void )") ||
275                         Token::simpleMatch(cast_end->tokAt(-1), "static_cast<void>")) {
276                         continue;
277                     }
278                 }
279     
280                 const Token *argument = tok->next()->astOperand2();
281                 if (argument && argument->isCalculation() && (!argument->isExpandedMacro() || printInconclusive))
282                     sizeofCalculationError(argument, argument->isExpandedMacro());
283             }
284         }
285     }

293     //-----------------------------------------------------------------------------
294     // Check for code like sizeof()*sizeof() or sizeof(ptr)/value
295     //-----------------------------------------------------------------------------
296     void CheckSizeof::suspiciousSizeofCalculation()
297     {
298         if (!_settings->isEnabled("warning") || !_settings->inconclusive)
299             return;
300     
301         // TODO: Use AST here. This should be possible as soon as sizeof without brackets is correctly parsed
302         for (const Token *tok = _tokenizer->tokens(); tok; tok = tok->next()) {
303             if (Token::simpleMatch(tok, "sizeof (")) {
304                 const Token* const end = tok->linkAt(1);
305                 const Variable* var = end->previous()->variable();
306                 if (end->strAt(-1) == "*" || (var && var->isPointer() && !var->isArray())) {
307                     if (end->strAt(1) == "/")
308                         divideSizeofError(tok);
309                 } else if (Token::simpleMatch(end, ") * sizeof") && end->next()->astOperand1() == tok->next())
310                     multiplySizeofError(tok);
311             }
312         }
313     }

329     void CheckSizeof::sizeofVoid()
330     {
331         if (!_settings->isEnabled("portability"))
332             return;
333     
334         for (const Token *tok = _tokenizer->tokens(); tok; tok = tok->next()) {
335             if (Token::simpleMatch(tok, "sizeof ( )")) { // "sizeof(void)" gets simplified to sizeof ( )
336                 sizeofVoidError(tok);
337             } else if (Token::simpleMatch(tok, "sizeof (") && tok->next()->astOperand2()) {
338                 const ValueType *vt = tok->next()->astOperand2()->valueType();
339                 if (vt && vt->type == ValueType::Type::VOID && vt->pointer == 0U)
340                     sizeofDereferencedVoidPointerError(tok, tok->strAt(3));
341             } else if (tok->str() == "-") {
342                 // only warn for: 'void *' - 'integral'
343                 const ValueType *vt1  = tok->astOperand1() ? tok->astOperand1()->valueType() : nullptr;
344                 const ValueType *vt2  = tok->astOperand2() ? tok->astOperand2()->valueType() : nullptr;
345                 bool op1IsvoidPointer = (vt1 && vt1->type == ValueType::Type::VOID && vt1->pointer == 1U);
346                 bool op2IsIntegral    = (vt2 && vt2->isIntegral() && vt2->pointer == 0U);
347                 if (op1IsvoidPointer && op2IsIntegral)
348                     arithOperationsOnVoidPointerError(tok, tok->astOperand1()->expressionString(), vt1->str());
349             } else if (Token::Match(tok, "+|++|--|+=|-=")) { // Arithmetic operations on variable of type "void*"
350                 const ValueType *vt1 = tok->astOperand1() ? tok->astOperand1()->valueType() : nullptr;
351                 const ValueType *vt2 = tok->astOperand2() ? tok->astOperand2()->valueType() : nullptr;
352     
353                 bool voidpointer1 = (vt1 && vt1->type == ValueType::Type::VOID && vt1->pointer == 1U);
354                 bool voidpointer2 = (vt2 && vt2->type == ValueType::Type::VOID && vt2->pointer == 1U);
355     
356                 if (voidpointer1)
357                     arithOperationsOnVoidPointerError(tok, tok->astOperand1()->expressionString(), vt1->str());
358     
359                 if (!tok->isAssignmentOp() && voidpointer2)
360                     arithOperationsOnVoidPointerError(tok, tok->astOperand2()->expressionString(), vt2->str());
361             }
362         }
363     }

365     void CheckSizeof::sizeofVoidError(const Token *tok)
366     {
367         const std::string message = "Behaviour of 'sizeof(void)' is not covered by the ISO C standard.";
368         const std::string verbose = message + " A value for 'sizeof(void)' is defined only as part of a GNU C extension, which defines 'sizeof(void)' to be 1.";
369         reportError(tok, Severity::portability, "sizeofVoid", message + "\n" + verbose);
370     }
