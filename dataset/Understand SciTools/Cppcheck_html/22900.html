45         {
46             tok = tok->tokAt(2);

54             return numpar;

83     {
84         if (tok->isStandardType())
85             return false;
86     
87         const Variable * var = tokenizer->getSymbolDatabase()->getVariableFromVarId(varid);
88     
89         // return false if the type is a simple record type without side effects
90         // a type that has no side effects (no constructors and no members with constructors)
91         /** @todo false negative: check base class for side effects */
92         /** @todo false negative: check constructors for side effects */
93         if (var && var->typeScope() && var->typeScope()->numConstructors == 0 &&
94             (var->typeScope()->varlist.empty() || var->type()->needInitialization == Type::True) &&
95             var->type()->derivedFrom.empty())

98         return true;

102     CheckMemoryLeak::AllocType CheckMemoryLeak::getAllocationType(const Token *tok2, unsigned int varid, std::list<const Function*> *callstack) const
103     {
104         // What we may have...
105         //     * var = (char *)malloc(10);
106         //     * var = new char[10];
107         //     * var = strdup("hello");
108         //     * var = strndup("hello", 3);
109         if (tok2 && tok2->str() == "(") {
110             tok2 = tok2->link();
111             tok2 = tok2 ? tok2->next() : nullptr;
112         }
113         if (! tok2)
114             return No;
115         if (tok2->str() == "::")
116             tok2 = tok2->next();
117         if (! tok2->isName())
118             return No;
119     
120         if (!Token::Match(tok2, "%name% ::|. %type%")) {
121             // Using realloc..
122             if (varid && Token::Match(tok2, "realloc ( %any% ,") && tok2->tokAt(2)->varId() != varid)
123                 return Malloc;
124     
125             if (tokenizer->isCPP() && tok2->str() == "new") {
126                 if (tok2->astOperand1() && (tok2->astOperand1()->str() == "[" || (tok2->astOperand1()->astOperand1() && tok2->astOperand1()->astOperand1()->str() == "[")))
127                     return NewArray;
128                 return New;
129             }
130     
131             if (settings1->standards.posix) {
132                 if (Token::Match(tok2, "open|openat|creat|mkstemp|mkostemp|socket (")) {
133                     // simple sanity check of function parameters..
134                     // TODO: Make such check for all these functions
135                     unsigned int num = countParameters(tok2);
136                     if (tok2->str() == "open" && num != 2 && num != 3)
137                         return No;
138     
139                     // is there a user function with this name?
140                     if (tok2->function())
141                         return No;
142                     return Fd;
143                 }
144     
145                 if (Token::simpleMatch(tok2, "popen ("))
146                     return Pipe;
147             }
148     
149             // Does tok2 point on a Library allocation function?
150             const int alloctype = settings1->library.alloc(tok2);
151             if (alloctype > 0) {
152                 if (alloctype == settings1->library.dealloc("free"))
153                     return Malloc;
154                 if (alloctype == settings1->library.dealloc("fclose"))
155                     return File;
156                 return Library::ismemory(alloctype) ? OtherMem : OtherRes;
157             }
158         }
159     
160         while (Token::Match(tok2,"%name% ::|. %type%"))
161             tok2 = tok2->tokAt(2);
162     
163         // User function
164         const Function* func = tok2->function();
165         if (func == nullptr)
166             return No;
167     
168         // Prevent recursion
169         if (callstack && std::find(callstack->begin(), callstack->end(), func) != callstack->end())
170             return No;
171     
172         std::list<const Function*> cs;
173         if (!callstack)
174             callstack = &cs;
175     
176         callstack->push_back(func);
177         return functionReturnType(func, callstack);
178     }

202     CheckMemoryLeak::AllocType CheckMemoryLeak::getDeallocationType(const Token *tok, unsigned int varid) const
203     {
204         if (tokenizer->isCPP() && tok->str() == "delete" && tok->astOperand1()) {
205             const Token* vartok = tok->astOperand1();
206             if (Token::Match(vartok, ".|::"))
207                 vartok = vartok->astOperand2();
208     
209             if (vartok && vartok->varId() == varid) {
210                 if (tok->strAt(1) == "[")
211                     return NewArray;
212                 return New;
213             }
214         }
215     
216         if (tok && tok->str() == "::")
217             tok = tok->next();
218     
219         if (Token::Match(tok, "%name% (")) {
220             if (Token::simpleMatch(tok, "fcloseall ( )"))
221                 return File;
222     
223             const Token* vartok = tok->tokAt(2);
224             while (Token::Match(vartok, "%name% .|::"))
225                 vartok = vartok->tokAt(2);
226     
227             if (Token::Match(vartok, "%varid% )|,|-", varid)) {
228                 if (tok->str() == "realloc" && Token::simpleMatch(vartok->next(), ", 0 )"))
229                     return Malloc;
230     
231                 if (settings1->standards.posix) {
232                     if (tok->str() == "close")
233                         return Fd;
234                     if (tok->str() == "pclose")
235                         return Pipe;
236                 }
237     
238                 // Does tok point on a Library deallocation function?
239                 const int dealloctype = settings1->library.dealloc(tok);
240                 if (dealloctype > 0) {
241                     if (dealloctype == settings1->library.dealloc("free"))
242                         return Malloc;
243                     if (dealloctype == settings1->library.dealloc("fclose"))
244                         return File;
245                     return Library::ismemory(dealloctype) ? OtherMem : OtherRes;
246                 }
247             }
248         }
249     
250         return No;
251     }

329     CheckMemoryLeak::AllocType CheckMemoryLeak::functionReturnType(const Function* func, std::list<const Function*> *callstack) const
330     {
331         if (!func || !func->hasBody())
332             return No;
333     
334         // Get return pointer..
335         unsigned int varid = 0;
336         for (const Token *tok2 = func->functionScope->classStart; tok2 != func->functionScope->classEnd; tok2 = tok2->next()) {
337             if (tok2->str() == "return") {
338                 AllocType allocType = getAllocationType(tok2->next(), 0, callstack);
339                 if (allocType != No)
340                     return allocType;
341     
342                 if (tok2->scope() != func->functionScope || !tok2->astOperand1())
343                     return No;
344                 const Token* tok = tok2->astOperand1();
345                 if (Token::Match(tok, ".|::"))
346                     tok = tok->astOperand2() ? tok->astOperand2() : tok->astOperand1();
347                 if (tok)
348                     varid = tok->varId();
349                 break;
350             }
351         }
352     
353         // Not returning pointer value..
354         if (varid == 0)
355             return No;
356     
357         // If variable is not local then alloctype shall be "No"
358         // Todo: there can be false negatives about mismatching allocation/deallocation.
359         //       => Generate "alloc ; use ;" if variable is not local?
360         const Variable *var = tokenizer->getSymbolDatabase()->getVariableFromVarId(varid);
361         if (!var || !var->isLocal() || var->isStatic())
362             return No;
363     
364         // Check if return pointer is allocated..
365         AllocType allocType = No;
366         for (const Token* tok = func->functionScope->classStart; tok != func->functionScope->classEnd; tok = tok->next()) {
367             if (Token::Match(tok, "%varid% =", varid)) {
368                 allocType = getAllocationType(tok->tokAt(2), varid, callstack);
369             }
370             if (Token::Match(tok, "= %varid% ;", varid)) {
371                 return No;
372             }
373             if (!tokenizer->isC() && Token::Match(tok, "[(,] %varid% [,)]", varid)) {
374                 return No;
375             }
376             if (Token::Match(tok, "[(,] & %varid% [.,)]", varid)) {
377                 return No;
378             }
379             if (allocType == No && tok->str() == "return")
380                 return No;
381         }
382     
383         return allocType;
384     }

387     const char *CheckMemoryLeak::functionArgAlloc(const Function *func, unsigned int targetpar, AllocType &allocType) const
388     {
389         allocType = No;
390     
391         if (!func || !func->functionScope)
392             return "";
393     
394         if (!Token::simpleMatch(func->retDef, "void"))
395             return "";
396     
397         std::list<Variable>::const_iterator arg = func->argumentList.begin();
398         for (; arg != func->argumentList.end(); ++arg) {
399             if (arg->index() == targetpar-1)
400                 break;
401         }
402         if (arg == func->argumentList.end())
403             return "";
404     
405         // Is **
406         if (!arg->isPointer())
407             return "";
408         const Token* tok = arg->typeEndToken();
409         tok = tok->previous();
410         if (tok->str() != "*")
411             return "";
412     
413         // Check if pointer is allocated.
414         int realloc = 0;
415         for (tok = func->functionScope->classStart; tok && tok != func->functionScope->classEnd; tok = tok->next()) {
416             if (tok->varId() == arg->declarationId()) {
417                 if (Token::Match(tok->tokAt(-3), "free ( * %name% )")) {
418                     realloc = 1;
419                     allocType = No;
420                 } else if (Token::Match(tok->previous(), "* %name% =")) {
421                     allocType = getAllocationType(tok->tokAt(2), arg->declarationId());
422                     if (allocType == No) {
423                         allocType = getReallocationType(tok->tokAt(2), arg->declarationId());
424                     }
425                     if (allocType != No) {
426                         if (realloc)
427                             return "realloc";
428                         return "alloc";
429                     }
430                 } else {
431                     // unhandled variable usage: bailout
432                     return "";
433                 }
434             }
435         }
436     
437         return "";
438     }

441     static bool notvar(const Token *tok, unsigned int varid)
442     {
443         if (!tok)
444             return false;
445         if (Token::Match(tok, "&&|;"))
446             return notvar(tok->astOperand1(),varid) || notvar(tok->astOperand2(),varid);
447         if (tok->str() == "(" && Token::Match(tok->astOperand1(), "UNLIKELY|LIKELY"))
448             return notvar(tok->astOperand2(), varid);
449         const Token *vartok = astIsVariableComparison(tok, "==", "0");
450         return vartok && (vartok->varId() == varid);
451     }

453     static bool ifvar(const Token *tok, unsigned int varid, const std::string &comp, const std::string &rhs)
454     {
455         if (!Token::simpleMatch(tok, "if ("))
456             return false;
457         const Token *condition = tok->next()->astOperand2();
458         if (condition && condition->str() == "(" && Token::Match(condition->astOperand1(), "UNLIKELY|LIKELY"))
459             condition = condition->astOperand2();
460         if (!condition || condition->str() == "&&")
461             return false;
462     
463         const Token *vartok = astIsVariableComparison(condition, comp, rhs);
464         return (vartok && vartok->varId() == varid);
465     }

468     {
469         if (!tok)
470             return false;

505     const char * CheckMemoryLeakInFunction::call_func(const Token *tok, std::list<const Token *> callstack, const unsigned int varid, AllocType &alloctype, AllocType &dealloctype, bool &allocpar, unsigned int sz)
506     {
507         if (test_white_list(tok->str(), _settings, tokenizer->isCPP())) {
508             if (call_func_keywords.find(tok->str())!=call_func_keywords.end())
509                 return 0;
510     
511             // is the varid a parameter?
512             for (const Token *tok2 = tok->tokAt(2); tok2 && tok2 != tok->linkAt(1); tok2 = tok2->next()) {
513                 if (tok2->str() == "(") {
514                     tok2 = tok2->nextArgument();
515                     if (!tok2)
516                         break;
517                 }
518                 if (tok2->varId() == varid) {
519                     if (tok->strAt(-1) == ".")
520                         return "use";
521                     else if (tok2->strAt(1) == "=")
522                         return "assign";
523                     else if (tok->str()=="printf")
524                         return "use"; // <- it is not certain printf dereference the pointer TODO: check the format string
525                     else
526                         return "use_";
527                 }
528             }
529     
530             return 0;
531         }
532     
533         if (_settings->library.isnoreturn(tok) && tok->strAt(-1) != "=")
534             return "exit";
535     
536         if (varid > 0 && (getReallocationType(tok, varid) != No || getDeallocationType(tok, varid) != No))
537             return 0;
538     
539         if (callstack.size() > 2)
540             return "dealloc_";
541     
542         const std::string& funcname(tok->str());
543         for (std::list<const Token *>::const_iterator it = callstack.begin(); it != callstack.end(); ++it) {
544             if ((*it) && (*it)->str() == funcname)
545                 return "recursive";
546         }
547         callstack.push_back(tok);
548     
549         // lock/unlock..
550         if (varid == 0) {
551             const Function* func = tok->function();
552             if (!func || !func->hasBody())
553                 return 0;
554     
555             Token *ftok = getcode(func->functionScope->classStart->next(), callstack, 0, alloctype, dealloctype, false, 1);
556             simplifycode(ftok);
557             const char *ret = nullptr;
558             if (Token::simpleMatch(ftok, "; alloc ; }"))
559                 ret = "alloc";
560             else if (Token::simpleMatch(ftok, "; dealloc ; }"))
561                 ret = "dealloc";
562             TokenList::deleteTokens(ftok);
563             return ret;
564         }
565     
566         // how many parameters is there in the function call?
567         const unsigned int numpar = countParameters(tok);
568         if (numpar == 0) {
569             // Taking return value => it is not a noreturn function
570             if (tok->strAt(-1) == "=")
571                 return nullptr;
572     
573             // Function is not noreturn
574             if (tok->function() && tok->function()->functionScope) {
575                 std::string temp;
576                 if (!_settings->library.isScopeNoReturn(tok->function()->functionScope->classEnd, &temp) && temp.empty())
577                     return nullptr;
578             } else if (_settings->library.isnotnoreturn(tok))
579                 return nullptr;
580     
581             return "callfunc";
582         }
583     
584         unsigned int par = 0;
585     
586         const bool dot(tok->previous()->str() == ".");
587         const bool eq(tok->previous()->str() == "=");
588     
589         const Token *functok = tok;
590     
591         tok = Token::findsimplematch(tok, "(");
592         if (tok)
593             tok = tok->next();
594     
595         for (; tok; tok = tok->nextArgument()) {
596             ++par;
597             if (Token::Match(tok, "%varid% [,()]", varid)) {
598                 if (dot)
599                     return "use";
600     
601                 const Function* function = functok->function();
602                 if (!function)
603                     return "use";
604     
605                 // how many parameters does the function want?
606                 if (numpar != function->argCount()) // TODO: Handle default parameters
607                     return "recursive";
608     
609                 if (!function->functionScope)
610                     return "use";
611                 const Variable* param = function->getArgumentVar(par-1);
612                 if (!param || !param->nameToken())
613                     return "use";
614                 Token *func = getcode(function->functionScope->classStart->next(), callstack, param->declarationId(), alloctype, dealloctype, false, sz);
615                 //simplifycode(func);
616                 const Token *func_ = func;
617                 while (func_ && func_->str() == ";")
618                     func_ = func_->next();
619     
620                 const char *ret = nullptr;
621                 /** @todo handle "goto" */
622                 if (Token::findsimplematch(func_, "dealloc"))
623                     ret = "dealloc";
624                 else if (Token::findsimplematch(func_, "use"))
625                     ret = "use";
626                 else if (Token::findsimplematch(func_, "&use"))
627                     ret = "&use";
628     
629                 TokenList::deleteTokens(func);
630                 return ret;
631             }
632             if (Token::Match(tok, "& %varid% [,()]", varid)) {
633                 const Function *func = functok->function();
634                 if (func == nullptr)
635                     continue;
636                 AllocType a;
637                 const char *ret = functionArgAlloc(func, par, a);
638     
639                 if (a != No) {
640                     if (alloctype == No)
641                         alloctype = a;
642                     else if (alloctype != a)
643                         alloctype = Many;
644                     allocpar = true;
645                     return ret;
646                 }
647             }
648             if (Token::Match(tok, "%varid% . %name% [,)]", varid))
649                 return "use";
650         }
651         return (eq || _settings->experimental) ? 0 : "callfunc";
652     }

655     static void addtoken(Token **rettail, const Token *tok, const std::string &str)
656     {
657         (*rettail)->insertToken(str);
658         (*rettail) = (*rettail)->next();
659         (*rettail)->linenr(tok->linenr());
660         (*rettail)->fileIndex(tok->fileIndex());
661     }

663     
664     Token *CheckMemoryLeakInFunction::getcode(const Token *tok, std::list<const Token *> callstack, const unsigned int varid, CheckMemoryLeak::AllocType &alloctype, CheckMemoryLeak::AllocType &dealloctype, bool classmember, unsigned int sz)
665     {
666         // variables whose value depends on if(!var). If one of these variables
667         // is used in a if-condition then generate "ifv" instead of "if".
668         std::set<unsigned int> extravar;
669     
670         // The first token should be ";"
671         Token* rethead = new Token(0);
672         rethead->str(";");
673         rethead->linenr(tok->linenr());
674         rethead->fileIndex(tok->fileIndex());
675         Token* rettail = rethead;
676     
677         int indentlevel = 0;
678         int parlevel = 0;
679         for (; tok; tok = tok->next()) {
680             if (tok->str() == "{") {
681                 addtoken(&rettail, tok, "{");
682                 ++indentlevel;
683             } else if (tok->str() == "}") {
684                 addtoken(&rettail, tok, "}");
685                 if (indentlevel <= 0)
686                     break;
687                 --indentlevel;
688             }
689     
690             else if (tok->str() == "(")
691                 ++parlevel;
692             else if (tok->str() == ")")
693                 --parlevel;
694     
695             if (parlevel == 0 && tok->str() == ";")
696                 addtoken(&rettail, tok, ";");
697     
698             // Start of new statement.. check if the statement has anything interesting
699             if (varid > 0 && parlevel == 0 && Token::Match(tok, "[;{}]")) {
700                 if (Token::Match(tok->next(), "[{};]"))
701                     continue;
702     
703                 // function calls are interesting..
704                 const Token *tok2 = tok;
705                 while (Token::Match(tok2->next(), "%name% ."))
706                     tok2 = tok2->tokAt(2);
707                 if (Token::Match(tok2->next(), "%name% ("))
708                     ;
709     
710                 else if (Token::Match(tok->next(), "continue|break|return|throw|goto|do|else"))
711                     ;
712     
713                 else {
714                     const Token *skipToToken = nullptr;
715     
716                     // scan statement for interesting keywords / varid
717                     for (tok2 = tok->next(); tok2; tok2 = tok2->next()) {
718                         if (tok2->str() == ";") {
719                             // nothing interesting found => skip this statement
720                             skipToToken = tok2->previous();
721                             break;
722                         }
723     
724                         if (tok2->varId() == varid ||
725                             tok2->str() == ":" || tok2->str() == "{" || tok2->str() == "}") {
726                             break;
727                         }
728                     }
729     
730                     if (skipToToken) {
731                         tok = skipToToken;
732                         continue;
733                     }
734                 }
735             }
736     
737             if (varid == 0) {
738                 if (!callstack.empty() && Token::Match(tok, "[;{}] __cppcheck_lock|__cppcheck_unlock ( ) ;")) {
739                     // Type of leak = Resource leak
740                     alloctype = dealloctype = CheckMemoryLeak::File;
741     
742                     if (tok->next()->str() == "__cppcheck_lock") {
743                         addtoken(&rettail, tok, "alloc");
744                     } else {
745                         addtoken(&rettail, tok, "dealloc");
746                     }
747     
748                     tok = tok->tokAt(3);
749                     continue;
750                 }
751     
752                 if (Token::simpleMatch(tok, "if (")) {
753                     addtoken(&rettail, tok, "if");
754                     tok = tok->next()->link();
755                     continue;
756                 }
757             } else {
758     
759                 if (Token::Match(tok, "%varid% = close ( %varid% )", varid)) {
760                     addtoken(&rettail, tok, "dealloc");
761                     addtoken(&rettail, tok, ";");
762                     addtoken(&rettail, tok, "assign");
763                     addtoken(&rettail, tok, ";");
764                     tok = tok->tokAt(5);
765                     continue;
766                 }
767     
768                 // var = strcpy|.. ( var ,
769                 if (Token::Match(tok, "[;{}] %varid% = memcpy|memmove|memset|strcpy|strncpy|strcat|strncat ( %varid% ,", varid)) {
770                     tok = tok->linkAt(4);
771                     continue;
772                 }
773     
774                 if (Token::Match(tok->previous(), "[(;{}] %varid% =", varid) ||
775                     Token::Match(tok, "asprintf|vasprintf ( & %varid% ,", varid)) {
776                     CheckMemoryLeak::AllocType alloc;
777     
778                     if (Token::Match(tok, "asprintf|vasprintf (")) {
779                         // todo: check how the return value is used.
780                         if (!Token::Match(tok->previous(), "[;{}]")) {
781                             TokenList::deleteTokens(rethead);
782                             return 0;
783                         }
784                         alloc = Malloc;
785                         tok = tok->next()->link();
786                     } else {
787                         alloc = getAllocationType(tok->tokAt(2), varid);
788                     }
789     
790                     if (sz > 1 &&
791                         Token::Match(tok->tokAt(2), "malloc ( %num% )") &&
792                         (MathLib::toLongNumber(tok->strAt(4)) % long(sz)) != 0) {
793                         mismatchSizeError(tok->tokAt(4), tok->strAt(4));
794                     }
795     
796                     if (alloc == CheckMemoryLeak::No) {
797                         alloc = getReallocationType(tok->tokAt(2), varid);
798                         if (alloc != CheckMemoryLeak::No) {
799                             addtoken(&rettail, tok, "realloc");
800                             addtoken(&rettail, tok, ";");
801                             tok = tok->tokAt(2);
802                             if (Token::Match(tok, "%name% ("))
803                                 tok = tok->next()->link();
804                             continue;
805                         }
806                     }
807     
808                     // don't check classes..
809                     if (alloc == CheckMemoryLeak::New) {
810                         if (Token::Match(tok->tokAt(2), "new struct| %type% [(;]")) {
811                             const int offset = tok->strAt(3) == "struct" ? 1 : 0;
812                             if (isclass(tok->tokAt(3 + offset), varid)) {
813                                 alloc = No;
814                             }
815                         } else if (Token::Match(tok->tokAt(2), "new ( nothrow ) struct| %type%")) {
816                             const int offset = tok->strAt(6) == "struct" ? 1 : 0;
817                             if (isclass(tok->tokAt(6 + offset), varid)) {
818                                 alloc = No;
819                             }
820                         } else if (Token::Match(tok->tokAt(2), "new ( std :: nothrow ) struct| %type%")) {
821                             const int offset = tok->strAt(8) == "struct" ? 1 : 0;
822                             if (isclass(tok->tokAt(8 + offset), varid)) {
823                                 alloc = No;
824                             }
825                         }
826     
827                         if (alloc == No && alloctype == No)
828                             alloctype = CheckMemoryLeak::New;
829                     }
830     
831                     if (alloc != No) {
832                         addtoken(&rettail, tok, "alloc");
833     
834                         if (alloctype != No && alloctype != alloc)
835                             alloc = Many;
836     
837                         if (alloc != Many && dealloctype != No && dealloctype != Many && dealloctype != alloc) {
838                             callstack.push_back(tok);
839                             mismatchAllocDealloc(callstack, Token::findmatch(_tokenizer->tokens(), "%varid%", varid)->str());
840                             callstack.pop_back();
841                         }
842     
843                         alloctype = alloc;
844     
845                         if (Token::Match(tok, "%name% = %type% (")) {
846                             tok = tok->linkAt(3);
847                             continue;
848                         }
849                     }
850     
851                     // assignment..
852                     else {
853                         // is the pointer in rhs?
854                         bool rhs = false;
855                         bool trailingSemicolon = false;
856                         bool used = false;
857                         for (const Token *tok2 = tok->next(); tok2; tok2 = tok2->next()) {
858                             if (tok2->str() == ";") {
859                                 trailingSemicolon = true;
860                                 if (rhs)
861                                     tok = tok2;
862                                 break;
863                             }
864     
865                             if (!used && !rhs) {
866                                 if (Token::Match(tok2, "[=+(,] %varid%", varid)) {
867                                     if (Token::Match(tok2, "[(,]")) {
868                                         used = true;
869                                         addtoken(&rettail, tok, "use");
870                                         addtoken(&rettail, tok, ";");
871                                     }
872                                     rhs = true;
873                                 }
874                             }
875                         }
876     
877                         if (!used) {
878                             if (!rhs)
879                                 addtoken(&rettail, tok, "assign");
880                             else {
881                                 addtoken(&rettail, tok, "use_");
882                                 if (trailingSemicolon)
883                                     addtoken(&rettail, tok, ";");
884                             }
885                         }
886                         continue;
887                     }
888                 }
889     
890                 if (Token::Match(tok->previous(), "%op%|;|{|}|) ::| %name%") || (Token::Match(tok->previous(), "( ::| %name%") && (!rettail || rettail->str() != "loop"))) {
891                     if (Token::Match(tok, "%varid% ?", varid))
892                         tok = tok->tokAt(2);
893     
894                     AllocType dealloc = getDeallocationType(tok, varid);
895     
896                     if (dealloc != No && tok->str() == "fcloseall" && alloctype != dealloc)
897                         ;
898     
899                     else if (dealloc != No) {
900                         addtoken(&rettail, tok, "dealloc");
901     
902                         if (dealloctype != No && dealloctype != dealloc)
903                             dealloc = Many;
904     
905                         if (dealloc != Many && alloctype != No && alloctype != Many && alloctype != dealloc) {
906                             callstack.push_back(tok);
907                             mismatchAllocDealloc(callstack, Token::findmatch(_tokenizer->tokens(), "%varid%", varid)->str());
908                             callstack.pop_back();
909                         }
910                         dealloctype = dealloc;
911     
912                         if (tok->strAt(2) == "(")
913                             tok = tok->linkAt(2);
914                         continue;
915                     }
916                 }
917     
918                 // if else switch
919                 if (Token::simpleMatch(tok, "if (")) {
920                     if (alloctype == Fd) {
921                         if (ifvar(tok, varid, ">",  "-1") ||
922                             ifvar(tok, varid, ">=", "0") ||
923                             ifvar(tok, varid, ">",  "0") ||
924                             ifvar(tok, varid, "!=", "-1")) {
925                             addtoken(&rettail, tok, "if(var)");
926                             tok = tok->next()->link();
927                             continue;
928                         } else if (ifvar(tok, varid, "==", "-1") ||
929                                    ifvar(tok, varid, "<", "0")) {
930                             addtoken(&rettail, tok, "if(!var)");
931                             tok = tok->next()->link();
932                             continue;
933                         }
934                     }
935     
936                     if (ifvar(tok, varid, "!=", "0")) {
937                         addtoken(&rettail, tok, "if(var)");
938     
939                         // Make sure the "use" will not be added
940                         tok = tok->next()->link();
941                         continue;
942                     } else if (ifvar(tok, varid, "==", "0")) {
943                         addtoken(&rettail, tok, "if(!var)");
944     
945                         // parse the if-body.
946                         // if a variable is assigned then add variable to "extravar".
947                         for (const Token *tok2 = tok->next()->link()->tokAt(2); tok2; tok2 = tok2->next()) {
948                             if (tok2->str() == "{")
949                                 tok2 = tok2->link();
950                             else if (tok2->str() == "}")
951                                 break;
952                             else if (Token::Match(tok2, "%var% ="))
953                                 extravar.insert(tok2->varId());
954                         }
955     
956                         tok = tok->next()->link();
957                         continue;
958                     } else {
959                         // Check if the condition depends on var or extravar somehow..
960                         bool dep = false;
961                         const Token* const end = tok->linkAt(1);
962                         for (const Token *tok2 = tok->next(); tok2 != end; tok2 = tok2->next()) {
963                             if (Token::Match(tok2, "close|pclose|fclose|closedir ( %varid% )", varid)) {
964                                 addtoken(&rettail, tok, "dealloc");
965                                 addtoken(&rettail, tok, ";");
966                                 dep = true;
967                                 break;
968                             } else if (alloctype == Fd && Token::Match(tok2, "%varid% !=|>=", varid)) {
969                                 dep = true;
970                             } else if (Token::Match(tok2, "! %varid%", varid)) {
971                                 dep = true;
972                             } else if (Token::Match(tok2, "%name% (") && !test_white_list(tok2->str(), _settings, tokenizer->isCPP())) {
973                                 bool use = false;
974                                 for (const Token *tok3 = tok2->tokAt(2); tok3; tok3 = tok3->nextArgument()) {
975                                     if (Token::Match(tok3->previous(), "(|, &| %varid% ,|)", varid)) {
976                                         use = true;
977                                         break;
978                                     }
979                                 }
980                                 if (use) {
981                                     addtoken(&rettail, tok, "use");
982                                     addtoken(&rettail, tok, ";");
983                                     dep = false;
984                                     break;
985                                 }
986                             } else if (tok2->varId() && extravar.find(tok2->varId()) != extravar.end()) {
987                                 dep = true;
988                             } else if (tok2->varId() == varid &&
989                                        (tok2->next()->isConstOp() || tok2->previous()->isConstOp()))
990                                 dep = true;
991                         }
992     
993                         if (notvar(tok->next()->astOperand2(), varid))
994                             addtoken(&rettail, tok, "if(!var)");
995                         else
996                             addtoken(&rettail, tok, (dep ? "ifv" : "if"));
997     
998                         tok = tok->next()->link();
999                         continue;
1000                     }
1001                 }
1002             }
1003     
1004             if ((tok->str() == "else") || (tok->str() == "switch")) {
1005                 addtoken(&rettail, tok, tok->str());
1006                 if (Token::simpleMatch(tok, "switch ("))
1007                     tok = tok->next()->link();
1008                 continue;
1009             }
1010     
1011             if ((tok->str() == "case")) {
1012                 addtoken(&rettail, tok, "case");
1013                 addtoken(&rettail, tok, ";");
1014                 if (Token::Match(tok, "case %any% :"))
1015                     tok = tok->tokAt(2);
1016                 continue;
1017             }
1018     
1019             if ((tok->str() == "default")) {
1020                 addtoken(&rettail, tok, "default");
1021                 addtoken(&rettail, tok, ";");
1022                 continue;
1023             }
1024     
1025             // Loops..
1026             else if ((tok->str() == "for") || (tok->str() == "while")) {
1027                 const Token* const end = tok->linkAt(1);
1028     
1029                 if ((Token::simpleMatch(tok, "while (") && alwaysTrue(tok->next()->astOperand2())) ||
1030                     Token::simpleMatch(tok, "for ( ; ; )")) {
1031                     addtoken(&rettail, tok, "while1");
1032                     tok = end;
1033                     continue;
1034                 }
1035     
1036                 else if (varid && getDeallocationType(tok->tokAt(2), varid) != No) {
1037                     addtoken(&rettail, tok, "dealloc");
1038                     addtoken(&rettail, tok, ";");
1039                 }
1040     
1041                 else if (alloctype == Fd && varid) {
1042                     if (Token::Match(tok, "while ( 0 <= %varid% )", varid) ||
1043                         Token::Match(tok, "while ( %varid% >= 0 )", varid) ||
1044                         Token::Match(tok, "while ( %varid% != -1 )", varid) ||
1045                         Token::Match(tok, "while ( -1 != %varid% )", varid)) {
1046                         addtoken(&rettail, tok, "while(var)");
1047                         tok = end;
1048                         continue;
1049                     } else if (Token::Match(tok, "while ( %varid% == -1 )", varid) ||
1050                                Token::Match(tok, "while ( -1 == %varid% )", varid) ||
1051                                Token::Match(tok, "while ( %varid% < 0 )", varid) ||
1052                                Token::Match(tok, "while ( 0 > %varid% )", varid)) {
1053                         addtoken(&rettail, tok, "while(!var)");
1054                         tok = end;
1055                         continue;
1056                     }
1057                 }
1058     
1059                 else if (varid && Token::Match(tok, "while ( %varid% )", varid)) {
1060                     addtoken(&rettail, tok, "while(var)");
1061                     tok = end;
1062                     continue;
1063                 } else if (varid && Token::simpleMatch(tok, "while (") && notvar(tok->next()->astOperand2(), varid)) {
1064                     addtoken(&rettail, tok, "while(!var)");
1065                     tok = end;
1066                     continue;
1067                 }
1068     
1069                 addtoken(&rettail, tok, "loop");
1070     
1071                 if (varid > 0 && notvar(tok->next()->astOperand2(), varid))
1072                     addtoken(&rettail, tok, "!var");
1073     
1074                 continue;
1075             }
1076             if ((tok->str() == "do")) {
1077                 addtoken(&rettail, tok, "do");
1078                 continue;
1079             }
1080     
1081             // continue / break..
1082             else if (tok->str() == "continue") {
1083                 addtoken(&rettail, tok, "continue");
1084             } else if (tok->str() == "break") {
1085                 addtoken(&rettail, tok, "break");
1086             } else if (tok->str() == "goto") {
1087                 addtoken(&rettail, tok, "goto");
1088             }
1089     
1090             // Return..
1091             else if (tok->str() == "return") {
1092                 addtoken(&rettail, tok, "return");
1093                 if (varid == 0) {
1094                     addtoken(&rettail, tok, ";");
1095                     while (tok && tok->str() != ";")
1096                         tok = tok->next();
1097                     if (!tok)
1098                         break;
1099                     continue;
1100                 }
1101     
1102                 // Returning a auto_ptr of this allocated variable..
1103                 if (Token::simpleMatch(tok->next(), "std :: auto_ptr <")) {
1104                     const Token *tok2 = tok->linkAt(4);
1105                     if (Token::Match(tok2, "> ( %varid% )", varid)) {
1106                         addtoken(&rettail, tok, "use");
1107                         tok = tok2->tokAt(3);
1108                     }
1109                 }
1110     
1111                 else if (varid && Token::Match(tok, "return strcpy|strncpy|memcpy ( %varid%", varid)) {
1112                     addtoken(&rettail, tok, "use");
1113                     tok = tok->tokAt(2);
1114                 }
1115     
1116                 else {
1117                     bool use = false;
1118     
1119                     std::stack<const Token *> functions;
1120     
1121                     for (const Token *tok2 = tok->next(); tok2; tok2 = tok2->next()) {
1122                         if (tok2->str() == ";") {
1123                             tok = tok2;
1124                             break;
1125                         }
1126     
1127                         if (tok2->str() == "(")
1128                             functions.push(tok2->previous());
1129                         else if (!functions.empty() && tok2->str() == ")")
1130                             functions.pop();
1131     
1132                         if (tok2->varId() == varid) {
1133                             // Read data..
1134                             if (!Token::Match(tok2->previous(), "&|(") &&
1135                                 tok2->strAt(1) == "[") {
1136                                 ;
1137                             } else if (functions.empty() ||
1138                                        !test_white_list(functions.top()->str(), _settings, tokenizer->isCPP()) ||
1139                                        getDeallocationType(functions.top(),varid)) {
1140                                 use = true;
1141                             }
1142                         }
1143                     }
1144                     if (use)
1145                         addtoken(&rettail, tok, "use");
1146                     addtoken(&rettail, tok, ";");
1147                 }
1148             }
1149     
1150             // throw..
1151             else if (tokenizer->isCPP() && Token::Match(tok, "try|throw|catch")) {
1152                 addtoken(&rettail, tok, tok->str());
1153                 if (tok->strAt(1) == "(")
1154                     tok = tok->next()->link();
1155             }
1156     
1157             // Assignment..
1158             if (varid) {
1159                 if (Token::simpleMatch(tok, "= {")) {
1160                     const Token* const end2 = tok->linkAt(1);
1161                     bool use = false;
1162                     for (const Token *tok2 = tok; tok2 != end2; tok2 = tok2->next()) {
1163                         if (tok2->varId() == varid) {
1164                             use = true;
1165                             break;
1166                         }
1167                     }
1168     
1169                     if (use) {
1170                         addtoken(&rettail, tok, "use");
1171                         addtoken(&rettail, tok, ";");
1172                         tok = tok->next()->link();
1173                         continue;
1174                     }
1175                 }
1176     
1177                 if (Token::Match(tok, "& %name% = %varid% ;", varid)) {
1178                     while (rethead->next())
1179                         rethead->deleteNext();
1180                     return rethead;
1181                 }
1182                 if (Token::Match(tok, "[)=] %varid% [+;)]", varid) ||
1183                     (Token::Match(tok, "%name% + %varid%", varid) &&
1184                      tok->strAt(3) != "[" &&
1185                      tok->strAt(3) != ".") ||
1186                     Token::Match(tok, "<< %varid% ;", varid) ||
1187                     Token::Match(tok, "= strcpy|strcat|memmove|memcpy ( %varid% ,", varid) ||
1188                     Token::Match(tok, "[;{}] %name% [ %varid% ]", varid)) {
1189                     addtoken(&rettail, tok, "use");
1190                 } else if (Token::Match(tok->previous(), ";|{|}|=|(|,|%cop% %varid% .|[", varid)) {
1191                     // warning is written for "dealloc ; use_ ;".
1192                     // but this use doesn't affect the leak-checking
1193                     addtoken(&rettail, tok, "use_");
1194                 }
1195             }
1196     
1197             // Investigate function calls..
1198             if (Token::Match(tok, "%name% (")) {
1199                 // A function call should normally be followed by ";"
1200                 if (Token::simpleMatch(tok->next()->link(), ") {")) {
1201                     if (!Token::Match(tok, "if|for|while|switch")) {
1202                         addtoken(&rettail, tok, "exit");
1203                         addtoken(&rettail, tok, ";");
1204                         tok = tok->next()->link();
1205                         continue;
1206                     }
1207                 }
1208     
1209                 // Calling setjmp / longjmp => bail out
1210                 else if (Token::Match(tok, "setjmp|longjmp")) {
1211                     while (rethead->next())
1212                         rethead->deleteNext();
1213                     return rethead;
1214                 }
1215     
1216                 // Inside class function.. if the var is passed as a parameter then
1217                 // just add a "::use"
1218                 // The "::use" means that a member function was probably called but it wasn't analysed further
1219                 else if (classmember) {
1220                     if (_settings->library.isnoreturn(tok))
1221                         addtoken(&rettail, tok, "exit");
1222

1921     const Token *CheckMemoryLeakInFunction::findleak(const Token *tokens)
1922     {
1923         const Token *result;
1924     
1925         if (Token::Match(tokens, "alloc ; if|if(var)|ifv break|continue|return ;")) {
1926             return tokens->tokAt(3);
1927         }
1928     
1929         if ((result = Token::findsimplematch(tokens, "loop alloc ;")) != nullptr) {
1930             return result;
1931         }
1932     
1933         if ((result = Token::findmatch(tokens, "alloc ; if|if(var)|ifv return ;")) != nullptr) {
1934             return result->tokAt(3);
1935         }
1936     
1937         if ((result = Token::findmatch(tokens, "alloc ; alloc|assign|return callfunc| ;")) != nullptr) {
1938             return result->tokAt(2);
1939         }
1940     
1941         if ((result = Token::findsimplematch(tokens, "; alloc ; if assign ;")) != nullptr) {
1942             return result->tokAt(4);
1943         }
1944     
1945         if (((result = Token::findsimplematch(tokens, "; alloc ; if dealloc ; }")) != nullptr) &&
1946             !result->tokAt(7)) {
1947             return result->tokAt(6);
1948         }
1949     
1950         if ((result = Token::findsimplematch(tokens, "alloc ; }")) != nullptr) {
1951             if (result->tokAt(3) == nullptr)
1952                 return result->tokAt(2);
1953         }
1954     
1955         // No deallocation / usage => report leak at the last token
1956         if (!Token::findmatch(tokens, "dealloc|use")) {
1957             const Token *last = tokens;
1958             while (last->next())
1959                 last = last->next();
1960     
1961             // not a leak if exit is called before the end of the function
1962             if (!Token::Match(last->tokAt(-2), "exit|callfunc ; }"))
1963                 return last;
1964         }
1965     
1966         return nullptr;
1967     }

1971     void CheckMemoryLeakInFunction::checkScope(const Token *startTok, const std::string &varname, unsigned int varid, bool classmember, unsigned int sz)
1972     {
1973         std::list<const Token *> callstack;
1974     
1975         AllocType alloctype = No;
1976         AllocType dealloctype = No;
1977     
1978         const Token *result;
1979     
1980         Token *tok = getcode(startTok, callstack, varid, alloctype, dealloctype, classmember, sz);
1981         //tok->printOut((std::string("Checkmemoryleak: getcode result for: ") + varname).c_str());
1982     
1983         const bool use_addr = bool(Token::findsimplematch(tok, "&use") != nullptr);
1984     
1985         // Simplify the code and check if freed memory is used..
1986         for (Token *tok2 = tok; tok2; tok2 = tok2->next()) {
1987             while (Token::Match(tok2, "[;{}] ;"))
1988                 tok2->deleteNext();
1989         }
1990         if ((result = Token::findmatch(tok, "[;{}] dealloc ; use_ ;")) != nullptr) {
1991             deallocuseError(result->tokAt(3), varname);
1992         }
1993     
1994         // Replace "&use" with "use". Replace "use_" with ";"
1995         for (Token *tok2 = tok; tok2; tok2 = tok2->next()) {
1996             if (tok2->str() == "&use")
1997                 tok2->str("use");
1998             else if (tok2->str() == "use_")
1999                 tok2->str(";");
2000             else if (Token::simpleMatch(tok2, "loop use_ {"))
2001                 tok2->deleteNext();
2002             else if (tok2->str() == "::use")    // Some kind of member function usage. Not analyzed very well.
2003                 tok2->str("use");
2004             else if (tok2->str() == "recursive")
2005                 tok2->str("use");
2006             else if (tok2->str() == "dealloc_")
2007                 tok2->str("dealloc");
2008             else if (tok2->str() == "realloc") {
2009                 tok2->str("dealloc");
2010                 tok2->insertToken("alloc");
2011                 tok2->insertToken(";");
2012             }
2013         }
2014     
2015         // If the variable is not allocated at all => no memory leak
2016         if (Token::findsimplematch(tok, "alloc") == 0) {
2017             TokenList::deleteTokens(tok);
2018             return;
2019         }
2020     
2021         simplifycode(tok);
2022     
2023         if (_settings->debug && _settings->verbose) {
2024             tok->printOut(("Checkmemoryleak: simplifycode result for: " + varname).c_str());
2025         }
2026     
2027         // If the variable is not allocated at all => no memory leak
2028         if (Token::findsimplematch(tok, "alloc") == 0) {
2029             TokenList::deleteTokens(tok);
2030             return;
2031         }
2032     
2033         /** @todo handle "goto" */
2034         if (Token::findsimplematch(tok, "goto")) {
2035             TokenList::deleteTokens(tok);
2036             return;
2037         }
2038     
2039         if ((result = findleak(tok)) != nullptr) {
2040             memoryLeak(result, varname, alloctype);
2041         }
2042     
2043         else if (!use_addr && (result = Token::findsimplematch(tok, "dealloc ; dealloc ;")) != nullptr) {
2044             deallocDeallocError(result->tokAt(2), varname);
2045         }
2046     
2047         // detect cases that "simplifycode" don't handle well..
2048         else if (tok && _settings->debugwarnings) {
2049             Token *first = tok;
2050             while (first && first->str() == ";")
2051                 first = first->next();
2052     
2053             bool noerr = false;
2054             noerr = noerr || Token::simpleMatch(first, "alloc ; }");
2055             noerr = noerr || Token::simpleMatch(first, "alloc ; dealloc ; }");
2056             noerr = noerr || Token::simpleMatch(first, "alloc ; return use ; }");
2057             noerr = noerr || Token::simpleMatch(first, "alloc ; use ; }");
2058             noerr = noerr || Token::simpleMatch(first, "alloc ; use ; return ; }");
2059             noerr = noerr || Token::simpleMatch(first, "alloc ; dealloc ; return ; }");
2060             noerr = noerr || Token::simpleMatch(first, "if alloc ; dealloc ; }");
2061             noerr = noerr || Token::simpleMatch(first, "if alloc ; return use ; }");
2062             noerr = noerr || Token::simpleMatch(first, "if alloc ; use ; }");
2063             noerr = noerr || Token::simpleMatch(first, "alloc ; ifv return ; dealloc ; }");
2064             noerr = noerr || Token::simpleMatch(first, "alloc ; if return ; dealloc; }");
2065     
2066             // Unhandled case..
2067             if (!noerr)
2068                 reportError(first, Severity::debug, "debug",
2069                             "inconclusive leak of " + varname + ": " + tok->stringifyList(false, false, false, false, false, 0, 0));
2070         }
2071     
2072         TokenList::deleteTokens(tok);
2073     }

2423         // Check struct..
2424         unsigned int indentlevel2 = 0;
2425         for (const Token *tok2 = variable->nameToken(); tok2 && tok2 != variable->scope()->classEnd; tok2 = tok2->next()) {
2426             if (tok2->str() == "{")
2427                 ++indentlevel2;
2428     
2429             else if (tok2->str() == "}") {
2430                 if (indentlevel2 == 0)
2431                     break;
2432                 --indentlevel2;
2433             }
2434     
2435             // Unknown usage of struct
2436             /** @todo Check how the struct is used. Only bail out if necessary */
2437             else if (Token::Match(tok2, "[(,] %varid% [,)]", variable->declarationId()))
2438                 break;
2439     
2440             // Struct member is allocated => check if it is also properly deallocated..
2441             else if (Token::Match(tok2->previous(), "[;{}] %varid% . %var% =", variable->declarationId())) {
2442                 if (getAllocationType(tok2->tokAt(4), tok2->tokAt(2)->varId()) == AllocType::No)
2443                     continue;
2444     
2445                 const unsigned int structid(variable->declarationId());
2446                 const unsigned int structmemberid(tok2->tokAt(2)->varId());
2447     
2448                 // This struct member is allocated.. check that it is deallocated
2449                 unsigned int indentlevel3 = indentlevel2;
2450                 for (const Token *tok3 = tok2; tok3; tok3 = tok3->next()) {
2451                     if (tok3->str() == "{")
2452                         ++indentlevel3;
2453     
2454                     else if (tok3->str() == "}") {
2455                         if (indentlevel3 == 0) {
2456                             memoryLeak(tok3, variable->name() + "." + tok2->strAt(2), Malloc);
2457                             break;
2458                         }
2459                         --indentlevel3;
2460                     }
2461     
2462                     // Deallocating the struct member..
2463                     else if (getDeallocationType(tok3, structmemberid) != AllocType::No) {
2464                         // If the deallocation happens at the base level, don't check this member anymore
2465                         if (indentlevel3 == 0)
2466                             break;
2467     
2468                         // deallocating and then returning from function in a conditional block =>
2469                         // skip ahead out of the block
2470                         bool ret = false;
2471                         while (tok3) {
2472                             if (tok3->str() == "return")
2473                                 ret = true;
2474                             else if (tok3->str() == "{" || tok3->str() == "}")
2475                                 break;
2476                             tok3 = tok3->next();
2477                         }
2478                         if (!ret || !tok3 || tok3->str() != "}")
2479                             break;
2480                         --indentlevel3;
2481                         continue;
2482                     }
2483     
2484                     // Deallocating the struct..
2485                     else if (Token::Match(tok3, "free|kfree ( %varid% )", structid)) {
2486                         if (indentlevel2 == 0)
2487                             memoryLeak(tok3, variable->name() + "." + tok2->strAt(2), Malloc);
2488                         break;
2489                     }
2490     
2491                     // failed allocation => skip code..
2492                     else if (Token::simpleMatch(tok3, "if (") &&
2493                              notvar(tok3->next()->astOperand2(), structmemberid)) {
2494                         // Goto the ")"
2495                         tok3 = tok3->next()->link();
2496     
2497                         // make sure we have ") {".. it should be
2498                         if (!Token::simpleMatch(tok3, ") {"))
2499                             break;
2500     
2501                         // Goto the "}"
2502                         tok3 = tok3->next()->link();
2503                     }
2504     
2505                     // succeeded allocation
2506                     else if (ifvar(tok3, structmemberid, "!=", "0")) {
2507                         // goto the ")"
2508                         tok3 = tok3->next()->link();
2509     
2510                         // check if the variable is deallocated or returned..
2511                         unsigned int indentlevel4 = 0;
2512                         for (const Token *tok4 = tok3; tok4; tok4 = tok4->next()) {
2513                             if (tok4->str() == "{")
2514                                 ++indentlevel4;
2515                             else if (tok4->str() == "}") {
2516                                 --indentlevel4;
2517                                 if (indentlevel4 == 0)
2518                                     break;
2519                             } else if (Token::Match(tok4, "free|kfree ( %var% . %varid% )", structmemberid)) {
2520                                 break;
2521                             }
2522                         }
2523     
2524                         // was there a proper deallocation?
2525                         if (indentlevel4 > 0)
2526                             break;
2527                     }
2528     
2529                     // Returning from function..
2530                     else if (tok3->str() == "return") {
2531                         // Returning from function without deallocating struct member?
2532                         if (!Token::Match(tok3, "return %varid% ;", structid) &&
2533                             !Token::Match(tok3, "return & %varid% .", structid)) {
2534                             memoryLeak(tok3, variable->name() + "." + tok2->strAt(2), Malloc);
2535                         }
2536                         break;
2537                     }
2538     
2539                     // struct assignment..
2540                     else if (Token::Match(tok3, "= %varid% ;", structid)) {
2541                         break;
2542                     } else if (Token::Match(tok3, "= %var% . %varid% ;", structmemberid)) {
2543                         break;
2544                     }
2545     
2546                     // goto isn't handled well.. bail out even though there might be leaks
2547                     else if (tok3->str() == "goto")
2548                         break;
2549     
2550                     // using struct in a function call..
2551                     else if (Token::Match(tok3, "%name% (")) {
2552                         // Calling non-function / function that doesn't deallocate?
2553                         if (CheckMemoryLeakInFunction::test_white_list(tok3->str(), _settings, tokenizer->isCPP()))
2554                             continue;
2555     
2556                         // Check if the struct is used..
2557                         bool deallocated = false;
2558                         const Token* const end4 = tok3->linkAt(1);
2559                         for (const Token *tok4 = tok3; tok4 != end4; tok4 = tok4->next()) {
2560                             if (Token::Match(tok4, "[(,] &| %varid% [,)]", structid)) {
2561                                 /** @todo check if the function deallocates the memory */
2562                                 deallocated = true;
2563                                 break;
2564                             }
2565     
2566                             if (Token::Match(tok4, "[(,] &| %varid% . %name% [,)]", structid)) {
2567                                 /** @todo check if the function deallocates the memory */
2568                                 deallocated = true;
2569                                 break;
2570                             }
2571                         }
2572     
2573                         if (deallocated)
2574                             break;
2575                     }
2576                 }
2577             }
2578         }
2579     }

2583     void CheckMemoryLeakNoVar::check()
2584     {
2585         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
2586     
2587         // only check functions
2588         const std::size_t functions = symbolDatabase->functionScopes.size();
2589         for (std::size_t i = 0; i < functions; ++i) {
2590             const Scope * scope = symbolDatabase->functionScopes[i];
2591     
2592             // Checks if a call to an allocation function like malloc() is made and its return value is not assigned.
2593             checkForUnusedReturnValue(scope);
2594     
2595             // Checks to see if a function is called with memory allocated for an argument that
2596             // could be leaked if a function called for another argument throws.
2597             checkForUnsafeArgAlloc(scope);
2598     
2599             // parse the executable scope until tok is reached...
2600             for (const Token *tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
2601                 // allocating memory in parameter for function call..
2602                 if (Token::Match(tok, "[(,] %name% (") && Token::Match(tok->linkAt(2), ") [,)]")) {
2603                     if (getAllocationType(tok->next(), 0) != No) {
2604                         // locate outer function call..
2605                         const Token* tok3 = tok;
2606                         while (tok3->astParent() && tok3->str() == ",")
2607                             tok3 = tok3->astParent();
2608                         if (tok3 && tok3->str() == "(") {
2609                             // Is it a function call..
2610                             if (!Token::Match(tok3->tokAt(-2), "= %name% (")) {
2611                                 const std::string& functionName = tok3->strAt(-1);
2612                                 if ((tokenizer->isCPP() && functionName == "delete") ||
2613                                     functionName == "free" ||
2614                                     functionName == "fclose" ||
2615                                     functionName == "realloc")
2616                                     break;
2617                                 if (CheckMemoryLeakInFunction::test_white_list(functionName, _settings, tokenizer->isCPP())) {
2618                                     functionCallLeak(tok, tok->strAt(1), functionName);
2619                                     break;
2620                                 }
2621                             }
2622                         }
2623                     }
2624                 }
2625             }
2626         }
2627     }

2633     {
2634         for (const Token *tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {

2636                 && (!tok->next()->astParent() || tok->next()->astParent()->str() == "!" || tok->next()->astParent()->isComparisonOp()) && tok->next()->astOperand1() == tok) {
2637                 const AllocType allocType = getAllocationType(tok, 0);
2638                 if (allocType != No)
2639                     returnValueNotUsedError(tok, tok->str());
2640             }

2644     //---------------------------------------------------------------------------
2645     // Check if an exception could cause a leak in an argument constructed with
2646     // shared_ptr/unique_ptr. For example, in the following code, it is possible
2647     // that if g() throws an exception, the memory allocated by "new int(42)"
2648     // could be leaked. See stackoverflow.com/questions/19034538/
2649     // why-is-there-memory-leak-while-using-shared-ptr-as-a-function-parameter
2650     //
2651     // void x() {
2652     //    f(shared_ptr<int>(new int(42)), g());
2653     // }
2654     //---------------------------------------------------------------------------
2655     void CheckMemoryLeakNoVar::checkForUnsafeArgAlloc(const Scope *scope)
2656     {
2657         // This test only applies to C++ source
2658         if (!_tokenizer->isCPP() || !_settings->inconclusive || !_settings->isEnabled("warning"))
2659             return;
2660     
2661         for (const Token *tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
2662             if (Token::Match(tok, "%name% (")) {
2663                 const Token *endParamToken = tok->next()->link();
2664                 const Token* pointerType = nullptr;
2665                 const Token* functionCalled = nullptr;
2666     
2667                 // Scan through the arguments to the function call
2668                 for (const Token *tok2 = tok->tokAt(2); tok2 && tok2 != endParamToken; tok2 = tok2->nextArgument()) {
2669                     const Function *func = tok2->function();
2670                     const bool isNothrow = func && (func->isAttributeNothrow() || func->isThrow());
2671     
2672                     if (Token::Match(tok2, "shared_ptr|unique_ptr <") && tok2->next()->link() && Token::Match(tok2->next()->link(), "> ( new %name%")) {
2673                         pointerType = tok2;
2674                     } else if (!isNothrow) {
2675                         if (Token::Match(tok2, "%name% ("))
2676                             functionCalled = tok2;
2677                         else if (tok2->isName() && tok2->next()->link() && Token::simpleMatch(tok2->next()->link(), "> ("))
2678                             functionCalled = tok2;
2679                     }
2680                 }
2681     
2682                 if (pointerType && functionCalled) {
2683                     std::string functionName = functionCalled->str();
2684                     if (functionCalled->strAt(1) == "<") {
2685                         functionName += '<';
2686                         for (const Token* tok2 = functionCalled->tokAt(2); tok2 != functionCalled->next()->link(); tok2 = tok2->next())
2687                             functionName += tok2->str();
2688                         functionName += '>';
2689                     }
2690                     std::string objectTypeName;
2691                     for (const Token* tok2 = pointerType->tokAt(2); tok2 != pointerType->next()->link(); tok2 = tok2->next())
2692                         objectTypeName += tok2->str();
2693     
2694                     unsafeArgAllocError(tok, functionName, pointerType->str(), objectTypeName);
2695                 }
2696             }
2697         }
2698     }
