19     //---------------------------------------------------------------------------
20     // You should use ++ and -- as prefix whenever possible as these are more
21     // efficient than postfix operators
22     //---------------------------------------------------------------------------
23     
24     #include "checkpostfixoperator.h"
25     #include "symboldatabase.h"
26     
27     //---------------------------------------------------------------------------
28     
29     
30     // Register this check class (by creating a static instance of it)
31     namespace {
32         CheckPostfixOperator instance;
33     }
34     
35     void CheckPostfixOperator::postfixOperator()
36     {
37         if (!_settings->isEnabled("performance"))
38             return;
39     
40         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
41     
42         const std::size_t functions = symbolDatabase->functionScopes.size();
43         for (std::size_t i = 0; i < functions; ++i) {
44             const Scope * scope = symbolDatabase->functionScopes[i];
45             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
46                 const Variable *var = tok->variable();
47                 if (!var || !Token::Match(tok, "%var% ++|--"))
48                     continue;
49     
50                 const Token* parent = tok->next()->astParent();
51                 if (!parent || parent->str() == ";" || (parent->str() == "," && (!parent->astParent() || parent->astParent()->str() != "("))) {
52                     if (var->isPointer() || var->isArray())
53                         continue;
54     
55                     if (Token::Match(var->nameToken()->previous(), "iterator|const_iterator|reverse_iterator|const_reverse_iterator")) {
56                         // the variable is an iterator
57                         postfixOperatorError(tok);
58                     } else if (var->type()) {
59                         // the variable is an instance of class
60                         postfixOperatorError(tok);
61                     }
62                 }
63             }
64         }
65     }
66     //---------------------------------------------------------------------------
67     
68     
69     void CheckPostfixOperator::postfixOperatorError(const Token *tok)
70     {
71         reportError(tok, Severity::performance, "postfixOperator",
72                     "Prefer prefix ++/-- operators for non-primitive types.\n"
73                     "Prefix ++/-- operators should be preferred for non-primitive types. "
74                     "Pre-increment/decrement can be more efficient than "
75                     "post-increment/decrement. Post-increment/decrement usually "
76                     "involves keeping a copy of the previous value around and "
77                     "adds a little extra code.");
78     }
