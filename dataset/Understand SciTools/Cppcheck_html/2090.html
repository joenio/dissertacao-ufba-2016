94     void SettingsDialog::AddIncludePath(const QString &path)
95     {
96         if (path.isNull() || path.isEmpty())
97             return;
98     
99         QListWidgetItem *item = new QListWidgetItem(path);
100         item->setFlags(item->flags() | Qt::ItemIsEditable);
101         mUI.mListIncludePaths->addItem(item);
102     }

104     void SettingsDialog::InitIncludepathsList()
105     {
106         QSettings settings;
107         const QString allPaths = settings.value(SETTINGS_GLOBAL_INCLUDE_PATHS).toString();
108         const QStringList paths = allPaths.split(";", QString::SkipEmptyParts);
109         foreach (QString path, paths) {
110             AddIncludePath(path);
111         }
112     }

114     void SettingsDialog::InitTranslationsList()
115     {
116         const QString current = mTranslator->GetCurrentLanguage();
117         QList<TranslationInfo> translations = mTranslator->GetTranslations();
118         foreach (TranslationInfo translation, translations) {
119             QListWidgetItem *item = new QListWidgetItem;
120             item->setText(translation.mName);
121             item->setData(LangCodeRole, QVariant(translation.mCode));
122             mUI.mListLanguages->addItem(item);
123             if (translation.mCode == current || translation.mCode == current.mid(0, 2))
124                 mUI.mListLanguages->setCurrentItem(item);
125         }
126     }

159     void SettingsDialog::SaveSettingValues() const
160     {
161         int jobs = mUI.mJobs->text().toInt();
162         if (jobs <= 0) {
163             jobs = 1;
164         }
165     
166         QSettings settings;
167         settings.setValue(SETTINGS_CHECK_THREADS, jobs);
168         SaveCheckboxValue(&settings, mUI.mForce, SETTINGS_CHECK_FORCE);
169         SaveCheckboxValue(&settings, mUI.mSaveAllErrors, SETTINGS_SAVE_ALL_ERRORS);
170         SaveCheckboxValue(&settings, mUI.mSaveFullPath, SETTINGS_SAVE_FULL_PATH);
171         SaveCheckboxValue(&settings, mUI.mShowFullPath, SETTINGS_SHOW_FULL_PATH);
172         SaveCheckboxValue(&settings, mUI.mShowNoErrorsMessage, SETTINGS_SHOW_NO_ERRORS);
173         SaveCheckboxValue(&settings, mUI.mShowDebugWarnings, SETTINGS_SHOW_DEBUG_WARNINGS);
174         SaveCheckboxValue(&settings, mUI.mInlineSuppressions, SETTINGS_INLINE_SUPPRESSIONS);
175         SaveCheckboxValue(&settings, mUI.mEnableInconclusive, SETTINGS_INCONCLUSIVE_ERRORS);
176         SaveCheckboxValue(&settings, mUI.mShowErrorId, SETTINGS_SHOW_ERROR_ID);
177     
178         const QListWidgetItem *currentLang = mUI.mListLanguages->currentItem();
179         if (currentLang) {
180             const QString langcode = currentLang->data(LangCodeRole).toString();
181             settings.setValue(SETTINGS_LANGUAGE, langcode);
182         }
183     
184         const int count = mUI.mListIncludePaths->count();
185         QString includePaths;
186         for (int i = 0; i < count; i++) {
187             QListWidgetItem *item = mUI.mListIncludePaths->item(i);
188             includePaths += item->text();
189             includePaths += ";";
190         }
191         settings.setValue(SETTINGS_GLOBAL_INCLUDE_PATHS, includePaths);
192     }

207             mUI.mListWidget->addItem(app.getName());

211     void SettingsDialog::RemoveApplication()
212     {
213         QList<QListWidgetItem *> selected = mUI.mListWidget->selectedItems();
214         foreach (QListWidgetItem *item, selected) {
215             const int removeIndex = mUI.mListWidget->row(item);
216             const int currentDefault = mTempApplications->GetDefaultApplication();
217             mTempApplications->RemoveApplication(removeIndex);
218             if (removeIndex == currentDefault)
219                 // If default app is removed set default to unknown
220                 mTempApplications->SetDefault(-1);
221             else if (removeIndex < currentDefault)
222                 // Move default app one up if earlier app was removed
223                 mTempApplications->SetDefault(currentDefault - 1);
224         }
225         mUI.mListWidget->clear();
226         PopulateApplicationList();
227     }

229     void SettingsDialog::EditApplication()
230     {
231         QList<QListWidgetItem *> selected = mUI.mListWidget->selectedItems();
232         QListWidgetItem *item = 0;
233         foreach (item, selected) {
234             int row = mUI.mListWidget->row(item);
235             Application& app = mTempApplications->GetApplication(row);
236             ApplicationDialog dialog(tr("Modify an application"), app, this);
237     
238             if (dialog.exec() == QDialog::Accepted) {
239                 QString name = app.getName();
240                 if (mTempApplications->GetDefaultApplication() == row)
241                     name += tr(" [Default]");
242                 item->setText(name);
243             }
244         }
245     }

247     void SettingsDialog::DefaultApplication()
248     {
249         QList<QListWidgetItem *> selected = mUI.mListWidget->selectedItems();
250         if (!selected.isEmpty()) {
251             int index = mUI.mListWidget->row(selected[0]);
252             mTempApplications->SetDefault(index);
253             mUI.mListWidget->clear();
254             PopulateApplicationList();
255         }
256     }
257     
258     void SettingsDialog::PopulateApplicationList()
259     {
260         const int defapp = mTempApplications->GetDefaultApplication();
261         for (int i = 0; i < mTempApplications->GetApplicationCount(); i++) {
262             const Application& app = mTempApplications->GetApplication(i);
263             QString name = app.getName();
264             if (i == defapp) {
265                 name += " ";
266                 name += tr("[Default]");
267             }
268             mUI.mListWidget->addItem(name);
269         }
270     
271         // Select default application, or if there is no default app then the
272         // first item.
273         if (defapp == -1)
274             mUI.mListWidget->setCurrentRow(0);
275         else {
276             if (mTempApplications->GetApplicationCount() > defapp)
277                 mUI.mListWidget->setCurrentRow(defapp);
278             else
279                 mUI.mListWidget->setCurrentRow(0);
280         }
281     }

319     void SettingsDialog::AddIncludePath()
320     {
321         QString selectedDir = QFileDialog::getExistingDirectory(this,
322                               tr("Select include directory"),
323                               GetPath(SETTINGS_LAST_INCLUDE_PATH));
324     
325         if (!selectedDir.isEmpty()) {
326             AddIncludePath(selectedDir);
327             SetPath(SETTINGS_LAST_INCLUDE_PATH, selectedDir);
