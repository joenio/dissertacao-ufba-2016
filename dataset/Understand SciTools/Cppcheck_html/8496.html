20     //---------------------------------------------------------------------------
21     #ifndef templatesimplifierH
22     #define templatesimplifierH
23     //---------------------------------------------------------------------------
24     
25     #include <set>
26     #include <list>
27     #include <string>
28     #include <vector>
29     #include "config.h"
30     
31     class Token;
32     class TokenList;
33     class ErrorLogger;
34     class Settings;
35     
36     
37     /// @addtogroup Core
38     /// @{
39     
40     /** @brief Simplify templates from the preprocessed and partially simplified code. */
41     class CPPCHECKLIB TemplateSimplifier {
42         TemplateSimplifier();
43         ~TemplateSimplifier();
44     public:
45     
46         /**
47          * Used after simplifyTemplates to perform a little cleanup.
48          * Sometimes the simplifyTemplates isn't fully successful and then
49          * there are function calls etc with "wrong" syntax.
50          */
51         static void cleanupAfterSimplify(Token *tokens);
52     
53         /**
54          * \param[in] tokens token list
55          * @return false if there are no syntax errors or true
56          */
57         static void checkComplicatedSyntaxErrorsInTemplates(const Token *tokens);
58     
59         /**
60          * is the token pointing at a template parameters block
61          * < int , 3 > => yes
62          * \param tok start token that must point at "<"
63          * \return number of parameters (invalid parameters => 0)
64          */
65         static unsigned int templateParameters(const Token *tok);
66     
67         /**
68          * Expand specialized templates : "template<>.."
69          * @return names of expanded templates
70          */
71         static std::set<std::string> expandSpecialized(Token *tokens);
72     
73         /**
74          * Get template declarations
75          * @return list of template declarations
76          */
77         static std::list<Token *> getTemplateDeclarations(Token *tokens, bool &codeWithTemplates);
78     
79         /**
80          * Get template instantiations
81          * @return list of template instantiations
82          */
83         static std::list<Token *> getTemplateInstantiations(Token *tokens);
84     
85         /**
86          * simplify template instantiations (use default argument values)
87          * @param templates list of template declarations
88          * @param templateInstantiations list of template instantiations
89          */
90         static void useDefaultArgumentValues(const std::list<Token *> &templates,
91                                              std::list<Token *> *templateInstantiations);
92     
93         /**
94          * Match template declaration/instantiation
95          * @param instance template instantiation
96          * @param name name of template
97          * @param numberOfArguments number of template arguments
98          * @param patternAfter pattern that must match the tokens after the ">"
99          * @return match => true
100          */
101         static bool instantiateMatch(const Token *instance, const std::string &name, std::size_t numberOfArguments, const char patternAfter[]);
102     
103         /**
104          * Match template declaration/instantiation
105          * @param tok The ">" token e.g. before "class"
106          * @return -1 to bail out or positive integer to identity the position
107          * of the template name.
108          */
109         static int getTemplateNamePosition(const Token *tok);
110     
111         static void expandTemplate(
112             TokenList& tokenlist,
113             const Token *tok,
114             const std::string &name,
115             std::vector<const Token *> &typeParametersInDeclaration,
116             const std::string &newName,
117             std::vector<const Token *> &typesUsedInTemplateInstantiation,
118             std::list<Token *> &templateInstantiations);
119     
120         /**
121          * @brief TemplateParametersInDeclaration
122          * @param tok  template < typename T, typename S >
123          *                        ^ tok
124          * @param typeParametersInDeclaration  template < typename T, typename S >
125          *                                                         ^ [0]       ^ [1]
126          * @return  template < typename T, typename S >
127          *                                              ^ return
128          */
129         static const Token * TemplateParametersInDeclaration(
130             const Token * tok,
131             std::vector<const Token *> & typeParametersInDeclaration);
132     
133         /**
134          * Simplify templates : expand all instantiations for a template
135          * @todo It seems that inner templates should be instantiated recursively
136          * @param tokenlist token list
137          * @param errorlogger error logger
138          * @param _settings settings
139          * @param tok token where the template declaration begins
140          * @param templateInstantiations a list of template usages (not necessarily just for this template)
141          * @param expandedtemplates all templates that has been expanded so far. The full names are stored.
142          * @return true if the template was instantiated
143          */
144         static bool simplifyTemplateInstantiations(
145             TokenList& tokenlist,
146             ErrorLogger* errorlogger,
147             const Settings *_settings,
148             const Token *tok,
149             std::list<Token *> &templateInstantiations,
150             std::set<std::string> &expandedtemplates);
151     
152         /**
153          * Simplify templates
154          * @param tokenlist token list
155          * @param errorlogger error logger
156          * @param _settings settings
157          * @param _codeWithTemplates output parameter that is set if code contains templates
158          */
159         static void simplifyTemplates(
160             TokenList& tokenlist,
161             ErrorLogger* errorlogger,
162             const Settings *_settings,
163             bool &_codeWithTemplates);
164     
165         /**
166          * Simplify constant calculations such as "1+2" => "3"
167          * @param tok start token
168          * @return true if modifications to token-list are done.
169          *         false if no modifications are done.
170          */
171         static bool simplifyNumericCalculations(Token *tok);
172     
173         /**
174          * Simplify constant calculations such as "1+2" => "3".
175          * This also performs simple cleanup of parentheses etc.
176          * @param _tokens start token
177          * @return true if modifications to token-list are done.
178          *         false if no modifications are done.
179          */
180         static bool simplifyCalculations(Token *_tokens);
181     
182     private:
183     
184         /**
185          * Remove a specific "template < ..." template class/function
186          */
187         static bool removeTemplate(Token *tok);
188     
189         /** Syntax error */
190         static void syntaxError(const Token *tok);
191     
192     };
193     
194     /// @}
195     //---------------------------------------------------------------------------
196     #endif // templatesimplifierH
