73     enum OpenMode { CLOSED, READ_MODE, WRITE_MODE, RW_MODE, UNKNOWN_OM };
74     static OpenMode getMode(const std::string& str)
75     {
76         if (str.find('+', 1) != std::string::npos)
77             return RW_MODE;
78         else if (str.find('w') != std::string::npos || str.find('a') != std::string::npos)
79             return WRITE_MODE;
80         else if (str.find('r') != std::string::npos)
81             return READ_MODE;
82         return UNKNOWN_OM;
83     }

88         enum Operation {NONE, UNIMPORTANT, READ, WRITE, POSITIONING, OPEN, CLOSE, UNKNOWN_OP} lastOperation;

91         AppendMode append_mode;
92         explicit Filepointer(OpenMode mode_ = UNKNOWN_OM)
93             : mode(mode_), mode_indent(0), lastOperation(NONE), op_indent(0), append_mode(UNKNOWN_AM) {

102     void CheckIO::checkFileUsage()
103     {
104         const bool windows = _settings->isWindowsPlatform();
105         const bool printPortability = _settings->isEnabled("portability");
106         const bool printWarnings = _settings->isEnabled("warning");
107     
108         std::map<unsigned int, Filepointer> filepointers;
109     
110         const SymbolDatabase* symbolDatabase = _tokenizer->getSymbolDatabase();
111         std::size_t varListSize = symbolDatabase->getVariableListSize();
112         for (std::size_t i = 1; i < varListSize; ++i) {
113             const Variable* var = symbolDatabase->getVariableFromVarId(i);
114             if (!var || !var->declarationId() || var->isArray() || !Token::simpleMatch(var->typeStartToken(), "FILE *"))
115                 continue;
116     
117             if (var->isLocal()) {
118                 if (var->nameToken()->strAt(1) == "(") // initialize by calling "ctor"
119                     filepointers.insert(std::make_pair(var->declarationId(), Filepointer(UNKNOWN_OM)));
120                 else
121                     filepointers.insert(std::make_pair(var->declarationId(), Filepointer(CLOSED)));
122             } else {
123                 filepointers.insert(std::make_pair(var->declarationId(), Filepointer(UNKNOWN_OM)));
124                 // TODO: If all fopen calls we find open the file in the same type, we can set Filepointer::mode
125             }
126         }
127     
128         std::size_t functions = symbolDatabase->functionScopes.size();
129         for (std::size_t j = 0; j < functions; ++j) {
130             const Scope * scope = symbolDatabase->functionScopes[j];
131             unsigned int indent = 0;
132             for (const Token *tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
133                 if (tok->str() == "{")
134                     indent++;
135                 else if (tok->str() == "}") {
136                     indent--;
137                     for (std::map<unsigned int, Filepointer>::iterator i = filepointers.begin(); i != filepointers.end(); ++i) {
138                         if (indent < i->second.mode_indent) {
139                             i->second.mode_indent = 0;
140                             i->second.mode = UNKNOWN_OM;
141                         }
142                         if (indent < i->second.op_indent) {
143                             i->second.op_indent = 0;
144                             i->second.lastOperation = Filepointer::UNKNOWN_OP;
145                         }
146                     }
147                 } else if (tok->str() == "return" || tok->str() == "continue" || tok->str() == "break") { // Reset upon return, continue or break
148                     for (std::map<unsigned int, Filepointer>::iterator i = filepointers.begin(); i != filepointers.end(); ++i) {
149                         i->second.mode_indent = 0;
150                         i->second.mode = UNKNOWN_OM;
151                         i->second.op_indent = 0;
152                         i->second.lastOperation = Filepointer::UNKNOWN_OP;
153                     }
154                 } else if (Token::Match(tok, "%var% =") &&
155                            (tok->strAt(2) != "fopen" && tok->strAt(2) != "freopen" && tok->strAt(2) != "tmpfile" &&
156                             (windows ? (tok->str() != "_wfopen" && tok->str() != "_wfreopen") : true))) {
157                     std::map<unsigned int, Filepointer>::iterator i = filepointers.find(tok->varId());
158                     if (i != filepointers.end()) {
159                         i->second.mode = UNKNOWN_OM;
160                         i->second.lastOperation = Filepointer::UNKNOWN_OP;
161                     }
162                 } else if (Token::Match(tok, "%name% (") && tok->previous() && (!tok->previous()->isName() || Token::Match(tok->previous(), "return|throw"))) {
163                     std::string mode;
164                     const Token* fileTok = 0;
165                     Filepointer::Operation operation = Filepointer::NONE;
166     
167                     if ((tok->str() == "fopen" || tok->str() == "freopen" || tok->str() == "tmpfile" ||
168                          (windows && (tok->str() == "_wfopen" || tok->str() == "_wfreopen"))) &&
169                         tok->strAt(-1) == "=") {
170                         if (tok->str() != "tmpfile") {
171                             const Token* modeTok = tok->tokAt(2)->nextArgument();
172                             if (modeTok && modeTok->tokType() == Token::eString)
173                                 mode = modeTok->strValue();
174                         } else
175                             mode = "wb+";
176                         fileTok = tok->tokAt(-2);
177                         operation = Filepointer::OPEN;
178                     } else if (windows && Token::Match(tok, "fopen_s|freopen_s|_wfopen_s|_wfreopen_s ( & %name%")) {
179                         const Token* modeTok = tok->tokAt(2)->nextArgument()->nextArgument();
180                         if (modeTok && modeTok->tokType() == Token::eString)
181                             mode = modeTok->strValue();
182                         fileTok = tok->tokAt(3);
183                         operation = Filepointer::OPEN;
184                     } else if ((tok->str() == "rewind" || tok->str() == "fseek" || tok->str() == "fsetpos" || tok->str() == "fflush") ||
185                                (windows && tok->str() == "_fseeki64")) {
186                         if (printPortability && tok->str() == "fflush") {
187                             fileTok = tok->tokAt(2);
188                             if (fileTok) {
189                                 if (fileTok->str() == "stdin")
190                                     fflushOnInputStreamError(tok, fileTok->str());
191                                 else {
192                                     Filepointer& f = filepointers[fileTok->varId()];
193                                     if (f.mode == READ_MODE)
194                                         fflushOnInputStreamError(tok, fileTok->str());
195                                 }
196                             }
197                         }
198     
199                         fileTok = tok->tokAt(2);
200                         operation = Filepointer::POSITIONING;
201                     } else if (tok->str() == "fgetc" || tok->str() == "fgetwc" ||
202                                tok->str() == "fgets" || tok->str() == "fgetws" || tok->str() == "fread" ||
203                                tok->str() == "fscanf" || tok->str() == "fwscanf" || tok->str() == "getc" ||
204                                (windows && (tok->str() == "fscanf_s" || tok->str() == "fwscanf_s"))) {
205                         if (tok->str().find("scanf") != std::string::npos)
206                             fileTok = tok->tokAt(2);
207                         else
208                             fileTok = tok->linkAt(1)->previous();
209                         operation = Filepointer::READ;
210                     } else if (tok->str() == "fputc" || tok->str() == "fputwc" ||
211                                tok->str() == "fputs" || tok->str() == "fputws" || tok->str() == "fwrite" ||
212                                tok->str() == "fprintf" || tok->str() == "fwprintf" || tok->str() == "putcc" ||
213                                (windows && (tok->str() == "fprintf_s" || tok->str() == "fwprintf_s"))) {
214                         if (tok->str().find("printf") != std::string::npos)
215                             fileTok = tok->tokAt(2);
216                         else
217                             fileTok = tok->linkAt(1)->previous();
218                         operation = Filepointer::WRITE;
219                     } else if (tok->str() == "fclose") {
220                         fileTok = tok->tokAt(2);
221                         operation = Filepointer::CLOSE;
222                     } else if (whitelist.find(tok->str()) != whitelist.end()) {
223                         fileTok = tok->tokAt(2);
224                         if ((tok->str() == "ungetc" || tok->str() == "ungetwc") && fileTok)
225                             fileTok = fileTok->nextArgument();
226                         operation = Filepointer::UNIMPORTANT;
227                     } else if (!Token::Match(tok, "if|for|while|catch|switch") && _settings->library.functionpure.find(tok->str()) == _settings->library.functionpure.end()) {
228                         const Token* const end2 = tok->linkAt(1);
229                         if (scope->functionOf && scope->functionOf->isClassOrStruct() && !scope->function->isStatic() && ((tok->strAt(-1) != "::" && tok->strAt(-1) != ".") || tok->strAt(-2) == "this")) {
230                             if (!tok->function() || (tok->function()->nestedIn && tok->function()->nestedIn->isClassOrStruct())) {
231                                 for (std::map<unsigned int, Filepointer>::iterator i = filepointers.begin(); i != filepointers.end(); ++i) {
232                                     const Variable* var = symbolDatabase->getVariableFromVarId(i->first);
233                                     if (!var || !(var->isLocal() || var->isGlobal() || var->isStatic())) {
234                                         i->second.mode = UNKNOWN_OM;
235                                         i->second.mode_indent = 0;
236                                         i->second.op_indent = indent;
237                                         i->second.lastOperation = Filepointer::UNKNOWN_OP;
238                                     }
239                                 }
240                                 continue;
241                             }
242                         }
243                         for (const Token* tok2 = tok->tokAt(2); tok2 != end2; tok2 = tok2->next()) {
244                             if (tok2->varId() && filepointers.find(tok2->varId()) != filepointers.end()) {
245                                 fileTok = tok2;
246                                 operation = Filepointer::UNKNOWN_OP; // Assume that repositioning was last operation and that the file is opened now
247                                 break;
248                             }
249                         }
250                     }
251     
252                     while (Token::Match(fileTok, "%name% ."))
253                         fileTok = fileTok->tokAt(2);
254     
255                     if (!fileTok || !fileTok->varId() || fileTok->strAt(1) == "[")
256                         continue;
257     
258                     if (filepointers.find(fileTok->varId()) == filepointers.end()) { // function call indicates: Its a File
259                         filepointers.insert(std::make_pair(fileTok->varId(), Filepointer(UNKNOWN_OM)));
260                     }
261                     Filepointer& f = filepointers[fileTok->varId()];
262     
263                     switch (operation) {
264                     case Filepointer::OPEN:
265                         f.mode = getMode(mode);
266                         if (mode.find('a') != std::string::npos) {
267                             if (f.mode == RW_MODE)
268                                 f.append_mode = Filepointer::APPEND_EX;
269                             else
270                                 f.append_mode = Filepointer::APPEND;
271                         } else
272                             f.append_mode = Filepointer::UNKNOWN_AM;
273                         f.mode_indent = indent;
274                         break;
275                     case Filepointer::POSITIONING:
276                         if (f.mode == CLOSED)
277                             useClosedFileError(tok);
278                         else if (f.append_mode == Filepointer::APPEND && tok->str() != "fflush" && printWarnings)
279                             seekOnAppendedFileError(tok);
280                         break;
281                     case Filepointer::READ:
282                         if (f.mode == CLOSED)
283                             useClosedFileError(tok);
284                         else if (f.mode == WRITE_MODE)
285                             readWriteOnlyFileError(tok);
286                         else if (f.lastOperation == Filepointer::WRITE)
287                             ioWithoutPositioningError(tok);
288                         break;
289                     case Filepointer::WRITE:
290                         if (f.mode == CLOSED)
291                             useClosedFileError(tok);
292                         else if (f.mode == READ_MODE)
293                             writeReadOnlyFileError(tok);
294                         else if (f.lastOperation == Filepointer::READ)
295                             ioWithoutPositioningError(tok);
296                         break;
297                     case Filepointer::CLOSE:
298                         if (f.mode == CLOSED)
299                             useClosedFileError(tok);
300                         else
301                             f.mode = CLOSED;
302                         f.mode_indent = indent;
303                         break;
304                     case Filepointer::UNIMPORTANT:
305                         if (f.mode == CLOSED)
306                             useClosedFileError(tok);
307                         break;
308                     case Filepointer::UNKNOWN_OP:
309                         f.mode = UNKNOWN_OM;
310                         f.mode_indent = 0;
311                         break;
312                     default:
313                         break;
314                     }
315                     if (operation != Filepointer::NONE && operation != Filepointer::UNIMPORTANT) {
316                         f.op_indent = indent;
317                         f.lastOperation = operation;
318                     }
319                 }
320             }
321             for (std::map<unsigned int, Filepointer>::iterator i = filepointers.begin(); i != filepointers.end(); ++i) {
322                 i->second.op_indent = 0;
323                 i->second.mode = UNKNOWN_OM;
324                 i->second.lastOperation = Filepointer::UNKNOWN_OP;
325             }
326         }
327     }

368     // scanf without field width limits can crash with huge input data
369     //---------------------------------------------------------------------------
370     void CheckIO::invalidScanf()
371     {
372         if (!_settings->isEnabled("warning"))
373             return;
374     
375         const SymbolDatabase * const symbolDatabase = _tokenizer->getSymbolDatabase();
376         std::size_t functions = symbolDatabase->functionScopes.size();
377         for (std::size_t j = 0; j < functions; ++j) {
378             const Scope * scope = symbolDatabase->functionScopes[j];
379             for (const Token *tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
380                 const Token *formatToken = nullptr;
381                 if (Token::Match(tok, "scanf|vscanf ( %str% ,"))
382                     formatToken = tok->tokAt(2);
383                 else if (Token::Match(tok, "sscanf|vsscanf|fscanf|vfscanf (")) {
384                     const Token* nextArg = tok->tokAt(2)->nextArgument();
385                     if (nextArg && nextArg->tokType() == Token::eString)
386                         formatToken = nextArg;
387                     else
388                         continue;
389                 } else
390                     continue;
391     
392                 bool format = false;
393     
394                 // scan the string backwards, so we do not need to keep states
395                 const std::string &formatstr(formatToken->str());
396                 for (std::size_t i = 1; i < formatstr.length(); i++) {
397                     if (formatstr[i] == '%')
398                         format = !format;
399     
400                     else if (!format)
401                         continue;
402     
403                     else if (std::isdigit(formatstr[i]) || formatstr[i] == '*') {
404                         format = false;
405                     }
406     
407                     else if (std::isalpha((unsigned char)formatstr[i]) || formatstr[i] == '[') {
408                         if (formatstr[i] == 's' || formatstr[i] == '[' || formatstr[i] == 'S' || (formatstr[i] == 'l' && formatstr[i+1] == 's'))  // #3490 - field width limits are only necessary for string input
409                             invalidScanfError(tok);
410                         format = false;
411                     }
412                 }
413             }
414         }
415     }

450     static bool findFormat(unsigned int arg, const Token *firstArg,
451                            const Token **formatStringTok, const Token **formatArgTok)
452     {
453         const Token* argTok = firstArg;
454     
455         for (unsigned int i = 0; i < arg && argTok; ++i)
456             argTok = argTok->nextArgument();
457     
458         if (Token::Match(argTok, "%str% [,)]")) {
459             *formatArgTok = argTok->nextArgument();
460             *formatStringTok = argTok;
461             return true;
462         } else if (Token::Match(argTok, "%var% [,)]") &&
463                    argTok->variable() &&
464                    Token::Match(argTok->variable()->typeStartToken(), "char|wchar_t") &&
465                    (argTok->variable()->isPointer() ||
466                     (argTok->variable()->dimensions().size() == 1 &&
467                      argTok->variable()->dimensionKnown(0) &&
468                      argTok->variable()->dimension(0) != 0))) {
469             *formatArgTok = argTok->nextArgument();
470             if (argTok->values.size() >= 1 && argTok->values.front().tokvalue && argTok->values.front().tokvalue->tokType() == Token::eString)
471                 *formatStringTok = argTok->values.front().tokvalue;
472             return true;
473         }
474         return false;
475     }

483     void CheckIO::checkWrongPrintfScanfArguments()
484     {
485         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
486         const bool isWindows = _settings->isWindowsPlatform();
487     
488         std::size_t functions = symbolDatabase->functionScopes.size();
489         for (std::size_t j = 0; j < functions; ++j) {
490             const Scope * scope = symbolDatabase->functionScopes[j];
491             for (const Token *tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
492                 if (!tok->isName()) continue;
493     
494                 const Token* argListTok = 0; // Points to first va_list argument
495                 const Token* formatStringTok = 0; // Points to format string token
496     
497                 bool scan = false;
498                 bool scanf_s = false;
499                 int formatStringArgNo = -1;
500     
501                 if (tok->strAt(1) == "(" && _settings->library.formatstr_function(tok->str())) {
502                     const std::map<int, Library::ArgumentChecks>& argumentChecks = _settings->library.argumentChecks.at(tok->str());
503                     for (std::map<int, Library::ArgumentChecks>::const_iterator i = argumentChecks.cbegin(); i != argumentChecks.cend(); ++i) {
504                         if (i->second.formatstr) {
505                             formatStringArgNo = i->first - 1;
506                             break;
507                         }
508                     }
509     
510                     scan = _settings->library.formatstr_scan(tok->str());
511                     scanf_s = _settings->library.formatstr_secure(tok->str());
512                 }
513     
514                 if (formatStringArgNo >= 0) {
515                     // formatstring found in library. Find format string and first argument belonging to format string.
516                     if (!findFormat(static_cast<unsigned int>(formatStringArgNo), tok->tokAt(2), &formatStringTok, &argListTok))
517                         continue;
518                 } else if (isWindows && Token::Match(tok, "Format|AppendFormat (") &&
519                            Token::Match(tok->tokAt(-2), "%var% .") && tok->tokAt(-2)->variable() &&
520                            tok->tokAt(-2)->variable()->typeStartToken()->str() == "CString") {
521                     // Find second parameter and format string
522                     if (!findFormat(0, tok->tokAt(2), &formatStringTok, &argListTok))
523                         continue;
524                 } else if (Token::simpleMatch(tok, "swprintf (") && Token::Match(tok->tokAt(2)->nextArgument(), "%str%")) {
525                     // Find third parameter and format string
526                     if (!findFormat(1, tok->tokAt(2), &formatStringTok, &argListTok))
527                         continue;
528                 } else if (Token::simpleMatch(tok, "swprintf (") && !Token::Match(tok->tokAt(2)->nextArgument(), "%str%")) {
529                     // Find forth parameter and format string
530                     if (!findFormat(2, tok->tokAt(2), &formatStringTok, &argListTok))
531                         continue;
532                 } else if (isWindows && Token::Match(tok, "sprintf_s|swprintf_s (")) {
533                     // template <size_t size> int sprintf_s(char (&buffer)[size], const char *format, ...);
534                     if (findFormat(1, tok->tokAt(2), &formatStringTok, &argListTok)) {
535                         if (!formatStringTok)
536                             continue;
537                     }

542                     }
543                 } else if (isWindows && Token::Match(tok, "_snprintf_s|_snwprintf_s (")) {

548                     }

553                     }
554                 } else {
555                     continue;
556                 }
557     
558                 if (!formatStringTok)
559                     continue;
560     
561                 checkFormatString(tok, formatStringTok, argListTok, scan, scanf_s);
562             }
563         }

1506                     }
1507                     typeToken = tempToken;
1508                     return;
1509                 }
1510     
1511                 // check for std::vector::at() and std::string::at()
1512                 else if (Token::Match(tok1->previous(), "%var% . at (") &&
1513                          Token::Match(tok1->linkAt(2), ") [,)]")) {
1514                     varTok = tok1->previous();
1515                     variableInfo = varTok->variable();
1516     
1517                     if (!variableInfo || !isStdVectorOrString()) {
1518                         variableInfo = 0;
1519                         typeToken = 0;
1520                     }
1521     
1522                     return;
1523                 } else if (!(tok1->str() == "." || tok1->tokType() == Token::eVariable || tok1->tokType() == Token::eFunction))
1524                     return;
1525             }
1526     
1527             if (varTok) {
1528                 variableInfo = varTok->variable();
1529                 element = tok1->previous()->str() == "]";
1530     
1531                 // look for std::vector operator [] and use template type as return type
1532                 if (variableInfo) {
1533                     if (element && isStdVectorOrString()) { // isStdVectorOrString sets type token if true
1534                         element = false;    // not really an array element
1535                     } else
1536                         typeToken = variableInfo->typeStartToken();
1537                 }
1538     
1539                 return;
1540             }
1541         }

1560     {
1561         if (!isCPP)
1562             return false;
1563         if (variableInfo->isStlType(stl_vector)) {
1564             typeToken = variableInfo->typeStartToken()->tokAt(4);
1565             _template = true;
1566             return true;
1567         } else if (variableInfo->isStlType(stl_string)) {
1568             tempToken = new Token(0);
1569             tempToken->fileIndex(variableInfo->typeStartToken()->fileIndex());
1570             tempToken->linenr(variableInfo->typeStartToken()->linenr());
1571             if (variableInfo->typeStartToken()->strAt(2) == "string")
1572                 tempToken->str("char");
1573             else
1574                 tempToken->str("wchar_t");
1575             typeToken = tempToken;
1576             return true;
1577         } else if (variableInfo->type() && !variableInfo->type()->derivedFrom.empty()) {
1578             const std::vector<Type::BaseInfo>& derivedFrom = variableInfo->type()->derivedFrom;
1579             for (std::size_t i = 0, size = derivedFrom.size(); i < size; ++i) {
1580                 const Token* nameTok = derivedFrom[i].nameTok;
1581                 if (Token::Match(nameTok, "std :: vector|array <")) {
1582                     typeToken = nameTok->tokAt(4);
1583                     _template = true;
1584                     return true;
1585                 } else if (Token::Match(nameTok, "std :: string|wstring")) {
1586                     tempToken = new Token(0);
1587                     tempToken->fileIndex(variableInfo->typeStartToken()->fileIndex());
1588                     tempToken->linenr(variableInfo->typeStartToken()->linenr());
1589                     if (nameTok->strAt(2) == "string")
1590                         tempToken->str("char");
1591                     else
1592                         tempToken->str("wchar_t");
1593                     typeToken = tempToken;
1594                     return true;
1595                 }
1596             }
1597         } else if (variableInfo->type()) {
1598             const Scope * classScope = variableInfo->type()->classScope;
1599             if (classScope) {
1600                 std::list<Function>::const_iterator functions;
1601                 for (functions = classScope->functionList.begin();
1602                      functions != classScope->functionList.end(); ++functions) {
1603                     if (functions->name() == "operator[]") {
1604                         if (Token::Match(functions->retDef, "%type% &")) {
1605                             typeToken = functions->retDef;
1606                             return true;
1607                         }
1608                     }
1609                 }
1610             }
1611         }
1612     
1613         return false;
1614     }

1927     void CheckIO::argumentType(std::ostream& os, const ArgumentInfo * argInfo)
1928     {
1929         if (argInfo) {
1930             os << "\'";
1931             const Token *type = argInfo->typeToken;
1932             if (type->tokType() == Token::eString) {
1933                 if (type->isLong())
1934                     os << "const wchar_t *";
1935                 else
1936                     os << "const char *";
1937             } else {
1938                 if (type->originalName().empty()) {
1939                     if (type->strAt(-1) == "const")
1940                         os << "const ";
1941                     while (Token::Match(type, "const|struct")) {
1942                         os << type->str() << " ";
1943                         type = type->next();
1944                     }
1945                     while (Token::Match(type, "%any% ::")) {
1946                         os << type->str() << "::";
1947                         type = type->tokAt(2);
1948                     }
1949                     type->stringify(os, false, true, false);
1950                     if (type->strAt(1) == "*" && !argInfo->element)
1951                         os << " *";
1952                     else if (argInfo->variableInfo && !argInfo->element && argInfo->variableInfo->isArray())
1953                         os << " *";
1954                     else if (type->strAt(1) == "*" && argInfo->variableInfo && argInfo->element && argInfo->variableInfo->isArray())
1955                         os << " *";
1956                     if (argInfo->address)
1957                         os << " *";
1958                 } else {
1959                     if ((type->originalName() == "__int64" || type->originalName() == "__int32") && type->isUnsigned())
1960                         os << "unsigned ";
1961                     os << type->originalName();
1962                     if (type->strAt(1) == "*" || argInfo->address)
1963                         os << " *";
1964                     os << " {aka ";
1965                     type->stringify(os, false, true, false);
1966                     if (type->strAt(1) == "*" || argInfo->address)
1967                         os << " *";
1968                     os << "}";
1969                 }
1970             }
1971             os << "\'";
1972         } else
1973             os << "Unknown";
1974     }
