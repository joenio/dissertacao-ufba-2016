23     #include <string>
24     #include <map>
25     #include <ctime>
26     #include "config.h"
27     
28     enum SHOWTIME_MODES {
29         SHOWTIME_NONE = 0,
30         SHOWTIME_FILE,
31         SHOWTIME_SUMMARY,
32         SHOWTIME_TOP5
33     };
34     
35     class CPPCHECKLIB TimerResultsIntf {
36     public:
37         virtual ~TimerResultsIntf() { }
38     
39         virtual void AddResults(const std::string& str, std::clock_t clocks) = 0;
40     };
41     
42     struct TimerResultsData {
43         std::clock_t _clocks;
44         long _numberOfResults;
45     
46         TimerResultsData()
47             : _clocks(0)
48             , _numberOfResults(0) {
49         }
50     
51         double seconds() const {
52             double ret = (double)((unsigned long)_clocks) / (double)CLOCKS_PER_SEC;
53             return ret;
54         }
55     };
56     
57     class CPPCHECKLIB TimerResults : public TimerResultsIntf {
58     public:
59         TimerResults() {
60         }
61     
62         void ShowResults(SHOWTIME_MODES mode) const;
63         virtual void AddResults(const std::string& str, std::clock_t clocks);
64     
65     private:
66         std::map<std::string, struct TimerResultsData> _results;
67     };
68     
69     class CPPCHECKLIB Timer {
70     public:
71         Timer(const std::string& str, unsigned int showtimeMode, TimerResultsIntf* timerResults = NULL);
72         ~Timer();
73         void Stop();
74     
75     private:
76         Timer& operator=(const Timer&); // disallow assignments
77     
78         const std::string _str;
79         TimerResultsIntf* _timerResults;
80         std::clock_t _start;
81         const unsigned int _showtimeMode;
82         bool _stopped;
83     };
84     //---------------------------------------------------------------------------
85     #endif // timerH
