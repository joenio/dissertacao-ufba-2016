79     bool CppCheckExecutor::parseFromArgs(CppCheck *cppcheck, int argc, const char* const argv[])
80     {
81         Settings& settings = cppcheck->settings();
82         CmdLineParser parser(&settings);
83         const bool success = parser.ParseFromArgs(argc, argv);
84     
85         if (success) {
86             if (parser.GetShowVersion() && !parser.GetShowErrorMessages()) {
87                 const char * const extraVersion = cppcheck->extraVersion();
88                 if (*extraVersion != 0)
89                     std::cout << "Cppcheck " << cppcheck->version() << " ("
90                               << extraVersion << ')' << std::endl;
91                 else
92                     std::cout << "Cppcheck " << cppcheck->version() << std::endl;
93             }
94     
95             if (parser.GetShowErrorMessages()) {
96                 errorlist = true;
97                 std::cout << ErrorLogger::ErrorMessage::getXMLHeader(settings.xml_version);
98                 cppcheck->getErrorMessages();
99                 std::cout << ErrorLogger::ErrorMessage::getXMLFooter(settings.xml_version) << std::endl;
100             }
101     
102             if (parser.ExitAfterPrinting()) {
103                 settings.terminate();
104                 return true;
105             }
106         } else {
107             return false;
108         }
109     
110         // Check that all include paths exist
111         {
112             std::list<std::string>::iterator iter;
113             for (iter = settings.includePaths.begin();
114                  iter != settings.includePaths.end();
115                 ) {
116                 const std::string path(Path::toNativeSeparators(*iter));
117                 if (FileLister::isDirectory(path))
118                     ++iter;
119                 else {
120                     // If the include path is not found, warn user and remove the non-existing path from the list.
121                     if (settings.isEnabled("information"))
122                         std::cerr << "(information) Couldn't find path given by -I '" << path << '\'' << std::endl;
123                     iter = settings.includePaths.erase(iter);
124                 }
125             }
126         }
127     
128         // Output a warning for the user if he tries to exclude headers
129         bool warn = false;
130         const std::vector<std::string>& ignored = parser.GetIgnoredPaths();
131         for (std::vector<std::string>::const_iterator i = ignored.cbegin(); i != ignored.cend(); ++i) {
132             if (Path::isHeader(*i)) {
133                 warn = true;
134                 break;
135             }
136         }
137         if (warn) {
138             std::cout << "cppcheck: filename exclusion does not apply to header (.h and .hpp) files." << std::endl;
139             std::cout << "cppcheck: Please use --suppress for ignoring results from the header files." << std::endl;
140         }
141     
142         const std::vector<std::string>& pathnames = parser.GetPathNames();
143     
144     #if defined(_WIN32)
145         // For Windows we want case-insensitive path matching
146         const bool caseSensitive = false;
147     #else
148         const bool caseSensitive = true;
149     #endif
150         if (!pathnames.empty()) {
151             // Execute recursiveAddFiles() to each given file parameter
152             PathMatch matcher(ignored, caseSensitive);
153             for (std::vector<std::string>::const_iterator iter = pathnames.begin(); iter != pathnames.end(); ++iter)
154                 FileLister::recursiveAddFiles(_files, Path::toNativeSeparators(*iter), _settings->library.markupExtensions(), matcher);
155         }
156     
157         if (_files.empty()) {
158             std::cout << "cppcheck: error: could not find or open any of the paths given." << std::endl;
159             if (!ignored.empty())
160                 std::cout << "cppcheck: Maybe all paths were ignored?" << std::endl;
161             return false;
162         }
163         return true;
164     }

166     int CppCheckExecutor::check(int argc, const char* const argv[])
167     {
168         Preprocessor::missingIncludeFlag = false;
169         Preprocessor::missingSystemIncludeFlag = false;
170     
171         CppCheck cppCheck(*this, true);
172     
173         const Settings& settings = cppCheck.settings();
174         _settings = &settings;
175     
176         if (!parseFromArgs(&cppCheck, argc, argv)) {
177             return EXIT_FAILURE;
178         }
179         if (settings.terminated()) {
180             return EXIT_SUCCESS;
181         }
182     
183         if (cppCheck.settings().exceptionHandling) {
184             return check_wrapper(cppCheck, argc, argv);
185         } else {
186             return check_internal(cppCheck, argc, argv);
187         }
188     }

203         /*
204          * Try to print the callstack.
205          * That is very sensitive to the operating system, hardware, compiler and runtime!
206          * The code is not meant for production environment, it's using functions not whitelisted for usage in a signal handler function.
207          */
208         void print_stacktrace(FILE* output, bool demangling, int maxdepth, bool lowMem)
209         {
210     #if defined(USE_UNIX_BACKTRACE_SUPPORT)
211     // 32 vs. 64bit
212     #define ADDRESSDISPLAYLENGTH ((sizeof(long)==8)?12:8)
213             const int fd = fileno(output);
214             void *array[32]= {0}; // the less resources the better...
215             const int currentdepth = backtrace(array, (int)GetArrayLength(array));
216             const int offset=2; // some entries on top are within our own exception handling code or libc
217             if (maxdepth<0)
218                 maxdepth=currentdepth-offset;
219             else
220                 maxdepth = std::min(maxdepth, currentdepth);
221             if (lowMem) {
222                 fputs("Callstack (symbols only):\n", output);
223                 fflush(output);
224                 backtrace_symbols_fd(array+offset, maxdepth, fd);
225                 fflush(output);
226             } else {
227                 char **symbolstrings = backtrace_symbols(array, currentdepth);
228                 if (symbolstrings) {
229                     fputs("Callstack:\n", output);
230                     for (int i = offset; i < maxdepth; ++i) {
231                         const char * const symbol = symbolstrings[i];
232                         char * realname = nullptr;
233                         const char * const firstBracketName     = strchr(symbol, '(');
234                         const char * const firstBracketAddress  = strchr(symbol, '[');
235                         const char * const secondBracketAddress = strchr(firstBracketAddress, ']');
236                         const char * const beginAddress         = firstBracketAddress+3;
237                         const int addressLen = int(secondBracketAddress-beginAddress);
238                         const int padLen     = int(ADDRESSDISPLAYLENGTH-addressLen);
239                         if (demangling && firstBracketName) {
240                             const char * const plus = strchr(firstBracketName, '+');
241                             if (plus && (plus>(firstBracketName+1))) {
242                                 char input_buffer[512]= {0};
243                                 strncpy(input_buffer, firstBracketName+1, plus-firstBracketName-1);
244                                 char output_buffer[1024]= {0};
245                                 size_t length = GetArrayLength(output_buffer);
246                                 int status=0;
247                                 realname = abi::__cxa_demangle(input_buffer, output_buffer, &length, &status); // non-NULL on success
248                             }
249                         }
250                         const int ordinal=i-offset;
251                         fprintf(output, "#%-2d 0x",
252                                 ordinal);
253                         if (padLen>0)
254                             fprintf(output, "%0*d",
255                                     padLen, 0);
256                         if (realname) {
257                             fprintf(output, "%.*s in %s\n",
258                                     (int)(secondBracketAddress-firstBracketAddress-3), firstBracketAddress+3,
259                                     realname);
260                         } else {
261                             fprintf(output, "%.*s in %.*s\n",
262                                     (int)(secondBracketAddress-firstBracketAddress-3), firstBracketAddress+3,
263                                     (int)(firstBracketAddress-symbol), symbol);
264                         }
265                     }
266                     free(symbolstrings);
267                 } else {
268                     fputs("Callstack could not be obtained\n", output);
269                 }
270             }
271     #undef ADDRESSDISPLAYLENGTH
272     #endif
273         }

275         const size_t MYSTACKSIZE = 16*1024+SIGSTKSZ; // wild guess about a reasonable buffer
276         char mytstack[MYSTACKSIZE]= {0}; // alternative stack for signal handler
277         bool bStackBelowHeap=false; // lame attempt to locate heap vs. stack address space. See CppCheckExecutor::check_wrapper()

279         /*
280          * \param[in] ptr address to be examined.
281          * \return true if address is supposed to be on stack (contrary to heap or elsewhere). If ptr is 0 false will be returned.
282          * If unknown better return false.
283          */
284         bool IsAddressOnStack(const void* ptr)
285         {
286             if (nullptr==ptr)
287                 return false;
288             char a;
289             if (bStackBelowHeap)
290                 return ptr < &a;
291             else
292                 return ptr > &a;
293         }

295         /* (declare this list here, so it may be used in signal handlers in addition to main())
296          * A list of signals available in ISO C
297          * Check out http://pubs.opengroup.org/onlinepubs/009695399/basedefs/signal.h.html
298          * For now we only want to detect abnormal behaviour for a few selected signals:
299          */
300     
301     #define DECLARE_SIGNAL(x) << std::make_pair(x, #x)
302         typedef std::map<int, std::string> Signalmap_t;
303         const Signalmap_t listofsignals = make_container< Signalmap_t > ()
304                                           DECLARE_SIGNAL(SIGABRT)
305                                           DECLARE_SIGNAL(SIGBUS)
306                                           DECLARE_SIGNAL(SIGFPE)
307                                           DECLARE_SIGNAL(SIGILL)
308                                           DECLARE_SIGNAL(SIGINT)
309                                           DECLARE_SIGNAL(SIGQUIT)
310                                           DECLARE_SIGNAL(SIGSEGV)
311                                           DECLARE_SIGNAL(SIGSYS)
312                                           // don't care: SIGTERM
313                                           DECLARE_SIGNAL(SIGUSR1)
314                                           DECLARE_SIGNAL(SIGUSR2)
315                                           ;
316     #undef DECLARE_SIGNAL
317         /*
318          * Entry pointer for signal handlers
319          * It uses functions which are not safe to be called from a signal handler,
320          * but when ending up here something went terribly wrong anyway.
321          * And all which is left is just printing some information and terminate.
322          */
323         void CppcheckSignalHandler(int signo, siginfo_t * info, void * context)
324         {
325             int type = -1;
326             pid_t killid = getpid();
327     #if defined(__linux__) && defined(REG_ERR)
328             const ucontext_t* const uc = reinterpret_cast<const ucontext_t*>(context);
329             killid = (pid_t) syscall(SYS_gettid);
330             if (uc) {
331                 type = (int)uc->uc_mcontext.gregs[REG_ERR] & 2;
332             }
333     #endif
334             const Signalmap_t::const_iterator it=listofsignals.find(signo);
335             const char * const signame = (it==listofsignals.end()) ? "unknown" : it->second.c_str();
336             bool printCallstack=true;
337             bool lowMem=false;
338             bool unexpectedSignal=true;
339             const bool isaddressonstack = IsAddressOnStack(info->si_addr);
340             FILE* output = CppCheckExecutor::getExceptionOutput();
341             switch (signo) {
342             case SIGABRT:
343                 fputs("Internal error: cppcheck received signal ", output);
344                 fputs(signame, output);
345                 fputs(" - out of memory?\n", output);
346                 lowMem=true; // educated guess
347                 break;
348             case SIGBUS:
349                 fputs("Internal error: cppcheck received signal ", output);
350                 fputs(signame, output);
351                 switch (info->si_code) {
352                 case BUS_ADRALN: // invalid address alignment
353                     fputs(" - BUS_ADRALN", output);
354                     break;
355                 case BUS_ADRERR: // nonexistent physical address
356                     fputs(" - BUS_ADRERR", output);
357                     break;
358                 case BUS_OBJERR: // object-specific hardware error
359                     fputs(" - BUS_OBJERR", output);
360                     break;
361     #ifdef BUS_MCEERR_AR
362                 case BUS_MCEERR_AR: // Hardware memory error consumed on a machine check;
363                     fputs(" - BUS_MCEERR_AR", output);
364                     break;
365     #endif
366     #ifdef BUS_MCEERR_AO
367                 case BUS_MCEERR_AO: // Hardware memory error detected in process but not consumed
368                     fputs(" - BUS_MCEERR_AO", output);
369                     break;
370     #endif
371                 default:
372                     break;
373                 }
374                 fprintf(output, " (at 0x%lx).\n",
375                         (unsigned long)info->si_addr);
376                 break;
377             case SIGFPE:
378                 fputs("Internal error: cppcheck received signal ", output);
379                 fputs(signame, output);
380                 switch (info->si_code) {
381                 case FPE_INTDIV: //     integer divide by zero
382                     fputs(" - FPE_INTDIV", output);
383                     break;
384                 case FPE_INTOVF: //     integer overflow
385                     fputs(" - FPE_INTOVF", output);
386                     break;
387                 case FPE_FLTDIV: //     floating-point divide by zero
388                     fputs(" - FPE_FLTDIV", output);
389                     break;
390                 case FPE_FLTOVF: //     floating-point overflow
391                     fputs(" - FPE_FLTOVF", output);
392                     break;
393                 case FPE_FLTUND: //     floating-point underflow
394                     fputs(" - FPE_FLTUND", output);
395                     break;
396                 case FPE_FLTRES: //     floating-point inexact result
397                     fputs(" - FPE_FLTRES", output);
398                     break;
399                 case FPE_FLTINV: //     floating-point invalid operation
400                     fputs(" - FPE_FLTINV", output);
401                     break;
402                 case FPE_FLTSUB: //     subscript out of range
403                     fputs(" - FPE_FLTSUB", output);
404                     break;
405                 default:
406                     break;
407                 }
408                 fprintf(output, " (at 0x%lx).\n",
409                         (unsigned long)info->si_addr);
410                 break;
411             case SIGILL:
412                 fputs("Internal error: cppcheck received signal ", output);
413                 fputs(signame, output);
414                 switch (info->si_code) {
415                 case ILL_ILLOPC: //     illegal opcode
416                     fputs(" - ILL_ILLOPC", output);
417                     break;
418                 case ILL_ILLOPN: //    illegal operand
419                     fputs(" - ILL_ILLOPN", output);
420                     break;
421                 case ILL_ILLADR: //    illegal addressing mode
422                     fputs(" - ILL_ILLADR", output);
423                     break;
424                 case ILL_ILLTRP: //    illegal trap
425                     fputs(" - ILL_ILLTRP", output);
426                     break;
427                 case ILL_PRVOPC: //    privileged opcode
428                     fputs(" - ILL_PRVOPC", output);
429                     break;
430                 case ILL_PRVREG: //    privileged register
431                     fputs(" - ILL_PRVREG", output);
432                     break;
433                 case ILL_COPROC: //    coprocessor error
434                     fputs(" - ILL_COPROC", output);
435                     break;
436                 case ILL_BADSTK: //    internal stack error
437                     fputs(" - ILL_BADSTK", output);
438                     break;
439                 default:
440                     break;
441                 }
442                 fprintf(output, " (at 0x%lx).%s\n",
443                         (unsigned long)info->si_addr,
444                         (isaddressonstack)?" Stackoverflow?":"");
445                 break;
446             case SIGINT:
447                 unexpectedSignal=false;
448                 fputs("cppcheck received signal ", output);
449                 fputs(signame, output);
450                 printCallstack=true;
451                 fputs(".\n", output);
452                 break;
453             case SIGSEGV:
454                 fputs("Internal error: cppcheck received signal ", output);
455                 fputs(signame, output);
456                 switch (info->si_code) {
457                 case SEGV_MAPERR: //    address not mapped to object
458                     fputs(" - SEGV_MAPERR", output);
459                     break;
460                 case SEGV_ACCERR: //    invalid permissions for mapped object
461                     fputs(" - SEGV_ACCERR", output);
462                     break;
463                 default:
464                     break;
465                 }
466                 fprintf(output, " (%sat 0x%lx).%s\n",
467                         (type==-1)? "" :
468                         (type==0) ? "reading " : "writing ",
469                         (unsigned long)info->si_addr,
470                         (isaddressonstack)?" Stackoverflow?":""
471                        );
472                 break;
473             case SIGUSR1:
474                 unexpectedSignal=false;
475                 fputs("cppcheck received signal ", output);
476                 fputs(signame, output);
477                 fputs(".\n", output);
478                 break;
479             default:
480                 fputs("Internal error: cppcheck received signal ", output);
481                 fputs(signame, output);
482                 fputs(".\n", output);
483                 break;
484             }
485             if (printCallstack) {
486                 print_stacktrace(output, true, -1, lowMem);
487             }
488             if (unexpectedSignal) {
489                 fputs("\nPlease report this to the cppcheck developers!\n", output);
490             }
491             fflush(output);
492     
493             // now let things proceed, shutdown and hopefully dump core for post-mortem analysis
494             signal(signo, SIG_DFL);
495             kill(killid, signo);
496         }
497     }

539         void PrintCallstack(FILE* outputFile, PEXCEPTION_POINTERS ex)
540         {
541             if (!loadDbgHelp())
542                 return;
543             const HANDLE hProcess   = GetCurrentProcess();
544             const HANDLE hThread    = GetCurrentThread();
545             BOOL result = pSymInitialize(
546                               hProcess,
547                               0,
548                               TRUE
549                           );
550             CONTEXT             context = *(ex->ContextRecord);
551             STACKFRAME64        stack= {0};
552     #ifdef _M_IX86
553             stack.AddrPC.Offset    = context.Eip;
554             stack.AddrPC.Mode      = AddrModeFlat;
555             stack.AddrStack.Offset = context.Esp;
556             stack.AddrStack.Mode   = AddrModeFlat;
557             stack.AddrFrame.Offset = context.Ebp;
558             stack.AddrFrame.Mode   = AddrModeFlat;
559     #else
560             stack.AddrPC.Offset    = context.Rip;
561             stack.AddrPC.Mode      = AddrModeFlat;
562             stack.AddrStack.Offset = context.Rsp;
563             stack.AddrStack.Mode   = AddrModeFlat;
564             stack.AddrFrame.Offset = context.Rsp;
565             stack.AddrFrame.Mode   = AddrModeFlat;
566     #endif
567             IMAGEHLP_SYMBOL64_EXT symbol;
568             symbol.SizeOfStruct  = sizeof(IMAGEHLP_SYMBOL64);
569             symbol.MaxNameLength = maxnamelength;
570             DWORD64 displacement   = 0;
571             int beyond_main=-1; // emergency exit, see below
572             for (ULONG frame = 0; ; frame++) {
573                 result = pStackWalk64
574                          (
575     #ifdef _M_IX86
576                              IMAGE_FILE_MACHINE_I386,
577     #else
578                              IMAGE_FILE_MACHINE_AMD64,
579     #endif
580                              hProcess,
581                              hThread,
582                              &stack,
583                              &context,
584                              nullptr,
585                              pSymFunctionTableAccess64,
586                              pSymGetModuleBase64,
587                              nullptr
588                          );
589                 if (!result)  // official end...
590                     break;
591                 pSymGetSymFromAddr64(hProcess, (ULONG64)stack.AddrPC.Offset, &displacement, &symbol);
592                 TCHAR undname[maxnamelength]= {0};
593                 pUnDecorateSymbolName((const TCHAR*)symbol.Name, (PTSTR)undname, (DWORD)GetArrayLength(undname), UNDNAME_COMPLETE);
594                 if (beyond_main>=0)
595                     ++beyond_main;
596                 if (_tcscmp(undname, _T("main"))==0)
597                     beyond_main=0;
598                 fprintf(outputFile,
599                         "%lu. 0x%08I64X in ",
600                         frame, (ULONG64)stack.AddrPC.Offset);
601                 fputs((const char *)undname, outputFile);
602                 fputc('\n', outputFile);
603                 if (0==stack.AddrReturn.Offset || beyond_main>2) // StackWalk64() sometimes doesn't reach any end...
604                     break;
605             }
606     
607             FreeLibrary(hLibDbgHelp);
608             hLibDbgHelp=0;
609         }

611         void writeMemoryErrorDetails(FILE* outputFile, PEXCEPTION_POINTERS ex, const char* description)
612         {
613             fputs(description, outputFile);
614             fprintf(outputFile, " (instruction: 0x%p) ", ex->ExceptionRecord->ExceptionAddress);
615             // Using %p for ULONG_PTR later on, so it must have size identical to size of pointer
616             // This is not the universally portable solution but good enough for Win32/64
617             C_ASSERT(sizeof(void*) == sizeof(ex->ExceptionRecord->ExceptionInformation[1]));
618             switch (ex->ExceptionRecord->ExceptionInformation[0]) {
619             case 0:
620                 fprintf(outputFile, "reading from 0x%p",
621                         reinterpret_cast<void*>(ex->ExceptionRecord->ExceptionInformation[1]));
622                 break;
623             case 1:
624                 fprintf(outputFile, "writing to 0x%p",
625                         reinterpret_cast<void*>(ex->ExceptionRecord->ExceptionInformation[1]));
626                 break;
627             case 8:
628                 fprintf(outputFile, "data execution prevention at 0x%p",
629                         reinterpret_cast<void*>(ex->ExceptionRecord->ExceptionInformation[1]));
630                 break;
631             default:
632                 break;
633             }
634         }

637          * Any evaluation of the exception needs to be done here!
638          */
639         int filterException(int code, PEXCEPTION_POINTERS ex)
640         {
641             FILE *outputFile = stdout;
642             fputs("Internal error: ", outputFile);
643             switch (ex->ExceptionRecord->ExceptionCode) {
644             case EXCEPTION_ACCESS_VIOLATION:
645                 writeMemoryErrorDetails(outputFile, ex, "Access violation");
646                 break;
647             case EXCEPTION_ARRAY_BOUNDS_EXCEEDED:
648                 fputs("Out of array bounds", outputFile);
649                 break;
650             case EXCEPTION_BREAKPOINT:
651                 fputs("Breakpoint", outputFile);
652                 break;
653             case EXCEPTION_DATATYPE_MISALIGNMENT:
654                 fputs("Misaligned data", outputFile);
655                 break;
656             case EXCEPTION_FLT_DENORMAL_OPERAND:
657                 fputs("Denormalized floating-point value", outputFile);
658                 break;
659             case EXCEPTION_FLT_DIVIDE_BY_ZERO:
660                 fputs("Floating-point divide-by-zero", outputFile);
661                 break;
662             case EXCEPTION_FLT_INEXACT_RESULT:
663                 fputs("Inexact floating-point value", outputFile);
664                 break;
665             case EXCEPTION_FLT_INVALID_OPERATION:
666                 fputs("Invalid floating-point operation", outputFile);
667                 break;
668             case EXCEPTION_FLT_OVERFLOW:
669                 fputs("Floating-point overflow", outputFile);
670                 break;
671             case EXCEPTION_FLT_STACK_CHECK:
672                 fputs("Floating-point stack overflow", outputFile);
673                 break;
674             case EXCEPTION_FLT_UNDERFLOW:
675                 fputs("Floating-point underflow", outputFile);
676                 break;
677             case EXCEPTION_GUARD_PAGE:
678                 fputs("Page-guard access", outputFile);
679                 break;
680             case EXCEPTION_ILLEGAL_INSTRUCTION:
681                 fputs("Illegal instruction", outputFile);
682                 break;
683             case EXCEPTION_IN_PAGE_ERROR:
684                 writeMemoryErrorDetails(outputFile, ex, "Invalid page access");
685                 break;
686             case EXCEPTION_INT_DIVIDE_BY_ZERO:
687                 fputs("Integer divide-by-zero", outputFile);
688                 break;
689             case EXCEPTION_INT_OVERFLOW:
690                 fputs("Integer overflow", outputFile);
691                 break;
692             case EXCEPTION_INVALID_DISPOSITION:
693                 fputs("Invalid exception dispatcher", outputFile);
694                 break;
695             case EXCEPTION_INVALID_HANDLE:
696                 fputs("Invalid handle", outputFile);
697                 break;
698             case EXCEPTION_NONCONTINUABLE_EXCEPTION:
699                 fputs("Non-continuable exception", outputFile);
700                 break;
701             case EXCEPTION_PRIV_INSTRUCTION:
702                 fputs("Invalid instruction", outputFile);
703                 break;
704             case EXCEPTION_SINGLE_STEP:
705                 fputs("Single instruction step", outputFile);
706                 break;
707             case EXCEPTION_STACK_OVERFLOW:
708                 fputs("Stack overflow", outputFile);
709                 break;
710             default:
711                 fprintf(outputFile, "Unknown exception (%d)\n",
712                         code);
713                 break;
714             }
715             fputc('\n', outputFile);
716             PrintCallstack(outputFile, ex);
717             fflush(outputFile);
718             return EXCEPTION_EXECUTE_HANDLER;
719         }

723     /**
724      * Signal/SEH handling
725      * Has to be clean for using with SEH on windows, i.e. no construction of C++ object instances is allowed!
726      * TODO Check for multi-threading issues!
727      *
728      */
729     int CppCheckExecutor::check_wrapper(CppCheck& cppcheck, int argc, const char* const argv[])
730     {
731     #ifdef USE_WINDOWS_SEH
732         FILE *outputFile = stdout;
733         __try {
734             return check_internal(cppcheck, argc, argv);
735         } __except (filterException(GetExceptionCode(), GetExceptionInformation())) {
736             // reporting to stdout may not be helpful within a GUI application...
737             fputs("Please report this to the cppcheck developers!\n", outputFile);
738             return -1;
739         }
740     #elif defined(USE_UNIX_SIGNAL_HANDLING)
741         // determine stack vs. heap
742         char stackVariable;
743         char *heapVariable=(char*)malloc(1);
744         bStackBelowHeap = &stackVariable < heapVariable;
745         free(heapVariable);
746     
747         // set up alternative stack for signal handler
748         stack_t segv_stack;
749         segv_stack.ss_sp = mytstack;
750         segv_stack.ss_flags = 0;
751         segv_stack.ss_size = MYSTACKSIZE;
752         sigaltstack(&segv_stack, nullptr);
753     
754         // install signal handler
755         struct sigaction act;
756         memset(&act, 0, sizeof(act));
757         act.sa_flags=SA_SIGINFO|SA_ONSTACK;
758         act.sa_sigaction=CppcheckSignalHandler;
759         for (std::map<int, std::string>::const_iterator sig=listofsignals.begin(); sig!=listofsignals.end(); ++sig) {
760             sigaction(sig->first, &act, nullptr);
761         }
762         return check_internal(cppcheck, argc, argv);
763     #else
764         return check_internal(cppcheck, argc, argv);
765     #endif
766     }

768     /*
769      * That is a method which gets called from check_wrapper
770      * */
771     int CppCheckExecutor::check_internal(CppCheck& cppcheck, int /*argc*/, const char* const argv[])
772     {
773         Settings& settings = cppcheck.settings();
774         _settings = &settings;
775         bool std = tryLoadLibrary(settings.library, argv[0], "std.cfg");
776         bool posix = true;
777         if (settings.standards.posix)
778             posix = tryLoadLibrary(settings.library, argv[0], "posix.cfg");
779         bool windows = true;
780         if (settings.isWindowsPlatform())
781             windows = tryLoadLibrary(settings.library, argv[0], "windows.cfg");
782     
783         if (!std || !posix || !windows) {
784             const std::list<ErrorLogger::ErrorMessage::FileLocation> callstack;
785             const std::string msg("Failed to load " + std::string(!std ? "std.cfg" : !posix ? "posix.cfg" : "windows.cfg") + ". Your Cppcheck installation is broken, please re-install.");
786     #ifdef CFGDIR
787             const std::string details("The Cppcheck binary was compiled with CFGDIR set to \"" +
788                                       std::string(CFGDIR) + "\" and will therefore search for "
789                                       "std.cfg in that path.");
790     #else
791             const std::string cfgfolder(Path::fromNativeSeparators(Path::getPathFromFilename(argv[0])) + "cfg");
792             const std::string details("The Cppcheck binary was compiled without CFGDIR set. Either the "
793                                       "std.cfg should be available in " + cfgfolder + " or the CFGDIR "
794                                       "should be configured.");
795     #endif
796             ErrorLogger::ErrorMessage errmsg(callstack, Severity::information, msg+" "+details, "failedToLoadCfg", false);
797             reportErr(errmsg);
798             return EXIT_FAILURE;
799         }
800     
801         if (settings.reportProgress)
802             time1 = std::time(0);
803     
804         if (settings.xml) {
805             reportErr(ErrorLogger::ErrorMessage::getXMLHeader(settings.xml_version));
806         }
807     
808         unsigned int returnValue = 0;
809         if (settings.jobs == 1) {
810             // Single process
811             settings.jointSuppressionReport = true;
812     
813             std::size_t totalfilesize = 0;
814             for (std::map<std::string, std::size_t>::const_iterator i = _files.begin(); i != _files.end(); ++i) {
815                 totalfilesize += i->second;
816             }
817     
818             std::size_t processedsize = 0;
819             unsigned int c = 0;
820             for (std::map<std::string, std::size_t>::const_iterator i = _files.begin(); i != _files.end(); ++i) {
821                 if (!_settings->library.markupFile(i->first)
822                     || !_settings->library.processMarkupAfterCode(i->first)) {
823                     returnValue += cppcheck.check(i->first);
824                     processedsize += i->second;
825                     if (!settings.quiet)
826                         reportStatus(c + 1, _files.size(), processedsize, totalfilesize);
827                     c++;
828                 }
829             }
830     
831             // second loop to parse all markup files which may not work until all
832             // c/cpp files have been parsed and checked
833             for (std::map<std::string, std::size_t>::const_iterator i = _files.begin(); i != _files.end(); ++i) {
834                 if (_settings->library.markupFile(i->first) && _settings->library.processMarkupAfterCode(i->first)) {
835                     returnValue += cppcheck.check(i->first);
836                     processedsize += i->second;
837                     if (!settings.quiet)
838                         reportStatus(c + 1, _files.size(), processedsize, totalfilesize);
839                     c++;
840                 }
841             }
842             cppcheck.analyseWholeProgram();
843         } else if (!ThreadExecutor::isEnabled()) {
844             std::cout << "No thread support yet implemented for this platform." << std::endl;
845         } else {
846             // Multiple processes
847             ThreadExecutor executor(_files, settings, *this);
848             returnValue = executor.check();
849         }
850     
851         if (settings.isEnabled("information") || settings.checkConfiguration) {
852             const bool enableUnusedFunctionCheck = cppcheck.unusedFunctionCheckIsEnabled();
853     
854             if (settings.jointSuppressionReport) {
855                 for (std::map<std::string, std::size_t>::const_iterator i = _files.begin(); i != _files.end(); ++i) {
856                     reportUnmatchedSuppressions(settings.nomsg.getUnmatchedLocalSuppressions(i->first, enableUnusedFunctionCheck));
857                 }
858             }
859     
860             reportUnmatchedSuppressions(settings.nomsg.getUnmatchedGlobalSuppressions(enableUnusedFunctionCheck));
861         }
862     
863         if (!settings.checkConfiguration) {
864             cppcheck.tooManyConfigsError("",0U);
865     
866             if (settings.isEnabled("missingInclude") && (Preprocessor::missingIncludeFlag || Preprocessor::missingSystemIncludeFlag)) {
867                 const std::list<ErrorLogger::ErrorMessage::FileLocation> callStack;
868                 ErrorLogger::ErrorMessage msg(callStack,
869                                               Severity::information,
870                                               "Cppcheck cannot find all the include files (use --check-config for details)\n"
871                                               "Cppcheck cannot find all the include files. Cppcheck can check the code without the "
872                                               "include files found. But the results will probably be more accurate if all the include "
873                                               "files are found. Please check your project's include directories and add all of them "
874                                               "as include directories for Cppcheck. To see what files Cppcheck cannot find use "
875                                               "--check-config.",
876                                               Preprocessor::missingIncludeFlag ? "missingInclude" : "missingIncludeSystem",
877                                               false);
878                 reportInfo(msg);
879             }
880         }
881     
882         if (settings.xml) {
883             reportErr(ErrorLogger::ErrorMessage::getXMLFooter(settings.xml_version));
884         }
885     
886         _settings = 0;
887         if (returnValue)
888             return settings.exitCode;
889         else
890             return 0;
891     }

909     {
910         (void)filename;
911     
912         if (!time1)
913             return;
914     
915         // Report progress messages every 10 seconds
916         const std::time_t time2 = std::time(nullptr);
917         if (time2 >= (time1 + 10)) {
918             time1 = time2;
919     
920             // format a progress message
921             std::ostringstream ostr;
922             ostr << "progress: "
923                  << stage
924                  << ' ' << value << '%';
925     
926             // Report progress message
927             reportOut(ostr.str());
928         }

969     bool CppCheckExecutor::tryLoadLibrary(Library& destination, const char* basepath, const char* filename)
970     {
971         const Library::Error err = destination.load(basepath, filename);
972     
973         if (err.errorcode == Library::UNKNOWN_ELEMENT)
974             std::cout << "cppcheck: Found unknown elements in configuration file '" << filename << "': " << err.reason << std::endl;
975         else if (err.errorcode != Library::OK) {
976             std::string errmsg;
977             switch (err.errorcode) {
978             case Library::OK:
979                 break;
980             case Library::FILE_NOT_FOUND:
981                 errmsg = "File not found";
982                 break;
983             case Library::BAD_XML:
984                 errmsg = "Bad XML";
985                 break;
986             case Library::UNKNOWN_ELEMENT:
987                 errmsg = "Unexpected element";
988                 break;
989             case Library::MISSING_ATTRIBUTE:
990                 errmsg = "Missing attribute";
991                 break;
992             case Library::BAD_ATTRIBUTE_VALUE:
993                 errmsg = "Bad attribute value";
994                 break;
995             case Library::UNSUPPORTED_FORMAT:
996                 errmsg = "File is of unsupported format version";
997                 break;
998             case Library::DUPLICATE_PLATFORM_TYPE:
999                 errmsg = "Duplicate platform type";
1000                 break;
1001             case Library::PLATFORM_TYPE_REDEFINED:
1002                 errmsg = "Platform type redefined";
1003                 break;
1004             }
1005             if (!err.reason.empty())
1006                 errmsg += " '" + err.reason + "'";
1007             std::cout << "cppcheck: Failed to load library configuration file '" << filename << "'. " << errmsg << std::endl;
1008             return false;
1009         }
1010         return true;
1011     }
