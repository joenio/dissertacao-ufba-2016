32     class CPPCHECKLIB VarInfo {
33     public:
34         enum AllocStatus { DEALLOC = -1, NOALLOC = 0, ALLOC = 1 };
35         struct AllocInfo {
36             AllocStatus status;
37             int type;
38             AllocInfo(int type_ = 0, AllocStatus status_ = NOALLOC) : status(status_), type(type_) {}
39         };
40         std::map<unsigned int, AllocInfo> alloctype;
41         std::map<unsigned int, std::string> possibleUsage;
42         std::set<unsigned int> conditionalAlloc;
43         std::set<unsigned int> referenced;
44     
45         void clear() {
46             alloctype.clear();
47             possibleUsage.clear();
48             conditionalAlloc.clear();
49             referenced.clear();
50         }
51     
52         void erase(unsigned int varid) {
53             alloctype.erase(varid);
54             possibleUsage.erase(varid);
55             conditionalAlloc.erase(varid);
56         }
57     
58         void swap(VarInfo &other) {
59             alloctype.swap(other.alloctype);
60             possibleUsage.swap(other.possibleUsage);
61             conditionalAlloc.swap(other.conditionalAlloc);
62             referenced.swap(other.referenced);
63         }
64     
65         /** set possible usage for all variables */
66         void possibleUsageAll(const std::string &functionName);
67     
68         void print();
69     };

72     /// @addtogroup Checks
73     /// @{
74     
75     /**
76      * @brief Check for leaks
77      */
78     
79     class CPPCHECKLIB CheckLeakAutoVar : public Check {
80     public:
81         /** This constructor is used when registering the CheckLeakAutoVar */
82         CheckLeakAutoVar() : Check(myName()) {
83         }
84     
85         /** This constructor is used when running checks. */
86         CheckLeakAutoVar(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
87             : Check(myName(), tokenizer, settings, errorLogger) {
88         }
89     
90         /** @brief Run checks against the simplified token list */
91         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
92             CheckLeakAutoVar checkLeakAutoVar(tokenizer, settings, errorLogger);
93             checkLeakAutoVar.check();
94         }
95     
96     private:
97     
98         /** check for leaks in all scopes */
99         void check();
100     
101         /** check for leaks in a function scope */
102         void checkScope(const Token * const startToken,
103                         VarInfo *varInfo,
104                         std::set<unsigned int> notzero);
105     
106         /** parse function call */
107         void functionCall(const Token *tok, VarInfo *varInfo, const VarInfo::AllocInfo& allocation);
108     
109         /** parse changes in allocation status */
110         void changeAllocStatus(VarInfo *varInfo, const VarInfo::AllocInfo& allocation, const Token* tok, const Token* arg);
111     
112         /** return. either "return" or end of variable scope is seen */
113         void ret(const Token *tok, const VarInfo &varInfo);
114     
115         /** if variable is allocated then there is a leak */
116         void leakIfAllocated(const Token *vartok, const VarInfo &varInfo);
117     
118         void leakError(const Token* tok, const std::string &varname, int type);
119         void mismatchError(const Token* tok, const std::string &varname);
120         void deallocUseError(const Token *tok, const std::string &varname);
121         void deallocReturnError(const Token *tok, const std::string &varname);
122         void doubleFreeError(const Token *tok, const std::string &varname, int type);
123     
124         /** message: user configuration is needed to complete analysis */
125         void configurationInfo(const Token* tok, const std::string &functionName);
126     
127         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
128             CheckLeakAutoVar c(0, settings, errorLogger);
129             c.deallocReturnError(0, "p");
130             c.configurationInfo(0, "f");  // user configuration is needed to complete analysis
131             c.doubleFreeError(0, "varname", 0);
132         }
133     
134         static std::string myName() {
135             return "Leaks (auto variables)";
136         }
137     
138         std::string classInfo() const {
139             return "Detect when a auto variable is allocated but not deallocated or deallocated twice.\n";
140         }
141     };
