791          * @param one_line true=>copy all tokens to the same line as dest. false=>copy all tokens to dest while keeping the 'line breaks'
792          * @return new location of last token copied
793          */
794         static Token *copyTokens(Token *dest, const Token *first, const Token *last, bool one_line = true);
795     
796         /**
