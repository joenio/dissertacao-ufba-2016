17     #ifndef REDIRECT_H
18     #define REDIRECT_H
19     
20     #include <iostream>
21     
22     extern std::ostringstream errout;
23     extern std::ostringstream output;
24     /**
25       * @brief Utility class for capturing cout and cerr to ostringstream buffers
26       * for later use. Uses RAII to stop redirection when the object goes out of
27       * scope.
28       */
29     class RedirectOutputError {
30     public:
31         /** Set up redirection, flushing anything in the pipes. */
32         RedirectOutputError() {
33             // flush all old output
34             std::cout.flush();
35             std::cerr.flush();
36     
37             _oldCout = std::cout.rdbuf(); // back up cout's streambuf
38             _oldCerr = std::cerr.rdbuf(); // back up cerr's streambuf
39     
40             std::cout.rdbuf(_out.rdbuf()); // assign streambuf to cout
41             std::cerr.rdbuf(_err.rdbuf()); // assign streambuf to cerr
42         }
43     
44         /** Revert cout and cerr behaviour */
45         ~RedirectOutputError() {
46             std::cout.rdbuf(_oldCout); // restore cout's original streambuf
47             std::cerr.rdbuf(_oldCerr); // restore cerrs's original streambuf
48     
49             errout << _err.str();
50             output << _out.str();
51         }
52     
53         /** Return what would be printed to cout. See also clearOutput() */
54         std::string getOutput() const {
55             return _out.str();
56         }
57     
58         /** Normally called after getOutput() to prevent same text to be returned
59         twice. */
60         void clearOutput() {
61             _out.str("");
62         }
63     
64         /** Return what would be printed to cerr. See also clearErrout() */
65         std::string getErrout() const {
66             return _err.str();
67         }
68     
69         /** Normally called after getErrout() to prevent same text to be returned
70         twice. */
71         void clearErrout() {
72             _err.str("");
73         }
74     
75     private:
76         std::ostringstream _out;
77         std::ostringstream _err;
78         std::streambuf *_oldCout;
79         std::streambuf *_oldCerr;
80     };
81     
82     #define REDIRECT RedirectOutputError redir;
83     #define GET_REDIRECT_OUTPUT redir.getOutput()
84     #define CLEAR_REDIRECT_OUTPUT redir.clearOutput()
85     #define GET_REDIRECT_ERROUT redir.getErrout()
86     #define CLEAR_REDIRECT_ERROUT redir.clearErrout()
87     
88     #endif
