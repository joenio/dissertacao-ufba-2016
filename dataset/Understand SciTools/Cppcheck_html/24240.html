26     #include <string>

28     
29     class TestThreadExecutor : public TestFixture {
30     public:
31         TestThreadExecutor() : TestFixture("TestThreadExecutor") {
32         }
33     
34     private:
35         Settings settings;
36     
37         /**
38          * Execute check using n jobs for y files which are have
39          * identical data, given within data.
40          */
41         void check(unsigned int jobs, int files, int result, const std::string &data) {
42             errout.str("");
43             output.str("");
44             if (!ThreadExecutor::isEnabled()) {
45                 // Skip this check on systems which don't use this feature
46                 return;
47             }
48     
49             std::map<std::string, std::size_t> filemap;
50             for (int i = 1; i <= files; ++i) {
51                 std::ostringstream oss;
52                 oss << "file_" << i << ".cpp";
53                 filemap[oss.str()] = 1;
54             }
55     
56             settings.jobs = jobs;
57             ThreadExecutor executor(filemap, settings, *this);
58             for (std::map<std::string, std::size_t>::const_iterator i = filemap.begin(); i != filemap.end(); ++i)
59                 executor.addFileContent(i->first, data);
60     
61             ASSERT_EQUALS(result, executor.check());
62         }
63     
64         void run() {
65             LOAD_LIB_2(settings.library, "std.cfg");
66     
67             TEST_CASE(deadlock_with_many_errors);
68             TEST_CASE(many_threads);
69             TEST_CASE(no_errors_more_files);
70             TEST_CASE(no_errors_less_files);
71             TEST_CASE(no_errors_equal_amount_files);
72             TEST_CASE(one_error_less_files);
73             TEST_CASE(one_error_several_files);
74         }
75     
76         void deadlock_with_many_errors() {
77             std::ostringstream oss;
78             oss << "int main()\n"
79                 << "{\n";
80             for (int i = 0; i < 500; i++)
81                 oss << "  {char *a = malloc(10);}\n";
82     
83             oss << "  return 0;\n"
84                 << "}\n";
85             check(2, 3, 3, oss.str());
86         }
87     
88         void many_threads() {
89             check(16, 100, 100,
90                   "int main()\n"
91                   "{\n"
92                   "  char *a = malloc(10);\n"
93                   "  return 0;\n"
94                   "}");
95         }
96     
97         void no_errors_more_files() {
98             check(2, 3, 0,
99                   "int main()\n"
100                   "{\n"
101                   "  return 0;\n"
102                   "}");
103         }
104     
105         void no_errors_less_files() {
106             check(2, 1, 0,
107                   "int main()\n"
108                   "{\n"
109                   "  return 0;\n"
110                   "}");
111         }
112     
113         void no_errors_equal_amount_files() {
114             check(2, 2, 0,
115                   "int main()\n"
116                   "{\n"
117                   "  return 0;\n"
118                   "}");
119         }
120     
121         void one_error_less_files() {
122             check(2, 1, 1,
123                   "int main()\n"
124                   "{\n"
125                   "  {char *a = malloc(10);}\n"
126                   "  return 0;\n"
127                   "}");
128         }
129     
130         void one_error_several_files() {
131             check(2, 20, 20,
132                   "int main()\n"
133                   "{\n"
134                   "  {char *a = malloc(10);}\n"
135                   "  return 0;\n"
136                   "}");
137         }
138     };
