241         void function_arg_minsize() const {
242             const char xmldata[] = "<?xml version=\"1.0\"?>\n"
243                                    "<def>\n"
244                                    "  <function name=\"foo\">\n"
245                                    "    <arg nr=\"1\"><minsize type=\"strlen\" arg=\"2\"/></arg>\n"
246                                    "    <arg nr=\"2\"><minsize type=\"argvalue\" arg=\"3\"/></arg>\n"
247                                    "    <arg nr=\"3\"/>\n"
248                                    "  </function>\n"
249                                    "</def>";
250     
251             Library library;
252             readLibrary(library, xmldata);
253     
254             TokenList tokenList(nullptr);
255             std::istringstream istr("foo(a,b,c);");
256             tokenList.createTokens(istr);
257             tokenList.front()->next()->astOperand1(tokenList.front());
258     
259             // arg1: type=strlen arg2
260             const std::list<Library::ArgumentChecks::MinSize> *minsizes = library.argminsizes(tokenList.front(),1);
261             ASSERT_EQUALS(true, minsizes != nullptr);
262             ASSERT_EQUALS(1U, minsizes ? minsizes->size() : 1U);
263             if (minsizes && minsizes->size() == 1U) {
264                 const Library::ArgumentChecks::MinSize &m = minsizes->front();
265                 ASSERT_EQUALS(Library::ArgumentChecks::MinSize::STRLEN, m.type);
266                 ASSERT_EQUALS(2, m.arg);
267             }
268     
269             // arg2: type=argvalue arg3
270             minsizes = library.argminsizes(tokenList.front(), 2);
271             ASSERT_EQUALS(true, minsizes != nullptr);
272             ASSERT_EQUALS(1U, minsizes ? minsizes->size() : 1U);
273             if (minsizes && minsizes->size() == 1U) {
274                 const Library::ArgumentChecks::MinSize &m = minsizes->front();
275                 ASSERT_EQUALS(Library::ArgumentChecks::MinSize::ARGVALUE, m.type);
276                 ASSERT_EQUALS(3, m.arg);
277             }
278         }

309         void function_warn() const {
310             const char xmldata[] = "<?xml version=\"1.0\"?>\n"
311                                    "<def>\n"
312                                    "  <function name=\"a\">\n"
313                                    "    <warn severity=\"style\" cstd=\"c99\">Message</warn>\n"
314                                    "  </function>\n"
315                                    "  <function name=\"b\">\n"
316                                    "    <warn severity=\"performance\" cppstd=\"c++11\" reason=\"Obsolescent\" alternatives=\"c,d,e\"/>\n"
317                                    "  </function>\n"
318                                    "</def>";
319     
320             Library library;
321             readLibrary(library, xmldata);
322     
323             TokenList tokenList(nullptr);
324             std::istringstream istr("a(); b();");
325             tokenList.createTokens(istr);
326     
327             const Library::WarnInfo* a = library.getWarnInfo(tokenList.front());
328             const Library::WarnInfo* b = library.getWarnInfo(tokenList.front()->tokAt(4));
329     
330             ASSERT_EQUALS(2, library.functionwarn.size());
331             ASSERT(a && b);
332             if (a && b) {
333                 ASSERT_EQUALS("Message", a->message);
334                 ASSERT_EQUALS(Severity::style, a->severity);
335                 ASSERT_EQUALS(Standards::C99, a->standards.c);
336                 ASSERT_EQUALS(Standards::CPP03, a->standards.cpp);
337     
338                 ASSERT_EQUALS("Obsolescent function 'b' called. It is recommended to use 'c', 'd' or 'e' instead.", b->message);
339                 ASSERT_EQUALS(Severity::performance, b->severity);
340                 ASSERT_EQUALS(Standards::C89, b->standards.c);
341                 ASSERT_EQUALS(Standards::CPP11, b->standards.cpp);
342             }

421                                    "<def>\n"
422                                    "  <container id=\"A\" startPattern=\"std :: A &lt;\" endPattern=\"&gt; !!::\" itEndPattern=\"&gt; :: iterator\">\n"
423                                    "    <type templateParameter=\"1\"/>\n"
424                                    "    <size templateParameter=\"4\">\n"
425                                    "      <function name=\"resize\" action=\"resize\"/>\n"
426                                    "      <function name=\"clear\" action=\"clear\"/>\n"
427                                    "      <function name=\"size\" yields=\"size\"/>\n"
428                                    "      <function name=\"empty\" yields=\"empty\"/>\n"
429                                    "      <function name=\"push_back\" action=\"push\"/>\n"
430                                    "      <function name=\"pop_back\" action=\"pop\"/>\n"
431                                    "    </size>\n"
432                                    "    <access>\n"
433                                    "      <function name=\"at\" yields=\"at_index\"/>\n"
434                                    "      <function name=\"begin\" yields=\"start-iterator\"/>\n"
435                                    "      <function name=\"end\" yields=\"end-iterator\"/>\n"
436                                    "      <function name=\"data\" yields=\"buffer\"/>\n"
437                                    "      <function name=\"c_str\" yields=\"buffer-nt\"/>\n"
438                                    "      <function name=\"front\" yields=\"item\"/>\n"
439                                    "      <function name=\"find\" action=\"find\"/>\n"
440                                    "    </access>\n"
441                                    "  </container>\n"
442                                    "  <container id=\"B\" startPattern=\"std :: B &lt;\" inherits=\"A\" opLessAllowed=\"false\">\n"
443                                    "    <size templateParameter=\"3\"/>\n" // Inherits all but templateParameter
444                                    "  </container>\n"
445                                    "  <container id=\"C\">\n"
446                                    "    <type string=\"std-like\"/>\n"
447                                    "    <access indexOperator=\"array-like\"/>\n"
448                                    "  </container>\n"
449                                    "</def>";
450     
451             Library library;
452             readLibrary(library, xmldata);
453     
454             Library::Container& A = library.containers["A"];
455             Library::Container& B = library.containers["B"];
456             Library::Container& C = library.containers["C"];
457     
458             ASSERT_EQUALS(A.type_templateArgNo, 1);
459             ASSERT_EQUALS(A.size_templateArgNo, 4);
460             ASSERT_EQUALS(A.startPattern, "std :: A <");
461             ASSERT_EQUALS(A.endPattern, "> !!::");
462             ASSERT_EQUALS(A.itEndPattern, "> :: iterator");
463             ASSERT_EQUALS(A.stdStringLike, false);
464             ASSERT_EQUALS(A.arrayLike_indexOp, false);
465             ASSERT_EQUALS(A.opLessAllowed, true);
466             ASSERT_EQUALS(Library::Container::SIZE, A.getYield("size"));
467             ASSERT_EQUALS(Library::Container::EMPTY, A.getYield("empty"));
468             ASSERT_EQUALS(Library::Container::AT_INDEX, A.getYield("at"));
469             ASSERT_EQUALS(Library::Container::START_ITERATOR, A.getYield("begin"));
470             ASSERT_EQUALS(Library::Container::END_ITERATOR, A.getYield("end"));
471             ASSERT_EQUALS(Library::Container::BUFFER, A.getYield("data"));
472             ASSERT_EQUALS(Library::Container::BUFFER_NT, A.getYield("c_str"));
473             ASSERT_EQUALS(Library::Container::ITEM, A.getYield("front"));
474             ASSERT_EQUALS(Library::Container::NO_YIELD, A.getYield("foo"));
475             ASSERT_EQUALS(Library::Container::RESIZE, A.getAction("resize"));
476             ASSERT_EQUALS(Library::Container::CLEAR, A.getAction("clear"));
477             ASSERT_EQUALS(Library::Container::PUSH, A.getAction("push_back"));
478             ASSERT_EQUALS(Library::Container::POP, A.getAction("pop_back"));
479             ASSERT_EQUALS(Library::Container::FIND, A.getAction("find"));
480             ASSERT_EQUALS(Library::Container::NO_ACTION, A.getAction("foo"));
481     
482             ASSERT_EQUALS(B.type_templateArgNo, 1);
483             ASSERT_EQUALS(B.size_templateArgNo, 3);
484             ASSERT_EQUALS(B.startPattern, "std :: B <");
485             ASSERT_EQUALS(B.endPattern, "> !!::");
486             ASSERT_EQUALS(B.itEndPattern, "> :: iterator");
487             ASSERT_EQUALS(B.functions.size(), A.functions.size());
488             ASSERT_EQUALS(B.opLessAllowed, false);
489     
490             ASSERT(C.functions.empty());
491             ASSERT_EQUALS(C.type_templateArgNo, -1);
492             ASSERT_EQUALS(C.size_templateArgNo, -1);
493             ASSERT_EQUALS(C.stdStringLike, true);
494             ASSERT_EQUALS(C.arrayLike_indexOp, true);
495         }
