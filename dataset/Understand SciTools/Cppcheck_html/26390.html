31     void TestProjectFile::loadSimple()
32     {
33         const QString filepath(QString(SRCDIR) + "/../data/projectfiles/simple.cppcheck");
34         ProjectFile pfile(filepath);
35         QVERIFY(pfile.Read());
36         QCOMPARE(pfile.GetRootPath(), QString("../.."));
37         QStringList includes = pfile.GetIncludeDirs();
38         QCOMPARE(includes.size(), 2);
39         QCOMPARE(includes[0], QString("lib/"));
40         QCOMPARE(includes[1], QString("cli/"));
41         QStringList paths = pfile.GetCheckPaths();
42         QCOMPARE(paths.size(), 2);
43         QCOMPARE(paths[0], QString("gui/"));
44         QCOMPARE(paths[1], QString("test/"));
45         QStringList excludes = pfile.GetExcludedPaths();
46         QCOMPARE(excludes.size(), 1);
47         QCOMPARE(excludes[0], QString("gui/temp/"));
48         QStringList defines = pfile.GetDefines();
49         QCOMPARE(defines.size(), 1);
50         QCOMPARE(defines[0], QString("FOO"));
51     }
