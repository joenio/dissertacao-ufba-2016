31     class ErrorLogger;
32     class Settings;

34     /// @addtogroup Core
35     /// @{
36     
37     /**
38      * @brief The cppcheck preprocessor.
39      * The preprocessor has special functionality for extracting the various ifdef
40      * configurations that exist in a source file.
41      */
42     class CPPCHECKLIB Preprocessor {
43     public:
44     
45         /**
46          * Include file types.
47          */
48         enum HeaderTypes {
49             NoHeader = 0,
50             UserHeader,
51             SystemHeader
52         };
53     
54         /** character that is inserted in expanded macros */
55         static char macroChar;
56     
57         Preprocessor(Settings& settings, ErrorLogger *errorLogger = nullptr);
58     
59         static bool missingIncludeFlag;
60         static bool missingSystemIncludeFlag;
61     
62         /**
63          * Extract the code for each configuration
64          * @param istr The (file/string) stream to read from.
65          * @param result The map that will get the results
66          * @param filename The name of the file to check e.g. "src/main.cpp"
67          * @param includePaths List of paths where include files should be searched from,
68          * single path can be e.g. in format "include/".
69          * There must be a path separator at the end. Default parameter is empty list.
70          * Note that if path from given filename is also extracted and that is used as
71          * a last include path if include file was not found from earlier paths.
72          */
73         void preprocess(std::istream &istr, std::map<std::string, std::string> &result, const std::string &filename, const std::list<std::string> &includePaths = std::list<std::string>());
74     
75         /**
76          * Extract the code for each configuration. Use this with getcode() to get the
77          * file data for each individual configuration.
78          *
79          * @param srcCodeStream The (file/string) stream to read from.
80          * @param processedFile Give reference to empty string as a parameter,
81          * function will fill processed file here. Use this also as a filedata parameter
82          * to getcode() if you received more than once configurations.
83          * @param resultConfigurations List of configurations. Pass these one by one
84          * to getcode() with processedFile.
85          * @param filename The name of the file to check e.g. "src/main.cpp"
86          * @param includePaths List of paths where include files should be searched from,
87          * single path can be e.g. in format "include/".
88          * There must be a path separator at the end. Default parameter is empty list.
89          * Note that if path from given filename is also extracted and that is used as
90          * a last include path if include file was not found from earlier paths.
91          */
92         void preprocess(std::istream &srcCodeStream, std::string &processedFile, std::list<std::string> &resultConfigurations, const std::string &filename, const std::list<std::string> &includePaths);
93     
94         /** Just read the code into a string. Perform simple cleanup of the code */
95         std::string read(std::istream &istr, const std::string &filename);
96     
97         /** read preprocessor statements into a string. */
98         static std::string readpreprocessor(std::istream &istr, const unsigned int bom);
99     
100         /** should __cplusplus be defined? */
101         static bool cplusplus(const Settings *settings, const std::string &filename);
102     
103         /**
104          * Get preprocessed code for a given configuration
105          * @param filedata file data including preprocessing 'if', 'define', etc
106          * @param cfg configuration to read out
107          * @param filename name of source file
108          */
109         std::string getcode(const std::string &filedata, const std::string &cfg, const std::string &filename);
110     
111         /**
112          * simplify condition
113          * @param variables Variable values
114          * @param condition The condition to simplify
115          * @param match if true, 'defined(A)' is replaced with 0 if A is not defined
116          */
117         void simplifyCondition(const std::map<std::string, std::string> &variables, std::string &condition, bool match);
118     
119         /**
120          * preprocess all whitespaces
121          * @param processedFile The data to be processed
122          */
123         static void preprocessWhitespaces(std::string &processedFile);
124     
125         /**
126          * make sure empty configuration macros are not used in code. the given code must be a single configuration
127          * @param code The input code
128          * @param cfg configuration
129          * @return true => configuration is valid
130          */
131         bool validateCfg(const std::string &code, const std::string &cfg);
132         void validateCfgError(const std::string &cfg, const std::string &macro);
133     
134         void handleUndef(std::list<std::string> &configurations) const;
135     
136         /**
137          * report error
138          * @param fileName name of file that the error was found in
139          * @param linenr linenr in file
140          * @param errorLogger Error logger to write error to
141          * @param errorType id string for error
142          * @param errorText Plain text
143          */
144         static void writeError(const std::string &fileName, const unsigned int linenr, ErrorLogger *errorLogger, const std::string &errorType, const std::string &errorText);
145     
146         /**
147          * Replace "#if defined" with "#ifdef" where possible
148          *
149          * @param str The string to be converted
150          */
151         void replaceIfDefined(std::string &str) const;
152     
153         /**
154          * expand macros in code. ifdefs etc are ignored so the code must be a single configuration
155          * @param code The input code
156          * @param filename filename of source file
157          * @param cfg user given -D configuration
158          * @param errorLogger Error logger to write errors to (if any)
159          * @return the expanded string
160          */
161         static std::string expandMacros(const std::string &code, std::string filename, const std::string &cfg, ErrorLogger *errorLogger);
162     
163         /**
164          * Remove comments from code. This should only be called from read().
165          * If there are inline suppressions, the _settings member is modified
166          * @param str Code processed by read().
167          * @param filename filename
168          * @return code without comments
169          */
170         std::string removeComments(const std::string &str, const std::string &filename);
171     
172         /**
173          * Cleanup 'if 0' from the code
174          * @param code Code processed by read().
175          * @return code without 'if 0'
176          */
177         static std::string removeIf0(const std::string &code);
178     
179         /**
180          * Remove redundant parentheses from preprocessor commands. This should only be called from read().
181          * @param str Code processed by read().
182          * @return code with reduced parentheses
183          */
184         static std::string removeParentheses(const std::string &str);
185     
186         /**
187          * clean up #-preprocessor lines (only)
188          * @param processedFile The data to be processed
189          */
190         static std::string preprocessCleanupDirectives(const std::string &processedFile);
191     
192         /**
193          * Returns the string between double quote characters or \< \> characters.
194          * @param str e.g. \code#include "menu.h"\endcode or \code#include <menu.h>\endcode
195          * After function call it will contain e.g. "menu.h" without double quotes.
196          * @return NoHeader empty string if double quotes or \< \> were not found.
197          *         UserHeader if file surrounded with "" was found
198          *         SystemHeader if file surrounded with \<\> was found
199          */
200         static Preprocessor::HeaderTypes getHeaderFileName(std::string &str);
201     private:
202     
203         /**
204          * Remove space that has new line character on left or right side of it.
205          *
206          * @param str The string to be converted
207          * @return The string where space characters have been removed.
208          */
209         static std::string removeSpaceNearNL(const std::string &str);
210     
211         static std::string getdef(std::string line, bool def);
212     
213     public:
214     
215         /**
216          * Get all possible configurations sorted in alphabetical order.
217          * By looking at the ifdefs and ifndefs in filedata
218          */
219         std::list<std::string> getcfgs(const std::string &filedata, const std::string &filename, const std::map<std::string, std::string> &defs);
220     
221         /**
222          * Remove asm(...) from a string
223          * @param str Code
224          */
225         static void removeAsm(std::string &str);
226     
227         /**
228          * Evaluate condition 'numerically'
229          * @param cfg configuration
230          * @param def condition
231          * @return result when evaluating the condition
232          */
233         bool match_cfg_def(std::map<std::string, std::string> cfg, std::string def);
234     
235         static void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings);
236     
237         /**
238          * handle includes for a specific configuration
239          * @param code code in string
240          * @param filePath filename of code
241          * @param includePaths Paths where headers might be
242          * @param defs defines (only values)
243          * @param pragmaOnce includes that has already been included and contains a \#pragma once statement
244          * @param includes provide a empty list. this is just used to prevent recursive inclusions.
245          * \return resulting string
246          */
247         std::string handleIncludes(const std::string &code, const std::string &filePath, const std::list<std::string> &includePaths, std::map<std::string,std::string> &defs, std::set<std::string> &pragmaOnce, std::list<std::string> includes);
248     
249         void setFile0(const std::string &f) {
250             file0 = f;
251         }
252     
253     private:
254         void missingInclude(const std::string &filename, unsigned int linenr, const std::string &header, HeaderTypes headerType);
255     
256         void error(const std::string &filename, unsigned int linenr, const std::string &msg);
257     
258         /**
259          * Search includes from code and append code from the included
260          * file
261          * @param[in,out] code The source code to modify
262          * @param filePath Relative path to file to check e.g. "src/main.cpp"
263          * @param includePaths List of paths where include files should be searched from,
264          * single path can be e.g. in format "include/".
265          * There must be a path separator at the end. Default parameter is empty list.
266          * Note that if path from given filename is also extracted and that is used as
267          * a last include path if include file was not found from earlier paths.
268          */
269         void handleIncludes(std::string &code, const std::string &filePath, const std::list<std::string> &includePaths);
270     
271         Settings& _settings;
272         ErrorLogger *_errorLogger;
273     
274         /** filename for cpp/c file - useful when reporting errors */
275         std::string file0;
276     };
277     
278     /// @}
279     //---------------------------------------------------------------------------
280     #endif // preprocessorH
