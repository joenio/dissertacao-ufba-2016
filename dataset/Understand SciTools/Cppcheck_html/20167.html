68     void ThreadHandler::Check(const Settings &settings, bool all)
69     {
70         if (mResults.GetFileCount() == 0 || mRunningThreadCount > 0 || settings.jobs == 0) {
71             qDebug() << "Can't start checking if there's no files to check or if check is in progress.";
72             emit Done();
73             return;
74         }
75     
76         SetThreadCount(settings.jobs);
77     
78         mRunningThreadCount = mThreads.size();
79     
80         if (mResults.GetFileCount() < mRunningThreadCount) {
81             mRunningThreadCount = mResults.GetFileCount();
82         }
83     
84         for (int i = 0; i < mRunningThreadCount; i++) {
85             mThreads[i]->Check(settings);
86         }
87     
88         // Date and time when checking starts..
89         mCheckStartTime = QDateTime::currentDateTime();
90     
91         mTime.start();
92     }

99     void ThreadHandler::SetThreadCount(const int count)
100     {
101         if (mRunningThreadCount > 0 ||
102             count == mThreads.size() ||
103             count <= 0) {
104             return;
105         }
106     
107         //Remove unused old threads
108         RemoveThreads();
109         //Create new threads
110         for (int i = mThreads.size(); i < count; i++) {
111             mThreads << new CheckThread(mResults);
112             connect(mThreads.last(), SIGNAL(Done()),
113                     this, SLOT(ThreadDone()));
114             connect(mThreads.last(), SIGNAL(FileChecked(const QString &)),
115                     &mResults, SLOT(FileChecked(const QString &)));
116         }
117     
118     }

121     void ThreadHandler::RemoveThreads()
122     {
123         for (int i = 0; i < mThreads.size(); i++) {
124             mThreads[i]->terminate();
125             disconnect(mThreads.last(), SIGNAL(Done()),
126                        this, SLOT(ThreadDone()));
127             disconnect(mThreads.last(), SIGNAL(FileChecked(const QString &)),
128                        &mResults, SLOT(FileChecked(const QString &)));
129     
130             delete mThreads[i];
131         }
132     
133         mThreads.clear();
134     }

136     void ThreadHandler::ThreadDone()
137     {
138         mRunningThreadCount--;
139         if (mRunningThreadCount == 0) {
140             emit Done();
141     
142             mScanDuration = mTime.elapsed();
143     
144             // Set date/time used by the recheck
145             if (!mCheckStartTime.isNull()) {
146                 mLastCheckTime = mCheckStartTime;
147                 mCheckStartTime = QDateTime();
148             }
149         }
150     }

152     void ThreadHandler::Stop()
153     {
154         mCheckStartTime = QDateTime();
155         for (int i = 0; i < mThreads.size(); i++) {
156             mThreads[i]->stop();
157         }
158     }

200     QStringList ThreadHandler::GetReCheckFiles(bool all) const
201     {
202         if (mLastCheckTime.isNull() || all)
203             return mLastFiles;
204     
205         std::set<QString> modified;
206         std::set<QString> unmodified;
207     
208         QStringList files;
209         for (int i = 0; i < mLastFiles.size(); ++i) {
210             if (NeedsReCheck(mLastFiles[i], modified, unmodified))
211                 files.push_back(mLastFiles[i]);
212         }
213         return files;
214     }

216     bool ThreadHandler::NeedsReCheck(const QString &filename, std::set<QString> &modified, std::set<QString> &unmodified) const
217     {
218         if (modified.find(filename) != modified.end())
219             return true;
220     
221         if (unmodified.find(filename) != unmodified.end())
222             return false;
223     
224         if (QFileInfo(filename).lastModified() > mLastCheckTime) {
225             return true;
226         }
227     
228         // Parse included files recursively
229         QFile f(filename);
230         if (!f.open(QIODevice::ReadOnly | QIODevice::Text))
231             return false;
232     
233         // prevent recursion..
234         unmodified.insert(filename);
235     
236         QTextStream in(&f);
237         while (!in.atEnd()) {
238             QString line = in.readLine();
239             if (line.startsWith("#include \"")) {
240                 line.remove(0,10);
241                 int i = line.indexOf("\"");
242                 if (i > 0) {
243                     line.remove(i,line.length());
244                     line = QFileInfo(filename).absolutePath() + "/" + line;
245                     if (NeedsReCheck(line, modified, unmodified)) {
246                         modified.insert(line);
247                         return true;
248                     }
249                 }
250             }
251         }
252     
253         return false;
254     }
