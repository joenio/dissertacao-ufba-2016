37     /**
38      * @brief buffer overruns and array index out of bounds
39      *
40      * Buffer overrun and array index out of bounds are pretty much the same.
41      * But I generally use 'array index' if the code contains []. And the given
42      * index is out of bounds.
43      * I generally use 'buffer overrun' if you for example call a strcpy or
44      * other function and pass a buffer and reads or writes too much data.
45      */
46     class CPPCHECKLIB CheckBufferOverrun : public Check {
47     public:
48     
49         /** This constructor is used when registering the CheckClass */
50         CheckBufferOverrun() : Check(myName()) {
51         }
52     
53         /** This constructor is used when running checks. */
54         CheckBufferOverrun(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
55             : Check(myName(), tokenizer, settings, errorLogger) {
56         }
57     
58         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
59             CheckBufferOverrun checkBufferOverrun(tokenizer, settings, errorLogger);
60             checkBufferOverrun.bufferOverrun();
61             checkBufferOverrun.bufferOverrun2();
62             checkBufferOverrun.arrayIndexThenCheck();
63             checkBufferOverrun.negativeArraySize();
64         }

67         void bufferOverrun();
68

70         void bufferOverrun2();

72         /** @brief Using array index before bounds check */
73         void arrayIndexThenCheck();
74     
75         /** @brief negative size for array */
76         void negativeArraySize();

90         void checkGlobalAndLocalVariable();

101         /** Information about N-dimensional array */
102         class CPPCHECKLIB ArrayInfo {
103         private:
104             /** number of elements of array */
105             std::vector<MathLib::bigint> _num;
106     
107             /** full name of variable as pattern */
108             std::string _varname;
109     
110             /** size of each element in array */
111             MathLib::bigint _element_size;
112     
113             /** declaration id */
114             unsigned int _declarationId;
115     
116         public:
117             ArrayInfo();
118             ArrayInfo(const Variable *var, const Tokenizer *tokenizer, const Library *library, const unsigned int forcedeclid = 0);
119     
120             /**
121              * Create array info with specified data
122              * The intention is that this is only a temporary solution.. all
123              * checking should be based on ArrayInfo from the start and then
124              * this will not be needed as the declare can be used instead.
125              */
126             ArrayInfo(unsigned int id, const std::string &name, MathLib::bigint size1, MathLib::bigint n);
127     
128             /** Create a copy ArrayInfo where the number of elements have been limited by a value */
129             ArrayInfo limit(MathLib::bigint value) const;
130     
131             /** array sizes */
132             const std::vector<MathLib::bigint> &num() const {
133                 return _num;
134             }
135     
136             /** array size */
137             MathLib::bigint num(std::size_t index) const {
138                 return _num[index];
139             }
140             void num(std::size_t index, MathLib::bigint number) {
141                 _num[index] = number;
142             }
143     
144             /** size of each element */
145             MathLib::bigint element_size() const {
146                 return _element_size;
147             }
148     
149             /** Variable name */
150             unsigned int declarationId() const {
151                 return _declarationId;
152             }
153             void declarationId(unsigned int id) {
154                 _declarationId = id;
155             }
156     
157             /** Variable name */
158             const std::string &varname() const {
159                 return _varname;
160             }
161             void varname(const std::string &name) {
162                 _varname = name;
163             }
164     
165             MathLib::bigint numberOfElements() const;
166             MathLib::bigint totalIndex(const std::vector<ValueFlow::Value> &indexes) const;
167         };

240     public:
241         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
242             CheckBufferOverrun c(0, settings, errorLogger);
243             const std::vector<MathLib::bigint> indexes(2, 1);
244             c.arrayIndexOutOfBoundsError(0, ArrayInfo(0, "array", 1, 2), indexes);
245             c.bufferOverrunError(0, std::string("buffer"));
246             c.strncatUsageError(0);
247             c.outOfBoundsError(0, "index", true, 2, 1);
248             c.sizeArgumentAsCharError(0);
249             c.terminateStrncpyError(0, "buffer");
250             c.bufferNotZeroTerminatedError(0, "buffer", "strncpy");
251             c.negativeIndexError(0, -1);
252             c.cmdLineArgsError(0);
253             c.pointerOutOfBoundsError(nullptr, nullptr, 0);
254             c.arrayIndexThenCheckError(0, "index");
255             c.possibleBufferOverrunError(0, "source", "destination", false);
256             c.argumentSizeError(0, "function", "array");
257             c.negativeMemoryAllocationSizeError(0);
258             c.negativeArraySizeError(0);
259             c.reportError(nullptr, Severity::warning, "arrayIndexOutOfBoundsCond", "Array 'x[10]' accessed at index 20, which is out of bounds. Otherwise condition 'y==20' is redundant.");
260         }

267         std::string classInfo() const {
268             return "Out of bounds checking:\n"
269                    "- Array index out of bounds detection by value flow analysis\n"
270                    "- Dangerous usage of strncat()\n"
271                    "- char constant passed as size to function like memset()\n"
272                    "- strncpy() leaving string unterminated\n"
273                    "- Accessing array with negative index\n"
274                    "- Unsafe usage of main(argv, argc) arguments\n"
275                    "- Accessing array with index variable before checking its value\n"
276                    "- Check for large enough arrays being passed to functions\n"
277                    "- Allocating memory with a negative size\n";
278         }
