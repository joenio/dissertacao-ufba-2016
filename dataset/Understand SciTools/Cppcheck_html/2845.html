20     //---------------------------------------------------------------------------
21     #ifndef checkstringH
22     #define checkstringH
23     //---------------------------------------------------------------------------
24     
25     #include "config.h"
26     #include "check.h"
27     
28     /// @addtogroup Checks
29     /// @{
30     
31     
32     /** @brief Detect misusage of C-style strings and related standard functions */
33     
34     class CPPCHECKLIB CheckString : public Check {
35     public:
36         /** @brief This constructor is used when registering the CheckClass */
37         CheckString() : Check(myName()) {
38         }
39     
40         /** @brief This constructor is used when running checks. */
41         CheckString(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
42             : Check(myName(), tokenizer, settings, errorLogger) {
43         }
44     
45         /** @brief Run checks against the normal token list */
46         void runChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
47             CheckString checkString(tokenizer, settings, errorLogger);
48     
49             // Checks
50             checkString.strPlusChar();
51             checkString.checkSuspiciousStringCompare();
52         }
53     
54         /** @brief Run checks against the simplified token list */
55         void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) {
56             CheckString checkString(tokenizer, settings, errorLogger);
57     
58             // Checks
59             checkString.checkIncorrectStringCompare();
60             checkString.checkAlwaysTrueOrFalseStringCompare();
61             checkString.sprintfOverlappingData();
62             checkString.stringLiteralWrite();
63         }
64     
65         /** @brief undefined behaviour, writing string literal */
66         void stringLiteralWrite();
67     
68         /** @brief str plus char (unusual pointer arithmetic) */
69         void strPlusChar();
70     
71         /** @brief %Check for using bad usage of strncmp and substr */
72         void checkIncorrectStringCompare();
73     
74         /** @brief %Check for comparison of a string literal with a char* variable */
75         void checkSuspiciousStringCompare();
76     
77         /** @brief %Check for suspicious code that compares string literals for equality */
78         void checkAlwaysTrueOrFalseStringCompare();
79     
80         /** @brief %Check for overlapping source and destination passed to sprintf() */
81         void sprintfOverlappingData();
82     
83     private:
84         void stringLiteralWriteError(const Token *tok, const Token *strValue);
85         void sprintfOverlappingDataError(const Token *tok, const std::string &varname);
86         void strPlusCharError(const Token *tok);
87         void incorrectStringCompareError(const Token *tok, const std::string& func, const std::string &string);
88         void incorrectStringBooleanError(const Token *tok, const std::string& string);
89         void alwaysTrueFalseStringCompareError(const Token *tok, const std::string& str1, const std::string& str2);
90         void alwaysTrueStringVariableCompareError(const Token *tok, const std::string& str1, const std::string& str2);
91         void suspiciousStringCompareError(const Token* tok, const std::string& var);
92         void suspiciousStringCompareError_char(const Token* tok, const std::string& var);
93     
94         void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const {
95             CheckString c(0, settings, errorLogger);
96     
97             c.stringLiteralWriteError(0,0);
98             c.sprintfOverlappingDataError(0, "varname");
99             c.strPlusCharError(0);
100             c.incorrectStringCompareError(0, "substr", "\"Hello World\"");
101             c.suspiciousStringCompareError(0, "foo");
102             c.suspiciousStringCompareError_char(0, "foo");
103             c.incorrectStringBooleanError(0, "\"Hello World\"");
104             c.alwaysTrueFalseStringCompareError(0, "str1", "str2");
105             c.alwaysTrueStringVariableCompareError(0, "varname1", "varname2");
106         }
107     
108         static std::string myName() {
109             return "String";
110         }
111     
112         std::string classInfo() const {
113             return "Detect misusage of C-style strings:\n"
114                    "- overlapping buffers passed to sprintf as source and destination\n"
115                    "- incorrect length arguments for 'substr' and 'strncmp'\n"
116                    "- suspicious condition (runtime comparison of string literals)\n"
117                    "- suspicious condition (string literals as boolean)\n"
118                    "- suspicious comparison of a string literal with a char* variable\n"
119                    "- suspicious comparison of '\\0' with a char* variable\n";
120         }
121     };
122     /// @}
123     //---------------------------------------------------------------------------
124     #endif // checkstringH
