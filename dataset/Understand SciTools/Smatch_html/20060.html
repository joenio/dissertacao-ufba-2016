18     /*
19      * Record parameter types in the database.
20      *
21      */
22     
23     #include "smatch.h"
24     #include "smatch_slist.h"
25     
26     static int my_id;
27     
28     static void match_def(struct symbol *sym)
29     {
30     	struct symbol *arg;
31     	int i;
32     
33     	i = -1;
34     	FOR_EACH_PTR(sym->ctype.base_type->arguments, arg) {
35     		i++;
36     		sql_insert_function_type_info(i, type_to_str(get_real_base_type(arg)));
37     	} END_FOR_EACH_PTR(arg);
38     }
39     
40     void register_function_info(int id)
41     {
42     	my_id = id;
43     	add_hook(match_def, FUNC_DEF_HOOK);
44     }
