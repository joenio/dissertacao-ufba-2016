44     static void clean_up_symbols(struct symbol_list *list)
45     {
46     	struct symbol *sym;
47     
48     	FOR_EACH_PTR(list, sym) {
49     		expand_symbol(sym);
50     		emit_one_symbol(sym);
51     	} END_FOR_EACH_PTR(sym);
52     }
53     
54     int main(int argc, char **argv)
55     {
56     	char *file;
57     	struct string_list *filelist = NULL;
58     
59     	clean_up_symbols(sparse_initialize(argc, argv, &filelist));
60     	FOR_EACH_PTR_NOTAG(filelist, file) {
61     		struct symbol_list *list;
62     		const char *basename = strrchr(file, '/');
63     		basename = basename ?  basename+1 : file;
64     
65     		list = sparse(file);
66     
67     		// Do type evaluation and simplification
68     		emit_unit_begin(basename);
69     		clean_up_symbols(list);
70     		emit_unit_end();
71     	} END_FOR_EACH_PTR_NOTAG(file);

84     }
