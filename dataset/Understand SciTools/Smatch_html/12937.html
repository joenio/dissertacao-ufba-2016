24     int open_data_file(const char *filename)
25     {
26     	int fd;
27     	char buf[256];
28     
29     	fd = open(filename, O_RDONLY);
30     	if (fd >= 0)
31     		goto exit;
32     	if (!data_dir)
33     		goto exit;
34     	snprintf(buf, 256, "%s/%s", data_dir, filename);
35     	fd = open(buf, O_RDONLY);
36     exit:
37     	return fd;
38     }
39     
40     struct token *get_tokens_file(const char *filename)
41     {
42     	int fd;
43     	struct token *token;
44     
45     	if (option_no_data)
46     		return NULL;
47     	fd = open_data_file(filename);
48     	if (fd < 0)
49     		return NULL;
50     	token = tokenize(filename, fd, NULL, NULL);
51     	close(fd);
52     	return token;
53     }
