18     #include "smatch.h"
19     #include "smatch_extra.h"
20     
21     static int my_id;
22     
23     STATE(derefed);
24     
25     static void underef(struct sm_state *sm, struct expression *mod_expr)
26     {
27     	set_state(my_id, sm->name, sm->sym, &undefined);
28     }
29     
30     static void match_dereference(struct expression *expr)
31     {
32     	if (expr->type != EXPR_PREOP)
33     		return;
34     	expr = strip_expr(expr->unop);
35     	if (!is_pointer(expr))
36     		return;
37     	if (implied_not_equal(expr, 0))
38     		return;
39     
40     	if (is_impossible_path())
41     		return;
42     
43     	set_state_expr(my_id, expr, &derefed);
44     }
45     
46     static void set_param_dereferenced(struct expression *arg, char *key, char *unused)
47     {
48     	struct symbol *sym;
49     	char *name;
50     
51     	name = get_variable_from_key(arg, key, &sym);
52     	if (!name || !sym)
53     		goto free;
54     
55     	if (implied_not_equal_name_sym(name, sym, 0))
56     		return;
57     	set_state(my_id, name, sym, &derefed);
58     
59     free:
60     	free_string(name);
61     }
62     
63     static void match_condition(struct expression *expr)
64     {
65     	struct sm_state *sm;
66     
67     	if (__in_pre_condition)
68     		return;
69     
70     	if (get_macro_name(expr->pos))
71     		return;
72     
73     	if (!is_pointer(expr))
74     		return;
75     
76     	sm = get_sm_state_expr(my_id, expr);
77     	if (!sm || sm->state != &derefed)
78     		return;
79     
80     	sm_msg("warn: variable dereferenced before check '%s' (see line %d)", sm->name, sm->line);
81     	set_state_expr(my_id, expr, &undefined);
82     }
83     
84     void check_deref_check(int id)
85     {
86     	my_id = id;
87     	add_hook(&match_dereference, DEREF_HOOK);
88     	add_hook(&match_condition, CONDITION_HOOK);
89     	select_call_implies_hook(DEREFERENCE, &set_param_dereferenced);
90     	add_modification_hook(my_id, &underef);
91     }
