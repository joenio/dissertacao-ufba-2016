18     /*
19      * This is like check_deref_check.c except that it complains about code like:
20      * if (a)
21      *        a->foo = 42;
22      * a->bar = 7;
23      *
24      * Of course, Smatch has complained about these for forever but the problem is
25      * the old scripts were too messy and complicated and generated too many false
26      * positives.
27      *
28      * This check is supposed to be simpler because it only looks for one kind of 
29      * null dereference bug instead of every kind.  It also gets rid of the false 
30      * positives caused by the checks that happen inside macros.
31      *
32      */
33     
34     #include "smatch.h"
35     #include "smatch_slist.h"
36     #include "smatch_extra.h"
37     
38     static int my_id;
39     
40     STATE(null);
41     STATE(ok);
42     
43     static void is_ok(struct sm_state *sm, struct expression *mod_expr)
44     {
45     	set_state(my_id, sm->name, sm->sym, &ok);
46     }
47     
48     static void check_dereference(struct expression *expr)
49     {
50     	struct sm_state *sm;
51     	struct sm_state *tmp;
52     
53     	expr = strip_expr(expr);
54     
55     	sm = get_sm_state_expr(my_id, expr);
56     	if (!sm)
57     		return;
58     	if (is_ignored(my_id, sm->name, sm->sym))
59     		return;
60     	if (implied_not_equal(expr, 0))
61     		return;
62     
63     	FOR_EACH_PTR(sm->possible, tmp) {
64     		if (tmp->state == &merged)
65     			continue;
66     		if (tmp->state == &ok)
67     			continue;
68     		if (tmp->state == &null) {
69     			sm_msg("error: we previously assumed '%s' could be null (see line %d)",
70     			       tmp->name, tmp->line);
71     			add_ignore(my_id, sm->name, sm->sym);
72     			return;
73     		}
74     	} END_FOR_EACH_PTR(tmp);
75     }
76     
77     static void check_dereference_name_sym(char *name, struct symbol *sym)
78     {
79     	struct sm_state *sm;
80     	struct sm_state *tmp;
81     
82     	sm = get_sm_state(my_id, name, sym);
83     	if (!sm)
84     		return;
85     	if (is_ignored(my_id, sm->name, sm->sym))
86     		return;
87     	if (implied_not_equal_name_sym(name, sym, 0))
88     		return;
89     
90     	FOR_EACH_PTR(sm->possible, tmp) {
91     		if (tmp->state == &merged)
92     			continue;
93     		if (tmp->state == &ok)
94     			continue;
95     		if (tmp->state == &null) {
96     			sm_msg("error: we previously assumed '%s' could be null (see line %d)",
97     			       tmp->name, tmp->line);
98     			add_ignore(my_id, sm->name, sm->sym);
99     			return;
100     		}
101     	} END_FOR_EACH_PTR(tmp);
102     }
103     
104     static void match_dereferences(struct expression *expr)
105     {
106     	if (expr->type != EXPR_PREOP)
107     		return;
108     	check_dereference(expr->unop);
109     }
110     
111     static void match_pointer_as_array(struct expression *expr)
112     {
113     	if (!is_array(expr))
114     		return;
115     	check_dereference(get_array_base(expr));
116     }
117     
118     static void set_param_dereferenced(struct expression *arg, char *key, char *unused)
119     {
120     	struct symbol *sym;
121     	char *name;
122     
123     	name = get_variable_from_key(arg, key, &sym);
124     	if (!name || !sym)
125     		goto free;
126     
127     	check_dereference_name_sym(name, sym);
128     free:
129     	free_string(name);
130     }
131     
132     static void match_condition(struct expression *expr)
133     {
134     	if (get_macro_name(expr->pos))
135     		return;
136     
137     	if (!is_pointer(expr))
138     		return;
139     
140     	if (expr->type == EXPR_ASSIGNMENT) {
141     		match_condition(expr->right);
142     		match_condition(expr->left);
143     	}
144     	set_true_false_states_expr(my_id, expr, &ok, &null);
145     }
146     
147     void check_check_deref(int id)
148     {
149     	my_id = id;
150     
151     	add_modification_hook(my_id, &is_ok);
152     	add_hook(&match_dereferences, DEREF_HOOK);
153     	add_hook(&match_pointer_as_array, OP_HOOK);
154     	select_call_implies_hook(DEREFERENCE, &set_param_dereferenced);
155     	add_hook(&match_condition, CONDITION_HOOK);
156     }
