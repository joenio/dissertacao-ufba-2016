20     ALLOCATOR(var_sym, "var_sym structs");
21     
22     struct var_sym *alloc_var_sym(const char *var, struct symbol *sym)
23     {
24     	struct var_sym *tmp;
25     
26     	tmp = __alloc_var_sym(0);
27     	tmp->var = alloc_string(var);
28     	tmp->sym = sym;
29     	return tmp;
30     }
31     
32     struct var_sym_list *expr_to_vsl(struct expression *expr)
33     {
34     	struct var_sym_list *ret = NULL;
35     	char *var;
36     	struct symbol *sym;
37     
38     	expr = strip_expr(expr);
39     	if (!expr)
40     		return NULL;
41     
42     	if (expr->type == EXPR_PREOP && expr->op == '*')
43     		return expr_to_vsl(expr->unop);
44     
45     	if (expr->type == EXPR_BINOP ||
46     	    expr->type == EXPR_LOGICAL ||
47     	    expr->type == EXPR_COMPARE) {
48     		struct var_sym_list *left, *right;
49     
50     		left = expr_to_vsl(expr->left);
51     		right = expr_to_vsl(expr->right);
52     		ret = combine_var_sym_lists(left, right);
53     		free_var_syms_and_list(&left);
54     		free_var_syms_and_list(&right);
55     		return ret;
56     	}
57     	var = expr_to_var_sym(expr, &sym);
58     	if (!var || !sym) {
59     		free_string(var);
60     		return NULL;
61     	}
62     	add_var_sym(&ret, var, sym);
63     	return ret;
64     }
65     
66     void add_var_sym(struct var_sym_list **list, const char *var, struct symbol *sym)
67     {
68     	struct var_sym *tmp;
69     
70     	if (in_var_sym_list(*list, var, sym))
71     		return;
72     	tmp = alloc_var_sym(var, sym);
73     	add_ptr_list(list, tmp);
74     }
75     
76     void add_var_sym_expr(struct var_sym_list **list, struct expression *expr)
77     {
78     	char *var;
79     	struct symbol *sym;
80     
81     	var = expr_to_var_sym(expr, &sym);
82     	if (!var || !sym)
83     		goto free;
84     	add_var_sym(list, var, sym);
85     free:
86     	free_string(var);
87     }
88     
89     static void free_var_sym(struct var_sym *vs)
90     {
91     	free_string(vs->var);
92     	__free_var_sym(vs);
93     }
94     
95     void del_var_sym(struct var_sym_list **list, const char *var, struct symbol *sym)
96     {
97     	struct var_sym *tmp;
98     
99     	FOR_EACH_PTR(*list, tmp) {
100     		if (tmp->sym == sym && strcmp(tmp->var, var) == 0) {
101     			DELETE_CURRENT_PTR(tmp);
102     			free_var_sym(tmp);
103     			return;
104     		}
105     	} END_FOR_EACH_PTR(tmp);
106     }
107     
108     int in_var_sym_list(struct var_sym_list *list, const char *var, struct symbol *sym)
109     {
110     	struct var_sym *tmp;
111     
112     	FOR_EACH_PTR(list, tmp) {
113     		if (tmp->sym == sym && strcmp(tmp->var, var) == 0)
114     			return 1;
115     	} END_FOR_EACH_PTR(tmp);
116     	return 0;
117     }
118     
119     struct var_sym_list *clone_var_sym_list(struct var_sym_list *from_vsl)
120     {
121     	struct var_sym *tmp, *clone_vs;
122     	struct var_sym_list *to_vsl = NULL;
123     
124     	FOR_EACH_PTR(from_vsl, tmp) {
125     		clone_vs = alloc_var_sym(tmp->var, tmp->sym);
126     		add_ptr_list(&to_vsl, clone_vs);
127     	} END_FOR_EACH_PTR(tmp);
128     	return to_vsl;
129     }
130     
131     void merge_var_sym_list(struct var_sym_list **dest, struct var_sym_list *src)
132     {
133     	struct var_sym *tmp;
134     
135     	FOR_EACH_PTR(src, tmp) {
136     		add_var_sym(dest, tmp->var, tmp->sym);
137     	} END_FOR_EACH_PTR(tmp);
138     }
139     
140     struct var_sym_list *combine_var_sym_lists(struct var_sym_list *one, struct var_sym_list *two)
141     {
142     	struct var_sym_list *to_vsl;
143     
144     	to_vsl = clone_var_sym_list(one);
145     	merge_var_sym_list(&to_vsl, two);
146     	return to_vsl;
147     }
148     
149     void free_var_sym_list(struct var_sym_list **list)
150     {
151     	__free_ptr_list((struct ptr_list **)list);
152     }
153     
154     void free_var_syms_and_list(struct var_sym_list **list)
155     {
156     	struct var_sym *tmp;
157     
158     	FOR_EACH_PTR(*list, tmp) {
159     		free_var_sym(tmp);
160     	} END_FOR_EACH_PTR(tmp);
161     	free_var_sym_list(list);
162     }
