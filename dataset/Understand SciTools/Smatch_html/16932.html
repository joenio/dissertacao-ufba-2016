84     static int
85     skip_atoi(const char **s)
86     {
87     	int i = 0;
88     
89     	while (isdigit(**s))
90     		i = i*10 + *((*s)++) - '0';
91     
92     	return i;
93     }

95     static int
96     format_decode(const char *fmt, struct printf_spec *spec)
97     {
98     	const char *start = fmt;
99     	char qualifier;
100     
101     	/* we finished early by reading the field width */
102     	if (spec->type == FORMAT_TYPE_WIDTH) {
103     		if (spec->field_width < 0) {
104     			spec->field_width = -spec->field_width;
105     			spec->flags |= LEFT;
106     		}
107     		spec->type = FORMAT_TYPE_NONE;
108     		goto precision;
109     	}
110     
111     	/* we finished early by reading the precision */
112     	if (spec->type == FORMAT_TYPE_PRECISION) {
113     		if (spec->precision < 0)
114     			spec->precision = 0;
115     
116     		spec->type = FORMAT_TYPE_NONE;
117     		goto qualifier;
118     	}
119     
120     	/* By default */
121     	spec->type = FORMAT_TYPE_NONE;
122     
123     	for (; *fmt ; ++fmt) {
124     		if (*fmt == '%')
125     			break;
126     	}
127     
128     	/* Return the current non-format string */
129     	if (fmt != start || !*fmt)
130     		return fmt - start;
131     
132     	/* Process flags */
133     	spec->flags = 0;
134     
135     	while (1) { /* this also skips first '%' */
136     		bool found = true;
137     
138     		++fmt;
139     
140     		switch (*fmt) {
141     		case '-': spec->flags |= LEFT;    break;
142     		case '+': spec->flags |= PLUS;    break;
143     		case ' ': spec->flags |= SPACE;   break;
144     		case '#': spec->flags |= SPECIAL; break;
145     		case '0': spec->flags |= ZEROPAD; break;
146     		default:  found = false;
147     		}
148     
149     		if (!found)
150     			break;
151     	}
152     
153     	/* get field width */
154     	spec->field_width = -1;
155     
156     	if (isdigit(*fmt))
157     		spec->field_width = skip_atoi(&fmt);
158     	else if (*fmt == '*') {
159     		/* it's the next argument */
160     		spec->type = FORMAT_TYPE_WIDTH;
161     		return ++fmt - start;
162     	}
163     
164     precision:
165     	/* get the precision */
166     	spec->precision = -1;
167     	if (*fmt == '.') {
168     		++fmt;
169     		if (isdigit(*fmt)) {
170     			spec->precision = skip_atoi(&fmt);
171     			if (spec->precision < 0)
172     				spec->precision = 0;
173     		} else if (*fmt == '*') {
174     			/* it's the next argument */
175     			spec->type = FORMAT_TYPE_PRECISION;
176     			return ++fmt - start;
177     		}
178     	}
179     
180     qualifier:
181     	/* get the conversion qualifier */
182     	qualifier = 0;
183     	if (*fmt == 'h' || _tolower(*fmt) == 'l' ||
184     	    _tolower(*fmt) == 'z' || *fmt == 't') {
185     		qualifier = *fmt++;
186     		if (qualifier == *fmt) {
187     			if (qualifier == 'l') {
188     				qualifier = 'L';
189     				++fmt;
190     			} else if (qualifier == 'h') {
191     				qualifier = 'H';
192     				++fmt;
193     			} else {
194     				sm_msg("warn: invalid repeated qualifier '%c'", *fmt);
195     			}
196     		}
197     	}
198     
199     	/* default base */
200     	spec->base = 10;
201     	switch (*fmt) {
202     	case 'c':
203     		if (qualifier)
204     			sm_msg("warn: qualifier '%c' ignored for %%c specifier", qualifier);
205     
206     		spec->type = FORMAT_TYPE_CHAR;
207     		return ++fmt - start;
208     
209     	case 's':
210     		if (qualifier)
211     			sm_msg("warn: qualifier '%c' ignored for %%s specifier", qualifier);
212     
213     		spec->type = FORMAT_TYPE_STR;
214     		return ++fmt - start;
215     
216     	case 'p':
217     		spec->type = FORMAT_TYPE_PTR;
218     		return ++fmt - start;
219     
220     	case '%':
221     		spec->type = FORMAT_TYPE_PERCENT_CHAR;
222     		return ++fmt - start;
223     
224     	/* integer number formats - set up the flags and "break" */
225     	case 'o':
226     		spec->base = 8;
227     		break;
228     
229     	case 'x':
230     		spec->flags |= SMALL;
231     
232     	case 'X':
233     		spec->base = 16;
234     		break;
235     
236     	case 'd':
237     	case 'i':
238     		spec->flags |= SIGN;
239     	case 'u':
240     		break;
241     
242     	case 'n':
243     		spec->type = FORMAT_TYPE_NRCHARS;
244     		return ++fmt - start;
245     
246     	case 'a': case 'A':
247     	case 'e': case 'E':
248     	case 'f': case 'F':
249     	case 'g': case 'G':
250     		spec->type = FORMAT_TYPE_FLOAT;
251     		return ++fmt - start;
252     
253     	default:
254     		spec->type = FORMAT_TYPE_INVALID;
255     		/* Unlike the kernel code, we 'consume' the invalid
256     		 * character so that it can get included in the
257     		 * report. After that, we bail out. */
258     		return ++fmt - start;
259     	}
260     
261     	if (qualifier == 'L')
262     		spec->type = FORMAT_TYPE_LONG_LONG;
263     	else if (qualifier == 'l') {
264     		if (spec->flags & SIGN)
265     			spec->type = FORMAT_TYPE_LONG;
266     		else
267     			spec->type = FORMAT_TYPE_ULONG;
268     	} else if (_tolower(qualifier) == 'z') {
269     		spec->type = FORMAT_TYPE_SIZE_T;
270     	} else if (qualifier == 't') {
271     		spec->type = FORMAT_TYPE_PTRDIFF;
272     	} else if (qualifier == 'H') {
273     		if (spec->flags & SIGN)
274     			spec->type = FORMAT_TYPE_BYTE;
275     		else
276     			spec->type = FORMAT_TYPE_UBYTE;
277     	} else if (qualifier == 'h') {
278     		if (spec->flags & SIGN)
279     			spec->type = FORMAT_TYPE_SHORT;
280     		else
281     			spec->type = FORMAT_TYPE_USHORT;
282     	} else {
283     		if (spec->flags & SIGN)
284     			spec->type = FORMAT_TYPE_INT;
285     		else
286     			spec->type = FORMAT_TYPE_UINT;
287     	}
288     
289     	return ++fmt - start;
290     }

292     static int is_struct_tag(struct symbol *type, const char *tag)
293     {
294     	return type->type == SYM_STRUCT && type->ident && !strcmp(type->ident->name, tag);

297     static int has_struct_tag(struct symbol *type, const char *tag)
298     {
299     	struct symbol *tmp;
300     
301     	if (type->type == SYM_STRUCT)
302     		return is_struct_tag(type, tag);
303     	if (type->type == SYM_UNION) {
304     		FOR_EACH_PTR(type->symbol_list, tmp) {
305     			tmp = get_real_base_type(tmp);
306     			if (tmp && is_struct_tag(tmp, tag))
307     				return 1;
308     		} END_FOR_EACH_PTR(tmp);
309     	}
310     	return 0;
311     }

328     static struct symbol *_typedef_lookup(const char *name)
329     {
330     	struct ident *id;
331     	struct symbol *node;
332     
333     	id = built_in_ident(name);
334     	if (!id)
335     		return NULL;
336     	node = lookup_symbol(id, NS_TYPEDEF);
337     	if (!node || node->type != SYM_NODE)
338     		return NULL;
339     	return node->ctype.base_type;
340     }

342     static void typedef_lookup(struct typedef_lookup *tl)
343     {
344     	if (tl->sym || tl->failed)
345     		return;
346     	tl->sym = _typedef_lookup(tl->name);
347     	if (!tl->sym) {
348     		sm_msg("internal error: could not find typedef '%s'", tl->name);
349     		tl->failed = 1;
350     	}

354     static void ip4(const char *fmt, struct symbol *type, struct symbol *basetype, int vaidx)
355     {
356     	enum { ENDIAN_BIG, ENDIAN_LITTLE, ENDIAN_HOST } endian = ENDIAN_BIG;
357     
358     	assert(fmt[0] == 'i' || fmt[0] == 'I');
359     	assert(fmt[1] == '4');
360     
361     	if (isalnum(fmt[2])) {
362     		switch (fmt[2]) {
363     		case 'h':
364     			endian = ENDIAN_HOST;
365     			break;
366     		case 'l':
367     			endian = ENDIAN_LITTLE;
368     			break;
369     		case 'n':
370     		case 'b':
371     			endian = ENDIAN_BIG;
372     			break;
373     		default:
374     			sm_msg("warn: '%%p%c4' can only be followed by one of [hnbl], not '%c'", fmt[0], fmt[2]);
375     		}
376     		if (isalnum(fmt[3]))
377     			sm_msg("warn: '%%p%c4' can only be followed by precisely one of [hnbl]", fmt[0]);
378     	}
379     
380     
381     	if (type->ctype.modifiers & MOD_NODEREF)
382     		sm_msg("error: passing __user pointer to '%%p%c4'", fmt[0]);
383     
384     	/*
385     	 * If we have a pointer to char/u8/s8, we expect the caller to
386     	 * handle endianness; I don't think there's anything we can
387     	 * do. I'd like to check that if we're passed a pointer to a
388     	 * __bitwise u32 (most likely a __be32), we should have endian
389     	 * == ENDIAN_BIG. But I can't figure out how to get that
390     	 * information (it also seems to require ensuring certain
391     	 * macros are defined). But struct in_addr certainly consists
392     	 * of only a single __be32, so in that case we can do a check.
393     	 */
394     	if (is_char_type(basetype))
395     		return;
396     
397     	if (is_struct_tag(basetype, "in_addr") && endian != ENDIAN_BIG)
398     		sm_msg("warn: passing struct in_addr* to '%%p%c4%c', is the endianness ok?", fmt[0], fmt[2]);
399     
400     	/* ... */
401     }

421     static void ipS(const char *fmt, struct symbol *type, struct symbol *basetype, int vaidx)
422     {
423     	const char *f;
424     
425     	assert(tolower(fmt[0]) == 'i');
426     	assert(fmt[1] == 'S');
427     
428     	for (f = fmt+2; isalnum(*f); ++f) {
429     		/* It's probably too anal checking for duplicate flags. */
430     		if (!strchr("pfschnbl", *f))
431     			sm_msg("warn: '%%p%cS' cannot be followed by '%c'", fmt[0], *f);
432     	}
433     
434     	/*
435     	 * XXX: Should we also allow passing a pointer to a union, one
436     	 * member of which is a struct sockaddr? It may be slightly
437     	 * cleaner actually passing &u.raw instead of just &u, though
438     	 * the generated code is of course exactly the same. For now,
439     	 * we do accept struct sockaddr_in and struct sockaddr_in6,
440     	 * since those are easy to handle and rather harmless.
441     	 */
442     	if (!has_struct_tag(basetype, "sockaddr") &&
443     	    !has_struct_tag(basetype, "sockaddr_in") &&
444     	    !has_struct_tag(basetype, "sockaddr_in6") &&
445     	    !has_struct_tag(basetype, "__kernel_sockaddr_storage"))
446     		sm_msg("error: '%%p%cS' expects argument of type struct sockaddr *, "
447     			"argument %d has type '%s'", fmt[0], vaidx, type_to_str(type));
448     }

567     static void address_val(const char *fmt, struct symbol *type, struct symbol *basetype, int vaidx)
568     {
569     	static struct typedef_lookup dma = { .name = "dma_addr_t" };
570     	static struct typedef_lookup phys = { .name = "phys_addr_t" };
571     	struct typedef_lookup *which = &phys;
572     	const char *suf = "";
573     	assert(fmt[0] == 'a');
574     
575     	if (isalnum(fmt[1])) {
576     		switch (fmt[1]) {
577     		case 'd':
578     			which = &dma;
579     			suf = "d";
580     			break;
581     		case 'p':
582     			suf = "p";
583     			break;
584     		default:
585     			sm_msg("error: '%%pa' can only be followed by one of [dp]");
586     		}
587     		if (isalnum(fmt[2]))
588     			sm_msg("error: '%%pa%c' cannot be followed by '%c'", fmt[1], fmt[2]);
589     	}
590     
591     	typedef_lookup(which);
592     	if (!which->sym)
593     		return;
594     	if (basetype != which->sym) {
595     		sm_msg("error: '%%pa%s' expects argument of type '%s*', argument %d has type '%s'",
596     			suf, which->name, vaidx, type_to_str(type));
597     	}

600     static void flag_string(const char *fmt, struct symbol *type, struct symbol *basetype, int vaidx)
601     {
602     	static struct typedef_lookup gfp = { .name = "gfp_t" };
603     
604     	assert(fmt[0] == 'g');
605     	if (!isalnum(fmt[1])) {
606     		sm_msg("error: %%pg must be followed by one of [gpv]");
607     		return;
608     	}
609     	switch (fmt[1]) {
610     	case 'p':
611     	case 'v':
612     		if (basetype != &ulong_ctype)
613     			sm_msg("error: '%%pg%c' expects argument of type 'unsigned long *', argument %d has type '%s'",

615     		break;
616     	case 'g':
617     		typedef_lookup(&gfp);
618     		if (basetype != gfp.sym)
619     			sm_msg("error: '%%pgg' expects argument of type 'gfp_t *', argument %d has type '%s'",
620     				vaidx, type_to_str(type));
621     		break;
622     	default:
623     		sm_msg("error: '%%pg' must be followed by one of [gpv]");
624     	}

628     static void
629     pointer(const char *fmt, struct expression *arg, int vaidx)
630     {
631     	struct symbol *type, *basetype;
632     
633     	type = get_type(arg);
634     	if (!type) {
635     		sm_msg("warn: could not determine type of argument %d", vaidx);
636     		return;
637     	}
638     	if (!is_ptr_type(type)) {
639     		sm_msg("error: %%p expects pointer argument, but argument %d has type '%s'",
640     			vaidx, type_to_str(type));
641     		return;
642     	}
643     	/* Just plain %p, nothing to check. */
644     	if (!isalnum(*fmt))
645     		return;
646     
647     	basetype = get_real_base_type(type);
648     	/*
649     	 * Passing a pointer-to-array is harmless, but most likely one
650     	 * meant to pass pointer-to-first-element. If basetype is
651     	 * array type, we issue a notice and "dereference" the types
652     	 * once more.
653     	 */
654     	if (basetype->type == SYM_ARRAY) {
655     		spam("note: passing pointer-to-array; is the address-of redundant?");
656     		type = basetype;
657     		basetype = get_real_base_type(type);
658     	}
659     
660     	/*
661     	 * We pass both the type and the basetype to the helpers. If,
662     	 * for example, the pointer is really a decayed array which is
663     	 * passed to %pI4, we might want to check that it is in fact
664     	 * an array of four bytes. But most are probably only
665     	 * interested in whether the basetype makes sense. Also, the
666     	 * pointer may carry some annotation such as __user which
667     	 * might be worth checking in the handlers which actually
668     	 * dereference the pointer.
669     	 */
670     
671     	switch (*fmt) {
672     	case 'b':
673     	case 'F':
674     	case 'f':
675     	case 'S':
676     	case 's':
677     	case 'B':
678     		/* Can we do anything sensible? Check that the arg is a function pointer, for example? */
679     		break;
680     
681     	case 'R':
682     	case 'r':
683     		resource_string(fmt, type, basetype, vaidx);
684     		break;
685     	case 'M':
686     	case 'm':
687     		mac_address_string(fmt, type, basetype, vaidx);
688     		break;
689     	case 'I':
690     	case 'i':
691     		switch (fmt[1]) {
692     		case '4':
693     			ip4(fmt, type, basetype, vaidx);
694     			break;
695     		case '6':
696     			ip6(fmt, type, basetype, vaidx);
697     			break;
698     		case 'S':
699     			ipS(fmt, type, basetype, vaidx);
700     			break;
701     		default:
702     			sm_msg("warn: '%%p%c' must be followed by one of [46S]", fmt[0]);
703     			break;
704     		}
705     		break;
706            /*
707     	* %pE and %ph can handle any valid pointer. We still check
708     	* whether all the subsequent alphanumerics are valid for the
709     	* particular %pX conversion.
710     	*/
711     	case 'E':
712     		escaped_string(fmt, type, basetype, vaidx);
713     		break;
714     	case 'h':
715     		hex_string(fmt, type, basetype, vaidx);
716     		break;
717     	case 'U': /* TODO */
718     		break;
719     	case 'V':
720     		va_format(fmt, type, basetype, vaidx);
721     		break;
722     	case 'K': /* TODO */
723     		break;
724     	case 'N':
725     		netdev_feature(fmt, type, basetype, vaidx);
726     		break;
727     	case 'a':
728     		address_val(fmt, type, basetype, vaidx);
729     		break;
730     	case 'D':
731     	case 'd':
732     		dentry_file(fmt, type, basetype, vaidx);
733     		break;
734     	case 'C':
735     		clock(fmt, type, basetype, vaidx);
736     		break;
737     	case 'j':
738     		flag_string(fmt, type, basetype, vaidx);
739     		break;
740     	default:
741     		sm_msg("error: unrecognized %%p extension '%c', treated as normal %%p", *fmt);
742     	}
743     }

745     /*
746      * A common error is to pass a "char" or "signed char" to %02x (or
747      * %.2X or some other variant). This can actually be a security
748      * problem, because a lot of code expects this to produce exactly two
749      * characters of output. Unfortunately this also produces false
750      * positives, since we're sometimes in arch-specific code on an arch
751      * where char is always unsigned.
752      */
753     static void
754     hexbyte(const char *fmt, int fmt_len, struct expression *arg, int vaidx, struct printf_spec spec)
755     {
756     	struct symbol *type;
757     
758     	/*
759     	 * For now, just check the most common and obvious, which is
760     	 * roughly %[.0]2[xX].
761     	 */
762     	if (spec.field_width != 2 && spec.precision != 2)
763     		return;
764     	if (spec.base != 16)
765     		return;
766     
767     	type = get_type(arg);
768     	if (!type) {
769     		sm_msg("warn: could not determine type of argument %d", vaidx);
770     		return;
771     	}
772     	if (type == &char_ctype || type == &schar_ctype)
773     		sm_msg("warn: argument %d to %.*s specifier has type '%s'",
774     		       vaidx, fmt_len, fmt, type_to_str(type));
775     }

777     static int
778     check_format_string(const char *fmt, const char *caller)
779     {
780     	const char *f;
781     
782     	for (f = fmt; *f; ++f) {
783     		unsigned char c = *f;
784     		switch (c) {
785     		case KERN_SOH_ASCII:
786     			/*
787     			 * This typically arises from bad conversion
788     			 * to pr_*, e.g. pr_warn(KERN_WARNING "something").
789     			 */
790     			if (f != fmt)
791     				sm_msg("warn: KERN_* level not at start of string");
792     			/*
793     			 * In a very few cases, the level is actually
794     			 * computed and passed via %c, as in KERN_SOH
795     			 * "%c...". printk explicitly supports
796     			 * this.
797     			 */
798     			if (!(('0' <= f[1] && f[1] <= '7') ||
799     			      f[1] == 'd' ||
800     			      (f[1] == '%' && f[2] == 'c')))
801     				sm_msg("warn: invalid KERN_* level: KERN_SOH_ASCII followed by '\\x%02x'", (unsigned char)f[1]);
802     			break;
803     		case '\t':
804     		case '\n':
805     		case '\r':
806     		case 0x20 ... 0x7e:
807     			break;
808     		case 0x80 ... 0xff:
809     			sm_msg("warn: format string contains non-ascii character '\\x%02x'", c);
810     			break;
811     		case 0x08:
812     			if (f == fmt)
813     				break;
814     			/* fall through */
815     		default:
816     			sm_msg("warn: format string contains unusual character '\\x%02x'", c);
817     			break;
818     		}
819     	}
820     
821     	f = strstr(fmt, caller);
822     	if (f && strstr(f+1, caller))
823     		sm_msg("note: format string contains name of enclosing function '%s' twice", caller);
824     
825     	return f != NULL;
826     }

843     static int is_array_of_const_char(struct symbol *sym)
844     {
845     	struct symbol *base = sym->ctype.base_type;
846     	if (base->type != SYM_ARRAY)
847     		return 0;
848     	if (!(base->ctype.modifiers & MOD_CONST))
849     		return 0;
850     	if (!is_char_type(base->ctype.base_type)) {
851     		spam("weird: format argument is array of const '%s'", type_to_str(base->ctype.base_type));
852     		return 0;
853     	}
854     	return 1;
855     }

857     static int is_const_pointer_to_const_char(struct symbol *sym)
858     {
859     	struct symbol *base = sym->ctype.base_type;
860     	if (!(sym->ctype.modifiers & MOD_CONST))
861     		return 0;
862     	if (base->type != SYM_PTR)
863     		return 0;
864     	if (!(base->ctype.modifiers & MOD_CONST))
865     		return 0;
866     	if (!is_char_type(base->ctype.base_type)) {
867     		spam("weird: format argument is pointer to const '%s'", type_to_str(base->ctype.base_type));
868     		return 0;
869     	}
870     	return 1;
871     }

913     static void
914     do_check_printf_call(const char *caller, const char *name, struct expression *callexpr, struct expression *fmtexpr, int vaidx)
915     {
916     	struct printf_spec spec = {0};
917     	const char *fmt, *orig_fmt;
918     	int caller_in_fmt;
919     
920     	fmtexpr = strip_parens(fmtexpr);
921     	if (fmtexpr->type == EXPR_CONDITIONAL) {
922     		do_check_printf_call(caller, name, callexpr, fmtexpr->cond_true ? : fmtexpr->conditional, vaidx);
923     		do_check_printf_call(caller, name, callexpr, fmtexpr->cond_false, vaidx);
924     		return;
925     	}
926     	if (fmtexpr->type == EXPR_SYMBOL) {
927     		/*
928     		 * If the symbol has an initializer, we can handle
929     		 *
930     		 *   const char foo[] = "abc";         and
931     		 *   const char * const foo = "abc";
932     		 *
933     		 * We simply replace fmtexpr with the initializer
934     		 * expression. If foo is not one of the above, or if
935     		 * the initializer expression is somehow not a string
936     		 * literal, fmtexpr->type != EXPR_STRING will trigger
937     		 * below and we'll spam+return.
938     		 */
939     		struct symbol *sym = fmtexpr->symbol;
940     		if (sym->initializer &&
941     		    (is_array_of_const_char(sym) ||
942     		     is_const_pointer_to_const_char(sym))) {
943     			fmtexpr = strip_parens(sym->initializer);
944     		}
945     	}
946     
947     	if (fmtexpr->type != EXPR_STRING) {
948     		if (!unknown_format(fmtexpr))
949     			return;
950     		/*
951     		 * Since we're now handling both ?: and static const
952     		 * char[] arguments, we don't get as much noise. It's
953     		 * still spammy, though.
954     		 */
955     		spam("warn: call of '%s' with non-constant format argument", name);
956     		return;
957     	}
958     
959     	orig_fmt = fmt = fmtexpr->string->data;
960     	caller_in_fmt = check_format_string(fmt, caller);
961     
962     	while (*fmt) {
963     		const char *old_fmt = fmt;
964     		int read = format_decode(fmt, &spec);
965     		struct expression *arg;
966     
967     		fmt += read;
968     		if (spec.type == FORMAT_TYPE_NONE ||
969     		    spec.type == FORMAT_TYPE_PERCENT_CHAR)
970     			continue;
971     
972     		/*
973     		 * vaidx is currently the correct 0-based index for
974     		 * get_argument_from_call_expr. We post-increment it
975     		 * here so that it is the correct 1-based index for
976     		 * all the handlers below. This of course requires
977     		 * that we handle all FORMAT_TYPE_* things not taking
978     		 * an argument above.
979     		 */
980     		arg = get_argument_from_call_expr(callexpr->args, vaidx++);
981     
982     		if (spec.flags & SPECIAL && has_hex_prefix(orig_fmt, old_fmt))
983     			sm_msg("warn: '%.2s' prefix is redundant when # flag is used", old_fmt-2);
984     		if (is_integer_specifier(spec.type) && spec.base != 16 && has_hex_prefix(orig_fmt, old_fmt))
985     			sm_msg("warn: '%.2s' prefix is confusing together with '%.*s' specifier",
986     			       old_fmt-2, (int)(fmt-old_fmt), old_fmt);
987     
988     		switch (spec.type) {
989     		/* case FORMAT_TYPE_NONE: */
990     		/* case FORMAT_TYPE_PERCENT_CHAR: */
991     		/* 	break; */
992     
993     		case FORMAT_TYPE_INVALID:
994     			sm_msg("error: format specifier '%.*s' invalid", read, old_fmt);
995     			return;
996     
997     		case FORMAT_TYPE_FLOAT:
998     			sm_msg("error: no floats in the kernel; invalid format specifier '%.*s'", read, old_fmt);
999     			return;
1000     
1001     		case FORMAT_TYPE_NRCHARS:
1002     			sm_msg("error: %%n not supported in kernel");
1003     			return;
1004     
1005     		case FORMAT_TYPE_WIDTH:
1006     		case FORMAT_TYPE_PRECISION:
1007     			/* check int argument */
1008     			break;
1009     
1010     		case FORMAT_TYPE_STR:
1011     			/*
1012     			 * If the format string already contains the
1013     			 * function name, it probably doesn't make
1014     			 * sense to pass __func__ as well (or rather
1015     			 * vice versa: If pr_fmt(fmt) has been defined
1016     			 * to '"%s: " fmt, __func__', it doesn't make
1017     			 * sense to use a format string containing the
1018     			 * function name).
1019     			 *
1020     			 * This produces a lot of hits. They are not
1021     			 * false positives, but it is easier to handle
1022     			 * the things which don't occur that often
1023     			 * first, so we use spam().
1024     			 */
1025     			if (caller_in_fmt) {
1026     				if (arg_is___func__(arg))
1027     					spam("warn: passing __func__ while the format string already contains the name of the function '%s'",
1028     					     caller);
1029     				else if (arg_contains_caller(arg, caller))
1030     					sm_msg("warn: passing string constant '%s' containing '%s' which is already part of the format string",
1031     					       arg->string->data, caller);
1032     			}
1033     			break;
1034     
1035     		case FORMAT_TYPE_PTR:
1036     			/* This is the most important part: Checking %p extensions. */
1037     			pointer(fmt, arg, vaidx);
1038     			while (isalnum(*fmt))
1039     				fmt++;
1040     			break;
1041     
1042     		case FORMAT_TYPE_CHAR:
1043     
1044     		case FORMAT_TYPE_UBYTE:
1045     		case FORMAT_TYPE_BYTE:
1046     		case FORMAT_TYPE_USHORT:
1047     		case FORMAT_TYPE_SHORT:
1048     		case FORMAT_TYPE_INT:
1049     			/* argument should have integer type of width <= sizeof(int) */
1050     			break;
1051     
1052     		case FORMAT_TYPE_UINT:
1053     			hexbyte(old_fmt, fmt-old_fmt, arg, vaidx, spec);
1054     		case FORMAT_TYPE_LONG:
1055     		case FORMAT_TYPE_ULONG:
1056     		case FORMAT_TYPE_LONG_LONG:
1057     		case FORMAT_TYPE_PTRDIFF:
1058     		case FORMAT_TYPE_SIZE_T:
1059     			break;
1060     		}
1061     
1062     
1063     	}
1064     
1065     	if (get_argument_from_call_expr(callexpr->args, vaidx))
1066     		sm_msg("warn: excess argument passed to '%s'", name);
1067     
1068     
1069     }

1071     static void
1072     check_printf_call(const char *name, struct expression *callexpr, void *_info)
1073     {
1074     	/*
1075     	 * Note: attribute(printf) uses 1-based indexing, but
1076     	 * get_argument_from_call_expr() uses 0-based indexing.
1077     	 */
1078     	int info = PTR_INT(_info);
1079     	int fmtidx = (info & 0xff) - 1;
1080     	int vaidx = ((info >> 8) & 0xff) - 1;
1081     	struct expression *fmtexpr;
1082     	const char *caller = get_function();
1083     
1084     	/*
1085     	 * Calling a v*printf function with a literal format arg is
1086     	 * extremely rare, so we don't bother doing the only checking
1087     	 * we could do, namely checking that the format string is
1088     	 * valid.
1089     	 */
1090     	if (vaidx < 0)
1091     		return;
1092     
1093     	/*
1094     	 * For the things we use the name of the calling function for,
1095     	 * it is more appropriate to skip a potential SyS_ prefix; the
1096     	 * same goes for leading underscores.
1097     	 */
1098     	if (!strncmp(caller, "SyS_", 4))
1099     		caller += 4;
1100     	while (*caller == '_')
1101     		++caller;
1102     
1103     	/* Lack of format argument is a bug. */
1104     	fmtexpr = get_argument_from_call_expr(callexpr->args, fmtidx);
1105     	if (!fmtexpr) {
1106     		sm_msg("error: call of '%s' with no format argument", name);
1107     		return;
1108     	}
1109     
1110     	do_check_printf_call(caller, name, callexpr, fmtexpr, vaidx);
1111     }
