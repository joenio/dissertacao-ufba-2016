18     #include "smatch.h"
19     
20     static int my_id;
21     
22     static void match_condition(struct expression *expr)
23     {
24     	sval_t val;
25     
26     	if (expr->type != EXPR_ASSIGNMENT || expr->op != '=')
27     		return;
28     
29     	if (!get_value(expr->right, &val))
30     		return;
31     	sm_msg("warn: was '== %s' instead of '='", sval_to_str(val));
32     }
33     
34     void check_assign_vs_compare(int id)
35     {
36     	my_id = id;
37     	add_hook(&match_condition, CONDITION_HOOK);
38     }
