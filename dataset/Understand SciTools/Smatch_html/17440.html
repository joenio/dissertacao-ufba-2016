18     /*
19      * This is almost the same as smatch_param_filter.c.  The difference is that
20      * this only deals with values passed on the stack and param filter only deals
21      * with values changed so that the caller sees the new value.  It other words
22      * the key for these should always be "$" and the key for param_filter should
23      * never be "$".  Also smatch_param_set() should never use "$" as the key.
24      * Param set should work together with param_filter to determine the value that
25      * the caller sees at the end.
26      *
27      * This is for functions like this:
28      *
29      * int foo(int a)
30      * {
31      *        if (a >= 0 && a < 10) {
32      *                 a = 42;
33      *                 return 1;
34      *        }
35      *        return 0;
36      * }
37      *
38      * If we pass in 5, it returns 1.
39      *
40      * It's a bit complicated because we can't just consider the final value, we
41      * have to always consider the passed in value.
42      *
43      */

50     static int my_id;
51     
52     static struct stree *start_states;
53     static struct stree_stack *saved_stack;
54     
55     static void save_start_states(struct statement *stmt)
56     {
57     	start_states = get_all_states_stree(SMATCH_EXTRA);
58     }
59     
60     static void free_start_states(void)
61     {
62     	free_stree(&start_states);
63     }

65     static struct smatch_state *unmatched_state(struct sm_state *sm)
66     {
67     	struct smatch_state *state;
68     
69     	state = get_state(SMATCH_EXTRA, sm->name, sm->sym);
70     	if (state)
71     		return state;
72     	return alloc_estate_whole(estate_type(sm->state));
73     }
74     
75     struct smatch_state *get_orig_estate(const char *name, struct symbol *sym)
76     {
77     	struct smatch_state *state;
78     
79     	state = get_state(my_id, name, sym);
80     	if (state)
81     		return state;
82     
83     	state = get_state(SMATCH_EXTRA, name, sym);
84     	if (state)
85     		return state;
86     	return alloc_estate_rl(alloc_whole_rl(get_real_base_type(sym)));
87     }
88     
89     struct smatch_state *get_orig_estate_type(const char *name, struct symbol *sym, struct symbol *type)
90     {
91     	struct smatch_state *state;
92     
93     	state = get_state(my_id, name, sym);
94     	if (state)
95     		return state;
96     
97     	state = get_state(SMATCH_EXTRA, name, sym);
98     	if (state)
99     		return state;
100     	return alloc_estate_rl(alloc_whole_rl(type));
101     }

103     static void print_return_value_param(int return_id, char *return_ranges, struct expression *expr)
104     {
105     	struct smatch_state *state, *old;
106     	struct sm_state *tmp;
107     	const char *param_name;
108     	int param;
109     
110     	FOR_EACH_MY_SM(SMATCH_EXTRA, __get_cur_stree(), tmp) {
111     		param = get_param_num_from_sym(tmp->sym);
112     		if (param < 0)
113     			continue;
114     
115     		param_name = get_param_name(tmp);
116     		if (!param_name)
117     			continue;
118     
119     		state = get_state(my_id, tmp->name, tmp->sym);
120     		if (!state)
121     			state = tmp->state;
122     
123     		if (estate_is_whole(state))
124     			continue;
125     		old = get_state_stree(start_states, SMATCH_EXTRA, tmp->name, tmp->sym);
126     		if (old && estates_equiv(old, state))
127     			continue;
128     
129     		sql_insert_return_states(return_id, return_ranges, PARAM_LIMIT,
130     					 param, param_name, state->name);
131     	} END_FOR_EACH_SM(tmp);
132     }

134     static void extra_mod_hook(const char *name, struct symbol *sym, struct smatch_state *state)
135     {
136     	struct smatch_state *orig_vals;
137     	int param;
138     
139     	param = get_param_num_from_sym(sym);
140     	if (param < 0)
141     		return;
142     
143     	orig_vals = get_orig_estate_type(name, sym, estate_type(state));
144     	set_state(my_id, name, sym, orig_vals);
145     }
146     
147     static void match_save_states(struct expression *expr)
148     {
149     	push_stree(&saved_stack, start_states);
150     	start_states = NULL;
151     }
152     
153     static void match_restore_states(struct expression *expr)
154     {
155     	free_stree(&start_states);
156     	start_states = pop_stree(&saved_stack);
157     }
158     
159     void register_param_limit(int id)
160     {
161     	my_id = id;
162     
163     	add_hook(&save_start_states, AFTER_DEF_HOOK);
164     	add_hook(&free_start_states, END_FUNC_HOOK);
165     
166     	add_extra_mod_hook(&extra_mod_hook);
167     	add_unmatched_state_hook(my_id, &unmatched_state);
168     	add_merge_hook(my_id, &merge_estates);
169     
170     	add_hook(&match_save_states, INLINE_FN_START);
171     	add_hook(&match_restore_states, INLINE_FN_END);
172     
173     	add_split_return_callback(&print_return_value_param);
174     }
