22     static int my_id;
23     
24     STATE(err_ptr);
25     STATE(checked);
26     
27     sval_t err_ptr_min = {
28     	.type = &int_ctype,
29     	{.value = -4095},
30     };
31     
32     sval_t err_ptr_max = {
33     	.type = &int_ctype,
34     	{.value = -1},
35     };
36     
37     struct range_list *err_ptr_rl;
38     
39     static void ok_to_use(struct sm_state *sm, struct expression *mod_expr)
40     {
41     	if (sm->state != &checked)
42     		set_state(my_id, sm->name, sm->sym, &checked);
43     }
44     
45     static void check_is_err_ptr(struct expression *expr)
46     {
47     	struct sm_state *sm;
48     	struct range_list *rl;
49     
50     	sm = get_sm_state_expr(my_id, expr);
51     	if (!sm)
52     		return;
53     
54     	if (!slist_has_state(sm->possible, &err_ptr))
55     		return;
56     
57     	get_absolute_rl(expr, &rl);
58     	if (!possibly_true_rl(rl, SPECIAL_EQUAL, err_ptr_rl))
59     		return;
60     
61     	sm_msg("error: '%s' dereferencing possible ERR_PTR()", sm->name);
62     	set_state(my_id, sm->name, sm->sym, &checked);
63     }
64     
65     static void match_returns_err_ptr(const char *fn, struct expression *expr,
66     				void *info)
67     {
68     	set_state_expr(my_id, expr->left, &err_ptr);
69     }

72     static void match_checked(const char *fn, struct expression *call_expr,
73     			struct expression *assign_expr, void *unused)
74     {
75     	struct expression *arg;
76     
77     	arg = get_argument_from_call_expr(call_expr->args, 0);
78     	arg = strip_expr(arg);
79     	while (arg->type == EXPR_ASSIGNMENT)
80     		arg = strip_expr(arg->left);
81     	set_state_expr(my_id, arg, &checked);
82     }
83     
84     static void match_err(const char *fn, struct expression *call_expr,
85     			struct expression *assign_expr, void *unused)
86     {
87     	struct expression *arg;
88     
89     	arg = get_argument_from_call_expr(call_expr->args, 0);
90     	arg = strip_expr(arg);
91     	while (arg->type == EXPR_ASSIGNMENT)
92     		arg = strip_expr(arg->left);
93     	set_state_expr(my_id, arg, &err_ptr);
94     }
95     
96     static void match_dereferences(struct expression *expr)
97     {
98     	if (expr->type != EXPR_PREOP)
99     		return;
100     	check_is_err_ptr(expr->unop);
101     }
102     
103     static void match_kfree(const char *fn, struct expression *expr, void *_arg_nr)
104     {
105     	int arg_nr = PTR_INT(_arg_nr);
106     	struct expression *arg;
107     
108     	arg = get_argument_from_call_expr(expr->args, arg_nr);
109     	check_is_err_ptr(arg);
110     }
111     
112     static void match_condition(struct expression *expr)
113     {
114     	if (expr->type == EXPR_ASSIGNMENT) {
115     		match_condition(expr->right);
116     		match_condition(expr->left);
117     	}
118     	if (!get_state_expr(my_id, expr))
119     		return;
120     	/* If we know the variable is zero that means it's not an ERR_PTR */
121     	set_true_false_states_expr(my_id, expr, NULL, &checked);
122     }

124     static void register_err_ptr_funcs(void)
125     {
126     	struct token *token;
127     	const char *func;
128     
129     	token = get_tokens_file("kernel.returns_err_ptr");
130     	if (!token)
131     		return;
132     	if (token_type(token) != TOKEN_STREAMBEGIN)
133     		return;
134     	token = token->next;
135     	while (token_type(token) != TOKEN_STREAMEND) {
136     		if (token_type(token) != TOKEN_IDENT)
137     			return;
138     		func = show_ident(token->ident);
139     		add_function_assign_hook(func, &match_returns_err_ptr, NULL);
140     		token = token->next;
141     	}
142     	clear_token_alloc();
143     }

145     static void match_err_ptr_positive_const(const char *fn, struct expression *expr, void *unused)
146     {
147     	struct expression *arg;
148     	sval_t sval;
149     
150     	arg = get_argument_from_call_expr(expr->args, 0);
151     
152     	if (!get_value(arg, &sval))
153     		return;
154     	if (sval_is_positive(sval) && sval_cmp_val(sval, 0) != 0)
155     		sm_msg("error: passing non negative %s to ERR_PTR", sval_to_str(sval));
156     }

158     static void match_err_ptr(const char *fn, struct expression *expr, void *unused)
159     {
160     	struct expression *arg;
161     	struct sm_state *sm;
162     	struct sm_state *tmp;
163     	sval_t tmp_min;
164     	sval_t tmp_max;
165     	sval_t min = sval_type_max(&llong_ctype);
166     	sval_t max = sval_type_min(&llong_ctype);
167     
168     	arg = get_argument_from_call_expr(expr->args, 0);
169     	sm = get_sm_state_expr(SMATCH_EXTRA, arg);
170     	if (!sm)
171     		return;
172     	FOR_EACH_PTR(sm->possible, tmp) {
173     		tmp_min = estate_min(tmp->state);
174     		if (!sval_is_a_min(tmp_min) && sval_cmp(tmp_min, min) < 0)
175     			min = tmp_min;
176     		tmp_max = estate_max(tmp->state);
177     		if (!sval_is_a_max(tmp_max) && sval_cmp(tmp_max, max) > 0)
178     			max = tmp_max;
179     	} END_FOR_EACH_PTR(tmp);
180     	if (sval_is_negative(min) && sval_cmp_val(min, -4095) < 0)
181     		sm_msg("error: %s too low for ERR_PTR", sval_to_str(min));
182     	if (sval_is_positive(max) && sval_cmp_val(max, 0) != 0)
183     		sm_msg("error: passing non negative %s to ERR_PTR", sval_to_str(max));
184     }

186     void check_err_ptr_deref(int id)
187     {
188     	if (option_project != PROJ_KERNEL)
189     		return;
190     
191     	my_id = id;
192     	return_implies_state("IS_ERR", 0, 0, &match_checked, NULL);
193     	return_implies_state("IS_ERR", 1, 1, &match_err, NULL);
194     	return_implies_state("IS_ERR_OR_NULL", 0, 0, &match_checked, NULL);
195     	return_implies_state("IS_ERR_OR_NULL", 1, 1, &match_err, NULL);
196     	return_implies_state("PTR_RET", 0, 0, &match_checked, NULL);
197     	return_implies_state("PTR_RET", -4096, -1, &match_err, NULL);
198     	register_err_ptr_funcs();
199     	add_hook(&match_dereferences, DEREF_HOOK);
200     	add_function_hook("ERR_PTR", &match_err_ptr_positive_const, NULL);
201     	add_function_hook("ERR_PTR", &match_err_ptr, NULL);
202     	add_hook(&match_condition, CONDITION_HOOK);
203     	add_modification_hook(my_id, &ok_to_use);
204     	add_function_hook("kfree", &match_kfree, INT_PTR(0));
205     	add_function_hook("brelse", &match_kfree, INT_PTR(0));
206     	add_function_hook("kmem_cache_free", &match_kfree, INT_PTR(1));
207     	add_function_hook("vfree", &match_kfree, INT_PTR(0));
208     
209     	err_ptr_rl = clone_rl_permanent(alloc_rl(err_ptr_min, err_ptr_max));
210     }
