204     #define sm_msg(msg...) \
205     do {                                                           \
206     	print_implied_debug_msg();                             \
207     	if (!option_debug && !final_pass && !local_debug)      \
208     		break;                                         \
209     	if (!option_info && is_silenced_function())	       \
210     		break;					       \
211     	sm_prefix();					       \
212             sm_printf(msg);                                        \
213             sm_printf("\n");                                       \
214     } while (0)
215     
216     #define local_debug(msg...)					\
217     do {								\
218     	if (local_debug)					\
219     		sm_msg(msg);					\
220     } while (0)

222     extern char *implied_debug_msg;
223     static inline void print_implied_debug_msg(void)
224     {
225     	static struct symbol *last_printed = NULL;
226     
227     	if (!implied_debug_msg)
228     		return;
229     	if (last_printed == cur_func_sym)
230     		return;
231     	last_printed = cur_func_sym;
232     	sm_msg("%s", implied_debug_msg);
233     }
234     
235     #define sm_debug(msg...) do { if (option_debug) sm_printf(msg); } while (0)
236     
237     #define sm_info(msg...) do {					\
238     	if (option_debug || (option_info && final_pass)) {	\
239     		sm_prefix();					\
240     		sm_printf("info: ");				\
241     		sm_printf(msg);					\
242     		sm_printf("\n");				\
243     	}							\
244     } while(0)

686     #define run_sql(call_back, data, sql...)    \
687     do {                                  \
688     	char sql_txt[1024];           \
689     	if (option_no_db)             \
690     		break;                \
691     	snprintf(sql_txt, 1024, sql); \
692     	debug_sql(sql_txt);  	      \
693     	sql_exec(call_back, data, sql_txt); \
694     } while (0)
695     
696     /* like run_sql() but for the in-memory database */
697     #define mem_sql(call_back, data, sql...)						\
698     do {										\
699     	char sql_txt[1024];							\
700     										\
701     	snprintf(sql_txt, sizeof(sql_txt), sql);				\
702     	sm_debug("in-mem: %s\n", sql_txt);					\
703     	debug_mem_sql(sql_txt);  	      					\
704     	sql_mem_exec(call_back, data, sql_txt);					\
705     } while (0)
