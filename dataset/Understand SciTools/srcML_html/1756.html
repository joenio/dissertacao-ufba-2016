24     #include <srcml_options.hpp>
25     #include <src_input_libarchive.hpp>
26     #include <src_prefix.hpp>
27     
28     // Convert input to a ParseRequest and assign request to the processing queue
29      void src_input_text(ParseQueue& queue,
30      	srcml_archive* srcml_arch,
31      	const srcml_request_t& srcml_request,
32      	const std::string& input_file) {
33     
34         // form the parsing request
35      	ParseRequest* prequest = new ParseRequest;
36     
37      	if (srcml_request.command & SRCML_COMMAND_NOARCHIVE)
38      		prequest->disk_dir = srcml_request.output_filename;
39     
40         prequest->filename = srcml_request.att_filename;
41      	prequest->url = srcml_request.att_url;
42      	prequest->version = srcml_request.att_version;
43      	prequest->srcml_arch = srcml_arch;
44      	prequest->language = srcml_request.att_language ? *srcml_request.att_language : "";
45     
46         prequest->status = 0; //!language.empty() ? 0 : SRCML_STATUS_UNSET_LANGUAGE;
47     
48         std::string raw_text = src_prefix_resource(input_file);
49     
50         // fill up the parse request buffer
51         if (!prequest->status) {
52         	// copy from the text directly into a buffer
53         	// perform newline and tab expansion
54         	// TODO: Do this more efficiently
55         	// TODO: Make test cases for each part
56         	// TODO: Support \nnn, \xnnn, \unnn, \Unnnnnnnn
57         	bool startescape = false;
58         	for (std::string::const_iterator p = raw_text.begin(); p != raw_text.end(); ++p) {
59     
60         		if (!startescape && *p == '\\') {
61     
62         			startescape = true;
63     
64         		} else if (startescape) {
65     
66         			if (*p == 'n') {
67         				prequest->buffer.push_back('\n');
68         			} else if (*p == 't') {
69         				prequest->buffer.push_back('\t');
70         			} else if (*p == 'f') {
71         				prequest->buffer.push_back('\f');
72         			} else if (*p == 'a') {
73         				prequest->buffer.push_back('\a');
74         			} else if (*p == 'b') {
75         				prequest->buffer.push_back('\b');
76         			} else if (*p == 'e') {
77         				prequest->buffer.push_back('\e');
78         			} else if (*p == 'r') {
79         				prequest->buffer.push_back('\r');
80         			} else if (*p == 'v') {
81         				prequest->buffer.push_back('\v');
82         			} else {
83         				prequest->buffer.push_back('\\');
84         				prequest->buffer.push_back(*p);
85         			}
86     
87        				startescape = false;
88         		} else {
89     
90         			prequest->buffer.push_back(*p);
91         		}
92         	}
93         	++prequest->loc;
94         }
95     
96         // schedule for parsing
97         queue.schedule(prequest);
98     
99     }
