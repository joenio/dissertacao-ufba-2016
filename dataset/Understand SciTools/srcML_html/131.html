65     /**
66      * SRCML_TRANSORM_TYPE
67      *
68      * Transformation types, xpath, xslt, and relaxng
69      */
70     enum SRCML_TRANSFORM_TYPE { SRCML_XPATH, SRCML_XSLT, SRCML_RELAXNG };
71     
72     /**
73      * xpath_arguments
74      *
75      * Data structure to hold xpath arguments
76      */
77      struct xpath_arguments {
78     
79         /** the xpath expression */
80         boost::optional<std::string> str;
81     
82         /** the optional element prefix */
83         boost::optional<std::string> prefix;
84         /** the element uri */
85         boost::optional<std::string> uri;
86         /** the tag name */
87         boost::optional<std::string> element;
88     
89         /** the optional attribute prefix */
90         boost::optional<std::string> attr_prefix;
91         /** the attribute uri */
92         boost::optional<std::string> attr_uri;
93         /** the attribute name */
94         boost::optional<std::string> attr_name;
95         /** the optional attribute value */
96         boost::optional<std::string> attr_value;
97     
98     
99      };
100     
101     /**
102      * transform
103      *
104      * Struct to hold transformation information for latter application.
105      */
106     struct transform {
107         
108         /** a transformation type */
109         SRCML_TRANSFORM_TYPE type;
110     
111         /** XSLT parameters */
112         std::vector<const char *> xsl_parameters;
113     
114         /** the transformation to perform for XPath */
115         struct xpath_arguments arguments;
116     
117         /** the transformation to perform for XSLT and relaxng */
118         xmlDocPtr doc;
119     
120         xmlXPathCompExprPtr compiled_xpath;
121     };
122     
123     /**
124      * SRCML_ARCHIVE_TYPE
125      *
126      * Archive type read, write, or read/write.
127      */
128     enum SRCML_ARCHIVE_TYPE { SRCML_ARCHIVE_INVALID, SRCML_ARCHIVE_RW, SRCML_ARCHIVE_READ, SRCML_ARCHIVE_WRITE };
129     
130     /**
131      * srcml_archive
132      *
133      * Holds data for a srcML archive read/write.
134      */
135     struct srcml_archive {
136     
137         /** type archive type read/write */
138         SRCML_ARCHIVE_TYPE type;
139     
140         /** @todo rename to xml_encoding */
141         /** an attribute for the xml encoding */
142         boost::optional<std::string> encoding;
143         /** source encoding */
144         boost::optional<std::string> src_encoding;
145         /** an attribute for a revision */
146         boost::optional<std::string> revision;
147         /** an attribute for a language */
148         boost::optional<std::string> language;
149         /** an attribute for a url path */
150         boost::optional<std::string> url;
151         /** an attribute for a version string */
152         boost::optional<std::string> version;
153         /** an array of name-value attribute pairs */
154         std::vector<std::string> attributes;
155     
156         /** srcml options */
157         OPTION_TYPE options;
158     
159         /** size of tabstop */
160         size_t tabstop;
161     
162         /** an array of XML namespace prefixes */
163         std::vector<std::string> prefixes;
164     
165         /** namespaces an array of XML namespaces */
166         std::vector<std::string> namespaces;
167     
168         /** target/data pair for processing instruction */
169         boost::optional<std::pair<std::string, std::string> > processing_instruction;
170     
171         /** an array of registered extension language pairs */
172         language_extension_registry registered_languages;
173     
174         /** an array of user defined macros and their types */
175         std::vector<std::string> user_macro_list;
176     
177         /** a srcMLTranslator for writing and parsing */
178         srcml_translator * translator;
179     
180         /** a srcMLReader for reading */
181         srcml_sax2_reader * reader;
182     
183         /** xmlParserInputBuffer for reading */
184         xmlParserInputBufferPtr input;
185      
186         /** an array of transformations to apply */
187         std::vector<transform> transformations;
188     
189         /** libxml2 callback wrapper context.  Only needed for IO functions */
190         boost::any context;
191     
192     };
193     
194     /**
195      * srcml_unit
196      *
197      * Hold data pertaining to a unit in an archive,
198      * such as the transformed or collected unit.
199      */
200     struct srcml_unit {
201         /** the archive the unit is created from */
202         srcml_archive* archive;
203     
204         /** source encoding */
205         boost::optional<std::string> encoding;
206         /** an attribute for a revision */
207         boost::optional<std::string> revision;
208         /** an attribute for a language */
209         boost::optional<std::string> language;
210         /** an attribute name for a file */
211         boost::optional<std::string> filename;
212         /** an attribute for a url path */
213         boost::optional<std::string> url;
214         /** an attribute for a version string */
215         boost::optional<std::string> version;
216         /** an attribute for a timestamp string */
217         boost::optional<std::string> timestamp;
218         /** an attribute for a hash string */
219         boost::optional<std::string> hash;
220         /** an array of name-value attribute pairs */
221         std::vector<std::string> attributes;
222         /** the type of eol to output with source code */
223         size_t eol;
224     
225         /** output buffer to hold streaming creation of unit */
226         xmlBuffer * output_buffer;
227     
228         /** a unit srcMLTranslator for writing and parsing as a stream */
229         srcml_translator * unit_translator;
230     
231         /** store if attributes have been read */
232         bool read_header;
233     
234         /** a buffer to store srcml from read and after parsing */
235         boost::optional<std::string> unit;
236     
237         /** libxml2 callback wrapper context.  Only needed for IO functions */
238         boost::any context;
239     
240     };
