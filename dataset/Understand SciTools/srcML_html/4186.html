46     /**
47      * strnstr
48      * @param s1 string to search
49      * @param s2 string to search for in s1
50      * @param n number of items to search
51      * 
52      * strnstr is only on BSD by default 
53      * Bounded substring search. Search for s2 in s1 only search first n characters.
54      *
55      * @returns location of first match or NULL.
56      */
57     char * strnstr(const char *s1, const char *s2, size_t n) {
58     
59       char save_char = s1[n];
60       ((char *)s1)[n] = 0;
61        char * ret = (char *)strstr(s1, s2);
62       ((char *)s1)[n] = save_char;
63     
64       return ret;
65     
66     }

187     /**
188      * close
189      *
190      * Close the srcml_translator and the srcML.
191      */
192     void srcml_translator::close() {
193     
194         if(first && !text_only && (options & SRCML_OPTION_ARCHIVE) > 0) {
195     
196             // Open for write;
197             out.initWriter();
198             out.initNamespaces(prefix, uri);
199     
200             out.outputXMLDecl();
201             out.outputPreRootProcessingInstruction();
202     
203             // root unit for compound srcML documents
204             out.startUnit(0, revision, url, filename, version, 0, 0, 0, attributes, true);
205         }
206     
207         first = false;
208     
209         if(is_outputting_unit) add_end_unit();
210     
211         /* FIXME: Crashes when deleted */
212         out.close();
213     
214         if(str_buffer && buffer->use) {
215     
216           (*str_buffer) = (char *)malloc(buffer->use * sizeof(char));
217           memcpy(*str_buffer, buffer->content, (size_t)buffer->use);
218           if(size && *str_buffer) *size = (size_t)buffer->use;
219     
220         }
221     
222     }

224     /**
225      * translate
226      *
227      * Translate a single unit and output.  No xml declaration is added.
228      */
229     void srcml_translator::translate(UTF8CharBuffer * parser_input) {
230     
231         if(first) {
232       
233             // Open for write;
234             out.initWriter();
235             out.initNamespaces(prefix, uri);
236     
237         }
238     
239         first = false;
240     
241         // output as inner unit
242         if(isoption(options, SRCML_OPTION_ARCHIVE)) out.setDepth(1);
243     
244         //options |= SRCML_OPTION_ARCHIVE;
245     
246         try {
247     
248             // master lexer with multiple streams
249             antlr::TokenStreamSelector selector;
250     
251             // srcML lexical analyzer from standard input
252             KeywordLexer lexer(parser_input, getLanguage(), options, user_macro_list);
253             lexer.setSelector(&selector);
254             lexer.setTabsize((int)tabsize);
255     
256             // pure block comment lexer
257             CommentTextLexer textlexer(lexer.getInputState());
258             textlexer.setSelector(&selector);
259     
260             // switching between lexers
261             selector.addInputStream(&lexer, "main");
262             selector.addInputStream(&textlexer, "text");
263             selector.select(&lexer);
264     
265             // base stream parser srcML connected to lexical analyzer
266             StreamMLParser parser(selector, getLanguage(), options);
267     
268             // connect local parser to attribute for output
269             out.setTokenStream(parser);
270     
271             // parse and form srcML output with unit attributes
272             out.consume(getLanguageString(), revision, url, filename, version, timestamp, hash, encoding);
273     
274         } catch (const std::exception& e) {
275             fprintf(stderr, "SRCML Exception: %s\n", e.what());
276         }
277         catch (UTF8FileError) {
278             fprintf(stderr, "UTF8 file error\n");
279         }
280         catch (...) {
281             fprintf(stderr, "srcML translator error\n");
282         }
283     
284         // set back to root
285         out.setDepth(0);
286     
287     }

289     /**
290      * add_unit
291      * @param unit srcML to add to archive/non-archive with configuration options
292      * @param xml the xml to output
293      *
294      * Add a unit as string directly to the archive.  If not an archive
295      * and supplied unit does not have src namespace add it.  Also, write out
296      * a supplied hash as part of output unit if specified.
297      * Can not be in by element mode.
298      *
299      * @returns if succesfully added.
300      */
301     bool srcml_translator::add_unit(const srcml_unit * unit, const char * xml) {
302     
303       if(is_outputting_unit) return false;
304     
305       bool is_archive = (options & SRCML_OPTION_ARCHIVE) > 0;
306     
307       if(first) {
308     
309         // Open for write;
310         out.initWriter();
311         out.initNamespaces(prefix, uri);
312     
313         if ((options & SRCML_OPTION_XML_DECL) > 0)
314           out.outputXMLDecl();
315       
316         out.outputPreRootProcessingInstruction();
317     
318         // root unit for compound srcML documents
319     
320         if(is_archive)
321             out.startUnit(0, revision, url, filename, version, 0, 0, 0, attributes, true);
322     
323         if (is_archive)
324             out.processText("\n\n", 2);
325     
326       }
327     
328       first = false;
329     
330       const char * end_start_unit = (char *)strchr(xml, '>');
331       if(!end_start_unit) return false;
332     
333       /** extract language */
334       char * language_start_name = strnstr(xml, "language", end_start_unit - xml);
335     
336       char * language_start_value = 0;
337       char * language_end_value = 0;
338       if(language_start_name) {
339     
340         language_start_value = (char *)strchr(language_start_name, '"');
341         language_end_value = (char *)strchr(language_start_value + 1, '"');
342         (*language_end_value) = '\0';
343     
344       } 
345     
346       /** is there a cpp namespace */
347       bool is_cpp = false;
348       for(int pos = 0; SRCML_URI_PREFIX[pos][0]; ++pos) {
349     
350         std::string cpp_uri = SRCML_URI_PREFIX[pos] + "srcML/cpp";
351     
352         is_cpp = strnstr(xml, cpp_uri.c_str(), end_start_unit - xml) != 0;
353         if(is_cpp) break;
354     
355       }
356     
357       OPTION_TYPE save_options = options;
358       if(is_cpp) options |= SRCML_OPTION_CPP;
359     
360       out.startUnit(language_start_value ? language_start_value + 1 : 0, is_archive && unit->revision ? unit->revision->c_str() : revision, unit->url ? unit->url->c_str() : 0, unit->filename ? unit->filename->c_str() : 0,
361                            unit->version ? unit->version->c_str() : 0, unit->timestamp ? unit->timestamp->c_str() : 0, unit->hash ? unit->hash->c_str() : 0, 
362                            unit->encoding ? unit->encoding->c_str() : 0, unit->attributes, false);
363     
364       if(language_start_name) (*language_end_value) = '"';
365     
366       options = save_options;
367     
368       size_t size = strlen(end_start_unit);
369     
370       if(size > 1) {
371     
372         while(end_start_unit[--size] != '<')
373           ;
374     
375         xmlTextWriterWriteRawLen(out.getWriter(), (xmlChar *)end_start_unit + 1, (int)size - 1);
376     
377       }
378     
379       out.srcMLTextWriterEndElement(out.getWriter());
380     
381       if ((options & SRCML_OPTION_ARCHIVE) > 0)
382           out.processText("\n\n", 2);
383     
384       return true;
385     
386     }

551      * Add the end tag of a unit disable worther element processing.
552      * Outputs all ending tags that are currently open within the unit first.
553      *
554      * @returns if succesfully added.
555      */
556     bool srcml_translator::add_end_unit() {
557     
558         if(!is_outputting_unit) return false;
559     
560         while(output_unit_depth--) xmlTextWriterEndElement(out.getWriter());
561     
562         output_unit_depth = 0;
563         is_outputting_unit = false;
564     
565         bool success = xmlTextWriterEndElement(out.getWriter()) != -1;
566     
567         return success;

660     /**
661      * add_string.
662      * @param content the string to write out
663      *
664      * Write the string/text to a started unit.
665      * i.e being added to start tag.
666      * add_start_unit must be called first.
667      *
668      * @returns if succesfully added.
669      */
670     bool srcml_translator::add_string(const char * content) {
671     
672         if(!is_outputting_unit || content == 0) return false;
673     
674         int ret = 0;
675         char * text = (char *)content;
676         for(char * pos = text; *pos; ++pos) {
677     
678           if(*pos != '"') continue;
679     
680           *pos = 0;
681           ret = xmlTextWriterWriteString(out.getWriter(), (const xmlChar *)text);
682           if(ret == -1) return false;
683     
684           *pos = '\"';
685           xmlTextWriterWriteRaw(out.getWriter(), (const xmlChar *)"\"");
686           if(ret == -1) return false;
687     
688           text = pos + 1;
689     
690       }
691     
692       ret = xmlTextWriterWriteString(out.getWriter(), (const xmlChar *)text);
693     
694     
695     
696       return ret != -1;
697     
698     }
