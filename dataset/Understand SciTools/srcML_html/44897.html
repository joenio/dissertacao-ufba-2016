139     template <int option>
140     void option_markup(bool opt) {
141         /*
142           If we have markup options the NULL optional arguement needs to
143           first be initializied before the bitwise work can be done.
144         */
145         if (!srcml_request.markup_options)
146             srcml_request.markup_options = 0;
147     
148         if (opt)
149             *srcml_request.markup_options |= option;
150     }

179     template <>
180     void option_field<&srcml_request_t::src_encoding>(const std::string& value) {
181     
182         if (value.empty() || srcml_check_encoding(value.c_str()) == 0) {
183             std::cerr << "srcml: invalid src encoding \"" << value.c_str() << "\"\n";
184             exit(4);
185         }
186         srcml_request.src_encoding = value;
187     }

213     template <>
214     void option_field<&srcml_request_t::tabs>(int value) {
215     
216         // check tabstop
217         if (value < 1) {

222         srcml_request.tabs = value;

231     void option_xmlns_uri(const std::string& value) {
232         srcml_request.xmlns_namespaces[""] = value;
233         srcml_request.xmlns_namespace_uris[value] = "";
234     }
235     
236     void option_xmlns_prefix(const std::vector<std::string>& values) {
237         BOOST_FOREACH( std::string value, values )
238         {
239           std::size_t delim = value.find("=");
240           if (delim == std::string::npos) {
241             std::cerr << "srcml: xmlns format missing \"=\"\n";
242             exit(1); //ERROR CODE TBD
243           }
244     
245           srcml_request.xmlns_namespaces[value.substr(0, delim)] = value.substr(delim + 1);
246           srcml_request.xmlns_namespace_uris[value.substr(delim + 1)] = value.substr(0, delim);
247         }
248     }

251     void option_to_dir(const std::string& value) {
252         srcml_request.output_filename = value;

254         srcml_request.command |= SRCML_COMMAND_NOARCHIVE;

257     void positional_args(const std::vector<std::string>& value) {
258         srcml_request.input.reserve(srcml_request.input.size() + value.size());
259     
260         BOOST_FOREACH(const std::string& iname, value) {
261     
262             // record the position of stdin
263             if (iname == "-" || iname == "stdin://-")
264                 srcml_request.stdindex = (int) srcml_request.input.size();
265     
266             srcml_request.input.push_back(src_prefix_add_uri(iname));
267         }
268     }

279     void option_help(const std::string& help_opt) {
280         if (help_opt.empty()) {
281             // TODO: A new header and footer for the general option
282             std::cout << SRCML_HEADER << "\n";
283             std::cout << general << "\n";
284             
285             std::cout << src2srcml << "\n";
286             std::cout << srcml2src << "\n";
287     
288             std::cout << SRCML_FOOTER << "\n";
289         }
290         else if (help_opt == "src2srcml") {
291             std::cout << SRC2SRCML_HEADER << "\n";
292             std::cout << src2srcml << "\n";
293             std::cout << SRC2SRCML_FOOTER << "\n";
294         }
295         else if (help_opt == "srcml2src") {
296             std::cout << SRCML2SRC_HEADER << "\n";
297             std::cout << srcml2src << "\n";
298             std::cout << SRCML2SRC_FOOTER << "\n";
299         }
300         else {
301             std::cout << "Unknown module '"
302                       << help_opt << "' in --help\n";
303             exit(1);
304         }
305         exit(0);
306     }

438             all.add(general).add(src2srcml_options).add(markup_options).add(xml_form).add(metadata_options).add(srcml2src_options).
439                 add(query_transform).add(positional_options).add(deprecated_options).add(debug_options).add(experimental_options);
440     
441             // Positional Args
442             input_file.add("input-files", -1);
443     
444             // Assign the CLI args to the map
445             prog_opts::variables_map cli_map;
446             
447             const prog_opts::basic_parsed_options< char >& cliopts = prog_opts::command_line_parser(argc, argv).options(all).
448                              positional(input_file).extra_parser(custom_parser).run();
449     
450             std::vector< prog_opts::basic_option< char > > parsedOptions = cliopts.options;
451     
452             // loop the cli options in the order they were processed/received
453             BOOST_FOREACH(const prog_opts::basic_option< char >& option, parsedOptions) {
454               if (option.string_key == "relaxng" || option.string_key == "xpath" || option.string_key == "xslt" || option.string_key == "xpathparam" 
455                  || option.string_key == "element" || option.string_key == "attribute") {
456     
457                 if (option.string_key == "xpath")
458                   srcml_request.xpath_query_support.push_back(std::make_pair(boost::none,boost::none));
459     
460                 BOOST_FOREACH(const std::basic_string< char >& vals, option.value) {
461                   if (option.string_key == "element" && srcml_request.xpath_query_support.size() < 1) {
462                     std::cerr << "srcml: element option must follow an --xpath option\n";
463                     exit(SRCML_STATUS_INVALID_ARGUMENT);
464                   }
465                   if (option.string_key == "attribute" && srcml_request.xpath_query_support.size() < 1) {
466                     std::cerr << "srcml: attribute option must follow an --xpath option\n";
467                     exit(SRCML_STATUS_INVALID_ARGUMENT);
468                   }
469     
470                   if (option.string_key == "element") {
471                     srcml_request.xpath_query_support.at(srcml_request.xpath_query_support.size() - 1).first = clean_element_input(vals);
472                   }
473                   else if (option.string_key == "attribute") {
474                     srcml_request.xpath_query_support.at(srcml_request.xpath_query_support.size() - 1).second = clean_attribute_input(vals);
475                   }
476                   else {
477                     srcml_request.transformations.push_back(src_prefix_add_uri(option.string_key, vals));
478                   }
479                 }
480               }
481             }
482     
483             prog_opts::store(cliopts , cli_map);
484             prog_opts::notify(cli_map);
485     
486             // Check option conflicts
487             conflicting_options(cli_map, "quiet", "verbose");
488             conflicting_options(cli_map, "output", "to-dir");
489             conflicting_options(cli_map, "cpp-text-else", "cpp-markup-else");
490             conflicting_options(cli_map, "cpp-text-if0", "cpp-markup-if0");
491             conflicting_options(cli_map, "output-src", "output-xml");
492     
493             // Check dependent options
494             // Format: option_dependency(cli_map, [option], [option]);
495             option_dependency(cli_map, "text", "language");
496     
497             // If input was from stdin, then artificially put a "-" into the list of input files
498             if (srcml_request.input.empty())
499               positional_args(std::vector<std::string>(1, "stdin://-"));
500     
501             // If position option is used without tabs...set default tab of 8
502             if ((*srcml_request.markup_options & SRCML_OPTION_POSITION && srcml_request.tabs == 0) || srcml_request.tabs == 0)
503               srcml_request.tabs = 8;
504     
505     #if defined(__GNUG__) && !defined(__MINGW32__)
506             // automatic interactive use from stdin (not on redirect or pipe)
507             if (isatty(STDIN_FILENO))
508                 option_command<SRCML_COMMAND_INTERACTIVE>(true);
509     #endif
510     
511             // Ideally get this from libsrcml
512             const std::vector<std::string> reserved_namespaces = {
513             "=http://www.srcML.org/srcML/src",
514             "cpp=http://www.srcML.org/srcML/cpp",
515             "java=http://www.srcML.org/srcML/java",};
516     
517             BOOST_FOREACH(const std::string& ns, reserved_namespaces) {
518               size_t delim = ns.find('=');
519               std::string prefix = ns.substr(0, delim);
520               std::string uri = ns.substr(delim + 1);
521     
522               // A reserved prefix wasn't used or it was set to the same uri
523               if (srcml_request.xmlns_namespaces.find(prefix) == srcml_request.xmlns_namespaces.end() || srcml_request.xmlns_namespaces[prefix] == uri) {
524                 continue;
525               }
526     
527               // A reserved uri is set to a different prefix
528               if (srcml_request.xmlns_namespace_uris.find(uri) != srcml_request.xmlns_namespaces.end() && srcml_request.xmlns_namespace_uris[uri] != prefix) {
529                 continue;
530               }
531     
532               std::cerr << "srcml: prefix " << "\"" << prefix << "\" assigned multiple URIs \"" << uri << "\", \"" << srcml_request.xmlns_namespaces[prefix] << "\"\n";
533               exit(1); // TODO Need a real error code
534             }
535     
536         }
537         // Unknown Option
538         catch(boost::program_options::unknown_option& e) {
539             std::cerr << "srcml: " << e.what() << "\n";
540             exit(3);
541         }
542         // Missing Option Value
543         catch(boost::program_options::error_with_option_name& e) {
544             std::cerr << "srcml: " << e.what() << "\n";
545             exit(7);
546         }
547         // Catch all other issues with generic error
548         catch(std::exception& e) {
549             std::cerr << "srcml: " << e.what() << "\n";
550             exit(1);
551         }
552     
553         return srcml_request;
554     }

556     // Custom parser for xmlns: option
557     std::pair<std::string, std::string> custom_parser(const std::string& s) {
558         if (s.find("--xmlns:") == 0)
559             return std::make_pair(std::string("xmlns:"), std::string(s.substr(s.find(":")+1)));
560         else
561             return std::make_pair(std::string(), std::string());
562     }

564     // Set to detect option conflicts
565     void conflicting_options(const prog_opts::variables_map& vm, const char* opt1, const char* opt2) {
566         if (vm.count(opt1) && !vm[opt1].defaulted() && vm.count(opt2) && !vm[opt2].defaulted()) {
567             std::cerr << "srcml: " << "Conflicting options '" << opt1 << "' and '" << opt2 << "'.\n";
568             exit(15);
569         }
570     }
571     
572     // Check for dependent options
573     void option_dependency(const prog_opts::variables_map& vm,
574                             const char* option, const char* dependent_option)
575     {
576         if (vm.count(option) && !vm[option].defaulted()) {
577             if (vm.count(dependent_option) == 0) {
578                 throw std::logic_error(std::string("Option '") + option 
579                                         + "' requires option '" + dependent_option + "'.");
580             }
581         }
582     }

584     element clean_element_input(const std::basic_string< char >& element_input) {
585       std::string vals = element_input;
586       size_t elemn_index = vals.find(":");
587     
588       // Element requires a prefix
589       if (elemn_index == std::string::npos) {
590         exit(1);
591       }
592     
593       element elem;
594       elem.prefix = vals.substr(0, elemn_index);
595       elem.name = vals.substr(elemn_index + 1);
596       return elem;
597     }
598     
599     attribute clean_attribute_input(const std::basic_string< char >& attribute_input) {
600       std::string vals = attribute_input;
601       size_t attrib_colon = vals.find(":");
602       size_t attrib_equals = vals.find("=");
603       
604       // Attribute must have a value
605       if (attrib_equals == std::string::npos) {
606         exit(1);
607       }
608     
609       // Missing prefix requires an element with a prefix
610       if (attrib_colon == std::string::npos && !(srcml_request.xpath_query_support.at(srcml_request.xpath_query_support.size() - 1).first)) {
611         exit(1);
612       }
613     
614       attribute attrib;
615       
616       if (attrib_colon != std::string::npos) {  
617         attrib.prefix = vals.substr(0, attrib_colon);
618         attrib.name = vals.substr(attrib_colon + 1, attrib_equals - attrib_colon - 1);
619       }
620       else {
621         attrib.prefix = srcml_request.xpath_query_support.at(srcml_request.xpath_query_support.size() - 1).first->prefix;
622         attrib.name = vals.substr(0, attrib_equals);
623       }
624         
625       attrib.value = vals.substr(attrib_equals + 1);
626     
627       return attrib;
628     }
