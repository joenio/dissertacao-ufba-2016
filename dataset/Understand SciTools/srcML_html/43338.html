46      * srcml_append_transform_xpath
47      * @param archive a srcml archive
48      * @param xpath_string an XPath expression
49      *
50      * Append the XPath expression to the list
51      * of transformation/queries.  As of yet no way to specify context
52      *
53      * @returns Returns SRCML_STATUS_OK on success and a status error codes on failure.
54      */
55     int srcml_append_transform_xpath(srcml_archive* archive, const char* xpath_string) {
56     
57         if(archive == NULL || xpath_string == 0) return SRCML_STATUS_INVALID_ARGUMENT;
58         if(archive->type != SRCML_ARCHIVE_READ && archive->type != SRCML_ARCHIVE_RW) return SRCML_STATUS_INVALID_IO_OPERATION;
59     
60         struct xpath_arguments arguments = { optional_string_create(xpath_string), boost::optional<std::string>(),
61             boost::optional<std::string>(), boost::optional<std::string>(), boost::optional<std::string>(),
62             boost::optional<std::string>(), boost::optional<std::string>(), boost::optional<std::string>() };
63     
64         transform tran = { SRCML_XPATH, std::vector<const char *>(1, (const char *)0), arguments, 0, 0 };
65         archive->transformations.push_back(tran);
66     
67         return SRCML_STATUS_OK;
68     
69     }

342     /**
343      * srcml_append_transform_param
344      * @param archive a srcml archive
345      * @param xpath_param_name name of a parameter
346      * @param xpath_param_value value of the named parameter
347      *
348      * Append the parameter to the last transformation.
349      *
350      * @returns Returns SRCML_STATUS_OK on success and a status errors code on failure.
351      */
352     int srcml_append_transform_param(srcml_archive* archive, const char* xpath_param_name, const char* xpath_param_value) {
353     
354         if(archive == NULL || xpath_param_name == NULL || xpath_param_value == NULL) return SRCML_STATUS_INVALID_ARGUMENT;
355         if(archive->type != SRCML_ARCHIVE_READ && archive->type != SRCML_ARCHIVE_RW) return SRCML_STATUS_INVALID_IO_OPERATION;
356         if(archive->transformations.size() == 0) return SRCML_STATUS_NO_TRANSFORMATION;
357     
358         archive->transformations.back().xsl_parameters.pop_back();
359         archive->transformations.back().xsl_parameters.push_back(xpath_param_name);
360         archive->transformations.back().xsl_parameters.push_back(strdup(xpath_param_value));
361         archive->transformations.back().xsl_parameters.push_back(0);
362     
363         return SRCML_STATUS_OK;
364     
365     }

367     /**
368      * srcml_append_transform_stringparam
369      * @param archive a srcml archive
370      * @param xpath_param_name name of a parameter
371      * @param xpath_param_value value of the named parameter will be wrapped in "
372      *
373      * Append the parameter to the last transformation with the value wrapped in ".
374      *
375      * @returns Returns SRCML_STATUS_OK on success and a status errors code on failure.
376      */
377     int srcml_append_transform_stringparam(srcml_archive* archive, const char* xpath_param_name, const char* xpath_param_value) {
378     
379         if(archive == NULL || xpath_param_name == NULL || xpath_param_value == NULL) return SRCML_STATUS_INVALID_ARGUMENT;
380         if(archive->type != SRCML_ARCHIVE_READ && archive->type != SRCML_ARCHIVE_RW) return SRCML_STATUS_INVALID_IO_OPERATION;
381         if(archive->transformations.size() == 0) return SRCML_STATUS_NO_TRANSFORMATION;
382     
383         archive->transformations.back().xsl_parameters.pop_back();
384         archive->transformations.back().xsl_parameters.push_back(xpath_param_name);
385     
386         size_t xpath_param_value_length = strlen(xpath_param_value);
387         char * string_value = new char[xpath_param_value_length + 3];
388         string_value[0] = '"';
389         strncpy(string_value + 1, xpath_param_value, xpath_param_value_length);
390         string_value[xpath_param_value_length + 1] = '"';
391         string_value[xpath_param_value_length + 2] = 0;
392     
393         archive->transformations.back().xsl_parameters.push_back(string_value);
394         archive->transformations.back().xsl_parameters.push_back(0);
395     
396         return SRCML_STATUS_OK;
397     
398     }

400     /**
401      * srcml_clear_transforms
402      * @param archive an archive
403      *
404      * Remove all transformations from archive.
405      *
406      * @returns SRCML_STATUS_OK on success and SRCML_STATUS_INVALID_ARGUMENT on failure.
407      */
408     int srcml_clear_transforms(srcml_archive * archive) {
409     
410         if(archive == NULL) return SRCML_STATUS_INVALID_ARGUMENT;
411     
412         for(std::vector<transform>::iterator itr = archive->transformations.begin(); itr != archive->transformations.end(); ++itr) {
413     
414             for(std::vector<const char *>::size_type pos = 1; pos < itr->xsl_parameters.size(); pos += 2)
415                 delete itr->xsl_parameters[pos];
416     
417             if(itr->type == SRCML_XSLT || itr->type == SRCML_RELAXNG)
418                 xmlFreeDoc(itr->doc);
419     
420         }
421     
422         archive->transformations.clear();
423     
424         return SRCML_STATUS_OK;
425     
426     }

428     /**
429      * srcml_apply_transforms
430      * @param iarchive an input srcml archive
431      * @param oarchive and output srcml archive
432      *
433      * Apply appended transformations inorder added and consecutively.
434      * Intermediate results are stored in a temporary file.
435      * Transformations are cleared.
436      *
437      * @returns Returns SRCML_STATUS_OK on success and a status error codes on failure.
438      */
439     
440     std::vector<transform> global_transformations;
441     
442     int srcml_apply_transforms(srcml_archive* iarchive, srcml_archive* oarchive) {
443     
444         if(iarchive == NULL || oarchive == NULL) return SRCML_STATUS_INVALID_ARGUMENT;
445         if((iarchive->type != SRCML_ARCHIVE_READ && iarchive->type != SRCML_ARCHIVE_RW)
446             || (oarchive->type != SRCML_ARCHIVE_WRITE && oarchive->type != SRCML_ARCHIVE_RW)) return SRCML_STATUS_INVALID_IO_OPERATION;
447     
448         global_transformations = iarchive->transformations;
449     
450         for(std::vector<transform>::size_type i = 0; i < iarchive->transformations.size(); ++i) {
451     
452             xmlParserInputBufferPtr pinput = iarchive->input;
453     
454             int error = 0;
455             try {
456     
457                 switch(iarchive->transformations.at(i).type) {
458     
459                 case SRCML_XPATH: {
460     
461                     error = srcml_xpath(pinput, "src:unit",
462                                         optional_get_c_str(iarchive->transformations.at(i).arguments.str),
463                                         optional_get_c_str(iarchive->transformations.at(i).arguments.prefix), optional_get_c_str(iarchive->transformations.at(i).arguments.uri),
464                                         optional_get_c_str(iarchive->transformations.at(i).arguments.element),
465                                         optional_get_c_str(iarchive->transformations.at(i).arguments.attr_prefix), optional_get_c_str(iarchive->transformations.at(i).arguments.attr_uri),
466                                         optional_get_c_str(iarchive->transformations.at(i).arguments.attr_name), optional_get_c_str(iarchive->transformations.at(i).arguments.attr_value),
467                                         oarchive->options, oarchive);
468                     break;
469                 }
470     
471     #ifdef WITH_LIBXSLT
472                 case SRCML_XSLT: {
473     
474                     error = srcml_xslt(pinput, "src:unit",
475                                        iarchive->transformations.at(i).doc,
476                                        &iarchive->transformations.at(i).xsl_parameters.front(), 0, oarchive->options, oarchive);
477                     break;
478                 }
479     #endif
480     
481                 case SRCML_RELAXNG: {
482     
483                     error = srcml_relaxng(pinput,
484                                           iarchive->transformations.at(i).doc,
485                                           oarchive->options, oarchive);
486                     break;
487                 }
488     
489                 default : break;
490     
491                 }
492     
493             } catch(...) {
494     
495                 return SRCML_STATUS_INVALID_INPUT;
496             }
497     
498             break;
499     
500         }
501     
502         //srcml_clear_transforms(iarchive);
503     
504         return SRCML_STATUS_OK;
505     
506     }
