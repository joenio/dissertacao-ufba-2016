21     /*
22       Example program of the use of the C API for srcML.
23     
24       A straightforward translation of source code to the srcML format.
25       Translates the file "a.cpp" to the srcML format in "a.cpp.xml", however
26       in this case the language is specified:
27     
28       * This creates a single-unit srcML file, i.e., a non-archive srcML
29       * The srcML attribute filename will be the name of the file passed as the first
30       parameter.
31     */
32     
33     #include "srcml.h"
34     
35     int main(int argc, char* argv[]) {
36     
37         /* Translate from a source-code file to a srcML file */
38         srcml_set_language(SRCML_LANGUAGE_CXX);
39         srcml("a.cpp", "a.cpp.xml");
40     
41         return 0;
42     }
