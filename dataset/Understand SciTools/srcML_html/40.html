52      enum STATES { INDETERMINATE, SRC, SRCML };
53     
54      class srcml_input_src {
55      public:
56     
57         srcml_input_src() : unit(0) {}
58         srcml_input_src(const std::string& other) : arch(0), state(INDETERMINATE), isdirectory(false), unit(0) { 
59     
60     
61             filename = src_prefix_add_uri(other);
62     
63             // since boost::filesystem does not support URIs, separate out the protocol
64             src_prefix_split_uri(filename, protocol, resource);
65     
66             // boost::filesystem does not handle multiple extensions
67             // so extract
68             boost::filesystem::path rpath(resource.c_str());
69     
70             if (protocol == "file")
71                 isdirectory = boost::filesystem::is_directory(rpath);
72     
73             if (!isdirectory) {
74     
75                 // gather compression and archive extensions together, as an
76                 // extension could be both
77                 for ( ; rpath.has_extension() && (is_compressed(rpath.extension().string()) || is_archive(rpath.extension().string())); rpath = rpath.stem()) {
78                     // collect compressions
79                     if (is_compressed(rpath.extension().string()))
80                         compressions.push_back(rpath.extension().string());
81     
82                     // collect archives
83                     if (is_archive(rpath.extension().string()))
84                         archives.push_back(rpath.extension().string());
85                 }
86     
87                 // collect real extension
88                 extension = rpath.has_extension() ? rpath.extension().string() : (!archives.empty() ? archives.back() : "");
89             }
90     
91             plainfile = rpath.string();
92     
93             if (resource != "-" && protocol != "text")
94                 state = extension == ".xml" ? SRCML : SRC;
95     
96             if (protocol == "text")
97                 state = SRC;
98     
99             if (protocol == "stdin")
100                 fd = STDIN_FILENO;
101             if (protocol == "stdout")
102                 fd = STDOUT_FILENO;
103         }
104     
105         srcml_input_src(const std::string& other, int fds) : unit(0) {
106     
107             srcml_input_src s(other);
108             s = fds;
109     
110             swap(s);
111         }
112     
113         srcml_input_src(int fds) : unit(0) {
114     
115             srcml_input_src s("-");
116             s = fds;
117     
118             swap(s);
119         }
120     
121         srcml_input_src& operator=(const std::string& other) { srcml_input_src t(other); swap(t); return *this; }
122         srcml_input_src& operator=(FILE* other) { fileptr = other; return *this; }
123         srcml_input_src& operator=(int other) { fd = other; return *this; }
124     
125         operator const std::string&() const { return resource; }
126         operator FILE*() const { return *fileptr; }
127         operator int() const { return *fd; }
128     
129         bool operator==(const std::string& other) const { return other == filename; }
130         bool operator!=(const char* other) const { return filename != other; }
131     
132         const char* c_str() const { return resource.c_str(); }
133     
134         void swap(srcml_input_src& other) {
135     
136             std::swap(filename, other.filename);
137             std::swap(protocol, other.protocol);
138             std::swap(resource, other.resource);
139             std::swap(plainfile, other.plainfile);
140             std::swap(extension, other.extension);
141             std::swap(fileptr, other.fileptr);
142             std::swap(fd, other.fd);
143             std::swap(arch, other.arch);
144             std::swap(state, other.state);
145             std::swap(compressions, other.compressions);
146             std::swap(archives, other.archives);
147             std::swap(isdirectory, other.isdirectory);
148         }
149     
150         std::string filename;
151         std::string protocol;
152         std::string resource;
153         std::string plainfile;
154         std::string extension;
155         boost::optional<FILE*> fileptr;
156         boost::optional<int> fd;
157         archive* arch;
158         enum STATES state;
159         std::list<std::string> compressions;
160         std::list<std::string> archives;
161         bool isdirectory;
162         int unit;
163     };

187     inline std::ostream& operator<<(std::ostream& out, const srcml_input_src& input) {
188     
189         out << "filename:" << input.filename << '\n';
190         out << "protocol:" << input.protocol << '\n';
191         out << "resource:" << input.resource << '\n';
192         out << "plainfile:" << input.plainfile << '\n';
193         out << "extension:" << input.extension << '\n';
194         if (input.fileptr)
195             out << "fileptr:" << *input.fileptr << '\n';
196         if (input.fd)
197             out << "fd:" << *input.fd << '\n';
198         out << "state:" << input.state << '\n';
199         BOOST_FOREACH(const std::string& compression, input.compressions)
200             out << "compression:" << compression << '\n';
201         BOOST_FOREACH(const std::string& archive, input.archives)
202             out << "archive:" << archive << '\n';
203         out << "isdirectory:" << input.isdirectory << '\n';
204     
205         return out;
206     }
