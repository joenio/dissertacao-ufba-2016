21     #include <iostream>
22     #include <string>
23     #include <fstream>
24     #include <cmath>
25     #include <sstream>
26     
27     #pragma GCC diagnostic push
28     #pragma GCC diagnostic ignored "-Wshorten-64-to-32"
29     #include <boost/date_time/posix_time/posix_time.hpp>
30     #pragma GCC diagnostic pop
31     
32     int main(int argc, char * argv[]) {
33     
34         if(argc < 3) {
35     
36             std::cerr << "timing_driver input_file log_file\n";
37             exit(1);
38     
39         }
40     
41         std::string input_file = argv[1];
42     
43         std::ifstream in(input_file);
44     
45         size_t lines_of_code = 0;
46     
47         bool is_text = false;
48         while(in) {
49     
50             char c;
51             in.get(c);
52     
53             if(c == '\n') {
54     
55                 ++lines_of_code;
56                 is_text = false;
57     
58             } else {
59     
60                 is_text = true;
61     
62             }
63     
64         }
65     
66         if(lines_of_code && !is_text) lines_of_code--;
67         if(is_text) ++lines_of_code;
68     
69         std::string extension = input_file.substr(input_file.rfind('.') + 1);
70     
71         std::string language = "";
72         if(extension == "hpp" || extension == "cpp") language = "C++";
73         if(extension == "h" || extension == "c") language = "C";
74         if(extension == "cs") language = "C#";
75         if(extension == "java") language = "Java";
76     
77         std::string str_command = "srcml " + input_file + " -o " + input_file + ".xml";
78     
79         //std::cout << "Command: " << str_command << " Line count: " << lines_of_code << '\n';
80     
81         const char * command = str_command.c_str();
82     
83         boost::posix_time::ptime start_time = boost::posix_time::microsec_clock::universal_time();
84     
85         FILE * sub_process = popen(command, "r");
86         pclose(sub_process);
87     
88         boost::posix_time::time_duration elapsed_time_duration = boost::posix_time::microsec_clock::universal_time() - start_time;
89     
90         long double elapsed_seconds = elapsed_time_duration.total_seconds() + (elapsed_time_duration.fractional_seconds() / pow((long double)10, boost::posix_time::time_duration::num_fractional_digits()));
91     
92         unsigned long long lines_of_code_per_second = (lines_of_code / elapsed_seconds) + 0.5;
93     
94         std::ostringstream lines_of_code_per_second_stream;
95         if(lines_of_code_per_second > 1000)
96             lines_of_code_per_second_stream << lines_of_code_per_second / 1000 << ',' << std::setfill('0') << std::setw(3) << (lines_of_code_per_second % 1000);
97         else
98             lines_of_code_per_second_stream << lines_of_code_per_second;
99     
100         std::cout << std::left << std::setw(9) << language << std::right << std::setw(6) << lines_of_code_per_second_stream.str() << " LOC/sec\n";
101     
102         std::string log_file_name = argv[2];
103     
104         std::ofstream log_file(log_file_name, std::ios_base::app);
105     
106         log_file << std::left << std::setw(9) << language << std::right << std::setw(6) << lines_of_code_per_second_stream.str() << " LOC/sec\n";
107     
108         return 0;
109     }
