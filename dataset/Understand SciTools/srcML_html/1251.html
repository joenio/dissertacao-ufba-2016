24     /**
25      *  XPath Signature: bool has_return()
26      *  
27      *  Checks all block children and all subsequent children for a return
28      *  statement. This is scope respecting, in that it doesn't search within
29      *  closures functions or local classes.
30      *  
31      *  has_return can be called on almost anything but the typical case 
32      *  is for it to be used on statements or functions, The behavior it has
33      *  when called on non-complex statements (those that don't combine multiple
34      *  statements using a block such as return, throw, expr_stmt and decl_stmt)
35      *  is to search those statements for a return including the sub elements 
36      *  excluding  lambdas and delegates, This means that because only the sub
37      *  elements are searched that a return doesn't have a return (because it is
38      *  one).
39      *  
40      *  Languages: C, C++, C#, Java
41      *  
42      *  Works with srcML elements:
43      *      block
44      *      function
45      *      constructor
46      *      destructor
47      *      while
48      *      if
49      *      then
50      *      elseif
51      *      else
52      *      try
53      *      catch
54      *      finally
55      *      do
56      *      for
57      *      foreach
58      *      switch
59      *      lambda
60      *      delegate
61      *      using_stmt
62      *      fixed
63      *      lock
64      *      synchronized
65      *      unsafe
66      *      static
67      *      checked
68      *      unchecked
69      *  
70      */
71     void xpath_exfun_has_return(xmlXPathParserContextPtr ctxt, int nargs);
