24     
25     #include <string.h>
26     #include <iostream>
27     
28     const char* PROTOCOL_SEPARATOR = "://";
29     
30     std::string src_prefix_add_uri(const std::string& protocol, const std::string& resource) {
31         return protocol + PROTOCOL_SEPARATOR + resource;
32     }
33     
34     std::string src_prefix_add_uri(const std::string& input_file) {
35     
36         // Only add a prefix if the input doesn't have one already (IE. http://)
37         size_t prefixPos = input_file.find(PROTOCOL_SEPARATOR);
38         if (prefixPos == std::string::npos) {
39     
40             if (input_file.compare("/dev/stdin") == 0 || input_file.compare("-") == 0)
41                 return src_prefix_add_uri("stdin", "-");
42     
43             // Check for local file
44             if (input_file.find("http:") == std::string::npos || input_file.find("https:") == std::string::npos) {
45                 return src_prefix_add_uri("file", input_file);
46             }
47         }
48     
49         return input_file;
50     }
51     
52     void src_prefix_split_uri(const std::string& input_file, std::string& protocol, std::string& resource){
53         // Extract function split_uri(input_file, protocol, resource)
54         size_t prefixPos = input_file.find(PROTOCOL_SEPARATOR);
55     
56         if (prefixPos == std::string::npos) {
57             protocol = "";
58             resource = input_file;
59             return;
60         }
61         protocol = input_file.substr(0, prefixPos);
62         resource = input_file.substr(prefixPos + strlen(PROTOCOL_SEPARATOR));
63     }
64     
65     std::string src_prefix_resource(const std::string& input_file) {
66     
67         size_t prefixPos = input_file.find(PROTOCOL_SEPARATOR);
68     
69         if (prefixPos == std::string::npos)
70             return input_file;
71     
72         return input_file.substr(prefixPos + strlen(PROTOCOL_SEPARATOR));
73     }
74     
75     std::string src_prefix_protocol(const std::string& input_file) {
76     
77         size_t prefixPos = input_file.find(PROTOCOL_SEPARATOR);
78     
79         if (prefixPos == std::string::npos)
80             return "";
81     
82         return input_file.substr(0, prefixPos);
83     }
