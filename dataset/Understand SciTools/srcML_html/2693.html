97         /**
98          * meta_tag
99          *
100          * Store a meta tag for later output.
101          */
102          struct meta_tag {
103     
104             /** metatags localname */
105             const char * localname;
106     
107             /** metatags prefix */
108             const char * prefix;
109     
110             /** metatags number of attributes */
111             int num_attributes;
112     
113             /** meta tags attributes */
114             struct srcsax_attribute * attributes;
115     
116             /**
117              * meta_tag
118              * @param localname the meta tag name
119              * @param prefix the meta tag prefix
120              * @param num_attributes the number attributes on the meta tag
121              * @param attributes the attributes on the meta tag
122              *
123              * Construct meta_tag from SAX data.
124              */
125             meta_tag(const char * localname, const char * prefix, int num_attributes, const struct srcsax_attribute * attributes) {
126     
127                 this->localname = localname ? strdup(localname) : 0;
128                 this->prefix = prefix ? strdup(prefix) : 0;
129                 this->num_attributes = num_attributes;
130                 this->attributes = (struct srcsax_attribute *)calloc(num_attributes, sizeof(struct srcsax_attribute));
131                 for(int pos = 0; pos < num_attributes; ++pos) {
132     
133                     this->attributes[pos].localname = attributes[pos].localname ? strdup(attributes[pos].localname) : 0;
134                     this->attributes[pos].prefix = attributes[pos].prefix ? strdup(attributes[pos].prefix) : 0;
135                     this->attributes[pos].uri = attributes[pos].uri ? strdup(attributes[pos].uri) : 0;
136                     this->attributes[pos].value = attributes[pos].value ? strdup(attributes[pos].value) : 0;
137     
138                 }
139     
140             }
141     
142             /**
143              * meta_tag
144              * @param other another meta_tag
145              *
146              * Copy constructor.
147              */
148             meta_tag(const meta_tag & other) {
149     
150                 this->localname = other.localname ? strdup(other.localname) : 0;
151                 this->prefix = other.prefix ? strdup(other.prefix) : 0;
152                 this->num_attributes = other.num_attributes;
153                 this->attributes = (struct srcsax_attribute *)calloc(other.num_attributes, sizeof(struct srcsax_attribute));
154                 for(int pos = 0; pos < other.num_attributes; ++pos) {
155     
156                     this->attributes[pos].localname = other.attributes[pos].localname ? strdup(other.attributes[pos].localname) : 0;
157                     this->attributes[pos].prefix = other.attributes[pos].prefix ? strdup(other.attributes[pos].prefix) : 0;
158                     this->attributes[pos].uri = other.attributes[pos].uri ? strdup(other.attributes[pos].uri) : 0;
159                     this->attributes[pos].value = other.attributes[pos].value ? strdup(other.attributes[pos].value) : 0;
160     
161                 }
162     
163             }
164     
165     
166             /**
167              * operator=
168              * @param other another meta_tag
169              *
170              * Overloaded assignment operator
171              * Returns the assigned to meta_tag
172              */
173             meta_tag & operator=(meta_tag other) {
174     
175     
176                 this->swap(other);
177                 return *this;
178     
179             }
180     
181             /**
182              * swap
183              * @param other another meta_tag
184              *
185              * swap the contents of the meta tags.
186              */
187             void swap(meta_tag & other) {
188     
189                 std::swap(localname, other.localname);
190                 std::swap(prefix, other.prefix);
191                 std::swap(num_attributes, other.num_attributes);
192                 std::swap(attributes, other.attributes);
193     
194             }
195     
196             /**
197              * ~meta_tag
198              *
199              * Destructor
200              */
201             ~meta_tag() {
202     
203                 if(localname) free((void *)localname), localname = 0;
204                 if(prefix) free((void *)prefix), prefix = 0;
205     
206                 if(attributes) {
207     
208                     for(int pos = 0; pos < num_attributes; ++pos) {
209     
210                         if(attributes[pos].localname) free((void *)attributes[pos].localname);
211                         if(attributes[pos].prefix) free((void *)attributes[pos].prefix);
212                         if(attributes[pos].uri) free((void *)attributes[pos].uri);
213                         if(attributes[pos].value) free((void *)attributes[pos].value);
214     
215                     }
216     
217                     free(attributes), attributes = 0;
218     
219                 }
220     
221             }
222     
223          };

243         std::string attribute_revision(const std::string & attribute) {
244     
245             if(!revision) return attribute;
246     
247             std::string::size_type pos = attribute.find('|');
248             if(pos == std::string::npos) return attribute;
249     
250             if(*revision == ORIGINAL) return attribute.substr(0, pos);
251     
252             return attribute.substr(pos + 1, std::string::npos);
253     
254         }

290          * wait
291          *
292          * Allows calling thread to wait until reached
293          * end of unit.
294          */
295         void wait() {
296     
297             boost::unique_lock<boost::mutex> lock(mutex);
298     
299             if(is_done) return;
300     
301             if(wait_root) cond.wait(lock);
302     
303         }

306          * resume

308          * Resume SAX2 execution.

310         void resume() {
311     
312             boost::unique_lock<boost::mutex> lock(mutex);

360         /**
361          * startRoot
362          * @param localname the name of the element tag
363          * @param prefix the tag prefix
364          * @param URI the namespace of tag
365          * @param num_namespaces number of namespaces definitions
366          * @param namespaces the defined namespaces
367          * @param num_attributes the number of attributes on the tag
368          * @param attributes list of attributes
369          *
370          * Overidden startRoot to handle collection of root attributes. Stop before continue
371          */
372         virtual void startRoot(const char * localname, const char * prefix, const char * URI,
373                                int num_namespaces, const struct srcsax_namespace * namespaces, int num_attributes,
374                                const struct srcsax_attribute * attributes) {
375     
376     #ifdef SRCSAX_DEBUG
377             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
378     #endif
379     
380             if(!is_archive) srcml_archive_disable_option(archive, SRCML_OPTION_ARCHIVE);
381     
382             // collect attributes
383             for(int pos = 0; pos < num_attributes; ++pos) {
384     
385                 std::string attribute = attributes[pos].localname;
386                 std::string value = attributes[pos].value;
387     
388                 // Note: these are ignore instead of placing in attributes.
389                 if(attribute == "timestamp")
390                     ;
391                 else if(attribute == "hash")
392                     ;
393                 else if(attribute == "language")
394                     ;
395                 else if(attribute == "revision")
396                     archive->revision = value;
397                 else if(attribute == "filename")
398                     ;
399                 else if(attribute == "url")
400                     srcml_archive_set_url(archive, value.c_str());
401                 else if(attribute == "version")
402                     srcml_archive_set_version(archive, value.c_str());
403                 else if(attribute == "tabs")
404                     archive->tabstop = atoi(value.c_str());
405                 else if(attribute == "options") {
406     
407                     while(!value.empty()) {
408     
409                         std::string::size_type pos = value.find(",");
410                         std::string option = value.substr(0, pos);
411                         if(pos == std::string::npos)
412                             value = "";
413                         else
414                             value = value.substr(value.find(",") + 1);
415     
416                         if(option == "XMLDECL")
417                             archive->options |= SRCML_OPTION_XML_DECL;
418                         else if(option == "NAMESPACEDECL")
419                             archive->options |= SRCML_OPTION_NAMESPACE_DECL;
420                         else if(option == "CPP_TEXT_ELSE")
421                             archive->options |= SRCML_OPTION_CPP_TEXT_ELSE;
422                         else if(option == "CPP_MARKUP_IF0")
423                             archive->options |= SRCML_OPTION_CPP_MARKUP_IF0;
424                         else if(option == "EXPRESSION")
425                             archive->options |= SRCML_OPTION_EXPRESSION;
426                         else if(option == "LINE")
427                             archive->options |= SRCML_OPTION_LINE;
428                         else if(option == "NESTIF")
429                             archive->options |= SRCML_OPTION_NESTIF;
430                         else if(option == "CPPIF_CHECK")
431                             archive->options |= SRCML_OPTION_CPPIF_CHECK;
432                         else if(option == "WRAP_TEMPLATE")
433                             archive->options |= SRCML_OPTION_WRAP_TEMPLATE;
434                         else if(option == "OPERATOR_TERNARY")
435                             archive->options &= ~SRCML_OPTION_TERNARY;
436     
437                     }
438     
439                 } else if(attribute == "hash") 
440                     ;
441                 else {
442     
443                     archive->attributes.push_back(attribute);
444                     archive->attributes.push_back(value);
445     
446                 }
447     
448             }
449     
450             // collect namespaces
451             for(int pos = 0; pos < num_namespaces; ++pos) {
452     
453                 std::string prefix = namespaces[pos].prefix ? namespaces[pos].prefix : "";
454                 std::string uri = namespaces[pos].uri ? namespaces[pos].uri  : "";
455                 srcml_uri_normalize(uri);
456     
457                 if(uri == SRCML_CPP_NS_URI) {
458     
459                     if(archive->language != 0) {
460     
461                         if(*archive->language == "C++" || *archive->language == "C" || *archive->language == "Objective-C")
462                             archive->options |= SRCML_OPTION_CPP | SRCML_OPTION_CPP_NOMACRO;
463                         else if(*archive->language == "C#")
464                             archive->options |= SRCML_OPTION_CPP_NOMACRO;
465     
466                     }
467     
468                 } else if(uri == SRCML_ERR_NS_URI)
469                     archive->options |= SRCML_OPTION_DEBUG;
470                 else if(uri == SRCML_EXT_LITERAL_NS_URI)
471                     archive->options |= SRCML_OPTION_LITERAL;
472                 else if(uri == SRCML_EXT_OPERATOR_NS_URI)
473                     archive->options |= SRCML_OPTION_OPERATOR;
474                 else if(uri == SRCML_EXT_MODIFIER_NS_URI)
475                     archive->options |= SRCML_OPTION_MODIFIER;
476                 else if(uri == SRCML_EXT_POSITION_NS_URI)
477                     archive->options |= SRCML_OPTION_POSITION;
478                 else if(uri == SRCML_EXT_OPENMP_NS_URI)
479                     archive->options |= SRCML_OPTION_OPENMP;
480     
481                 srcml_archive_register_namespace(archive, prefix.c_str(), uri.c_str());
482     
483             }
484     
485     
486     
487     #ifdef SRCSAX_DEBUG
488             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
489     #endif
490     
491         }
492     
493         /**
494          * startUnit
495          * @param localname the name of the element tag
496          * @param prefix the tag prefix
497          * @param URI the namespace of tag
498          * @param num_namespaces number of namespaces definitions
499          * @param namespaces the defined namespaces
500          * @param num_attributes the number of attributes on the tag
501          * @param attributes list of attributes
502          *
503          * Overidden startUnit to handle collection of Unit attributes and tag. Stop before continue
504          * if collecting attributes.
505          */
506         virtual void startUnit(const char * localname, const char * prefix, const char * URI,
507                                int num_namespaces, const struct srcsax_namespace * namespaces, int num_attributes,
508                                const struct srcsax_attribute * attributes) {
509     
510     #ifdef SRCSAX_DEBUG
511             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
512     #endif
513     
514             srcdiff_stack.push(COMMON);
515     
516             // pause
517             // @todo this may need to change because, meta tags have separate call now
518             if(!read_root) {
519     
520                 {
521                     
522                     boost::unique_lock<boost::mutex> lock(mutex);
523                     if(terminate) stop_parser();
524                     wait_root = false;
525                     cond.notify_all();
526                     cond.wait(lock);
527                     read_root = true;
528             
529                 }
530     
531                 if(terminate) {
532     
533                     stop_parser();
534                     return;
535     
536                 }
537     
538             }
539     
540             unit = srcml_unit_create(archive);
541             unit->unit = "";
542     
543             is_empty = true;
544     
545             // collect attributes
546             for(int pos = 0; pos < num_attributes; ++pos) {
547     
548                 std::string attribute = attributes[pos].localname;
549                 std::string value = attribute_revision(attributes[pos].value);
550     
551                 if(attribute == "timestamp")
552                     srcml_unit_set_timestamp(unit, value.c_str());
553                 else if(attribute == "hash")
554                     srcml_unit_set_hash(unit, value.c_str());
555                 else if(attribute == "language")
556                     srcml_unit_set_language(unit, value.c_str());
557                 else if(attribute == "revision")
558                     unit->revision = value;
559                 else if(attribute == "filename")
560                     srcml_unit_set_filename(unit, value.c_str());
561                 else if(attribute == "url")
562                     srcml_unit_set_url(unit, value.c_str());
563                 else if(attribute == "version")
564                     srcml_unit_set_version(unit, value.c_str());
565                 else if(attribute == "tabs" || attribute == "options" || attribute == "hash")
566                     ;
567                 else if(attribute == "src-encoding")
568                     archive->options |= SRCML_OPTION_STORE_ENCODING, srcml_unit_set_src_encoding(unit, value.c_str());
569                 else {
570     
571                     unit->attributes.push_back(attribute);
572                     unit->attributes.push_back(value);
573     
574                 }
575     
576             }
577     
578             if(collect_unit_attributes) {
579     
580                 // pause
581                 boost::unique_lock<boost::mutex> lock(mutex);
582                 if(terminate) stop_parser();
583                 cond.notify_all();
584                 cond.wait(lock);
585     
586             }
587     
588             if(skip) {
589     
590                 get_controller().enable_startElement(false);
591                 get_controller().enable_charactersUnit(false);
592                 get_controller().enable_comment(false);
593                 get_controller().enable_cdataBlock(false);
594     
595             }
596     
597             if(collect_srcml) {
598     
599                 write_startTag(localname, prefix, num_namespaces, namespaces, num_attributes, attributes);
600     
601                 if(!is_archive) {
602     
603                     if(meta_tags.size()) {
604     
605                         *unit->unit += ">";
606                         is_empty = false;
607     
608                     }
609     
610                     for(std::vector<meta_tag>::size_type i = 0; i < meta_tags.size(); ++i) {
611     
612     
613                         try {
614     
615                             meta_tag & meta_tag = meta_tags.at(i);
616                             write_startTag(meta_tag.localname, meta_tag.prefix, 0, 0, meta_tag.num_attributes, meta_tag.attributes);
617                             write_endTag(meta_tag.localname, meta_tag.prefix, true);
618     
619                         } catch(...) { /** @todo handle */ continue; }
620     
621                     }
622     
623                 }
624     
625             }
626     
627             if(terminate) stop_parser();
628     
629     #ifdef SRCSAX_DEBUG
630             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
631     #endif
632     
633         }
634     
635         /**
636          * startElement
637          * @param localname the name of the element tag
638          * @param prefix the tag prefix
639          * @param URI the namespace of tag
640          * @param num_namespaces number of namespaces definitions
641          * @param namespaces the defined namespaces
642          * @param num_attributes the number of attributes on the tag
643          * @param attributes list of attributes
644          *
645          * Overidden startElementNs to handle collection of srcML elements.
646          */
647         virtual void startElement(const char * localname, const char * prefix, const char * URI,
648                                     int num_namespaces, const struct srcsax_namespace * namespaces, int num_attributes,
649                                     const struct srcsax_attribute * attributes) {
650     
651     #ifdef SRCSAX_DEBUG
652             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
653     #endif
654     
655             if(URI && std::string(URI) == SRCML_DIFF_NS_URI) {
656     
657                 std::string local_name(localname);
658     
659                 if(local_name == "common")
660                     srcdiff_stack.push(COMMON);
661                 else if(local_name == "delete")
662                     srcdiff_stack.push(DELETE);
663                 else
664                     srcdiff_stack.push(INSERT);
665     
666             }
667     
668             if(revision) {
669     
670                 if(std::string(URI) == SRCML_DIFF_NS_URI) return;
671                 if(*revision == ORIGINAL && srcdiff_stack.top() == INSERT) return;
672                 if(*revision == MODIFIED && srcdiff_stack.top() == DELETE) return;
673     
674             }
675     
676             if(collect_src && localname[0] == 'e' && localname[1] == 's'
677                && strcmp((const char *)localname, "escape") == 0) {
678     
679                 char value = (int)strtol((const char*) attributes[0].value, NULL, 0);
680     
681                 charactersUnit(&value, 1);
682     
683     
684             }
685     
686             if(is_empty && collect_srcml) *unit->unit += ">";
687             is_empty = true;
688     
689             if(collect_srcml) {
690     
691                 write_startTag(localname, prefix, num_namespaces, namespaces, num_attributes, attributes);
692     
693             }
694     
695             if(terminate) stop_parser();
696     
697     #ifdef SRCSAX_DEBUG
698             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
699     #endif
700     
701         }
702     
703         /**
704          * endRoot
705          * @param localname tag name
706          * @param prefix prefix for the tag
707          * @param URI uri for tag
708          *
709          * Overidden endRoot to indicate done with parsing and free any waiting process.
710          */
711         virtual void endRoot(const char * localname, const char * prefix, const char * URI) {
712     
713     #ifdef SRCSAX_DEBUG
714             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
715     #endif
716             if(!read_root) {
717     
718                 {
719                     
720                     boost::unique_lock<boost::mutex> lock(mutex);
721                     if(terminate) stop_parser();
722                     wait_root = false;
723                     cond.notify_all();
724                     cond.wait(lock);
725                     read_root = true;
726             
727                 }
728     
729                 if(terminate) {
730     
731                     stop_parser();
732                     return;
733     
734                 }
735     
736             }
737     
738             {
739                 boost::unique_lock<boost::mutex> lock(mutex);
740                 if(terminate) stop_parser();
741                 is_done = true;
742                 cond.notify_all();
743             }
744     
745             if(terminate) stop_parser();
746     
747     #ifdef SRCSAX_DEBUG
748             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
749     #endif
750     
751         }
752     
753         /**
754          * endUnit
755          * @param localname tag name
756          * @param prefix prefix for the tag
757          * @param URI uri for tag
758          *
759          * Overidden endUnit to collect srcml and stop parsing.  Clear collect srcML after pause.
760          */
761         virtual void endUnit(const char * localname, const char * prefix, const char * URI) {
762     
763     #ifdef SRCSAX_DEBUG
764             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
765     #endif
766     
767             srcdiff_stack.pop();
768     
769             if(skip) {
770     
771                 get_controller().enable_startElement(true);
772                 get_controller().enable_charactersUnit(true);
773                 get_controller().enable_comment(true);
774                 get_controller().enable_cdataBlock(true);
775     
776             }
777     
778     
779             //if(is_empty) *unit->unit += ">";
780             if(collect_srcml) {
781     
782                 write_endTag(localname, prefix, is_empty);
783     
784             }
785     
786             if(collect_srcml || collect_src) {
787     
788                 // pause
789                 boost::unique_lock<boost::mutex> lock(mutex);
790                 if(terminate) stop_parser();
791                 cond.notify_all();
792                 cond.wait(lock);
793     
794             }
795     
796             is_empty = false;
797     
798             srcml_unit_free(unit);
799             unit = 0;
800     
801             if(terminate) stop_parser();
802     
803     #ifdef SRCSAX_DEBUG
804             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
805     #endif
806     
807         }
808     
809         /**
810          * endElementNs
811          * @param localname tag name
812          * @param prefix prefix for the tag
813          * @param URI uri for tag
814          *
815          * Overidden endElementNs to collect srcML.
816          */
817         virtual void endElement(const char * localname, const char * prefix, const char * URI) {
818     
819     #ifdef SRCSAX_DEBUG
820             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
821     #endif
822     
823             if(!skip && URI && std::string(URI) == SRCML_DIFF_NS_URI)
824                 srcdiff_stack.pop();
825     
826             if(revision) {
827     
828                 if(std::string(URI) == SRCML_DIFF_NS_URI) return;
829                 if(*revision == ORIGINAL && srcdiff_stack.top() == INSERT) return;
830                 if(*revision == MODIFIED && srcdiff_stack.top() == DELETE) return;
831     
832             }
833     
834             if(collect_srcml) {
835     
836                 write_endTag(localname, prefix, is_empty);
837             }
838     
839             is_empty = false;
840     
841             if(terminate) stop_parser();
842     
843     #ifdef SRCSAX_DEBUG
844             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, (const char *)localname);
845     #endif
846     
847         }

849         /**
850          * charactersUnit
851          * @param ch the characters
852          * @param len length of the characters
853          *
854          * Overidden charactersUnit to collect srcML.
855          */
856         virtual void charactersUnit(const char * ch, int len) {
857     
858     #ifdef SRCSAX_DEBUG
859             std::string chars;
860             chars.append((const char *)ch, len);
861             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, chars.c_str());
862     #endif
863     
864             if(revision) {
865     
866                 if(*revision == ORIGINAL && srcdiff_stack.top() == INSERT) return;
867                 if(*revision == MODIFIED && srcdiff_stack.top() == DELETE) return;
868     
869             }        
870     
871             if(is_empty && collect_srcml) *unit->unit += ">";
872             is_empty = false;
873     
874             if(collect_src) {
875     
876                 xmlOutputBufferWrite(output_buffer, len, (const char *)ch);
877     
878             } else {
879     
880                 for(int i = 0; i < len; ++i) {
881                     char character = (char)ch[i];
882     
883                     if(character == '&')
884                         (*unit->unit) += "&amp;";
885                     else if(character == '<')
886                         (*unit->unit) += "&lt;";
887                     else if(character == '>')
888                         (*unit->unit) += "&gt;";
889                     else
890                         (*unit->unit) += character;
891                 }
892     
893             }
894     
895             if(terminate) stop_parser();
896     
897     #ifdef SRCSAX_DEBUG
898             fprintf(stderr, "HERE: %s %s %d '%s'\n", __FILE__, __FUNCTION__, __LINE__, chars.c_str());
899     #endif
900     
901         }

903         /**
904          * metaTag
905          * @param localname the name of the element tag
906          * @param prefix the tag prefix
907          * @param URI the namespace of tag
908          * @param num_namespaces number of namespaces definitions
909          * @param namespaces the defined namespaces
910          * @param num_attributes the number of attributes on the tag
911          * @param attributes list of attributes\
912          *
913          * SAX handler function for a meta tags.
914          * Overide for desired behaviour.
915          */
916         virtual void metaTag(const char * localname, const char * prefix, const char * URI,
917                                int num_namespaces, const struct srcsax_namespace * namespaces, int num_attributes,
918                                const struct srcsax_attribute * attributes) {
919     
920             if(strcmp(localname, "macro-list") == 0) {
921     
922                 std::string token("");
923                 std::string type("");
924     
925                 for(int pos = 0; pos < num_attributes; ++pos) {
926     
927                     if(strcmp(attributes[pos].localname, "token") == 0)
928                         token = attributes[pos].value;
929                     else if(strcmp(attributes[pos].localname, "type") == 0)
930                         type = attributes[pos].value;
931     
932     
933                 }
934     
935                 if(token != "" && type != "") {
936     
937                     archive->user_macro_list.push_back(token);
938                     archive->user_macro_list.push_back(type);
939     
940                 }
941     
942             }
943     
944             if(!is_archive) {
945     
946                 meta_tags.push_back(meta_tag(localname, prefix, num_attributes, attributes));
947     
948             }
949     
950         }

1007                 *unit->unit += "\"";
1008
