41     /**
42      * srcml_extract_text
43      * @param input_buffer srcML to extract text
44      * @param size size of input buffer
45      * @param output_buffer output buffer to write text
46      * @param options srcml options
47      * @param unit unit number to extract
48      *
49      * Extract a given unit from supplied srcML input buffer.
50      *
51      * @returns Return SRCML_STATUS_OK on success and SRCML_STATUS_INVALID_ARGUMENT on failure.
52      */
53     int srcml_extract_text(const char * input_buffer, size_t size, xmlOutputBufferPtr output_buffer, OPTION_TYPE options, int unit) {
54     
55         if(input_buffer == NULL || size == 0) return SRCML_STATUS_INVALID_ARGUMENT;
56     
57         xmlParserInputBufferPtr input = xmlParserInputBufferCreateMem(input_buffer, (int)size, XML_CHAR_ENCODING_NONE);
58     
59         if(input == NULL) return SRCML_STATUS_IO_ERROR;
60     
61         srcml_sax2_reader reader(input);
62         reader.read_src(output_buffer);
63     
64         
65         xmlFreeParserInputBuffer(input);
66     
67         return SRCML_STATUS_OK;
68     
69     }

71     /**
72      * srcml_extract_text_filename
73      * @param ifilename name of srcML file to extract text
74      * @param ofilename name of output file to put source
75      * @param encoding output encoding
76      * @param compression amount of compression 0 none to 9 max
77      * @param unit unit number to extract
78      *
79      * Extract a given unit from supplied srcML directly to file.
80      *
81      * @returns Return SRCML_STATUS_OK on success and a status error code on failure.
82      */
83     int srcml_extract_text_filename(const char * ifilename, const char * ofilename, const char * encoding, unsigned short compression, int unit) {
84     
85         if(compression > 9) compression = 9;
86     
87         xmlOutputBufferPtr output_buffer = xmlOutputBufferCreateFilename(ofilename, xmlFindCharEncodingHandler(encoding), compression);
88     
89         srcml_sax2_reader reader(ifilename);
90         reader.read_src(output_buffer);
91     
92         xmlOutputBufferClose(output_buffer);
93     
94         return SRCML_STATUS_OK;
95     
96     }

98     /**
99      * srcml_xpath
100      * @param input_buffer a parser input buffer
101      * @param context_element a srcML element that is to be used as the context
102      * @param xpath the xpath expression
103      * @param prefix an element namepace prefix
104      * @param uri an element namepace uri
105      * @param element the tag name
106      * @param attr_prefix an attribute namepace prefix
107      * @param attr_uri an attribute namepace uri
108      * @param attr_name the attribute name
109      * @param attr_value the attribute value
110      * @param options srcml options
111      *
112      * XPath evaluation of the nested units.
113      *
114      * @returns Return SRCML_STATUS_OK on success and a status error code on failure.
115      */
116     int srcml_xpath(xmlParserInputBufferPtr input_buffer, const char * context_element, const char * xpath,
117                     const char * prefix, const char * uri, const char * element, const char * attr_prefix, const char * attr_uri, const char * attr_name, const char * attr_value,
118                     OPTION_TYPE options, srcml_archive* out_archive) {
119     
120         if(input_buffer == NULL || context_element == NULL ||
121            xpath == NULL) return SRCML_STATUS_INVALID_ARGUMENT;
122     
123         // relative xpath changed to at any level
124         std::string s = xpath;
125         //  if (s[0] != '/')
126         //    s = "//" + s;
127     
128         // compile the xpath that will be applied to each unit
129         xmlXPathCompExprPtr compiled_xpath = xmlXPathCompile(BAD_CAST s.c_str());
130         if (compiled_xpath == 0) {
131             fprintf(stderr, "libsrcml:  Unable to compile XPath '%s'\n", s.c_str());
132             return SRCML_STATUS_INVALID_INPUT;
133         }
134     
135         // setup process handling
136         xpath_query_units process(options, compiled_xpath, out_archive, prefix, uri, element, attr_prefix, attr_uri, attr_name, attr_value);
137         srcSAXController control(input_buffer);
138     
139         try {
140     
141             control.parse(&process);
142     
143         } catch(SAXError error) {
144     
145             fprintf(stderr, "Error Parsing: %s\n", error.message.c_str());
146     
147         }
148     
149         xmlXPathFreeCompExpr(compiled_xpath);
150     
151         return 0;//status;
152     
153     }

156     /**
157      * dlexsltRegisterAll
158      *
159      * Allow for all exslt functions by dynamic load
160      * of exslt library.
161      */
162     void dlexsltRegisterAll(void * handle) {
163     
164     #if defined(__GNUG__) && !defined(__MINGW32__) && !defined(NO_DLLOAD)
165         typedef void * __attribute__ ((__may_alias__)) VOIDPTR;
166         typedef void (*exsltRegisterAll_function)();
167         dlerror();
168         exsltRegisterAll_function exsltRegisterAll;
169         *(VOIDPTR *)(&exsltRegisterAll) = dlsym(handle, "exsltRegisterAll");
170         char* error;
171         if ((error = dlerror()) != NULL) {
172             dlclose(handle);
173             return;
174         }
175     
176         // allow for all exslt functions
177         exsltRegisterAll();
178     
179     #endif
180     
181     }

184     /**
185      * srcml_xslt
186      * @param input_buffer a parser input buffer
187      * @param context_element a srcml element to be used as the context
188      * @param xslt xmlDocPtr containing an XSLT program
189      * @param params NULL-terminated list of XSLT parameters
190      * @param paramcount number of XSLT parameters
191      * @param options srcml options
192      *
193      * XSLT evaluation of the nested units.
194      *
195      * @returns Return SRCML_STATUS_OK on success and a status error code on failure.
196      */
197     int srcml_xslt(xmlParserInputBufferPtr input_buffer, const char* context_element, xmlDocPtr xslt, const char* params[], int paramcount, OPTION_TYPE options,
198                     srcml_archive* out_archive) {
199     
200         if(input_buffer == NULL || context_element == NULL ||
201            xslt == NULL) return SRCML_STATUS_INVALID_ARGUMENT;
202     
203         xmlInitParser();
204     
205     #if defined(__GNUG__) && !defined(__MINGW32__) && !defined(NO_DLLOAD)
206         typedef void * __attribute__ ((__may_alias__)) VOIDPTR;
207         typedef xsltStylesheetPtr (*xsltParseStylesheetDoc_function) (xmlDocPtr);
208         typedef void (*xsltCleanupGlobals_function)();
209         typedef void (*xsltFreeStylesheet_function)(xsltStylesheetPtr);
210     
211         void* handle = dlopen("libexslt.so", RTLD_LAZY);
212         if (!handle) {
213             handle = dlopen("libexslt.so.0", RTLD_LAZY);
214             if (!handle) {
215                 handle = dlopen("libexslt.dylib", RTLD_LAZY);
216                 if (!handle) {
217                     fprintf(stderr, "Unable to open libexslt library\n");
218                     return SRCML_STATUS_ERROR;
219                 }
220             }
221         }
222     
223         // allow for all exstl functions
224         dlexsltRegisterAll(handle);
225     
226         dlerror();
227         xsltParseStylesheetDoc_function xsltParseStylesheetDoc;
228         *(VOIDPTR *)(&xsltParseStylesheetDoc) = dlsym(handle, "xsltParseStylesheetDoc");
229         char* error;
230         if ((error = dlerror()) != NULL) {
231             dlclose(handle);
232             return SRCML_STATUS_ERROR;
233         }
234     
235         dlerror();
236         xsltCleanupGlobals_function xsltCleanupGlobals;
237         *(VOIDPTR *)(&xsltCleanupGlobals) = dlsym(handle, "xsltCleanupGlobals");
238         if ((error = dlerror()) != NULL) {
239             dlclose(handle);
240             return SRCML_STATUS_ERROR;
241         }
242     
243         dlerror();
244         xsltFreeStylesheet_function xsltFreeStylesheet;
245         *(VOIDPTR *)(&xsltFreeStylesheet) = dlsym(handle, "xsltFreeStylesheet");
246         if ((error = dlerror()) != NULL) {
247             dlclose(handle);
248             return SRCML_STATUS_ERROR;
249         }
250     #endif
251     
252         // parse the stylesheet
253         xsltStylesheetPtr stylesheet = xsltParseStylesheetDoc(xslt);
254         if (!stylesheet)
255             return SRCML_STATUS_ERROR;
256     
257     
258     
259         xsltsrcMLRegister();
260     
261         // setup process handling
262         xslt_units process(context_element, options, stylesheet, params, out_archive);
263         srcSAXController control(input_buffer);
264     
265         try {
266     
267             control.parse(&process);
268     
269         } catch(SAXError error) {
270     
271             fprintf(stderr, "Error Parsing: %s\n", error.message.c_str());
272     
273         }
274     
275         stylesheet->doc = 0;
276         xsltFreeStylesheet(stylesheet);
277         xsltCleanupGlobals();
278     
279     #if defined(__GNUG__) && !defined(__MINGW32__) && !defined(NO_DLLOAD)
280         dlclose(handle);
281     #endif
282     
283         return 0;//status;
284     
285     }
286     #endif

288     /**
289      * srcml_relaxng
290      * @param input_buffer a parser input buffer
291      * @param relaxng xmlDocPtr containing a RelaxNG schema
292      * @param fd output file descriptor
293      * @param options srcml options
294      *
295      * RelaxNG evaluation of the nested units.
296      *
297      * @returns Returns SRCML_STATUS_OK on success and a status error code on failure.
298      */
299     int srcml_relaxng(xmlParserInputBufferPtr input_buffer, xmlDocPtr relaxng, OPTION_TYPE options, srcml_archive* oarchive) {
300     
301         if(input_buffer == NULL || relaxng == NULL) return SRCML_STATUS_INVALID_ARGUMENT;
302     
303         xmlRelaxNGParserCtxtPtr relaxng_parser_ctxt = xmlRelaxNGNewDocParserCtxt(relaxng);
304         xmlRelaxNGPtr rng = xmlRelaxNGParse(relaxng_parser_ctxt);
305         xmlRelaxNGValidCtxtPtr rngctx = xmlRelaxNGNewValidCtxt(rng);
306     
307         relaxng_units process(options, rngctx, oarchive);
308         srcSAXController control(input_buffer);
309     
310         try {
311     
312             control.parse(&process);
313     
314         } catch(SAXError error) {
315     
316             fprintf(stderr, "Error Parsing: %s\n", error.message.c_str());
317     
318         }
319     
320         xmlRelaxNGFreeValidCtxt(rngctx);
321         xmlRelaxNGFree(rng);
322         xmlRelaxNGFreeParserCtxt(relaxng_parser_ctxt);
323     
324         return 0;//status;
325     
326     }
