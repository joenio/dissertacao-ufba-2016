21     /*
22       Example program of the use of the C API for srcML.
23     
24       Create an archive, file by file, with an output file descriptor
25     */
26     
27     #include "srcml.h"
28     #include <fcntl.h>
29     #include <sys/stat.h>
30     #ifdef _MSC_BUILD
31     #include <io.h>
32     #else
33     #include <unistd.h>
34     #endif
35     
36     #include <srcml_macros.hpp>
37     
38     
39     int main(int argc, char* argv[]) {
40         int i;
41         int srcml_output;
42         int srcml_input;
43         struct srcml_archive* archive;
44         struct srcml_unit* unit;
45     
46         /* create a new srcml archive structure */
47         archive = srcml_archive_create();
48     
49         /* setup our output file using a file descriptor */
50         srcml_output = OPEN("project.xml", O_WRONLY | O_CREAT, S_IRUSR | S_IWUSR);
51     
52         /* open a srcML archive for output */
53         srcml_archive_write_open_fd(archive, srcml_output);
54     
55         /* add all the files to the archive */
56         for (i = 1; i < argc; ++i) {
57     
58             unit = srcml_unit_create(archive);
59     
60             srcml_unit_set_language(unit, srcml_archive_check_extension(archive, argv[i]));
61     
62             /* Translate to srcml */
63             srcml_input = OPEN(argv[i], O_RDONLY, 0);
64             srcml_unit_parse_fd(unit, srcml_input);
65     
66             /* Append to the archive */
67             srcml_write_unit(archive, unit);
68     
69             srcml_unit_free(unit);
70             CLOSE(srcml_input);
71         }
72     
73         /* close the srcML archive */
74         srcml_archive_close(archive);
75     
76         /* file can now be closed also */
77         CLOSE(srcml_output);
78     
79         /* free the srcML archive data */
80         srcml_archive_free(archive);
81     
82         return 0;
83     }
