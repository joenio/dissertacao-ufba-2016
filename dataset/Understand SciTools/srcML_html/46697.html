23     #include <compress_srcml.hpp>
24     #include <archive.h>
25     #include <archive_entry.h>
26     #include <boost/foreach.hpp>
27     
28     #if ARCHIVE_VERSION_NUMBER > 3001002
29     void compress_srcml(const srcml_request_t& /* srcml_request */,
30                         const srcml_input_t& input_sources,
31                         const srcml_output_dest& destination) {
32     
33         archive* ar = archive_write_new();
34     
35         archive_write_set_format_raw(ar);
36     
37         // setup compressions
38         BOOST_FOREACH(const std::string& ext, destination.compressions)
39             archive_write_set_compression_by_extension(ar, ext.c_str());
40     
41         int status = ARCHIVE_OK;
42         if (contains<int>(destination)) {
43             status = archive_write_open_fd(ar, destination);
44         } else {
45             status = archive_write_open_filename(ar, destination.resource.c_str());
46         }
47         if (status != ARCHIVE_OK) {
48             std::cerr << status;
49             exit(1);
50         }
51     
52         archive_entry* entry = archive_entry_new();
53     
54         archive_entry_set_pathname(entry, "test");
55         archive_entry_set_filetype(entry, AE_IFREG);
56     
57         if ((status = archive_write_header(ar, entry)) != ARCHIVE_OK) {
58             std::cerr << status;
59             exit(1);
60         }
61     
62         // write the data into the archive
63         std::vector<char> buffer(4092);
64         while (ssize_t s = read(*input_sources[0].fd, &buffer.front(), buffer.size())) {
65     
66             ssize_t status = archive_write_data(ar, &buffer.front(), s);
67             if (status == 0)
68                 break;
69         }
70     
71         archive_write_close(ar);
72         archive_write_free(ar);
73     }
74     #endif
