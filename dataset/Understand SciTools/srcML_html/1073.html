23     #ifndef PARSE_QUEUE_HPP
24     #define PARSE_QUEUE_HPP
25     
26     #include <parse_request.hpp>
27     #pragma GCC diagnostic push
28     #pragma GCC diagnostic ignored "-Wshorten-64-to-32"
29     #include <threadpool.hpp>
30     #pragma GCC diagnostic pop
31     #include <boost/function.hpp>
32     #include <write_queue.hpp>
33     
34     class ParseQueue {
35     public:
36     
37     	ParseQueue(int max_threads, boost::function<void(ParseRequest*)> consumearg, WriteQueue& write_queue)
38     	    : consume(consumearg), pool(max_threads), counter(0), wqueue(write_queue) {}
39     
40     	inline void schedule(ParseRequest* pvalue) {
41     
42     	    pvalue->position = ++counter;
43     
44     	    // error passthrough to output for proper output in trace
45     	    if (pvalue->status) {
46     	        pvalue->unit = 0;
47     	        wqueue.schedule(pvalue);
48     	        return;
49     	    }
50     
51     	    pool.schedule( boost::bind(consume, pvalue));
52     	}
53     
54     	inline void wait() {
55     
56     		pool.wait();
57     	}
58     
59     private:
60         boost::function<void(ParseRequest*)> consume;
61         boost::threadpool::pool pool;
62         int counter;
63         WriteQueue& wqueue;
64     };
65     
66     #endif
