54         bool curl_supported(const std::string& input_protocol) {
55             const char* const* curl_types = curl_version_info(CURLVERSION_NOW)->protocols;
56             for (int i = 0; curl_types[i] != NULL; ++i) {
57                 if (strcmp(curl_types[i], input_protocol.c_str()) == 0)
58                     return true;
59             }
60             return false;
61         }
62     }

64     // Convert input to a ParseRequest and assign request to the processing queue
65     void src_input_libarchive(ParseQueue& queue,
66                               srcml_archive* srcml_arch,
67                               const srcml_request_t& srcml_request,
68                               const srcml_input_src& input_file) {
69     
70         // don't process if non-archive, non-compressed, and we don't handle the extension
71         // this is to prevent trying to open, with srcml_archive_open_filename(), a non-srcml file,
72         // which then hangs
73         // Note: may need to fix in libsrcml
74         if (!contains<int>(input_file) && !contains<FILE*>(input_file) && input_file.compressions.empty() && input_file.archives.empty() && !srcml_check_extension(input_file.plainfile.c_str())) {
75     
76             // if we are not verbose, then just end this attemp
77             if (!(SRCML_COMMAND_VERBOSE & SRCMLOptions::get())) {
78                 return;
79             }
80     
81             // form the parsing request
82             ParseRequest* prequest = new ParseRequest;
83             prequest->filename = input_file.resource;
84             prequest->url = srcml_request.att_url;
85             prequest->version = srcml_request.att_version;
86             prequest->srcml_arch = srcml_arch;
87             prequest->language = "";
88             prequest->status = SRCML_STATUS_UNSET_LANGUAGE;
89     
90             // schedule for parsing
91             queue.schedule(prequest);
92     
93             return;
94         }
95     
96         archive* arch = archive_read_new();
97     
98         archive_read_support_format_ar(arch);
99         archive_read_support_format_cpio(arch);
100         archive_read_support_format_gnutar(arch);
101         archive_read_support_format_iso9660(arch);
102         archive_read_support_format_mtree(arch);
103         archive_read_support_format_tar(arch);
104         archive_read_support_format_xar(arch);
105         archive_read_support_format_zip(arch);
106         archive_read_support_format_raw(arch);
107         archive_read_support_format_empty(arch);
108     
109         /* Check libarchive version enable version specific features/syntax */
110     #if ARCHIVE_VERSION_NUMBER < 3000000
111         // V2 Only Settings
112         // Compressions
113         archive_read_support_compression_all(arch);
114     #else
115         // V3 Only Settings
116         // File Formats
117         archive_read_support_format_7zip(arch);
118         archive_read_support_format_cab(arch);
119         archive_read_support_format_lha(arch);
120         archive_read_support_format_rar(arch);
121     
122         // Compressions
123         archive_read_support_filter_all(arch);
124     #endif
125     
126         int status;
127         curl curling;
128         if (contains<int>(input_file)) {
129     
130             status = archive_read_open_fd(arch, input_file, 16384);
131     
132         } else if (contains<FILE*>(input_file)) {
133     
134             status = archive_read_open_FILE(arch, input_file);
135     
136         } else if (input_file.protocol != "file" && curl_supported(input_file.protocol)) {
137     
138             curling.source = input_file.filename;
139             status = archive_read_open(arch, &curling, archive_curl_open, archive_curl_read, archive_curl_close);
140     
141         } else {
142     
143             status = archive_read_open_filename(arch, input_file.c_str(), 16384);
144         }
145     
146         if (status != ARCHIVE_OK) {
147             std::cerr << "Unable to open file " << input_file.filename << '\n';
148             exit(1);
149         }
150     
151         /* In general, go through this once for each time the header can be read
152            Exception: if empty, go through the loop exactly once */
153         int count = 0;
154         archive_entry *entry;
155         while (status == ARCHIVE_OK &&
156                (((status = archive_read_next_header(arch, &entry)) == ARCHIVE_OK) ||
157                 (status == ARCHIVE_EOF && !count))) {
158     
159             // skip any directories
160             if (status == ARCHIVE_OK && archive_entry_filetype(entry) == AE_IFDIR)
161                 continue;
162     
163             // default is filename from archive entry (if not empty)
164             std::string filename = status == ARCHIVE_OK ? archive_entry_pathname(entry) : "";
165     
166             // stdin, single files require a explicit filename
167             if (filename == "data" && !srcml_request.att_language && input_file.filename == "stdin://-") {
168                 std::cerr << "Language required for stdin single files" << '\n';
169                 exit(1);
170             }
171     
172             if (count == 0 && filename != "data" && status != ARCHIVE_EOF) {
173                 srcml_archive_enable_option(srcml_arch, SRCML_OPTION_ARCHIVE);
174                 srcml_archive_enable_option(srcml_arch, SRCML_OPTION_HASH);
175             }
176     
177             // archive entry filename for non-archive input is "data"
178             if (filename.empty() || filename == "data")
179                 filename = input_file.resource;
180     
181             if (srcml_request.att_filename && !(srcml_archive_get_options(srcml_arch) & SRCML_OPTION_ARCHIVE))
182                 filename = *srcml_request.att_filename;
183     
184             // language may have been explicitly set
185             std::string language;
186     
187             if (srcml_request.att_language)
188                 language = *srcml_request.att_language;
189     
190             // if not explicitly set, language comes from extension
191             // we have to do this ourselves, since libsrcml can't for memory
192             if (language.empty())
193                 if (const char* l = srcml_archive_check_extension(srcml_arch, filename.c_str()))
194                     language = l;
195     
196             // if we don't have a language, and are not verbose, then just end this attemp
197             if (language.empty() && !(SRCML_COMMAND_VERBOSE & SRCMLOptions::get())) {
198                 ++count;
199                 continue;
200             }
201     
202             // form the parsing request
203             ParseRequest* prequest = new ParseRequest;
204     
205             if (srcml_request.command & SRCML_COMMAND_NOARCHIVE)
206                 prequest->disk_dir = srcml_request.output_filename;
207     
208             if (srcml_request.att_filename || (filename != "-"))
209                 prequest->filename = filename;
210     
211     
212             prequest->url = srcml_request.att_url;
213             prequest->version = srcml_request.att_version;
214             prequest->srcml_arch = srcml_arch;
215             prequest->language = language;
216             prequest->status = !language.empty() ? 0 : SRCML_STATUS_UNSET_LANGUAGE;
217     
218             if (SRCML_COMMAND_TIMESTAMP & SRCMLOptions::get()) {
219     
220                 //Long time provided by libarchive needs to be time_t
221                 time_t mod_time(archive_entry_mtime(entry));
222     
223                 //Standard ctime output and prune '/n' from string
224                 char* c_time = ctime(&mod_time);
225                 c_time[strlen(c_time) - 1] = 0;
226                 
227                 prequest->time_stamp = c_time;
228             }
229     
230             // fill up the parse request buffer
231             if (!status && !prequest->status) {
232                 // if we know the size, create the right sized data_buffer
233                 if (archive_entry_size_is_set(entry))
234                     prequest->buffer.reserve(archive_entry_size(entry));
235     
236                 const char* buffer;
237                 size_t size;
238     #if ARCHIVE_VERSION_NUMBER < 3000000
239                 off_t offset;
240     #else
241                 int64_t offset;
242     #endif
243                 while (status == ARCHIVE_OK && archive_read_data_block(arch, (const void**) &buffer, &size, &offset) == ARCHIVE_OK)
244                     prequest->buffer.insert(prequest->buffer.end(), buffer, buffer + size);
245     
246                 // LOC count
247                 prequest->loc = std::count(prequest->buffer.begin(), prequest->buffer.end(), '\n');
248                 if (!prequest->buffer.empty() && prequest->buffer.back() != '\n')
249                     ++prequest->loc;
250             }
251             
252             // schedule for parsing
253             queue.schedule(prequest);
254     
255             ++count;
256         }
257         archive_read_finish(arch);
258     }

262         size_t curl_cb(void* buffer, size_t len, size_t nmemb, void* data) {
263     
264             curl* curldata = (curl*) data;
265     
266             curldata->data_buffer = (char*)buffer;
267             curldata->data_len = len * nmemb;
268     
269             return curldata->data_len;
270         }

272         int archive_curl_open(archive*, void* client_data) {
273     
274             curl* curldata = (curl*) client_data;
275     
276             curl_global_init(CURL_GLOBAL_DEFAULT);
277     
278             curldata->handle = curl_easy_init();
279     
280             curl_easy_setopt(curldata->handle, CURLOPT_SSL_VERIFYPEER, 0L);
281             curl_easy_setopt(curldata->handle, CURLOPT_SSL_VERIFYHOST, 0L);
282             curl_easy_setopt(curldata->handle, CURLOPT_HTTPAUTH, (long)CURLAUTH_ANY);
283             curl_easy_setopt(curldata->handle, CURLOPT_FOLLOWLOCATION, 1L);
284             curl_easy_setopt(curldata->handle, CURLOPT_VERBOSE, 0L);
285             curl_easy_setopt(curldata->handle, CURLOPT_WRITEFUNCTION, curl_cb);
286             curl_easy_setopt(curldata->handle, CURLOPT_WRITEDATA, curldata);
287             curl_easy_setopt(curldata->handle, CURLOPT_URL, curldata->source.c_str());
288     
289             // Quick check to see if the remote location exists or is available
290             CURL* ping = curl_easy_duphandle(curldata->handle);
291             curl_easy_setopt(ping, CURLOPT_NOBODY, 1L);
292             //curl_easy_setopt(ping, CURLOPT_HEADER, 1L);
293             curl_easy_perform(ping);
294             
295             long http_code = 0;
296             double data_size = 0;
297             curl_easy_getinfo (ping, CURLINFO_RESPONSE_CODE, &http_code);
298             curl_easy_getinfo (ping, CURLINFO_CONTENT_LENGTH_DOWNLOAD, &data_size);
299     
300             curl_easy_cleanup(ping);
301             if (http_code != 200)
302             {   
303                 return -1;
304             }
305     
306             // The resource is there, so lets go get it!
307             curldata->multi_handle = curl_multi_init();
308             curl_multi_add_handle(curldata->multi_handle, curldata->handle);
309             curl_multi_perform(curldata->multi_handle, &curldata->still_running);
310             curldata->stopwatch = Timer(10); // 10 second timeout
311     
312             return ARCHIVE_OK;
313         }
314     
315         ssize_t archive_curl_read(archive*, void* client_data, const void** buff) {
316     
317             curl* curldata = (curl*) client_data;
318     
319             curldata->data_len = 0;
320             curldata->stopwatch.start();
321             while (curldata->data_len == 0 && curldata->still_running) {
322                 curl_multi_perform(curldata->multi_handle, &curldata->still_running);
323                
324                // Connection lost mid-transfer time to give up
325                 if (curldata->stopwatch.is_expired()) {
326                     curldata->data_len = 0;
327                     curldata->data_buffer = 0;
328                     break;
329                 }
330             }
331     
332             *buff = curldata->data_buffer;
333     
334             return curldata->data_len;
335         }
336     
337         int archive_curl_close(archive*, void* client_data) {
338     
339             curl* curldata = (curl*) client_data;
340             curl_multi_cleanup(curldata->multi_handle);
341             curl_easy_cleanup(curldata->handle);
342     
343             return 0;
344         }
