33     /**
34      * thread_args
35      *
36      * Structure to hold information to pass
37      * to thread function.
38      */
39     struct thread_args {
40     
41         /** control for sax processing */
42         srcSAXController * control;
43     
44         /** handler with hooks for sax processing */
45         srcml_reader_handler * handler;
46     
47     };
48     
49     /**
50      * start_routine
51      * @param arguments thread_args structure with control and handler
52      *
53      * Starts the parsing of the document.
54      */
55     void * start_routine(void * arguments) {
56     
57         thread_args * args = (thread_args *)arguments;
58     
59         try {
60             args->control->parse(args->handler);
61         } catch(SAXError error) {
62     
63             if(!(error.error_code == XML_ERR_EXTRA_CONTENT || error.error_code == XML_ERR_DOCUMENT_END)) {
64                 fprintf(stderr, "Error Parsing: %s\n", error.message.c_str());
65                 args->handler->stop();
66             }
67     
68             // might have to release a lock here or set is_done;
69         }
70     
71         return 0;
72     
73     }
74     
75     void srcml_sax2_reader::init_constructor() {
76     
77     
78         thread_args args = { &control, &handler };
79     
80         thread = new boost::thread(start_routine, &args);
81         handler.wait();
82     
83     }

125     /**
126      * read_root_unit_attributes
127      * @param language a location to store the language attribute
128      * @param url a location to store the url attribute
129      * @param version a location to store the version attribute
130      * @param attributes array to store other attributes gathered
131      * @param prefixes an array to store gathered XML namespace prefixes
132      * @param namespaces an array to store gathered XML naamespaces
133      * @param processing_instruction a location to store the pre-root processing-instruction
134      * @param options a variable to set used options
135      * @param tabstop a variable to set the tabstop
136      * @param user_macro_list a variable to set the list of user defined macros
137      *
138      * Read attributes and namespace information fromt the root unit,
139      * setting the necessary options.
140      *
141      * @returns 1 on success and 0 on failure.
142      */
143     int srcml_sax2_reader::read_root_unit_attributes(boost::optional<std::string> & encoding,
144                                                      boost::optional<std::string> & language,
145                                                      boost::optional<std::string> & url, boost::optional<std::string> & version,
146                                                      std::vector<std::string> & attributes,
147                                                      std::vector<std::string> & prefixes,
148                                                      std::vector<std::string> & namespaces,
149                                                      boost::optional<std::pair<std::string, std::string> > & processing_instruction,
150                                                      OPTION_TYPE & options,
151                                                      size_t & tabstop,
152                                                      std::vector<std::string> & user_macro_list) {
153     
154         if(read_root || handler.read_root) return 0;
155     
156         encoding.swap(handler.archive->encoding);
157         language.swap(handler.archive->language);
158         url.swap(handler.archive->url);
159         version.swap(handler.archive->version);
160         attributes.swap(handler.archive->attributes);
161         prefixes.swap(handler.archive->prefixes);
162         namespaces.swap(handler.archive->namespaces);
163         processing_instruction.swap(handler.archive->processing_instruction);
164         options = handler.archive->options;
165         tabstop = handler.archive->tabstop;
166         user_macro_list.swap(handler.archive->user_macro_list);
167     
168         read_root = true;
169     
170         return 1;
171     }

173     /**
174      * read_unit_attributes
175      * @param language a location to store the language attribute
176      * @param filename a location to store the filename attribute
177      * @param url a location to store the url attribute
178      * @param version a location to store the version attribute
179      *
180      * Read attributes from next unit.
181      *
182      * @returns 1 on success and 0 on failure.
183      */
184     int srcml_sax2_reader::read_unit_attributes(boost::optional<std::string> & language, boost::optional<std::string> & filename,
185                                                 boost::optional<std::string> & url, boost::optional<std::string> & version,
186                                                 boost::optional<std::string> & timestamp, boost::optional<std::string> & hash,
187                                                 std::vector<std::string> & attributes) {
188     
189         if(handler.is_done) return 0;
190         handler.skip = true;
191         handler.collect_unit_attributes = true;
192         handler.resume_and_wait();
193         handler.collect_unit_attributes = false;
194         handler.skip = false;
195         if(handler.is_done) return 0;
196     
197         language.swap(handler.unit->language);
198         filename.swap(handler.unit->filename);
199         url.swap(handler.unit->url);
200         version.swap(handler.unit->version);
201         hash.swap(handler.unit->hash);
202         timestamp.swap(handler.unit->timestamp);
203         attributes.swap(handler.unit->attributes);
204     
205         return 1;
206     
207     }

210     /**
211      * read_srcml
212      * @param unit location in which to read srcML unit.
213      *
214      * Read the next unit from a srcML Archive
215      * and return in the passed string parameter.
216      *
217      * @returns 1 on success and 0 if done
218      */
219     int srcml_sax2_reader::read_srcml(boost::optional<std::string> & unit) {
220     
221         if(unit) unit = boost::optional<std::string>();
222     
223         if(handler.is_done) return 0;
224         handler.collect_srcml = true;
225         handler.resume_and_wait();
226         handler.collect_srcml = false;
227         if(handler.is_done) return 0;
228     
229         unit.swap(handler.unit->unit);
230     
231         return unit ? 1 : 0;
232     }

234     /**
235      * read_src
236      * @param output_buffer output buffer to write text
237      *
238      * Read the next unit from a srcML Archive
239      * and return in the passed string parameter.
240      *
241      * @returns 1 on success and 0 if done
242      */
243     int srcml_sax2_reader::read_src(xmlOutputBufferPtr output_buffer) {
244     
245         if(handler.is_done) return 0;
246         control.enable_comment(false);
247         control.enable_cdataBlock(false);
248         handler.output_buffer = output_buffer;
249         handler.collect_src = true;
250         handler.resume_and_wait();
251         handler.collect_src = false;
252         handler.output_buffer = 0;
253         control.enable_comment(true);
254         control.enable_cdataBlock(true);
255         if(handler.is_done) return 0;
256     
257         return 1;
258     }
