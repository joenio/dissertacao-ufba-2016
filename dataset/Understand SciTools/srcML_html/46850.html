23     #include <srcml_execute.hpp>
24     #pragma GCC diagnostic push
25     #pragma GCC diagnostic ignored "-Wshorten-64-to-32"
26     #include <boost/thread.hpp>
27     #pragma GCC diagnostic pop
28     #include <boost/foreach.hpp>
29     
30     #if defined(_MSC_BUILD) || defined(__MINGW32__)
31     #include <io.h>
32     #include <fcntl.h>
33     #include <windows.h>
34     #endif
35     
36     void srcml_execute(const srcml_request_t& srcml_request,
37                        std::list<process_srcml>& pipeline,
38                        const srcml_input_t& input_sources,
39                        const srcml_output_dest& destination) {
40     
41         // create a thread for each step, creating pipes between adjoining steps
42         boost::thread_group pipeline_threads;
43         int fds[2] = { -1, -1 };
44         BOOST_FOREACH(process_srcml command, pipeline) {
45     
46             // special handling for first and last steps
47             bool first = command == pipeline.front();
48             bool last  = command == pipeline.back();
49     
50             // pipe between each step
51             int prevoutfd = fds[0];
52             fds[0] = fds[1] = -1;
53             if (pipeline.size() > 1 && !last) {
54     #if !defined(_MSC_BUILD) && !defined(__MINGW32__)
55                 pipe(fds);
56     #else
57                 HANDLE read_pipe;
58                 HANDLE write_pipe;
59                 CreatePipe(&read_pipe,&write_pipe, NULL, 0);
60     
61                 fds[1] = _open_osfhandle((intptr_t)write_pipe, 0);
62                 fds[0] = _open_osfhandle((intptr_t)read_pipe, _O_RDONLY);
63     #endif
64             }
65     
66             /* run this step in the sequence */
67             pipeline_threads.create_thread(
68                 boost::bind(
69                     command,
70                     srcml_request,
71                     /* first process_srcml uses input_source, rest input from previous output pipe */
72                     first ? input_sources : srcml_input_t(1, srcml_input_src("stdin://-", prevoutfd)),
73                     /* last process_srcml uses destination, rest output to pipe */                
74                     last  ? destination   : srcml_output_dest("-", fds[1])
75                 )
76             );
77     
78         }
79     
80         pipeline_threads.join_all();
81     }
