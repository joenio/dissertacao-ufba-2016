42     
43     
44     std::string src = "int a;\n";
45     
46     std::string asrcml = "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>\n"
47         "<unit xmlns=\"http://www.srcML.org/srcML/src\" xmlns:cpp=\"http://www.srcML.org/srcML/cpp\" revision=\"" SRCML_VERSION_STRING "\" language=\"C++\" filename=\"a.cpp\" hash=\"0123456789abcdef\"><decl_stmt><decl><type><name>int</name></type> <name>a</name></decl>;</decl_stmt>\n"
48         "</unit>\n";
49     
50     std::string srcml_c = "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>\n"
51         "<unit xmlns=\"http://www.srcML.org/srcML/src\" xmlns:cpp=\"http://www.srcML.org/srcML/cpp\" revision=\"" SRCML_VERSION_STRING "\" language=\"C\" filename=\"a.cpp\" hash=\"0123456789abcdef\"><decl_stmt><decl><type><name>int</name></type> <name>a</name></decl>;</decl_stmt>\n"
52         "</unit>\n";
53     
54     std::string srcml_full = "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>\n"
55         "<s:unit xmlns:s=\"http://www.srcML.org/srcML/src\" xmlns:cpp=\"http://www.srcML.org/srcML/cpp\" revision=\"" SRCML_VERSION_STRING "\" language=\"C++\" url=\"url\" filename=\"file\" version=\"1\" hash=\"0123456789abcdef\"><s:decl_stmt><s:decl><s:type><s:name>int</s:name></s:type> <s:name>a</s:name></s:decl>;</s:decl_stmt>\n"
56         "</s:unit>\n";
57     
58     int main() {
59     
60         assert(srcml_get_options() & SRCML_OPTION_HASH);
61         srcml_disable_option(SRCML_OPTION_HASH);
62         global_unit.hash = "0123456789abcdef";
63     
64         std::ofstream src_file("a.cpp");
65         src_file << src;
66         src_file.close();
67     
68         std::ofstream asrcml_file("project.xml");
69         asrcml_file << asrcml;
70         asrcml_file.close();
71     
72         std::ofstream srcml_ext_file("project.srcML");
73         srcml_ext_file << asrcml;
74         srcml_ext_file.close();
75     
76         std::ofstream srcml_c_file("project_c.xml");
77         srcml_c_file << srcml_c;
78         srcml_c_file.close();
79     
80         std::ofstream srcml_xml_file("project");
81         srcml_xml_file << asrcml;
82         srcml_xml_file.close();
83     
84         std::ofstream srcml_full_file("project_full.xml");
85         srcml_full_file << srcml_full;
86         srcml_full_file.close();
87     
88         /*
89           srcml
90         */
91     
92         {
93             srcml("a.cpp", "project.cpp.xml");
94             std::string res_srcml;
95             std::ifstream project("project.cpp.xml");
96             char c = 0;
97             while(project.get(c)) {
98                 res_srcml += c;
99             }
100     
101             dassert(res_srcml, asrcml);
102     
103         }
104     
105         {
106             srcml_set_language(SRCML_LANGUAGE_C);
107             srcml("a.cpp", "project.c.xml");
108             std::string res_srcml;
109             std::ifstream project("project.c.xml");
110             char c = 0;
111             while(project.get(c)) {
112                 res_srcml += c;
113             }
114     
115             dassert(res_srcml, srcml_c);
116     
117             srcml_set_language(SRCML_LANGUAGE_NONE);
118     
119         }
120     
121         {
122     
123             srcml_set_filename("file");
124             srcml_set_url("url");
125             srcml_set_version("1");
126             srcml_register_namespace("s", "http://www.srcML.org/srcML/src");
127             srcml("a.cpp", "project_full.cpp.xml");
128             std::string res_srcml;
129             std::ifstream project("project_full.cpp.xml");
130             char c = 0;
131             while(project.get(c)) {
132                 res_srcml += c;
133             }
134     
135             dassert(res_srcml, srcml_full);
136     
137         }
138     
139         {
140             srcml("project.xml", "inta.cpp");
141             std::string res_src;
142             std::ifstream project("inta.cpp");
143             char c = 0;
144             while(project.get(c)) {
145                 res_src += c;
146             }
147     
148             dassert(res_src, src);
149     
150         }
151     
152         {
153             srcml("project.srcML", "inta.cpp");
154             std::string res_src;
155             std::ifstream project("inta.cpp");
156             char c = 0;
157             while(project.get(c)) {
158                 res_src += c;
159             }
160     
161             dassert(res_src, src);
162     
163         }
164     
165         {
166             srcml_set_language(SRCML_LANGUAGE_XML);
167             srcml("project", "inta.cpp");
168             std::string res_src;
169             std::ifstream project("inta.cpp");
170             char c = 0;
171             while(project.get(c)) {
172                 res_src += c;
173             }
174     
175             dassert(res_src, src);
176     
177             srcml_set_language(SRCML_LANGUAGE_NONE);
178     
179         }
180     
181         {
182             srcml("project_c.xml", "inta.cpp");
183             std::string res_src;
184             std::ifstream project("inta.cpp");
185             char c = 0;
186             while(project.get(c)) {
187                 res_src += c;
188             }
189     
190             dassert(res_src, src);
191     
192         }
193     
194         {
195             srcml("project_full.xml", "inta.cpp");
196             std::string res_src;
197             std::ifstream project("inta.cpp");
198             char c = 0;
199             while(project.get(c)) {
200                 res_src += c;
201             }
202     
203             dassert(res_src, src);
204     
205         }
206     
207         {
208             dassert(srcml("foo.c", "foo.xml"), SRCML_STATUS_IO_ERROR);
209         }
210     
211         srcml_cleanup_globals();
212     
213         UNLINK("a.cpp");
214         UNLINK("project.xml");
215         UNLINK("project.srcML");
216         UNLINK("project_c.xml");
217         UNLINK("project");
218         UNLINK("project_full.xml");
219         UNLINK("project.cpp.xml");
220         UNLINK("project.c.xml");
221         UNLINK("inta.cpp");
222         UNLINK("project_full.cpp.xml");
223     
224         return 0;
