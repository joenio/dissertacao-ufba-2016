42     /**
43      * relaxng_units
44      *
45      * Extends unit_dom to execute RelaxNG grammar and write results.
46      */
47     class relaxng_units : public unit_dom {
48     public :
49     
50         /**
51          * relaxng_units
52          * @param options list of srcML options
53          * @param rngctx RelaxNG context pointer
54          * @param fd file descriptor in which to write
55          *
56          * Constructor.
57          */
58         relaxng_units(OPTION_TYPE options, xmlRelaxNGValidCtxtPtr rngctx, srcml_archive* oarchive)
59             : unit_dom(options), rngctx(rngctx), oarchive(oarchive) {
60         }
61     
62         /**
63          * ~relaxng_units
64          *
65          * Destructor.
66          */
67         virtual ~relaxng_units() {}
68     
69         /**
70          * start_output
71          *
72          * Create output buffer.
73          */
74         virtual void start_output() {}
75     
76         /**
77          * apply
78          *
79          * Apply RelaxNG grammar writing results.
80          * 
81          * @returns true on success false on failure.
82          */
83         virtual bool apply() {
84     
85             // validate
86             int n = xmlRelaxNGValidateDoc(rngctx, ctxt->myDoc);
87             if (n != 0)
88                 return true;
89     
90             // output if it validates
91     
92             // get the root node of current unit
93             xmlNodePtr node = xmlDocGetRootElement(ctxt->myDoc);
94             if (!node)
95                 return true;
96     
97             outputResult(node);
98     
99             return true;
100         }
101     
102         virtual void outputResult(xmlNodePtr a_node) {
103     
104             static xmlBufferPtr lbuffer = xmlBufferCreate();
105             int size = xmlNodeDump(lbuffer, ctxt->myDoc, a_node, 0, 0);
106             if (size == 0)
107                 return;
108     
109             oarchive->translator->add_unit_raw((const char*) xmlBufferContent(lbuffer), size);
110     
111             xmlBufferEmpty(lbuffer);
112         }
113     
114         /**
115          * end_output
116          *
117          * Finish the archive and close buffer.
118          */
119         virtual void end_output() {}
120     
121     private :
122     
123         xmlRelaxNGValidCtxtPtr rngctx;
124         srcml_archive* oarchive;
125     
126     };
