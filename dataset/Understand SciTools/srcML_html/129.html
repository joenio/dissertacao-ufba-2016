32     /** set language_pair as type that is a pair of string/int */
33     typedef std::pair<std::string, int> language_pair;
34     
35     /** get the string representation of a language from a language pair */
36     #define get_language_string(OBJ) OBJ.first
37     
38     /** get the number representation of a language from a language pair */
39     #define get_language_number(OBJ) OBJ.second

42      * Namespace for language names

47         const char* const LANGUAGE_NONE = "";

49         /** String constant for C language */
50         const char* const LANGUAGE_C = "C";

52         /** String constant for C# language */

70     /**
71      * Language
72      *
73      * Class representing a source code language.  Also, provides static members
74      * for determining a language from an extension.
75      */
76     class Language {
77     
78     public:
79     
80         /**
81          * Languages
82          *
83          * Enum constants for each handled language.
84          * Each language is a bit.
85          */
86         enum Languages {
87     
88             /** Interger Constant for No language */
89             LANGUAGE_NONE = 0,
90     
91             /** Interger Constant for C language */
92             LANGUAGE_C = 1,
93     
94             /** Interger Constant for C++ language */
95             LANGUAGE_CXX = 2,
96     
97             /** Interger Constant for Java language */
98             LANGUAGE_JAVA = 4,
99     
100             /** Interger Constant for AspectJ language */
101             LANGUAGE_ASPECTJ = 8,
102     
103             /** Interger Constant for C# language */
104             LANGUAGE_CSHARP = 16,
105     
106             /** Interger Constant for Objective-C language */
107             LANGUAGE_OBJECTIVE_C = 32,
108     
109             /** Interger Constant for languages in C++ family i.e. C++/C#  */
110             LANGUAGE_CXX_FAMILY = LANGUAGE_CXX | LANGUAGE_CSHARP,
111     
112             /** Interger Constant for languages in C family i.e. C/C++/C# */
113             LANGUAGE_C_FAMILY = LANGUAGE_C | LANGUAGE_CXX_FAMILY | Language::LANGUAGE_OBJECTIVE_C,
114     
115             /** Interger Constant for languages in Java family i.e. Java/AspectJ */
116             LANGUAGE_JAVA_FAMILY = LANGUAGE_JAVA | LANGUAGE_ASPECTJ,
117     
118             /** Interger Constant for Object-Oriented languages i.e. C++/C#/Java */
119             LANGUAGE_OO = LANGUAGE_CXX_FAMILY | LANGUAGE_JAVA_FAMILY,
120     
121             /** Interger Constant for all languages */
122             LANGUAGE_ALL = ~0
123         };
124     
125         /**
126          * Language
127          * @param lang numeric represenation to for language object
128          *
129          * Constructor taking int to set language.
130          */
131         Language(int lang)
132             : language(lang) {
133     
134         }
135     
136         /**
137          * inLanguage
138          *
139          * Predicate method to test if in a current language
140          *
141          * @returns if in the current language.
142          */
143         inline bool inLanguage(int l) const {
144             return (l & language) > 0;
145         }
146     
147         /**
148          * getLanguage
149          *
150          * Accessor method that gets the current language.
151          *
152          * @returns the numberic representation of the language.
153          */
154         inline int getLanguage() const {
155             return language;
156         }
157     
158         /**
159          * getLanguage
160          *
161          * Static method to translate a string representation of a language
162          * into numeric version
163          *
164          * @returns the numberic representation of the language.
165          */
166         static int getLanguage(const char* const s) {
167     
168             for(int i = 0; i < lang2intcount; ++i)
169                 if (get_language_string(lang2int[i]) == s)
170                     return get_language_number(lang2int[i]);
171     
172             return 0;
173         }
174     
175         /**
176          * getLanguageString
177          *
178          * Get the current language as a string.
179          *
180          * @returns the current language as a string.
181          */
182         const char* getLanguageString() const {
183     
184             for(int i = 0; i < lang2intcount; ++i)
185                 if (get_language_number(lang2int[i]) == language)
186                     return get_language_string(lang2int[i]).c_str();
187     
188             return "";
189         }
190     
191         /**
192          * ~Language
193          *
194          * Destructor.
195          */
196         ~Language() {}
197     
198     private:
199     
200         /** the current langauge */
201         const int language;
202     
203         static int lang2intcount;
204         static language_pair lang2int[];
205     };
206     
207     #endif
