30      int apply_xpath(srcml_archive* in_arch, const std::string& transform_input, const std::pair< boost::optional<element>, boost::optional<attribute> >& xpath_support, const std::map<std::string,std::string>& xmlns_namespaces) {
31      	// FIRST IS ELEMENT / SECOND IS ATTRIBUTE
32     
33      	// Check for element
34      	if (xpath_support.first){
35     
36             // check first if namespace is already declared
37             const char* element_uri = srcml_archive_get_uri_from_prefix(in_arch, xpath_support.first->prefix->c_str());
38             
39             // if not declared, check for xmlns from cli
40             if (!element_uri) {
41                 std::map<std::string,std::string>::const_iterator it = xmlns_namespaces.find(*(xpath_support.first->prefix));
42                 if (it != xmlns_namespaces.end())            
43                     element_uri = it->second.c_str();
44             }
45     
46             // make sure we found it somewhere
47             if (!element_uri) {
48                 std::cerr << "srcml: no uri exists for prefix \"" << xpath_support.first->prefix->c_str() << "\"\n";
49                 return -1;
50             }
51     
52      		// See if an attribute is present as well
53      		if (xpath_support.second) {
54                 const char* attribute_uri = srcml_archive_get_uri_from_prefix(in_arch, xpath_support.second->prefix->c_str());
55                 
56                 // if not declared, check for xmlns from cli
57                 if (!attribute_uri) {
58                     std::map<std::string,std::string>::const_iterator it = xmlns_namespaces.find(*(xpath_support.second->prefix));
59                     if (it != xmlns_namespaces.end())            
60                         attribute_uri = it->second.c_str();
61                 }
62     
63                 if (!attribute_uri) {
64                     std::cerr << "srcml: no uri exists for prefix \"" << xpath_support.second->prefix->c_str() << "\"\n";
65                     return -1;
66                 }
67     
68      			return srcml_append_transform_xpath_element (in_arch, transform_input.c_str(),
69                                                                 xpath_support.first->prefix->c_str(),
70                                                                 element_uri,
71                                                                 xpath_support.first->name->c_str(),
72                                                                 xpath_support.second->prefix->c_str(),
73                                                                 attribute_uri,
74                                                                 xpath_support.second->name->c_str(),
75                                                                 xpath_support.second->value->c_str());
76      		}
77      		else {
78      			return srcml_append_transform_xpath_element (in_arch, transform_input.c_str(),
79                                                                 xpath_support.first->prefix->c_str(),
80                                                                 element_uri,
81                                                                 xpath_support.first->name->c_str(),
82                                                                 NULL, NULL, NULL, NULL);
83      		}
84      	}
85     
86      	// Check for attribute
87      	if (xpath_support.second) {
88             const char* attribute_uri = srcml_archive_get_uri_from_prefix(in_arch, xpath_support.second->prefix->c_str());
89             
90             // if not declared, check for xmlns from cli
91             if (!attribute_uri) {
92                 std::map<std::string,std::string>::const_iterator it = xmlns_namespaces.find(*(xpath_support.second->prefix));
93                 if (it != xmlns_namespaces.end())            
94                     attribute_uri = it->second.c_str();
95             }
96     
97             if (!attribute_uri) {
98                 std::cerr << "srcml: no uri exists for prefix \"" << xpath_support.second->prefix->c_str() << "\"\n";
99                 return -1;
100             }
101      		
102             return srcml_append_transform_xpath_attribute (in_arch, transform_input.c_str(),
103                                                                 xpath_support.second->prefix->c_str(),
104                                                                 attribute_uri,
105                                                                 xpath_support.second->name->c_str(),
106                                                                 xpath_support.second->value->c_str());
107      	}
108     
109      	return srcml_append_transform_xpath(in_arch, transform_input.c_str());
110      }

112      int apply_xslt(srcml_archive* in_arch, const std::string& transform_input) {
113      	// xslt has file input, which may need to be processed
114      	int status;
115     	srcml_input_src xslt_file(transform_input.c_str());
116     	input_file(xslt_file);
117     
118         if (contains<int>(xslt_file))
119             status = srcml_append_transform_xslt_fd(in_arch, xslt_file);
120         else if (contains<FILE*>(xslt_file))
121             status = srcml_append_transform_xslt_FILE(in_arch, xslt_file);
122         else
123             status = srcml_append_transform_xslt_filename(in_arch, xslt_file.c_str());
124     
125         return status;
126      }

144     // transform srcml with query or transformation
145     void transform_srcml(const srcml_request_t& srcml_request,
146                          const srcml_input_t& input_sources,
147                          const srcml_output_dest& output) {
148     
149     	// Convert output into srcml archive
150     	int status;
151     	srcml_archive* out_arch = srcml_archive_create();
152         if (contains<int>(output))
153             status = srcml_archive_write_open_fd(out_arch, output);
154         else if (contains<FILE*>(output))
155             status = srcml_archive_write_open_FILE(out_arch, output);
156         else
157             status = srcml_archive_write_open_filename(out_arch, output.c_str(), 0);
158         if (status != SRCML_STATUS_OK) {
159             std::cerr << "srcml: error with output archive for transformation\n";
160             exit(-1);
161         }
162     
163         // register xml namespaces
164         std::map<std::string, std::string>::const_iterator itr;
165         for (itr = srcml_request.xmlns_namespaces.begin(); itr != srcml_request.xmlns_namespaces.end(); ++itr){
166             srcml_archive_register_namespace(out_arch, (*itr).first.c_str(), (*itr).second.c_str());
167         }
168     
169         // Convert inputs into srcml archive
170     	BOOST_FOREACH(const srcml_input_src& input, input_sources) {
171             srcml_archive* in_arch = srcml_archive_create();
172             if (contains<int>(input))
173                 status = srcml_archive_read_open_fd(in_arch, input);
174             else if (contains<FILE*>(input))
175                 status = srcml_archive_read_open_FILE(in_arch, input);
176             else
177                 status = srcml_archive_read_open_filename(in_arch, input.c_str());
178             if (status != SRCML_STATUS_OK) {
179                 std::cerr << "srcml: error with input archive for transformation\n";
180                 exit(-1);
181             }
182     
183             // see if we have any XPath output
184             bool isxpath = false;
185             BOOST_FOREACH(const std::string& trans, srcml_request.transformations) {
186                 std::string protocol;
187                 std::string resource;
188                 src_prefix_split_uri(trans, protocol, resource);
189     
190                 if (protocol == "xpath") {
191                     isxpath = true;
192                     break;
193                 }
194             }
195     
196             // for non-archive input, then we want non-archive output, fool
197             if (!isxpath && !(srcml_archive_get_options(in_arch) & SRCML_OPTION_ARCHIVE)) {
198                 srcml_archive_disable_option(out_arch, SRCML_OPTION_ARCHIVE);
199             }
200     
201             // copy input xml namespaces
202             // TODO: This assumes namespaces on first input. Need to open all, figure out
203             // output namespaces, then process
204             //if (srcml_archive_get_options(in_arch) & SRCML_OPTION_ARCHIVE) {
205                 for (int i = 0; i < (int)srcml_archive_get_namespace_size(in_arch); ++i) {
206     
207                     // do not register the srcML namespace, unless the prefix is different
208                     if (std::string(srcml_archive_get_namespace_uri(in_arch, i)) == "http://www.srcML.org/srcML/src")
209                         continue;
210     
211                     srcml_archive_register_namespace(out_arch, srcml_archive_get_namespace_prefix(in_arch, i), srcml_archive_get_namespace_uri(in_arch, i));
212                 }
213             //}
214     
215     		// iterate through all transformations added during cli parsing
216     		int xpath_index = -1;
217             BOOST_FOREACH(const std::string& trans, srcml_request.transformations) {
218                 std::string protocol;
219                 std::string resource;
220                 src_prefix_split_uri(trans, protocol, resource);
221     
222                 if (protocol == "xpath") {
223                     // TODO: FIX BUG
224     				if (apply_xpath(in_arch, resource, srcml_request.xpath_query_support.at(++xpath_index), srcml_request.xmlns_namespaces) != SRCML_STATUS_OK) {
225     					std::cerr << "srcml: error with xpath transformation\n";
226                         exit(-1);
227                     }
228     			}
229     			else if (protocol == "xslt") {
230     		        if (apply_xslt(in_arch, resource) != SRCML_STATUS_OK) {
231     		            std::cerr << "srcml: error with xslt transformation\n";
232                         exit(-1);
233                     }
234     				
235     				//std::cerr << protocol << " : " << resource << "\n"; // Debug Printout
236     			}
237     			else if (protocol == "xpathparam") {
238     				//std::cerr << protocol << " : " << resource << "\n"; // Stub
239     			}
240     			else if (protocol == "relaxng") {
241     		        if (apply_relaxng(in_arch, resource) != SRCML_STATUS_OK) {
242     		            std::cerr << "srcml: error with relaxng transformation\n";
243                         exit(-1);
244                     }
245     
246     				//std::cerr << protocol << " : " << resource << "\n"; //Debug Printout
247     			}
248     		}
249     		srcml_apply_transforms(in_arch, out_arch);
250     
251     		srcml_archive_close(in_arch);
252     		srcml_archive_free(in_arch);
253     	}
254     
255     	srcml_archive_close(out_arch);
256     	srcml_archive_free(out_arch);
257     }
