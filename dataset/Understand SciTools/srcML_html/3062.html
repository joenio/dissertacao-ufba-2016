45         bool curl_supported(const std::string& input_protocol) {
46             const char* const* curl_types = curl_version_info(CURLVERSION_NOW)->protocols;
47             for (int i = 0; curl_types[i] != NULL; ++i) {
48                 if (strcmp(curl_types[i], input_protocol.c_str()) == 0)
49                     return true;
50             }
51             return false;
52         }

55     void decompress_srcml(const srcml_request_t& /* srcml_request */,
56         const srcml_input_t& input_sources,
57         const srcml_output_dest& destination) {
58     
59         archive* libarchive_srcml = archive_read_new();
60     
61         // just a bunch of bytes
62         archive_read_support_format_raw(libarchive_srcml);
63     
64         /* Check libarchive version enable version specific features/syntax */
65     #if ARCHIVE_VERSION_NUMBER < 3000000
66         // V2 Only Settings
67         // Compressions
68         archive_read_support_compression_all(libarchive_srcml);
69     
70     #else
71         // V3 Only Settings
72         // File Formats
73         archive_read_support_format_7zip(libarchive_srcml);
74         archive_read_support_format_cab(libarchive_srcml);
75         archive_read_support_format_lha(libarchive_srcml);
76         archive_read_support_format_rar(libarchive_srcml);
77     
78         // Compressions
79         archive_read_support_filter_all(libarchive_srcml);
80     #endif    // setup decompressions
81     
82         int status = ARCHIVE_OK;
83         curl curling;
84     
85         if (contains<int>(input_sources[0])) {
86             status = archive_read_open_fd(libarchive_srcml, input_sources[0], 16384);
87         } else if (curl_supported(input_sources[0].protocol)) {
88     
89             curling.source = input_sources[0].filename;
90             status = archive_read_open(libarchive_srcml, &curling, archive_curl_open, archive_curl_read, archive_curl_close);
91     
92         } else {
93             status = archive_read_open_filename(libarchive_srcml, input_sources[0].resource.c_str(), 16384);
94         }
95         if (status != ARCHIVE_OK) {
96             std::cerr << status;
97             exit(1);
98         }
99         archive_entry *entry;
100         status = archive_read_next_header(libarchive_srcml, &entry);
101     
102         // copy from the libarchive decompressed data into the destination file descriptor
103         // for the next stage in the pipeline
104         archive_read_data_into_fd(libarchive_srcml, *destination.fd);
105     
106         // important to close, since this is how the file descriptor reader get an EOF
107         close(*destination.fd);
108     
109         archive_read_close(libarchive_srcml);
110         archive_read_finish(libarchive_srcml);
111     }

115         size_t curl_cb(void* buffer, size_t len, size_t nmemb, void* data) {
116     
117             curl* curldata = (curl*) data;
118     
119             curldata->data_buffer = (char*)buffer;
120             curldata->data_len = len * nmemb;
121     
122             return curldata->data_len;
123         }
124     
125         int archive_curl_open(archive*, void* client_data) {
126     
127             curl* curldata = (curl*) client_data;
128     
129             curl_global_init(CURL_GLOBAL_DEFAULT);
130     
131             curldata->handle = curl_easy_init();
132     
133             curl_easy_setopt(curldata->handle, CURLOPT_SSL_VERIFYPEER, 0L);
134             curl_easy_setopt(curldata->handle, CURLOPT_SSL_VERIFYHOST, 0L);
135             curl_easy_setopt(curldata->handle, CURLOPT_HTTPAUTH, (long)CURLAUTH_ANY);
136             curl_easy_setopt(curldata->handle, CURLOPT_FOLLOWLOCATION, 1L);
137             curl_easy_setopt(curldata->handle, CURLOPT_VERBOSE, 0L);
138             curl_easy_setopt(curldata->handle, CURLOPT_WRITEFUNCTION, curl_cb);
139             curl_easy_setopt(curldata->handle, CURLOPT_WRITEDATA, curldata);
140             curl_easy_setopt(curldata->handle, CURLOPT_URL, curldata->source.c_str());
141     
142             curldata->multi_handle = curl_multi_init();
143             curl_multi_add_handle(curldata->multi_handle, curldata->handle);
144             curl_multi_perform(curldata->multi_handle, &curldata->still_running);
145     
146             // TODO: SOMETHING HERE TO MAKE SURE THE FILE IS ACTUALLY PRESENT
147             return ARCHIVE_OK;
148         }
149     
150         ssize_t archive_curl_read(archive*, void* client_data, const void** buff) {
151     
152             curl* curldata = (curl*) client_data;
153     
154             curldata->data_len = 0;
155             while (curldata->data_len == 0 && curldata->still_running) {
156                 curl_multi_perform(curldata->multi_handle, &curldata->still_running);
157             }
158     
159             *buff = curldata->data_buffer;
160     
161             return curldata->data_len;
162         }
163     
164         int archive_curl_close(archive*, void* client_data) {
165     
166             curl* curldata = (curl*) client_data;
167             curl_multi_cleanup(curldata->multi_handle);
168             curl_easy_cleanup(curldata->handle);
169     
170             return 0;
171         }
