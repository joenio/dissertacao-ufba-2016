23     #include <src_input_stdin.hpp>
24     
25     #include <stdio.h>
26     
27     #ifndef WIN32
28     #include <unistd.h>
29     #include <sys/select.h>
30     #else
31     #include <io.h>
32     #include <winsock2.h>
33     #define  STDIN_FILENO   0       /* standard input file descriptor */
34     #define STDOUT_FILENO   1       /* standard output file descriptor */
35     #define STDERR_FILENO   2       /* standard error file descriptor */
36     #endif
37     
38     #include <iostream>
39     
40     // check stdin for data
41     bool src_input_stdin() {
42     
43         // init file descriptor with stdin
44         fd_set fds;
45         FD_ZERO(&fds);
46         FD_SET(STDIN_FILENO, &fds);
47     
48         /*
49           Need a timeout so the application doesn't
50           hang waiting for input that never comes
51         */
52         struct timeval timeout;
53         timeout.tv_sec = 5;
54         timeout.tv_usec = 0;
55     
56         // use select to see if stdin has data
57         int selectRetVal = select(sizeof(fds)*8, &fds, NULL, NULL, &timeout);
58     
59         if (selectRetVal == -1) {
60             std::cerr << "SELECT FAILED!\n";
61             return false;
62         }
63         if (selectRetVal == 0) {
64             std::cerr << "NO DATA TO FETCH!\n";
65             return false;
66         }
67         return true;
68     }
