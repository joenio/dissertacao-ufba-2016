28     // determine if XML from first four bytes in almost any encoding
29     bool isxml(unsigned char ar[], int size);

31     // determine if XML
32     bool isxml(FILE* infd) {
33     
34         // determine if the input is srcML or src from the first up-to 4 bytes
35         unsigned char data[4];
36     
37         return isxml(data, peek4char(infd, data));
38     }

40     // determine if XML from first four bytes in almost any encoding
41     bool isxml(unsigned char ar[], int size) {
42     
43         // bail if not valid array
44         if (!ar)
45             return false;
46     
47         // xml is always at least 4 bytes for xml declaration, i.e., '<?xm'
48         // or for xml without xml declaration, i.e., '<uni'
49         if (size < 4)
50             return false;
51     
52         // treat unsigned int field as just 4 bytes regardless of endianness
53         union { unsigned char d[4]; uint32_t i; } data = { { ar[0], ar[1], ar[2], ar[3] } };
54         // data.d[0] = ar[0]; data.d[1] = ar[1]; data.d[2] = ar[2]; data.d[3] = ar[3];
55     
56         // determine if XML by looking for xml declaration (i.e., '<?xm')
57         // in first four bytes in almost any encoding
58     //    fprintf(stderr, "DEBUG:  %s %s %d DATA: %x\n", __FILE__,  __FUNCTION__, __LINE__, data.i);
59     
60         return
61             (data.i == 0x6D783F3C) || // '<?xm' UTF-8
62             (data.i == 0x696E753C) || // '<uni' UTF-8
63             (data.i == 0x003F003C) || // '<?xm' UTF-16LE
64             (data.i == 0x0075003C) || // '<uni' UTF-16LE
65             (data.i == 0x3F003C00) || // '<?xm' UTF-16BE
66             (data.i == 0x75003C00) || // '<uni' UTF-16BE
67             (data.i == 0x3C000000) || // <?xm' UCS-4BE
68     //        (data.i == 0x3C000000) || // '<uni' UCS-4BE
69             (data.i == 0x0000003C) || // '<?xm' UCS-4LE
70     //        (data.i == 0x0000003C) || // '<uni' UCS-4LE
71             (data.i == 0x003C0000) || // '<?xm' UCS-4_2143
72     //        (data.i == 0x003C0000) || // '<uni' UCS-4_2143
73             (data.i == 0x00003C00) || // '<?xm' UCS-4_3412
74     //        (data.i == 0x00003C00) || // '<uni' UCS-4_3412
75             (data.i == 0x94A76F4C) || // '<?xm' EBCDIC
76     //        (data.i == 0x94A76F4C) || // '<uni' EBCDIC
77             false;
78     }
