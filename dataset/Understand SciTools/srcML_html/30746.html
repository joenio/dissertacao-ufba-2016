23     #include <src_language.hpp>
24     #include <boost/filesystem.hpp>
25     
26     std::string src_language(const std::string& inputFile) {
27         boost::filesystem::path localPath(inputFile);
28         std::string language;
29     
30         if (is_directory(localPath))
31             language = "dir";
32     
33         if (is_regular_file(localPath)) {
34             if (localPath.extension().string().compare(".xml") == 0) {
35                 language = "xml";
36             }
37             else {
38                 const char* lang = srcml_check_extension(inputFile.c_str());
39                 language = lang ? lang : "";
40             }
41         }
42         return language;
43     }
