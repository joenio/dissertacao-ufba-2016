23     #include <srcml_input_srcml.hpp>
24     #include <parse_queue.hpp>
25     #include <srcml_input_src.hpp>
26     #include <srcml.h>
27     #include <srcml_options.hpp>
28     #include <srcml_cli.hpp>
29     
30     void srcml_input_srcml(ParseQueue& queue,
31                            srcml_archive* srcml_output_archive,
32                            const srcml_input_src& srcml_input) {
33     
34         // open the srcml input archive
35         srcml_archive* srcml_input_archive = srcml_archive_create();
36         if (contains<int>(srcml_input))
37             srcml_archive_read_open_fd(srcml_input_archive, srcml_input);
38         else if (contains<FILE*>(srcml_input))
39             srcml_archive_read_open_FILE(srcml_input_archive, srcml_input);
40         else
41             srcml_archive_read_open_filename(srcml_input_archive, srcml_input.c_str());
42     
43         if (SRCML_COMMAND_XML & SRCMLOptions::get()) {
44             if (SRCML_OPTION_ARCHIVE & srcml_archive_get_options(srcml_input_archive) && srcml_input.unit == 0) {
45                 srcml_archive_enable_option(srcml_output_archive, SRCML_OPTION_ARCHIVE);
46             }
47     
48             size_t nsSize = srcml_archive_get_namespace_size(srcml_input_archive);
49             
50             for (size_t i = 0; i < nsSize; ++i) {
51                 srcml_archive_register_namespace(srcml_output_archive,
52                     srcml_archive_get_namespace_prefix(srcml_input_archive, i),
53                     srcml_archive_get_namespace_uri(srcml_input_archive, i));
54             }
55         }
56     
57         // move to the correct unit
58         for (int i = 1; i < srcml_input.unit; ++i) {
59             srcml_unit* unit = srcml_read_unit_header(srcml_input_archive);
60             srcml_unit_free(unit);
61         }
62     
63         bool unitPresent = false;
64     
65         // process each entry in the input srcml archive
66         while (srcml_unit* unit = srcml_read_unit(srcml_input_archive)) {
67             unitPresent = true;
68             // form the parsing request
69             ParseRequest* prequest = new ParseRequest;
70             prequest->srcml_arch = srcml_output_archive;
71             prequest->unit = unit;
72     
73             // hand request off to the processing queue
74             queue.schedule(prequest);
75     
76             // one-time through for individual unit
77             if (srcml_input.unit)
78                 break;
79         }
80     
81         if (!unitPresent) {
82             std::cerr << "Requested unit " << srcml_input.unit << " out of range.\n";
83             exit(4);
84         }
85     
86     }
