47     constexpr bool isequal(char const* s1, char const* s2) {
48       return *s1 && *s2 ? *s1 == *s2 && isequal(s1 + 1, s2 + 1) : !*s1 && !*s2;
49     }
50     
51     constexpr bool islessthanorequal(char const* s1, char const* s2) {
52       return *s1 && *s2 ? (*s1 != *s2 ? *s1 < *s2 : islessthanorequal(s1 + 1, s2 + 1)) : !*s1;
53     }
54     static_assert(islessthanorequal("a", "a"), "islessthanorequal failure");
55     static_assert(islessthanorequal("a", "b"), "islessthanorequal failure");
56     static_assert(!islessthanorequal("b", "a"), "islessthanorequal failure");
57     static_assert(islessthanorequal("aa", "aa"), "islessthanorequal failure");
58     static_assert(islessthanorequal("aa", "ab"), "islessthanorequal failure");
59     static_assert(!islessthanorequal("ab", "aa"), "islessthanorequal failure");
60     
61     constexpr bool isordered(const archive_calls_t* p) {
62         return !(p->name) || !((p + 1)->name) ? true : islessthanorequal(p->name, (p+1)->name) && isordered(p + 1);
63     }

146     bool compare(const archive_calls_t& call, const char* extension) {
147     
148         return strcmp(extension, call.name) == 0;
149     }
150     
151     int archive_write_set_format_by_extension(struct archive* ar, const char* extension) {
152     
153         const archive_calls_t* end = format_calls + sizeof(format_calls) / sizeof(format_calls[0]) - 1;
154         archive_calls_t searchvalue = { extension, 0 };
155     
156         const archive_calls_t* extcall = std::lower_bound(format_calls, end, searchvalue);
157         if (extcall != end && extcall->setter)
158             return (extcall->setter)(ar);
159     
160         archive_set_error(ar, EINVAL, "No such format for this extension '%s'", extension);
161         return ARCHIVE_FATAL;
162     }
163     
164     int archive_write_set_compression_by_extension(struct archive* ar, const char* extension) {
165     
166         const archive_calls_t* end = compression_calls + sizeof(compression_calls) / sizeof(compression_calls[0]) - 1;
167         archive_calls_t searchvalue = { extension, 0 };
168     
169         const archive_calls_t* extcall = std::lower_bound(compression_calls, end, searchvalue);
170         if (extcall != end && extcall->setter)
171             return (extcall->setter)(ar);
172     
173         archive_set_error(ar, EINVAL, "No such compression for this extension '%s'", extension);
174         return ARCHIVE_FATAL;
175     }

177     bool operator<(const archive_calls_t& call, const archive_calls_t& call2) {
178     
179         return strcmp(call.name, call2.name) < 0;
180     }
181     
182     bool is_archive(const std::string& extension) {
183     
184         const archive_calls_t* end = format_calls + sizeof(format_calls) / sizeof(format_calls[0]) - 1;
185         archive_calls_t searchvalue = { extension.c_str(), 0 };
186     
187         return std::binary_search(format_calls, end, searchvalue);
188     }
189     
190     bool is_compressed(const std::string& extension) {
191     
192         const archive_calls_t* end = compression_calls + sizeof(compression_calls) / sizeof(compression_calls[0]) - 1;
193         archive_calls_t searchvalue = { extension.c_str(), 0 };
194     
195         return std::binary_search(compression_calls, end, searchvalue);
196     }
