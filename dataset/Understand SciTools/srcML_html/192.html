41     /**
42      * srcml_sax2_reader
43      *
44      * Extend XML Text Reader interface to
45      * progressively read a srcML Archive collecting
46      * units and reading unit attributes.
47      */
48     class srcml_sax2_reader {
49     
50     private :
51     
52         /** control for sax parsing */
53         srcSAXController control;
54         /** boolean for marking if read root */
55         bool read_root;
56         /** handler with hooks for sax parsing */
57         srcml_reader_handler handler;
58     
59         /** thread running execution */
60         boost::thread * thread;
61     
62         void init_constructor();
63     
64     public :
65     
66         // constructors
67         srcml_sax2_reader(const char * filename, const char * encoding = 0);
68         srcml_sax2_reader(xmlParserInputBufferPtr input);
69     
70     
71         // destructors
72         ~srcml_sax2_reader();
73     
74         // read attribute and namespace information from root unit.  Does not advance read.
75         int read_root_unit_attributes(boost::optional<std::string> & encoding, 
76                                       boost::optional<std::string> & language,
77                                       boost::optional<std::string> & url, boost::optional<std::string> & version,
78                                       std::vector<std::string> & attributes,
79                                       std::vector<std::string> & prefixes,
80                                       std::vector<std::string> & namespaces,
81                                       boost::optional<std::pair<std::string, std::string> > & processing_instruction,
82                                       OPTION_TYPE & options,
83                                       size_t & tabstop,
84                                       std::vector<std::string> & user_macro_list);
85     
86     
87         /* finds next unit tag if not current unit and sets attributes.  Consumes unit.
88            Unit is still avaible for readsrcML or read.  But not readUnitAttributes.
89         */
90         int read_unit_attributes(boost::optional<std::string> & language, boost::optional<std::string> & filename,
91                                  boost::optional<std::string> & url, boost::optional<std::string> & version,
92                                  boost::optional<std::string> & timestamp, boost::optional<std::string> & hash,
93                                  std::vector<std::string> & attributes);
94     
95         // reads the next unit and returns it in parameter as srcML
96         int read_srcml(boost::optional<std::string> & unit);
97     
98         // reads the next unit and returns it in parameter as src
99         int read_src(xmlOutputBufferPtr output_buffer);
100     
101         // set the revision number to retrieve
102         void revision_number(boost::optional<size_t> revision_number);
103     
104     
105     };
