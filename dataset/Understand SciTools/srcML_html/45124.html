23     #include <src_output_libarchive.hpp>
24     #include <srcml.h>
25     #include <archive.h>
26     #include <archive_entry.h>
27     #include <stdlib.h>
28     
29     void src_output_libarchive(srcml_archive* srcml_arch, archive* src_archive) {
30     
31         long arch_status = ARCHIVE_OK;
32         while (srcml_unit* unit = srcml_read_unit_header(srcml_arch)) {
33     
34             // setup the entry header
35             archive_entry* entry = archive_entry_new();
36             if (!entry)
37                 break;
38     
39             char* buffer;
40             size_t buffer_size;
41             srcml_unit_unparse_memory(unit, &buffer, &buffer_size);
42     
43             archive_entry_set_pathname(entry, srcml_unit_get_filename(unit));
44             archive_entry_set_size(entry, buffer_size);
45             archive_entry_set_filetype(entry, AE_IFREG);
46             archive_entry_set_perm(entry, 0644);
47     
48             time_t now = time(NULL);
49             archive_entry_set_atime(entry, now, 0);
50             archive_entry_set_ctime(entry, now, 0);
51             archive_entry_set_mtime(entry, now, 0);
52     
53             if ((arch_status = archive_write_header(src_archive, entry)) != ARCHIVE_OK) {
54                 break;
55             }
56     
57             // write the data into the archive
58             arch_status = archive_write_data(src_archive, buffer, (size_t) buffer_size);
59             if (arch_status != ARCHIVE_OK /* || arch_status != buffer_size */)
60                 break;
61     
62             free(buffer);
63     
64             archive_entry_free(entry);
65     
66             srcml_unit_free(unit);
67         }
68     }
