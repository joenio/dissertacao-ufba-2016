23       Test cases for srcml_archive_create.
24     */
25     #include <stdio.h>
26     #include <string.h>
27     #include <cassert>
28     
29     #include <srcml.h>
30     #include <srcml_types.hpp>
31     #include <srcmlns.hpp>
32     
33     #include <unit_tests.hpp>
34     
35     int main() {
36     
37         {
38     
39             srcml_archive * archive = srcml_archive_create();
40             archive->type = SRCML_ARCHIVE_READ;
41             archive->encoding = "e";
42             archive->language = "l";
43             archive->url = "u";
44             archive->version = "v";
45             archive->attributes.push_back("a");
46             archive->attributes.push_back("a");
47             archive->options = 1 | 2;
48             archive->tabstop = 4;
49             archive->prefixes.at(0) = "s";
50             archive->prefixes.push_back("foo");
51             archive->namespaces.push_back("bar");
52             archive->user_macro_list.push_back("foo");
53             archive->user_macro_list.push_back("bar");
54             srcml_archive_register_file_extension(archive, "foo", "bar");
55             archive->translator = (srcml_translator *)1;
56             archive->reader = (srcml_sax2_reader *)1;
57             archive->input = (xmlParserInputBufferPtr)1;
58             transform trans;
59             trans.type = SRCML_XPATH;
60             trans.arguments.str = "//src:unit";
61             archive->transformations.push_back(trans);
62     
63             srcml_archive * new_archive = srcml_archive_clone(archive);
64     
65             dassert(new_archive->type, SRCML_ARCHIVE_INVALID);
66             dassert(*new_archive->encoding, "e");
67             dassert(new_archive->src_encoding, 0);
68             dassert(*new_archive->language, "l");
69             dassert(*new_archive->url, "u");
70             dassert(*new_archive->version, "v");
71             dassert(new_archive->attributes.size(), 2);
72             dassert(new_archive->attributes.at(0), "a");
73             dassert(new_archive->attributes.at(1), "a");
74     
75             dassert(new_archive->options, (1 | 2));
76     
77             dassert(new_archive->tabstop, 4);
78             dassert(new_archive->prefixes.size(), 2);
79             dassert(new_archive->namespaces.size(), 2);
80             dassert(new_archive->registered_languages.size(), archive->registered_languages.size());
81     
82             dassert(new_archive->prefixes.at(0), "s");
83             dassert(new_archive->prefixes.at(1), "foo");
84             dassert(new_archive->namespaces.at(1), "bar");
85     
86             dassert(new_archive->user_macro_list.size(), 2);
87             dassert(new_archive->user_macro_list.at(0), "foo");
88             dassert(new_archive->user_macro_list.at(1), "bar");
89     
90             dassert(new_archive->translator, 0);
91             dassert(new_archive->reader, 0);
92             dassert(new_archive->input, 0);
93     
94             dassert(new_archive->transformations.size(), 0);
95     
96             archive->translator = 0;
97             archive->reader = 0;
98             archive->input = 0;
99             srcml_archive_free(archive);
100             srcml_archive_free(new_archive);
101         }
102     
103         {
104     
105             dassert(srcml_archive_clone(0), 0);
106     
107         }
108     
109         return 0;
110     }
