3     /**
4      * is_srcml_namespace
5      * @param uri the uri to compare
6      * @param srcml_uri the srcML URI to compare to
7      *
8      * Compare uri to the srcML uri idependent of prefix
9      * i.e., www.sdml.info or www.srcML.org.
10      *
11      * @returns if the two uris are equal.
12      */
13     bool is_srcml_namespace(const std::string & uri, const std::string & srcml_uri) {
14     
15     	if(uri == srcml_uri) return true;
16     
17     	std::string uri_suffix = uri, srcml_uri_suffix = srcml_uri;
18     
19     	for(int pos = 0; SRCML_URI_PREFIX[pos][0]; ++pos) {
20     
21     		std::string::size_type str_pos = uri.find(SRCML_URI_PREFIX[pos]);
22     		if(str_pos == 0) {
23     
24     			uri_suffix = uri.substr(SRCML_URI_PREFIX[pos].size());
25     			break;
26     
27     		}
28     
29     	}
30     
31     	for(int pos = 0; SRCML_URI_PREFIX[pos][0]; ++pos) {
32     
33     		std::string::size_type str_pos = srcml_uri.find(SRCML_URI_PREFIX[pos]);
34     		if(str_pos == 0) {
35     
36     			srcml_uri_suffix = srcml_uri.substr(SRCML_URI_PREFIX[pos].size());
37     			break;
38     
39     		}
40     
41     	}
42     
43     	return uri_suffix == srcml_uri_suffix;
44     
45     }

65     		}

67     	}
