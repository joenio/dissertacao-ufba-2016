23     #include <src_input_file.hpp>
24     #include <srcml_options.hpp>
25     #include <src_input_libarchive.hpp>
26     
27     // Convert input to a ParseRequest and assign request to the processing queue
28     void src_input_file(ParseQueue& queue,
29                         srcml_archive* srcml_arch,
30                         const srcml_request_t& srcml_request,
31                         const std::string& input_file) {
32     
33         if (SRCML_COMMAND_VERBOSE & SRCMLOptions::get()) {
34             src_input_libarchive(queue, srcml_arch, srcml_request, input_file);
35             return;
36         }
37     
38         // form the parsing request
39         ParseRequest* prequest = new ParseRequest;
40        
41         if (srcml_request.command & SRCML_COMMAND_NOARCHIVE)
42             prequest->disk_dir = srcml_request.output_filename;
43     
44         if (srcml_request.att_filename)
45             prequest->filename = *srcml_request.att_filename;
46         else if (input_file != "_")
47             prequest->filename = input_file;
48     
49         prequest->url = srcml_request.att_url;
50         prequest->version = srcml_request.att_version;
51         prequest->srcml_arch = srcml_arch;
52         prequest->language = srcml_request.att_language ? *srcml_request.att_language : "";
53     
54         if (prequest->language.empty())
55                 if (const char* l = srcml_archive_check_extension(srcml_arch, prequest->filename->c_str()))
56                     prequest->language = l;
57         
58         prequest->disk_filename = input_file;
59     
60         // Hand request off to the processing queue
61         queue.schedule(prequest);
62     }
