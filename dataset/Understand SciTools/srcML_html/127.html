48     /**
49     * srcml_translator
50     *
51     * Class for translation of individual files
52     * and projects into srcML.  Inherits from Language
53     * to set what language to use for parsing.
54     */
55     class srcml_translator : public Language {
56     public:
57     
58         // constructor
59         srcml_translator(char ** str_buf,
60                          size_t * size,
61                          const char* xml_encoding,
62                          OPTION_TYPE & op,
63                          std::vector<std::string> & prefix,
64                          std::vector<std::string> & uri,
65                          boost::optional<std::pair<std::string, std::string> > processing_instruction,
66                          size_t tabsize,
67                          int language,
68                          const char* revision,
69                          const char* url,
70                          const char* filename,
71                          const char* version,
72                          const std::vector<std::string> & attributes,
73                          const char* timestamp,
74                          const char* hash,
75                          const char* encoding);
76     
77         // constructor
78         srcml_translator(xmlOutputBuffer * output_buffer,
79                          const char* xml_encoding,
80                          OPTION_TYPE& op,
81                          std::vector<std::string> & prefix,
82                          std::vector<std::string> & uri,
83                          boost::optional<std::pair<std::string, std::string> > processing_instruction,
84                          size_t tabsize,
85                          int language,
86                          const char* revision,
87                          const char* url,
88                          const char* filename,
89                          const char* version,
90                          const std::vector<std::string> & attributes,
91                          const char* timestamp, 
92                          const char* hash,
93                          const char* encoding);
94     
95         void set_macro_list(std::vector<std::string> & list);
96     
97         void close();
98     
99         void translate(UTF8CharBuffer* parser_input);
100     
101         bool add_unit(const srcml_unit * unit, const char * xml);
102         bool add_unit_content(const srcml_unit * unit, const char * xml, int size);
103         bool add_unit_raw(const char * xml, int size);
104         bool add_start_unit(const srcml_unit * unit);
105         bool add_end_unit();
106         bool add_start_element(const char * prefix, const char * name, const char * uri);
107         bool add_end_element();
108         bool add_namespace(const char * prefix, const char * uri);
109         bool add_attribute(const char * prefix, const char * name, const char * uri, const char * content);
110         bool add_string(const char * content);
111     
112         xmlOutputBufferPtr output_buffer() { return out.output_buffer; }
113     
114         xmlTextWriterPtr output_textwriter() { return out.xout; }
115     
116         void set_text_only() { text_only = true; }
117     
118         // destructor
119         ~srcml_translator();
120     
121     private:
122     
123         /** is this the first unit */
124         bool first;
125     
126         /** the unit revision attribute */
127         const char* revision;
128     
129         /** the unit url attribute */
130         const char* url;
131     
132         /** the unit filename attribute */
133         const char* filename;
134     
135         /** the unit version attribute */
136         const char* version;
137     
138         /** the unit timestamp attribute */
139         const char* timestamp;
140     
141         /** the unit hash attribute */
142         const char* hash;
143     
144         /** the unit encoding attribute */
145         const char* encoding;
146     
147         /** an array of name-value attribute pairs */
148         const std::vector<std::string> & attributes;
149     
150         std::vector<std::string> & prefix;
151         std::vector<std::string> & uri;
152     
153         /** translation options */
154         OPTION_TYPE& options;
155     
156         /** buffer to hold output to memory */
157         xmlBuffer * buffer;
158     
159         /** member for handling output */
160         srcMLOutput out;
161     
162         /** size of tabstop */
163         size_t tabsize;
164     
165         /** list of user defined macros */
166         std::vector<std::string> user_macro_list;
167     
168         /** location to store output when output is to memory */
169         char ** str_buffer;
170         
171         /** location to store size of output when output is to memory */
172         size_t * size;
173     
174         /** mark if have outputted starting unit tag for by element writing */
175         bool is_outputting_unit;
176     
177         /** track depth for by element writing */
178         int output_unit_depth;
179     
180         /** text-only mode (no XML) */
181         bool text_only;
182     
183     };
