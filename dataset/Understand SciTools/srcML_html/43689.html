23     /**
24      * Adds stream parsing capabilities to a Parser class.
25      *
26      * Fundamentally it provides a nextToken method so that another Parser
27      * can use this as a TokenStream.
28      *
29      * A new class is created with the template parameter Base and
30      * TokenStream as base classes.  This allows other Parsers to use it as
31      * a TokenStream in their constructor.
32      *
33      * It also implements the TokenStream nextToken method and provides
34      * pushToken methods to fill the token buffer.
35      *
36      * Parser grammars that use this class must also inherit from TokenParser.hpp
37      */
38     
39     #ifndef INCLUDED_SRCMLTOKENSTREAM_HPP
40     #define INCLUDED_SRCMLTOKENSTREAM_HPP
41     
42     #include <antlr/TokenStream.hpp>
43     
44     /**
45      * TokenStream
46      *
47      * Class for a token stream.
48      */
49     class TokenStream {
50     
51     public:
52     
53         /** abstract method for getting next token */
54         virtual const antlr::RefToken& nextToken() = 0;
55     
56         /**
57          * ~TokenStream
58          *
59          * Destructor.
60          */
61         virtual ~TokenStream() {}
62     
63     };
64     
65     #endif
