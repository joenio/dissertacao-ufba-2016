23     #ifndef WRITE_QUEUE_HPP
24     #define WRITE_QUEUE_HPP
25     
26     #include <srcml.h>
27     #include <parse_request.hpp>
28     #pragma GCC diagnostic push
29     #pragma GCC diagnostic ignored "-Wshorten-64-to-32"
30     #include <threadpool.hpp>
31     #pragma GCC diagnostic pop
32     #include <boost/function.hpp>
33     #include <prio_scheduler_strict.hpp>
34     
35     class WriteQueue {
36     public:
37     
38         WriteQueue(boost::function<void(ParseRequest*)> writearg, bool order)
39             : write(writearg), pool(1) {
40     
41             set_ordering(order);
42         }
43     
44         /* puts an element in the back of the queue by swapping with parameter */
45         inline void schedule(ParseRequest* pvalue) {
46     
47             pool.schedule(prio_strict_task_func(pvalue->position, boost::bind(write, pvalue)));
48         }
49     
50         inline void wait() {
51     
52             pool.wait();
53         }
54     
55     private:
56         boost::function<void(ParseRequest*)> write;
57         prio_strict_pool pool;
58     };
59     
60     #endif
