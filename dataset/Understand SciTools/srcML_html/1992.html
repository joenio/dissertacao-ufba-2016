31     struct pretty_template_t {
32         boost::optional<std::string> header;
33         std::vector<std::string> header_args;
34         boost::optional<std::string> body;
35         std::vector<std::string> body_args;
36         boost::optional<std::string> footer;
37         std::vector<std::string> footer_args;
38         boost::optional<size_t> error_location;
39     };
40     
41     /*
42     PRETTY Args
43     - %C: total number of units (only for footer)
44     - %F: file attribute on the archive
45     - %f: file name attribute on the unit
46     - %h: hash attribute on the unit
47     - %i: index of the unit within the archive
48     - %l: unit language 
49     - %N: print xmlns namespace %N = full namespace, %N:u = namespace uri, %N:p = namespace prefix
50     - %S: source encoding attribute on the archive
51     - %s: source encoding attribute on the unit
52     - %t: timestamp on unit
53     - %U: directory attribute on the archive
54     - %u: directory attribute on the unit
55     - %V: version attribute on the archive
56     - %v: version attribute on the unit
57     - %X: XML encoding on the archive
58     - %x: XML encoding attribute on the unit
59     */
60     
61     const std::string valid_header_args = "FNSUVX";
62     const std::string valid_body_args = "FfhilNSstUuVvXx";
63     const std::string valid_footer_args = "CFNSUVX";
64     
65     int srcml_pretty(srcml_archive* srcml_arch, const std::string& pretty_input);
