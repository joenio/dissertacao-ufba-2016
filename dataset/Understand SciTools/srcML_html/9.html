23     #include <trace_log.hpp>
24     #include <srcml_cli.hpp>
25     #include <iomanip>
26     #include <stdio.h>
27     
28     TraceLog::TraceLog(int options)
29         : count(0), num_skipped(), num_error(0) {
30     
31         enabled = options & SRCML_COMMAND_VERBOSE;
32     }
33     
34     void TraceLog::header() {
35     
36         if (!enabled)
37             return;
38     
39         fprintf(stderr, "Source encoding:  (null)\nXML encoding:  UTF-8\n");
40     }
41     
42     void TraceLog::report() {
43     
44         if (!enabled)
45             return;
46     
47         fprintf(stderr, "\nTranslated: %d\tSkipped: %d\tError: %d\tTotal: %d\n", count, num_skipped, num_error, (count + num_skipped + num_error));
48     }
49     
50     TraceLog& operator<<(TraceLog& tlog, char c) {
51     
52         if (!tlog.enabled)
53             return tlog;
54     
55         if (c != '-') {
56             fprintf(stderr, "%5d ", ++tlog.count);
57         } else {
58             fprintf(stderr, "%5c ", '-');
59             ++tlog.num_skipped;
60         }
61     
62         return tlog;
63     }
64     
65     TraceLog& operator<<(TraceLog& tlog, const std::string& s) {
66     
67         if (!tlog.enabled)
68             return tlog;
69     
70         fprintf(stderr, "%s\n", s.c_str());
71     
72         return tlog;
73     }
