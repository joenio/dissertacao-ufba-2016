456     /**
457      * srcMLOutput
458      * @param ints a token stream
459      * @param filename if output to a file
460      * @param language the unit language
461      * @param xml_enc output encoding
462      * @param op output operations
463      * @param ts tabstop size
464      * @param output_buffer if output is to a output buffer
465      *
466      * Constructor. Handles all outputs.
467      */
468     srcMLOutput::srcMLOutput(TokenStream* ints,
469                              xmlOutputBuffer * output_buffer,
470                              const char* language,
471                              const char* xml_enc,
472                              OPTION_TYPE& op,
473                              const std::vector<std::string> & attributes,
474                              boost::optional<std::pair<std::string, std::string> > processing_instruction,
475                              size_t ts)
476         : last_line(0), last_line2(0), last_column(0), end_position_output(false), input(ints), xout(0), output_buffer(output_buffer), unit_language(language), unit_url(0), unit_filename(0),
477           unit_version(0), options(op), xml_encoding(xml_enc), unit_attributes(attributes), processing_instruction(processing_instruction),
478           openelementcount(0), curline(0), curcolumn(0), tabsize(ts), depth(0), 
479           debug_time_start(boost::posix_time::microsec_clock::universal_time())
480     {
481     
482         if(!isoption(options, SRCML_OPTION_OPTIONAL_MARKUP)) {
483     
484             ElementPrefix[SSTRING]   = SRCML_SRC_NS_URI_POS;
485             ElementPrefix[SCHAR]     = SRCML_SRC_NS_URI_POS;
486             ElementPrefix[SLITERAL]  = SRCML_SRC_NS_URI_POS;
487             ElementPrefix[SBOOLEAN]  = SRCML_SRC_NS_URI_POS;
488             ElementPrefix[SNULL]     = SRCML_SRC_NS_URI_POS;
489             ElementPrefix[SCOMPLEX]  = SRCML_SRC_NS_URI_POS;
490             ElementPrefix[SNIL]      = SRCML_SRC_NS_URI_POS;
491             ElementPrefix[SOPERATOR] = SRCML_SRC_NS_URI_POS;
492             ElementPrefix[SMODIFIER] = SRCML_SRC_NS_URI_POS;
493     
494         }
495     
496         if (isoption(options, SRCML_OPTION_POSITION) && isoption(options, SRCML_OPTION_LINE))
497           num2process[2] = &srcMLOutput::processTextPositionLine;
498         else if (isoption(options, SRCML_OPTION_POSITION))
499           num2process[2] = &srcMLOutput::processTextPosition;
500     
501     }

530         num2prefix.push_back(SRCML_SRC_NS_PREFIX_DEFAULT);
531         num2prefix.push_back(SRCML_CPP_NS_PREFIX_DEFAULT);
532         num2prefix.push_back(SRCML_ERR_NS_PREFIX_DEFAULT);
533         num2prefix.push_back(SRCML_EXT_LITERAL_NS_PREFIX_DEFAULT);
534         num2prefix.push_back(SRCML_EXT_OPERATOR_NS_PREFIX_DEFAULT);
535         num2prefix.push_back(SRCML_EXT_MODIFIER_NS_PREFIX_DEFAULT);
536         num2prefix.push_back(SRCML_EXT_POSITION_NS_PREFIX_DEFAULT);
537         num2prefix.push_back(SRCML_EXT_OPENMP_NS_PREFIX_DEFAULT);
538     
539         num2uri.push_back(SRCML_SRC_NS_URI);
540         num2uri.push_back(SRCML_CPP_NS_URI);
541         num2uri.push_back(SRCML_ERR_NS_URI);
542         num2uri.push_back(SRCML_EXT_LITERAL_NS_URI);
543         num2uri.push_back(SRCML_EXT_OPERATOR_NS_URI);
544         num2uri.push_back(SRCML_EXT_MODIFIER_NS_URI);
545         num2uri.push_back(SRCML_EXT_POSITION_NS_URI);
546         num2uri.push_back(SRCML_EXT_OPENMP_NS_URI);
547     
548         for(std::vector<std::string>::size_type outer_pos = 0; outer_pos < uri.size(); ++ outer_pos) {
549     
550             std::vector<std::string>::size_type pos;
551             for(pos = 0; pos < num2uri.size() && num2uri[pos] != uri[outer_pos]; ++pos)
552                 ;
553     
554             if(pos < num2uri.size()) {
555     
556                 num2prefix[pos] = prefix[outer_pos];
557     
558             } else {
559     
560                 num2prefix.push_back(prefix[outer_pos]);
561                 num2uri.push_back(uri[outer_pos]);
562     
563             }
564     
565         }
566     
567         // setup attributes names for line/column position if used
568         if (isoption(options, SRCML_OPTION_POSITION)) {
569     
570             lineAttribute = num2prefix[SRCML_EXT_POSITION_NS_URI_POS];
571             lineAttribute += ":line";
572     
573             line2Attribute = num2prefix[SRCML_EXT_POSITION_NS_URI_POS];
574             line2Attribute += ":line2";
575     
576             columnAttribute = num2prefix[SRCML_EXT_POSITION_NS_URI_POS];
577             columnAttribute += ":column";
578         }
579     
580     }

897     /**
898      * outputNamesapces
899      * @param xout the xml writer to write namespaces
900      * @param options the current set options
901      * @param depth the depth in the archive number of output units
902      * @param outer is this an outer unit or inner unit
903      *
904      * Output the namespaces on the units.
905      */
906     void srcMLOutput::outputNamespaces(xmlTextWriterPtr xout, const OPTION_TYPE& options, int depth) {
907     
908         // figure out which namespaces are needed
909         char const * const ns[] = {
910     
911             // main srcML namespace declaration always used
912             (depth == 0) ? SRCML_SRC_NS_URI : 0,
913     
914             // main cpp namespace declaration
915             isoption(options, SRCML_OPTION_CPP) && (isoption(options, SRCML_OPTION_ARCHIVE) == !(depth == 0)) ? SRCML_CPP_NS_URI : 0,
916     
917             // optional debugging xml namespace
918             (depth == 0) && isoption(options, SRCML_OPTION_DEBUG)    ? SRCML_ERR_NS_URI : 0,
919     
920             // optional literal xml namespace
921             (depth == 0) && isoption(options, SRCML_OPTION_OPTIONAL_MARKUP) && isoption(options, SRCML_OPTION_LITERAL)  ? SRCML_EXT_LITERAL_NS_URI : 0,
922     
923             // optional operator xml namespace
924             (depth == 0) && isoption(options, SRCML_OPTION_OPTIONAL_MARKUP) && isoption(options, SRCML_OPTION_OPERATOR) ? SRCML_EXT_OPERATOR_NS_URI : 0,
925     
926             // optional modifier xml namespace
927             (depth == 0) && isoption(options, SRCML_OPTION_OPTIONAL_MARKUP) && isoption(options, SRCML_OPTION_MODIFIER) ? SRCML_EXT_MODIFIER_NS_URI : 0,
928     
929             // optional position xml namespace
930             (depth == 0) && isoption(options, SRCML_OPTION_POSITION) ? SRCML_EXT_POSITION_NS_URI : 0,
931     
932             // optional position xml namespace
933             (depth == 0) && isoption(options, SRCML_OPTION_OPENMP) ? SRCML_EXT_OPENMP_NS_URI : 0,
934     
935         };
936     
937         // output the namespaces
938         for (unsigned int i = 0; i < sizeof(ns) / sizeof(ns[0]); ++i) {
939             if (!ns[i])
940                 continue;
941     
942             std::string prefix = "xmlns";
943             if (num2prefix[i][0] != '\0') {
944                 prefix += ':';
945                 prefix += num2prefix[i];
946             }
947     
948             xmlTextWriterWriteAttribute(xout, BAD_CAST prefix.c_str(), BAD_CAST ns[i]);
949         }
950     
951         if(depth == 0) {
952     
953             for(std::vector<std::string>::size_type pos =  SRCML_EXT_OPENMP_NS_URI_POS + 1; pos < num2prefix.size(); ++pos) {
954     
955                 std::string prefix = "xmlns";
956                 if (num2prefix[pos][0] != '\0') {
957                     prefix += ':';
958                     prefix += num2prefix[pos];
959              }
960             
961                 xmlTextWriterWriteAttribute(xout, BAD_CAST prefix.c_str(), BAD_CAST num2uri[pos].c_str());
962                 
963             }
964     
965         }
966     
967     }

969     /**
970      * startUnit
971      * @param language the language attribute
972      * @param revision what version of srcML
973      * @param url the url attribute
974      * @param filename the filename attribute
975      * @param version the version attribute
976      * @param timestamp the timestamp attribute
977      * @param hash the hash attribute
978      * @param encoding the encoding attribute
979      * @param outer is this an outer or inner unit
980      *
981      * Output the start of a unit tag.
982      */
983     void srcMLOutput::startUnit(const char* language, const char* revision,
984                                 const char* url, const char* filename,
985                                 const char* version, const char* timestamp,
986                                 const char* hash,
987                                 const char* encoding,
988                                 const std::vector<std::string> & attributes,
989                                 bool output_macrolist) {
990     
991         const char * prefix = num2prefix[0].c_str();
992         std::string maintag = prefix ? prefix : "";
993         if (!maintag.empty())
994             maintag += ":";
995         maintag += "unit";
996     
997         // start of main tag
998         srcMLTextWriterStartElement(xout, BAD_CAST /* type2name(SUNIT) */ maintag.c_str());
999     
1000         // outer units have namespaces
1001         if (/* outer && */ isoption(options, SRCML_OPTION_NAMESPACE_DECL)) {
1002             outputNamespaces(xout, options, depth);
1003         }
1004     
1005         // setting up for tabs, even if not used
1006         std::ostringstream stabs;
1007         std::string tabattribute;
1008         if (isoption(options, SRCML_OPTION_POSITION)) {
1009             stabs << tabsize;
1010             tabattribute = num2prefix[SRCML_EXT_POSITION_NS_URI_POS];
1011             tabattribute.append(":tabs");
1012         }
1013     
1014         std::string soptions;
1015         std::string SEP;
1016         //if(isoption(options, SRCML_OPTION_XML_DECL))        { soptions = "XMLDECL"; }
1017         //if(isoption(options, SRCML_OPTION_NAMESPACE_DECL))  { if(soptions != "") SEP = ","; soptions += SEP + "NAMESPACEDECL"; }
1018         if(isoption(options, SRCML_OPTION_CPP_TEXT_ELSE))  { if(SEP.empty() && soptions != "") SEP = ","; soptions += SEP + "CPP_TEXT_ELSE"; }
1019         if(isoption(options, SRCML_OPTION_CPP_MARKUP_IF0)) { if(SEP.empty() && soptions != "") SEP = ","; soptions += SEP + "CPP_MARKUP_IF0"; }
1020         if(isoption(options, SRCML_OPTION_EXPRESSION))     { if(SEP.empty() && soptions != "") SEP = ","; soptions += SEP + "EXPRESSION"; }
1021         if(isoption(options, SRCML_OPTION_LINE))           { if(SEP.empty() && soptions != "") SEP = ","; soptions += SEP + "LINE"; }
1022         if(isoption(options, SRCML_OPTION_NESTIF))         { if(SEP.empty() && soptions != "") SEP = ","; soptions += SEP + "NESTIF"; }
1023         if(isoption(options, SRCML_OPTION_CPPIF_CHECK))    { if(SEP.empty() && soptions != "") SEP = ","; soptions += SEP + "CPPIF_CHECK"; }
1024         if(isoption(options, SRCML_OPTION_WRAP_TEMPLATE))  { if(SEP.empty() && soptions != "") SEP = ","; soptions += SEP + "WRAP_TEMPLATE"; }
1025         if(!isoption(options, SRCML_OPTION_TERNARY))       { if(SEP.empty() && soptions != "") SEP = ","; soptions += SEP + "OPERATOR_TERNARY"; }
1026     
1027         std::string stab = stabs.str();
1028     
1029         // list of attributes
1030         const char* const attrs[][2] = {
1031     
1032             { UNIT_ATTRIBUTE_REVISION, !isoption(options, SRCML_OPTION_NO_REVISION) ? revision : 0 },
1033     
1034             // language attribute
1035             { UNIT_ATTRIBUTE_LANGUAGE, language },
1036     
1037             // url attribute
1038             { UNIT_ATTRIBUTE_URL, url },
1039     
1040             // filename attribute
1041             { UNIT_ATTRIBUTE_FILENAME, filename },
1042     
1043             // version attribute
1044             { UNIT_ATTRIBUTE_VERSION, version },
1045     
1046             // position tab setting
1047             { tabattribute.c_str(), isoption(options, SRCML_OPTION_POSITION) ? stab.c_str() : 0 },
1048     
1049             // timestamp attribute
1050             { UNIT_ATTRIBUTE_TIMESTAMP, timestamp },
1051     
1052             // hash attribute
1053             { UNIT_ATTRIBUTE_HASH, hash },
1054     
1055             // source encoding attribute
1056             { UNIT_ATTRIBUTE_SOURCE_ENCODING, isoption(options, SRCML_OPTION_STORE_ENCODING) ? encoding : 0 },
1057     
1058             { UNIT_ATTRIBUTE_OPTIONS,  depth == 0 && (isoption(options, SRCML_OPTION_NESTIF)
1059              || isoption(options, SRCML_OPTION_CPPIF_CHECK) || isoption(options, SRCML_OPTION_WRAP_TEMPLATE) || !isoption(options, SRCML_OPTION_TERNARY)) ? soptions.c_str() : 0 },
1060     
1061         };
1062     
1063         // output attributes
1064         for (unsigned int i = 0; i < sizeof(attrs) / sizeof(attrs[0]); ++i) {
1065             if (!attrs[i][1])
1066                 continue;
1067     
1068             xmlTextWriterWriteAttribute(xout, BAD_CAST attrs[i][0], BAD_CAST attrs[i][1]);
1069         }
1070     
1071         for(std::vector<std::string>::size_type pos = 0; pos < attributes.size(); pos += 2) {
1072     
1073             xmlTextWriterWriteAttribute(xout, BAD_CAST attributes[pos].c_str(), BAD_CAST attributes[pos + 1].c_str());
1074     
1075         }
1076     
1077         if(output_macrolist) outputMacroList();
1078     
1079         ++depth;
1080     }
