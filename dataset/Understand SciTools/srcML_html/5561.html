27     #include <srcml.h>
28     #include <parse_request.hpp>
29     #include <write_queue.hpp>
30     #include <boost/static_assert.hpp>
31     #include <sha1utilities.hpp>
32     #include <srcml_options.hpp>
33     #include <srcml_cli.hpp>
34     #include <string>
35     #include <boost/filesystem.hpp>
36     
37     // creates initial unit, parses, and then sends unit to write queue
38     void srcml_consume(ParseRequest* request, WriteQueue* write_queue) {
39     
40         // error passthrough to output for proper output in trace
41         if (request->status) {
42             request->unit = 0;
43             write_queue->schedule(request);
44             return;
45         }
46     
47         // NOTE: thread task cannot throw exception
48     
49         // global access to options
50         bool isseparatearchive = SRCML_COMMAND_NOARCHIVE & SRCMLOptions::get();
51     
52         // current output archive
53         srcml_archive* srcml_arch = request->srcml_arch;
54         if (isseparatearchive) {
55             srcml_arch = srcml_archive_clone(request->srcml_arch);
56             srcml_archive_disable_option(srcml_arch, SRCML_OPTION_ARCHIVE);
57             srcml_archive_enable_option(srcml_arch, SRCML_OPTION_HASH);
58     
59             //Build the output filename
60     
61             //Filenames from directories come in as full paths
62             //Setup the directory on the filesystem
63             boost::filesystem::create_directories(*request->disk_dir);
64     
65             size_t pos = request->filename->find_last_of("/\\");
66     
67             if (pos != std::string::npos) {
68                 ++pos;
69             }
70             else {
71                 pos = 0;
72             }
73             
74             //Ensure that the directory path has a final "/" when appended to filename
75             if (request->disk_dir->at(request->disk_dir->length() -1) != '/')
76                 *request->disk_dir += "/";
77     
78             std::string xml_filename = *request->disk_dir + request->filename->substr(pos) + ".xml";
79             srcml_archive_write_open_filename(srcml_arch, xml_filename.c_str(), 0);
80             request->srcml_arch = srcml_arch;
81         }
82     
83         std::string original_filename;
84     
85         // construct and parse the unit
86         srcml_unit* unit = request->unit;
87         int status = SRCML_STATUS_OK;
88         try {
89     
90             // create the unit start tag
91             if (!unit)
92                 if (!(unit = srcml_unit_create(srcml_arch)))
93                     throw SRCML_STATUS_ERROR;
94     
95             // language attribute, required if from memory
96             if ((status = srcml_unit_set_language(unit, request->language.c_str())) != SRCML_STATUS_OK)
97                 throw status;
98     
99             // (optional) directory attribute
100             if (!(srcml_archive_get_options(srcml_arch) & SRCML_OPTION_ARCHIVE) && request->url && ((status = srcml_unit_set_url(unit, request->url->c_str())) != SRCML_STATUS_OK))
101                 throw status;
102     
103             // (optional) filename attribute
104             if (request->filename) {
105     
106                 original_filename = *request->filename;
107                 
108                 // Cleanup filename
109                 while (request->filename->at(0) == '.' || request->filename->at(0) == '/') {
110                     request->filename->erase(0,1);
111                 }
112                 
113                 if ((status = srcml_unit_set_filename(unit, request->filename->c_str())) != SRCML_STATUS_OK)
114                     throw status;
115             }
116     
117             // (optional) version attribute
118             if (request->version && ((status = srcml_unit_set_version(unit, request->version->c_str())) != SRCML_STATUS_OK))
119                 throw status;
120     
121             if (request->time_stamp)
122                 srcml_unit_set_timestamp(unit, request->time_stamp->c_str());
123     
124             // sha1 attribute, if hash is on
125             // sha1 value based on the code as encoded (source text encoding) in the original file
126             if (!request->disk_filename && (SRCML_COMMAND_NOARCHIVE & SRCMLOptions::get())) {
127             
128     #ifdef _MSC_BUILD
129                 unsigned char md[20];
130                 /** msvc hash provider object */
131                 HCRYPTPROV   crypt_provider;
132                 BOOL success = CryptAcquireContext(&crypt_provider, NULL, NULL, PROV_RSA_FULL, 0);
133                 if(! success && GetLastError() == NTE_BAD_KEYSET)
134                     success = CryptAcquireContext(&crypt_provider, NULL, NULL, PROV_RSA_FULL, CRYPT_NEWKEYSET);
135                 /** msvc hash object */
136                 HCRYPTHASH   crypt_hash;
137                 CryptCreateHash(crypt_provider, CALG_SHA1, 0, 0, &crypt_hash);
138                 CryptHashData(crypt_hash, (BYTE *)&request->buffer.front(), request->buffer.size(), 0);
139                 DWORD        SHA_DIGEST_LENGTH;
140                 DWORD        hash_length_size = sizeof(DWORD);
141                 CryptGetHashParam(crypt_hash, HP_HASHSIZE, (BYTE *)&SHA_DIGEST_LENGTH, &hash_length_size, 0);
142                 CryptGetHashParam(crypt_hash, HP_HASHVAL, (BYTE *)md, &SHA_DIGEST_LENGTH, 0);
143                 CryptDestroyHash(crypt_hash);
144                 CryptReleaseContext(crypt_provider, 0);
145     #else
146                 unsigned char md[SHA_DIGEST_LENGTH];
147                 if (SHA1((const unsigned char*)&request->buffer.front(), (SHA_LONG)request->buffer.size(), md) == 0)
148                     throw SRCML_STATUS_ERROR;
149     #endif
150     
151                 const char outmd[] = { HEXCHARASCII(md), '\0' };
152     
153     #ifndef _MSC_BUILD
154                 BOOST_STATIC_ASSERT_MSG(sizeof(outmd)/sizeof(outmd[0]) == (SHA_DIGEST_LENGTH * 2 + 1),
155                     "Wrong size for SHA_DIGEST_LENGTH conversion");
156     #endif
157     
158     //            fprintf(stderr, "DEBUG:  %s %s %d DATA: %s\n", __FILE__,  __FUNCTION__, __LINE__, outmd);
159                 
160                 srcml_unit_set_hash(unit, outmd);
161             }
162             
163             // parse the buffer/file (unless it is already form a srcml archive)
164             if (request->disk_filename)
165                 status = srcml_unit_parse_filename(unit, request->disk_filename->c_str());
166             else if (!request->unit)
167                 status = srcml_unit_parse_memory(unit, &request->buffer.front(), request->buffer.size());
168     
169             if (status != SRCML_STATUS_OK)
170                 // FIXME: Cannot throw exception from thread
171                 throw status;
172     
173         } catch (...) {
174     
175             fprintf(stderr, "srcml: Unable to open file %s\n", original_filename.c_str());
176             if (unit)
177                 srcml_unit_free(unit);
178             unit = 0;
179             if (request)
180                 delete request;
181             request = 0;
182             return;
183         }
184     
185         // schedule unit for output
186         request->unit = unit;
187         request->status = status;
188         write_queue->schedule(request);
189     }
