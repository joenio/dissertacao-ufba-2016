29     /**
30      * ModeStack
31      *
32      * Class representing a stack of modes that direct parsing.
33      * Modes also keep srcMLState such as open parethesis/curly braces/tags.
34      */
35     class ModeStack : public TokenParser {
36     
37     public:
38     
39         #include "Mode.hpp"
40     
41        /**
42          * ModeStack
43          * @param ptp the token parser
44          *
45          * Constructor.  Create mode stack from TokenParser and current language.
46          */
47         ModeStack(TokenParser* ptp)
48             : parser(ptp)
49         {}
50     
51         /**
52          * ~ModeStack
53          *
54          * Destructor.  Ends all open modes/states.
55          */
56         ~ModeStack() {
57     
58             // end all modes
59             endAllModes();
60     
61         }
62     
63          /** token parser */
64         TokenParser* parser;
65     
66         /** stack of states/modes */
67         std::list<srcMLState> st;
68     
69     protected:
70     
71         /**
72          * currentState
73          *
74          * Get the current state (const no modification).
75          *
76          * @returns the current state.
77          */
78         const srcMLState& currentState() const {
79             if (st.empty())
80                 throw Segmentation_Fault();
81     
82             return st.back();
83         }
84     
85         /**
86          * currentState
87          *
88          * Delegate to get the current state (allow modification).
89          *
90          * @returns the current state.
91          */
92         srcMLState& currentState() {
93             if (st.empty())
94                 throw Segmentation_Fault();
95     
96             return st.back();
97         }
98     
99         /**
100          * startNewMode
101          * @param m new modes to start
102          *
103          * Delegate to create a new mode and place it on top of stack.
104          */
105         void startNewMode(const srcMLState::MODE_TYPE& m) {
106     
107             // prepare for the new stack
108             st.push_back(srcMLState(m, !empty() ? getTransparentMode() : 0, !empty() ? getMode() : 0));
109         }
110     
111         /**
112          * endMode
113          *
114          * Delegate to remove the current mode (pop from stack).
115          */
116         void endMode() {
117     
118             if (st.size() == 1)
119                 throw Segmentation_Fault();
120     
121             popMode();
122         }
123     
124     #pragma GCC diagnostic push
125     #pragma GCC diagnostic ignored "-Wunused-parameter"
126     
127         /**
128          * endMode
129          * @param m mode to end
130          *
131          * Delegate to remove the current mode m (pop from stack).
132          * Does not actually check or use m.
133          */
134         void endMode(const srcMLState::MODE_TYPE& m) {
135     
136             if (st.size() <= 1)
137                 throw Segmentation_Fault();
138     
139             popMode();
140         }
141     
142     #pragma GCC diagnostic pop
143     
144         /**
145          * endLastMode
146          *
147          * Delegate to ends a mode (not necessarily last) popping from stack.
148          */
149         void endLastMode() {
150     
151             popMode();
152         }
153     
154         /**
155          * getMode
156          *
157          * Delegate to get the current mode from stack (top of stack).
158          *
159          * @returns the current mode.
160          */
161         srcMLState::MODE_TYPE getMode() const {
162     
163             return !st.empty() ? st.back().getMode() : 0;
164         }
165     
166         /**
167          * getPrevMode
168          *
169          * Delegate to get the previousmode from stack (top of stack).
170          *
171          * @returns the previous mode.
172          */
173         srcMLState::MODE_TYPE getPrevMode() const {
174     
175             return !st.empty() ? st.back().getMode() : 0;
176         }
177     
178         /**
179          * getTransparentMode
180          *
181          * Delegate to get the transparent mode from stack (top of stack).
182          *
183          * @returns the transparent mode.
184          */
185         srcMLState::MODE_TYPE getTransparentMode() const {
186     
187             return !st.empty() ? st.back().getTransparentMode() : 0;
188         }
189     
190     
191         /**
192          * getFirstMode
193          * @param m modes to query for
194          *
195          * Get the first mode matching m
196          *
197          * @returns the matching mode
198          */
199         srcMLState::MODE_TYPE getFirstMode(const srcMLState::MODE_TYPE& m) const {
200     
201             for(std::list<srcMLState>::const_reverse_iterator citr = st.rbegin(); citr != st.rend(); ++citr) {
202     
203                 if((citr->getMode() & m) != 0) return citr->getMode();
204     
205             }
206     
207             return 0;
208     
209         }
210             
211         /**
212          * setMode
213          * @param m modes to add to current mode
214          *
215          * Add the modes m to the current modes.
216          */
217         void setMode(const srcMLState::MODE_TYPE& m) {
218             if (st.empty())
219                 throw Segmentation_Fault();
220     
221             st.back().setMode(m);
222         }
223     
224         /**
225          * clearMode
226          * @param m modes to remove modes from current mode
227          *
228          * Delegate to removes the modes m from the current modes.
229          */
230          void clearMode(const srcMLState::MODE_TYPE& m) {
231             if (st.empty())
232                 throw Segmentation_Fault();
233     
234             st.back().clearMode(m);
235         }
236     
237         /**
238          * push
239          * @param id of open element to add to current mode open element stack
240          *
241          * Delegate to add the open element id to the top of the open element of the current mode.
242          */
243         void push(int id) {
244             if (st.empty())
245                 throw Segmentation_Fault();
246     
247             st.back().push(id);
248         }
249     
250         /**
251          * pop
252          *
253          * Delegate to remove the top open element form the current mode.
254          */
255          void pop() {
256             if (st.empty())
257                 throw Segmentation_Fault();
258     
259             st.back().pop();
260         }
261     
262         /**
263          * size
264          *
265          * Delegate to get the current modes number of open elements
266          *
267          * @returns the number of open element in current mode.
268          */
269         int size() const {
270     
271             return (int)st.size();
272         }
273     
274         /**
275          * empty
276          *
277          * Delegate Predicte method to test if state stack is empty
278          *
279          * @returns if states in stack.
280          */
281         bool empty() const {
282     
283             return st.empty();
284         }
285     
286        /**
287          * inMode
288          * @param m mode to test if currently in
289          *
290          * Delegate predicte method to test if currently in mode m.
291          *
292          * @returns if in mode m.
293          */
294         bool inMode(const srcMLState::MODE_TYPE& m) const {
295     
296             return !st.empty() ? st.back().inMode(m) : false;
297         }
298     
299         /**
300          * inPrevMode
301          * @param m mode to test if previously in
302          *
303          * Delegate predicte method to test if previously in mode m.
304          *
305          * @returns if in previous mode m.
306          */
307         bool inPrevMode(const srcMLState::MODE_TYPE& m) const {
308     
309             return st.size() > 1 ? st.back().inPrevMode(m) : false;
310         }
311     
312          /**
313          * inTransparentMode
314          * @param m mode to test if transaprently in
315          *
316          * Delegate predicte method to test if transparently in mode m.
317          *
318          * @returns if in transparent mode m.
319          */
320         bool inTransparentMode(const srcMLState::MODE_TYPE& m) const {
321     
322             return !st.empty() ? st.back().inTransparentMode(m) : false;
323         }

326          * getParen
327          *
328          * Delegate to get the open parenthesis in current mode.
329          *
330          * @returns number of open parethesis in current mode
331          */
332         int getParen() const {
333             return !st.empty() ? st.back().getParen() : 0;
334         }

336         /**
337          * incParen
338          *
339          * Delegate to increment the number of open parenthesis.

348         /**
349          * decParen
350          *
351          * Delegate to decrement the number of open parenthesis.

355                 throw Segmentation_Fault();

357             st.back().decParen();
358         }

361          * getCurly
362          *
363          * Delegate to get the open curly braces in current mode.
364          *
365          * @returns number of open curly braces in current mode

367         int getCurly() const {
368             return !st.empty() ? st.back().getCurly() : 0;
369         }
370     
371         /**
372          * incCurly
373          *
374          * Delegate to increment the number of open curly braces.

376         void incCurly() {
377             if (st.empty())

379

381         }

383         /**
384          * decCurly
385          *
386          * Delegate to decrement the number of open curly braces.

388         void decCurly() {
389             if (st.empty())
390                 throw Segmentation_Fault();
391     
392             st.back().decCurly();
393         }

395         /**
396          * getTypeCount count
397          *
398          * Delegate to get the types in current mode.
399          *
400          * @returns number of types in current mode
401          */
402         int getTypeCount() const {
403             return !st.empty() ? st.back().getTypeCount() : 0;
404         }

406         /**
407          * setTypeCount count
408          * @param n the number of types to set to
409          *
410          * Delegate to set the types in current mode.
411          */
412         void setTypeCount(int n) {
413             if (st.empty())
414                 throw Segmentation_Fault();

416             st.back().setTypeCount(n);
417         }

419         /**
420          * incTypeCount
421          *
422          * Delegate to increment the number of types.

425             if (st.empty())

432          * decTypeCount
433          *
434          * Delegate to decrement the number of types.

436         void decTypeCount() {
437             if (st.empty())
438                 throw Segmentation_Fault();
439     
440             st.back().decTypeCount();
441         }
442     
443         /**
444          * endAllModes
445          *
446          * End all modes/states on stack.
447          */
448         void endAllModes() {
449     
450             // end all modes
451             while (!st.empty()) {
452                 endMode(getMode());
453             }
454         }
455     
456         /**
457          * popMode
458          *
459          * Delegate to remove the current mode/state from stack.
460          */
461         void popMode() {
462             if (st.empty())
463                 throw Segmentation_Fault();
464     
465             // close all open elements
466             while (!st.empty() && !st.back().openelements.empty()) {
467                 parser->endElement(st.back().openelements.top());
468             }
469     
470             st.pop_back();
471         }
472     
473         /**
474          * endTopMode
475          *
476          * Delegate to remove/pop the mode on the top of the stack.
477          * Equivalent to endMode.
478          */
479         void endTopMode() {
480     
481             endMode();
482         }
483     
484         /**
485          * replaceMode
486          * @param oldm modes to clear
487          * @param newm modes to set
488          *
489          * Clear the current modes oldm, and sets then sets the modes new.
490          */
491         void replaceMode(const srcMLState::MODE_TYPE& oldm, const srcMLState::MODE_TYPE& newm) {
492     
493             clearMode(oldm);
494             setMode(newm);
495         }
496     
497         /**
498          * dupMode
499          *
500          * Duplicate mode on top of stack for cppif.
501          */
502         void dupMode(const std::stack<int> & open_elements) {
503     
504             srcMLState dup = st.back();
505             st.back().setMode(MODE_TOP | MODE_END_AT_ENDIF);
506     
507             dup.openelements = open_elements;
508             dup.setMode(MODE_ISSUE_EMPTY_AT_POP);
509             st.push_back(dup);
510     
511         }
512     
513         /**
514          * insertModeAfter
515          * @param m mode to stop on
516          * @param new_m mode to insert
517          * @param open_elements open elements for new_m
518          *
519          * Insert a new mode (new_m) with open_elements after first occurence of m
520          */
521         void insertModeAfter(const srcMLState::MODE_TYPE& m, const srcMLState::MODE_TYPE& new_m, const std::stack<int> & open_elements) {
522     
523             std::list<srcMLState> alist;
524             while((st.back().getMode() & m) != m) {
525     
526                 alist.push_front(st.back());
527                 st.pop_back();
528     
529             }
530     
531             st.push_back(new_m);
532             st.back().openelements = open_elements;
533     
534            for(std::list<srcMLState>::iterator i = alist.begin(); i != alist.end(); ++i)
535                 st.push_back(*i);
536     
537         }
538     
539         /**
540          * dupDownOverMode
541          * @param m mode to stop on
542          *
543          * Duplicate modes on stack down to and including m.
544          */
545         void dupDownOverMode(const srcMLState::MODE_TYPE& m) {
546     
547             std::list<srcMLState> alist;
548             while((st.back().getMode() & m).none()) {
549     
550                 alist.push_front(st.back());
551                 st.pop_back();
552     
553             }
554     
555             alist.push_front(st.back());
556             st.pop_back();
557     
558             alist.front().setMode(MODE_TOP | MODE_END_AT_ENDIF);
559             for(std::list<srcMLState>::iterator i = alist.begin(); i != alist.end(); ++i) {
560                 i->setMode(MODE_END_AT_ENDIF);
561                 st.push_back(*i);
562             }
563     
564             alist.front().openelements = std::stack<int>();
565             for(std::list<srcMLState>::iterator i = alist.begin(); i != alist.end(); ++i) {
566                 i->setMode(MODE_ISSUE_EMPTY_AT_POP);
567                 st.push_back(*i);
568     
569             }
570     
571     
572         }
573     
574         /**
575         * endDownToMode
576         * @param mode mode to end down to
577         *
578         * End elements down to but not including the mode mode.
579         */
580         void endDownToMode(const srcMLState::MODE_TYPE& mode) {
581     
582             if (!inTransparentMode(mode))
583                 return;
584     
585             while (size() > 1 && !inMode(mode))
586                 endMode();
587         }
588     
589         /**
590          * endDownToModeSet
591          * @param mode mode to end down to
592          *
593          * End elements down to but not including the mode mode.
594          */
595          void endDownToModeSet(const srcMLState::MODE_TYPE& mode) {
596     
597             //  if (getTransparentMode() & (mode == 0))
598             //      return;
599     
600             while (size() > 1 && (getMode() & mode) == 0)
601                 endMode();
602         }
603     
604         /**
605          * endDownOverMode
606          * @param mode mode to end down to
607          *
608          * End elements down to and including the mode mode.
609          */
610          void endDownOverMode(const srcMLState::MODE_TYPE& mode) {
611     
612             if (!inTransparentMode(mode))
613                 return;
614     
615             while (size() > 1 && !inMode(mode))
616                 endMode();
617             endMode();
618         }
619     
620         /**
621          * endWhileMode
622          * @param mode mode to down over
623          *
624          * End elements while in mode mode.
625          */
626          void endWhileMode(const srcMLState::MODE_TYPE& mode) {
627     
628             while (size() > 1 && inMode(mode))
629                 endMode();
630         }
