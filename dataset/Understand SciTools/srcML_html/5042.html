42     void srcml_handler_dispatch(ParseQueue& queue,
43                               srcml_archive* srcml_arch,
44                               const srcml_request_t& srcml_request,
45                               const srcml_input_src& input) {
46     
47         // call appropriate handler
48         if (input.state == SRCML) {
49             // libsrcml can apply gz decompression
50             // all other srcml compressions require a per-input decompression stage
51             srcml_input_src uninput = input;
52             input_file(uninput);
53             srcml_input_srcml(queue, srcml_arch, uninput);
54     
55         } else if (input.protocol == "text") {
56     
57             src_input_text(queue, srcml_arch, srcml_request, input.filename);
58     
59         } else if (input.protocol == "filelist") {
60     
61             src_input_filelist(queue, srcml_arch, srcml_request, input);
62     
63         } else if (input.protocol == "file" && input.isdirectory) {
64     
65             src_input_filesystem(queue, srcml_arch, srcml_request, input);
66     
67         } else if (input.protocol == "file" && input.archives.empty() && input.compressions.empty()) {
68            
69             src_input_file(queue, srcml_arch, srcml_request, input);
70     
71         } else {
72            
73             src_input_libarchive(queue, srcml_arch, srcml_request, input);
74         }
75     }

77     // create srcml from the current request
78     void create_srcml(const srcml_request_t& srcml_request,
79                       const srcml_input_t& input_sources,
80                       const srcml_output_dest& destination) {
81     
82         // create the output srcml archive
83         srcml_archive* srcml_arch = srcml_archive_create();
84     
85         // set options for the output srcml archive
86         if (srcml_request.att_xml_encoding)
87             srcml_archive_set_xml_encoding(srcml_arch, srcml_request.att_xml_encoding->c_str());
88     
89         if (srcml_request.src_encoding)
90             srcml_archive_set_src_encoding(srcml_arch, srcml_request.src_encoding->c_str());
91     
92         // for single input src archives (e.g., .tar), filename attribute is the source filename (if not already given)
93         if (srcml_request.att_url) {
94             srcml_archive_set_url(srcml_arch, src_prefix_resource(*srcml_request.att_url).c_str());
95         } else if (input_sources.size() == 1 && input_sources[0].archives.size() > 0) {
96     
97             // Cleanup filename
98             std::string url_name = src_prefix_resource(input_sources[0].filename);
99             while (url_name.at(0) == '.' || url_name.at(0) == '/') {
100                 url_name.erase(0,1);
101             }
102             
103             srcml_archive_set_url(srcml_arch, url_name.c_str());
104         }
105     
106         if (srcml_request.att_version)
107             srcml_archive_set_version(srcml_arch, srcml_request.att_version->c_str());
108     
109         if (srcml_request.markup_options)
110             srcml_archive_set_options(srcml_arch, srcml_archive_get_options(srcml_arch) | *srcml_request.markup_options);
111     
112         if (*srcml_request.markup_options & SRCML_OPTION_XML_DECL)
113             srcml_archive_disable_option(srcml_arch, SRCML_OPTION_XML_DECL);
114         else
115             srcml_archive_enable_option(srcml_arch, SRCML_OPTION_XML_DECL);
116     
117         if (*srcml_request.markup_options & SRCML_OPTION_NAMESPACE_DECL)
118             srcml_archive_disable_option(srcml_arch, SRCML_OPTION_NAMESPACE_DECL);
119         else
120             srcml_archive_enable_option(srcml_arch, SRCML_OPTION_NAMESPACE_DECL);
121     
122         if (srcml_request.att_language)
123             srcml_archive_set_language(srcml_arch, srcml_request.att_language->c_str());
124         else
125             srcml_archive_set_language(srcml_arch, SRCML_LANGUAGE_NONE);
126     
127         srcml_archive_set_tabstop(srcml_arch, srcml_request.tabs);
128     
129         // non-archive when:
130         //   only one input
131         //   no cli request to make it an archive
132         //   not a directory (if local file)
133         // TODO: check if a plain file. Source archives, i.e., .tar.gz, always produce srcml archives
134         if (input_sources.size() == 1 && input_sources[0].protocol != "filelist" &&
135             !(srcml_request.markup_options && (*srcml_request.markup_options & SRCML_OPTION_ARCHIVE)) &&
136             !input_sources[0].isdirectory && input_sources[0].archives.size() < 1) {
137     
138             srcml_archive_disable_option(srcml_arch, SRCML_OPTION_ARCHIVE);
139             
140             // If --hash is used, force hash for single input
141             if (*srcml_request.markup_options & SRCML_OPTION_HASH) {
142                 srcml_archive_enable_option(srcml_arch, SRCML_OPTION_HASH);
143             }
144             else {
145                 srcml_archive_disable_option(srcml_arch, SRCML_OPTION_HASH);
146             }
147     
148         } else {
149     
150             // if this is an archive, then no filename attribute is allowed
151             if (srcml_request.att_filename) {
152                 fprintf(stderr, "Attribute filename cannot be set for a srcML archive\n");
153                 exit(SRCML_STATUS_INVALID_ARGUMENT);
154             }
155     
156             srcml_archive_enable_option(srcml_arch, SRCML_OPTION_ARCHIVE);
157             srcml_archive_enable_option(srcml_arch, SRCML_OPTION_HASH);
158         }
159     
160         // register file extensions
161         BOOST_FOREACH(const std::string& ext, srcml_request.language_ext) {
162             size_t pos = ext.find('=');
163             srcml_archive_register_file_extension(srcml_arch, ext.substr(0, pos).c_str(), ext.substr(pos+1).c_str());
164         }
165     
166         // register xml namespaces
167         std::map<std::string, std::string>::const_iterator itr;
168         for(itr = srcml_request.xmlns_namespaces.begin(); itr != srcml_request.xmlns_namespaces.end(); ++itr){
169             srcml_archive_register_namespace(srcml_arch, (*itr).first.c_str(), (*itr).second.c_str());
170         }
171     
172         // create the srcML output file
173     
174         unsigned short compression = 0;
175         if (destination.extension == ".gz")
176             compression = 9;
177     
178         int status = 0;
179         if (SRCML_COMMAND_NOARCHIVE & SRCMLOptions::get()) {
180     
181             /* when no archive, this one is just used as a clone, so just don't open it */
182     
183         } else if (contains<int>(destination)) {
184     
185             status = srcml_archive_write_open_fd(srcml_arch, *destination.fd);
186         } else {
187     
188             status = srcml_archive_write_open_filename(srcml_arch, destination.c_str(), compression);
189         }
190     
191         // gzip compression available directly from libsrcml
192     
193         // setup the parsing queue
194         TraceLog log(SRCMLOptions::get());
195         log.header();
196         WriteQueue write_queue(boost::bind(srcml_write_request, _1, boost::ref(log)), srcml_request.command & SRCML_COMMAND_OUTPUT_ORDERED);
197         ParseQueue parse_queue(srcml_request.max_threads, boost::bind(srcml_consume, _1, &write_queue), write_queue);
198     
199         // process input sources
200         BOOST_FOREACH(const srcml_input_src& input, input_sources) {
201     /*
202     // if stdin, then there has to be data
203     // TODO: Safe to remove this? We already read data.
204     if (!contains<FILE*>(input) && (input.protocol == "stdin") && (srcml_request.command & SRCML_COMMAND_INTERACTIVE) &&
205     !src_input_stdin()) {
206     return; // stdin was requested, but no data was received
207     }
208     */
209             srcml_handler_dispatch(parse_queue, srcml_arch, srcml_request, input);
210         }
211     
212         // wait for the parsing and writing queues to finish
213         parse_queue.wait();
214         write_queue.wait();
215     
216         log.report();
217         
218         // close the created srcML archive
219         srcml_archive_close(srcml_arch);
220         srcml_archive_free(srcml_arch);
221     
222         // if we were writing to a file descriptor, then close it
223         if (contains<int>(destination))
224             close(*destination.fd);
225     }
