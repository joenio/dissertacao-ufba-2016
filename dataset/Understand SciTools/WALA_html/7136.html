12     package com.ibm.wala.cfg.exc.inter;
13     
14     import java.util.Map;
15     import java.util.Map.Entry;
16     
17     import com.ibm.wala.cfg.exc.ExceptionPruningAnalysis;
18     import com.ibm.wala.cfg.exc.InterprocAnalysisResult;
19     import com.ibm.wala.ipa.callgraph.CGNode;
20     import com.ibm.wala.ssa.SSAInstruction;
21     import com.ibm.wala.ssa.analysis.IExplodedBasicBlock;
22     
23     /**
24      * A wrapper for the interprocedural analysis result.
25      * 
26      * @author Juergen Graf <graf@kit.edu>
27      *
28      */
29     class InterprocAnalysisResultWrapper implements InterprocAnalysisResult<SSAInstruction, IExplodedBasicBlock> {
30     
31       private final Map<CGNode, IntraprocAnalysisState> map;
32       
33       InterprocAnalysisResultWrapper(final Map<CGNode, IntraprocAnalysisState> map) {
34         if (map == null) {
35           throw new IllegalArgumentException();
36         }
37     
38         this.map = map;
39       }
40       
41       @Override
42       public ExceptionPruningAnalysis<SSAInstruction, IExplodedBasicBlock> getResult(final CGNode n) {
43         if (!containsResult(n)) {
44           return null;
45         }
46         
47         return map.get(n);
48       }
49     
50       @Override
51       public boolean containsResult(final CGNode n) {
52         return map.containsKey(n) && map.get(n).canBeAnalyzed();
53       }
54     
55       @Override
56       public String toString() {
57         final StringBuilder sb = new StringBuilder();
58         
59         for (final Entry<CGNode, IntraprocAnalysisState> e : map.entrySet()) {
60           sb.append(e.getValue().hasExceptions() ? "THROWS " : "CLEAN  ");
61           sb.append(e.getKey().toString() + "\n");
62         }
63         
64         return sb.toString();
65       }
66     }
