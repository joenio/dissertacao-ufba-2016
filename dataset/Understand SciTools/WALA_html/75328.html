52     /**
53      * Base class that represents a set of files to analyze.
54      * 
55      * The analysis scope is partitioned by class loader. There are three pre-defined class loader scopes:
56      * <ul>
57      * <li>Primordial (for <code>rt.jar</code>, the core classes)
58      * <li>Extension (for extension libraries in $JRE/lib/ext)
59      * <li>Application (for the classes of the application)
60      * </ul>
61      * 
62      * Each class loader will load a set of classes described by a {@link Module}.
63      */
64     public class AnalysisScope {

93       /**
94        * Initialize the standard 3 class loaders for java analysis
95        */
96       protected void initCoreForJava() {
97         ClassLoaderReference primordial = new ClassLoaderReference(PRIMORDIAL, ClassLoaderReference.Java, null);
98         ClassLoaderReference extension = new ClassLoaderReference(EXTENSION, ClassLoaderReference.Java, primordial);
99         ClassLoaderReference application = new ClassLoaderReference(APPLICATION, ClassLoaderReference.Java, extension);
100     
101         loadersByName.put(PRIMORDIAL, primordial);
102         loadersByName.put(EXTENSION, extension);
103         loadersByName.put(APPLICATION, application);
104       }

109       protected void initSynthetic(ClassLoaderReference parent) {
110         ClassLoaderReference synthetic = new ClassLoaderReference(SYNTHETIC, ClassLoaderReference.Java, parent);
111         setLoaderImpl(synthetic, "com.ibm.wala.ipa.summaries.BypassSyntheticClassLoader");
112         loadersByName.put(SYNTHETIC, synthetic);

271         if (name == null) {

275           throw new IllegalArgumentException("empty atom is not a legal class loader name");

278          * if (Assertions.verifyAssertions) { if (name.getVal(0) > 'Z') { Assertions._assert(name.getVal(0) <= 'Z',

280          */
281         return loadersByName.get(name);
282       }

320       @Override
321       public String toString() {
322         StringBuffer result = new StringBuffer();
323         for (ClassLoaderReference loader : loadersByName.values()) {
324           result.append(loader.getName());
325           result.append("\n");
326           for (Module m : getModules(loader)) {
327             result.append(" ");
328             result.append(m);
329             result.append("\n");
330           }
331         }
332         result.append(getExclusionString());
333         result.append("\n");
334         return result.toString();
335       }

345        * Utility function. Useful when parsing input.
346        */
347       public MethodReference findMethod(Atom loader, String klass, Atom name, ImmutableByteArray desc) {
348         if (desc == null) {
349           throw new IllegalArgumentException("null desc");
350         }
351         ClassLoaderReference clr = getLoader(loader);
352         Descriptor ddesc = Descriptor.findOrCreate(languages.get(clr.getLanguage()), desc);
353         TypeReference type = TypeReference.findOrCreate(clr, TypeName.string2TypeName(klass));
354         return MethodReference.findOrCreate(type, name, ddesc);
355       }

370       /**
371        * @return the rt.jar (1.4) or core.jar (1.5) file, or null if not found.
372        */
373       private JarFile getRtJar() {
374         for (Iterator MS = getModules(getPrimordialLoader()).iterator(); MS.hasNext();) {
375           Module M = (Module) MS.next();
376           if (M instanceof JarFileModule) {
377             JarFile JF = ((JarFileModule) M).getJarFile();
378             if (JF.getName().endsWith(File.separator + "rt.jar")) {
379               return JF;
380             }
381             if (JF.getName().endsWith(File.separator + "core.jar")) {
382               return JF;
383             }
384             // hack for Mac
385             if (PlatformUtil.onMacOSX() && JF.getName().endsWith(File.separator + "classes.jar")) {
386               return JF;
387             }
388           }
389         }
390         return null;
391       }

393       public String getJavaLibraryVersion() throws IllegalStateException {
394         JarFile rtJar = getRtJar();
395         if (rtJar == null) {
396           throw new IllegalStateException("cannot find runtime libraries");
397         }
398         try {
399           Manifest man = rtJar.getManifest();
400           assert man != null : "runtime library has no manifest!";
401           String result = man.getMainAttributes().getValue("Specification-Version");
402           if (result == null) {
403             Attributes att = man.getMainAttributes();
404             System.err.println("main attributes:" + att);
405             Assertions.UNREACHABLE("Manifest for " + rtJar.getName() + " has no value for Specification-Version");
406           }
407           return result;
408         } catch (java.io.IOException e) {
409           Assertions.UNREACHABLE("error getting rt.jar manifest!");
410           return null;
411         }
412       }

434       /**
435        * Creates a "serializable" version of the analysis scope.
436        * 
437        * @return a "serializable" version of the analysis scope.
438        * @throws NotSerializableException
439        */
440       public ShallowAnalysisScope toShallowAnalysisScope() throws NotSerializableException {
441     
442         if (getArrayClassLoader().getNumberOfClasses() != 0) {
443           throw new NotSerializableException("Scope was already used for building array classes");
444         }
445         // Note: 'arrayClassLoader' object will be built from scratch in remote process
446     
447         // represent modules map as a set of strings (corresponding to analysis scope file lines.
448         List<String> moduleLines = new ArrayList<String>();
449         for (Map.Entry<ClassLoaderReference, List<Module>> e : moduleMap.entrySet()) {
450           ClassLoaderReference lrReference = e.getKey();
451           String moduleLdr = lrReference.getName().toString();
452           String moduleLang = lrReference.getLanguage().toString();
453           assert Language.JAVA.getName().equals(lrReference.getLanguage()) : "Java language only is currently supported";
454     
455           for (Module m : e.getValue()) {
456             String moduleType;
457             String modulePath;
458             if (m instanceof JarFileModule) {
459               moduleType = "jarFile";
460               modulePath = ((JarFileModule) m).getAbsolutePath();
461             } else if (m instanceof BinaryDirectoryTreeModule) {
462               moduleType = "binaryDir";
463               modulePath = ((BinaryDirectoryTreeModule) m).getPath();
464             } else if (m instanceof SourceDirectoryTreeModule) {
465               moduleType = "sourceDir";
466               modulePath = ((SourceDirectoryTreeModule) m).getPath();
467             } else if (m instanceof SourceFileModule) {
468               moduleType = "sourceFile";
469               modulePath = ((SourceFileModule) m).getAbsolutePath();
470             } else {
471               Assertions.UNREACHABLE("Module type isn't supported - " + m);
472               continue;
473             }
474             modulePath.replace("\\", "/");
475             String moduleDescrLine = String.format("%s,%s,%s,%s", moduleLdr, moduleLang, moduleType, modulePath);
476             moduleLines.add(moduleDescrLine);
477           }
478         }
479     
480         // represent loaderImplByRef map as set of strings
481         List<String> ldrImplLines = new ArrayList<String>();
482         for (Map.Entry<ClassLoaderReference, String> e : loaderImplByRef.entrySet()) {
483           ClassLoaderReference lrReference = e.getKey();
484           String ldrName = lrReference.getName().toString();
485           String ldrLang = lrReference.getLanguage().toString();
486           assert Language.JAVA.getName().equals(lrReference.getLanguage()) : "Java language only is currently supported";
487           String ldrImplName = e.getValue();
488           String ldrImplDescrLine = String.format("%s,%s,%s,%s", ldrName, ldrLang, "loaderImpl", ldrImplName);
489           ldrImplLines.add(ldrImplDescrLine);
490         }
491     
492         ShallowAnalysisScope shallowScope = new ShallowAnalysisScope(getExclusions(), moduleLines, ldrImplLines);
493         return shallowScope;
494       }
