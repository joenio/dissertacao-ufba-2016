11     package com.ibm.wala.dataflow.IFDS;
12     
13     import java.util.Iterator;
14     
15     import com.ibm.wala.util.graph.Graph;
16     import com.ibm.wala.util.graph.NumberedGraph;
17     
18     /**
19      * 
20      * A supergraph as defined by Reps, Horwitz, and Sagiv POPL95
21      * <p>
22      * In our implementation we don't require explicit entry and exit nodes. So, the first basic block in a method is implicitly the
23      * entry node, but might also be a call node too. Similarly for exit nodes. The solver is coded to deal with this appropriately.
24      * <p>
25      * Additionally, due to exceptional control flow, each method might have multiple exits or multiple entries.
26      * 
27      * T type of node in the supergraph P type of a procedure (like a box in an RSM)
28      */
29     public interface ISupergraph<T, P> extends NumberedGraph<T> {
30     
31       public static final byte CALL_EDGE = 0;
32     
33       public static final byte RETURN_EDGE = 1;
34     
35       public static final byte CALL_TO_RETURN_EDGE = 2;
36     
37       public static final byte OTHER = 3;
38     
39       /**
40        * @return the graph of procedures (e.g. a call graph) over which this supergraph is induced.
41        */
42       Graph<? extends P> getProcedureGraph();
43     
44       /**
45        * @param n a node in this supergraph
46        * @return true iff this node includes a call.
47        */
48       boolean isCall(T n);
49     
50       /**
51        * @param call a "call" node in the supergraph
52        * @return an Iterator of nodes that are targets of this call.
53        */
54       Iterator<? extends T> getCalledNodes(T call);
55     
56       /**
57        * @param call a "call" node in the supergraph
58        * @return an Iterator of nodes that are normal (non-call) successors of this call. This should only apply to backwards problems,
59        *         where we might have, say, a call and a goto flow into a return site.
60        */
61       Iterator<T> getNormalSuccessors(T call);
62     
63       /**
64        * @param call a "call" node in the supergraph
65        * @param callee a "called" "procedure" in the supergraph. if callee is null, answer return sites for which no callee was found.
66        * @return the corresponding return nodes. There may be many, because of exceptional control flow.
67        */
68       Iterator<? extends T> getReturnSites(T call, P callee);
69     
70       /**
71        * @param r a "return" node in the supergraph
72        * @param callee a "called" "procedure" in the supergraph. if callee is null, answer return sites for which no callee was found.
73        * @return the corresponding call nodes. There may be many.
74        */
75       Iterator<? extends T> getCallSites(T ret, P callee);
76     
77       /**
78        * @param n a node in the supergraph
79        * @return true iff this node is an exit node
80        */
81       boolean isExit(T n);
82     
83       /**
84        * @param n a node in the supergraph
85        * @return an object which represents the procedure which contains n
86        */
87       P getProcOf(T n);
88     
89       /**
90        * @return the blocks in the supergraph that represents entry nodes for procedure p
91        */
92       T[] getEntriesForProcedure(P procedure);
93     
94       /**
95        * @return the blocks in the supergraph that represents exit nodes for procedure p
96        */
97       T[] getExitsForProcedure(P procedure);
98     
99       /**
100        * @param procedure an object that represents a procedure
101        * @return the number of blocks from this procedure in this supergraph
102        */
103       int getNumberOfBlocks(P procedure);
104     
105       /**
106        * @param n a node in the supergraph
107        * @return the "logical" basic block number of n in its procedure
108        */
109       int getLocalBlockNumber(T n);
110     
111       /**
112        * @param procedure an object that represents a procedure
113        * @param i the "logical" basic block number of a node in the procedure
114        * @return the corresponding node in the supergraph
115        */
116       T getLocalBlock(P procedure, int i);
117     
118       /**
119        * @param n a node in this supergraph
120        * @return true iff this node is a return site.
121        */
122       boolean isReturn(T n);
123     
124       /**
125        * @return true iff this node is an entry node s_p for a procedure
126        */
127       boolean isEntry(T n);
128     
129       /**
130        * @param src node in the supergraph
131        * @param dest a successor of src in the supergraph
132        * @return one of CALL_EDGE, RETURN_EDGE, CALL_TO_RETURN_EDGE, or OTHER
133        */
134       byte classifyEdge(T src, T dest);
135     
136     }
