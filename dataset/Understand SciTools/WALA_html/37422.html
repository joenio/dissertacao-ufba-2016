11     package com.ibm.wala.cast.js.callgraph.fieldbased.flowgraph.vertices;
12     
13     
14     /**
15      * A parameter vertex represents a positional parameter of a function. It doesn't
16      * necessarily need to correspond to a named parameter.
17      * 
18      * <p>
19      * Numbering of positional parameters is 1-based, with parameter 0 being the
20      * <code>this</code> value.
21      * </p>
22      * 
23      * <p>
24      * A named parameter is an ordinary SSA variable, hence it is represented as
25      * a {@link VarVertex}. The flow graph builder sets up edges between parameter
26      * vertices and their corresponding variable vertices for named parameters.
27      * </p>
28      * 
29      * @author mschaefer
30      *
31      */
32     public class ParamVertex extends Vertex {
33     	private FuncVertex func;
34     	private int index;
35     
36     	ParamVertex(FuncVertex func, int index) {
37     		this.func = func;
38     		this.index = index;
39     	}
40     	
41     	public FuncVertex getFunc() {
42     	  return func;
43     	}
44     	
45     	public int getIndex() {
46     	  return index;
47     	}
48     
49     	@Override
50     	public <T> T accept(VertexVisitor<T> visitor) {
51     		return visitor.visitParamVertex(this);
52     	}
53     	
54     	@Override
55     	public String toString() {
56     	  return "Param(" + func + ", " + index + ")";
57     	}
58     }
