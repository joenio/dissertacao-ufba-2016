11     package com.ibm.wala.util.io;
12     
13     import java.io.ByteArrayOutputStream;
14     import java.io.IOException;
15     import java.io.InputStream;
16     
17     /**
18      * utilities for IO streams
19      */
20     public class Streams {
21     
22       /**
23        * @param in
24        * @return byte[] holding the contents of the stream
25        * @throws IllegalArgumentException  if in == null
26        */
27       public static byte[] inputStream2ByteArray(InputStream in) throws IllegalArgumentException, IOException {
28         if (in == null) {
29           throw new IllegalArgumentException("in == null");
30         }
31         ByteArrayOutputStream b = new ByteArrayOutputStream();
32         while (in.available() > 0) {
33           byte[] data = new byte[in.available()];
34           in.read(data);
35           b.write(data);
36         }
37         byte[] data = b.toByteArray();
38         return data;
39       }

41     }
