30     public class AndroidAnalysisScope {
31     
32     	private static final String BASIC_FILE = "primordial.txt";
33     
34     	public static AnalysisScope setUpAndroidAnalysisScope(URI classpath, String exclusions, ClassLoader loader, URI... androidLib) throws IOException {
35     		AnalysisScope scope;
36     		if (androidLib == null || androidLib.length == 0) {
37     			scope = AnalysisScopeReader.readJavaScope(BASIC_FILE, new File(exclusions), loader);
38     		} else {
39     			scope = AnalysisScope.createJavaAnalysisScope();
40     
41     			File exclusionsFile = new File(exclusions);
42     	        InputStream fs = exclusionsFile.exists()? new FileInputStream(exclusionsFile): FileProvider.class.getClassLoader().getResourceAsStream(exclusionsFile.getName());
43     	        scope.setExclusions(new FileOfClasses(fs));
44     	        
45     			scope.setLoaderImpl(ClassLoaderReference.Primordial,
46     					"com.ibm.wala.dalvik.classLoader.WDexClassLoaderImpl");
47     
48     			for(URI al : androidLib) {
49     				try {
50     					scope.addToScope(ClassLoaderReference.Primordial, DexFileModule.make(new File(al)));
51     				} catch (Exception e) {
52     					e.printStackTrace();
53     					scope.addToScope(ClassLoaderReference.Primordial, new JarFileModule(new JarFile(new File(al))));					
54     				}
55     			}
56     
57     		}
58     
59     		scope.setLoaderImpl(ClassLoaderReference.Application,
60     				"com.ibm.wala.dalvik.classLoader.WDexClassLoaderImpl");
61     
62     		scope.addToScope(ClassLoaderReference.Application, DexFileModule.make(new File(classpath)));
63     		
64     		return scope;
65     	}
66     	
67     	/**
68     	 * Handle .apk file.
69     	 * 
70     	 * @param classPath
71     	 * @param scope
72     	 * @param loader
73     	 */
74     	public static void addClassPathToScope(String classPath,
75     			AnalysisScope scope, ClassLoaderReference loader) {
76     		if (classPath == null) {
77     			throw new IllegalArgumentException("null classPath");
78     		}
79     		try {
80     			String[] paths = classPath.split(File.pathSeparator);
81     
82     			for (int i = 0; i < paths.length; i++) {
83     				if (paths[i].endsWith(".jar")
84     						|| paths[i].endsWith(".apk")
85     						|| paths[i].endsWith(".dex")) { // Handle android file.
86     					File f = new File(paths[i]);
87     					scope.addToScope(loader, DexFileModule.make(f));
88     				} else {
89     					File f = new File(paths[i]);
90     					if (f.isDirectory()) { // handle directory FIXME not working
91     											// for .dex and .apk files into that
92     											// directory
93     						scope.addToScope(loader, new BinaryDirectoryTreeModule(
94     								f));
95     					} else { // handle java class file.
96     						try {
97     							scope.addClassFileToScope(loader, f);
98     						} catch (InvalidClassFileException e) {
99     							throw new IllegalArgumentException(
100     									"Invalid class file");
101     						}
102     					}
103     				}
104     			}
105     
106     		} catch (IOException e) {
107     			Assertions.UNREACHABLE(e.toString());
108     		}
109     	}
110     }
