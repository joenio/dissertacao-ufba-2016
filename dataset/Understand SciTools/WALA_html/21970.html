11     package com.ibm.wala.classLoader;
12     
13     import java.util.HashMap;
14     
15     import com.ibm.wala.ipa.cha.IClassHierarchy;
16     import com.ibm.wala.types.TypeName;
17     import com.ibm.wala.types.TypeReference;
18     import com.ibm.wala.util.collections.HashMapFactory;
19     
20     /**
21      * Pseudo-classloader for all array classes; all other IClassLoader
22      * implementations should delegate to this one for array classes only.
23      */
24     public class ArrayClassLoader {
25     
26       private final static boolean DEBUG = false;
27     
28       /**
29        * map: TypeReference -> ArrayClass
30        */
31       final private HashMap<TypeReference, ArrayClass> arrayClasses = HashMapFactory.make();
32     
33     
34       /**
35        * @param className
36        *          name of the array class
37        * @param delegator
38        *          class loader to look up element type with
39        */
40       public IClass lookupClass(TypeName className, IClassLoader delegator, IClassHierarchy cha) throws IllegalArgumentException {
41         ArrayClass arrayClass;
42         if (DEBUG) {
43           assert className.toString().startsWith("[");
44         }
45         if (delegator == null) {
46           throw new IllegalArgumentException("delegator must not be null");
47         }
48     
49         TypeReference type = TypeReference.findOrCreate(delegator.getReference(), className);
50         TypeReference elementType = type.getArrayElementType();
51         if (elementType.isPrimitiveType()) {
52           TypeReference aRef = TypeReference.findOrCreateArrayOf(elementType);
53           arrayClass = arrayClasses.get(aRef);
54           IClassLoader primordial = getRootClassLoader(delegator);
55           if (arrayClass == null) {
56             arrayClasses.put(aRef, arrayClass=new ArrayClass(aRef,primordial,cha));
57           }
58         } else {
59           arrayClass = arrayClasses.get(type);
60           if (arrayClass == null) {
61     	// check that the element class is loadable. If not, return null.
62     	IClass elementCls = delegator.lookupClass(elementType.getName());
63             if (elementCls == null) {
64               return null;
65             }
66     	
67     	TypeReference realType = TypeReference.findOrCreateArrayOf(elementCls.getReference());
68     	arrayClass = arrayClasses.get(realType);
69     	
70     	if (arrayClass == null) {
71     	  arrayClass = new ArrayClass(realType, elementCls.getClassLoader(), cha);
72     	}
73           }
74           arrayClasses.put(type, arrayClass);
75         }
76         return arrayClass;
77       }
78     
79       private static IClassLoader getRootClassLoader(IClassLoader l) {
80         while (l.getParent() != null) {
81           l = l.getParent();
82         }
83         return l;
84       }
85       
86       public int getNumberOfClasses() {
87         return arrayClasses.size();
88       }
89     }
