11     package com.ibm.wala.util;
12     
13     /**
14      * An exception for when work is canceled in eclipse. This is identical to {@link CancelException}, but this one extends
15      * {@link RuntimeException}, so it need not be threaded through every API that uses it.
16      */
17     public class CancelRuntimeException extends RuntimeException {
18     
19       protected CancelRuntimeException(String msg) {
20         super(msg);
21       }
22     
23       public CancelRuntimeException(Exception cause) {
24         super(cause);
25       }
26     
27       public static CancelRuntimeException make(String msg) {
28         return new CancelRuntimeException(msg);
29       }
30     
31     }
