25     /**
26      * Floyd-Warshall algorithm to compute all-pairs shortest path in graph with no negative cycles.
27      * 
28      * TODO: this API should be cleaned up.
29      * 
30      * @param <T> node type in the graph
31      */
32     public class FloydWarshall<T> {
33     
34       public interface GetPath<T> {
35         List<T> getPath(T from, T to);
36       }
37     
38       public interface GetPaths<T> {
39         Set<List<T>> getPaths(T from, T to);
40       }
41     
42       protected final NumberedGraph<T> G;
43       
44       public FloydWarshall(NumberedGraph<T> g) {
45         G = g;
46       }
47     
48       protected int edgeCost(int from, int to) {
49         return 1;
50       }
51       
52       protected void pathCallback(int i, int j, int k) {
53         
54       }
55       
56       public int[][] allPairsShortestPaths() {
57         final int[][] result = new int[G.getNumberOfNodes()][G.getNumberOfNodes()];
58     
59         for(int i = 0; i < result.length; i++) {
60           for(int j = 0; j < result[i].length; j++) {
61             result[i][j] = Integer.MAX_VALUE;
62           }
63         }
64         
65         for(T from : G) {
66           final int fn = G.getNumber(from);
67           IntSet tos = G.getSuccNodeNumbers(from);
68           tos.foreach(new IntSetAction() {
69             @Override
70             public void act(int x) {
71               result[fn][x] = edgeCost(fn, x);
72             }
73           });
74         }
75         
76         for(T kn : G) {
77           int k = G.getNumber(kn);
78           for(T in : G) {
79             int i = G.getNumber(in);
80             for(T jn : G) {
81               int j = G.getNumber(jn);
82               long newLen = (long)result[i][k] + (long)result[k][j];
83               if (newLen <= result[i][j]) {
84                 pathCallback(i, j, k);
85               }
86               if (newLen < result[i][j]) {
87                 result[i][j] = (int)newLen;
88               }
89             }
90           }
91         }
92         
93         return result;
94       }

96       public static <T> int[][] shortestPathLengths(NumberedGraph<T> G) {
97         return new FloydWarshall<T>(G).allPairsShortestPaths();
98       }

100       public static <T> GetPath<T> allPairsShortestPath(final NumberedGraph<T> G) {
101          return new FloydWarshall<T>(G) {
102            int[][] next = new int[G.getNumberOfNodes()][G.getNumberOfNodes()];
103     
104            @Override
105            protected void pathCallback(int i, int j, int k) {
106              next[i][j] = k;
107            }
108            
109            private GetPath<T> doit() {
110              for(int i = 0; i < next.length; i++) {
111                for(int j = 0; j < next[i].length; j++) {
112                  next[i][j] = -1;
113                }
114              }
115              
116              final int[][] paths = allPairsShortestPaths();
117              return new GetPath<T>() {
118               @Override
119               public List<T> getPath(T from, T to) {
120                 int fn = G.getNumber(from);
121                 int tn = G.getNumber(to);
122                 if (paths[fn][tn] == Integer.MAX_VALUE) {
123                   throw new UnsupportedOperationException("no path from " + from + " to " + to);
124                 } else {
125                   int intermediate = next[fn][tn];
126                   if (intermediate == -1) {
127                     return Collections.emptyList();
128                   } else {
129                     T in = G.getNode(intermediate);
130                     List<T> result = new LinkedList<T>(getPath(from, in));
131                     result.add(in);
132                     result.addAll(getPath(in, to));
133                     return result;
134                   }
135                 }
136               } 
137              };
138            }
139          }.doit();
140       }

142       public static <T> GetPaths<T> allPairsShortestPaths(final NumberedGraph<T> G) {
143         return new FloydWarshall<T>(G) {
144           MutableIntSet[][] next = new MutableIntSet[G.getNumberOfNodes()][G.getNumberOfNodes()];
145     
146           @Override
147           protected void pathCallback(int i, int j, int k) {
148             if (next[i][j] == null) {
149               next[i][j] = IntSetUtil.make();
150             }
151             next[i][j].add(k);
152           }
153           
154           private GetPaths<T> doit() {        
155             final int[][] paths = allPairsShortestPaths();
156             return new GetPaths<T>() {
157              @Override
158             public Set<List<T>> getPaths(final T from, final T to) {
159                int fn = G.getNumber(from);
160                int tn = G.getNumber(to);
161                if (paths[fn][tn] == Integer.MAX_VALUE) {
162                  throw new UnsupportedOperationException("no path from " + from + " to " + to);
163                } else {
164                  MutableIntSet intermediate = next[fn][tn];
165                  if (intermediate == null) {
166                    List<T> none = Collections.emptyList();
167                    return Collections.singleton(none);
168                  } else {
169                    final Set<List<T>> result = new HashSet<List<T>>();
170                   
171                    intermediate.foreach(new IntSetAction() {
172                     @Override
173                     public void act(int x) {
174                       T in = G.getNode(x);
175                       for(List<T> pre : getPaths(from, in)) {
176                         for(List<T> post : getPaths(in, to)) {
177                           List<T> path = new LinkedList<T>(pre);
178                           path.add(in);
179                           path.addAll(post);
180                           result.add(path);
181                         }
182                       }
183                     }           
184                    });
185                    
186                    return result;
187                  }
188                }
189              } 
190             };
191           }
192         }.doit();
193      }
