12     package com.ibm.wala.classLoader;
13     
14     import java.io.IOException;
15     import java.io.Reader;
16     import java.util.Collection;
17     import java.util.Iterator;
18     import java.util.List;
19     import java.util.NoSuchElementException;
20     
21     import com.ibm.wala.ssa.SSAInstructionFactory;
22     import com.ibm.wala.types.ClassLoaderReference;
23     import com.ibm.wala.types.TypeName;
24     import com.ibm.wala.util.strings.Atom;
25     
26     /**
27      * Base class for an object that represents a single Java classloader for analysis purposes.
28      */
29     public interface IClassLoader {
30       /**
31        * Find and return the IClass defined by this class loader that corresponds to the given class name.
32        * 
33        * @param className name of the class
34        * @return the IClass defined by this class loader that corresponds to the given class name, or null if not found.
35        */
36       public abstract IClass lookupClass(TypeName className);
37     
38       /**
39        * Return the ClassLoaderReference for this class loader.
40        * 
41        * @return ClassLoaderReference
42        */
43       public abstract ClassLoaderReference getReference();
44     
45       /**
46        * @return an Iterator of all classes loaded by this loader
47        */
48       public abstract Iterator<IClass> iterateAllClasses();
49     
50       /**
51        * @return the number of classes in scope to be loaded by this loader
52        */
53       public abstract int getNumberOfClasses();
54     
55       /**
56        * @return the unique name that identifies this class loader.
57        */
58       Atom getName();
59     
60       /**
61        * @return the unique name that identifies the programming language from which this class loader loads code.
62        */
63       Language getLanguage();
64     
65       SSAInstructionFactory getInstructionFactory();
66       
67       public abstract int getNumberOfMethods();
68     
69       /**
70        * @param method The method for which information is desired
71        * @param offset an offset into the bytecode of the given method. 
72        * @return name of the source file corresponding to the given offset in the given method.  Note that 
73        * this api allows a single method to arise from multiple source files, which is deliberate as it
74        * can happen in some languages.
75        */
76       public abstract String getSourceFileName(IMethod method, int offset);
77       
78       /**
79        * @param method The method for which information is desired
80        * @param offset an offset into the bytecode of the given method. 
81        * @return input stream representing the source file for a given bytecode index of a given method,
82        * or null if not available
83        */
84       public abstract Reader getSource(IMethod method, int offset);
85       
86       /**
87        * @param klass the class for which information is desired.
88        * @return name of source file corresponding to the class, or null if not available
89        * @throws NoSuchElementException if this class was generated from more than one source file
90        * The assumption that a class is generated from a single source file is java
91        * specific, and will change in the future.  In place of this API, use the version that takes
92        * a method and an offset, since that is now the granularity at which source file information
93        * will be recorded.  
94        * SJF .. we should think about this deprecation.  postponing deprecation for now.  
95        */
96       public abstract String getSourceFileName(IClass klass) throws NoSuchElementException;
97     
98       /**
99        * @return input stream representing the source file for a class, or null if not available
100        * @throws NoSuchElementException if this class was generated from more than one source file
101        * The assumption that a class is generated from a single source file is java
102        * specific, and will change in the future.  In place of this API, use the version that takes
103        * a method and an offset, since that is now the granularity at which source file information
104        * will be recorded.
105        * SJF .. we should think about this deprecation.  postponing deprecation for now.  
106        */
107       public abstract Reader getSource(IClass klass) throws NoSuchElementException;
108     
109       /**
110        * @return the parent IClassLoader, if any, or null
111        */
112       public abstract IClassLoader getParent();
113     
114       /**
115        * Initialize internal data structures.
116        * 
117        * @throws IOException
118        * @throws IllegalArgumentException if modules is null
119        */
120       public void init(List<Module> modules) throws IOException;
121     
122       /**
123        * blow away references to any classes in the set
124        * 
125        * @param toRemove Collection<IClass>
126        */
127       public abstract void removeAll(Collection<IClass> toRemove);
128     }
