175       private void addHackedForInLoopSensitivity(JSCFABuilder builder) {
176         final ContextSelector orig = builder.getContextSelector();
177         builder.setContextSelector(new PropertyNameContextSelector(builder.getAnalysisCache(), orig));
178       }
