32     public class DynamicCallGraphTest extends DynamicCallGraphTestBase {
33     
34       private static String testJarLocation = getClasspathEntry("com.ibm.wala.core.testdata");
35         
36       private CallGraph staticCG(String mainClass, String exclusionsFile) throws IOException, ClassHierarchyException, IllegalArgumentException, CancelException {
37         AnalysisScope scope = CallGraphTestUtil.makeJ2SEAnalysisScope(TestConstants.WALA_TESTDATA, exclusionsFile != null? exclusionsFile: CallGraphTestUtil.REGRESSION_EXCLUSIONS);
38         ClassHierarchy cha = ClassHierarchy.make(scope);
39         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha, mainClass);
40         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
41         return CallGraphTestUtil.buildZeroOneCFA(options, new AnalysisCache(), cha, scope, false);
42       }
43     
44       @Test
45       public void testGraph() throws IOException, ClassNotFoundException, InvalidClassFileException, FailureException, SecurityException, NoSuchMethodException, IllegalArgumentException, IllegalAccessException, InvocationTargetException, ClassHierarchyException, CancelException, InterruptedException  {
46         instrument(testJarLocation);
47         run("dynamicCG.MainClass", null);
48         CallGraph staticCG = staticCG("LdynamicCG/MainClass", null);
49         checkEdges(staticCG);
50       }
51     
52       @Test
53       public void testCallbacks() throws IOException, ClassNotFoundException, InvalidClassFileException, FailureException, SecurityException, NoSuchMethodException, IllegalArgumentException, IllegalAccessException, InvocationTargetException, ClassHierarchyException, CancelException, InterruptedException  {
54         instrument(testJarLocation);
55         run("dynamicCG.CallbacksMainClass", null);
56         CallGraph staticCG = staticCG("LdynamicCG/CallbacksMainClass", null);
57         checkEdges(staticCG);
58       }
59     
60       @Test
61       public void testExclusions() throws IOException, ClassNotFoundException, InvalidClassFileException, FailureException, SecurityException, NoSuchMethodException, IllegalArgumentException, IllegalAccessException, InvocationTargetException, ClassHierarchyException, CancelException, InterruptedException  {
62         instrument(testJarLocation);
63         run("dynamicCG.MainClass", "ShrikeTestExclusions.txt");
64         CallGraph staticCG = staticCG("LdynamicCG/MainClass", "ShrikeTestExclusions.txt");
65         checkEdges(staticCG);
66       }
