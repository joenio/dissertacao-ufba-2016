11     package com.ibm.wala.util.warnings;
12     
13     import java.util.Collection;
14     import java.util.Iterator;
15     import java.util.TreeSet;
16     
17     import com.ibm.wala.util.collections.HashSetFactory;
18     
19     /**
20      * A global, static dictionary of warnings
21      */
22     public class Warnings {
23     
24       private final static Collection<Warning> warnings = HashSetFactory.make();
25     
26       public static synchronized boolean add(Warning w) {
27         return warnings.add(w);
28       }
29     
30       public static synchronized void clear() {
31         warnings.clear();
32       }
33     
34       public static synchronized String asString() {
35         TreeSet<Warning> T = new TreeSet<Warning>();
36         T.addAll(warnings);
37         Iterator<Warning> it = T.iterator();
38         StringBuffer result = new StringBuffer();
39         for (int i = 1; i <= T.size(); i++) {
40           result.append(i).append(". ");
41           result.append(it.next());
42           result.append("\n");
43         }
44         return result.toString();
45       }
46     
47       public static synchronized Iterator<Warning> iterator() {
48         return warnings.iterator();
49       }
50     }
