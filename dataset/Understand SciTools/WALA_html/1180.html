240          * int[] y= { 1, 2, 3, 4 } is represented in the IR as four array store

248             MethodReference mref = descriptorToMethodRef("Source#ArrayLiteral2#main#([Ljava/lang/String;)V", cg.getClassHierarchy());
249     
250             CGNode node = cg.getNodes(mref).iterator().next();
251     
252             final SSAInstruction[] instructions = node.getIR().getInstructions();
253             // test 1
254             {
255               SSAInstruction s1 = instructions[2];
256               if (s1 instanceof SSANewInstruction) {
257                 Assert.assertTrue("", ((SSANewInstruction) s1).getNewSite().getDeclaredType().isArrayType());
258               } else {
259                 Assert.assertTrue("Expected 3rd to be a new array instruction.", false);
260               }
261             }
262             // test 2
263             {
264               SSAInstruction s2 = instructions[3];
265               if (s2 instanceof SSANewInstruction) {
266                 Assert.assertTrue("", ((SSANewInstruction) s2).getNewSite().getDeclaredType().isArrayType());
267               } else {
268                 Assert.assertTrue("Expected 4th to be a new array instruction.", false);
269               }
270             }
271             // test 3: the last 4 instructions are of the form y[i] = i+1;
272             {
273               final SymbolTable symbolTable = node.getIR().getSymbolTable();
274               for (int i = 4; i <= 7; i++) {
275                 Assert.assertTrue("Expected only array stores.", instructions[i] instanceof SSAArrayStoreInstruction);
276     
277                 SSAArrayStoreInstruction as = (SSAArrayStoreInstruction) instructions[i];
278     
279                 Assert.assertEquals("Expected an array store to 'y'.", node.getIR().getLocalNames(i, as.getArrayRef())[0], "y");
280     
281                 final Integer valueOfArrayIndex = ((Integer) symbolTable.getConstantValue(as.getIndex()));
282                 final Integer valueAssigned = (Integer) symbolTable.getConstantValue(as.getValue());
283     
284                 Assert.assertEquals("Expected an array store to 'y' with value " + (valueOfArrayIndex + 1), valueAssigned.intValue(),
285                     valueOfArrayIndex + 1);
286     
287               }
288             }
289           }

588       private MethodReference getSliceRootReference(String className, String methodName, String methodDescriptor) {
589         TypeName clsName = TypeName.string2TypeName("L" + className.replace('.', '/'));
590         TypeReference clsRef = TypeReference.findOrCreate(JavaSourceAnalysisScope.SOURCE, clsName);
591     
592         Atom nameAtom = Atom.findOrCreateUnicodeAtom(methodName);
593         Descriptor descr = Descriptor.findOrCreateUTF8(Language.JAVA, methodDescriptor);
594     
595         return MethodReference.findOrCreate(clsRef, nameAtom, descr);
