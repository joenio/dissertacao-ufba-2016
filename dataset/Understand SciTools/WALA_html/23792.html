12     package com.ibm.wala.ipa.callgraph.propagation.cfa;
13     
14     import com.ibm.wala.ipa.callgraph.CGNode;
15     import com.ibm.wala.ipa.callgraph.Context;
16     import com.ibm.wala.ipa.callgraph.ContextItem;
17     import com.ibm.wala.ipa.callgraph.ContextKey;
18     
19     /**
20      * This is a {@link Context} which is defined by a pair consisting of <caller node, base context>.
21      * 
22      * The base context is typically some special case; e.g., a JavaTypeContext used for reflection.
23      */
24     public class CallerContextPair extends CallerContext {
25     
26       private final Context baseContext;
27     
28       /**
29        * @param caller the node which defines this context.
30        */
31       public CallerContextPair(CGNode caller, Context baseContext) {
32         super(caller);
33         this.baseContext = baseContext;
34         // avoid recursive contexts for now.
35         assert !(baseContext instanceof CallerContextPair);
36       }
37     
38       @Override
39       public ContextItem get(ContextKey name) {
40         if (name == null) {
41           throw new IllegalArgumentException("name is null");
42         }
43         if (name.equals(ContextKey.CALLER)) {
44           return super.get(name);
45         } else {
46           return baseContext.get(name);
47         }
48       }
49     
50       @Override
51       public boolean equals(Object obj) {
52         if (obj == null) {
53           return false;
54         }
55         if (getClass().equals(obj.getClass())) {
56           CallerContextPair other = (CallerContextPair) obj;
57           return getCaller().equals(other.getCaller()) && baseContext.equals(other.baseContext);
58         } else {
59           return false;
60         }
61       }
62     
63       @Override
64       public int hashCode() {
65         return 8377 * getCaller().hashCode() + baseContext.hashCode();
66       }
67     
68       @Override
69       public String toString() {
70         return "Caller: " + getCaller() + ",Base:" + baseContext;
71       }
72     
73     }
