25     /**
26      * An implementation of the class loader factory that produces ClassLoaderImpls
27      */
28     public class ClassLoaderFactoryImpl implements ClassLoaderFactory {
29     
30       /**
31        * Set of classes that class loaders should ignore; classloaders should
32        * pretend these classes don't exit.
33        */
34       final private SetOfClasses exclusions;
35     
36       /**
37        * A Mapping from ClassLoaderReference to IClassLoader
38        */
39       final private HashMap<ClassLoaderReference, IClassLoader> map = HashMapFactory.make(3);
40     
41       /**
42        * @param exclusions
43        *          A set of classes that class loaders should pretend don't exist.
44        */
45       public ClassLoaderFactoryImpl(SetOfClasses exclusions) {
46         this.exclusions = exclusions;
47       }
48     
49       /**
50        * Return a class loader corresponding to a given class loader identifier.
51        * Create one if necessary.
52        * 
53        * @param classLoaderReference
54        *          identifier for the desired class loader
55        */
56       @Override
57       public IClassLoader getLoader(ClassLoaderReference classLoaderReference, IClassHierarchy cha, AnalysisScope scope)
58           throws IOException {
59         if (classLoaderReference == null) {
60           throw new IllegalArgumentException("null classLoaderReference");
61         }
62         IClassLoader result = map.get(classLoaderReference);
63         if (result == null) {
64           ClassLoaderReference parentRef = classLoaderReference.getParent();
65           IClassLoader parent = null;
66           if (parentRef != null) {
67             parent = getLoader(parentRef, cha, scope);
68           }
69           IClassLoader cl = makeNewClassLoader(classLoaderReference, cha, parent, scope);
70           map.put(classLoaderReference, cl);
71           result = cl;
72         }
73         return result;
74       }
75     
76       /**
77        * Create a new class loader for a given key
78        * 
79        * @param classLoaderReference
80        *          the key
81        * @param parent
82        *          parent classloader to be used for delegation
83        * @return a new ClassLoaderImpl
84        * @throws IOException
85        *           if the desired loader cannot be instantiated, usually because the
86        *           specified module can't be found.
87        */
88       protected IClassLoader makeNewClassLoader(ClassLoaderReference classLoaderReference, IClassHierarchy cha, IClassLoader parent,
89           AnalysisScope scope) throws IOException {
90         String implClass = scope.getLoaderImpl(classLoaderReference);
91         IClassLoader cl;
92         if (implClass == null) {
93           cl = new ClassLoaderImpl(classLoaderReference, scope.getArrayClassLoader(), parent, exclusions, cha);
94         } else
95           try {
96             // this is fragile. why are we doing things this way again?
97             Class<?> impl = Class.forName(implClass);
98             Constructor<?> ctor = impl.getDeclaredConstructor(new Class[] { ClassLoaderReference.class, IClassLoader.class,
99                 SetOfClasses.class, IClassHierarchy.class });
100             cl = (IClassLoader) ctor.newInstance(new Object[] { classLoaderReference, parent, exclusions, cha });
101           } catch (Exception e) {
102             try {
103               Class<?> impl = Class.forName(implClass);
104               Constructor<?> ctor = impl.getDeclaredConstructor(new Class[] { ClassLoaderReference.class, ArrayClassLoader.class,
105                   IClassLoader.class, SetOfClasses.class, IClassHierarchy.class });
106               cl = (IClassLoader) ctor.newInstance(new Object[] { classLoaderReference, scope.getArrayClassLoader(), parent,
107                   exclusions, cha });
108             } catch (Exception e2) {
109               System.err.println("failed to load impl class " + implClass);
110               e2.printStackTrace(System.err);
111               Warnings.add(InvalidClassLoaderImplementation.create(implClass));
112               cl = new ClassLoaderImpl(classLoaderReference, scope.getArrayClassLoader(), parent, exclusions, cha);
113             }
114           }
115         cl.init(scope.getModules(classLoaderReference));
116         return cl;
117       }
118     
119       /**
120        * A waring when we fail to load an appropriate class loader implementation
121        */
122       private static class InvalidClassLoaderImplementation extends Warning {
123     
124         final String impl;
125     
126         InvalidClassLoaderImplementation(String impl) {
127           super(Warning.SEVERE);
128           this.impl = impl;
129         }
130     
131         @Override
132         public String getMsg() {
133           return getClass().toString() + " : " + impl;
134         }
135     
136         public static InvalidClassLoaderImplementation create(String impl) {
137           return new InvalidClassLoaderImplementation(impl);
138         }
139       }
140     
141       /**
142        * @return the set of classes that will be ignored.
143        */
144       public SetOfClasses getExclusions() {
145         return exclusions;
146       }
147     }
