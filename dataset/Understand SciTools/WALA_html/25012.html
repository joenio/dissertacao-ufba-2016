63       /*
64        * @see com.ibm.wala.dataflow.IFDS.ISupergraph#getCallSites(java.lang.Object)
65        */
66       @Override
67       public Iterator<? extends Statement> getCallSites(Statement r, PDG callee) {
68         switch (r.getKind()) {
69         case EXC_RET_CALLER: {
70           ExceptionalReturnCaller n = (ExceptionalReturnCaller) r;
71           SSAAbstractInvokeInstruction call = n.getInstruction();
72           PDG pdg = getProcOf(r);
73           return pdg.getCallStatements(call).iterator();
74         }
75         case NORMAL_RET_CALLER: {
76           NormalReturnCaller n = (NormalReturnCaller) r;
77           SSAAbstractInvokeInstruction call = n.getInstruction();
78           PDG pdg = getProcOf(r);
79           return pdg.getCallStatements(call).iterator();
80         }
81         case HEAP_RET_CALLER: {
82           HeapStatement.HeapReturnCaller n = (HeapStatement.HeapReturnCaller) r;
83           SSAAbstractInvokeInstruction call = n.getCall();
84           PDG pdg = getProcOf(r);
85           return pdg.getCallStatements(call).iterator();
86         }
87         default:
88           Assertions.UNREACHABLE(r.getKind().toString());
89           return null;
90         }
91       }

93       /*
94        * @see com.ibm.wala.dataflow.IFDS.ISupergraph#getCalledNodes(java.lang.Object)
95        */
96       @Override
97       public Iterator<? extends Statement> getCalledNodes(Statement call) {
98         switch (call.getKind()) {
99         case NORMAL:
100           Predicate f = new Predicate() {
101             @Override public boolean test(Object o) {
102               Statement s = (Statement) o;
103               return isEntry(s);
104             }
105           };
106           return new FilterIterator<Statement>(getSuccNodes(call), f);
107         case PARAM_CALLER:
108         case HEAP_PARAM_CALLER:
109           return getSuccNodes(call);
110         default:
111           Assertions.UNREACHABLE(call.getKind().toString());
112           return null;
113         }
114       }

117        * @see com.ibm.wala.dataflow.IFDS.ISupergraph#getEntriesForProcedure(java.lang.Object)
118        */
119       @Override
120       public Statement[] getEntriesForProcedure(PDG procedure) {
121         Statement[] normal = procedure.getParamCalleeStatements();
122         Statement[] result = new Statement[normal.length + 1];
123         result[0] = new MethodEntryStatement(procedure.getCallGraphNode());
124         System.arraycopy(normal, 0, result, 1, normal.length);
125         return result;
126       }

194        */
195       @Override
196       public Iterator<? extends Statement> getReturnSites(Statement call, PDG callee) {
197         switch (call.getKind()) {
198         case PARAM_CALLER: {
199           ParamCaller n = (ParamCaller) call;
200           SSAAbstractInvokeInstruction st = n.getInstruction();
201           PDG pdg = getProcOf(call);
202           return pdg.getCallerReturnStatements(st).iterator();
203         }
204         case HEAP_PARAM_CALLER: {
205           HeapStatement.HeapParamCaller n = (HeapStatement.HeapParamCaller) call;
206           SSAAbstractInvokeInstruction st = n.getCall();
207           PDG pdg = getProcOf(call);
208           return pdg.getCallerReturnStatements(st).iterator();
209         }
210         case NORMAL: {
211           NormalStatement n = (NormalStatement) call;
212           SSAAbstractInvokeInstruction st = (SSAAbstractInvokeInstruction) n.getInstruction();
213           PDG pdg = getProcOf(call);
214           return pdg.getCallerReturnStatements(st).iterator();
215         }
216         default:
217           Assertions.UNREACHABLE(call.getKind().toString());
218           return null;
219         }
220       }

222       /*
223        * @see com.ibm.wala.dataflow.IFDS.ISupergraph#isCall(java.lang.Object)
224        */
225       @Override
226       public boolean isCall(Statement n) {
227         switch (n.getKind()) {
228         case EXC_RET_CALLEE:
229         case EXC_RET_CALLER:
230         case HEAP_PARAM_CALLEE:
231         case NORMAL_RET_CALLEE:
232         case NORMAL_RET_CALLER:
233         case PARAM_CALLEE:
234         case PHI:
235         case HEAP_RET_CALLEE:
236         case HEAP_RET_CALLER:
237         case METHOD_ENTRY:
238         case METHOD_EXIT:
239         case CATCH:
240         case PI:
241           return false;
242         case HEAP_PARAM_CALLER:
243         case PARAM_CALLER:
244           return true;
245         case NORMAL:
246           if (sdg.getCOptions().equals(ControlDependenceOptions.NONE)) {
247             return false;
248           } else {
249             NormalStatement s = (NormalStatement) n;
250             return s.getInstruction() instanceof SSAAbstractInvokeInstruction;
251           }
252         default:
253           Assertions.UNREACHABLE(n.getKind() + " " + n.toString());
254           return false;
255         }
256       }
