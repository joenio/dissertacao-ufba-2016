40     public class JsViewerDriver extends JSCallGraphBuilderUtil {
41     	public static void main(String args[]) throws ClassHierarchyException, IllegalArgumentException, IOException, CancelException, Error, WalaException {
42     
43     		if (args.length != 1){
44     			System.out.println("Usage: <URL of html page to analyze>");
45     			System.exit(1);
46     		}
47     		boolean domless = false;
48     		
49     		URL url = new URL(args[0]); 
50     		
51     		// computing CG + PA
52     		JSCallGraphBuilderUtil.setTranslatorFactory(new CAstRhinoTranslatorFactory());
53     		JavaScriptLoader.addBootstrapFile(WebUtil.preamble);
54     
55     		SourceModule[] sources = getSources(domless, url);
56     		
57     		JSCFABuilder builder = makeCGBuilder(new WebPageLoaderFactory(translatorFactory), sources, CGBuilderType.ZERO_ONE_CFA, AstIRFactory.makeDefaultFactory());
58     		builder.setBaseURL(url);
59     
60     		CallGraph cg = builder.makeCallGraph(builder.getOptions());
61     		PointerAnalysis<InstanceKey> pa = builder.getPointerAnalysis();
62     
63     		new JsViewer(cg, pa);
64     	}
65     
66     	private static SourceModule[] getSources(boolean domless, URL url)
67     			throws IOException, Error {
68     		JSSourceExtractor sourceExtractor;
69     		if (domless ){
70     			sourceExtractor = new DomLessSourceExtractor(); 
71     		} else {
72     			sourceExtractor = new DefaultSourceExtractor();
73     		}
74     
75     		Set<MappedSourceModule> sourcesMap = sourceExtractor.extractSources(url, new JerichoHtmlParser(), new IdentityUrlResolver());
76     		SourceModule[] sources = new SourceFileModule[sourcesMap.size()];
77     		int i = 0;
78     		for (SourceModule m : sourcesMap){
79     			sources[i++] = m;
80     		}
81     		return sources;
82     	}
