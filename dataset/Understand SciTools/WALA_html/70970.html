11     package com.ibm.wala.cast.js.loader;
12     
13     import com.ibm.wala.cast.js.types.JavaScriptTypes;
14     import com.ibm.wala.cast.types.AstMethodReference;
15     import com.ibm.wala.classLoader.CallSiteReference;
16     import com.ibm.wala.shrikeBT.IInvokeInstruction;
17     import com.ibm.wala.types.MethodReference;
18     
19     public class JSCallSiteReference extends CallSiteReference {
20     
21       // this must be distinct from java invoke codes.
22       // see com.ibm.shrikeBT.BytecodeConstants
23       public static enum Dispatch implements IInvokeInstruction.IDispatch {
24         JS_CALL;
25     
26         @Override
27         public boolean hasImplicitThis() {
28           return false;
29         }
30       }
31     
32       public JSCallSiteReference(MethodReference ref, int pc) {
33         super(pc, ref);
34       }
35     
36       public JSCallSiteReference(int pc) {
37         this(AstMethodReference.fnReference(JavaScriptTypes.CodeBody), pc);
38       }
39     
40       @Override
41       public IInvokeInstruction.IDispatch getInvocationCode() {
42         return Dispatch.JS_CALL;
43       }
44     
45       @Override
46       public String toString() {
47         return "JSCall@" + getProgramCounter();
48       }
49     
50       public CallSiteReference cloneReference(int pc) {
51         return new JSCallSiteReference(pc);
52       }
53     
54       @Override
55       public boolean isDispatch() {
56         return true;
57       }
58     
59       @Override
60       public boolean isStatic() {
61         return false;
62       }
63     
64       @Override
65       public boolean isFixed() {
66         return false;
67       }
68     
69       
70     }
