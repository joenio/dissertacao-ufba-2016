12     
13     import java.util.Iterator;
14     
15     import com.ibm.wala.util.collections.SparseVector;
16     import com.ibm.wala.util.intset.BasicNaturalRelation;
17     import com.ibm.wala.util.intset.IBinaryNaturalRelation;
18     import com.ibm.wala.util.intset.IntPair;
19     import com.ibm.wala.util.intset.IntSet;
20     import com.ibm.wala.util.intset.MutableSparseIntSet;
21     import com.ibm.wala.util.intset.SparseLongIntVector;
22     import com.ibm.wala.util.math.LongUtil;
23     
24     /**
25      * A set of summary edges for a particular procedure.
26      */
27     public class LocalSummaryEdges {
28     
29       /**
30        * A map from integer n -> (IBinaryNonNegativeIntRelation)
31        * 
32        * Let s_p be an entry to this procedure, and x be an exit. n is a integer which uniquely identifies an (s_p,x) relation. For any
33        * such n, summaries[n] gives a relation R=(d1,d2) s.t. (<s_p, d1> -> <x,d2>) is a summary edge.
34        * 
35        * Note that this representation is a little different from the representation described in the PoPL 95 paper. We cache summary
36        * edges at the CALLEE, not at the CALLER!!! This allows us to avoid eagerly installing summary edges at all call sites to a
37        * procedure, which may be a win.
38        * 
39        * we don't technically need this class, since this information is redundantly stored in LocalPathEdges. However, we're keeping it
40        * cached for now for more efficient access when looking up summary edges.
41        * 
42        * TODO: more representation optimization.
43        */
44       private final SparseVector<IBinaryNaturalRelation> summaries = new SparseVector<IBinaryNaturalRelation>(1, 1.1f);
45     
46       /**
47        * Let (s_p,x) be an entry-exit pair, and let l := the long whose high word is s_p and low word is x.
48        * 
49        * Then entryExitMap(l) is an int which uniquely identifies (s_p,x)
50        * 
51        * we populate this map on demand!
52        */
53       private final static int UNASSIGNED = -1;
54     
55       private final SparseLongIntVector entryExitMap = new SparseLongIntVector(UNASSIGNED);
56     
57       private int nextEntryExitIndex = 0;
58     
59       /**
60        * 
61        */
62       public LocalSummaryEdges() {
63       }
64     
65       /**
66        * Record a summary edge for the flow d1 -> d2 from an entry s_p to an exit x.
67        * 
68        * @param s_p local block number an entry
69        * @param x local block number of an exit block
70        * @param d1 source dataflow fact
71        * @param d2 target dataflow fact
72        */
73       public void insertSummaryEdge(int s_p, int x, int d1, int d2) {
74         int n = getIndexForEntryExitPair(s_p, x);
75         IBinaryNaturalRelation R = summaries.get(n);
76         if (R == null) {
77           // we expect R to usually be sparse
78           R = new BasicNaturalRelation(new byte[] { BasicNaturalRelation.SIMPLE_SPACE_STINGY }, BasicNaturalRelation.SIMPLE);
79           summaries.set(n, R);
80         }
81         R.add(d1, d2);
82     //    if (TabulationSolver.DEBUG_LEVEL > 1) {
83     //      // System.err.println("recording summary edge, now n=" + n + " summarized by " + R);
84     //    }
85       }
86     
87       /**
88        * Does a particular summary edge exist?
89        * 
90        * @param s_p local block number an entry
91        * @param x local block number of an exit block
92        * @param d1 source dataflow fact
93        * @param d2 target dataflow fact
94        */
95       public boolean contains(int s_p, int x, int d1, int d2) {
96         int n = getIndexForEntryExitPair(s_p, x);
97         IBinaryNaturalRelation R = summaries.get(n);
98         if (R == null) {
99           return false;
100         } else {
101           return R.contains(d1, d2);
102         }
103       }
104     
105       /**
106        * @param s_p local block number an entry
107        * @param x local block number of an exit block
108        * @param d1 source dataflow fact
109        * @return set of d2 s.t. d1->d2 recorded as a summary edge for (s_p,x), or null if none
110        */
111       public IntSet getSummaryEdges(int s_p, int x, int d1) {
112         int n = getIndexForEntryExitPair(s_p, x);
113         IBinaryNaturalRelation R = summaries.get(n);
114         if (R == null) {
115           return null;
116         } else {
117           return R.getRelated(d1);
118         }
119       }
120     
121       /**
122        * Note: This is inefficient. Use with care.
123        * 
124        * @param s_p local block number an entry
125        * @param x local block number of an exit block
126        * @param d2 target dataflow fact
127        * @return set of d1 s.t. d1->d2 recorded as a summary edge for (s_p,x), or null if none
128        */
129       public IntSet getInvertedSummaryEdgesForTarget(int s_p, int x, int d2) {
130         int n = getIndexForEntryExitPair(s_p, x);
131         IBinaryNaturalRelation R = summaries.get(n);
132         if (R == null) {
133           return null;
134         } else {
135           MutableSparseIntSet result = MutableSparseIntSet.makeEmpty();
136           for (Iterator it = R.iterator(); it.hasNext();) {
137             IntPair p = (IntPair) it.next();
138             if (p.getY() == d2) {
139               result.add(p.getX());
140             }
141           }
142           return result;
143         }
144       }
145     
146       /**
147        * @return unique id n that represents the pair (s_p,x)
148        */
149       private int getIndexForEntryExitPair(int c, int r) {
150         long id = LongUtil.pack(c, r);
151         int result = entryExitMap.get(id);
152         if (result == UNASSIGNED) {
153           result = nextEntryExitIndex++;
154           entryExitMap.set(id, result);
155         }
156         return result;
157       }
158     
159     }
