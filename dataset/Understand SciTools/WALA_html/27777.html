18     package com.ibm.wala.sourcepos;
19     
20     /**
21      * Represents a source file position. Source file positions are integers in the
22      * format: {@code line-number << LINESHIFT + column-number}
23      * 
24      * @author Siegfried Weber
25      * @author Juergen Graf <juergen.graf@gmail.com>
26      */
27     public final class Position {
28     
29       /** Represents the undefined position. */
30       private static final int NOPOS = 0;
31       /** bits to shift to get the line number */
32       private static final int LINE_SHIFT = 10;
33       /** the bit mask of the column number */
34       private static final int COLUMN_MASK = 1023;
35     
36       /** Stores the position as unsigned integer. */
37       private int position;
38     
39       /**
40        * Creates the undefined position.
41        */
42       Position() {
43         position = NOPOS;
44       }
45     
46       /**
47        * Creates a new instance of Position.
48        * 
49        * @param position
50        *          the position as unsigned integer
51        * @throws InvalidPositionException
52        *           if the position is not undefined and the line or the column
53        *           number is 0
54        */
55       Position(int position) throws InvalidPositionException {
56         this.position = position;
57         if (!isUndefined() && getLine() == 0)
58           throw new InvalidPositionException(InvalidPositionException.Cause.LINE_NUMBER_ZERO);
59         if (!isUndefined() && getColumn() == 0)
60           throw new InvalidPositionException(InvalidPositionException.Cause.COLUMN_NUMBER_ZERO);
61       }
62     
63       /**
64        * Creates a new instance of Position.
65        * 
66        * @param line
67        *          the line number
68        * @param column
69        *          the column number
70        * @throws InvalidPositionException
71        *           if the line or the column number is out of range or if the
72        *           position is not undefined and the line or the column number is 0.
73        *           The maximum line number is 4194303. The maximum column number is
74        *           1023.
75        */
76       Position(int line, int column) throws InvalidPositionException {
77         if (line < 0 || line >= 4194304) // 4194304 = 2^32 >>> LINE_SHIFT
78           throw new InvalidPositionException(InvalidPositionException.Cause.LINE_NUMBER_OUT_OF_RANGE);
79         if (column < 0 || column > COLUMN_MASK)
80           throw new InvalidPositionException(InvalidPositionException.Cause.COLUMN_NUMBER_OUT_OF_RANGE);
81         if (line == 0 && column != 0)
82           throw new InvalidPositionException(InvalidPositionException.Cause.LINE_NUMBER_ZERO);
83         if (line != 0 && column == 0)
84           throw new InvalidPositionException(InvalidPositionException.Cause.COLUMN_NUMBER_ZERO);
85         position = (line << LINE_SHIFT) + column;
86       }
87     
88       /**
89        * Returns the line number.
90        * 
91        * @return the line number
92        */
93       public final int getLine() {
94         return position >>> LINE_SHIFT;
95       }
96     
97       /**
98        * Returns the column number.
99        * 
100        * @return the column number
101        */
102       public final int getColumn() {
103         return position & COLUMN_MASK;
104       }
105     
106       /**
107        * Tests whether this position is undefined.
108        * 
109        * @return true if this position is undefined
110        */
111       public final boolean isUndefined() {
112         return position == NOPOS;
113       }
114     
115       /**
116        * Tests whether this position is before the given position. If one of the
117        * positions is undefined or {@code p} is null then false is returned.
118        * 
119        * @param p
120        *          the position to test with
121        * @return true if this position is greater
122        */
123       boolean isBefore(Position p) {
124         return (p != null) && !isUndefined() && !p.isUndefined() && toLong() < p.toLong();
125       }
126     
127       /**
128        * Converts this position to a signed long variable.
129        * 
130        * @return this position as signed long
131        */
132       private long toLong() {
133         return position & 0xFFFFFFFFL;
134       }
135     
136       /**
137        * Returns this position as an unsigned integer.
138        * 
139        * @return this position as unsigned integer
140        */
141       int toUnsignedInt() {
142         return position;
143       }
144     
145       @Override
146       public String toString() {
147         return getLine() + ":" + getColumn();
148       }
149     }
