12     package com.ibm.wala.dataflow.graph;
13     
14     import com.ibm.wala.fixpoint.BitVectorVariable;
15     import com.ibm.wala.fixpoint.FixedPointConstants;
16     import com.ibm.wala.util.intset.IntSet;
17     import com.ibm.wala.util.intset.IntSetAction;
18     
19     /**
20      * Operator U(n) = U(n) n U(j)
21      */
22     public final class BitVectorIntersection extends AbstractMeetOperator<BitVectorVariable> implements FixedPointConstants {
23     
24       private static final BitVectorIntersection INSTANCE = new BitVectorIntersection();
25     
26       public static BitVectorIntersection instance() {
27         return INSTANCE;
28       }
29       
30       private BitVectorIntersection() {
31       }
32     
33       @Override
34       public byte evaluate(final BitVectorVariable lhs, final BitVectorVariable[] rhs) {
35         // as null is the initial value, we treat null as the neutral element to intersection
36         // which is a set of all possible elements.
37         IntSet intersect = lhs.getValue();
38         if (intersect == null) {
39           for (BitVectorVariable r : rhs) {
40             intersect = r.getValue();
41             if (intersect != null) { break; }
42           }
43           
44           if (intersect == null) {
45             // still null - so all rhs is null -> no change
46             return NOT_CHANGED;
47           }
48         } else if (intersect.isEmpty()) {
49           return NOT_CHANGED_AND_FIXED;
50         }
51         
52         for (final BitVectorVariable bv : rhs) {
53           final IntSet vlhs = bv.getValue();
54           if (vlhs != null) {
55             intersect = intersect.intersection(vlhs);
56           }
57         }
58     
59         if (lhs.getValue() != null && intersect.sameValue(lhs.getValue())) {
60           return NOT_CHANGED;
61         } else {
62           final BitVectorVariable bvv = new BitVectorVariable();
63           intersect.foreach(new IntSetAction() {
64             @Override
65             public void act(final int x) {
66               bvv.set(x);
67             }
68           });
69           lhs.copyState(bvv);
70     
71           return CHANGED;
72         }
73       }
74     
75       @Override
76       public int hashCode() {
77         return 9903;
78       }
79     
80       @Override
81       public boolean equals(final Object o) {
82         return o instanceof BitVectorIntersection;
83       }
84     
85       @Override
86       public String toString() {
87         return "INTERSECTION";
88       }
89     
90     }
