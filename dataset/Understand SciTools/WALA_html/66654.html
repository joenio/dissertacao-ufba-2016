11     package com.ibm.wala.util.collections;
12     
13     import java.io.Serializable;
14     import java.util.Iterator;
15     import java.util.Vector;
16     
17     import com.ibm.wala.util.debug.Assertions;
18     import com.ibm.wala.util.math.Logs;
19     
20     /**
21      * An {@link IVector} implementation which delegates to pages of int vectors.
22      */
23     public class TwoLevelVector<T> implements IVector<T>, Serializable {
24     
25       private static final int PAGE_SIZE = 4096;
26     
27       private static final int LOG_PAGE_SIZE = Logs.log2(PAGE_SIZE);
28     
29       /**
30        * Array of IVector: data.get(i) holds data[i*PAGE_SIZE] ...
31        * data[(i+1)*PAGESIZE - 1]
32        */
33       final private Vector<SparseVector<T>> data = new Vector<SparseVector<T>>(0);
34     
35       private int maxPage = -1;
36     
37       /*
38        * @see com.ibm.wala.util.intset.IntVector#get(int)
39        */
40       @Override
41       public T get(int x) {
42         if (x < 0) {
43           throw new IllegalArgumentException("invalid x: " + x);
44         }
45         int page = getPageNumber(x);
46         if (page >= data.size()) {
47           return null;
48         }
49         IVector<T> v = data.get(page);
50         if (v == null) {
51           return null;
52         }
53         int localX = x - getFirstIndexOnPage(page);
54         return v.get(localX);
55       }
56     
57       private int getFirstIndexOnPage(int page) {
58         return page << LOG_PAGE_SIZE;
59       }
60     
61       private int getPageNumber(int x) {
62         return x >> LOG_PAGE_SIZE;
63       }
64     
65       /*
66        * TODO: this can be optimized
67        * 
68        * @see com.ibm.wala.util.intset.IntVector#set(int, int)
69        */
70       @Override
71       public void set(int x, T value) {
72         if (x < 0) {
73           throw new IllegalArgumentException("illegal x: " + x);
74         }
75         int page = getPageNumber(x);
76         IVector<T> v = findOrCreatePage(page);
77         int localX = toLocalIndex(x, page);
78         v.set(localX, value);
79       }
80     
81       private int toLocalIndex(int x, int page) {
82         return x - getFirstIndexOnPage(page);
83       }
84     
85       private IVector<T> findOrCreatePage(int page) {
86         if (page >= data.size()) {
87           SparseVector<T> v = new SparseVector<T>();
88           data.setSize(page + 1);
89           data.add(page, v);
90           maxPage = Math.max(page, maxPage);
91           return v;
92         } else {
93           SparseVector<T> v = data.get(page);
94           if (v == null) {
95             v = new SparseVector<T>();
96             data.set(page, v);
97             maxPage = Math.max(page, maxPage);
98           }
99           return v;
100         }
101       }
102     
103       /*
104        * @see com.ibm.wala.util.debug.VerboseAction#performVerboseAction()
105        */
106       @Override
107       public void performVerboseAction() {
108         // do nothing;
109       }
110     
111       /*
112        * @see com.ibm.wala.util.intset.IVector#iterator()
113        */
114       @Override
115       public Iterator<T> iterator() {
116         return new Iterator<T>() {
117           final Iterator<SparseVector<T>> outer = data.iterator();
118     
119           Iterator<T> inner;
120           {
121             while (outer.hasNext()) {
122               IVector<T> v = outer.next();
123               if (v != null) {
124                 Iterator<T> it = v.iterator();
125                 if (it.hasNext()) {
126                   inner = it;
127                   break;
128                 }
129               }
130             }
131           }
132     
133           @Override
134           public boolean hasNext() {
135             return inner != null;
136           }
137     
138           @Override
139           public T next() {
140             T result = inner.next();
141             if (!inner.hasNext()) {
142               inner = null;
143               while (outer.hasNext()) {
144                 IVector<T> v = outer.next();
145                 if (v != null) {
146                   Iterator<T> it = v.iterator();
147                   if (it.hasNext()) {
148                     inner = it;
149                     break;
150                   }
151                 }
152               }
153             }
154             return result;
155           }
156     
157           @Override
158           public void remove() {
159             // TODO Auto-generated method stub
160             Assertions.UNREACHABLE();
161           }
162         };
163       }
164     
165       @Override
166       public int getMaxIndex() {
167         if (maxPage == -1) {
168           return -1;
169         } else {
170           IVector<T> v = data.get(maxPage);
171           int localMax = v.getMaxIndex();
172           return maxPage * PAGE_SIZE + localMax;
173         }
174     
175       }
176     
177     }
