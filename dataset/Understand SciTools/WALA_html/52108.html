88         @Override
89         public void handleEndTag(ITag tag) {
90           if (tag.getName().equalsIgnoreCase("script")) {
91             assert currentScriptTag != null;
92             currentScriptTag = null;
93           } else if (currentScriptTag != null && tag.getName().equals("!--")) {
94             assert currentCommentTag != null;
95             currentCommentTag = null;
96           }
97         }

99         @Override
100         public void handleText(Position p, String text) {
101           if (currentScriptTag != null && currentCommentTag == null) {
102             if (text.startsWith("<![CDATA[")) {
103              assert text.endsWith("]]>");
104              text = text.substring(9, text.length()-11);
105             }
106             
107             URL url = entrypointUrl;
108             try {
109               url = new URL(entrypointUrl, "#" + scriptNodeCounter);
110             } catch (MalformedURLException e) {
111               // TODO Auto-generated catch block
112               e.printStackTrace();
113             }
114     
115             scriptRegion.println(text, currentScriptTag.getContentPosition(), url, true);
116           }
117         }
118     
119         @Override
120         public void handleStartTag(ITag tag) {
121           if (tag.getName().equalsIgnoreCase("script")) {
122             handleScript(tag);
123             assert currentScriptTag == null;
124             currentScriptTag = tag;
125             scriptNodeCounter++;
126           } else if (currentScriptTag != null && tag.getName().equals("!--")){
127             currentCommentTag = tag;
128           }
129           handleDOM(tag);
130         }

132         private boolean isUsableIdentifier(String x) {
133           return x != null &&
134               LEGAL_JS_IDENTIFIER_REGEXP.matcher(x).matches() &&
135               !LEGAL_JS_KEYWORD_REGEXP.matcher(x).matches();
136         }

138         /**
139          * Model the HTML DOM
140          * 
141          * @param tag
142          *            - the HTML tag to module
143          */
144         protected void handleDOM(ITag tag) {
145           // Get the name of the modeling function either from the id attribute or a
146           // running counter
147           Pair<String,Position> idAttribute = tag.getAttributeByName("id");
148           String funcName;
149           if (idAttribute != null &&  isUsableIdentifier(idAttribute.fst)) {
150             funcName = idAttribute.fst;
151           } else {
152             funcName = "node" + (nodeCounter++);
153           }
154           handleDOM(tag, funcName);
155         }
156     
157         protected void handleDOM(ITag tag, String funcName) {
158           Map<String, Pair<String,Position>> attributeSet = tag.getAllAttributes();
159           for (Entry<String, Pair<String, Position>> a : attributeSet.entrySet()) {
160             handleAttribute(a, funcName, tag);
161           }
162         }
163     
164         private void handleAttribute(Entry<String, Pair<String,Position>> a, String funcName, ITag tag) {
165           URL url = entrypointUrl;
166           try {
167             url = new URL(entrypointUrl, "#" + tag.getElementPosition().getFirstOffset());
168           } catch (MalformedURLException e) {
169             // TODO Auto-generated catch block
170             if (DEBUG) {
171               e.printStackTrace();
172             }
173           }
174           Position pos = a.getValue().snd;
175           String attName = a.getKey();
176           String attValue = a.getValue().fst;
177           if (attName.toLowerCase().startsWith("on") || (attValue != null && attValue.toLowerCase().startsWith("javascript:"))) {
178             String fName = tag.getName().toLowerCase() + "_" + attName + "_" + funcName;
179             String signatureLine = "function " + fName + "(event) {";
180             // Defines the function  
181             domRegion.println(signatureLine + "\n" + extructJS(attValue) + "\n}", pos, url, true);
182             // Run it
183             entrypointRegion.println("\t" + fName + "(null);", pos, url, true);
184           }
185         }

187         protected static Pair<String,Character> quotify(String value) { 
188           char quote;
189           if (value.indexOf('"') < 0) {
190             quote= '"';
191           } else if (value.indexOf("'") < 0) {
192             quote= '"';
193           } else {
194             quote= '"';
195             value = value.replaceAll("\"", "\\\"");          
196           }
197     
198           if (value.indexOf('\n') >= 0) {
199             value = value.replaceAll("\n", "\\n");
200           }
201     
202           return Pair.make(value, quote);
203         }

205         private String extructJS(String attValue) {
206           if (attValue == null){
207             return "";
208           }
209           
210           String content;
211           if (attValue.toLowerCase().equals("javascript:")) {
212             content = attValue.substring("javascript:".length());
213           } else {
214             content = attValue;
215           }
216     
217           return content;
218         }

220         protected void handleScript(ITag tag) {
221     
222           Pair<String,Position> content = tag.getAttributeByName("src");
223     
224           try {
225             if (content != null) {
226               // script is out-of-line
227               getScriptFromUrl(content.fst, tag);
228             }
229     
230           } catch (IOException e) {
231             if (DEBUG) {
232               System.err.println("Error reading script file: " + e.getMessage());
233             }
234           }
235         }
236     
237         private void getScriptFromUrl(String urlAsString, ITag scriptTag) throws IOException, MalformedURLException {
238           URL scriptSrc = new URL(entrypointUrl, urlAsString);
239           Reader scriptInputStream;
240           try {
241             BOMInputStream bs = new BOMInputStream(scriptSrc.openConnection().getInputStream(), false, 
242                 ByteOrderMark.UTF_8, 
243                 ByteOrderMark.UTF_16LE, ByteOrderMark.UTF_16BE,
244                 ByteOrderMark.UTF_32LE, ByteOrderMark.UTF_32BE);
245             if (bs.hasBOM()) {
246               System.err.println("removing BOM " + bs.getBOM());
247             }
248             scriptInputStream = new InputStreamReader(bs);
249           } catch (Exception e) {
250             //it looks like this happens when we can't resolve the url?
251             if (DEBUG) {
252               System.err.println("Error reading script: " + scriptSrc);
253               System.err.println(e);
254               e.printStackTrace(System.err);
255             }
256             return;
257           }
258           
259           BufferedReader scriptReader = null;
260           try {
261             String line;
262             scriptReader = new BufferedReader(scriptInputStream);
263             StringBuffer x = new StringBuffer();
264             while ((line = scriptReader.readLine()) != null) {
265               x.append(line).append("\n");
266             }
267     
268             scriptRegion.println(x.toString(), scriptTag.getElementPosition(), scriptSrc, false);
269     
270           } finally {
271             if (scriptReader != null) {
272               scriptReader.close();
273             }
274           }
275         }

277         protected String getScriptName(URL url) throws MalformedURLException {
278           String file = url.getFile();
279           int lastIdxOfSlash = file.lastIndexOf('/');
280           file = (lastIdxOfSlash == (-1)) ? file : file.substring(lastIdxOfSlash + 1);
281           return file;
282         }

284         @Override
285         public void writeToFinalRegion(SourceRegion finalRegion) {
286           // wrapping the embedded scripts with a fake method of the window. Required for making this == window.
287           finalRegion.println("window.__MAIN__ = function __WINDOW_MAIN__(){");
288     
289           finalRegion.write(scriptRegion);
290     
291           finalRegion.write(domRegion);
292     
293           finalRegion.println("  document.URL = new String(\"" + entrypointUrl + "\");");
294     
295           finalRegion.println("while (true){ ");
296           finalRegion.write(entrypointRegion);
297           finalRegion.println("} // while (true)");
298             
299           finalRegion.println("} // end of window.__MAIN__");
300           finalRegion.println("window.__MAIN__();");
301         }

304       /**
305        * for storing the name of the temp file created by extractSources()
306        */
307       private File tempFile;
308       
309       @Override
310       public Set<MappedSourceModule> extractSources(URL entrypointUrl, IHtmlParser htmlParser, IUrlResolver urlResolver)
311       throws IOException, Error {
312     
313         Reader inputStreamReader = WebUtil.getStream(entrypointUrl);
314         IGeneratorCallback htmlCallback = createHtmlCallback(entrypointUrl, urlResolver); 
315         htmlParser.parse(entrypointUrl, inputStreamReader, htmlCallback, entrypointUrl.getFile());
316     
317         SourceRegion finalRegion = new SourceRegion();
318         htmlCallback.writeToFinalRegion(finalRegion);
319         
320         // writing the final region into one SourceFileModule.
321         File outputFile = createOutputFile(entrypointUrl, DELETE_UPON_EXIT, USE_TEMP_NAME);
322         tempFile = outputFile;
323         FileMapping fileMapping = finalRegion.writeToFile(new PrintWriter(new FileWriter(outputFile)));
324         if (fileMapping == null) {
325           fileMapping = new EmptyFileMapping();
326         }
327         MappedSourceModule singleFileModule = new MappedSourceFileModule(outputFile, outputFile.getName(), fileMapping);
328         return Collections.singleton(singleFileModule);
329       }
330     
331       protected IGeneratorCallback createHtmlCallback(URL entrypointUrl, IUrlResolver urlResolver) {
332         return new HtmlCallback(entrypointUrl, urlResolver);
333       }

335       private File createOutputFile(URL url, boolean delete, boolean useTempName) throws IOException {
336         File outputFile;
337         String fileName = new File(url.getFile()).getName();
338         if (fileName.length() < 5) {
339           fileName = "xxxx" + fileName; 
340         }
341         if (useTempName) {
342           outputFile = File.createTempFile(fileName, ".js");
343         } else {
344           outputFile = new File(fileName);
345         }
346         if (outputFile.exists()){
347           outputFile.delete();
348         }
349         if(delete){
350           outputFile.deleteOnExit();
351         }
352         return outputFile;
353       }
