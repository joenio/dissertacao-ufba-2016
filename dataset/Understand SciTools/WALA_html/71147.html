31      * TODO cache generated IRs
32      * 
33      * @see <a
34      *      href="https://developer.mozilla.org/en/JavaScript/Reference/Global_Objects/Function/Apply">MDN

37     public class JavaScriptFunctionApplyContextInterpreter extends AstContextInsensitiveSSAContextInterpreter {
38     
39       private static final TypeName APPLY_TYPE_NAME = TypeName.findOrCreate("Lprologue.js/Function_prototype_apply");
40     
41       public JavaScriptFunctionApplyContextInterpreter(AnalysisOptions options, AnalysisCache cache) {
42         super(options, cache);
43       }
44     
45       @Override
46       public boolean understands(CGNode node) {
47         return node.getMethod().getDeclaringClass().getName().equals(APPLY_TYPE_NAME);
48       }
49     
50       @Override
51       public IR getIR(CGNode node) {
52         assert understands(node);
53         @SuppressWarnings("unchecked")
54         ContextItem.Value<Boolean> isNonNullArray = (ContextItem.Value<Boolean>) node.getContext().get(JavaScriptFunctionApplyContextSelector.APPLY_NON_NULL_ARGS);
55         // isNonNullArray can be null if, e.g., due to recursion bounding we have no
56         // information on the arguments parameter
57         if (isNonNullArray == null || isNonNullArray.getValue()) {
58           return makeIRForArgList(node);
59         } else {
60           return makeIRForNoArgList(node);
61         }
62       }
63     
64       private IR makeIRForArgList(CGNode node) {
65         // we have: v1 is dummy apply method
66         // v2 is function to be invoked
67         // v3 is argument to be passed as 'this'
68         // v4 is array containing remaining arguments
69         // Ideally, we would take advantage of cases like constant arrays and
70         // precisely pass arguments in the appropriate slots. Unfortunately, in the
71         // pointer analysis fixed-point computation, it's possible that we will
72         // process the apply() call and then process some update to the arguments
73         // array, reflected only in its property values and object catalog. Perhaps
74         // eventually, we could create contexts based on the catalog of the object
75         // and then do a better job, but since the catalog is not passed directly as
76         // a parameter to apply(), this is not so easy.
77         // In the meantime, we do things imprecisely. We read an arbitrary
78         // enumerable property name of the argument list (via an
79         // EachElementGetInstruction), perform a dynamic read of that property, and
80         // then pass the resulting values in all argument positions (except 'this').
81         //
82         // NOTE: we don't know how many arguments the callee will take, whether it
83         // uses
84         // the arguments array, etc. For now, we use an unsound hack and pass the
85         // argument 10 times.
86         //
87         // NOTE: strictly speaking, using EachElementGet could be imprecise, as it
88         // should
89         // return properties inherited via the prototype chain. However, since this
90         // behavior
91         // is not modeled in WALA as of now, using the instruction is ok.
92         MethodReference ref = node.getMethod().getReference();
93         IClass declaringClass = node.getMethod().getDeclaringClass();
94         JSInstructionFactory insts = (JSInstructionFactory) declaringClass.getClassLoader().getInstructionFactory();
95         // nargs needs to match that of Function.apply(), even though no argsList
96         // argument was passed in this case
97         int nargs = 4;
98         JavaScriptSummary S = new JavaScriptSummary(ref, nargs);
99     
100         int numParamsToPass = 10;
101         int[] paramsToPassToInvoked = new int[numParamsToPass + 1];
102         // pass the 'this' argument first
103         paramsToPassToInvoked[0] = 3;
104     
105     //    int curValNum = passArbitraryPropertyValAsParams(insts, nargs, S, paramsToPassToInvoked);
106         int curValNum = passActualPropertyValsAsParams(insts, nargs, S, paramsToPassToInvoked);
107         
108         CallSiteReference cs = new JSCallSiteReference(S.getNextProgramCounter());
109     
110         // function being invoked is in v2
111         int resultVal = curValNum++;
112         int excVal = curValNum++;
113         S.addStatement(insts.Invoke(S.getNumberOfStatements(), 2, resultVal, paramsToPassToInvoked, excVal, cs));
114         S.getNextProgramCounter();
115     
116         S.addStatement(insts.ReturnInstruction(S.getNumberOfStatements(), resultVal, false));
117         S.getNextProgramCounter();
118     
119         JavaScriptSummarizedFunction t = new JavaScriptSummarizedFunction(ref, S, declaringClass);
120         return t.makeIR(node.getContext(), null);
121       }
122     
123       @SuppressWarnings("unused")
124       private int passArbitraryPropertyValAsParams(JSInstructionFactory insts, int nargs, JavaScriptSummary S, int[] paramsToPassToInvoked) {
125         // read an arbitrary property name via EachElementGet
126         int curValNum = nargs + 2;
127         int eachElementGetResult = curValNum++;
128         int nullPredVn = curValNum++;
129         S.addConstant(nullPredVn, new ConstantValue(null));
130         S.addStatement(insts.EachElementGetInstruction(S.getNumberOfStatements(), eachElementGetResult, 4, nullPredVn));
131         S.getNextProgramCounter();
132         // read value from the arbitrary property name
133         int propertyReadResult = curValNum++;
134         S.addStatement(insts.PropertyRead(S.getNumberOfStatements(), propertyReadResult, 4, eachElementGetResult));
135         S.getNextProgramCounter();
136         for (int i = 1; i < paramsToPassToInvoked.length; i++) {
137           paramsToPassToInvoked[i] = propertyReadResult;
138         }
139         return curValNum;
140       }
141       
142       private int passActualPropertyValsAsParams(JSInstructionFactory insts, int nargs, JavaScriptSummary S, int[] paramsToPassToInvoked) {
143         // read an arbitrary property name via EachElementGet
144         int curValNum = nargs + 2;
145         for (int i = 1; i < paramsToPassToInvoked.length; i++) {
146           // create a String constant for i-1
147           final int constVN = curValNum++;
148           // the commented line is correct, but it doesn't work because
149           // of our broken handling of int constants as properties.
150           // TODO fix property handling, and then fix this
151     //      S.addConstant(constVN, new ConstantValue(Integer.toString(i-1)));
152           S.addConstant(constVN, new ConstantValue(i-1));
153           int propertyReadResult = curValNum++;
154           // 4 is position of arguments array
155           S.addStatement(insts.PropertyRead(S.getNumberOfStatements(), propertyReadResult, 4, constVN));
156           S.getNextProgramCounter();
157           paramsToPassToInvoked[i] = propertyReadResult;
158         }
159         return curValNum;
160       }

162       private IR makeIRForNoArgList(CGNode node) {
163         // kind of a hack; re-use the summarized function infrastructure
164         MethodReference ref = node.getMethod().getReference();
165         IClass declaringClass = node.getMethod().getDeclaringClass();
166         JSInstructionFactory insts = (JSInstructionFactory) declaringClass.getClassLoader().getInstructionFactory();
167         // nargs needs to match that of Function.apply(), even though no argsList
168         // argument was passed in this case
169         int nargs = 4;
170         JavaScriptSummary S = new JavaScriptSummary(ref, nargs);
171     
172         // generate invocation instruction for the real method being invoked
173         int resultVal = nargs + 2;
174         CallSiteReference cs = new JSCallSiteReference(S.getNextProgramCounter());
175         int[] params = new int[1];
176         params[0] = 3;
177         // function being invoked is in v2
178         S.addStatement(insts.Invoke(S.getNumberOfStatements(), 2, resultVal, params, resultVal + 1, cs));
179         S.getNextProgramCounter();
180     
181         S.addStatement(insts.ReturnInstruction(S.getNumberOfStatements(), resultVal, false));
182         S.getNextProgramCounter();
183     
184         JavaScriptSummarizedFunction t = new JavaScriptSummarizedFunction(ref, S, declaringClass);
185         return t.makeIR(node.getContext(), null);
186       }
187     
188       @Override
189       public DefUse getDU(CGNode node) {
190         assert understands(node);
191         return new DefUse(getIR(node));
192       }
