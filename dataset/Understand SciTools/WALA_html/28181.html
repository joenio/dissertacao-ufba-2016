20     /**
21      * An exception for invalid positions.
22      * 
23      * @author Siegfried Weber
24      * @author Juergen Graf <juergen.graf@gmail.com>
25      */
26     class InvalidPositionException extends Exception {
27     
28       private static final long serialVersionUID = 7949660405524028349L;
29     
30       /** possible causes for this exception */
31       enum Cause {
32         LINE_NUMBER_ZERO, COLUMN_NUMBER_ZERO, LINE_NUMBER_OUT_OF_RANGE, COLUMN_NUMBER_OUT_OF_RANGE
33       }
34     
35       /** the cause for this exception */
36       private Cause cause;
37     
38       /**
39        * Constructs an instance of <code>InvalidRangeException</code> with the
40        * specified cause.
41        * 
42        * @param c
43        *          the cause
44        */
45       InvalidPositionException(Cause c) {
46         cause = c;
47       }
48     
49       /**
50        * Returns the cause for this exception.
51        * 
52        * @return the cause for this exception
53        */
54       Cause getThisCause() {
55         return cause;
56       }
57     }
