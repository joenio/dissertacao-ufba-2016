23     public class Util {
24     
25       public static Selector methodEntityToSelector(CAstEntity methodEntity) {
26         Atom name= Atom.findOrCreateUnicodeAtom(methodEntity.getName());
27         CAstType.Function signature= (CAstType.Function) methodEntity.getType();
28         // Use signature to determine # of args; (entity's args includes 'this')
29         TypeName retTypeName= 
30           TypeName.string2TypeName(signature.getReturnType().getName());
31         TypeName[] argTypeNames= 
32           (signature.getArgumentCount() == 0) ? 
33           null :
34           new TypeName[signature.getArgumentCount()];
35     
36         int i= 0;
37         for(Iterator iter= signature.getArgumentTypes().iterator(); 
38     	iter.hasNext(); i++)
39         {
40           CAstType argType= (CAstType) iter.next();
41           argTypeNames[i]= TypeName.string2TypeName(argType.getName());
42         }
43     
44         Descriptor desc= Descriptor.findOrCreate(argTypeNames, retTypeName);
45         
46         return new Selector(name, desc);
47       }
48     
49       public static Atom fieldEntityToAtom(CAstEntity fieldEntity) {
50         return Atom.findOrCreateUnicodeAtom(fieldEntity.getName());
51       }
52     }
