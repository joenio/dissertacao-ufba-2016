14     package com.ibm.wala.cast.java.types;
15     
16     import java.util.Collection;
17     import java.util.Collections;
18     import java.util.Map;
19     
20     import com.ibm.wala.cast.tree.CAstType;
21     import com.ibm.wala.util.collections.HashMapFactory;
22     
23     public class JavaPrimitiveTypeMap {
24       public static final Map<String, JavaPrimitiveType> primNameMap = HashMapFactory.make();
25     
26       public static class JavaPrimitiveType implements CAstType.Primitive {
27         String fLongName;
28     
29         String fShortName;
30     
31         private JavaPrimitiveType(String longName, String shortName) {
32           fLongName = longName;
33           fShortName = shortName;
34         }
35     
36         @Override
37         public String getName() {
38           return fShortName;
39         }
40     
41         public String getLongName() {
42           return fLongName;
43         }
44     
45         @Override
46         public Collection getSupertypes() {
47           return Collections.EMPTY_LIST;
48         }
49       }
50     
51       public static String getShortName(String longName) {
52         return primNameMap.get(longName).getName();
53       }
54     
55       public static JavaPrimitiveType lookupType(String longName) {
56         return primNameMap.get(longName);
57       }
58     
59       public static final JavaPrimitiveType VoidType = new JavaPrimitiveType("void", "V");
60     
61       static {
62         primNameMap.put("int", new JavaPrimitiveType("int", "I"));
63         primNameMap.put("long", new JavaPrimitiveType("long", "J"));
64         primNameMap.put("short", new JavaPrimitiveType("short", "S"));
65         primNameMap.put("char", new JavaPrimitiveType("char", "C"));
66         primNameMap.put("byte", new JavaPrimitiveType("byte", "B"));
67         primNameMap.put("boolean", new JavaPrimitiveType("boolean", "Z"));
68         primNameMap.put("float", new JavaPrimitiveType("float", "F"));
69         primNameMap.put("double", new JavaPrimitiveType("double", "D"));
70         primNameMap.put("void", VoidType);
71       }
72     }
