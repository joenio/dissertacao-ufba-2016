203       protected boolean unconditionallyAddConstraintsFromNode(CGNode node, IProgressMonitor monitor) throws CancelException {
204         this.monitor = monitor;
205         if (PERIODIC_WIPE_SOFT_CACHES) {
206           wipeCount++;
207           if (wipeCount >= WIPE_SOFT_CACHE_INTERVAL) {
208             wipeCount = 0;
209             ReferenceCleanser.clearSoftCaches();
210           }
211         }
212     
213         if (DEBUG) {
214           System.err.println("\n\nAdd constraints from node " + node);
215         }
216         IR ir = getCFAContextInterpreter().getIR(node);
217         if (DEBUG) {
218           if (ir == null) {
219             System.err.println("\n   No statements\n");
220           } else {
221             try {
222               System.err.println(ir.toString());
223             } catch (Error e) {
224               e.printStackTrace();
225             }
226           }
227         }
228     
229         if (ir == null) {
230           return false;
231         }
232     
233         addNodeInstructionConstraints(node, monitor);
234     
235         DefUse du = getCFAContextInterpreter().getDU(node);
236         addNodePassthruExceptionConstraints(node, ir, du);
237         // conservatively assume something changed
238         return true;
239       }

252       protected void addNodeInstructionConstraints(CGNode node, IProgressMonitor monitor) throws CancelException {
253         this.monitor = monitor;
254         ConstraintVisitor v = makeVisitor(node);
255     
256         IR ir = v.ir;
257         ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg = ir.getControlFlowGraph();
258         for (Iterator<ISSABasicBlock> x = cfg.iterator(); x.hasNext();) {
259           BasicBlock b = (BasicBlock) x.next();
260           addBlockInstructionConstraints(node, cfg, b, v, monitor);
261           if (wasChanged(node)) {
262             return;
263           }
264         }

268        * Add constraints for a particular basic block.
269        * @throws CancelException 
270        */
271       protected void addBlockInstructionConstraints(CGNode node, ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg, BasicBlock b,
272           ConstraintVisitor v, IProgressMonitor monitor) throws CancelException {
273         this.monitor = monitor;
274         v.setBasicBlock(b);
275     
276         // visit each instruction in the basic block.
277         for (Iterator<SSAInstruction> it = b.iterator(); it.hasNext();) {
278           MonitorUtil.throwExceptionIfCanceled(monitor);      
279           SSAInstruction s = it.next();
280           if (s != null) {
281             s.visit(v);
282             if (wasChanged(node)) {
283               return;
284             }
285           }
286         }
287     
288         addPhiConstraints(node, cfg, b, v);
289       }

291       private void addPhiConstraints(CGNode node, ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg, BasicBlock b,
292           ConstraintVisitor v) {
293         // visit each phi instruction in each successor block
294         for (Iterator sbs = cfg.getSuccNodes(b); sbs.hasNext();) {
295           BasicBlock sb = (BasicBlock) sbs.next();
296           if (!sb.hasPhi()) {
297             continue;
298           }
299           int n = 0;
300           for (Iterator<? extends IBasicBlock> back = cfg.getPredNodes(sb); back.hasNext(); n++) {
301             if (back.next() == b) {
302               break;
303             }
304           }
305           assert n < cfg.getPredNodeCount(sb);
306           for (Iterator<? extends SSAInstruction> phis = sb.iteratePhis(); phis.hasNext();) {
307             SSAPhiInstruction phi = (SSAPhiInstruction) phis.next();
308             if (phi == null) {
309               continue;
310             }
311             PointerKey def = getPointerKeyForLocal(node, phi.getDef());
312             if (hasNoInterestingUses(node, phi.getDef(), v.du)) {
313               system.recordImplicitPointsToSet(def);
314             } else {
315               // the following test restricts the constraints to reachable
316               // paths, according to verification constraints
317               if (phi.getUse(n) > 0) {
318                 PointerKey use = getPointerKeyForLocal(node, phi.getUse(n));
319                 if (contentsAreInvariant(v.symbolTable, v.du, phi.getUse(n))) {
320                   system.recordImplicitPointsToSet(use);
321                   InstanceKey[] ik = getInvariantContents(v.symbolTable, v.du, node, phi.getUse(n), this);
322                   for (int i = 0; i < ik.length; i++) {
323                     system.newConstraint(def, ik[i]);
324                   }
325                 } else {
326                   system.newConstraint(def, assignOperator, use);
327                 }
328               }
329             }
330           }
331         }
332       }

350       /**
351        * Generate constraints which assign exception values into an exception pointer
352        * 
353        * @param node governing node
354        * @param peis list of PEI instructions
355        * @param exceptionVar PointerKey representing a pointer to an exception value
356        * @param catchClasses the types "caught" by the exceptionVar
357        */
358       private void addExceptionDefConstraints(IR ir, DefUse du, CGNode node, List<ProgramCounter> peis, PointerKey exceptionVar,
359           Set<IClass> catchClasses) {
360         if (DEBUG) {
361           System.err.println("Add exception def constraints for node " + node);
362         }
363         for (Iterator<ProgramCounter> it = peis.iterator(); it.hasNext();) {
364           ProgramCounter peiLoc = it.next();
365           if (DEBUG) {
366             System.err.println("peiLoc: " + peiLoc);
367           }
368           SSAInstruction pei = ir.getPEI(peiLoc);
369     
370           if (DEBUG) {
371             System.err.println("Add exceptions from pei " + pei);
372           }
373     
374           if (pei instanceof SSAAbstractInvokeInstruction) {
375             SSAAbstractInvokeInstruction s = (SSAAbstractInvokeInstruction) pei;
376             PointerKey e = getPointerKeyForLocal(node, s.getException());
377     
378             if (!SHORT_CIRCUIT_SINGLE_USES || !hasUniqueCatchBlock(s, ir)) {
379               addAssignmentsForCatchPointerKey(exceptionVar, catchClasses, e);
380             }// else {
381             // System.err.println("SKIPPING ASSIGNMENTS TO " + exceptionVar + " FROM " +
382             // e);
383             // }
384           } else if (pei instanceof SSAAbstractThrowInstruction) {
385             SSAAbstractThrowInstruction s = (SSAAbstractThrowInstruction) pei;
386             PointerKey e = getPointerKeyForLocal(node, s.getException());
387     
388             if (contentsAreInvariant(ir.getSymbolTable(), du, s.getException())) {
389               InstanceKey[] ik = getInvariantContents(ir.getSymbolTable(), du, node, s.getException(), this);
390               for (int i = 0; i < ik.length; i++) {
391                 system.findOrCreateIndexForInstanceKey(ik[i]);
392                 assignInstanceToCatch(exceptionVar, catchClasses, ik[i]);
393               }
394             } else {
395               addAssignmentsForCatchPointerKey(exceptionVar, catchClasses, e);
396             }
397           }
398     
399           // Account for those exceptions for which we do not actually have a
400           // points-to set for
401           // the pei, but just instance keys
402           Collection<TypeReference> types = pei.getExceptionTypes();
403           if (types != null) {
404             for (Iterator<TypeReference> it2 = types.iterator(); it2.hasNext();) {
405               TypeReference type = it2.next();
406               if (type != null) {
407                 InstanceKey ik = getInstanceKeyForPEI(node, peiLoc, type, instanceKeyFactory);
408                 if (ik == null) {
409                   continue;
410                 }
411                 assert ik instanceof ConcreteTypeKey : "uh oh: need to implement getCaughtException constraints for instance " + ik;
412                 ConcreteTypeKey ck = (ConcreteTypeKey) ik;
413                 IClass klass = ck.getType();
414                 if (PropagationCallGraphBuilder.catches(catchClasses, klass, cha)) {
415                   system.newConstraint(exceptionVar, getInstanceKeyForPEI(node, peiLoc, type, instanceKeyFactory));
416                 }
417               }
418             }
419           }
420         }
421       }

426       protected static boolean hasUniqueCatchBlock(SSAAbstractInvokeInstruction call, IR ir) {
427         ISSABasicBlock[] bb = ir.getBasicBlocksForCall(call.getCallSite());
428         if (bb.length == 1) {
429           Iterator it = ir.getControlFlowGraph().getExceptionalSuccessors(bb[0]).iterator();
430           // check that there's exactly one element in the iterator
431           if (it.hasNext()) {
432             it.next();
433             return (!it.hasNext());
434           }
435         }
436         return false;
437       }

466       /**
467        * @return a List of Instructions that may transfer control to bb via an exceptional edge
468        * @throws IllegalArgumentException if ir is null
469        */
470       public static List<ProgramCounter> getIncomingPEIs(IR ir, ISSABasicBlock bb) {
471         if (ir == null) {
472           throw new IllegalArgumentException("ir is null");
473         }
474         if (DEBUG) {
475           System.err.println("getIncomingPEIs " + bb);
476         }
477         ControlFlowGraph<SSAInstruction, ISSABasicBlock> g = ir.getControlFlowGraph();
478         List<ProgramCounter> result = new ArrayList<ProgramCounter>(g.getPredNodeCount(bb));
479         for (Iterator it = g.getPredNodes(bb); it.hasNext();) {
480           BasicBlock pred = (BasicBlock) it.next();
481           if (DEBUG) {
482             System.err.println("pred: " + pred);
483           }
484           if (pred.isEntryBlock())
485             continue;
486           int index = pred.getLastInstructionIndex();
487           SSAInstruction pei = ir.getInstructions()[index];
488           // Note: pei might be null if pred is unreachable.
489           // TODO: consider pruning CFG for unreachable blocks.
490           if (pei != null && pei.isPEI()) {
491             if (DEBUG) {
492               System.err.println("PEI: " + pei + " index " + index + " PC " + g.getProgramCounter(index));
493             }
494             result.add(new ProgramCounter(g.getProgramCounter(index)));
495           }
496         }
497         return result;
498       }

1632           this.call = call;
1633           this.node = node;
1634           this.constParams = constParams;
1635           this.uniqueCatch = uniqueCatch;
1636           this.dispatchIndices = IntSetUtil.toArray(dispatchIndices);
1637           // we better always be interested in the receiver
1638           assert this.dispatchIndices[0] == 0;
1639           previousPtrs = new MutableIntSet[dispatchIndices.size()];

1647         /*
1648          * @see com.ibm.wala.dataflow.fixpoint.UnaryOperator#evaluate(com.ibm.wala.dataflow.fixpoint.IVariable,
1649          * com.ibm.wala.dataflow.fixpoint.IVariable)
1650          */
1651         @Override
1652         public byte evaluate(PointsToSetVariable lhs, final PointsToSetVariable[] rhs) {
1653           assert dispatchIndices.length >= rhs.length : "bad operator at " + call;
1654           // did evaluating the dispatch operation add a new possible target
1655           // to the call site?  
1656           final MutableBoolean addedNewTarget = new MutableBoolean();
1657           
1658           final MutableIntSet receiverVals;
1659           if (constParams != null && constParams[0] != null) {
1660             receiverVals = IntSetUtil.make();
1661             for(InstanceKey ik : constParams[0]) {
1662               receiverVals.add(system.getInstanceIndex(ik));
1663             }
1664           } else {
1665             receiverVals = rhs[0].getValue();
1666           }
1667           
1668           if (receiverVals == null) {
1669             // this constraint was put on the work list, probably by
1670             // initialization,
1671             // even though the right-hand-side is empty.
1672             // TODO: be more careful about what goes on the worklist to
1673             // avoid this.
1674             if (DEBUG) {
1675               System.err.println("EVAL dispatch with value null");
1676             }
1677             return NOT_CHANGED;        
1678             
1679           }
1680           // we handle the parameter positions one by one, rather than enumerating
1681           // the cartesian product of possibilities. this disallows
1682           // context-sensitivity policies like true CPA, but is necessary for
1683           // performance.
1684           InstanceKey keys[] = new InstanceKey[constParams == null? dispatchIndices[dispatchIndices.length-1]+1: constParams.length];

1686           // to check for redundancy
1687           boolean newReceiver = !receiverVals.isSubset(previousPtrs[0]);
1688           // keep separate rhsIndex, since it doesn't advance for constant
1689           // parameters
1690           int rhsIndex = (constParams != null && constParams[0] != null)? 0: 1;
1691           // this flag is set to true if we ever call handleAllReceivers() in the
1692           // loop below. we need to catch the case where we have a new receiver, but
1693           // there are no other dispatch indices with new values
1694           boolean propagatedReceivers = false;
1695           // we start at index 1 since we need to handle the receiver specially; see
1696           // below
1697           for (int index = 1; index < dispatchIndices.length; index++) {
1698             try {
1699               MonitorUtil.throwExceptionIfCanceled(monitor);
1700             } catch (CancelException e) {
1701               throw new CancelRuntimeException(e);
1702             }
1703             int paramIndex = dispatchIndices[index];
1704             assert keys[paramIndex] == null;
1705             final MutableIntSet prevAtIndex = previousPtrs[index];
1706             if (constParams != null && constParams[paramIndex] != null) {
1707               // we have a constant parameter.  only need to propagate again if we've never done it before or if we have a new receiver
1708               if (newReceiver || prevAtIndex.isEmpty()) {
1709                 for(int i = 0; i < constParams[paramIndex].length; i++) {
1710                   keys[paramIndex] = constParams[paramIndex][i];
1711                   handleAllReceivers(receiverVals,keys, addedNewTarget);
1712                   propagatedReceivers = true;
1713                   int ii = system.instanceKeys.getMappedIndex(constParams[paramIndex][i]);
1714                   prevAtIndex.add(ii);
1715                 }            
1716               }
1717             } else { // non-constant parameter
1718               PointsToSetVariable v = rhs[rhsIndex];
1719               if (v.getValue() != null) {
1720                 IntIterator ptrs = v.getValue().intIterator();
1721                 while (ptrs.hasNext()) {
1722                   int ptr = ptrs.next();
1723                   if (newReceiver || !prevAtIndex.contains(ptr)) {
1724                     keys[paramIndex] = system.getInstanceKey(ptr);
1725                     handleAllReceivers(receiverVals,keys, addedNewTarget);
1726                     propagatedReceivers = true;
1727                     prevAtIndex.add(ptr);
1728                   }
1729                 }
1730               } 
1731               rhsIndex++;
1732             }
1733             keys[paramIndex] = null;
1734           }
1735           if (newReceiver) {
1736             if (!propagatedReceivers) {
1737               // we have a new receiver value, and it wasn't propagated at all,
1738               // so propagate it now
1739               handleAllReceivers(receiverVals, keys, addedNewTarget);
1740             }
1741             // update receiver cache
1742             previousPtrs[0].addAll(receiverVals);
1743           }
1744     
1745           byte sideEffectMask = addedNewTarget.b ? (byte) SIDE_EFFECT_MASK : 0;
1746           return (byte) (NOT_CHANGED | sideEffectMask);
1747         }

1749         private void handleAllReceivers(MutableIntSet receiverVals, InstanceKey[] keys, MutableBoolean sideEffect) {
1750           assert keys[0] == null;
1751           IntIterator receiverIter = receiverVals.intIterator();
1752           while (receiverIter.hasNext()) {
1753             final int rcvr = receiverIter.next();
1754             keys[0] = system.getInstanceKey(rcvr);
1755             if (clone2Assign) {
1756               // for efficiency: assume that only call sites that reference
1757               // clone() might dispatch to clone methods
1758               if (call.getCallSite().getDeclaredTarget().getSelector().equals(cloneSelector)) {
1759                 IClass recv = (keys[0] != null) ? keys[0].getConcreteType() : null;
1760                 IMethod targetMethod = getOptions().getMethodTargetSelector().getCalleeTarget(node, call.getCallSite(), recv);
1761                 if (targetMethod != null && targetMethod.getReference().equals(CloneInterpreter.CLONE)) {
1762                   // treat this call to clone as an assignment
1763                   PointerKey result = getPointerKeyForLocal(node, call.getDef());
1764                   PointerKey receiver = getPointerKeyForLocal(node, call.getReceiver());
1765                   system.newConstraint(result, assignOperator, receiver);
1766                   return;
1767                 }
1768               }

1865       protected void iterateCrossProduct(final CGNode caller, final SSAAbstractInvokeInstruction call, IntSet parameters,
1866           final InstanceKey[][] invariants, final VoidFunction<InstanceKey[]> f) {
1867         final int params[] = IntSetUtil.toArray(parameters);
1868         final InstanceKey[] keys = new InstanceKey[call.getNumberOfParameters()];
1869         final CallSiteReference site = call.getCallSite();
1870         new Object() {
1871           private void rec(final int pi) {
1872             if (pi == params.length) {
1873               f.apply(keys);
1874             } else {
1875               final int p = params[pi];
1876               int vn = call.getUse(p);
1877               PointerKey var = getPointerKeyForLocal(caller, vn);
1878               InstanceKey[] ik = invariants != null ? invariants[p] : null;
1879               if (ik != null) {
1880                 if (ik.length > 0) {
1881                   for (int i = 0; i < ik.length; i++) {
1882                     system.findOrCreateIndexForInstanceKey(ik[i]);
1883                     keys[p] = ik[i];
1884                     rec(pi + 1);
1885                   }
1886                 } else {
1887                   if (!site.isDispatch() || p != 0) {
1888                     keys[p] = null;
1889                     rec(pi + 1);
1890                   }
1891                 }
1892               } else {
1893                 IntSet s = system.findOrCreatePointsToSet(var).getValue();
1894                 if (s != null && !s.isEmpty()) {
1895                   s.foreach(new IntSetAction() {
1896                     @Override
1897                     public void act(int x) {
1898                       keys[p] = system.getInstanceKey(x);
1899                       rec(pi + 1);
1900                     }
1901                   });
1902                 } else {
1903                   if (!site.isDispatch() || p != 0) {
1904                     keys[p] = null;
1905                     rec(pi + 1);
1906                   }
1907                 }
1908               }
1909             }
1910           }
1911         }.rec(0);
1912       }
