11     package com.ibm.wala.cast.tree.impl;
12     
13     import java.io.IOException;
14     import java.io.InputStreamReader;
15     import java.io.Reader;
16     import java.net.URL;
17     
18     public class LineNumberPosition extends AbstractSourcePosition {
19       private final URL url;
20       private final URL localFile;
21       private final int lineNumber;
22     
23       public LineNumberPosition(URL url, URL localFile, int lineNumber) {
24         this.url = url;
25         this.localFile = localFile;
26         this.lineNumber = lineNumber;
27       }
28     
29       @Override
30       public int getFirstLine() { return lineNumber; }
31     
32       @Override
33       public int getLastLine() { return lineNumber; }
34     
35       @Override
36       public int getFirstCol() { return -1; }
37     
38       @Override
39       public int getLastCol() { return -1; }
40       
41       @Override
42       public int getFirstOffset() { return -1; }
43      
44       @Override
45       public int getLastOffset() { return -1; }
46     
47       @Override
48       public URL getURL() { return url; }
49     	
50       @Override
51       public Reader getReader() throws IOException { 
52         return new InputStreamReader(localFile.openConnection().getInputStream());
53       }
54     
55       @Override
56       public String toString() {
57         String nm = url.getFile();
58         nm = nm.substring(nm.lastIndexOf('/') + 1);
59         return "[" + nm + ":"+lineNumber+"]";
60       }
61     }
