11     package com.ibm.wala.shrikeBT;
12     
13     import java.io.IOException;
14     import java.io.Writer;
15     
16     /**
17      * This is a very simple component to disassemble a ShrikeBT method. The disassembly is just the list of ShrikeBT instructions,
18      * annotated with exception handler blocks and the mapping back to the original bytecodes.
19      */
20     public class Disassembler {
21       final private IInstruction[] instructions;
22     
23       final private ExceptionHandler[][] handlers;
24     
25       final private int[] instructionsToBytecodes;
26     
27       /**
28        * Create a disassembler for a method.
29        */
30       public Disassembler(IInstruction[] instructions, ExceptionHandler[][] handlers, int[] instructionsToBytecodes) {
31         this.instructions = instructions;
32         this.handlers = handlers;
33         this.instructionsToBytecodes = instructionsToBytecodes;
34       }
35     
36       /**
37        * Create a disassembler for a method.
38        * 
39        * @throws NullPointerException if data is null
40        */
41       public Disassembler(MethodData data) throws NullPointerException {
42         this(data.getInstructions(), data.getHandlers(), data.getInstructionsToBytecodes());
43       }
44     
45       /**
46        * Write the disassembly to a stream. Each line is prefixed with 'prefix'.
47        */
48       public void disassembleTo(String prefix, Writer w) throws IOException {
49         for (int j = 0; j < instructions.length; j++) {
50           w.write(prefix + j + ": " + instructions[j] + " (" + instructionsToBytecodes[j] + ")\n");
51           for (int k = 0; k < handlers[j].length; k++) {
52             w.write(prefix + "    Handles " + handlers[j][k].catchClass + " at " + handlers[j][k].handler + "\n");
53           }
54         }
55       }
56     
57       /**
58        * Write the disassembly to a stream.
59        */
60       public void disassembleTo(Writer w) throws IOException {
61         disassembleTo("", w);
62       }
63     }
