11     package com.ibm.wala.cast.js.translator;
12     
13     import java.io.IOException;
14     import java.io.InputStreamReader;
15     import java.io.Reader;
16     import java.net.URL;
17     
18     import com.ibm.wala.cast.tree.CAstSourcePositionMap.Position;
19     import com.ibm.wala.cast.tree.impl.AbstractSourcePosition;
20     
21     public class RangePosition extends AbstractSourcePosition implements Position {
22       private final URL url;
23       private final int line;
24       private final int startOffset;
25       private final int endOffset;
26       
27       
28       public RangePosition(URL url, int line, int startOffset, int endOffset) {
29         super();
30         this.url = url;
31         this.line = line;
32         this.startOffset = startOffset;
33         this.endOffset = endOffset;
34       }
35     
36       @Override
37       public int compareTo(Object o) {
38         Position other = (Position) o;
39         if (startOffset != other.getFirstOffset()) {
40           return startOffset - other.getFirstOffset();
41         } else {
42           return endOffset - other.getLastOffset();
43         }
44       }
45     
46       @Override
47       public int getFirstLine() {
48         return line;
49       }
50     
51       @Override
52       public int getLastLine() {
53         return -1;
54       }
55     
56       @Override
57       public int getFirstCol() {
58         return -1;
59       }
60     
61       @Override
62       public int getLastCol() {
63         return -1;
64       }
65     
66       @Override
67       public int getFirstOffset() {
68         return startOffset;
69       }
70     
71       @Override
72       public int getLastOffset() {
73         return endOffset;
74       }
75     
76       @Override
77       public URL getURL() {
78         return url;
79       }
80     
81       @Override
82       public Reader getReader() throws IOException {
83         return new InputStreamReader(url.openStream());
84       }
85     }
