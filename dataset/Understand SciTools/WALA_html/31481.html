58     /**
59      * Class responsible for mapping JDT type system objects representing types, methods and fields to the corresponding WALA
60      * TypeReferences, MethodReferences and FieldReferences. Used during translation and by clients to help correlate WALA analysis
61      * results to the various AST nodes.
62      * 
63      * In English: keeps a hashtable of WALA "type references", "field references", etc. which describe types, fields, etc. Creates
64      * these from their JDT equivalents and keeps the hashtable linking the two representations.
65      * 
66      * @author rfuhrer
67      */
68     public class JDTIdentityMapper {
69       private final Map<String, TypeReference> fTypeMap = HashMapFactory.make();
70     
71       private final Map<String, FieldReference> fFieldMap = HashMapFactory.make();
72     
73       private final Map<String, MethodReference> fMethodMap = HashMapFactory.make();
74     
75       private final ClassLoaderReference fClassLoaderRef; // TAGALONG
76     
77       private final AST fAst;
78     
79       public JDTIdentityMapper(ClassLoaderReference clr, AST ast) {
80         fClassLoaderRef = clr;
81         fAst = ast;
82       }
83     
84       // TYPES
85     
86       /**
87        * Create (or reuse) a TypeReference for the given JDT Type Binding.<br>
88        * This method canonicalizes the TypeReferences
89        */
90       public TypeReference getTypeRef(ITypeBinding type) {
91         type = JDT2CAstUtils.getErasedType((ITypeBinding) type, fAst); // GENERICS: erasure...
92     
93         if (!fTypeMap.containsKey(type.getKey())) {
94           TypeName typeName = TypeName.string2TypeName(typeToTypeID(type));
95           TypeReference ref = TypeReference.findOrCreate(fClassLoaderRef, typeName);
96     
97           fTypeMap.put(type.getKey(), ref);
98           return ref;
99         }
100         return fTypeMap.get(type.getKey());
101       }
102     
103       /**
104        * Translates the given Polyglot type to a name suitable for use in a DOMO TypeReference (i.e. a bytecode-compliant type name).
105        */
106       public String typeToTypeID(ITypeBinding type) {
107         if (type.isPrimitive())
108           return type.getBinaryName();
109         else if (type.isArray())
110           // arrays' binary names in JDT are like "[Ljava.lang.String;"
111           return type.getBinaryName().replace('.', '/').replace(";", "");
112         else if (type.isLocal() || type.isAnonymous())
113           return anonLocalTypeToTypeID(type);
114         else if (type.isClass() || type.isEnum() || type.isInterface()) // in polyglot interfaces are classes too. not in JDT
115           // class binary names in JDT are like "java.lang.String"
116           return 'L' + type.getBinaryName().replace('.', '/'); // TODO:
117         else if (type.isTypeVariable()) {
118           return typeToTypeID(JDT2CAstUtils.getTypesVariablesBase(type, fAst));
119         }
120         Assertions.UNREACHABLE("typeToTypeID() encountered the type " + type + " that is neither primitive, array, nor class!");
121         return null;
122       }
123     
124       public String anonLocalTypeToTypeID(ITypeBinding type) {
125         String outerTypeID = typeToTypeID(type.getDeclaringClass());
126     
127         String metSelectorName;
128         IMethodBinding metBinding = type.getDeclaringMethod();
129         if (metBinding == null) // anonymous class declared in initializer or static initializer (rare case...)
130           metSelectorName = "<init>";
131         else
132           metSelectorName = getMethodRef(metBinding).getSelector().toString();
133     
134         String shortName = (type.isAnonymous()) ? JDT2CAstUtils.anonTypeName(type) : type.getName();
135     
136         return outerTypeID + '/' + metSelectorName + '/' + shortName;
137       }

139       // FIELDS
140     
141       public FieldReference getFieldRef(IVariableBinding field) {
142         if (!fFieldMap.containsKey(field.getKey())) {
143           // create one
144           ITypeBinding targetType = field.getDeclaringClass();
145           TypeReference targetTypeRef = TypeReference.findOrCreate(fClassLoaderRef, typeToTypeID(targetType));
146           ITypeBinding fieldType = field.getType();
147           TypeReference fieldTypeRef = TypeReference.findOrCreate(fClassLoaderRef, typeToTypeID(fieldType));
148           Atom fieldName = Atom.findOrCreateUnicodeAtom(field.getName());
149           FieldReference ref = FieldReference.findOrCreate(targetTypeRef, fieldName, fieldTypeRef);
150     
151           fFieldMap.put(field.getKey(), ref);
152           return ref;
153         }
154         return fFieldMap.get(field.getKey());
155       }
156     
157       public MethodReference fakeMethodRefNoArgs(String key, String typeID, String metName, String returnTypeID) {
158         if (!fMethodMap.containsKey(key)) {
159           // create one
160           TypeName ownerType = TypeName.string2TypeName(typeID);
161           TypeReference ownerTypeRef = TypeReference.findOrCreate(fClassLoaderRef, ownerType);
162     
163           // FAKE SELECTOR
164           Atom name = Atom.findOrCreateUnicodeAtom(metName);
165           TypeName[] argTypeNames = null;
166           TypeName retTypeName = TypeName.string2TypeName(returnTypeID);
167           Descriptor desc = Descriptor.findOrCreate(argTypeNames, retTypeName);
168           Selector selector = new Selector(name, desc);
169     
170           MethodReference ref = MethodReference.findOrCreate(ownerTypeRef, selector);
171     
172           fMethodMap.put(key, ref);
173           return ref;
174         }
175         return fMethodMap.get(key);
176       }
177     
178       // METHODS
179       public MethodReference getMethodRef(IMethodBinding met) {
180         if (!fMethodMap.containsKey(met.getKey())) {
181           // create one
182           TypeName ownerType = TypeName.string2TypeName(typeToTypeID(met.getDeclaringClass()));
183           TypeReference ownerTypeRef = TypeReference.findOrCreate(fClassLoaderRef, ownerType);
184           MethodReference ref = MethodReference.findOrCreate(ownerTypeRef, selectorForMethod(met));
185     
186           fMethodMap.put(met.getKey(), ref);
187           return ref;
188         }
189         return fMethodMap.get(met.getKey());
190       }
191     
192       private Selector selectorForMethod(IMethodBinding met) {
193         // TODO: have to handle default constructors?
194         // TODO: generics...
195         Atom name = (met.isConstructor()) ? MethodReference.initAtom : Atom.findOrCreateUnicodeAtom(met.getName());
196     
197         TypeName[] argTypeNames = null;
198         ITypeBinding[] formalTypes = met.getParameterTypes();
199     
200         int length = formalTypes.length;
201     
202         // ENUMS: hidden name and ID in constructor
203         if (met.isConstructor() && met.getDeclaringClass().isEnum())
204           length += 2;
205     
206         // Descriptor prefers null to an empty array
207         if (length > 0) {
208           argTypeNames = new TypeName[length];
209           int i = 0;
210           if (met.isConstructor() && met.getDeclaringClass().isEnum()) {
211             argTypeNames[0] = TypeName.string2TypeName(typeToTypeID(fAst.resolveWellKnownType("java.lang.String")));
212             argTypeNames[1] = TypeName.string2TypeName(typeToTypeID(fAst.resolveWellKnownType("int")));
213             i = 2;
214           }
215           for (ITypeBinding argType : formalTypes)
216             argTypeNames[i++] = TypeName.string2TypeName(typeToTypeID(argType));
217         }
218     
219         TypeName retTypeName = TypeName.string2TypeName(typeToTypeID(met.getReturnType()));
220         Descriptor desc = Descriptor.findOrCreate(argTypeNames, retTypeName);
221     
222         return new Selector(name, desc);
223       }
