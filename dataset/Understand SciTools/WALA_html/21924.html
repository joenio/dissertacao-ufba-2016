12     package com.ibm.wala.cast.js.ipa.callgraph.correlations;
13     
14     import java.util.Set;
15     
16     import com.ibm.wala.cast.ir.ssa.AbstractReflectiveGet;
17     import com.ibm.wala.cast.ir.ssa.AbstractReflectivePut;
18     import com.ibm.wala.cast.tree.CAstSourcePositionMap.Position;
19     
20     /**
21      * The most basic form of correlation: an intra-procedurally correlated pair of a dynamic property read
22      * and a dynamic property write.
23      * 
24      * @author mschaefer
25      *
26      */
27     public class ReadWriteCorrelation extends Correlation {
28     	private final AbstractReflectiveGet get;
29     	private final AbstractReflectivePut put;
30     	
31     	public ReadWriteCorrelation(AbstractReflectiveGet get, AbstractReflectivePut put,
32     	                            String indexName, Set<String> flownThroughLocals) {
33     	  super(indexName, flownThroughLocals);
34     		this.get = get;
35     		this.put = put;
36     	}
37     
38     	@Override
39     	public Position getStartPosition(SSASourcePositionMap positions) {
40     		return positions.getPosition(get);
41     	}
42     	
43     	@Override
44     	public Position getEndPosition(SSASourcePositionMap positions) {
45     	  return positions.getPosition(put);
46     	}
47     	
48     	@Override
49     	public String pp(SSASourcePositionMap positions) {
50     		return get + "@" + positions.getPosition(get) + " [" + getIndexName() + "]-> " + put + "@" + positions.getPosition(put);
51     	}
52     	
53     	@Override
54     	public <T> T accept(CorrelationVisitor<T> visitor) {
55     	  return visitor.visitReadWriteCorrelation(this);
56     	}
57     }
