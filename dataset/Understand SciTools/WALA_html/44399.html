133         /**
134          *  Make an intent.
135          *
136          *  @param  pack    The applications package. May be null if unknown - but this may yield an exception
137          *  @param  name    The Action this intent represents
138          *  @param  uri     The URI to match may be null
139          *  @throws IllegalArgumentException If name was null or starts with a dot and pack is null
140          *  @todo   Check Target-Types
141          */
142         public static Intent intent(String pack, String name, String uri) {
143             if ((name == null) || (name.isEmpty())) {
144                 throw new IllegalArgumentException ("name may not be null or empty");
145             }
146             Intent.IntentType type = Intent.IntentType.UNKNOWN_TARGET;
147             
148             if (name.startsWith(".")) {
149                 if ((pack == null) || (pack.isEmpty())) {
150                     throw new IllegalArgumentException("The pack is needed to resolve the full name of " + name + ", but it's empty");
151                 }
152                 name = pack + name;
153                 type = Intent.IntentType.INTERNAL_TARGET;   // TODO Ehhh...
154             } else if (!(name.contains("."))) {
155                 if ((pack == null) || (pack.isEmpty())) {
156                     throw new IllegalArgumentException("The pack is needed to resolve the full name of " + name + ", but it's empty");
157                 }
158                 name = pack + "." + name;
159                 type = Intent.IntentType.INTERNAL_TARGET;   // TODO Ehhh...
160             } else if ((pack != null) && (name.startsWith(pack))) {
161                 type = Intent.IntentType.INTERNAL_TARGET;   // TODO Ehhh...
162             } else if (name.startsWith("android.intent.action")) {
163                 type = Intent.IntentType.STANDARD_ACTION;
164             } 
165             
166             // convert name to the L-Slash format..
167             if ((name.startsWith("L") || name.contains("."))) {
168                 name = StringStuff.deployment2CanonicalTypeString(name);
169             }
170     
171             final Atom action = Atom.findOrCreateAsciiAtom(name);
172             final Intent ret;
173             Atom mUri = null;
174             if (uri != null) {
175                 mUri = Atom.findOrCreateAsciiAtom(uri);
176             }
177             if (type == Intent.IntentType.INTERNAL_TARGET) {
178                 if (uri != null) {
179                     ret = new InternalIntent(action, mUri);
180                 } else {
181                     ret = new InternalIntent(action);
182                 }
183             } else if (type == Intent.IntentType.STANDARD_ACTION) {
184                 if (uri != null) {
185                     ret = new StandardIntent(action, mUri);
186                 } else {
187                     ret = new StandardIntent(action);
188                 }
189             } else {
190                 if (uri != null) {
191                     ret = new StandardIntent(action, mUri);
192                 } else {
193                     ret = new StandardIntent(action);
194                 }
195             }
196     
197             return ret;
198         }

203         public static Intent intent(String fullyQualifiedAction, String uri) {
204             if (fullyQualifiedAction.startsWith(".")) {
205                 throw new IllegalArgumentException("The action " + fullyQualifiedAction + " is not fully qualified! Use " +
206                         " intent(String pack, String name, String uri) to build the intent!");
207             }
208             return intent(null, fullyQualifiedAction, uri);
209         }

211         public static Intent intent(String fullyQualifiedAction) {
212             if (fullyQualifiedAction.startsWith(".")) {
213                 throw new IllegalArgumentException("The action " + fullyQualifiedAction + " is not fully qualified! Use " +
214                         " intent(String pack, String name, null) to build the intent!");
215             }
216             return intent(null, fullyQualifiedAction, null);
217         }
