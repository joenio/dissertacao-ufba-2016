54       /**
55        * lexical access information for some entity scope. used during call graph
56        * construction to handle lexical accesses.
57        */
58       public interface LexicalInformation {
59     
60         /**
61          * names possibly accessed in a nested lexical scope, represented as pairs
62          * (name,nameOfDefiningEntity)
63          */
64         public Pair<String, String>[] getExposedNames();
65     
66         /**
67          * maps each exposed name (via its index in {@link #getExposedNames()}) to
68          * its value number at method exit.
69          */
70         public int[] getExitExposedUses();
71     
72         /**
73          * get a map from exposed name (via its index in {@link #getExposedNames()})
74          * to its value number at the instruction at offset instructionOffset.
75          */
76         public int[] getExposedUses(int instructionOffset);
77     
78         /**
79          * return all value numbers appearing as entries in either
80          * {@link #getExposedUses(int)} or {@link #getExitExposedUses()}
81          */
82         public IntSet getAllExposedUses();
83     
84         /**
85          * return the names of the enclosing methods declaring names that are
86          * lexically accessed by the entity
87          */
88         public String[] getScopingParents();
89     
90         /**
91          * returns true if name may be read in nested lexical scopes but cannot be
92          * written
93          */
94         public boolean isReadOnly(String name);
95     
96         /**
97          * get the name of this entity, as it appears in the definer portion of a
98          * lexical name
99          */
100         public String getScopingName();
101     
102       }

182       /**
183        * Parents of this method with respect to lexical scoping, that is, methods
184        * containing state possibly referenced lexically in this method
185        */
186       public static abstract class LexicalParent {
187         public abstract String getName();
188     
189         public abstract AstMethod getMethod();
190     
191         @Override
192         public int hashCode() {
193           return getName().hashCode() * getMethod().hashCode();
194         }
195     
196         @Override
197         public boolean equals(Object o) {
198           return (o instanceof LexicalParent) && getName().equals(((LexicalParent) o).getName())
199               && getMethod().equals(((LexicalParent) o).getMethod());
200         }
201       };
