25     /**
26      * 0-1-CFA Call graph builder, optimized to not disambiguate instances of "uninteresting" types.
27      */
28     public class ZeroXCFABuilder extends SSAPropagationCallGraphBuilder {
29     
30       public ZeroXCFABuilder(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache, ContextSelector appContextSelector,
31           SSAContextInterpreter appContextInterpreter, int instancePolicy) {
32     
33         super(cha, options, cache, new DefaultPointerKeyFactory());
34     
35         ContextSelector def = new DefaultContextSelector(options, cha);
36         ContextSelector contextSelector = appContextSelector == null ? def : new DelegatingContextSelector(appContextSelector, def);
37         setContextSelector(contextSelector);
38     
39         SSAContextInterpreter c = new DefaultSSAInterpreter(options, cache);
40         c = new DelegatingSSAContextInterpreter(ReflectionContextInterpreter.createReflectionContextInterpreter(cha, options,
41             getAnalysisCache()), c);
42         SSAContextInterpreter contextInterpreter = appContextInterpreter == null ? c : new DelegatingSSAContextInterpreter(
43             appContextInterpreter, c);
44         setContextInterpreter(contextInterpreter);
45     
46         ZeroXInstanceKeys zik = makeInstanceKeys(cha, options, contextInterpreter, instancePolicy);
47         setInstanceKeys(zik);
48       }
49     
50       /**
51        * subclasses can override as desired
52        */
53       protected ZeroXInstanceKeys makeInstanceKeys(IClassHierarchy cha, AnalysisOptions options,
54           SSAContextInterpreter contextInterpreter, int instancePolicy) {
55         ZeroXInstanceKeys zik = new ZeroXInstanceKeys(options, cha, contextInterpreter, instancePolicy);
56         return zik;
57       }
58     
59       /**
60        * @param options options that govern call graph construction
61        * @param cha governing class hierarchy
62        * @param cl classloader that can find WALA resources
63        * @param scope representation of the analysis scope
64        * @param xmlFiles set of Strings that are names of XML files holding bypass logic specifications.
65        * @return a 0-1-Opt-CFA Call Graph Builder.
66        * @throws IllegalArgumentException if options is null
67        * @throws IllegalArgumentException if xmlFiles == null
68        */
69       public static SSAPropagationCallGraphBuilder make(AnalysisOptions options, AnalysisCache cache, IClassHierarchy cha,
70           ClassLoader cl, AnalysisScope scope, String[] xmlFiles, byte instancePolicy) throws IllegalArgumentException {
71     
72         if (xmlFiles == null) {
73           throw new IllegalArgumentException("xmlFiles == null");
74         }
75         if (options == null) {
76           throw new IllegalArgumentException("options is null");
77         }
78         Util.addDefaultSelectors(options, cha);
79         for (int i = 0; i < xmlFiles.length; i++) {
80           Util.addBypassLogic(options, scope, cl, xmlFiles[i], cha);
81         }
82     
83         return new ZeroXCFABuilder(cha, options, cache, null, null, instancePolicy);
84       }
85     
86       public static ZeroXCFABuilder make(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache,
87           ContextSelector appContextSelector, SSAContextInterpreter appContextInterpreter, int instancePolicy) throws IllegalArgumentException {
88         if (options == null) {
89           throw new IllegalArgumentException("options == null");
90         }
91         return new ZeroXCFABuilder(cha, options, cache, appContextSelector, appContextInterpreter, instancePolicy);
92       }
93     
94     }
