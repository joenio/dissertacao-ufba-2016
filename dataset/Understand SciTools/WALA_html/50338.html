18     package com.ibm.wala.sourcepos;
19     
20     import java.io.ByteArrayInputStream;
21     import java.io.DataInputStream;
22     import java.io.IOException;
23     
24     /**
25      * This is the super class of all position attributes.
26      * 
27      * @author Siegfried Weber
28      * @author Juergen Graf <juergen.graf@gmail.com>
29      */
30     abstract class PositionsAttribute {
31     
32       /**
33        * Creates a new instance of PositionsAttribute.
34        * 
35        * @param data
36        *          the byte array containing the attribute
37        * @throws IOException
38        *           An IOException is thrown if the attribute can't be read or
39        *           {@code data} is null.
40        */
41       PositionsAttribute(byte[] data) throws IOException {
42         if (data == null)
43           throw new IOException();
44         DataInputStream in = new DataInputStream(new ByteArrayInputStream(data));
45         readData(in);
46       }
47     
48       /**
49        * Reads the attribute data from the input stream.
50        * 
51        * @param in
52        *          the input stream
53        * @throws IOException
54        *           if the input stream cannot be read.
55        */
56       protected abstract void readData(DataInputStream in) throws IOException;
57     }
