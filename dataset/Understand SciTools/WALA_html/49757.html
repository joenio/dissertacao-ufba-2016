26     /**
27      * Explore the result of an IFDS problem with an SWT viewer and ghostview.
28      */
29     public class IFDSExplorer {
30     
31       /**
32        * absolute path name to invoke dot
33        */
34       protected static String dotExe = null;
35     
36       /**
37        * Absolute path name to invoke viewer
38        */
39       protected static String viewerExe = null;
40     
41       private static final boolean PRINT_DOMAIN = true;
42     
43       public static void setDotExe(String newDotExe) {
44         dotExe = newDotExe;
45       }
46     
47       public static void setGvExe(String newGvExe) {
48         viewerExe = newGvExe;
49       }
50     
51       public static <T, P, F> void viewIFDS(TabulationResult<T, P, F> r, Collection<? extends P> roots) throws WalaException {
52         viewIFDS(r, roots, null);
53       }
54     
55       public static <T, P, F> void viewIFDS(TabulationResult<T, P, F> r, Collection<? extends P> roots, NodeDecorator labels)
56           throws WalaException {
57         Properties p = null;
58         try {
59           p = WalaProperties.loadProperties();
60         } catch (WalaException e) {
61           e.printStackTrace();
62           Assertions.UNREACHABLE();
63         }
64         String scratch = p.getProperty(WalaProperties.OUTPUT_DIR);
65         viewIFDS(r, roots, labels, scratch);
66       }
67     
68       public static <T, P, F> void viewIFDS(TabulationResult<T, P, F> r, Collection<? extends P> roots, NodeDecorator labels,
69           String scratchDirectory) throws WalaException {
70         if (r == null) {
71           throw new IllegalArgumentException("r is null");
72         }
73         assert dotExe != null;
74     
75         // dump the domain to stderr
76         if (PRINT_DOMAIN) {
77           System.err.println("Domain:\n" + r.getProblem().getDomain().toString());
78         }
79     
80         String irFileName = null;
81         switch (DotUtil.getOutputType()) {
82         case PDF:
83           irFileName = "ir.pdf";
84           break;
85         case PS:
86         case EPS:
87           irFileName = "ir.ps";
88           break;
89         case SVG:
90           irFileName = "ir.svg";
91           break;
92         }
93         String outputFile = scratchDirectory + File.separatorChar + irFileName;
94         String dotFile = scratchDirectory + File.separatorChar + "ir.dt";
95     
96         final SWTTreeViewer v = new SWTTreeViewer();
97         Graph<? extends P> g = r.getProblem().getSupergraph().getProcedureGraph();
98         v.setGraphInput(g);
99         v.setBlockInput(true);
100         v.setRootsInput(roots);
101         ViewIFDSLocalAction<T, P, F> action = (labels == null ? new ViewIFDSLocalAction<T, P, F>(v, r, outputFile, dotFile, dotExe,
102             viewerExe) : new ViewIFDSLocalAction<T, P, F>(v, r, outputFile, dotFile, dotExe, viewerExe, labels));
103         v.getPopUpActions().add(action);
104         v.run();
105     
106       }
107     
108       /**
109        * Calls {@link #viewIFDS(TabulationResult)} with roots computed by {@link InferGraphRoots}.
110        */
111       public static <T, P, F> void viewIFDS(TabulationResult<T, P, F> r) throws WalaException {
112         if (r == null) {
113           throw new IllegalArgumentException("null r");
114         }
115         Collection<? extends P> roots = InferGraphRoots.inferRoots(r.getProblem().getSupergraph().getProcedureGraph());
116         viewIFDS(r, roots);
117       }
118     
119       public static String getDotExe() {
120         return dotExe;
121       }
122     
123       public static String getGvExe() {
124         return viewerExe;
125       }

127     }
