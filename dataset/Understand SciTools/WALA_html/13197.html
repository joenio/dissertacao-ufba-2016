41     /**
42      * A SourceModuleTranslator whose implementation of loadAllSources() uses a suitably
43      * configured Polyglot-based compiler to generate WALA IR for the sources in the
44      * compile-time classpath.
45      * @author rfuhrer
46      */
47     public class PolyglotSourceModuleTranslator implements SourceModuleTranslator {
48       private final ExtensionInfo fExtInfo;
49     
50       /**
51        * A client-supplied ClassLoaderReference to identify the innermost class loader
52        * to use when populating the classpath with loader Modules.
53        */
54       private final ClassLoaderReference fSearchPathStart;
55     
56       protected String fClassPath;
57     
58       protected String fSourcePath;
59     
60       public PolyglotSourceModuleTranslator(AnalysisScope scope, IRTranslatorExtension extInfo,
61           PolyglotSourceLoaderImpl sourceLoader, ClassLoaderReference searchPathStart) {
62         fSearchPathStart = searchPathStart;
63         fExtInfo= (ExtensionInfo) extInfo;
64     	computeClassPath(scope);
65     	computeSourcePath(scope);
66     	extInfo.setSourceLoader(sourceLoader);	
67       }
68     
69       protected void computeClassPath(AnalysisScope scope) {
70         StringBuilder sb= new StringBuilder();
71         ClassLoaderReference cl= fSearchPathStart;
72     
73         while (cl != null) {
74           addModulesForLoader(scope, cl, sb);
75           cl= cl.getParent();
76         }
77         fClassPath= sb.toString();
78       }
79     
80       protected void addModulesForLoader(AnalysisScope scope, ClassLoaderReference cl, StringBuilder sb) {
81         for(Module m: scope.getModules(cl)) {
82           if (sb.length() > 0)
83             sb.append(File.pathSeparator);
84     
85           if (m instanceof JarFileModule) {
86             JarFileModule jarFileModule= (JarFileModule) m;
87     
88             sb.append(jarFileModule.getAbsolutePath());
89           } else if (m instanceof DirectoryTreeModule) {
90             DirectoryTreeModule directoryTreeModule= (DirectoryTreeModule) m;
91     
92             sb.append(directoryTreeModule.getPath());
93           } else if (m instanceof FileModule) {
94             // do nothing
95           } else
96             Assertions.UNREACHABLE("Module entry is neither jar file nor directory");
97         }
98       }
99     
100       protected void computeSourcePath(AnalysisScope scope) {
101         // Ordinarily, the source files to process are specified using absolute paths (see
102         // loadAllSources()), so the source path won't be needed. However, derived classes
103         // may choose to place entries on the source path to resolve source entities that
104         // were not explicitly specified. E.g., the X10 1.7 runtime jar contains source
105         // code, since class files don't contain all of the necessary type information.
106         // Given all of the above, the following value will either be correct or do no
107         // harm (even if the source path isn't used).
108         fSourcePath = ".";
109       }
110     
111       @Override
112       public void loadAllSources(Set<ModuleEntry> modules) {
113         Options opts= fExtInfo.getOptions();
114         opts.assertions = true;
115     
116         try {
117           opts.parseCommandLine(new String[] { "-noserial", "-cp", fClassPath, "-sourcepath", fSourcePath }, new HashSet<String>());
118         } catch (UsageError e) {
119           // Assertions.UNREACHABLE("Error parsing classpath spec???");
120         }
121     
122         Compiler compiler= new Compiler(fExtInfo);
123         List<Source> streams= new ArrayList<Source>();
124     
125         Globals.initialize(compiler);//PORT1.7 Must initialize before actually calling compiler
126     
127         // N.B.: 'modules' is a flat set of source file ModuleEntry's.
128         for(Iterator<ModuleEntry> it= modules.iterator(); it.hasNext(); ) {
129           SourceFileModule entry= (SourceFileModule) it.next();
130     
131           Assertions.productionAssertion(entry.isSourceFile());
132     
133           if (skipSourceFile(entry)) {
134             continue;
135           }
136     
137           try {
138             ModuleSource srcStream= new ModuleSource(entry);
139     
140             streams.add(srcStream);
141           } catch (IOException e) {
142             compiler.errorQueue().enqueue(new ErrorInfo(ErrorInfo.IO_ERROR, "Unable to open source file '" + entry.getName() + "'", Position.COMPILER_GENERATED));
143           }
144         }
145         compiler.compile(streams);
146         // At this point, WALA now "knows" about all the source-originated stuff
147       }
148     
149       /**
150        * @return true if the given source file module should not be processed,
151        * e.g. because it is generated on behalf of some upstream source.
152        */
153       protected boolean skipSourceFile(SourceFileModule entry) {
154         return false;
155       }
156     }
