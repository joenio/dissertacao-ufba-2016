43       public static String getJavaJar(AnalysisScope javaScope) throws IOException {
44         Module javaJar = javaScope.getModules(javaScope.getApplicationLoader()).iterator().next();
45         if (javaJar instanceof JarFileModule) {
46           String javaJarPath = ((JarFileModule)javaJar).getAbsolutePath();
47           return javaJarPath;
48         } else {
49           assert javaJar instanceof NestedJarFileModule : javaJar;
50           File F = File.createTempFile("android", ".jar");
51           //F.deleteOnExit();
52           System.err.println(F.getAbsolutePath());
53           TemporaryFile.streamToFile(F, ((NestedJarFileModule)javaJar).getNestedContents());
54           return F.getAbsolutePath();
55         }
56       }

77       public static URI[] androidLibs() {
78         List<URI> libs = new ArrayList<URI>();
79         if (walaProperties != null && walaProperties.getProperty(ANDROID_RT_DEX_DIR) != null) {
80           for(File lib : new File(walaProperties.getProperty(ANDROID_RT_DEX_DIR)).listFiles(new FilenameFilter() {
81             @Override
82             public boolean accept(File dir, String name) {
83               return name.endsWith("dex") || name.endsWith("jar") || name.endsWith("apk");
84             } 
85           })) {
86             libs.add(lib.toURI());
87           }
88         } else {
89           assert "Dalvik".equals(System.getProperty("java.vm.name"));
90           for(File f : new File("/system/framework/").listFiles(new FileFilter() {
91             @Override
92             public boolean accept(File pathname) {
93               String name = pathname.getName();
94               return 
95                   (name.startsWith("core") || name.startsWith("framework")) && 
96                   (name.endsWith("jar") || name.endsWith("apk"));
97             } 
98           })) 
99           {
100             System.out.println("adding " + f);
101             libs.add(f.toURI());
102           }
103         }
104         return libs.toArray(new URI[ libs.size() ]);
105       }

107       public static AnalysisScope makeDalvikScope(URI[] androidLibs, File androidAPIJar, String dexFileName) throws IOException {
108         if (androidLibs != null) {
109           return AndroidAnalysisScope.setUpAndroidAnalysisScope(
110             new File(dexFileName).toURI(), 
111             CallGraphTestUtil.REGRESSION_EXCLUSIONS,
112             CallGraphTestUtil.class.getClassLoader(),
113             androidLibs);
114           
115         } else {
116           AnalysisScope scope = AndroidAnalysisScope.setUpAndroidAnalysisScope(
117             new File(dexFileName).toURI(), 
118             CallGraphTestUtil.REGRESSION_EXCLUSIONS,
119             CallGraphTestUtil.class.getClassLoader());
120           
121           if (androidAPIJar != null) {
122             scope.addToScope(ClassLoaderReference.Primordial, new JarFileModule(new JarFile(androidAPIJar)));
123           }
124           
125           return scope;
126         }
127       }
