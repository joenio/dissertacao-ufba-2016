11     package com.ibm.wala.analysis.typeInference;
12     
13     import com.ibm.wala.classLoader.IClass;
14     import com.ibm.wala.types.TypeReference;
15     import com.ibm.wala.util.debug.Assertions;
16     
17     /**
18      * Represents a single concrete type.
19      */
20     public class PointType extends TypeAbstraction {
21     
22       private final IClass type;
23     
24       /**
25        * @throws IllegalArgumentException if type is null
26        */
27       public PointType(IClass type) {
28         if (type == null) {
29           throw new IllegalArgumentException("type is null");
30         }
31         this.type = type;
32         assert type.getReference().isReferenceType();
33       }
34     
35       @Override
36       public TypeAbstraction meet(TypeAbstraction rhs) {
37         if (rhs == TOP) {
38           return this;
39         } else {
40           if (rhs instanceof PointType) {
41             PointType other = (PointType) rhs;
42             if (type.equals(other.type)) {
43               return this;
44             } else if (type.isArrayClass() || other.type.isArrayClass()) {
45               // give up on arrays. We don't care anyway.
46               return new ConeType(type.getClassHierarchy().getRootClass());
47             } else {
48               return new ConeType(type.getClassHierarchy().getLeastCommonSuperclass(this.type, other.type));
49             }
50           } else if (rhs instanceof ConeType) {
51             ConeType other = (ConeType) rhs;
52             if (type.equals(other.getType())) {
53               // "this" and the cone type have the same underlying type, return the cone type
54               return other;
55             }
56             TypeReference T = other.getType().getReference();
57             if (type.isArrayClass() || T.isArrayType()) {
58               // give up on arrays. We don't care anyway.
59               return new ConeType(type.getClassHierarchy().getRootClass());
60             }
61             IClass typeKlass = type;
62             if (type.getClassHierarchy().isSubclassOf(typeKlass, other.getType())) {
63               return other;
64             } else if (other.isInterface()) {
65               if (type.getClassHierarchy().implementsInterface(typeKlass, other.getType())) {
66                 return other;
67               }
68             }
69             // if we get here, we need to do cha-based superclass and return a cone.
70             // TODO: avoid the allocation
71             return other.meet(new ConeType(this.getType()));
72           } else {
73             Assertions.UNREACHABLE("Unexpected type: " + rhs.getClass());
74             return null;
75           }
76         }
77       }
78     
79       @Override
80       public String toString() {
81         return "point: " + type.toString();
82       }
83     
84       @Override
85       public IClass getType() {
86         return type;
87       }
88     
89       @Override
90       public TypeReference getTypeReference() {
91         return type.getReference();
92       }
93     
94       @Override
95       public boolean equals(Object obj) {
96         if (!(obj instanceof PointType)) {
97           return false;
98         }
99         PointType other = (PointType) obj;
100         if (!type.getClassHierarchy().equals(other.type.getClassHierarchy())) {
101           Assertions.UNREACHABLE("different chas " + this + " " + other);
102         }
103         return type.equals(other.type);
104       }
105     
106       @Override
107       public int hashCode() {
108         return 37 * type.hashCode();
109       }
110     
111       public boolean isArrayType() {
112         return getType().isArrayClass();
113       }
114     
115       public IClass getIClass() {
116         return type;
117       }
118     }
