11     package com.ibm.wala.cast.ir.ssa;
12     
13     import java.util.Collection;
14     import java.util.Collections;
15     
16     import com.ibm.wala.ssa.SSAAbstractBinaryInstruction;
17     import com.ibm.wala.ssa.SSAInstruction;
18     import com.ibm.wala.ssa.SSAInstructionFactory;
19     import com.ibm.wala.ssa.SymbolTable;
20     import com.ibm.wala.types.TypeReference;
21     
22     /**
23      * This instruction represents iterating through the properties of its receiver object. The use represents an object,
24      * and the l-value represents a boolean indicating whether the object has more properties. 
25      * 
26      * Iterating across the fields or properties of a given object is a common idiom in scripting languages, which is why
27      * the IR has first-class support for it.
28      * 
29      * @author Julian Dolby (dolby@us.ibm.com)
30      */
31     public class EachElementHasNextInstruction extends SSAAbstractBinaryInstruction {
32     
33       public EachElementHasNextInstruction(int iindex, int lValue, int objectRef, int previousPropVal) {
34         super(iindex, lValue, objectRef, previousPropVal);
35       }
36     
37       @Override
38       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
39         return ((AstInstructionFactory)insts).EachElementHasNextInstruction(iindex, (defs == null) ? getDef(0) : defs[0], (uses == null) ? getUse(0) : uses[0], (uses == null) ? getUse(1) : uses[1]);
40       }
41     
42       @Override
43       public String toString(SymbolTable symbolTable) {
44         return getValueString(symbolTable, getDef(0)) + " = has next property: " + getValueString(symbolTable, getUse(0));
45       }
46     
47       @Override
48       public void visit(IVisitor v) {
49         ((AstInstructionVisitor) v).visitEachElementHasNext(this);
50       }
51     
52       @Override
53       public Collection<TypeReference> getExceptionTypes() {
54         return Collections.emptySet();
55       }
56     
57       @Override
58       public boolean isFallThrough() {
59         return true;
60       }
61     }
