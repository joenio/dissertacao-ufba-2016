32      */
33     public final class AnalysisUtil {
34     
35       private AnalysisUtil() {
36         throw new IllegalStateException("No instances of this class allowed.");
37       }
38       
39       /**
40        * Checks if a node is FakeRoot
41        * 
42        * @param node
43        *          the node to check
44        * @return true if node is FakeRoot
45        */
46       public static boolean isFakeRoot(CGNode node) {
47         return (node.getMethod().getName().equals(FakeRootMethod.rootMethod.getName()));
48       }
49       
50       /**
51        * Returns an array of <code>int</code> with the parameter's var nums of the
52        * invoked method in <code>invokeInstruction</code>.
53        * 
54        * @param invokeInstruction
55        *          The instruction that invokes the method.
56        * @return an array of <code>int</code> with all parameter's var nums
57        *         including the this pointer.
58        */
59       public static int[] getParameterNumbers(SSAAbstractInvokeInstruction invokeInstruction) {
60         final int number = invokeInstruction.getNumberOfParameters();
61         final int[] parameterNumbers = new int[number];
62         assert (parameterNumbers.length == invokeInstruction.getNumberOfUses());
63     
64         for (int i = 0; i < parameterNumbers.length; i++) {
65           parameterNumbers[i] = invokeInstruction.getUse(i);
66         }
67     
68         return parameterNumbers;
69       }
70     
71       /**
72        * Returns a Set of all blocks that invoke another method.
73        * 
74        * @param cfg
75        *          The Control Flow Graph to analyze
76        * @return a Set of all blocks that contain an invoke
77        */
78       public static Set<IExplodedBasicBlock> extractInvokeBlocks(final ControlFlowGraph<SSAInstruction, IExplodedBasicBlock> cfg) {
79         final HashSet<IExplodedBasicBlock> invokeBlocks = new HashSet<IExplodedBasicBlock>();
80     
81         for (final IExplodedBasicBlock block : cfg) {
82           if (block.getInstruction() instanceof SSAAbstractInvokeInstruction) {
83             invokeBlocks.add(block);
84           }
85         }
86     
87         return invokeBlocks;
88       }

90     }
