11     package com.ibm.wala.cast.loader;
12     
13     import com.ibm.wala.classLoader.ClassLoaderFactory;
14     import com.ibm.wala.classLoader.IClassLoader;
15     import com.ibm.wala.ipa.callgraph.AnalysisScope;
16     import com.ibm.wala.ipa.cha.IClassHierarchy;
17     import com.ibm.wala.types.ClassLoaderReference;
18     import com.ibm.wala.util.debug.Assertions;
19     
20     /**
21      * Abstract {@link ClassLoaderFactory} for languages modeled as having a single
22      * class loader. Subclasses provide the logic to create the classloader.
23      */
24     public abstract class SingleClassLoaderFactory implements ClassLoaderFactory {
25     
26       /**
27        * for caching the class loader, so we don't initialize more than once
28        */
29       private IClassLoader THE_LOADER = null;
30     
31       @Override
32       public IClassLoader getLoader(ClassLoaderReference classLoaderReference, IClassHierarchy cha, AnalysisScope scope) {
33         if (THE_LOADER == null) {
34           THE_LOADER = makeTheLoader(cha);
35           try {
36             THE_LOADER.init(scope.getModules(getTheReference()));
37           } catch (java.io.IOException e) {
38             Assertions.UNREACHABLE();
39           }
40         }
41     
42         assert classLoaderReference.equals(getTheReference());
43     
44         return THE_LOADER;
45       }
46     
47       public IClassLoader getTheLoader() {
48         return THE_LOADER;
49       }
50     
51       /**
52        * get the reference to the single class loader for the language
53        */
54       public abstract ClassLoaderReference getTheReference();
55     
56       protected abstract IClassLoader makeTheLoader(IClassHierarchy cha);
57     
58     }
