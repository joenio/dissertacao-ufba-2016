89     /**
90      * Specialized pointer analysis constraint generation for JavaScript.
91      * 
92      * <h2>Global object handling</h2>
93      * 
94      * The global object is represented by a {@link GlobalObjectKey} stored in
95      * {@link #globalObject}. {@link AstGlobalRead} and {@link AstGlobalWrite}
96      * instructions are treated as accessing properties of the global object; see
97      * {@link JSConstraintVisitor#visitAstGlobalRead(AstGlobalRead)},
98      * {@link JSConstraintVisitor#visitAstGlobalWrite(AstGlobalWrite)}, and
99      * {@link JSImplicitPointsToSetVisitor#visitAstGlobalRead(AstGlobalRead)}.
100      * Finally, we need to represent direct flow of the global object to handle
101      * receiver argument semantics (see
102      * {@link org.mozilla.javascript.RhinoToAstTranslator}). To do so, we create a
103      * reference to a global named {@link #GLOBAL_OBJ_VAR_NAME}, which is handled
104      * specially in {@link JSConstraintVisitor#visitAstGlobalRead(AstGlobalRead)}.
105      */
106     public class JSSSAPropagationCallGraphBuilder extends AstSSAPropagationCallGraphBuilder {

362             JSPointerAnalysisImpl jsAnalysis = (JSPointerAnalysisImpl) analysis;
363             FieldReference field = makeNonGlobalFieldReference(instruction.getDeclaredField());
364             assert !directGlobalObjectRef(field);
365             IField f = jsAnalysis.builder.getCallGraph().getClassHierarchy().resolveField(field);
366             assert f != null;
367             MutableSparseIntSet S = MutableSparseIntSet.makeEmpty();
368             InstanceKey globalObj = ((AstSSAPropagationCallGraphBuilder) jsAnalysis.builder).getGlobalObject(JavaScriptTypes.jsName);
369             PointerKey fkey = analysis.getHeapModel().getPointerKeyForInstanceField(globalObj, f);
370             if (fkey != null) {
371               OrdinalSet pointees = analysis.getPointsToSet(fkey);
372               IntSet set = pointees.getBackingSet();
373               if (set != null) {
374                 S.addAll(set);
375               }
376             }
377             pointsToSet = new OrdinalSet<InstanceKey>(S, analysis.getInstanceKeyMapping());
378           }

485         @Override
486         public void visitAstGlobalRead(AstGlobalRead instruction) {
487           int lval = instruction.getDef();
488           FieldReference field = makeNonGlobalFieldReference(instruction.getDeclaredField());
489           PointerKey def = getPointerKeyForLocal(lval);
490           assert def != null;
491           IField f = getClassHierarchy().resolveField(field);
492           assert f != null : "could not resolve referenced global " + field;
493           if (hasNoInterestingUses(lval)) {
494             system.recordImplicitPointsToSet(def);
495           } else {
496             InstanceKey globalObj = getBuilder().getGlobalObject(JavaScriptTypes.jsName);
497             if (directGlobalObjectRef(field)) {
498               // points-to set is just the global object
499               system.newConstraint(def, globalObj);
500             } else {
501               system.findOrCreateIndexForInstanceKey(globalObj);
502               PointerKey p = getPointerKeyForInstanceField(globalObj, f);
503               system.newConstraint(def, assignOperator, p);
504             }
505           }
506     
507         }

509         @Override
510         public void visitAstGlobalWrite(AstGlobalWrite instruction) {
511           int rval = instruction.getVal();
512           FieldReference field = makeNonGlobalFieldReference(instruction.getDeclaredField());
513           IField f = getClassHierarchy().resolveField(field);
514           assert f != null : "could not resolve referenced global " + field;
515           assert !f.getFieldTypeReference().isPrimitiveType();
516           InstanceKey globalObj = getBuilder().getGlobalObject(JavaScriptTypes.jsName);
517           system.findOrCreateIndexForInstanceKey(globalObj);
518           PointerKey p = getPointerKeyForInstanceField(globalObj, f);
519     
520           PointerKey rvalKey = getPointerKeyForLocal(rval);
521           if (contentsAreInvariant(symbolTable, du, rval)) {
522             system.recordImplicitPointsToSet(rvalKey);
523             InstanceKey[] ik = getInvariantContents(rval);
524             for (int i = 0; i < ik.length; i++) {
525               system.newConstraint(p, ik[i]);
526             }
527           } else {
528             system.newConstraint(p, assignOperator, rvalKey);
529           }
530     
531         }

585         private void handleJavascriptDispatch(final JavaScriptInvoke instruction, final InstanceKey receiverType) {
586           int functionVn = instruction.getFunction();
587     
588           ReflectedFieldAction fieldDispatchAction = new ReflectedFieldAction() {
589             @Override
590             public void action(final AbstractFieldPointerKey fieldKey) {
591                 class FieldValueDispatch extends UnaryOperator<PointsToSetVariable> {
592                   private JavaScriptInvoke getInstruction() { return instruction; }
593                   private InstanceKey getReceiver() { return receiverType; }
594                   private AbstractFieldPointerKey getProperty() { return fieldKey; }
595                   private CGNode getNode() { return node; }
596                   
597                   @Override
598                   public byte evaluate(PointsToSetVariable lhs, PointsToSetVariable ptrs) {
599                     if (ptrs.getValue() != null) {
600                       ptrs.getValue().foreach(new IntSetAction() {
601                         @Override
602                         public void act(int x) {
603                           final InstanceKey functionObj = system.getInstanceKey(x);
604                           visitInvokeInternal(instruction, new DefaultInvariantComputer() {
605                             @Override
606                             public InstanceKey[][] computeInvariantParameters(SSAAbstractInvokeInstruction call) {
607                               InstanceKey[][] x = super.computeInvariantParameters(call);
608                               if (x == null) {
609                                 x = new InstanceKey[call.getNumberOfUses()][];
610                               }
611                               x[0] = new InstanceKey[]{ functionObj };
612                               x[1] = new InstanceKey[]{ receiverType };
613                               return x;
614                             }
615                           });
616                         } 
617                       });
618                     }
619                     return NOT_CHANGED;
620                   }
621                   @Override
622                   public int hashCode() {
623                     return instruction.hashCode() * fieldKey.hashCode() * receiverType.hashCode();
624                   }
625                   @Override
626                   public boolean equals(Object o) {
627                     return o instanceof FieldValueDispatch &&
628                     ((FieldValueDispatch)o).getNode().equals(node) &&

632                   }

636                   } 
637                 };
638               
639                 system.newSideEffect(new FieldValueDispatch(), fieldKey);
640             }

642             public void dump(AbstractFieldPointerKey fieldKey, boolean constObj, boolean constProp) {
643               System.err.println("dispatch to " + receiverType + "." + fieldKey + " for " + instruction);
644             }
645           };
646     
647           TransitivePrototypeKey prototypeObjs = new TransitivePrototypeKey(receiverType);
648           InstanceKey[] objKeys = new InstanceKey[]{ receiverType };
649           if (contentsAreInvariant(symbolTable, du, functionVn)) {

651             newFieldOperationObjectAndFieldConstant(true, fieldDispatchAction, objKeys, fieldsKeys);

653           } else {

655             newFieldOperationOnlyObjectConstant(true, fieldDispatchAction, fieldKey, objKeys);
656             newFieldFullOperation(true, fieldDispatchAction, prototypeObjs, fieldKey);
657           }
658         }

720         private void handleBinaryOp(final SSABinaryOpInstruction instruction, final CGNode node, final SymbolTable symbolTable,
721             final DefUse du) {
722           int lval = instruction.getDef(0);
723           PointerKey lk = getPointerKeyForLocal(lval);
724           final PointsToSetVariable lv = system.findOrCreatePointsToSet(lk);
725     
726           final int arg1 = instruction.getUse(0);
727           final int arg2 = instruction.getUse(1);
728     
729           class BinaryOperator extends AbstractOperator<PointsToSetVariable> {
730

732               return node;

751               if (o instanceof BinaryOperator) {
752                 BinaryOperator op = (BinaryOperator) o;
753                 return op.getNode().equals(getNode()) && op.getInstruction().getUse(0) == getInstruction().getUse(0)
754                     && op.getInstruction().getUse(1) == getInstruction().getUse(1)
755                     && op.getInstruction().getDef(0) == getInstruction().getDef(0);
756               } else {
757                 return false;

761             private InstanceKey[] getInstancesArray(int vn) {
762               if (contentsAreInvariant(symbolTable, du, vn)) {
763                 return getInvariantContents(vn);
764               } else {
765                 PointsToSetVariable v = system.findOrCreatePointsToSet(getPointerKeyForLocal(vn));
766                 if (v.getValue() == null || v.size() == 0) {
767                   return new InstanceKey[0];
768                 } else {
769                   final Set<InstanceKey> temp = HashSetFactory.make();
770                   v.getValue().foreach(new IntSetAction() {
771                     @Override
772                     public void act(int keyIndex) {
773                       temp.add(system.getInstanceKey(keyIndex));
774                     }
775                   });
776     
777                   return temp.toArray(new InstanceKey[temp.size()]);
778                 }
779               }
780             }

782             private boolean isStringConstant(InstanceKey k) {
783               return (k instanceof ConstantKey) && k.getConcreteType().getReference().equals(JavaScriptTypes.String);
784             }
785     
786             private boolean addKey(InstanceKey k) {
787               int n = system.findOrCreateIndexForInstanceKey(k);
788               if (!lv.contains(n)) {
789                 lv.add(n);
790                 return true;
791               } else {
792                 return false;
793               }
794             }

796             @Override
797             public byte evaluate(PointsToSetVariable lhs, final PointsToSetVariable[] rhs) {
798               boolean doDefault = false;
799               byte changed = NOT_CHANGED;
800     
801               InstanceKey[] iks1 = getInstancesArray(arg1);
802               InstanceKey[] iks2 = getInstancesArray(arg2);
803     
804               if ((instruction.getOperator() == BinaryOpInstruction.Operator.ADD) && (getOptions().getTraceStringConstants())) {
805                 for (int i = 0; i < iks1.length; i++) {
806                   if (isStringConstant(iks1[i])) {
807                     for (int j = 0; j < iks2.length; j++) {
808                       if (isStringConstant(iks2[j])) {
809                         try {
810                           MonitorUtil.throwExceptionIfCanceled(builder.monitor);
811                         } catch (CancelException e) {
812                           throw new CancelRuntimeException(e);
813                         }
814                         String v1 = (String) ((ConstantKey) iks1[i]).getValue();
815                         String v2 = (String) ((ConstantKey) iks2[j]).getValue();
816                         if (v1.indexOf(v2) == -1 && v2.indexOf(v1) == -1) {
817                           InstanceKey lvalKey = getInstanceKeyForConstant(v1 + v2);
818                           if (addKey(lvalKey)) {
819                             changed = CHANGED;
820                           }
821                         } else {
822                           doDefault = true;
823                         }
824                       } else {
825                         doDefault = true;
826                       }
827                     }
828                   } else {
829                     doDefault = true;
830                   }
831                 }
832               } else {
833                 doDefault = true;
834               }
835     
836               if (doDefault) {
837                   for (int i = 0; i < iks1.length; i++) {
838                     for (int j = 0; j < iks2.length; j++) {
839                       try {
840                         MonitorUtil.throwExceptionIfCanceled(builder.monitor);
841                       } catch (CancelException e) {
842                         throw new CancelRuntimeException(e);
843                       }
844                       if (handleBinaryOperatorArgs(iks1[i], iks2[j])) {
845                         changed = CHANGED;
846                       }
847                     }
848                   }
849               }
850               
851               return changed;
852             }

854             private boolean isNumberType(Language l, TypeReference t) {
855               return l.isDoubleType(t)||l.isFloatType(t)||l.isIntType(t)||l.isLongType(t);
856             }

858             protected boolean handleBinaryOperatorArgs(InstanceKey left, InstanceKey right) {
859               Language l = node.getMethod().getDeclaringClass().getClassLoader().getLanguage();
860               if (l.isStringType(left.getConcreteType().getReference()) || l.isStringType(right.getConcreteType().getReference())) {
861                 return addKey(new ConcreteTypeKey(node.getClassHierarchy().lookupClass(l.getStringType())));
862               } else if (isNumberType(l, left.getConcreteType().getReference()) && isNumberType(l, right.getConcreteType().getReference())) {
863                 if (left instanceof ConstantKey && right instanceof ConstantKey) {
864                   return addKey(new ConcreteTypeKey(node.getClassHierarchy().lookupClass(JavaScriptTypes.Number)));
865                 } else {
866                   return addKey(left) || addKey(right);
867                 }
868               } else {
869                 return false;
870               }
871              }
872     
873           }
874     
875           BinaryOperator B = new BinaryOperator();
876           if (contentsAreInvariant(symbolTable, du, arg1)) {
877             if (contentsAreInvariant(symbolTable, du, arg2)) {
878               B.evaluate(null, null);
879             } else {
880               system.newConstraint(lk, B, getPointerKeyForLocal(arg2));
881             }
882           } else {
883             PointerKey k1 = getPointerKeyForLocal(arg1);
884             if (contentsAreInvariant(symbolTable, du, arg2)) {
885               system.newConstraint(lk, B, k1);
886             } else {
887               system.newConstraint(lk, B, k1, getPointerKeyForLocal(arg2));
888             }
889           }
890         }

992       protected void processCallingConstraints(CGNode caller, SSAAbstractInvokeInstruction instruction, CGNode target,
993           InstanceKey[][] constParams, PointerKey uniqueCatchKey) {
994         processCallingConstraintsInternal(this, caller, instruction, target, constParams, uniqueCatchKey);

997       public static void processCallingConstraintsInternal(AstSSAPropagationCallGraphBuilder builder, CGNode caller, SSAAbstractInvokeInstruction instruction, CGNode target,
998           InstanceKey[][] constParams, PointerKey uniqueCatchKey) {
999         IR sourceIR = builder.getCFAContextInterpreter().getIR(caller);
1000         SymbolTable sourceST = sourceIR.getSymbolTable();
1001     
1002         IR targetIR = builder.getCFAContextInterpreter().getIR(target);
1003         SymbolTable targetST = targetIR.getSymbolTable();
1004     
1005         JSConstraintVisitor targetVisitor = null;
1006         int av = -1;
1007         for (int v = 0; v <= targetST.getMaxValueNumber(); v++) {
1008           String[] vns = targetIR.getLocalNames(1, v);
1009           for (int n = 0; vns != null && n < vns.length; n++) {
1010             if ("arguments".equals(vns[n])) {
1011               av = v;
1012               targetVisitor = (JSConstraintVisitor) builder.makeVisitor(target);
1013               break;
1014             }
1015           }
1016         }
1017     
1018         int paramCount = targetST.getParameterValueNumbers().length;
1019         int argCount = instruction.getNumberOfParameters();
1020         
1021         // the first two arguments are the function object and the receiver, neither of which
1022         // should become part of the arguments array
1023         int num_pseudoargs = 2;
1024     
1025         // pass actual arguments to formals in the normal way
1026         for (int i = 0; i < Math.min(paramCount, argCount); i++) {
1027           InstanceKey[] fn = new InstanceKey[] { builder.getInstanceKeyForConstant(JavaScriptTypes.String, ""+(i-num_pseudoargs)) };
1028           PointerKey F = builder.getTargetPointerKey(target, i);
1029     
1030           if (constParams != null && constParams[i] != null) {
1031             for (int j = 0; j < constParams[i].length; j++) {
1032               builder.getSystem().newConstraint(F, constParams[i][j]);
1033             }
1034     
1035             if (av != -1 && i >= num_pseudoargs) {
1036               targetVisitor.newFieldWrite(target, av, fn, constParams[i]);
1037             }
1038             
1039           } else {
1040             PointerKey A = builder.getPointerKeyForLocal(caller, instruction.getUse(i));
1041             builder.getSystem().newConstraint(F, (F instanceof FilteredPointerKey) ? builder.filterOperator : assignOperator, A);
1042     
1043             if (av != -1 && i >= num_pseudoargs) {
1044               targetVisitor.newFieldWrite(target, av, fn, F);
1045             }
1046           }
1047         }
1048     
1049         // extra actual arguments get assigned into the ``arguments'' object
1050         if (paramCount < argCount) {
1051           if (av != -1) {
1052             for (int i = paramCount; i < argCount; i++) {
1053               InstanceKey[] fn = new InstanceKey[] { builder.getInstanceKeyForConstant(JavaScriptTypes.String, ""+(i-num_pseudoargs)) };
1054               if (constParams != null && constParams[i] != null && i >= num_pseudoargs) {
1055                   targetVisitor.newFieldWrite(target, av, fn, constParams[i]);
1056               } else if(i >= num_pseudoargs) {
1057                 PointerKey A = builder.getPointerKeyForLocal(caller, instruction.getUse(i));
1058                 targetVisitor.newFieldWrite(target, av, fn, A);
1059               }
1060             }
1061           }
1062         }
1063     
1064         // extra formal parameters get null (extra args are ignored here)
1065         else if (argCount < paramCount) {
1066           int nullvn = sourceST.getNullConstant();
1067           DefUse sourceDU = builder.getCFAContextInterpreter().getDU(caller);
1068           InstanceKey[] nullkeys = builder.getInvariantContents(sourceST, sourceDU, caller, nullvn, builder);
1069           for (int i = argCount; i < paramCount; i++) {
1070             PointerKey F = builder.getPointerKeyForLocal(target, targetST.getParameter(i));
1071             for (int k = 0; k < nullkeys.length; k++) {
1072               builder.getSystem().newConstraint(F, nullkeys[k]);
1073             }
1074           }
1075         }
1076     
1077         // write `length' in argument objects
1078         if (av != -1) {
1079           InstanceKey[] svn = new InstanceKey[] { builder.getInstanceKeyForConstant(JavaScriptTypes.Number, argCount-1) };
1080           InstanceKey[] lnv = new InstanceKey[] { builder.getInstanceKeyForConstant(JavaScriptTypes.String, "length") };
1081           targetVisitor.newFieldWrite(target, av, lnv, svn);
1082         }
1083     
1084         // return values
1085         if (instruction.getDef(0) != -1) {
1086           PointerKey RF = builder.getPointerKeyForReturnValue(target);
1087           PointerKey RA = builder.getPointerKeyForLocal(caller, instruction.getDef(0));
1088           builder.getSystem().newConstraint(RA, assignOperator, RF);
1089         }
1090     
1091         PointerKey EF = builder.getPointerKeyForExceptionalReturnValue(target);
1092         if (SHORT_CIRCUIT_SINGLE_USES && uniqueCatchKey != null) {
1093           // e has exactly one use. so, represent e implicitly
1094           builder.getSystem().newConstraint(uniqueCatchKey, assignOperator, EF);
1095         } else {
1096           PointerKey EA = builder.getPointerKeyForLocal(caller, instruction.getDef(1));
1097           builder.getSystem().newConstraint(EA, assignOperator, EF);
1098         }  
1099       }
