18     package com.ibm.wala.sourcepos;
19     
20     import java.io.DataInputStream;
21     import java.io.IOException;
22     
23     /**
24      * This class represents the MethodPositions attribute.
25      * 
26      * @author Siegfried Weber
27      * @author Juergen Graf <juergen.graf@gmail.com>
28      */
29     public final class MethodPositions extends PositionsAttribute {
30     
31       /** Stores the attribute name of this attribute */
32       public static final String ATTRIBUTE_NAME = "joana.sourceinfo.MethodPositions";
33     
34       private static final String ERR_COLUMN_ZERO = "Error in MethodPositions attribute: Invalid column number in %1$s.";
35     
36       private static final String ERR_LINE_ZERO = "Error in MethodPositions attribute: Invalid line number in %1$s.";
37     
38       private static final String ERR_RANGE_UNDEFINED = "Error in MethodPositions attribute: %1$s and %2$s are undefined.";
39     
40       private static final String ERR_SET_RANGE_UNDEFINED = "Error in MethodPositions attribute: Invalid positions, so %1$s and %2$s are set undefined.";
41     
42       private static final String ERR_POSITION_UNDEFINED = "Error in MethodPositions attribute: %1$s is undefined.";
43     
44       private static final String ERR_END_BEFORE_START = "Error in MethodPositions attribute: %2$s (%4$s) is before %1$s (%3$s).";
45     
46       private static final String WARN_INVALID_BLOCK_END = "Warning in MethodPositions attribute: Invalid method block end position.";
47     
48       private static final String WARN_PARAMETER_NOT_IN_DECLARATION = "Warning in MethodPositions attribute: Parameter not in the declaration range.";
49     
50       /** positions of the method declaration */
51       private Range declaration;
52       /** positions of the method parameters */
53       private Range parameter;
54       /** positions of the method block */
55       private Range block_end;
56     
57       /**
58        * Creates a new instance of MethodPositions
59        * 
60        * @param data
61        *          the byte array containing the attribute
62        * @throws IOException
63        *           if the attribute can't be read.
64        */
65       public MethodPositions(byte[] data) throws IOException {
66         super(data);
67         if (Debug.PRINT_CHARACTER_RANGE_TABLE) {
68           Debug.info("MethodPositions found: ");
69           Debug.info(toString());
70         }
71       }
72     
73       @Override
74       protected final void readData(DataInputStream in) throws IOException {
75         declaration = readRange(in, "declaration_start", "declaration_end", false);
76         parameter = readRange(in, "parameter_start", "parameter_end", true);
77         block_end = readRange(in, "block_end_start", "block_end_end", false);
78         if (!parameter.isUndefined()
79             && (!declaration.getStartPosition().isBefore(parameter.getStartPosition()) || !parameter.getEndPosition().isBefore(
80                 declaration.getEndPosition())))
81           Debug.warn(WARN_PARAMETER_NOT_IN_DECLARATION);
82         if (!declaration.getEndPosition().isBefore(block_end.getStartPosition()))
83           Debug.warn(WARN_INVALID_BLOCK_END);
84       }
85     
86       /**
87        * Reads a range from the input stream.
88        * 
89        * @param in
90        *          the input stream
91        * @param startVarName
92        *          the variable name for the start position
93        * @param endVarName
94        *          the variable name for the end position
95        * @param undefinedAllowed
96        *          {@code true} if the range may be undefined.
97        * @return the range
98        * @throws IOException
99        *           if the input stream cannot be read
100        */
101       private Range readRange(DataInputStream in, String startVarName, String endVarName, boolean undefinedAllowed) throws IOException {
102         boolean valid = true;
103         Range range = null;
104         Position start = null;
105         Position end = null;
106         try {
107           start = readPosition(in, startVarName);
108         } catch (InvalidPositionException e) {
109           valid = false;
110         }
111         try {
112           end = readPosition(in, endVarName);
113         } catch (InvalidPositionException e) {
114           valid = false;
115         }
116         if (valid) {
117           try {
118             range = new Range(start, end);
119           } catch (InvalidRangeException e) {
120             switch (e.getThisCause()) {
121             case END_BEFORE_START:
122               Debug.warn(ERR_END_BEFORE_START, startVarName, endVarName, start, end);
123             case START_UNDEFINED:
124               Debug.warn(ERR_POSITION_UNDEFINED, startVarName);
125             case END_UNDEFINED:
126               Debug.warn(ERR_POSITION_UNDEFINED, endVarName);
127             }
128           }
129         }
130         if (range == null) {
131           range = new Range();
132           Debug.warn(ERR_SET_RANGE_UNDEFINED, startVarName, endVarName);
133         }
134         if (range.isUndefined() && !undefinedAllowed) {
135           Debug.warn(ERR_RANGE_UNDEFINED, startVarName, endVarName);
136         }
137     
138         return range;
139       }
140     
141       /**
142        * Reads a position from the input stream.
143        * 
144        * @param in
145        *          the input stream
146        * @param varName
147        *          the variable name for this position
148        * @throws IOException
149        *           if the input stream cannot be read
150        * @throws InvalidPositionException
151        *           if the read position is invalid
152        */
153       private Position readPosition(DataInputStream in, String varName) throws IOException, InvalidPositionException {
154         Position pos = null;
155         try {
156           pos = new Position(in.readInt());
157         } catch (InvalidPositionException e) {
158           switch (e.getThisCause()) {
159           case LINE_NUMBER_ZERO:
160             Debug.warn(ERR_LINE_ZERO, varName);
161             throw e;
162           case COLUMN_NUMBER_ZERO:
163             Debug.warn(ERR_COLUMN_ZERO, varName);
164             throw e;
165           default:
166             assert false;
167           }
168         }
169         return pos;
170       }
171     
172       /**
173        * Returns the source position range of the method declaration.
174        * 
175        * @return the source position range of the method declaration
176        */
177       public final Range getHeaderInfo() {
178         return declaration;
179       }
180     
181       /**
182        * Returns the source position range of the method parameter declaration.
183        * 
184        * @return the source position range of the method parameter declaration
185        */
186       public final Range getMethodInfo() {
187         return parameter;
188       }
189     
190       /**
191        * Returns the source position range of the end of the method block.
192        * 
193        * @return the source position range of the end of the method block
194        */
195       public final Range getFooterInfo() {
196         return block_end;
197       }
198     
199       @Override
200       public String toString() {
201         return "header: " + declaration + " params: " + parameter + " footer:" + block_end;
202       }
203     }
