43     package com.ibm.wala.util.ssa;
44     
45     import com.ibm.wala.types.TypeReference;
46     import com.ibm.wala.types.TypeName;
47     import com.ibm.wala.types.MethodReference;

49     /**
50      *  A number representating an SSA-Value and its type.
51      *
52      *  WALA does not use this on a regular basis but it may come in handy for creating 
53      *  SyntheticMethods.
54      *
55      *  Use ParameterAccessor to get the parameters of a function as SSAValues.
56      *
57      *  @see    com.ibm.wala.util.ssa.TypeSafeInstructionFactory
58      *  @see    com.ibm.wala.util.ParameterAccessor
59      *
60      *  @author Tobias Blaschke <code@tobiasblaschke.de>
61      *  @since  2013-10-20
62      */
63     public class SSAValue {
64         /** The SSA Value itself */
65         protected final int number;
66         /** The type of this variable */
67         protected final TypeReference type;
68         /** All variables with the same name in the source code share a key. */ 
69         public final VariableKey key; // TODO: Protect again?
70         /** Method the variable is valid in */
71         protected final MethodReference mRef;
72         /** If an instruction wrote to this value (set manually) */
73         private boolean isAssigned;
74     
75         /**
76          *  All variables with the same name in the source code.
77          */
78         public interface VariableKey {}
79         /**
80          *  A key that cannot be recreated.
81          */
82         public static class UniqueKey implements VariableKey {
83             public UniqueKey() { }
84         }
85         /**
86          *  A key that matches variables by their type - does not compare to NamedKey.
87          */
88         public static class TypeKey implements VariableKey {
89             public final TypeName type;
90             public TypeKey(final TypeName type) {
91                 this.type = type;
92             }
93             @Override
94             public boolean equals(Object o) {
95                 if (o == null) {
96                     return false;
97                 } else if (o instanceof TypeKey) {
98                     TypeKey other = (TypeKey) o;
99                     return this.type.equals(other.type);
100                 } else if (o instanceof WeaklyNamedKey) {
101                     WeaklyNamedKey other = (WeaklyNamedKey) o;
102                     return this.type.equals(other.type);
103                 } else {
104                     return false;
105                 }
106             }
107             @Override
108             public int hashCode() {
109                 return this.type.hashCode();
110             }
111             @Override
112             public String toString() {
113                 return "<TypeKey type=\"" + this.type  + "\" />";
114             }
115         }
116         /**
117          *  This NamedKey also equals to TypeKeys.
118          */
119         public static class WeaklyNamedKey extends NamedKey {
120             public WeaklyNamedKey(final TypeName type, final String name) {
121                 super(type, name);
122             }
123             @Override
124             public boolean equals(Object o) {
125                 if (o == null) {
126                     return false;
127                 } else if (o instanceof NamedKey) {
128                     NamedKey other = (NamedKey) o;
129                     return (this.type.equals(other.type) && this.name.equals(other.name));
130                 } else if (o instanceof TypeKey) {
131                     TypeKey other = (TypeKey) o;
132                     return (this.type.equals(other.type));
133                 } else {
134                     return false;
135                 }
136             }
137             @Override
138             public int hashCode() {
139                 return this.type.hashCode() * ((this.name==null)?1:this.name.hashCode());
140             }
141             @Override
142             public String toString() {
143                 return "<WaklyNamedKey type=\"" + this.type  + "\" name=\"" + this.name + "\" />";
144             }
145         }
146         /**
147          *  Identify variables by a string and type.
148          */
149         public static class NamedKey implements VariableKey {
150             public final String name;
151             public final TypeName type;
152             public NamedKey(final TypeName type, final String name) {
153                 this.name = name;
154                 this.type = type;
155             }
156             @Override
157             public boolean equals(Object o) {
158                 if (o == null) {
159                     return false;
160                 } else if (o instanceof NamedKey) {
161                     NamedKey other = (NamedKey) o;
162                     return (this.type.equals(other.type) && this.name.equals(other.name));
163                 } else {
164                     return false;
165                 }
166             }
167             @Override
168             public int hashCode() {
169                 return this.type.hashCode() * ((this.name==null)?1:this.name.hashCode());
170             }
171             @Override
172             public String toString() {
173                 return "<NamedKey type=\"" + this.type  + "\" name=\"" + this.name + "\" />";
174             }
175         }

179         /**
180          *  Makes a SSAValue with number and type valid in the specified Method.
181          *
182          *  The number is the one to use with SSAInstructions. 
183          *
184          *  The MethodReference (validIn) is an optional value. However the TypeSafeInstructionFactory relies on it
185          *  to verify its ReturnInstruction so setting it does not hurt.
186          *
187          *  The variableName is optional and not really used yet. It might be handy for debugging.
188          *
189          *  @param  number  access the value using this number
190          *  @param  validIn optionally assign this value to a method
191          *  @param  variableName    optional name of the variable
192          *  @throws IllegalArgumentException on negative parameter number
193          */
194         public SSAValue(final int number, final TypeReference type, final MethodReference validIn, final VariableKey key) {
195             if (number < 0) {
196                 throw new IllegalArgumentException("A SSA-Value can't have a negative number, " + number + "given");
197             }
198             if (type == null) {
199                 throw new IllegalArgumentException("The type for the SSA-Variable may not be null");
200             }
201             if (type.equals(TypeReference.Void)) {
202                 throw new IllegalArgumentException("You can't create a SSA-Variable of type void");
203             }
204     
205             this.type = type;
206             this.number = number;
207             this.key = key;
208             this.mRef = validIn;
209             this.isAssigned = false;
210         }
211     
212         /**
213          *  Generates a SSAValue with a NamedKey (or TypeKey if name==null).
214          */
215         public SSAValue(final int number, final TypeReference type, final MethodReference validIn, final String variableName) {
216             this(number, type, validIn, ((variableName==null)?new TypeKey(type.getName()):new NamedKey(type.getName(), variableName)));
217         }
218     
219         /**
220          *  Generates a SSAValue with a UniqueKey.
221          */
222         public SSAValue(final int number, final TypeReference type, final MethodReference validIn) {
223             this(number, type, validIn, new UniqueKey());
224         }
225     
226     
227         /**
228          *  Create a new instance of the same type, validity and name.
229          *
230          *  Of course you still have to assign something to this value.
231          *
232          *  @param  number      the new number to use
233          *  @param  copyFrom    where to get the rest of the attributes
234          */
235         public SSAValue(final int number, SSAValue copyFrom) {
236             this(number, copyFrom.type, copyFrom.mRef, copyFrom.key);
237         }
238     
239         /**
240          *  The SSA-Value to use with SSAInstructions.
241          *
242          *  As an alternative one can generate Instructions using the TypeSafeInstructionFactory which takes
243          *  SSAValues as parameters.
244          */
245         public int getNumber() {
246             return this.number;
247         }
248     
249         /**
250          *  The type this SSA-Value represents.
251          */
252         public TypeReference getType() {
253             return this.type;
254         }
255     
256         /**
257          *  If setAssigned() was called on this variable.
258          */
259         public boolean isAssigned() {
260             return this.isAssigned;
261         }
262     
263         /**
264          *  Mark this variable as assigned.
265          *
266          *  Sets the value returned by isAssigned() to true. As a safety measure one can only call this method
267          *  once on a SSAValue, the second time raises an exception.
268          *
269          *  The TypeSafeInstructionFactory calls this method when writing to an SSAValue. It does however not check 
270          *  the setting when reading from an SSAValue.
271          *
272          *  This does obviously not prevent from generating a new SSAValue with the same number and double-assign
273          *  anyhow.
274          *
275          *  @throws IllegalStateException   if the variable was already assigned to
276          */
277         public void setAssigned() {
278             if (this.isAssigned) {
279                 throw new IllegalStateException("The SSA-Variable " + this + " was assigned to twice.");
280             }
281     
282             this.isAssigned = true;
283         }
284     
285         /**
286          *  Return the MethodReference this Variable was set valid in.
287          *
288          *  The value returned by this method is the one set in the constructor. As this parameter is optional to
289          *  it this function may return null if it was not set.
290          *
291          *  @return the argument validIn to the constructor
292          */
293         public MethodReference getValidIn() {
294             return this.mRef;
295         }
296         /**
297          *  Return the optional variable name.
298          *
299          *  @return the argument variableName to the constructor
300          */
301         public String getVariableName() {
302             if (this.key instanceof NamedKey) {
303                 return ((NamedKey)this.key).name;
304             } else {
305                 return null;    // TODO: build a name?
306             }
307         }
308     
309         @Override 
310         public String toString() {
311             return "<SSAValue " + this.number + " type=" + this.type + " validIn=" + this.mRef + ">";
312         }
313     
314         @Override
315         public boolean equals(Object o) {
316             if (o instanceof SSAValue) {
317                 final SSAValue other = (SSAValue) o;
318                 return ((this.number == other.number) && (this.mRef.equals(other.mRef)) && this.type.equals(other.type));
319             }
320             throw new IllegalArgumentException("Can't compare SSAValue to " + o.getClass());
321         }
322     
323         @Override
324         public int hashCode() {
325             return 157 * this.number * this.mRef.hashCode() * this.type.hashCode();
326         }

328     }
