11     package com.ibm.wala.util.collections;
12     
13     import java.util.Iterator;
14     
15     /**
16      * Converts an {@link Iterator} to an {@link Iterable}.
17      */
18     public class Iterator2Iterable<T> implements Iterable<T> {
19     
20       private final Iterator<T> iter;
21     
22       public static <T> Iterator2Iterable<T> make(Iterator<T> iter) {
23         return new Iterator2Iterable<T>(iter);
24       }
25     
26       public Iterator2Iterable(Iterator<T> iter) {
27         this.iter = iter;
28       }
29     
30       @Override
31       public Iterator<T> iterator() {
32         return iter;
33       }
34     
35     }
