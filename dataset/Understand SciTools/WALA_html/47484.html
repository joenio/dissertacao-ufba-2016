11     package com.ibm.wala.ipa.callgraph;
12     
13     import com.ibm.wala.analysis.typeInference.PointType;
14     import com.ibm.wala.classLoader.IClass;
15     import com.ibm.wala.ipa.callgraph.propagation.InstanceKey;
16     import com.ibm.wala.util.debug.Assertions;
17     
18     /**
19      * misc utilities for dealing with contexts
20      */
21     public class ContextUtil {
22     
23       /**
24        * @param c a context
25        * @return If this is an object-sensitive context that identifies a unique class for the receiver object, then return the unique
26        *         class. Else, return null.
27        * @throws IllegalArgumentException if c is null
28        */
29       public static IClass getConcreteClassFromContext(Context c) {
30         if (c == null) {
31           throw new IllegalArgumentException("c is null");
32         }
33         ContextItem item = c.get(ContextKey.RECEIVER);
34         if (item == null) {
35           return null;
36         } else {
37           if (item instanceof PointType) {
38             return ((PointType) item).getIClass();
39           } else if (item instanceof InstanceKey) {
40             return ((InstanceKey) item).getConcreteType();
41           } else {
42             Assertions.UNREACHABLE("Unexpected: " + item.getClass());
43             return null;
44           }
45         }
46       }
47     
48     }
