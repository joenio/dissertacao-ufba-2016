11     package com.ibm.wala.fixpoint;
12     
13     import com.ibm.wala.util.graph.INodeWithNumber;
14     
15     /**
16      * The general form of a statement definition in an iterative solver is: x &gt;= term, where
17      * term can be any complex expression whose free variables are among the
18      * IVariables of the constraint system
19      * this {@link IFixedPointStatement}is part of (x represents the left-hand side of the
20      * constraint). The interpretation of term (the right-hand side of the
21      * constraint) must be monotone.
22      * <p>
23      * The list of free variables in term is obtained by invoking {@link #getRHS()},
24      * and the left hand side variable is obtained by calling {@link #getLHS()}.
25      * </p>
26      * 
27      * Intuitively, a statement definition corresponds to an "equation" in dataflow parlance, or
28      * a "constraint" in constraint solvers.
29      */
30     @SuppressWarnings("rawtypes")
31     public interface IFixedPointStatement<T extends IVariable> extends INodeWithNumber {
32       /**
33        * @return the left-hand side of this statement.
34        */
35       public T getLHS();
36       
37       /**
38        * returns the list of free variables appearing in the right-hand side of the
39        * statement
40        */
41       public T[] getRHS();
42     
43       /**
44        * Evaluate this statement, setting a new value for the left-hand side. The
45        * return value is one of the following:
46        * <ul>
47        * {@link FixedPointConstants#CHANGED},
48        * {@link FixedPointConstants#CHANGED_AND_FIXED},
49        * {@link FixedPointConstants#NOT_CHANGED}or
50        * {@link FixedPointConstants#NOT_CHANGED_AND_FIXED}.
51        * </ul>
52        */
53       abstract byte evaluate();
54     
55       /**
56        * Does this statement definition contain an appearance of a given variable?
57        * 
58        * @param v the variable in question
59        */
60       public boolean hasVariable(T v);
61     }
