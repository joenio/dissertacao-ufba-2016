47       public static SourceFileModule makeSourceModule(URL script, String dir, String name) {
48         // DO NOT use File.separator here, since this name is matched against
49         // URLs. It seems that, in DOS, URL.getFile() does not return a
50         // \-separated file name, but rather one with /'s. Rather makes one
51         // wonder why the function is called get_File_ :(
52         return makeSourceModule(script, dir + "/" + name);
53       }

55       public static SourceFileModule makeSourceModule(URL script, String scriptName) {
56         String hackedName = script.getFile().replaceAll("%5c", "/").replaceAll("%20", " ");
57     
58         File scriptFile = new File(hackedName);
59     
60         assert hackedName.endsWith(scriptName) : scriptName + " does not match file " + script.getFile();
61     
62         return new SourceFileModule(scriptFile, scriptName, null) {
63           @Override
64           public InputStream getInputStream() {
65             BOMInputStream bs = new BOMInputStream(super.getInputStream(), false, 
66                 ByteOrderMark.UTF_8, 
67                 ByteOrderMark.UTF_16LE, ByteOrderMark.UTF_16BE,
68                 ByteOrderMark.UTF_32LE, ByteOrderMark.UTF_32BE);
69             try {
70               if (bs.hasBOM()) {
71                 System.err.println("removing BOM " + bs.getBOM());
72               }
73               return bs;
74             } catch (IOException e) {
75               return super.getInputStream();
76             }
77           }
78         };
79       }

101       public static String getShortName(IMethod method) {
102         String origName = method.getName().toString();
103         String result = origName;
104         if (origName.equals("do") || origName.equals("ctor")) {
105           result = method.getDeclaringClass().getName().toString();
106           result = result.substring(result.lastIndexOf('/') + 1);
107           if (origName.equals("ctor")) {
108             if (result.equals("LFunction")) {
109               String s = method.toString();
110               if (s.indexOf('(') != -1) {
111                 String functionName = s.substring(s.indexOf('(') + 1, s.indexOf(')'));
112                 functionName = functionName.substring(functionName.lastIndexOf('/') + 1);
113                 result += " " + functionName;
114               }
115             }
116             result = "ctor of " + result;
117           }
118         } 
119         return result;
120       }

122       public static void dumpCG(PointerAnalysis<InstanceKey> PA, CallGraph CG) {
123         if (AVOID_DUMP)
124           return;
125         for (Iterator x = CG.iterator(); x.hasNext();) {
126           CGNode N = (CGNode) x.next();
127           System.err.print("callees of node " + getShortName(N) + " : [");
128           boolean fst = true;
129           for (Iterator<? extends CGNode> ns = CG.getSuccNodes(N); ns.hasNext();) {
130             if (fst)
131               fst = false;
132             else
133               System.err.print(", ");
134             System.err.print(getShortName(ns.next()));
135           }
136           System.err.println("]");
137           System.err.println("\nIR of node " + N.getGraphNodeId() + ", context " + N.getContext());
138           IR ir = N.getIR();
139           if (ir != null) {
140             System.err.println(ir);
141           } else {
142             System.err.println("no IR!");
143           }
144         }
145     
146         System.err.println("pointer analysis");
147         for (Iterator x = PA.getPointerKeys().iterator(); x.hasNext();) {
148           PointerKey n = (PointerKey) x.next();
149           try {
150             System.err.println((n + " --> " + PA.getPointsToSet(n)));
151           } catch (Throwable e) {
152             System.err.println(("error computing set for " + n));
153           }
154         }
155       }

157       public static SourceFileModule[] handleFileNames(String[] fileNameArgs) {
158         SourceFileModule[] fileNames = new SourceFileModule[fileNameArgs.length];
159         for (int i = 0; i < fileNameArgs.length; i++) {
160           if (new File(fileNameArgs[i]).exists()) {
161             try {
162               fileNames[i] = CAstCallGraphUtil.makeSourceModule(new File(fileNameArgs[i]).toURI().toURL(), fileNameArgs[i]);
163             } catch (MalformedURLException e) {
164               Assertions.UNREACHABLE(e.toString());
165             }
166           } else {
167             URL url = CAstCallGraphUtil.class.getClassLoader().getResource(fileNameArgs[i]);
168             fileNames[i] = CAstCallGraphUtil.makeSourceModule(url, fileNameArgs[i]);
169           }
170         }
171     
172         return fileNames;
173       }
