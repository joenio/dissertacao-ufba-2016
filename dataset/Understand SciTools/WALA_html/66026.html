13     import java.io.File;
14     import java.net.URL;
15     import java.util.Properties;
16     
17     import com.ibm.wala.properties.WalaProperties;
18     import com.ibm.wala.util.io.FileProvider;
19     
20     public final class WalaExamplesProperties {
21     
22       public static final String PDFVIEW_EXE = "pdfview_exe"; //$NON-NLS-1$
23     
24       public static final String DOT_EXE = "dot_exe"; //$NON-NLS-1$
25     
26       public final static String PROPERTY_FILENAME = "wala.examples.properties"; //$NON-NLS-1$
27     
28       public static Properties loadProperties()  {
29     
30         try {
31           Properties result = WalaProperties.loadPropertiesFromFile(WalaExamplesProperties.class.getClassLoader(), PROPERTY_FILENAME);
32     
33           return result;
34         } catch (Exception e) {
35           e.printStackTrace();
36           throw new IllegalStateException("Unable to set up wala examples properties ", e);
37         }
38       }
39     
40       public static String getWalaCoreTestsHomeDirectory()  {
41         final URL url = WalaExamplesProperties.class.getClassLoader().getResource(PROPERTY_FILENAME);
42         if (url == null) {
43           throw new IllegalStateException("failed to find URL for wala.examples.properties");
44         }
45     
46         return new File((new FileProvider()).filePathFromURL(url)).getParentFile().getParentFile().getAbsolutePath();
47       }
