25     /**
26      * Exploded interprocedural control-flow graph, constructed lazily.
27      */
28     public class ExplodedInterproceduralCFG extends AbstractInterproceduralCFG<IExplodedBasicBlock> {
29     
30       /**
31        * Caching to improve runtime .. hope it doesn't turn into a memory leak.
32        */
33       private Map<CGNode, ExplodedControlFlowGraph> cfgMap;
34     
35       public static ExplodedInterproceduralCFG make(CallGraph cg) {
36         return new ExplodedInterproceduralCFG(cg);
37       }
38     
39       protected ExplodedInterproceduralCFG(CallGraph cg) {
40         super(cg);
41       }
42       
43       public ExplodedInterproceduralCFG(CallGraph cg, Predicate<CGNode> filter) {
44         super(cg, filter);
45       }
46     
47       /**
48        * @return the cfg for n, or null if none found
49        * @throws IllegalArgumentException if n == null
50        */
51       @Override
52       public ControlFlowGraph<SSAInstruction, IExplodedBasicBlock> getCFG(CGNode n) throws IllegalArgumentException {
53         if (n == null) {
54           throw new IllegalArgumentException("n == null");
55         }
56         if (cfgMap == null) {
57           // we have to initialize this lazily since this might be called from a super() constructor
58           cfgMap = HashMapFactory.make();
59         }
60         ExplodedControlFlowGraph result = cfgMap.get(n);
61         if (result == null) {
62           IR ir = n.getIR();
63           if (ir == null) {
64             return null;
65           }
66           result = ExplodedControlFlowGraph.make(ir);
67           cfgMap.put(n, result);
68         }
69         return result;
70       }
71     
72     }
