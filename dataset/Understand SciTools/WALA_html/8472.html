155        * @see com.ibm.wala.demandpa.flowgraph.IFlowLabelGraph#visitSuccs(java.lang.Object,
156        * com.ibm.wala.demandpa.flowgraph.IFlowLabel.IFlowLabelVisitor)

158       @Override
159       public void visitSuccs(Object node, IFlowLabelVisitor v) {
160         for (Iterator<? extends IFlowLabel> succLabelIter = getSuccLabels(node); succLabelIter.hasNext();) {
161           final IFlowLabel label = succLabelIter.next();
162           for (Iterator<? extends Object> succNodeIter = getSuccNodes(node, label); succNodeIter.hasNext();) {
163             label.visit(v, succNodeIter.next());
164           }
165         }
166       }

172       @Override
173       public void visitPreds(Object node, IFlowLabelVisitor v) {
174         for (Iterator<? extends IFlowLabel> predLabelIter = getPredLabels(node); predLabelIter.hasNext();) {
175           final IFlowLabel label = predLabelIter.next();
176           for (Iterator<? extends Object> predNodeIter = getPredNodes(node, label); predNodeIter.hasNext();) {
177             label.visit(v, predNodeIter.next());
178           }
179         }

182       /**
183        * For each invocation in the method, add nodes for actual parameters and return values
184        * 
185        * @param node
186        */
187       protected void addNodesForInvocations(CGNode node, IR ir) {
188         for (Iterator<CallSiteReference> iter = ir.iterateCallSites(); iter.hasNext();) {
189           CallSiteReference site = iter.next();
190           SSAAbstractInvokeInstruction[] calls = ir.getCalls(site);
191           for (SSAAbstractInvokeInstruction invokeInstr : calls) {
192             for (int i = 0; i < invokeInstr.getNumberOfUses(); i++) {
193               // just make nodes for parameters; we'll get to them when
194               // traversing
195               // from the callee
196               PointerKey use = heapModel.getPointerKeyForLocal(node, invokeInstr.getUse(i));
197               addNode(use);
198               Set<SSAAbstractInvokeInstruction> s = MapUtil.findOrCreateSet(callParams, use);
199               s.add(invokeInstr);
200             }
201     
202             // for any def'd values, keep track of the fact that they are def'd
203             // by a call
204             if (invokeInstr.hasDef()) {
205               PointerKey def = heapModel.getPointerKeyForLocal(node, invokeInstr.getDef());
206               addNode(def);
207               callDefs.put(def, invokeInstr);
208             }
209             PointerKey exc = heapModel.getPointerKeyForLocal(node, invokeInstr.getException());
210             addNode(exc);
211             callDefs.put(exc, invokeInstr);
212     
213           }
214         }

261       @Override
262       public Iterator<PointerKey> getWritesToInstanceField(PointerKey pk, IField f) {
263         // TODO: cache this!!
264         if (f == ArrayContents.v()) {
265           return getArrayWrites(pk);
266         }
267         pk = convertPointerKeyToHeapModel(pk, mam.getHeapModel());
268         Collection<MemoryAccess> writes = mam.getFieldWrites(pk, f);
269         for (MemoryAccess a : writes) {
270           addSubgraphForNode(a.getNode());
271         }
272         ArrayList<PointerKey> written = new ArrayList<PointerKey>();
273         for (MemoryAccess a : writes) {
274           IR ir = a.getNode().getIR();
275           SSAPutInstruction s = (SSAPutInstruction) ir.getInstructions()[a.getInstructionIndex()];
276           if (s == null) {
277             // s can be null because the memory access map may be constructed from bytecode,
278             // and the write instruction may have been eliminated from SSA because it's dead
279             // TODO clean this up
280             continue;
281           }
282           PointerKey r = heapModel.getPointerKeyForLocal(a.getNode(), s.getVal());
283           // if (Assertions.verifyAssertions) {
284           // Assertions._assert(containsNode(r));
285           // }
286           written.add(r);
287         }
288         return written.iterator();
289       }

291       /**
292        * convert a pointer key to one in the memory access map's heap model
293        * 
294        * TODO move this somewhere more appropriate
295        * 
296        * @throws UnsupportedOperationException if it doesn't know how to handle a {@link PointerKey}
297        */
298       public static PointerKey convertPointerKeyToHeapModel(PointerKey pk, HeapModel h) {
299         if (pk == null) {
300           throw new IllegalArgumentException("null pk");
301         }
302         if (pk instanceof LocalPointerKey) {
303           LocalPointerKey lpk = (LocalPointerKey) pk;
304           return h.getPointerKeyForLocal(lpk.getNode(), lpk.getValueNumber());
305         } else if (pk instanceof ArrayContentsKey) {
306           ArrayContentsKey ack = (ArrayContentsKey) pk;
307           InstanceKey ik = ack.getInstanceKey();
308           if (ik instanceof NormalAllocationInNode) {
309             NormalAllocationInNode nain = (NormalAllocationInNode) ik;
310             ik = h.getInstanceKeyForAllocation(nain.getNode(), nain.getSite());
311           } else {
312             assert false : "need to handle " + ik.getClass();
313           }
314           return h.getPointerKeyForArrayContents(ik);
315         } else if (pk instanceof ReturnValueKey) {
316           ReturnValueKey rvk = (ReturnValueKey) pk;
317           return h.getPointerKeyForReturnValue(rvk.getNode());
318         }
319         throw new UnsupportedOperationException("need to handle " + pk.getClass());
320       }

322       @Override
323       public Iterator<PointerKey> getReadsOfInstanceField(PointerKey pk, IField f) {
324         // TODO: cache this!!
325         if (f == ArrayContents.v()) {
326           return getArrayReads(pk);
327         }
328         pk = convertPointerKeyToHeapModel(pk, mam.getHeapModel());
329         Collection<MemoryAccess> reads = mam.getFieldReads(pk, f);
330         for (MemoryAccess a : reads) {
331           addSubgraphForNode(a.getNode());
332         }
333         ArrayList<PointerKey> readInto = new ArrayList<PointerKey>();
334         for (MemoryAccess a : reads) {
335           IR ir = a.getNode().getIR();
336           SSAGetInstruction s = (SSAGetInstruction) ir.getInstructions()[a.getInstructionIndex()];
337           if (s == null) {
338             // actually dead code
339             continue;
340           }
341           PointerKey r = heapModel.getPointerKeyForLocal(a.getNode(), s.getDef());
342           // if (Assertions.verifyAssertions) {
343           // Assertions._assert(containsNode(r));
344           // }
345           readInto.add(r);
346         }
347         return readInto.iterator();
348       }

350       Iterator<PointerKey> getArrayWrites(PointerKey arrayRef) {
351         arrayRef = convertPointerKeyToHeapModel(arrayRef, mam.getHeapModel());
352         Collection<MemoryAccess> arrayWrites = mam.getArrayWrites(arrayRef);
353         for (MemoryAccess a : arrayWrites) {
354           addSubgraphForNode(a.getNode());
355         }
356         ArrayList<PointerKey> written = new ArrayList<PointerKey>();
357         for (MemoryAccess a : arrayWrites) {
358           final CGNode node = a.getNode();
359           IR ir = node.getIR();
360           SSAInstruction instruction = ir.getInstructions()[a.getInstructionIndex()];
361           if (instruction == null) {
362             // this means the array store found was in fact dead code
363             // TODO detect this earlier and don't keep it in the MemoryAccessMap
364             continue;
365           }
366           if (instruction instanceof SSAArrayStoreInstruction) {
367             SSAArrayStoreInstruction s = (SSAArrayStoreInstruction) instruction;
368             PointerKey r = heapModel.getPointerKeyForLocal(node, s.getValue());
369             written.add(r);
370           } else if (instruction instanceof SSANewInstruction) {
371             NewMultiDimInfo multiDimInfo = DemandPointerFlowGraph.getInfoForNewMultiDim((SSANewInstruction) instruction, heapModel,
372                 node);
373             for (Pair<PointerKey, PointerKey> arrStoreInstr : multiDimInfo.arrStoreInstrs) {
374               written.add(arrStoreInstr.snd);
375             }
376           } else {
377             Assertions.UNREACHABLE();
378           }
379         }
380         return written.iterator();
381       }

383       protected Iterator<PointerKey> getArrayReads(PointerKey arrayRef) {
384         arrayRef = convertPointerKeyToHeapModel(arrayRef, mam.getHeapModel());
385         Collection<MemoryAccess> arrayReads = mam.getArrayReads(arrayRef);
386         for (Iterator<MemoryAccess> it = arrayReads.iterator(); it.hasNext();) {
387           MemoryAccess a = it.next();
388           addSubgraphForNode(a.getNode());
389         }
390         ArrayList<PointerKey> read = new ArrayList<PointerKey>();
391         for (Iterator<MemoryAccess> it = arrayReads.iterator(); it.hasNext();) {
392           MemoryAccess a = it.next();
393           IR ir = a.getNode().getIR();
394           SSAArrayLoadInstruction s = (SSAArrayLoadInstruction) ir.getInstructions()[a.getInstructionIndex()];
395           if (s == null) {
396             // actually dead code
397             continue;
398           }
399           PointerKey r = heapModel.getPointerKeyForLocal(a.getNode(), s.getDef());
400           // if (Assertions.verifyAssertions) {
401           // Assertions._assert(containsNode(r));
402           // }
403           read.add(r);
404         }
405         return read.iterator();
406       }

408       /**
409        * Add constraints to represent the flow of exceptions to the exceptional return value for this node
410        */
411       protected void addNodePassthruExceptionConstraints(CGNode node, IR ir) {
412         // add constraints relating to thrown exceptions that reach the exit
413         // block.
414         List<ProgramCounter> peis = SSAPropagationCallGraphBuilder.getIncomingPEIs(ir, ir.getExitBlock());
415         PointerKey exception = heapModel.getPointerKeyForExceptionalReturnValue(node);
416         IClass c = node.getClassHierarchy().lookupClass(TypeReference.JavaLangThrowable);
417     
418         addExceptionDefConstraints(ir, node, peis, exception, Collections.singleton(c));
419       }

421       /**
422        * Generate constraints which assign exception values into an exception pointer
423        * 
424        * @param node governing node
425        * @param peis list of PEI instructions
426        * @param exceptionVar PointerKey representing a pointer to an exception value
427        * @param catchClasses the types "caught" by the exceptionVar
428        */
429       protected void addExceptionDefConstraints(IR ir, CGNode node, List<ProgramCounter> peis, PointerKey exceptionVar,
430           Set<IClass> catchClasses) {
431         for (Iterator<ProgramCounter> it = peis.iterator(); it.hasNext();) {
432           ProgramCounter peiLoc = it.next();
433           SSAInstruction pei = ir.getPEI(peiLoc);
434     
435           if (pei instanceof SSAAbstractInvokeInstruction) {
436             SSAAbstractInvokeInstruction s = (SSAAbstractInvokeInstruction) pei;
437             PointerKey e = heapModel.getPointerKeyForLocal(node, s.getException());
438             addNode(exceptionVar);
439             addNode(e);
440             addEdge(exceptionVar, e, AssignLabel.noFilter());
441     
442           } else if (pei instanceof SSAAbstractThrowInstruction) {
443             SSAAbstractThrowInstruction s = (SSAAbstractThrowInstruction) pei;
444             PointerKey e = heapModel.getPointerKeyForLocal(node, s.getException());
445             addNode(exceptionVar);
446             addNode(e);
447             addEdge(exceptionVar, e, AssignLabel.noFilter());
448           }
449     
450           // Account for those exceptions for which we do not actually have a
451           // points-to set for
452           // the pei, but just instance keys
453           Collection<TypeReference> types = pei.getExceptionTypes();
454           if (types != null) {
455             for (Iterator<TypeReference> it2 = types.iterator(); it2.hasNext();) {
456               TypeReference type = it2.next();
457               if (type != null) {
458                 InstanceKey ik = heapModel.getInstanceKeyForPEI(node, peiLoc, type);
459                 if (ik == null) {
460                   // ugh.  hope someone somewhere else knows what they are doing.
461                   // this is probably due to analysis scope exclusions.
462                   continue;
463                 }
464                 if (!(ik instanceof ConcreteTypeKey)) {
465                   assert ik instanceof ConcreteTypeKey : "uh oh: need to implement getCaughtException constraints for instance " + ik;
466                 }
467                 ConcreteTypeKey ck = (ConcreteTypeKey) ik;
468                 IClass klass = ck.getType();
469                 if (PropagationCallGraphBuilder.catches(catchClasses, klass, cha)) {
470                   addNode(exceptionVar);
471                   addNode(ik);
472                   addEdge(exceptionVar, ik, NewLabel.v());
473                 }
474               }
475             }
476           }
477         }
478       }

480       /**
481        * add constraints for reference constants assigned to vars
482        */
483       protected void addNodeConstantConstraints(CGNode node, IR ir) {
484         SymbolTable symbolTable = ir.getSymbolTable();
485         for (int i = 1; i <= symbolTable.getMaxValueNumber(); i++) {
486           if (symbolTable.isConstant(i)) {
487             Object v = symbolTable.getConstantValue(i);
488             if (!(v instanceof Number)) {
489               Object S = symbolTable.getConstantValue(i);
490               TypeReference type = node.getMethod().getDeclaringClass().getClassLoader().getLanguage().getConstantType(S);
491               if (type != null) {
492                 InstanceKey ik = heapModel.getInstanceKeyForConstant(type, S);
493                 if (ik != null) {
494                   PointerKey pk = heapModel.getPointerKeyForLocal(node, i);
495                   addNode(pk);
496                   addNode(ik);
497                   addEdge(pk, ik, NewLabel.v());
498                 }
499               }
500             }
501           }
502         }
503       }
