11     package com.ibm.wala.cast.java.ipa.callgraph;

13     import com.ibm.wala.ipa.callgraph.AnalysisCache;
14     import com.ibm.wala.ipa.callgraph.AnalysisOptions;

22     
23     /**
24      * 0-1-CFA Call graph builder, optimized to not disambiguate instances of "uninteresting" types.
25      */
26     public class AstJavaZeroXCFABuilder extends AstJavaCFABuilder {
27     
28       public AstJavaZeroXCFABuilder(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache,
29           ContextSelector appContextSelector, SSAContextInterpreter appContextInterpreter, int instancePolicy) {
30         super(cha, options, cache);
31     
32         SSAContextInterpreter contextInterpreter = makeDefaultContextInterpreters(appContextInterpreter, options, cha);
33         setContextInterpreter(contextInterpreter);
34     
35         ContextSelector def = new DefaultContextSelector(options, cha);
36         ContextSelector contextSelector = appContextSelector == null ? def : new DelegatingContextSelector(appContextSelector, def);
37     
38         setContextSelector(contextSelector);
39     
40         setInstanceKeys(new JavaScopeMappingInstanceKeys(cha, this, new ZeroXInstanceKeys(options, cha, contextInterpreter,
41             instancePolicy)));
42       }
43     
44       /**
45        * @param options options that govern call graph construction
46        * @param cha governing class hierarchy
47        * @param cl classloader that can find DOMO resources
48        * @param scope representation of the analysis scope
49        * @param xmlFiles set of Strings that are names of XML files holding bypass logic specifications.
50        * @return a 0-1-Opt-CFA Call Graph Builder.
51        */
52       public static AstJavaCFABuilder make(AnalysisOptions options, AnalysisCache cache, IClassHierarchy cha, ClassLoader cl,
53           AnalysisScope scope, String[] xmlFiles, byte instancePolicy) {
54     
55         com.ibm.wala.ipa.callgraph.impl.Util.addDefaultSelectors(options, cha);
56         for (int i = 0; i < xmlFiles.length; i++) {
57           com.ibm.wala.ipa.callgraph.impl.Util.addBypassLogic(options, scope, cl, xmlFiles[i], cha);
58         }
59     
60         return new AstJavaZeroXCFABuilder(cha, options, cache, null, null, instancePolicy);
61       }
62     
63     }
