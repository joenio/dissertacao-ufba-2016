11     package com.ibm.wala.cast.js.ssa;
12     
13     import java.util.Collection;
14     
15     import com.ibm.wala.ssa.SSAAbstractUnaryInstruction;
16     import com.ibm.wala.ssa.SSAInstruction;
17     import com.ibm.wala.ssa.SSAInstructionFactory;
18     import com.ibm.wala.ssa.SymbolTable;
19     import com.ibm.wala.types.TypeReference;
20     
21     public class JavaScriptTypeOfInstruction extends SSAAbstractUnaryInstruction {
22     
23       public JavaScriptTypeOfInstruction(int iindex, int lval, int object) {
24         super(iindex, lval, object);
25       }
26     
27       @Override
28       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
29         return ((JSInstructionFactory)insts).TypeOfInstruction(iindex, (defs != null ? defs[0] : getDef(0)), (uses != null ? uses[0] : getUse(0)));
30       }
31     
32       @Override
33       public String toString(SymbolTable symbolTable) {
34         return getValueString(symbolTable, getDef(0)) + " = typeof(" + getValueString(symbolTable, getUse(0)) + ")";
35       }
36     
37       @Override
38       public void visit(IVisitor v) {
39         ((JSInstructionVisitor) v).visitTypeOf(this);
40       }
41     
42       @Override
43       public Collection<TypeReference> getExceptionTypes() {
44         return Util.noExceptions();
45       }
46     
47     }
