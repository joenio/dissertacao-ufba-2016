35     /**
36      * An SSA IR.
37      * 
38      * The IR (Intermediate Representation) is the central data structure that represents the instructions of a particular method.
39      * The IR represents a method's instructions in a language close to JVM bytecode, but in an SSA-based register transfer language
40      * which eliminates the stack abstraction, relying instead on a set of symbolic registers. The IR organizes instructions in a
41      * control-flow graph of basic blocks, as typical in compiler textbooks.
42      * 
43      * See http://wala.sourceforge.net/wiki/index.php/UserGuide:IR for more details on the IR API.
44      */
45     public abstract class IR {

116       /**
117        * create mappings from call sites, new sites, and PEIs to instruction index
118        */
119       protected void setupLocationMap() {
120         for (int i = 0; i < instructions.length; i++) {
121           SSAInstruction x = instructions[i];
122           if (x != null) {
123             if (x instanceof SSAAbstractInvokeInstruction) {
124               callSiteMapping.add(((SSAAbstractInvokeInstruction) x).getCallSite().getProgramCounter(), i);
125             }
126             if (x instanceof SSANewInstruction) {
127               newSiteMapping.put(((SSANewInstruction) x).getNewSite(), new Integer(i));
128             }
129             if (x.isPEI()) {
130                peiMapping.put(new ProgramCounter(cfg.getProgramCounter(i)), new Integer(i));
131             }
132           }
133         }
134       }

141       @Override
142       public String toString() {
143         Collection<? extends SSAIndirectionData.Name> names = null;
144         if (getIndirectionData() != null) {
145           names = getIndirectionData().getNames();
146         }
147         StringBuffer result = new StringBuffer(method.toString());
148         result.append("\nCFG:\n");
149         result.append(cfg.toString());
150         result.append("Instructions:\n");
151         for (int i = 0; i <= cfg.getMaxNumber(); i++) {
152           BasicBlock bb = cfg.getNode(i);
153           int start = bb.getFirstInstructionIndex();
154           int end = bb.getLastInstructionIndex();
155           result.append("BB").append(bb.getNumber());
156           if (bb instanceof ExceptionHandlerBasicBlock) {
157             result.append("<Handler>");
158           }
159           result.append("\n");
160           for (Iterator it = bb.iteratePhis(); it.hasNext();) {
161             SSAPhiInstruction phi = (SSAPhiInstruction) it.next();
162             if (phi != null) {
163               result.append("           " + phi.toString(symbolTable)).append("\n");
164             }
165           }
166           if (bb instanceof ExceptionHandlerBasicBlock) {
167             ExceptionHandlerBasicBlock ebb = (ExceptionHandlerBasicBlock) bb;
168             SSAGetCaughtExceptionInstruction s = ebb.getCatchInstruction();
169             if (s != null) {
170               result.append("           " + s.toString(symbolTable)).append("\n");
171             } else {
172               result.append("           " + " No catch instruction. Unreachable?\n");
173             }
174           }
175           for (int j = start; j <= end; j++) {
176             if (instructions[j] != null) {
177               if (names != null) {
178                 boolean any = false;
179                 for(SSAIndirectionData.Name n : names) {
180                   if (getIndirectionData().getUse(j, n) != -1) {
181                     result.append(" " + n + " -> " + getIndirectionData().getUse(j, n));
182                     any = true;
183                   }
184                 }
185                 if (any) {
186                   result.append("\n");
187                 }
188               }
189               StringBuffer x = new StringBuffer(j + "   " + instructions[j].toString(symbolTable));
190               StringStuff.padWithSpaces(x, 45);
191               result.append(x);
192               result.append(instructionPosition(j));
193               
194               Map<Integer,Set<String>> valNames = HashMapFactory.make();
195               for(int v = 0; v < instructions[j].getNumberOfDefs(); v++) {
196                 int valNum = instructions[j].getDef(v);
197                 addNames(j, valNames, valNum);
198               }
199               for(int v = 0; v < instructions[j].getNumberOfUses(); v++) {
200                 int valNum = instructions[j].getUse(v);
201                 addNames(j, valNames, valNum);
202               }
203               if (!valNames.isEmpty()) {
204                 result.append(" ").append(valNames);
205               }
206      
207               result.append("\n");
208               
209               if (names != null) {
210                 boolean any = false;
211                 for(SSAIndirectionData.Name n : names) {
212                   if (getIndirectionData().getDef(j, n) != -1) {
213                     result.append(" " + n + " <- " + getIndirectionData().getDef(j, n));
214                     any = true;
215                   }
216                 }
217                 if (any) {
218                   result.append("\n");
219                 }
220               }
221             }
222           }
223           for (Iterator it = bb.iteratePis(); it.hasNext();) {
224             SSAPiInstruction pi = (SSAPiInstruction) it.next();
225             if (pi != null) {
226               result.append("           " + pi.toString(symbolTable)).append("\n");
227             }
228           }
229         }
230         return result.toString();
231       }

233       private void addNames(int j, Map<Integer, Set<String>> valNames, int valNum) {
234         if (getLocalNames(j, valNum) != null && getLocalNames(j, valNum).length > 0) {
235           if (! valNames.containsKey(valNum)) {
236             valNames.put(valNum, HashSetFactory.<String>make());
237           }
238           for(String s : getLocalNames(j, valNum)) {
239             valNames.get(valNum).add(s);
240           }
241         }

293       /**
294        * An {@link Iterator} over all {@link SSAInstruction}s of a certain type, retrieved by iterating over the {@link BasicBlock}s,
295        * one at a time.
296        * 
297        * TODO: this looks buggy to me .. looks like it's hardcoded for Phis. Does it work for Pis?
298        */
299       abstract private class TwoLevelIterator implements Iterator<SSAInstruction> {
300         // invariant: if currentBlockIndex != -1, then
301         // currentBlockIterator.hasNext()
302         private Iterator<? extends SSAInstruction> currentBlockIterator;
303     
304         private int currentBlockIndex;
305     
306         abstract Iterator<? extends SSAInstruction> getBlockIterator(BasicBlock b);
307     
308         TwoLevelIterator() {
309           currentBlockIndex = 0;
310           currentBlockIterator = cfg.getNode(0).iteratePhis();
311           if (!currentBlockIterator.hasNext()) {
312             advanceBlock();
313           }
314         }
315     
316         @Override
317         public boolean hasNext() {
318           return currentBlockIndex != -1;
319         }
320     
321         @Override
322         public SSAInstruction next() {
323           SSAInstruction result = currentBlockIterator.next();
324           if (!currentBlockIterator.hasNext()) {
325             advanceBlock();
326           }
327           return result;
328         }
329     
330         @Override
331         public void remove() {
332           Assertions.UNREACHABLE();
333         }
334     
335         private void advanceBlock() {
336           for (int i = currentBlockIndex + 1; i <= cfg.getMaxNumber(); i++) {
337             Iterator<? extends SSAInstruction> it = getBlockIterator(cfg.getNode(i));
338             if (it.hasNext()) {
339               currentBlockIndex = i;
340               currentBlockIterator = it;
341               return;
342             }
343           }
344           currentBlockIterator = null;
345           currentBlockIndex = -1;
346         }
347       }

393        * TODO: looks like this should be merged into {@link TwoLevelIterator}, above?
394        */
395       private class CatchIterator implements Iterator<SSAInstruction> {
396         // invariant: if currentBlockIndex != -1, then
397         // then block[currentBlockIndex] is a handler block
398         private int currentBlockIndex;
399     
400         private boolean hasCatch(Object x) {
401           return (x instanceof ExceptionHandlerBasicBlock) && (((ExceptionHandlerBasicBlock) x).getCatchInstruction() != null);
402         }
403     
404         CatchIterator() {
405           currentBlockIndex = 0;
406           if (!hasCatch(cfg.getNode(0))) {
407             advanceBlock();
408           }
409         }
410     
411         @Override
412         public boolean hasNext() {
413           return currentBlockIndex != -1;
414         }
415     
416         @Override
417         public SSAInstruction next() {
418           ExceptionHandlerBasicBlock bb = (ExceptionHandlerBasicBlock) cfg.getNode(currentBlockIndex);
419           SSAInstruction result = bb.getCatchInstruction();
420           advanceBlock();
421           return result;
422         }
423     
424         @Override
425         public void remove() {
426           Assertions.UNREACHABLE();
427         }
428     
429         private void advanceBlock() {
430           for (int i = currentBlockIndex + 1; i < cfg.getMaxNumber(); i++) {
431             if (hasCatch(cfg.getNode(i))) {
432               currentBlockIndex = i;
433               return;
434             }
435           }
436           currentBlockIndex = -1;
437         }
438       }

465       private class NormalIterator implements Iterator<SSAInstruction> {
466         int nextIndex = -1;
467     
468         final SSAInstruction[] instructions = getInstructions();
469     
470         NormalIterator() {
471           advanceIndex(0);
472         }
473     
474         private void advanceIndex(int start) {
475           for (int i = start; i < instructions.length; i++) {
476             if (instructions[i] != null) {
477               nextIndex = i;
478               return;
479             }
480           }
481           nextIndex = -1;
482         }
483     
484         @Override
485         public boolean hasNext() {
486           return nextIndex != -1;
487         }
488     
489         @Override
490         public void remove() {
491           Assertions.UNREACHABLE();
492         }
493     
494         @Override
495         public SSAInstruction next() {
496           SSAInstruction result = instructions[nextIndex];
497           advanceIndex(nextIndex + 1);
498           return result;
499         }
500       }

517       /**
518        * Return the invoke instructions corresponding to a call site
519        * 
520        * Note that Shrike may inline JSRS. This can lead to multiple copies of a single bytecode instruction in a particular IR. So we
521        * may have more than one instruction index for a particular call site from bytecode.
522        */
523       public SSAAbstractInvokeInstruction[] getCalls(CallSiteReference site) {
524         if (site == null) {
525           throw new IllegalArgumentException("site is null");
526         }
527         IntSet s = callSiteMapping.getRelated(site.getProgramCounter());
528         if (s == null) {
529           throw new IllegalArgumentException("no calls at site's pc");
530         }
531         SSAAbstractInvokeInstruction[] result = new SSAAbstractInvokeInstruction[s.size()];
532         int index = 0;
533         for (IntIterator it = s.intIterator(); it.hasNext();) {
534           int i = it.next();
535           result[index++] = (SSAAbstractInvokeInstruction) instructions[i];
536         }
537         return result;
538       }

585       /**
586        * @return an {@link Iterator} of all the call sites ( {@link CallSiteReference}s ) in this IR
587        */
588       public Iterator<CallSiteReference> iterateCallSites() {
589         return new Iterator<CallSiteReference>() {
590           private final int limit = callSiteMapping.maxKeyValue();
591     
592           private int i = -1;
593     
594           {
595             advance();
596           }
597     
598           private void advance() {
599             while (callSiteMapping.getRelatedCount(++i) == 0 && i <= limit)
600               ;
601           }
602     
603           @Override
604           public boolean hasNext() {
605             return i <= limit;
606           }
607     
608           @Override
609           public CallSiteReference next() {
610             int index = callSiteMapping.getRelated(i).max();
611             advance();
612             return ((SSAAbstractInvokeInstruction) instructions[index]).getCallSite();
613           }
614     
615           @Override
616           public void remove() {
617             throw new UnsupportedOperationException();
618           }
619         };
620       }

622       /**
623        * @param site a call site in this method
624        * @return the basic block corresponding to this instruction
625        * @throws IllegalArgumentException if site is null
626        */
627       public ISSABasicBlock[] getBasicBlocksForCall(final CallSiteReference site) {
628         if (site == null) {
629           throw new IllegalArgumentException("site is null");
630         }
631         final IntSet s = callSiteMapping.getRelated(site.getProgramCounter());
632         if (s == null) {
633           throw new IllegalArgumentException("invalid site: " + site);
634         }
635         final ISSABasicBlock[] result = new ISSABasicBlock[s.size()];
636         int index = 0;
637         for (final IntIterator it = s.intIterator(); it.hasNext();) {
638           final int i = it.next();
639           result[index++] = getControlFlowGraph().getBlockForInstruction(i);
640         }
641         return result;
642       }

651           mapInstructions2Blocks();

656       private void mapInstructions2Blocks() {
657         instruction2Block = HashMapFactory.make();
658         for (ISSABasicBlock b : cfg) {
659           for (SSAInstruction s : b) {
660             instruction2Block.put(s, b);
661           }
662         }
