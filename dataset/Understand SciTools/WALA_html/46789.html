11     package com.ibm.wala.util.collections;
12     
13     import java.io.Serializable;
14     import java.util.Iterator;
15     import java.util.NoSuchElementException;
16     
17     import com.ibm.wala.util.debug.Assertions;
18     import com.ibm.wala.util.intset.IntIterator;
19     import com.ibm.wala.util.intset.MutableSparseIntSet;
20     import com.ibm.wala.util.intset.TunedMutableSparseIntSet;
21     
22     /**
23      * An {@link IVector} implementation designed for low occupancy. Note that get() from this
24      * vector is a binary search.
25      * 
26      * This should only be used for small sets ... insertion and deletion are linear
27      * in size of set.
28      */
29     public class SparseVector<T> implements IVector<T>, Serializable {
30     
31       private final static int DEF_INITIAL_SIZE = 5;
32     
33       /**
34        * if indices[i] = x, then data[i] == get(x)
35        */
36       private MutableSparseIntSet indices = MutableSparseIntSet.makeEmpty();
37     
38       private Object[] data;
39     
40       public SparseVector() {
41         data = new Object[DEF_INITIAL_SIZE];
42         indices = MutableSparseIntSet.makeEmpty();
43       }
44     
45       /**
46        * @param initialSize
47        * @param expansion
48        */
49       public SparseVector(int initialSize, float expansion) {
50         data = new Object[DEF_INITIAL_SIZE];
51         indices = new TunedMutableSparseIntSet(initialSize, expansion);
52       }
53     
54       /*
55        * @see com.ibm.wala.util.intset.IntVector#get(int)
56        */
57       @Override
58       @SuppressWarnings("unchecked")
59       public T get(int x) {
60         int index = indices.getIndex(x);
61         if (index == -1) {
62           return null;
63         } else {
64           return (T) data[index];
65         }
66       }
67     
68       /**
69        * TODO: this can be optimized
70        * 
71        * @see com.ibm.wala.util.intset.IntVector#set(int, int)
72        */
73       @Override
74       public void set(int x, T value) {
75         int index = indices.getIndex(x);
76         if (index == -1) {
77           indices.add(x);
78           index = indices.getIndex(x);
79           ensureCapacity(indices.size() + 1);
80           if (index < (data.length - 1)) {
81             System.arraycopy(data, index, data, index + 1, indices.size() - index);
82           }
83         }
84         data[index] = value;
85       }
86     
87       private void ensureCapacity(int capacity) {
88         if (data.length < capacity + 1) {
89           Object[] old = data;
90           data = new Object[1 + (int) (capacity * indices.getExpansionFactor())];
91           System.arraycopy(old, 0, data, 0, old.length);
92         }
93       }
94     
95       /*
96        * @see com.ibm.wala.util.debug.VerboseAction#performVerboseAction()
97        */
98       @Override
99       public void performVerboseAction() {
100         System.err.println((getClass() + " stats: "));
101         System.err.println(("data.length " + data.length));
102         System.err.println(("indices.size() " + indices.size()));
103     
104       }
105     
106       /*
107        * @see com.ibm.wala.util.intset.IVector#iterator()
108        */
109       @Override
110       public Iterator<T> iterator() {
111         return new Iterator<T>() {
112     
113           int i = 0;
114     
115           @Override
116           public boolean hasNext() {
117             return i < indices.size();
118           }
119     
120           @Override
121           @SuppressWarnings("unchecked")
122           public T next() {
123             if (!hasNext()) {
124               throw new NoSuchElementException();
125             }
126             return (T) data[i++];
127           }
128     
129           @Override
130           public void remove() {
131             // TODO Auto-generated method stub
132             Assertions.UNREACHABLE();
133           }
134     
135         };
136       }
137     
138       /**
139        * @return max i s.t get(i) != null
140        */
141       @Override
142       public int getMaxIndex() throws IllegalStateException {
143         return indices.max();
144       }
145     
146       public int size() {
147         return indices.size();
148       }
149     
150       public IntIterator iterateIndices() {
151         return indices.intIterator();
152       }
153     
154       /**
155        * This iteration _will_ cover all indices even when remove is called while
156        * the iterator is active.
157        */
158       public IntIterator safeIterateIndices() {
159         return MutableSparseIntSet.make(indices).intIterator();
160       }
161     
162       public void clear() {
163         data = new Object[DEF_INITIAL_SIZE];
164         indices = MutableSparseIntSet.makeEmpty();
165       }
166     
167       public void remove(int x) {
168         int index = indices.getIndex(x);
169         if (index == -1) {
170           return;
171         } else {
172           System.arraycopy(data, index + 1, data, index, size() - index - 1);
173           indices.remove(x);
174         }
175       }
176     }
