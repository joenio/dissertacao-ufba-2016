30     /**
31      * Basic analysis; context-insensitive except for newInstance and clone
32      */
33     public class DefaultSSAInterpreter extends DefaultRTAInterpreter implements SSAContextInterpreter {
34     
35       private final CloneInterpreter cloneInterpreter;
36     
37       private final ContextInsensitiveSSAInterpreter defaultInterpreter;
38     
39       public DefaultSSAInterpreter(AnalysisOptions options, AnalysisCache cache) {
40         super(options, cache);
41         cloneInterpreter = new CloneInterpreter();
42         defaultInterpreter = new ContextInsensitiveSSAInterpreter(options, cache);
43       }
44     
45       private SSAContextInterpreter getCFAInterpreter(CGNode node) {
46         if (cloneInterpreter.understands(node)) {
47           return cloneInterpreter;
48         } else {
49           return defaultInterpreter;
50         }
51     
52       }
53     
54       @Override
55       public IR getIR(CGNode node) {
56         return getCFAInterpreter(node).getIR(node);
57       }
58     
59       @Override
60       public int getNumberOfStatements(CGNode node) {
61         return getCFAInterpreter(node).getNumberOfStatements(node);
62       }
63     
64       @Override
65       public Iterator<NewSiteReference> iterateNewSites(CGNode node) {
66         return getCFAInterpreter(node).iterateNewSites(node);
67       }
68     
69       @Override
70       public Iterator<CallSiteReference> iterateCallSites(CGNode node) {
71         return getCFAInterpreter(node).iterateCallSites(node);
72       }
73     
74       @Override
75       public boolean recordFactoryType(CGNode node, IClass klass) {
76         // do nothing; we don't understand factory methods.
77         return false;
78       }
79     
80       @Override
81       public ControlFlowGraph<SSAInstruction, ISSABasicBlock> getCFG(CGNode N) {
82         return getCFAInterpreter(N).getCFG(N);
83       }
84     
85       @Override
86       public DefUse getDU(CGNode node) {
87         return getCFAInterpreter(node).getDU(node);
88       }
89     
90     }
