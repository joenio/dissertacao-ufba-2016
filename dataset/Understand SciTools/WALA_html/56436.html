31       public static final CAstQualifier STATIC = new CAstQualifier("static");
32       public static final CAstQualifier PRIVATE = new CAstQualifier("private");
33       public static final CAstQualifier PROTECTED = new CAstQualifier("protected");
34       public static final CAstQualifier PUBLIC = new CAstQualifier("public");

39         sQualifiers.add(ANNOTATION);

52         sQualifiers.add(CONST);
53       }
54     
55       private static int sNextBitNum = 0;
56     
57       private String fName;
58     
59       private long fBit;
60     
61       public CAstQualifier(String name) {
62         super();
63         fBit = 1L << sNextBitNum++;
64         fName = name;
65         sQualifiers.add(this);
66       }
67     
68       public long getBit() {
69         return fBit;
70       }
71     
72       public String getName() {
73         return fName;
74       }
75     
76       @Override
77       public boolean equals(Object o) {
78         if (!(o instanceof CAstQualifier))
79           return false;
80         CAstQualifier other = (CAstQualifier) o;
81         return other.fName.equals(fName) && (fBit == other.fBit);
82       }
83     
84       @Override
85       public int hashCode() {
86         int result = 37;
87         result = result * 13 + fName.hashCode();
88         return result;
89       }
90     
91       @Override
92       public String toString() {
93         return fName;
94       }
95     }
