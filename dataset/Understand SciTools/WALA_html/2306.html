38     package com.ibm.wala.cast.java.translator.jdt;
39     
40     import java.io.IOException;
41     
42     import com.ibm.wala.cast.java.ipa.callgraph.JavaSourceAnalysisScope;
43     import com.ibm.wala.classLoader.ClassLoaderFactoryImpl;
44     import com.ibm.wala.classLoader.ClassLoaderImpl;
45     import com.ibm.wala.classLoader.IClassLoader;
46     import com.ibm.wala.ipa.callgraph.AnalysisScope;
47     import com.ibm.wala.ipa.cha.IClassHierarchy;
48     import com.ibm.wala.types.ClassLoaderReference;
49     import com.ibm.wala.util.config.SetOfClasses;
50     
51     public class JDTClassLoaderFactory extends ClassLoaderFactoryImpl {
52       private boolean dump;
53       
54       // /**
55       // * A map from ClassLoaderReference to IRTranslatorExtension, so that source files
56       // * in different languages are processed by the right kind of IRTranslatorExtension.
57       // */
58       // that's nice, but there's only one kind of IRTransatorExtension.
59     
60       public JDTClassLoaderFactory(SetOfClasses exclusions) {
61         this(exclusions, false);
62       }
63     
64       public JDTClassLoaderFactory(SetOfClasses exclusions, boolean dump) {
65         super(exclusions);
66         this.dump = dump;
67       }
68     
69       @Override
70       protected IClassLoader makeNewClassLoader(ClassLoaderReference classLoaderReference, IClassHierarchy cha, IClassLoader parent,
71           AnalysisScope scope) throws IOException {
72         if (classLoaderReference.equals(JavaSourceAnalysisScope.SOURCE)) {
73           ClassLoaderImpl cl = makeSourceLoader(classLoaderReference, cha, parent);
74           cl.init(scope.getModules(classLoaderReference));
75           return cl;
76         } else {
77           return super.makeNewClassLoader(classLoaderReference, cha, parent, scope);
78         }
79       }
80     
81       protected JDTSourceLoaderImpl makeSourceLoader(ClassLoaderReference classLoaderReference, IClassHierarchy cha, IClassLoader parent)
82           throws IOException {
83         return new JDTSourceLoaderImpl(classLoaderReference, parent, getExclusions(), cha, dump);
84       }
85     }
