29     public abstract class AstTypeInference extends TypeInference {
30     
31       private final TypeAbstraction booleanType;
32     
33       protected class AstTypeOperatorFactory extends TypeOperatorFactory implements AstInstructionVisitor {
34         @Override
35         public void visitAstLexicalRead(AstLexicalRead inst) {
36           result = new DeclaredTypeOperator(new ConeType(cha.getRootClass()));
37         }
38     
39         @Override
40         public void visitAstLexicalWrite(AstLexicalWrite inst) {
41         }
42     
43         @Override
44         public void visitAstGlobalRead(AstGlobalRead instruction) {
45           result = new DeclaredTypeOperator(new ConeType(cha.getRootClass()));
46         }
47     
48         @Override
49         public void visitAstGlobalWrite(AstGlobalWrite instruction) {
50         }
51     
52         @Override
53         public void visitAssert(AstAssertInstruction instruction) {
54         }
55     
56         @Override
57         public void visitEachElementGet(EachElementGetInstruction inst) {
58           result = new DeclaredTypeOperator(new ConeType(cha.getRootClass()));
59         }
60     
61         @Override
62         public void visitEachElementHasNext(EachElementHasNextInstruction inst) {
63         }
64     
65         @Override
66         public void visitIsDefined(AstIsDefinedInstruction inst) {
67           if (doPrimitives) {
68             result = new DeclaredTypeOperator(booleanType);
69           }
70         }
71     
72         @Override
73         public void visitEcho(AstEchoInstruction inst) {
74     
75         }
76       };
77     
78       public AstTypeInference(IR ir, IClassHierarchy cha, TypeAbstraction booleanType, boolean doPrimitives) {
79         super(ir, doPrimitives);
80         this.booleanType = booleanType;
81       }
82     
83       @Override
84       protected void initialize() {
85         init(ir, new TypeVarFactory(), new AstTypeOperatorFactory());
86       }
87     
88     }
