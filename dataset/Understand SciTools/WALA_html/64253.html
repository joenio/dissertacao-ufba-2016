16     /**
17      * The MethodEditor is the core of the ShrikeBT code rewriting mechanism. To rewrite code, construct a MethodEditor initialized with
18      * the intial code for the method. Then perform a series of passes. In each pass you call beginPass(), insert a number of patches
19      * using the insert...() or replace...() methods, then optionally call applyPatches() to update the code with your changes, then
20      * call endPass(). The end of each pass updates the code in the MethodData that you passed in, so the new code can be extracted from
21      * that MethodData object. Note that if applyPatches() is not called, or it is called but no patches had been inserted, then the
22      * code will not be updated and that pass is essentially aborted.
23      * 
24      * A patch is simply a subclass of MethodEditor.Patch, representing a code sequence to insert into the method. Each patch class
25      * implements one method, emitTo(), which writes the patch code into the code stream using the provided MethodEditor.Output
26      * instance. Anonymous inner classes are very useful for writing patches.
27      * 
28      * Patches can be inserted at the following points:
29      * <ul>
30      * <li>Before the start of the method
31      * <li>Before or after a particular instruction
32      * <li>Replacing a particular instruction
33      * <li>Handling an exception on a particular instruction
34      * <li>Handling an exception anywhere in the method
35      * <li>After the end of the method, where code will not normally be executed, but can be branched to by another patch
36      * </ul>
37      * Patch application is deterministic; if two patches are applied at the same point, then the order of application determines the
38      * order of code generation, in a way specified by the particular application point. See the patch application methods below.
39      * 
40      * MethodEditor relies on labels. A label is an integer representing a point in the code. Labels are valid only during a single
41      * pass; at the end of each pass, instructions are reordered and old labels become invalid. At the beginning of a pass every
42      * instruction in the instructions array is labelled with the index of that instruction in the array. During instrumentation new
43      * labels can be allocated by calling MethodEditor.allocateLabel(); control instructions can be created referring to these new
44      * labels or the existing labels. At the end of a pass, as patch code is spliced into the method body, all instructions are updated
45      * to refer to the new labels which are simply the indices of instructions in the instruction array.
46      */

148       private String getStateMessage(int state) {
149         switch (state) {
150         case BEFORE_PASS:
151           return "This operation can only be performed before or after an editing pass";
152         case DURING_PASS:
153           return "This operation can only be performed during an editing pass";
154         case EMITTING_CODE:
155           return "This operation can only be performed while applying patches and emitting code";
156         case BEFORE_END_PASS:
157           return "This operation can only be performed after applying patches";
158         default:
159           return "This operation cannot be performed in this state";
160         }
161       }

187       static ExceptionHandler[] mergeHandlers(ExceptionHandler[] h1, ExceptionHandler[] h2) {
188         if (h1.length == 0) {
189           return h2;
190         } else if (h2.length == 0) {
191           return h1;
192         } else {
193           ExceptionHandler[] r = new ExceptionHandler[h1.length + h2.length];
194           System.arraycopy(h1, 0, r, 0, h1.length);
195           System.arraycopy(h2, 0, r, h1.length, h2.length);
196           return r;
197         }
198       }

238         /**
239          * Emit an instruction with some exception handlers at the current point in the code.
240          */
241         public void emit(Instruction i, ExceptionHandler[] handlers) {
242           codeChanged = true;
243     
244           int s = newInstructions.size();
245           if (s + 1 > instructionsToBytecodes.length) {
246             int[] t = new int[instructionsToBytecodes.length * 2];
247             System.arraycopy(instructionsToBytecodes, 0, t, 0, instructionsToBytecodes.length);
248             instructionsToBytecodes = t;
249           }
250     
251           instructionsToBytecodes[s] = originalBytecode;
252           newInstructions.add(i);
253           newInstructionHandlers.add(mergeHandlers(handlers, additionalHandlers));
254         }

256         void internalEmitInstruction(IInstruction i) {
257           int s = newInstructions.size();
258           if (s + 1 > instructionsToBytecodes.length) {
259             int[] t = new int[instructionsToBytecodes.length * 2];
260             System.arraycopy(instructionsToBytecodes, 0, t, 0, instructionsToBytecodes.length);
261             instructionsToBytecodes = t;
262           }
263     
264           instructionsToBytecodes[s] = originalBytecode;
265           newInstructions.add(i);
266           newInstructionHandlers.add(additionalHandlers);
267         }

269         /**
270          * Emit a list of instructions at the current point in the code.
271          */
272         public void emit(Instruction[] instrs) {
273           emit(instrs, noHandlers);
274         }
275     
276         /**
277          * Emit a list of instructions with some exception handlers at the current point in the code. All the instructions are covered
278          * by all the handlers.
279          */
280         public void emit(Instruction[] instrs, ExceptionHandler[] handlers) {
281           if (instrs.length == 0) {
282             return;
283           }
284           codeChanged = true;
285     
286           int s = newInstructions.size();
287           if (s + instrs.length > instructionsToBytecodes.length) {
288             int[] t = new int[instructionsToBytecodes.length * 2 + instrs.length];
289             System.arraycopy(instructionsToBytecodes, 0, t, 0, instructionsToBytecodes.length);
290             instructionsToBytecodes = t;
291           }
292     
293           ExceptionHandler[] hs = mergeHandlers(handlers, additionalHandlers);
294           for (int i = 0; i < instrs.length; i++) {
295             instructionsToBytecodes[s + i] = originalBytecode;
296             newInstructions.add(instrs[i]);
297             newInstructionHandlers.add(hs);
298           }
299         }
300       }

366       /**
367        * Insert code to be executed whenever the method is entered. This code is not protected by any exception handlers (other than
368        * handlers declared in the patch).
369        * 
370        * When multiple 'start' patches are given, the last one added is first in execution order.
371        * 
372        * @throws IllegalArgumentException if p is null
373        */
374       public void insertAtStart(Patch p) {
375         if (p == null) {
376           throw new IllegalArgumentException("p is null");
377         }
378         verifyState(DURING_PASS);
379         methodStartPatches = p.insert(methodStartPatches);
380         patchCount++;

383       /**
384        * Insert code to be executed before the instruction. Branches to the instruction will branch to this code. Exception handlers
385        * that cover the instruction will be extended to cover the patch.
386        * 
387        * When multiple 'before' patches are given, the last one added is first in execution order.
388        * 
389        * @throws IllegalArgumentException if p is null
390        */
391       public void insertBefore(int i, Patch p) {
392         if (p == null) {
393           throw new IllegalArgumentException("p is null");
394         }
395         verifyState(DURING_PASS);
396         beforePatches[i] = p.insert(beforePatches[i]);
397         patchCount++;
398       }
399     
400       /**
401        * Insert code to be executed after the instruction. This code will only execute if the instruction "falls through". For example,
402        * code inserted after a "goto" will never be executed. Likewise if the instruction throws an execution the 'after' code will not
403        * be executed. Exception handlers that cover the instruction will be extended to cover the patch.
404        * 
405        * When multiple 'after' patches are given, the last one added is LAST in execution order.
406        */
407       public void insertAfter(int i, Patch p) {
408         verifyState(DURING_PASS);
409         try {
410           if (afterPatches[i] == null) {
411             lastAfterPatches[i] = afterPatches[i] = p.insert(null);
412           } else {
413             lastAfterPatches[i].next = p;
414             lastAfterPatches[i] = p.insert(null);
415           }
416           patchCount++;
417         } catch (ArrayIndexOutOfBoundsException e) {
418           throw new IllegalArgumentException("invalid i");
419         }
420       }
421     
422       /**
423        * Insert code to replace the instruction. Exception handlers that cover the instruction will cover the patch.
424        * 
425        * Multiple replacements are not allowed.
426        * 
427        * @throws NullPointerException if p is null
428        * @throws IllegalArgumentException if p is null
429        */
430       public void replaceWith(int i, Patch p) throws NullPointerException {
431         if (p == null) {
432           throw new IllegalArgumentException("p is null");
433         }
434         verifyState(DURING_PASS);
435         if (replacementPatches[i] != null) {
436           throw new IllegalArgumentException("Instruction " + i + " cannot be replaced more than once");
437         }
438         replacementPatches[i] = p.insert(null);
439         patchCount++;
440       }

442       /**
443        * An "instruction exception handler" handles exceptions generated by a specific instruction (including patch code that may be
444        * inserted before, after, or instead of the instruction in this pass). If the patch code falls through, control resumes with the
445        * next instruction. This exception handler handles exceptions before any exception handler already attached to the instruction.
446        * Furthermore, the patch itself is covered by the exception handlers already attached to the instruction.
447        * 
448        * If multiple instruction exception handlers are given, then the last one added handles the exception first; if an exception is
449        * rethrown, then the next-to-last one added handles that exception, etc.
450        */
451       public void addInstructionExceptionHandler(int i, String catchClass, Patch p) {
452         verifyState(DURING_PASS);
453         try {
454           instructionHandlerPatches[i] = new HandlerPatch(instructionHandlerPatches[i], catchClass, allocateLabel(), p);
455           patchCount++;
456         } catch (ArrayIndexOutOfBoundsException e) {
457           throw new IllegalArgumentException("invalid i: " + i);
458         }
459       }

500       private static ExceptionHandler[] makeExceptionArray(HandlerPatch hp) {
501         if (hp == null) {
502           return noHandlers;
503         } else {
504           int count = 0;
505           for (HandlerPatch hpIterator = hp; hpIterator != null; hpIterator = hpIterator.next) {
506             count++;
507           }
508     
509           ExceptionHandler[] patchedHandlers = new ExceptionHandler[count];
510           count = 0;
511           for (HandlerPatch hpIterator = hp; hpIterator != null; hpIterator = hpIterator.next) {
512             patchedHandlers[count] = new ExceptionHandler(hpIterator.label, hpIterator.catchClass);
513             count++;
514           }
515     
516           return patchedHandlers;
517         }
518       }

520       /**
521        * This method finishes a pass. All code is updated; instructions are reordered and old labels may not be valid.
522        * 
523        * If no patches were issued, we don't need to do anything at all; this case is detected quickly and no updates are made.
524        * 
525        * @return true iff non-trivial patches were applied
526        */
527       public boolean applyPatches() throws IllegalArgumentException {
528         verifyState(DURING_PASS);
529         state = EMITTING_CODE;
530     
531         if (patchCount == 0) {
532           state = BEFORE_END_PASS;
533           // no patches issued; drop out
534           return false;
535         }
536     
537         Output w = new Output(nextLabel);
538     
539         int[] oldInstructionsToNew = new int[instructions.length];
540     
541         w.additionalHandlers = noHandlers;
542         w.originalBytecode = 0;
543         for (Patch p = methodStartPatches; p != null; p = p.next) {
544           p.emitTo(w);
545         }
546     
547         ExceptionHandler[] methodHandlers = makeExceptionArray(methodHandlerPatches);
548         if (methodHandlers.length > 0) {
549           // if a method handler is added, the code has changed even though
550           // the patch might not emit any actual code
551           w.codeChanged = true;
552         }
553     
554         for (int i = 0; i < instructions.length; i++) {
555           ExceptionHandler[] basicHandlers = mergeHandlers(handlers[i], methodHandlers);
556           HandlerPatch hp = instructionHandlerPatches[i];
557     
558           w.emitLabel(i);
559           w.originalBytecode = instructionsToBytecodes[i];
560     
561           w.additionalHandlers = basicHandlers;
562           for (Patch p = beforePatches[i]; p != null; p = p.next) {
563             p.emitTo(w);
564           }
565     
566           w.additionalHandlers = mergeHandlers(makeExceptionArray(hp), basicHandlers);
567           Patch replace = replacementPatches[i];
568           if (replace == null) {
569             oldInstructionsToNew[i] = w.newInstructions.size();
570             w.internalEmitInstruction(instructions[i]);
571           } else {
572             // a patch might delete an instruction, in which case the
573             // code is definitely modified
574             w.codeChanged = true;
575             oldInstructionsToNew[i] = -1;
576             replace.emitTo(w);
577           }
578     
579           w.additionalHandlers = basicHandlers;
580           for (Patch p = afterPatches[i]; p != null; p = p.next) {
581             p.emitTo(w);
582           }
583     
584           if (hp != null) {
585             // if an instruction handler is added, the code has changed even though
586             // the patch might not emit any actual code
587             w.codeChanged = true;
588     
589             GotoInstruction branchOver = GotoInstruction.make(i + 1);
590     
591             w.internalEmitInstruction(branchOver);
592             for (HandlerPatch hpIterator = hp; hpIterator != null; hpIterator = hpIterator.next) {
593               w.additionalHandlers = mergeHandlers(makeExceptionArray(hpIterator.next), basicHandlers);
594               w.emitLabel(hpIterator.label);
595               hpIterator.patch.emitTo(w);
596               w.internalEmitInstruction(branchOver);
597             }
598           }
599         }
600     
601         w.originalBytecode = 0;
602     
603         for (HandlerPatch hpIterator = methodHandlerPatches; hpIterator != null; hpIterator = hpIterator.next) {
604           w.additionalHandlers = makeExceptionArray(hpIterator.next);
605           w.emitLabel(hpIterator.label);
606           hpIterator.patch.emitTo(w);
607         }
608     
609         w.additionalHandlers = noHandlers;
610         for (Patch p = afterMethodPatches; p != null; p = p.next) {
611           p.emitTo(w);
612         }
613     
614         state = BEFORE_END_PASS;
615     
616         if (!w.codeChanged) {
617           return false;
618         }
619     
620         instructions = new Instruction[w.newInstructions.size()];
621         handlers = new ExceptionHandler[instructions.length][];
622         instructionsToBytecodes = new int[instructions.length];
623     
624         w.newInstructions.toArray(instructions);
625         w.newInstructionHandlers.toArray(handlers);
626         System.arraycopy(w.instructionsToBytecodes, 0, instructionsToBytecodes, 0, instructionsToBytecodes.length);
627     
628         int[] labelDefs = w.labelDefs;
629     
630         int[] newInstructionsToOld = new int[instructions.length];
631     
632         for (int i = 0; i < instructions.length; i++) {
633           instructions[i] = instructions[i].redirectTargets(labelDefs);
634           newInstructionsToOld[i] = -1;
635         }
636     
637         // We want to update each exception handler array exactly once
638         IdentityHashMap<ExceptionHandler, Object> adjustedHandlers = null;
639         for (int i = 0; i < handlers.length; i++) {
640           ExceptionHandler[] hs = handlers[i];
641           if (hs.length > 0 && (i == 0 || hs != handlers[i - 1])) {
642             if (adjustedHandlers == null) {
643               adjustedHandlers = new IdentityHashMap<ExceptionHandler, Object>();
644             }
645     
646             for (int j = 0; j < hs.length; j++) {
647               ExceptionHandler h = hs[j];
648               if (!adjustedHandlers.containsKey(h)) {
649                 adjustedHandlers.put(h, null);
650                 h.handler = labelDefs[h.handler]; // breaks invariant of ExceptionHandler: immutable!
651               }
652             }
653           }
654         }
655     
656         if (methodInfo != null) {
657           for (int i = 0; i < oldInstructionsToNew.length; i++) {
658             if (oldInstructionsToNew[i] != -1) {
659               newInstructionsToOld[oldInstructionsToNew[i]] = i;
660             }
661           }
662           methodInfo.update(instructions, handlers, newInstructionsToOld, instructionsToBytecodes);
663         }
664     
665         return true;
666       }

697         /**
698          * @return the index of the current instruction in the code array
699          */
