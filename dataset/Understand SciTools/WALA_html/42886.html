54     public class JDTTypeDictionary extends CAstTypeDictionaryImpl {
55     
56       // TODO: better way of getting type "ObjecT" that doesn't require us to keep AST? although this is similar to
57       // polyglot.
58       protected final AST fAst; // TAGALONG
59     
60       protected final JDTIdentityMapper fIdentityMapper; // TAGALONG
61     
62       /**
63        * 
64        * @param ast Needed to get root type "java.lang.Object"
65        * @param translator
66        */
67       public JDTTypeDictionary(AST ast, JDTIdentityMapper identityMapper) {
68         fAst = ast;
69         fIdentityMapper = identityMapper;
70       }
71     
72       @Override
73       public CAstType getCAstTypeFor(Object astType) {
74     
75         ITypeBinding jdtType = JDT2CAstUtils.getErasedType((ITypeBinding) astType, fAst);
76     
77         CAstType type = super.getCAstTypeFor(astType); // check cache first
78         // Handle the case where we haven't seen an AST decl for some type before
79         // processing a reference. This can certainly happen with classes in byte-
80         // code libraries, for which we never see an AST decl.
81         // In this case, just create a new type and return that.
82         if (type == null) {
83     
84           if (jdtType.isClass() || jdtType.isEnum() || jdtType.isInterface()) // in JDT interfaces are not classes
85             type = new JdtJavaType(jdtType);
86           else if (jdtType.isPrimitive()) {
87             type = JavaPrimitiveTypeMap.lookupType(jdtType.getName());
88           } else if (jdtType.isArray()) {
89             type = new JdtJavaArrayType(jdtType);
90           } else
91             Assertions.UNREACHABLE("getCAstTypeFor() passed type that is not primitive, array, or class?");
92           super.map(astType, type); // put in cache
93         }
94         return type;
95       }
96     
97       private final class JdtJavaArrayType implements CAstType.Array {
98         private final ITypeBinding fEltJdtType;
99     
100         private final CAstType fEltCAstType;
101     
102         private JdtJavaArrayType(ITypeBinding arrayType) {
103           super();
104           fEltJdtType = arrayType.getComponentType();
105           fEltCAstType = getCAstTypeFor(fEltJdtType);
106         }
107     
108         @Override
109         public int getNumDimensions() {
110           return 1; // always 1 for Java
111         }
112     
113         @Override
114         public CAstType getElementType() {
115           return fEltCAstType;
116         }
117     
118         @Override
119         public String getName() {
120           return "[" + fEltCAstType.getName();
121         }
122     
123         @Override
124         @SuppressWarnings("unchecked")
125         public Collection getSupertypes() {
126           if (fEltJdtType.isPrimitive())
127             return Collections.singleton(getCAstTypeFor(fAst.resolveWellKnownType("java.lang.Object")));
128           // TODO: there is no '.isReference()' as in Polyglot: is this right? enum? I think if it's another array it will
129           // just ignore it
130           // TEST DOUBLE ARRAYS! and maybe ask someone?
131           assert fEltJdtType.isArray() || fEltJdtType.isClass() : "Non-primitive, non-reference array element type!";
132           Collection<CAstType> supers = new ArrayList<CAstType>();
133           for (ITypeBinding type : fEltJdtType.getInterfaces()) {
134             supers.add(getCAstTypeFor(type));
135           }
136           if (fEltJdtType.getSuperclass() != null)
137             supers.add(getCAstTypeFor(fEltJdtType.getSuperclass()));
138           return supers;
139         }
140       }
141     
142       public final class JdtJavaType implements JavaType {
143         private final ITypeBinding fType;
144     
145         private Collection<CAstType> fSuperTypes = null;
146     
147         @Override
148         public String toString() {
149           return super.toString() + ":" + getName();
150         }
151     
152         public JdtJavaType(ITypeBinding type) {
153           super();
154           fType = type;
155         }
156     
157         @Override
158         public String getName() {
159           return fIdentityMapper.getTypeRef(fType).getName().toString();
160         }
161     
162         @Override
163         @SuppressWarnings("unchecked")
164         public Collection getSupertypes() {
165           if (fSuperTypes == null) {
166             buildSuperTypes();
167           }
168           return fSuperTypes;
169         }
170     
171         private void buildSuperTypes() {
172           // TODO this is a source entity, but it might actually be the root type
173           // (Object), so assume # intfs + 1
174           ITypeBinding superType = (fType.getSuperclass() == null) ? fAst.resolveWellKnownType("java.lang.Object") : fType
175               .getSuperclass();
176           int N = fType.getInterfaces().length + 1;
177     
178           fSuperTypes = new ArrayList<CAstType>(N);
179           // Following assumes that noone can call getSupertypes() before we have
180           // created CAstType's for every type in the program being analyzed.
181           fSuperTypes.add(getCAstTypeFor(superType));
182           for (ITypeBinding t : fType.getInterfaces())
183             fSuperTypes.add(getCAstTypeFor(t));
184         }
185     
186         @Override
187         public Collection<CAstQualifier> getQualifiers() {
188           return JDT2CAstUtils.mapModifiersToQualifiers(fType.getModifiers(), fType.isInterface(), fType.isAnnotation());
189         }
190     
191         @Override
192         public boolean isInterface() {
193           return fType.isInterface();
194         }
195       }
