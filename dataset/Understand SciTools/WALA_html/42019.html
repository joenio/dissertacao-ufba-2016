11     package com.ibm.wala.cast.js.html;
12     
13     import java.io.IOException;
14     import java.io.InputStreamReader;
15     import java.io.Reader;
16     import java.net.URL;
17     
18     import com.ibm.wala.cast.tree.CAstSourcePositionMap.Position;
19     import com.ibm.wala.cast.tree.impl.AbstractSourcePosition;
20     
21     public class RangeFileMapping implements FileMapping {
22     
23       public final static class Range {
24         private final int rangeStart;
25         private final int rangeEnd;
26         private final int rangeStartingLine;
27         private final int rangeEndingLine;
28     
29         public boolean includes(Position offset) {
30           return 
31             offset.getFirstOffset() != -1? 
32             rangeStart <= offset.getFirstOffset() && offset.getLastOffset() <= rangeEnd:
33             rangeStartingLine <= offset.getFirstLine() && 
34               (offset.getLastLine() == -1? offset.getFirstLine(): offset.getLastLine()) <= rangeEndingLine;
35         }
36     
37         public Range(int rangeStart, int rangeEnd, int rangeStartingLine, int rangeEndingLine) {
38           super();
39           this.rangeStart = rangeStart;
40           this.rangeEnd = rangeEnd;
41           this.rangeStartingLine = rangeStartingLine;
42           this.rangeEndingLine = rangeEndingLine;
43         }
44     
45         public int getStart() {
46           return rangeStart;
47         }
48     
49         public int getEnd() {
50           return rangeEnd;
51         }
52     
53         public int getStartingLine() {
54           return rangeStartingLine;
55         }
56     
57         public int getEndingLine() {
58           return rangeEndingLine;
59         }
60          
61         @Override
62         public String toString() {
63           return "{"+rangeStart+"->"+rangeEnd+"}";
64         }
65       }
66     
67       private final Range range;
68       private final URL includedURL;
69       private final Position includePosition;
70         
71       public RangeFileMapping(int rangeStart, int rangeEnd, int rangeStartingLine, int rangeEndingLine, Position parentPosition, URL url) {
72         assert parentPosition != null;
73         this.range = new Range(rangeStart, rangeEnd, rangeStartingLine, rangeEndingLine);
74         this.includePosition = parentPosition;
75         includedURL = url;
76       }
77     
78       @Override
79       public String toString() {
80         return range + ":" + includePosition;
81       }
82       
83       @Override
84       public IncludedPosition getIncludedPosition(final Position offset) {
85         if (range.includes(offset)) {
86           class Pos extends AbstractSourcePosition implements IncludedPosition {
87             @Override
88             public int getFirstLine() {
89               // line numbers are decreed to start at 1, rather than 0
90               return offset.getFirstLine() - range.getStartingLine() + 1;
91             }
92             @Override
93             public int getLastLine() {
94               return offset.getLastLine() == -1? -1: offset.getLastLine()-range.getStartingLine()+1;
95             }
96             @Override
97             public int getFirstCol() {
98               return offset.getFirstCol();
99             }
100             @Override
101             public int getLastCol() {
102               return offset.getLastCol();
103             }
104             @Override
105             public int getFirstOffset() {
106               return offset.getFirstOffset() == -1? -1: offset.getFirstOffset()-range.getStart();
107             }
108             @Override
109             public int getLastOffset() {
110               return offset.getLastOffset() == -1? -1: offset.getLastOffset()-range.getStart();
111             }
112             @Override
113             public URL getURL() {
114               return includedURL;
115             }
116             @Override
117             public Reader getReader() throws IOException {
118               return RangeFileMapping.this.getInputStream();
119              }
120             @Override
121             public Position getIncludePosition() {
122               return includePosition;
123             }
124             @Override
125             public String toString() {
126               return "[include:"+includePosition+"]"+super.toString();
127             }
128           };
129           
130           return new Pos();
131         } else {
132           return null;
133         }
134       }
135     
136        public Reader getInputStream() throws IOException {
137          return new InputStreamReader(includedURL.openStream());
138       }
139     
140     }
