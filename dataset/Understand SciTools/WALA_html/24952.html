32     /**
33      * Computes interprocedural field accesses for a given method.
34      * 
35      * @author Martin Seidel
36      * @author Juergen Graf <juergen.graf@gmail.com>
37      *
38      */
39     public final class ModRefFieldAccess {
40     	
41     	private class TwoMaps {
42     		private Map<IClass, Set<IField>> mods;
43     		private Map<IClass, Set<IField>> refs;
44     		
45     		public TwoMaps(Map<IClass, Set<IField>> mods, Map<IClass, Set<IField>> refs) {
46     			this.mods = mods;
47     			this.refs = refs;
48     			if (mods == null) {
49     				this.mods = new HashMap<IClass, Set<IField>>();
50     			}
51     			if (refs == null) {
52     				this.refs = new HashMap<IClass, Set<IField>>();
53     			}
54     		}
55     		
56     		public Map<IClass, Set<IField>> getMods() {
57     			return mods;
58     		}
59     
60     		public Map<IClass, Set<IField>> getRefs() {
61     			return refs;
62     		}
63     	}
64     
65     	private final CallGraph cg;
66     	
67     	private Map<CGNode, Map<IClass, Set<IField>>> mods;
68     	private Map<CGNode, Map<IClass, Set<IField>>> refs;
69     	private Map<CGNode, Map<IClass, Set<IField>>> tmods;
70     	private Map<CGNode, Map<IClass, Set<IField>>> trefs;
71     	
72     	private List<CGNode> done;
73     
74     	private ModRefFieldAccess(CallGraph cg) {
75     		this.cg = cg;
76     		this.refs = new HashMap<CGNode, Map<IClass, Set<IField>>>();
77     		this.mods = new HashMap<CGNode, Map<IClass, Set<IField>>>();
78     		this.trefs = new HashMap<CGNode, Map<IClass, Set<IField>>>();
79     		this.tmods = new HashMap<CGNode, Map<IClass, Set<IField>>>();
80     		this.done = new LinkedList<CGNode>();
81     	}
82     	
83     	public static ModRefFieldAccess compute(CallGraph cg) {
84     		ModRefFieldAccess fa = new ModRefFieldAccess(cg);
85     		fa.run();
86     		
87     		return fa;
88     	}
89     	
90     	public Map<IClass, Set<IField>> getMod(CGNode node) {
91     		return mods.get(node);
92     	}
93     	
94     	public Map<IClass, Set<IField>> getRef(CGNode node) {
95     		return refs.get(node);
96     	}
97     	
98     	public Map<IClass, Set<IField>> getTransitiveMod(CGNode node) {
99     		return tmods.get(node);
100     	}
101     	
102     	public Map<IClass, Set<IField>> getTransitiveRef(CGNode node) {
103     		return trefs.get(node);
104     	}
105     	
106     	private void run() {
107     
108     		for (CGNode cgNode : cg) {
109     			if (!refs.containsKey(cgNode)) {
110     				refs.put(cgNode, new HashMap<IClass, Set<IField>>());
111     			}
112     			if (!mods.containsKey(cgNode)) {
113     				mods.put(cgNode, new HashMap<IClass, Set<IField>>());
114     			}
115     			
116     			final IR ir = cgNode.getIR();
117     			
118     			if (ir == null) {
119     				continue;
120     			}
121     			
122     			for (Iterator<SSAInstruction> it = ir.iterateNormalInstructions(); it.hasNext();) {
123     				SSAInstruction instr = it.next();
124     				if (instr instanceof SSAGetInstruction) {
125     					SSAGetInstruction get = (SSAGetInstruction) instr;
126     					FieldReference fref = get.getDeclaredField();
127     					IField field = cg.getClassHierarchy().resolveField(fref);
128     					if(field != null) {
129     						IClass cls = field.getDeclaringClass();
130     						if(cls != null) {
131     							if (!refs.get(cgNode).containsKey(cls)) {
132     								refs.get(cgNode).put(cls, new HashSet<IField>());
133     							}
134     							refs.get(cgNode).get(cls).add(field);
135     						}
136     					}
137     				} else if (instr instanceof SSAPutInstruction) {
138     					SSAPutInstruction put = (SSAPutInstruction) instr;
139     					FieldReference fput  = put.getDeclaredField();
140     					IField field = cg.getClassHierarchy().resolveField(fput);
141     					if(field != null) {
142     						IClass cls = field.getDeclaringClass();
143     						if(cls != null) {
144     							if (!mods.get(cgNode).containsKey(cls)) {
145     								mods.get(cgNode).put(cls, new HashSet<IField>());
146     							}
147     							mods.get(cgNode).get(cls).add(field);
148     						}
149     					}
150     				}
151     			}
152     		}
153     		
154     		recAdd(cg.getFakeRootNode());
155     		
156     	}
157     	
158     	private TwoMaps recAdd(CGNode node) {
159     		if (!trefs.containsKey(node)) {
160     			trefs.put(node, new HashMap<IClass, Set<IField>>());
161     		}
162     		if (!tmods.containsKey(node)) {
163     			tmods.put(node, new HashMap<IClass, Set<IField>>());
164     		}
165     		
166     		final IR ir = node.getIR();
167     		if (ir != null) {
168     			for (Iterator<SSAInstruction> it = ir.iterateNormalInstructions(); it.hasNext();) {
169     				SSAInstruction instr = it.next();
170     				if (instr instanceof SSAGetInstruction) {
171     					SSAGetInstruction get = (SSAGetInstruction) instr;
172     					FieldReference fref = get.getDeclaredField();
173     					IField field = cg.getClassHierarchy().resolveField(fref);
174     					if(field != null) {
175     						IClass cls = field.getDeclaringClass();
176     						if(cls != null) {
177     							if (!trefs.get(node).containsKey(cls)) {
178     								trefs.get(node).put(cls, new HashSet<IField>());
179     							}
180     							trefs.get(node).get(cls).add(field);
181     						}
182     					}
183     				} else if (instr instanceof SSAPutInstruction) {
184     					SSAPutInstruction put = (SSAPutInstruction) instr;
185     					FieldReference fput  = put.getDeclaredField();
186     					IField field = cg.getClassHierarchy().resolveField(fput);
187     					if(field != null) {
188     						IClass cls = field.getDeclaringClass();
189     						if(cls != null) {
190     							if (!tmods.get(node).containsKey(cls)) {
191     								tmods.get(node).put(cls, new HashSet<IField>());
192     							}
193     							tmods.get(node).get(cls).add(field);
194     						}
195     					}
196     				}
197     			}
198     		}
199     		
200     		for (Iterator<CGNode> it = cg.getSuccNodes(node); it.hasNext();) {
201     			CGNode n = it.next();
202     			if (!done.contains(n)) {
203     				done.add(n);
204     				TwoMaps t = recAdd(n);
205     				for (IClass c : t.getRefs().keySet()) {
206     					if (trefs.get(node).containsKey(c)) {
207     						trefs.get(node).get(c).addAll(t.getRefs().get(c));
208     					} else {
209     						trefs.get(node).put(c, t.getRefs().get(c));
210     					}
211     				}
212     				for (IClass c : t.getMods().keySet()) {
213     					if (tmods.get(node).containsKey(c)) {
214     						tmods.get(node).get(c).addAll(t.getMods().get(c));
215     					} else {
216     						tmods.get(node).put(c, t.getMods().get(c));
217     					}
218     				}
219     				
220     			}
221     		}
222     		
223     		return new TwoMaps(tmods.get(node), trefs.get(node));
224     	}
