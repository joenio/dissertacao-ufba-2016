60     /**
61      *  Instructions can be added in a non-ascending manner.
62      *
63      *  The later position of the instruction is determined by it's iindex.
64      *  Additionally this Summary may be instructed to prune unnecessary 
65      *  instructions.
66      *
67      *  However don't go berserk with the iindex as this will consume loads of
68      *  memory.
69      *
70      *  You can get an ordinary MethodSummary using the {@link #getMethodSummary()}-Method.
71      *
72      *  It extends the MethodSummarys capabilities by the functions:
73      *  * {@link #getStatementAt(int)}
74      *  * {@link #getStatementsAfter(int)}
75      *  * {@link #reserveProgramCounters(int)}
76      *  * {@link #allowReserved(boolean)}
77      *
78      *  @see com.ibm.wala.dalvik.ssa.AndroidModelInstructionFectory
79      *  @see com.ibm.wala.dalvik.ipa.callgraph.impl.DexFakeRootMethod
80      *  @see com.ibm.wala.ipa.summaries.MethodSummary
81      *
82      *  @author     Tobias Blaschke <code@tobiasblaschke.de>
83      *  @since      2013-09-08
84      */

99             public Reserved () { super(SSAInstruction.NO_INDEX); }

101                 throw new IllegalStateException();

107         }

124         /**
125          *  @param  programCounter  the ProgramCounter to retrieve the Instruction from
126          *  @return The instruction or null if there is none
127          *  @throws IllegalArgumentException if the ProgramCounter is negative
128          */
129         public SSAInstruction getStatementAt(int programCounter) {
130             if (programCounter < 0) {
131                 throw new IllegalArgumentException("Program-Counter may not be negative!");
132             }
133             if (this.instructions.size() <= programCounter) {
134                 return null;
135             }
136             if (this.instructions.get(programCounter).equals(RESERVED)) {
137                 return null;
138             }
139             return this.instructions.get(programCounter);
140         }

142         /**
143          *  Reserves an amount of ProgramCounters for later use.
144          *
145          *  This method reserves a count of ProgramCounters and thus affects the value
146          *  returned by getNextProgramCounter. It also marks these ProgramCounters as
147          *  reserved so you can't use them unless you explicitly allow it by 
148          *  {@link #allowReserved(boolean)}.
149          *
150          *  @param  count   The amount of ProgramCounters to reserve ongoing from the
151          *      current ProgramCounter
152          *  @throws IllegalArgumentException if the count is negative (a count of zero 
153          *      is however ok)
154          */
155         public void reserveProgramCounters(int count) {
156             if (count < 0) {
157                 throw new IllegalArgumentException("The count of ProgramCounters to reserve may not be negative");
158             }
159             for (int i=0; i<count; ++i) {
160                 instructions.add(RESERVED);
161             }
162             currentProgramCounter += count;
163         }

165         /**
166          *  (Dis-)allows the usage of reserved ProgramCounters.
167          *
168          *  The setting of this function defaults to disallow upon class creation
169          *
170          *  @param  enable  A value of true allows the usage of all reserved ProgramCounters
171          *  @return the previous value of allowReserved 
172          */
173         public boolean allowReserved(boolean enable) {
174             boolean prev = this.allowReservedPC;
175             this.allowReservedPC = enable;
176             return prev;
177         }

179         /**
180          *  Returns if the ProgramCounter is reserved.
181          *
182          *  @param programCounter   the ProgramCounter in question
183          *  @return true if the position is reserved
184          *  @throws IllegalArgumentException if the ProgramCounter is negative
185          */
186         public boolean isReserved(int programCounter) {
187             if (programCounter < 0) {
188                 throw new IllegalArgumentException("The Program-Counter may not be negative");
189             }
190     
191             if (instructions.size() - 1 < programCounter) return false;
192             if (instructions.get(programCounter) == null) return false;
193             return (instructions.get(programCounter).equals(RESERVED));
194         }

196         /**
197          *  Returns if the ProgramCounter is writable.
198          *
199          *  The ProgramCounter is not writable if there is already an instruction located at that 
200          *  ProgramCounter or if the reserved ProgramCounters forbid usage. Thus the answer may
201          *  depend on the setting of {@link #allowReserved(boolean)}.
202          *
203          *  @param programCounter   the ProgramCounter in question
204          *  @return true if you may write to the location
205          *  @throws IllegalArgumentException if the ProgramCounter is negative
206          */
207         public boolean isFree(int programCounter) {
208             if (programCounter < 0) {
209                 throw new IllegalArgumentException("The Program-Counter may not be negative");
210             }
211     
212             if (instructions.size() - 1 < programCounter) return true;
213             if (instructions.get(programCounter) == null) return true;
214             if (instructions.get(programCounter).equals(RESERVED)) return false;
215             return false;
216         }
217     
218         /**
219          *  Not exactly dual to {@link #isFree(boolean)}.
220          *
221          *  Returns whether an instruction is located at ProgramCounter. Thus it is a  shortcut
222          *  to {@link #getStatementAt(int)} != null.
223          *  <p>
224          *  It is not the exact dual to {@link #isFree(boolean)} as it does not consider reserved
225          *  ProgramCounters.
226          *
227          *  @param programCounter   the ProgramCounter in question
228          *  @return true if there's an instruction at that program counter
229          *  @throws IllegalArgumentException if the ProgramCounter is negative
230         */
231         public boolean isUsed(int programCounter) {
232             if (programCounter < 0) {
233                 throw new IllegalArgumentException("The Program-Counter may not be negative");
234             }
235     
236             if (instructions.size() - 1 < programCounter) return false;
237             if (instructions.get(programCounter) == null) return false;
238             if (instructions.get(programCounter).equals(RESERVED)) return false;
239             return true;
240         }

242         /**
243          *  Like {@link addStatement(SSAInstructionWithPC <? extends SSAInstruction>) but may replace an existing one.
244          *
245          *  @param  statement    The statement to add without care of overwriting
246          *  @return true if a statement has actually been overwritten 
247          *  @throws IllegalStateException if you may not write to the ProgramCounter due to
248          *      the setting of {@link #allowReserved(boolean)} or {@link #getMethodSummary()} has
249          *      been called and thus this summary got locked.
250          *  @throws NullPointerException if statement is null
251          *  @throws IllegalArgumentException if the statement has set an invalid ProgramCounter
252          */
253         public boolean overwriteStatement(SSAInstruction statement) {
254             if (this.locked) {
255                 throw new IllegalStateException("Summary locked due to call to getMethodSummary().");
256             }
257             if (statement == null) {
258                 throw new NullPointerException("Statement is null!");
259             }
260             if (statement.iindex < 0) {
261                 throw new IllegalArgumentException("Statement has a negative iindex");
262             }
263             if ((!this.allowReservedPC) && isReserved(statement.iindex)) {
264                 throw new IllegalStateException("ProgramCounter " + statement.iindex + " is reserved! Use allowReserved(true).");
265             }
266             if (statement.iindex > this.currentProgramCounter) {
267                 throw new IllegalArgumentException("IIndex " + statement.iindex + " is greater than currentProgramCounter. Use getNextProgramCounter.");
268             }
269     
270             boolean didOverwrite = isUsed(statement.iindex);
271             while (this.instructions.size() - 1 < statement.iindex) this.instructions.add(null);
272             if (DEBUG) { System.err.printf("Setting {} to {}", statement.iindex, statement); }
273             this.instructions.set(statement.iindex, statement);
274             return didOverwrite;
275         }

277         /**
278          *  Generates the MethodSummary and locks class.
279          *
280          *  @throws IllegalStateException if you altered the referenced (by constructor) summary
281          *  @return the finished MethodSummary
282          */
283         public MethodSummary getMethodSummary() {
284             if (locked) {
285                 // Already generated
286                 return this.summary;
287             }
288             if (summary.getNumberOfStatements() > 0) {
289                 throw new IllegalStateException("Meanwhile Statements have been added to the summary given " +
290                         "to the constructor. This behavior is not supported!");
291             }
292             this.locked = true;
293             for (int i = 0; i < this.instructions.size(); ++i) {
294                 final SSAInstruction inst = this.instructions.get(i);
295                 if (inst == null) {
296                   if (DEBUG) { System.err.printf("No instruction at iindex {}", i); }
297                   this.summary.addStatement(null);
298                 } else if (inst == RESERVED) {
299                   // replace reserved slots by 'goto next' statements
300                   this.summary.addStatement(new SSAGotoInstruction(i, i+1));
301                 } else {
302                   if (DEBUG) { System.err.printf("Adding @{}: ", inst); }
303                   this.summary.addStatement(inst);
304                 }
305             }
306             
307             // Let the GC free instructions..
308             this.instructions = null;
309     
310             return this.summary;
311         }

331         /**
332          *  Adds a statement to the MethodSummary.
333          *
334          *  @param  statement    The statement to be added
335          *  @throws IllegalStateException if you may not write to the ProgramCounter due to
336          *      the setting of {@link #allowReserved(boolean)} or {@link #getMethodSummary()} has
337          *      been called and thus this summary got locked.
338          *  @throws NullPointerException if statement is null
339          *  @throws IllegalArgumentException if the statement has set an invalid ProgramCounter or
340          *      if there is already a statement at the statements iindex. In this case you can use 
341          *      {@link #overwritetatement(SSAInstruction)}.
342          */
343         public void addStatement(SSAInstruction statement) {
344             if (isUsed(statement.iindex)) {
345                 throw new IllegalArgumentException("ProgramCounter " + statement.iindex + " is in use! By " +
346                         getStatementAt(statement.iindex) + " Use overwriteStatement().");
347             }
348          
349             overwriteStatement(statement);
350         }

429         /**
430          *  Gets you a non-reserved ProgramCounter you can write to.
431          *
432          *  This function returns the next ProgramCounter for which not({@link #isUsed(ProgramCounter)})
433          *  holds. Thus it will _not_ give you a ProgramCounter which is reserved even if you enabled
434          *  writing to reserved ProgramCounters using {@link #allowReserved(boolean)}! You'll have to
435          *  keep track of them on your own.
436          *
437          * @return A non-reserved writable ProgramCounter
438          */
439         public int getNextProgramCounter() {
440           while (isUsed(this.currentProgramCounter) || isReserved(this.currentProgramCounter)) {
441             this.currentProgramCounter++;
442           }
443           while (this.instructions.size() < this.currentProgramCounter) this.instructions.add(null);
444           return this.currentProgramCounter;
445         }

508          *  Get all statements added to the Summary.
509          *
510          *  This builds a copy of the internal list and may contain 'null'-values if no
511          *  instruction has been placed at a particular pc.
512          *
513          *  @return The statements of the summary
514          */
515         public SSAInstruction[] getStatements() {
516             SSAInstruction[] ret = new SSAInstruction[this.instructions.size()];
517             ret = this.instructions.toArray(ret);
518             
519             // Remove Reserved
520             for (int i=0; i<ret.length; ++i) {
521                 if (ret[i].equals(RESERVED)) {
522                     ret[i] = null;
523                 }
524             }
525             return ret;
526         }
