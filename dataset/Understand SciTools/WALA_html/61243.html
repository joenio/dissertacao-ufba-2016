11     package com.ibm.wala.ipa.cfg;
12     
13     import com.ibm.wala.cfg.ControlFlowGraph;
14     import com.ibm.wala.ipa.callgraph.CGNode;
15     import com.ibm.wala.ipa.callgraph.CallGraph;
16     import com.ibm.wala.ssa.ISSABasicBlock;

20     /**
21      * Interprocedural control-flow graph.
22      * 
23      * TODO: think about a better implementation; perhaps a lazy view of the constituent CFGs Lots of ways this can be optimized?
24      */
25     public class InterproceduralCFG extends AbstractInterproceduralCFG<ISSABasicBlock> {
26     
27       public InterproceduralCFG(CallGraph CG) {
28         super(CG);
29       }
30     
31       public InterproceduralCFG(CallGraph cg, Filtersection<CGNode> filtersection) {
32         super(cg, filtersection);
33       }
34     
35       /**
36        * @return the cfg for n, or null if none found
37        * @throws IllegalArgumentException if n == null
38        */
39       @Override
40       public ControlFlowGraph<SSAInstruction, ISSABasicBlock> getCFG(CGNode n) throws IllegalArgumentException {
41         if (n == null) {
42           throw new IllegalArgumentException("n == null");
43         }
44         if (n.getIR() == null) {
45           return null;
46         }
47         ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg = n.getIR().getControlFlowGraph();
48         if (cfg == null) {
49           return null;
50         }
51     
52         return cfg;
53       }
54     
55     }
