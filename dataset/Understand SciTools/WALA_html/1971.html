11     package com.ibm.wala.cast.js.ipa.callgraph;
12     
13     import com.ibm.wala.cast.ir.translator.AstTranslator.AstLexicalInformation;
14     import com.ibm.wala.cast.js.ipa.summaries.JavaScriptConstructorFunctions.JavaScriptConstructor;
15     import com.ibm.wala.classLoader.CallSiteReference;
16     import com.ibm.wala.classLoader.IMethod;
17     import com.ibm.wala.ipa.callgraph.CGNode;
18     import com.ibm.wala.ipa.callgraph.Context;
19     import com.ibm.wala.ipa.callgraph.ContextSelector;
20     import com.ibm.wala.ipa.callgraph.propagation.InstanceKey;
21     import com.ibm.wala.ipa.callgraph.propagation.cfa.OneLevelSiteContextSelector;
22     import com.ibm.wala.ipa.callgraph.propagation.cfa.nCFAContextSelector;
23     import com.ibm.wala.util.intset.IntSet;
24     
25     public class JavaScriptConstructorContextSelector implements ContextSelector {
26       private final ContextSelector base;
27     
28       /**
29        * for generating contexts with one-level of call strings, to match standard
30        * Andersen's heap abstraction
31        */
32       private final nCFAContextSelector oneLevelCallStrings;
33     
34       private final OneLevelSiteContextSelector oneLevelCallerSite;
35       
36       public JavaScriptConstructorContextSelector(ContextSelector base) {
37         this.base = base;
38         this.oneLevelCallStrings = new nCFAContextSelector(1, base);
39         this.oneLevelCallerSite = new OneLevelSiteContextSelector(base);
40       }
41     
42       @Override
43       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
44         return base.getRelevantParameters(caller, site);
45       }
46     
47       @Override
48       public Context getCalleeTarget(final CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] receiver) {
49         if (callee instanceof JavaScriptConstructor) {
50           final Context oneLevelCallStringContext = oneLevelCallStrings.getCalleeTarget(caller, site, callee, receiver);
51           if (AstLexicalInformation.hasExposedUses(caller, site)) {
52             // use a caller-site context, to enable lexical scoping lookups (via caller CGNode)
53             return oneLevelCallerSite.getCalleeTarget(caller, site, callee, receiver);
54           } else {
55             // use at least one-level of call-string sensitivity for constructors
56             // always
57             return oneLevelCallStringContext;
58           }
59         } else {
60           return base.getCalleeTarget(caller, site, callee, receiver);
61         }
62       }
63     
64     }
