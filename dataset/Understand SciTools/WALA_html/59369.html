41      * @author Manu Sridharan
42      * 
43      */
44     public class MatchBarLabel implements IFlowLabel {
45     
46       private static final MatchBarLabel theInstance = new MatchBarLabel();
47     
48       private MatchBarLabel() {
49       }
50     
51       public static MatchBarLabel v() {
52         return theInstance;
53       }
54     
55       /*
56        * @see demandGraph.IFlowLabel#bar()
57        */
58       @Override
59       public MatchLabel bar() {
60         return MatchLabel.v();
61       }
62     
63       /*
64        * @see demandGraph.IFlowLabel#visit(demandGraph.IFlowLabel.IFlowLabelVisitor,
65        *      java.lang.Object)
66        */
67       @Override
68       public void visit(IFlowLabelVisitor v, Object dst) throws IllegalArgumentException {
69         if (v == null) {
70           throw new IllegalArgumentException("v == null");
71         }
72         v.visitMatchBar(this, dst);
73       }
74     
75       @Override
76       public boolean isBarred() {
77         return true;
78       }
79     
80     }
