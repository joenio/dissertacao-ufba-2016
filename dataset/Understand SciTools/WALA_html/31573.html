31     /**
32      * A representation of an entrypoint in the call graph.
33      */
34     public abstract class Entrypoint implements BytecodeConstants {
35     
36       /**
37        * The method to be called
38        */
39       protected final IMethod method;
40     
41       /**
42        * @param method the method to be called for this entrypoint
43        */
44       protected Entrypoint(IMethod method) {
45         if (method == null) {
46           throw new IllegalArgumentException("method is null");
47         }
48         this.method = method;
49         assert method.getDeclaringClass() != null : "null declaring class";
50       }
51     
52       protected Entrypoint(MethodReference method, IClassHierarchy cha) {
53         if (cha == null) {
54           throw new IllegalArgumentException("cha is null");
55         }
56         IMethod m = cha.resolveMethod(method);
57         if (m == null) {
58           Assertions.UNREACHABLE("could not resolve " + method);
59         }
60         this.method = m;
61       }
62     
63       /**
64        * Create a call site reference representing a call to this entrypoint
65        * 
66        * @param programCounter the bytecode index of the synthesize call
67        * @return the call site reference, or null if failed to find entrypoint
68        */
69       public CallSiteReference makeSite(int programCounter) {
70     
71         if (method.getSelector().equals(MethodReference.clinitSelector)) {
72           assert method.isStatic();
73           return CallSiteReference.make(programCounter, method.getReference(), IInvokeInstruction.Dispatch.STATIC);
74         } else if (method.getSelector().equals(MethodReference.initSelector)) {
75           assert !method.isStatic();
76           return CallSiteReference.make(programCounter, method.getReference(), IInvokeInstruction.Dispatch.SPECIAL);
77         } else {
78           if (method.getDeclaringClass().isInterface()) {
79             return CallSiteReference.make(programCounter, method.getReference(), IInvokeInstruction.Dispatch.INTERFACE);
80           } else {
81             if (method.isStatic()) {
82               return CallSiteReference.make(programCounter, method.getReference(), IInvokeInstruction.Dispatch.STATIC);
83             } else {
84               return CallSiteReference.make(programCounter, method.getReference(), IInvokeInstruction.Dispatch.VIRTUAL);
85             }
86           }
87         }
88       }
89     
90       /**
91        * Add allocation statements to the fake root method for each possible value of parameter i. If necessary, add a phi to combine
92        * the values.
93        * 
94        * @return value number holding the parameter to the call; -1 if there was some error
95        */
96       protected int makeArgument(AbstractRootMethod m, int i) {
97         TypeReference[] p = getParameterTypes(i);
98         if (p.length == 0) {
99           return -1;
100         } else if (p.length == 1) {
101           if (p[0].isPrimitiveType()) {
102             return m.addLocal();
103           } else {
104             SSANewInstruction n = m.addAllocation(p[0]);
105             return (n == null) ? -1 : n.getDef();
106           }
107         } else {
108           int[] values = new int[p.length];
109           int countErrors = 0;
110           for (int j = 0; j < p.length; j++) {
111             SSANewInstruction n = m.addAllocation(p[j]);
112             int value = (n == null) ? -1 : n.getDef();
113             if (value == -1) {
114               countErrors++;
115             } else {
116               values[j - countErrors] = value;
117             }
118           }
119           if (countErrors > 0) {
120             int[] oldValues = values;
121             values = new int[oldValues.length - countErrors];
122             System.arraycopy(oldValues, 0, values, 0, values.length);
123           }
124     
125           TypeAbstraction a;
126           if (p[0].isPrimitiveType()) {
127             a = PrimitiveType.getPrimitive(p[0]);
128             for (i = 1; i < p.length; i++) {
129               a = a.meet(PrimitiveType.getPrimitive(p[i]));
130             }
131           } else {
132             IClassHierarchy cha = m.getClassHierarchy();
133             IClass p0 = cha.lookupClass(p[0]);
134             a = new ConeType(p0);
135             for (i = 1; i < p.length; i++) {
136               IClass pi = cha.lookupClass(p[i]);
137               a = a.meet(new ConeType(pi));
138             }
139           }
140     
141           return m.addPhi(values);
142         }
143       }
144     
145       @Override
146       public boolean equals(Object obj) {
147         // assume these are managed canonically
148         return this == obj;
149       }
150     
151       /**
152        * Add a call to this entrypoint from the fake root method
153        * 
154        * @param m the Fake Root Method
155        * @return the call instruction added, or null if the operation fails
156        */
157       public SSAAbstractInvokeInstruction addCall(AbstractRootMethod m) {
158         int paramValues[];
159         CallSiteReference site = makeSite(0);
160         if (site == null) {
161           return null;
162         }
163         paramValues = new int[getNumberOfParameters()];
164         for (int j = 0; j < paramValues.length; j++) {
165           paramValues[j] = makeArgument(m, j);
166           if (paramValues[j] == -1) {
167             // there was a problem
168             return null;
169           }
170         }
171     
172         return m.addInvocation(paramValues, site);
173       }
174     
175       /**
176        * @return the method this call invokes
177        */
178       public IMethod getMethod() {
179         return method;
180       }
181     
182       /**
183        * @return types to allocate for parameter i; for non-static methods, parameter 0 is "this"
184        */
185       public abstract TypeReference[] getParameterTypes(int i);
186     
187       /**
188        * @return number of parameters to this call, including "this" for non-statics
189        */
190       public abstract int getNumberOfParameters();
191     
192       @Override
193       public String toString() {
194         StringBuffer result = new StringBuffer(method.toString());
195         result.append("(");
196         for (int i = 0; i < getNumberOfParameters() - 1; i++) {
197           result.append(Arrays.toString(getParameterTypes(i)));
198           result.append(",");
199         }
200         if (getNumberOfParameters() > 0) {
201           result.append(Arrays.toString(getParameterTypes(getNumberOfParameters() - 1)));
202         }
203         result.append(")");
204         return result.toString();
205       }
206     
207       @Override
208       public int hashCode() {
209         return method.hashCode() * 1009;
210       }
211     }
