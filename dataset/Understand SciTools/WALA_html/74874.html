11     package com.ibm.wala.cast.java.translator.polyglot;
12     
13     import polyglot.frontend.ExtensionInfo;
14     import polyglot.frontend.Job;
15     import polyglot.frontend.Scheduler;
16     import polyglot.frontend.SourceGoal_c;
17     
18     import com.ibm.wala.cast.java.loader.JavaSourceLoaderImpl;
19     import com.ibm.wala.cast.java.translator.Java2IRTranslator;
20     
21     /**
22      * A kind of EndGoal that indicates that WALA IR has been generated for the given compilation unit.
23      * @author rfuhrer
24      */
25     public class IRGoal extends SourceGoal_c /* PORT1.7 removed 'implements EndGoal' */ {
26       /**
27        * 
28        */
29       private static final long serialVersionUID = -8023929848709826817L;
30     
31       private JavaSourceLoaderImpl fSourceLoader;
32     
33       protected Java2IRTranslator fTranslator;
34     
35       public IRGoal(Job job, JavaSourceLoaderImpl sourceLoader) {
36         super(job);
37         fSourceLoader = sourceLoader;
38     
39         Scheduler scheduler= job.extensionInfo().scheduler();
40     
41         addPrereq(scheduler.TypeChecked(job));
42         // PORT1.7 - TypeChecked will suffice for what used to require ConstantsChecked.
43         // Need ConstantsChecked in order to make sure that case statements have non-zero labels.
44     //  addPrereq(scheduler.ConstantsChecked(job));
45         // Need to add an AscriptionGoal as a prereq to make sure that empty array initializers get a type ascribed.
46         addPrereq(new AscriptionGoal(job));
47       }
48     
49       @Override
50       public boolean runTask() {
51         ExtensionInfo extInfo= job.extensionInfo();
52     
53         fTranslator= new Java2IRTranslator(
54                 fSourceLoader,
55                 ((IRTranslatorExtension)extInfo).getCAstRewriterFactory());
56         ModuleSource src = (ModuleSource) job.source();
57         fTranslator.translate( 
58             src.getModule(),
59             new PolyglotJava2CAstTranslator(
60                 job.ast(),
61                 fSourceLoader.getReference(),
62                 extInfo.nodeFactory(),
63                 extInfo.typeSystem(),
64                 new PolyglotIdentityMapper(fSourceLoader.getReference()),
65                 ((IRTranslatorExtension)extInfo).getReplicateForDoLoops()).translateToCAst());
66         return true;
67       }
68     
69       @Override
70       public String name() {
71         return "<WALA IR goal for " + job.source().path() + ">";
72       }
73     }
