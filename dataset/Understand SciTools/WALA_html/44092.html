15     
16     import com.ibm.wala.analysis.typeInference.PrimitiveType;

25     import com.ibm.wala.util.strings.Atom;
26     
27     /**
28      * Main interface for language-specific information. This interface helps build
29      * analyses which can operate over multiple languages.
30      *
31      */
32     public interface Language {
33     
34       /**
35        * The canonical {@link Language} implementation for Java
36        */
37       public static JavaLanguage JAVA = new JavaLanguage();
38     
39       /**
40        * What is the name of the language?
41        */
42       Atom getName();
43     
44       /**
45        * If this language is "derived" from some other langauge, which one?
46        */
47       Language getBaseLanguage();
48     
49       /**
50        * Yuck? Languages are mutable?
51        */
52       void registerDerivedLanguage(Language l);
53     
54       Set<Language> getDerivedLanguages();
55     
56       /**
57        * What is the root type in a type hierarchy for this language? e.g.
58        * java.lang.Object in Java.
59        */
60       TypeReference getRootType();
61     
62       /**
63        * What is the root type of exceptions in this language? e.g.
64        * java.lang.Throwable in Java
65        */
66       TypeReference getThrowableType();
67     
68       /**
69        * Given a Java constant o, return the appropriate language type to associate
70        * with the constant. Possible types for o can be language dependent, but
71        * typically include Boolean, String, Integer, Float, etc.
72        */
73       TypeReference getConstantType(Object o);
74     
75       /**
76        * Is t the type of the language's null value? Should return true if
77        * <code>t == null</code> (?).
78        */
79       boolean isNullType(TypeReference t);
80     
81       boolean isIntType(TypeReference t);
82     
83       boolean isLongType(TypeReference t);
84     
85       boolean isVoidType(TypeReference t);
86     
87       boolean isFloatType(TypeReference t);
88     
89       boolean isDoubleType(TypeReference t);
90     
91       boolean isStringType(TypeReference t);
92     
93       /**
94        * Is t a "metadata" type for the language, i.e., a type describing some other
95        * type (e.g., java.lang.Class for Java)?
96        */
97       boolean isMetadataType(TypeReference t);
98     
99       boolean isCharType(TypeReference t);
100     
101       boolean isBooleanType(TypeReference t);
102     
103       /**
104        * Get the representation of the meta-data corresponding to value. For
105        * example, in Java, if value represents some type, the returned object should
106        * be the corresponding {@link TypeReference}. The returned object should be
107        * appropriate for use as the token in an {@link SSALoadMetadataInstruction}
108        * for the language
109        * 
110        */
111       Object getMetadataToken(Object value);
112     
113       /**
114        * get the interfaces implemented by all arrays in the language
115        */
116       TypeReference[] getArrayInterfaces();
117     
118       /**
119        * Given a source-level primitive type name, get the corresponding "low-level"
120        * type name, e.g., the corresponding character to use in a Java method
121        * descriptor
122        */
123       TypeName lookupPrimitiveType(String name);
124     
125       SSAInstructionFactory instructionFactory();
126     
127       /**
128        * determine the set of possible exception types a call to target may throw
129        */
130       Collection<TypeReference> inferInvokeExceptions(MethodReference target, IClassHierarchy cha) throws InvalidClassFileException;
131     
132       TypeReference getStringType();
133     
134       TypeReference getPointerType(TypeReference pointee);
135     
136       /**
137        * get the abstraction of a primitive type to be used for type inference
138        * 
139        * @see TypeInference
140        */
141       PrimitiveType getPrimitive(TypeReference reference);
142     }
