41     package com.ibm.wala.dalvik.util.androidEntryPoints;
42     
43     import java.util.List;
44     
45     import com.ibm.wala.dalvik.ipa.callgraph.impl.AndroidEntryPoint.ExecutionOrder;
46     import com.ibm.wala.dalvik.util.AndroidComponent;
47     import com.ibm.wala.dalvik.util.AndroidEntryPointLocator.AndroidPossibleEntryPoint;
48     
49     /**
50      *  Hardcoded specifications of androids loader call-backs.
51      *
52      *  The specifications are read and handled by AndroidEntryPointLocator if the flags of
53      *  AndroidEntryPointLocator are set to include call-backs.
54      *
55      *  @see    com.ibm.wala.dalvik.util.AndroidEntryPointLocator
56      *  @author Tobias Blaschke <code@tobiasblaschke.de>
57      */
58     public final class LoaderCB {
59         
60         /**
61          * Instantiate and return a new Loader for the given ID.
62          */
63         public static final AndroidPossibleEntryPoint onCreateLoader = new AndroidPossibleEntryPoint(AndroidComponent.LOADER_CB, "onCreateLoader",
64             ExecutionOrder.AT_FIRST); // TODO: Baaad position
65     
66         /**
67          *  Called when a previously created loader has finished its load.
68          */
69         public static final AndroidPossibleEntryPoint onLoadFinished = new AndroidPossibleEntryPoint(AndroidComponent.LOADER_CB, "onLoadFinished",
70                 ExecutionOrder.after(onCreateLoader)); // TODO: Baaad position
71     
72         /**
73          *  Called when a previously created loader is being reset, and thus making its data unavailable.
74          */
75         public static final AndroidPossibleEntryPoint onLoaderReset = new AndroidPossibleEntryPoint(AndroidComponent.LOADER_CB, "onLoaderReset",
76                 ExecutionOrder.after(onCreateLoader)); // TODO: Baaad position
77     
78         /**
79          *  Add the EntryPoint specifications defined in this file to the given list.
80          *
81          *  @param  possibleEntryPoints the list to extend.
82          */
83     	public static void populate(List<? super AndroidPossibleEntryPoint> possibleEntryPoints) {
84     		possibleEntryPoints.add(onCreateLoader);
85     		possibleEntryPoints.add(onLoadFinished);
86     		possibleEntryPoints.add(onLoaderReset);
87     	}
88     }
