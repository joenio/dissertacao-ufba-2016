11     package com.ibm.wala.dataflow.IFDS;

13     import com.ibm.wala.util.MonitorUtil.IProgressMonitor;
14     
15     
16     
17     /**
18      * A {@link TabulationSolver} that gives up after a finite bound.
19      * 
20      * @param <T> type of node in the supergraph 
21      * @param <P> type of a procedure (like a box in an RSM)
22      * @param <F> type of factoids propagated when solving this problem
23      * 
24      */
25     public class BoundedTabulationSolver<T, P, F> extends TabulationSolver<T, P, F> {
26     
27       public static <T, P, F> BoundedTabulationSolver<T, P, F> createBoundedTabulationSolver(TabulationProblem<T, P, F> p, int bound,
28           IProgressMonitor monitor) {
29         return new BoundedTabulationSolver<T, P, F>(p, bound, monitor);
30       }
31     
32       private final int bound;
33       
34       private int numSteps = 0;
35       
36       protected BoundedTabulationSolver(TabulationProblem<T, P, F> p, int bound, IProgressMonitor monitor) {
37         super(p, monitor);
38         this.bound = bound;
39       }
40       
41       @Override
42       protected boolean propagate(T s_p, int i,T n, int j) {
43         if (numSteps < bound) {
44           numSteps++;
45           return super.propagate(s_p, i, n, j);
46         }
47         return false;
48       }
49       
50       public int getNumSteps() {
51         return numSteps;
52       }
53       
54       public void resetBound() {
55         numSteps = 0;
56       }
57     
58     }
