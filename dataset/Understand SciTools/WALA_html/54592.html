11     package com.ibm.wala.cast.js.types;
12     
13     import com.ibm.wala.cast.js.loader.JavaScriptLoader;
14     import com.ibm.wala.cast.types.AstMethodReference;
15     import com.ibm.wala.types.Descriptor;
16     import com.ibm.wala.types.MethodReference;
17     import com.ibm.wala.types.TypeReference;
18     import com.ibm.wala.util.strings.Atom;
19     
20     public class JavaScriptMethods extends AstMethodReference {
21     
22       public final static String ctorAtomStr = "ctor";
23       public final static Atom ctorAtom = Atom.findOrCreateUnicodeAtom(ctorAtomStr);
24       public final static String ctorDescStr = "()LRoot;";
25       public final static Descriptor ctorDesc = Descriptor.findOrCreateUTF8(JavaScriptLoader.JS, ctorDescStr);
26       public final static MethodReference ctorReference =
27         MethodReference.findOrCreate(JavaScriptTypes.CodeBody, ctorAtom, ctorDesc);
28     
29       public static MethodReference makeCtorReference(TypeReference cls) {
30         return MethodReference.findOrCreate(cls, ctorAtom, ctorDesc);
31       }
32     
33       public final static String dispatchAtomStr = "dispatch";
34       public final static Atom dispatchAtom = Atom.findOrCreateUnicodeAtom(dispatchAtomStr);
35       public final static String dispatchDescStr = "()LRoot;";
36       public final static Descriptor dispatchDesc = Descriptor.findOrCreateUTF8(JavaScriptLoader.JS, dispatchDescStr);
37       public final static MethodReference dispatchReference =
38         MethodReference.findOrCreate(JavaScriptTypes.CodeBody, dispatchAtom, dispatchDesc);
39     
40     }
