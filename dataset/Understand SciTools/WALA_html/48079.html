22     /**
23      * An InstanceKey serves as the representative for an equivalence class of
24      * objects in the heap, that can be pointed to.
25      * 
26      * For example, for 0-CFA, an InstanceKey would embody an <IClass>... we model
27      * all instances of a particular class
28      * 
29      * For 0-1-CFA, an InstanceKey could be <IMethod,statement #>, representing a
30      * particular allocation statement in a particular method.
31      */
32     public interface InstanceKey extends ContextItem {
33     
34       /**
35        * For now, we assert that each InstanceKey represents a set of classes which
36        * are all of the same concrete type (modulo the fact that all arrays of
37        * references are considered concrete type []Object;)
38        */
39       IClass getConcreteType();
40     
41       /**
42        * Get the creation sites of <code>this</code>, i.e., the statements that may
43        * allocate objects represented by <code>this</code>. A creation site is a
44        * pair (n,s), where n is the containing {@link CGNode} in the given
45        * {@link CallGraph} <code>CG</code> and s is the allocating
46        * {@link NewSiteReference}.
47        * 
48        */
49       Iterator<Pair<CGNode, NewSiteReference>> getCreationSites(CallGraph CG);
50     
51     }
