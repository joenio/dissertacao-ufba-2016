64     public class JDTJavaSourceAnalysisEngine extends EclipseProjectSourceAnalysisEngine<IJavaProject> {
65       private boolean dump;
66       
67       public JDTJavaSourceAnalysisEngine(IJavaProject project) throws IOException, CoreException {
68         super(project);
69       }
70     
71       public JDTJavaSourceAnalysisEngine(String projectName) throws IOException, CoreException {
72         this(JdtUtil.getNamedProject(projectName));
73       }
74     
75       public void setDump(boolean dump) {
76         this.dump = dump;
77       }
78       
79       @Override
80       protected ClassLoaderFactory makeClassLoaderFactory(SetOfClasses exclusions) {
81     	  return new JDTClassLoaderFactory(exclusions, dump);	
82       }
83     
84       @Override
85       protected AnalysisScope makeAnalysisScope() {
86     	  return new JavaSourceAnalysisScope();
87       }
88       
89       @Override
90       protected ClassLoaderReference getSourceLoader() {
91     	  return JavaSourceAnalysisScope.SOURCE;
92       }
93     
94       @Override
95       public AnalysisCache makeDefaultCache() {
96         return new AnalysisCache(AstIRFactory.makeDefaultFactory());
97       }
98     
99       @Override
100       protected EclipseProjectPath<?, IJavaProject> createProjectPath(
101     		  IJavaProject project) throws IOException, CoreException {
102         project.open(new NullProgressMonitor());
103     	  return JavaEclipseProjectPath.make(project, AnalysisScopeType.SOURCE_FOR_PROJ_AND_LINKED_PROJS);	
104       }
105     
106       @Override
107       protected CallGraphBuilder getCallGraphBuilder(IClassHierarchy cha,
108     		  AnalysisOptions options, AnalysisCache cache) {
109     	    return new ZeroCFABuilderFactory().make(options, cache, cha, scope, false);
110       }
111     
112     }
