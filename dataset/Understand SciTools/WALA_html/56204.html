75     /**
76      * The nodes in this graph are PointerKeys corresponding to local variables and static fields, InstanceKeys, and FieldRefs (see
77      * below).
78      * 
79      * This graph is constructed on-demand during a traversal.
80      * 
81      * The edges represent
82      * <ul>
83      * <li>flow from local -> local representing assignment (i.e. phi,pi)
84      * <li>flow from instancekey -> local for news
85      * <li>flow from formal -> actual parameter
86      * <li>flow from return value -> local
87      * <li>match edges
88      * <li>local -> local edges representing loads/stores (e.g. x = y.f will have a edge x->y, labelled with f) for a getstatic x = Y.f,
89      * we have an edge from x -> Y.f.
90      * </ul>
91      * 
92      * N.B: Edges go OPPOSITE the flow of values.
93      * 
94      * Edges carry labels if they arise from loads/stores, or calls
95      */
96     public class SimpleDemandPointerFlowGraph extends SlowSparseNumberedGraph<Object> {

148       public void addSubgraphForNode(CGNode node) {
149         int n = cg.getNumber(node);
150         if (!cgNodesVisited.contains(n)) {
151           cgNodesVisited.add(n);
152           unconditionallyAddConstraintsFromNode(node);
153           addNodesForParameters(node);
154         }

157       /**
158        * add nodes for parameters and return values
159        * 
160        * @param node
161        */
162       private void addNodesForParameters(CGNode node) {
163         // TODO Auto-generated method stub
164         IR ir = node.getIR();
165         TypeInference ti = TypeInference.make(ir, false);
166         SymbolTable symbolTable = ir.getSymbolTable();
167         for (int i = 0; i < symbolTable.getNumberOfParameters(); i++) {
168           int parameter = symbolTable.getParameter(i);
169           TypeAbstraction t = ti.getType(parameter);
170           if (t != null) {
171             PointerKey paramPk = heapModel.getPointerKeyForLocal(node, parameter);
172             addNode(paramPk);
173             params.put(paramPk, node);
174           }
175         }
176         addNode(heapModel.getPointerKeyForReturnValue(node));
177         addNode(heapModel.getPointerKeyForExceptionalReturnValue(node));
178       }

249        * @see com.ibm.capa.util.graph.AbstractGraph#getSuccNodes(java.lang.Object)

251       @Override
252       public Iterator<Object> getSuccNodes(Object N) {
253         if (N instanceof StaticFieldKey) {
254           addNodesThatWriteToStaticField(((StaticFieldKey) N).getField());
255         } else {
256           IField f = getFieldDefs.get(N);
257           if (f != null) {
258             addMatchEdges((LocalPointerKey) N, f);
259           } else {
260             SSAInvokeInstruction callInstr = callDefs.get(N);
261             if (callInstr != null) {
262               addReturnEdges((LocalPointerKey) N, callInstr);
263             } else {
264               CGNode node = params.get(N);
265               if (node != null) {
266                 addParamEdges((LocalPointerKey) N, node);
267               } else {
268                 if (arrayDefs.contains(N)) {
269                   addArrayMatchEdges((LocalPointerKey) N);
270                 }
271               }
272             }
273           }
274         }
275         return super.getSuccNodes(N);
276       }

278       private void addArrayMatchEdges(LocalPointerKey pk) {
279         Collection<MemoryAccess> arrayWrites = fam.getArrayWrites(null);
280         for (MemoryAccess a : arrayWrites) {
281           addSubgraphForNode(a.getNode());
282         }
283         for (MemoryAccess a : arrayWrites) {
284           IR ir = a.getNode().getIR();
285           SSAArrayStoreInstruction s = (SSAArrayStoreInstruction) ir.getInstructions()[a.getInstructionIndex()];
286           PointerKey r = heapModel.getPointerKeyForLocal(a.getNode(), s.getValue());
287           assert containsNode(r);
288           assert containsNode(pk);
289           addMatchEdge(pk, r);
290         }
291       }

293       private void addParamEdges(LocalPointerKey pk, CGNode node) {
294         // get parameter position: value number - 1?
295         int paramPos = pk.getValueNumber() - 1;
296         // iterate over callers
297         for (CGNode caller : cg) {
298           // TODO we don't need to add the graph if null is passed
299           // as the argument
300           addSubgraphForNode(caller);
301           IR ir = caller.getIR();
302           for (Iterator<CallSiteReference> iterator = ir.iterateCallSites(); iterator.hasNext();) {
303             CallSiteReference call = iterator.next();
304             if (cg.getPossibleTargets(caller, call).contains(node)) {
305               SSAAbstractInvokeInstruction[] callInstrs = ir.getCalls(call);
306               for (int i = 0; i < callInstrs.length; i++) {
307                 SSAAbstractInvokeInstruction callInstr = callInstrs[i];
308                 PointerKey actualPk = heapModel.getPointerKeyForLocal(caller, callInstr.getUse(paramPos));
309                 assert containsNode(actualPk);
310                 assert containsNode(pk);
311                 addEdge(pk, actualPk);
312     
313               }
314             }
315           }
316         }
317       }

319       /**
320        * @param pk value being def'fed by a call instruction (either normal or exceptional)
321        */
322       private void addReturnEdges(LocalPointerKey pk, SSAInvokeInstruction callInstr) {
323         boolean isExceptional = pk.getValueNumber() == callInstr.getException();
324     
325         // get call targets
326         Collection<CGNode> possibleCallees = cg.getPossibleTargets(pk.getNode(), callInstr.getCallSite());
327         // construct graph for each target
328         for (CGNode callee : possibleCallees) {
329           addSubgraphForNode(callee);
330           PointerKey retVal = isExceptional ? heapModel.getPointerKeyForExceptionalReturnValue(callee) : heapModel
331               .getPointerKeyForReturnValue(callee);
332           assert containsNode(retVal);
333           addEdge(pk, retVal);
334         }
335       }

338         Collection<MemoryAccess> fieldWrites = fam.getFieldWrites(null, f);
339         addMatchHelper(pk, fieldWrites);

342       private void addMatchHelper(LocalPointerKey pk, Collection<MemoryAccess> writes) {
343         for (MemoryAccess a : writes) {
344           addSubgraphForNode(a.getNode());
345         }
346         for (MemoryAccess a : writes) {
347           IR ir = a.getNode().getIR();
348           SSAPutInstruction s = (SSAPutInstruction) ir.getInstructions()[a.getInstructionIndex()];
349           PointerKey r = heapModel.getPointerKeyForLocal(a.getNode(), s.getVal());
350           assert containsNode(r);
351           assert containsNode(pk);
352           addMatchEdge(pk, r);
353         }
354       }

377       protected void unconditionallyAddConstraintsFromNode(CGNode node) {
378     
379         if (DEBUG) {
380           System.err.println(("Visiting CGNode " + node));
381         }
382     
383         if (SSAPropagationCallGraphBuilder.PERIODIC_WIPE_SOFT_CACHES) {
384           wipeCount++;
385           if (wipeCount >= SSAPropagationCallGraphBuilder.WIPE_SOFT_CACHE_INTERVAL) {
386             wipeCount = 0;
387             ReferenceCleanser.clearSoftCaches();
388           }
389         }
390     
391         IR ir = node.getIR();
392         debugPrintIR(ir);
393     
394         if (ir == null) {
395           return;
396         }
397     
398         DefUse du = node.getDU();
399         addNodeInstructionConstraints(node, ir, du);
400         addNodePassthruExceptionConstraints(node, ir);
401       }

403       /**
404        * Add constraints to represent the flow of exceptions to the exceptional return value for this node
405        */
406       protected void addNodePassthruExceptionConstraints(CGNode node, IR ir) {
407         // add constraints relating to thrown exceptions that reach the exit block.
408         List<ProgramCounter> peis = SSAPropagationCallGraphBuilder.getIncomingPEIs(ir, ir.getExitBlock());
409         PointerKey exception = heapModel.getPointerKeyForExceptionalReturnValue(node);
410         IClass c = node.getClassHierarchy().lookupClass(TypeReference.JavaLangThrowable);
411     
412         addExceptionDefConstraints(ir, node, peis, exception, Collections.singleton(c));
413       }

415       /**
416        * Generate constraints which assign exception values into an exception pointer
417        * 
418        * @param node governing node
419        * @param peis list of PEI instructions
420        * @param exceptionVar PointerKey representing a pointer to an exception value
421        * @param catchClasses the types "caught" by the exceptionVar
422        */
423       private void addExceptionDefConstraints(IR ir, CGNode node, List<ProgramCounter> peis, PointerKey exceptionVar,
424           Set<IClass> catchClasses) {
425         for (ProgramCounter peiLoc : peis) {
426           SSAInstruction pei = ir.getPEI(peiLoc);
427     
428           if (pei instanceof SSAAbstractInvokeInstruction) {
429             SSAAbstractInvokeInstruction s = (SSAAbstractInvokeInstruction) pei;
430             PointerKey e = heapModel.getPointerKeyForLocal(node, s.getException());
431             addNode(exceptionVar);
432             addNode(e);
433             addEdge(exceptionVar, e);
434     
435           } else if (pei instanceof SSAAbstractThrowInstruction) {
436             SSAAbstractThrowInstruction s = (SSAAbstractThrowInstruction) pei;
437             PointerKey e = heapModel.getPointerKeyForLocal(node, s.getException());
438             addNode(exceptionVar);
439             addNode(e);
440             addEdge(exceptionVar, e);
441           }
442     
443           // Account for those exceptions for which we do not actually have a
444           // points-to set for
445           // the pei, but just instance keys
446           Collection<TypeReference> types = pei.getExceptionTypes();
447           if (types != null) {
448             for (TypeReference type : types) {
449               if (type != null) {
450                 InstanceKey ik = heapModel.getInstanceKeyForPEI(node, peiLoc, type);
451                 assert ik instanceof ConcreteTypeKey : "uh oh: need to implement getCaughtException constraints for instance " + ik;
452                 ConcreteTypeKey ck = (ConcreteTypeKey) ik;
453                 IClass klass = ck.getType();
454                 if (PropagationCallGraphBuilder.catches(catchClasses, klass, cha)) {
455                   addNode(exceptionVar);
456                   addNode(ik);
457                   addEdge(exceptionVar, ik);
458                 }
459               }
460             }
461           }
462         }
463       }

468       protected void addNodeInstructionConstraints(CGNode node, IR ir, DefUse du) {
469         StatementVisitor v = makeVisitor((ExplicitCallGraph.ExplicitNode) node, ir, du);
470         ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg = ir.getControlFlowGraph();
471         for (ISSABasicBlock b : cfg) {
472           addBlockInstructionConstraints(node, cfg, b, v);
473         }

476       /**
477        * Add constraints for a particular basic block.
478        */
479       protected void addBlockInstructionConstraints(CGNode node, ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg,
480           ISSABasicBlock b, StatementVisitor v) {
481         v.setBasicBlock(b);
482     
483         // visit each instruction in the basic block.
484         for (Iterator<SSAInstruction> it = b.iterator(); it.hasNext();) {
485           SSAInstruction s = it.next();
486           if (s != null) {
487             s.visit(v);
488           }
489         }
490     
491         addPhiConstraints(node, cfg, b);
492       }
493     
494       private void addPhiConstraints(CGNode node, ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg, ISSABasicBlock b) {
495     
496         // visit each phi instruction in each successor block
497         for (Iterator<? extends IBasicBlock> sbs = cfg.getSuccNodes(b); sbs.hasNext();) {
498           ISSABasicBlock sb = (ISSABasicBlock) sbs.next();
499           if (sb.isExitBlock()) {
500             // an optimization based on invariant that exit blocks should have no
501             // phis.
502             continue;
503           }
504           int n = 0;
505           // set n to be whichPred(this, sb);
506           for (Iterator<? extends IBasicBlock> back = cfg.getPredNodes(sb); back.hasNext(); n++) {
507             if (back.next() == b) {
508               break;
509             }
510           }
511           assert n < cfg.getPredNodeCount(sb);
512           for (Iterator<SSAPhiInstruction> phis = sb.iteratePhis(); phis.hasNext();) {
513             SSAPhiInstruction phi = phis.next();
514             if (phi == null) {
515               continue;
516             }
517             PointerKey def = heapModel.getPointerKeyForLocal(node, phi.getDef());
518             if (phi.getUse(n) > 0) {
519               PointerKey use = heapModel.getPointerKeyForLocal(node, phi.getUse(n));
520               addNode(def);
521               addNode(use);
522               addEdge(def, use);
523             }
524             // }
525             // }
526           }
527         }
528       }
