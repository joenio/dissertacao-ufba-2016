27     /**
28      * Utility class for computing an analysis result for call graph nodes and their
29      * transitive callees, given the results for individual nodes.
30      * 
31      */
32     public class CallGraphTransitiveClosure {
33     
34       
35       /**
36        * Compute the transitive closure of an analysis result over all callees.
37        * 
38        * @param cg the call graph
39        * @param nodeResults analysis result for each individual node
40        * @return a map from each node to the analysis result for the node and its transitive callees
41        */
42       public static <T> Map<CGNode, OrdinalSet<T>> transitiveClosure(CallGraph cg, Map<CGNode, Collection<T>> nodeResults) {
43         try {
44           // invert the call graph, to compute the bottom-up result
45           GenReach<CGNode, T> gr = new GenReach<CGNode, T>(GraphInverter.invert(cg), nodeResults);
46           BitVectorSolver<CGNode> solver = new BitVectorSolver<CGNode>(gr);
47           solver.solve(null);
48           Map<CGNode, OrdinalSet<T>> result = HashMapFactory.make();
49           for (Iterator<? extends CGNode> it = cg.iterator(); it.hasNext();) {
50             CGNode n = it.next();
51             BitVectorVariable bv = solver.getOut(n);
52             result.put(n, new OrdinalSet<T>(bv.getValue(), gr.getLatticeValues()));
53           }
54           return result;
55         } catch (CancelException e) {
56           throw new CancelRuntimeException(e);
57         }
58       }
59       
60       /**
61        * Collect analysis result for each {@link CGNode} in a {@link Map}.
62        */
63       public static <T> Map<CGNode, Collection<T>> collectNodeResults(CallGraph cg, Function<CGNode, Collection<T>> nodeResultComputer) {
64         Map<CGNode, Collection<T>> result = HashMapFactory.make();
65         for (Iterator<? extends CGNode> it = cg.iterator(); it.hasNext();) {
66           CGNode n = it.next();
67           result.put(n, nodeResultComputer.apply(n));
68         }
69         return result;
70         
71       }
72     }
