11     package com.ibm.wala.ide.classloader;
12     
13     import java.io.File;
14     import java.util.regex.Pattern;
15     
16     import org.eclipse.core.resources.IFile;
17     import org.eclipse.core.resources.IWorkspace;
18     import org.eclipse.core.resources.IWorkspaceRoot;
19     import org.eclipse.core.resources.ResourcesPlugin;
20     import org.eclipse.core.runtime.IPath;
21     
22     import com.ibm.wala.classLoader.FileModule;
23     import com.ibm.wala.classLoader.SourceDirectoryTreeModule;
24     import com.ibm.wala.ide.util.EclipseProjectPath;
25     
26     public class EclipseSourceDirectoryTreeModule extends SourceDirectoryTreeModule {
27     
28       private final IPath rootIPath;
29       private final Pattern[] excludePatterns;
30       
31       private static Pattern interpretPattern(IPath pattern) {
32         return Pattern.compile("^" + pattern.toString().replace(".",  "\\.").replace("**", "~~~").replace("*", "[^/]*").replace("~~~",  ".*") + "$");
33       }
34       
35       private static Pattern[] interpretExcludes(IPath[] excludes) {
36         if (excludes == null) {
37           return null;
38         } else {
39           Pattern[] stuff = new Pattern[ excludes.length ];
40           for(int i = 0; i < excludes.length; i++) {
41             stuff[i] = interpretPattern(excludes[i]);
42           }
43           return stuff;
44           }
45       }
46       
47       public EclipseSourceDirectoryTreeModule(IPath root, IPath[] excludePaths) {
48         super(EclipseProjectPath.makeAbsolute(root).toFile());
49         this.rootIPath = root;
50         this.excludePatterns = interpretExcludes(excludePaths);
51       }
52     
53       public EclipseSourceDirectoryTreeModule(IPath root, IPath[] excludePaths, String fileExt) {
54         super(EclipseProjectPath.makeAbsolute(root).toFile(), fileExt);
55         this.rootIPath = root;
56         this.excludePatterns = interpretExcludes(excludePaths);
57     }
58     
59       @Override
60       protected FileModule makeFile(File file) {
61         IPath p = rootIPath.append(file.getPath().substring(root.getPath().length()));
62         IWorkspace ws = ResourcesPlugin.getWorkspace();
63         IWorkspaceRoot root = ws.getRoot();
64         IFile ifile = root.getFile(p);
65         assert ifile.exists();
66         return EclipseSourceFileModule.createEclipseSourceFileModule(ifile);
67       }
68       
69       @Override
70       protected boolean includeFile(File file) {
71         if (!super.includeFile(file)) {
72           return false;
73         } else {
74           if (excludePatterns != null) {
75             IPath p = rootIPath.append(file.getPath().substring(root.getPath().length()));
76             for(Pattern exclude : excludePatterns) {
77               if (exclude.matcher(p.toOSString()).matches()) {
78                 return false;
79               }
80             }
81           }
82           return true;
83         }
84       }
85     
86       @Override
87       public String toString() {
88         return "EclipseSourceDirectoryTreeModule:" + rootIPath;
89       }
90     
91     }
