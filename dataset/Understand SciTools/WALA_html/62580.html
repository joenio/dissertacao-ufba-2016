258       public CAstEntity translateToCAst() {
259     
260         List<CAstEntity> declEntities = new ArrayList<CAstEntity>();
261     
262         for (Iterator<CAstEntity> iter = cu.types().iterator(); iter.hasNext();) {
263           AbstractTypeDeclaration decl = (AbstractTypeDeclaration) iter.next();
264           // can be of type AnnotationTypeDeclaration, EnumDeclaration, TypeDeclaration
265           declEntities.add(visit(decl, new RootContext()));
266         }
267     
268         if (dump) {
269           for(CAstEntity d : declEntities) {
270             CAstPrinter.printTo(d, new PrintWriter(System.err));
271           }
272         }
273         
274         return new CompilationUnitEntity(cu.getPackage(), declEntities);
275       }

416       /**
417        * 
418        * @param n
419        * @param bodyDecls
420        * @param enumConstants
421        * @param typeBinding
422        * @param name Used in creating default constructor, and passed into new ClassEntity()
423        * @param context
424        * @return
425        */
426       private CAstEntity createClassDeclaration(ASTNode n, List/* <BodyDeclaration> */bodyDecls,
427           List/* EnumConstantDeclaration */enumConstants, ITypeBinding typeBinding, String name, int modifiers, 
428           boolean isInterface, boolean isAnnotation, WalkContext context) {
429         final List<CAstEntity> memberEntities = new ArrayList<CAstEntity>();
430     
431         // find and collect all initializers (type Initializer) and field initializers (type VariableDeclarationFragment).
432         // instance initializer code will be inserted into each constructors.
433         // all static initializer code will be grouped together in its own entity.
434         ArrayList<ASTNode> inits = new ArrayList<ASTNode>();
435         ArrayList<ASTNode> staticInits = new ArrayList<ASTNode>();
436     
437         if (enumConstants != null) {
438           for (Object decl : enumConstants) {
439             EnumConstantDeclaration ecd = (EnumConstantDeclaration) decl;
440             staticInits.add(ecd); // always (implicitly) static,final (actually, no modifiers allowed)
441           }
442         }
443     
444         for (Object decl : bodyDecls) {
445           if (decl instanceof Initializer) {
446             Initializer initializer = (Initializer) decl;
447             boolean isStatic = ((initializer.getModifiers() & Modifier.STATIC) != 0);
448             (isStatic ? staticInits : inits).add(initializer);
449           } else if (decl instanceof FieldDeclaration) {
450             FieldDeclaration fd = (FieldDeclaration) decl;
451     
452             for (Object f : fd.fragments()) {
453               VariableDeclarationFragment frag = (VariableDeclarationFragment) f;
454               if (frag.getInitializer() != null) {
455                 boolean isStatic = ((fd.getModifiers() & Modifier.STATIC) != 0);
456                 (isStatic ? staticInits : inits).add(frag);
457               }
458             }
459           }
460         }
461     
462         // process entities. initializers will be folded in here.
463         if (enumConstants != null) {
464           for (Object decl : enumConstants) {
465             memberEntities.add(visit((EnumConstantDeclaration) decl, context));
466           }
467         }
468     
469         for (Object d : bodyDecls) {
470           BodyDeclaration decl = (BodyDeclaration) d;
471           if (decl instanceof FieldDeclaration) {
472             FieldDeclaration fieldDecl = (FieldDeclaration) decl;
473             Collection<CAstQualifier> quals = JDT2CAstUtils.mapModifiersToQualifiers(fieldDecl.getModifiers(), false, false);
474             for (Object f : fieldDecl.fragments()) {
475               VariableDeclarationFragment fieldFrag = (VariableDeclarationFragment) f;
476               IVariableBinding fieldBinding = fieldFrag.resolveBinding();
477     		memberEntities.add(new FieldEntity(fieldFrag.getName().getIdentifier(), fieldBinding.getType(), quals,
478                   makePosition(fieldFrag.getStartPosition(), fieldFrag.getStartPosition() + fieldFrag.getLength()),
479                   handleAnnotations(fieldBinding)));
480             }
481           } else if (decl instanceof Initializer) {
482             // Initializers are inserted into constructors when making constructors.
483           } else if (decl instanceof MethodDeclaration) {
484             MethodDeclaration metDecl = (MethodDeclaration) decl;
485     
486             if (typeBinding.isEnum() && metDecl.isConstructor())
487               memberEntities.add(createEnumConstructorWithParameters(metDecl.resolveBinding(), metDecl, context, inits, metDecl));
488             else {
489               memberEntities.add(visit(metDecl, typeBinding, context, inits));
490     
491               // /////////////// Java 1.5 "overridden with subtype" thing (covariant return type) ///////////
492               Collection<IMethodBinding> overriddenMets = JDT2CAstUtils.getOverriddenMethod(metDecl.resolveBinding());
493               if (overriddenMets != null) {
494                 for (IMethodBinding overridden : overriddenMets)
495                   if (!JDT2CAstUtils.sameErasedSignatureAndReturnType(metDecl.resolveBinding(), overridden))
496                     memberEntities.add(makeSyntheticCovariantRedirect(metDecl, metDecl.resolveBinding(), overridden, context));
497               }
498             }
499           } else if (decl instanceof AbstractTypeDeclaration) {
500             memberEntities.add(visit((AbstractTypeDeclaration) decl, context));
501           } else if (decl instanceof AnnotationTypeMemberDeclaration) {
502             // TODO: need to decide what to do with these
503           } else {
504             Assertions.UNREACHABLE("BodyDeclaration not Field, Initializer, or Method");
505           }
506         }
507     
508         // add default constructor(s) if necessary
509         // most default constructors have no parameters; however, those created by anonymous classes will have parameters
510         // (they just call super with those parameters)
511         for (Object m : typeBinding.getDeclaredMethods()) {
512           IMethodBinding met = (IMethodBinding) m;
513           if (met.isDefaultConstructor()) {
514             if (typeBinding.isEnum())
515               memberEntities.add(createEnumConstructorWithParameters(met, n, context, inits, null));
516             else if (met.getParameterTypes().length > 0)
517               memberEntities.add(createDefaultConstructorWithParameters(met, n, context, inits));
518             else
519               memberEntities.add(createDefaultConstructor(name, typeBinding, context, inits, n));
520           }
521         }
522     
523         if (typeBinding.isEnum() && !typeBinding.isAnonymous())
524           doEnumHiddenEntities(typeBinding, staticInits, memberEntities, context);
525     
526         // collect static inits
527         if (!staticInits.isEmpty()) {
528           Map<CAstNode, CAstEntity> childEntities = HashMapFactory.make();
529           final MethodContext newContext = new MethodContext(context, childEntities);
530           // childEntities is the same one as in the ProcedureEntity. later visit(New), etc. may add to this.
531     
532           CAstNode[] bodyNodes = new CAstNode[staticInits.size()];
533           for (int i = 0; i < staticInits.size(); i++)
534             bodyNodes[i] = visitFieldInitNode(staticInits.get(i), newContext);
535           CAstNode staticInitAst = makeNode(newContext, fFactory, n, CAstNode.BLOCK_STMT, bodyNodes);
536           memberEntities.add(new ProcedureEntity(staticInitAst, typeBinding, childEntities, newContext, null));
537         }
538     
539         Collection<CAstQualifier> quals = JDT2CAstUtils.mapModifiersToQualifiers(modifiers, isInterface, isAnnotation);
540     
541         Set<CAstAnnotation> annotations = handleAnnotations(typeBinding);
542     
543         return new ClassEntity(typeBinding, name, quals, memberEntities, makePosition(n), annotations);
544       }

548             JDT2CAstUtils.anonTypeName(n.resolveBinding()), 0 /* no modifiers */, false, false, context);

551       private CAstNode visit(TypeDeclarationStatement n, WalkContext context) {
552         // TODO 1.6: enums of course...
553         AbstractTypeDeclaration decl = n.getDeclaration();
554         assert decl instanceof TypeDeclaration : "Local enum declaration not yet supported";
555         CAstEntity classEntity = visitTypeDecl((TypeDeclaration) decl, context);
556     
557         // these statements doin't actually do anything, just define a type
558         final CAstNode lcdNode = makeNode(context, fFactory, n, CAstNode.EMPTY);
559     
560         // so define it!
561         context.addScopedEntity(lcdNode, classEntity);
562         return lcdNode;
563       }

572        * Make a constructor with parameters that calls super(...) with parameters. Used for anonymous classes with arguments to a
573        * constructor, like new Foo(arg1,arg2) { }
574        */
575       private CAstEntity createDefaultConstructorWithParameters(IMethodBinding ctor, ASTNode n, WalkContext oldContext,
576           ArrayList<ASTNode> inits) {
577         // PART I: find super ctor to call
578         ITypeBinding newType = ctor.getDeclaringClass();
579         ITypeBinding superType = newType.getSuperclass();
580         IMethodBinding superCtor = null;
581     
582         for (IMethodBinding m : superType.getDeclaredMethods())
583           if (m.isConstructor() && Arrays.equals(m.getParameterTypes(), ctor.getParameterTypes()))
584             superCtor = m;
585     
586         assert superCtor != null : "couldn't find constructor for anonymous class";
587     
588         // PART II: make ctor with simply "super(a,b,c...)"
589         final Map<CAstNode, CAstEntity> memberEntities = new LinkedHashMap<CAstNode, CAstEntity>();
590         final MethodContext context = new MethodContext(oldContext, memberEntities);
591         MethodDeclaration fakeCtor = ast.newMethodDeclaration();
592         fakeCtor.setConstructor(true);
593         fakeCtor.setSourceRange(n.getStartPosition(), n.getLength());
594         fakeCtor.setBody(ast.newBlock());
595     
596         // PART IIa: make a fake JDT constructor method with the proper number of args
597         // Make fake args that will be passed
598         String[] fakeArguments = new String[superCtor.getParameterTypes().length + 1];
599         ArrayList<CAstType> paramTypes = new ArrayList<CAstType>(superCtor.getParameterTypes().length);
600         for (int i = 0; i < fakeArguments.length; i++)
601           fakeArguments[i] = (i == 0) ? "this" : ("argument" + i); // TODO: change to invalid name and don't use
602                                                                    // singlevariabledeclaration below
603         for (int i = 1; i < fakeArguments.length; i++) {
604           // the name
605           SingleVariableDeclaration svd = ast.newSingleVariableDeclaration();
606           svd.setName(ast.newSimpleName(fakeArguments[i]));
607           fakeCtor.parameters().add(svd);
608     
609           // the type
610           paramTypes.add(fTypeDict.getCAstTypeFor(ctor.getParameterTypes()[i - 1]));
611         }
612     
613         // PART IIb: create the statements in the constructor
614         // one super() call plus the inits
615         CAstNode[] bodyNodes = new CAstNode[inits.size() + 1];
616     
617         // make super(...) call
618         // this, call ref, args
619         CAstNode[] children = new CAstNode[fakeArguments.length + 1];
620         children[0] = makeNode(context, fFactory, n, CAstNode.SUPER);
621         CallSiteReference callSiteRef = CallSiteReference.make(0, fIdentityMapper.getMethodRef(superCtor),
622             IInvokeInstruction.Dispatch.SPECIAL);
623         children[1] = fFactory.makeConstant(callSiteRef);
624         for (int i = 1; i < fakeArguments.length; i++) {
625           CAstNode argName = fFactory.makeConstant(fakeArguments[i]);
626           CAstNode argType = fFactory.makeConstant(paramTypes.get(i-1));
627           children[i + 1] = makeNode(context, fFactory, n, CAstNode.VAR, argName, argType);
628         }
629         bodyNodes[0] = makeNode(context, fFactory, n, CAstNode.CALL, children);
630         // QUESTION: no handleExceptions?
631     
632         for (int i = 0; i < inits.size(); i++)
633           bodyNodes[i + 1] = visitFieldInitNode(inits.get(i), context);
634     
635         // finally, make the procedure entity
636         CAstNode ast = makeNode(context, fFactory, n, CAstNode.BLOCK_STMT, bodyNodes);
637         return new ProcedureEntity(ast, fakeCtor, newType, memberEntities, context, paramTypes, null, null);
638     
639       }

661       /**
662        * Setup constructor body. Here we add the initializer code (both initalizer blocks and initializers in field declarations). We
663        * may also need to add an implicit super() call.
664        * 
665        * @param n
666        * @param classBinding Used so we can use this with fake MethodDeclaration nodes, as in the case of creating a default
667        *          constructor.
668        * @param context
669        * @param inits
670        * @return
671        */
672       private CAstNode createConstructorBody(MethodDeclaration n, ITypeBinding classBinding, WalkContext context,
673           ArrayList<ASTNode> inits) {
674         // three possibilites: has super(), has this(), has neither.
675     
676         Statement firstStatement = null;
677         if (!n.getBody().statements().isEmpty())
678           firstStatement = (Statement) n.getBody().statements().get(0);
679         if (firstStatement instanceof SuperConstructorInvocation) {
680           // Split at call to super:
681           // super();
682           // field initializer code
683           // remainder of ctor body
684           ArrayList<CAstNode> origStatements = createBlock(n.getBody(), context);
685           CAstNode[] bodyNodes = new CAstNode[inits.size() + origStatements.size()];
686           int idx = 0;
687           bodyNodes[idx++] = origStatements.get(0);
688           for (ASTNode init : inits)
689             bodyNodes[idx++] = visitFieldInitNode(init, context); // visit each in this constructor's context, ensuring
690           // proper handling of exceptions (we can't just reuse the
691           // CAstNodes)
692           for (int i = 1; i < origStatements.size(); i++)
693             bodyNodes[idx++] = origStatements.get(i);
694     
695           return makeNode(context, fFactory, n.getBody(), CAstNode.BLOCK_STMT, bodyNodes); // QUESTION: why no LOCAL_SCOPE?
696           // that's the way it is in
697           // polyglot.
698         } else if (firstStatement instanceof ConstructorInvocation) {
699           return visitNode(n.getBody(), context); // has this(...) call; initializers will be set somewhere else.
700         } else {
701           // add explicit call to default super()
702           // QUESTION their todo: following superClass lookup of default ctor won't work if we
703           // process Object in source...
704     
705           ITypeBinding superType = classBinding.getSuperclass();
706           
707           // find default constructor. IT is an error to have a constructor
708           // without super() when the default constructor of the superclass does not exist.
709           IMethodBinding defaultSuperCtor = findDefaultCtor(superType);
710           CallSiteReference callSiteRef = CallSiteReference.make(0, fIdentityMapper.getMethodRef(defaultSuperCtor),
711               IInvokeInstruction.Dispatch.SPECIAL);
712     
713           // QUESTION: why isn't first arg this like in visit(ConstructorInvocation) ?
714           // why don't we handle exceptions like in visit(ConstructorInvocation) ? (these two things are same in polyglot
715           // implementation)
716     
717           CAstNode superCall = makeNode(context, fFactory, n.getBody(), CAstNode.CALL, makeNode(context, fFactory, n.getBody(),
718               CAstNode.SUPER), fFactory.makeConstant(callSiteRef));
719           Object mapper = new Object(); // dummy used for mapping this node in CFG
720           handleThrowsFromCall(defaultSuperCtor, mapper, context);
721           context.cfg().map(mapper, superCall);
722     
723           ArrayList<CAstNode> origStatements = createBlock(n.getBody(), context);
724           CAstNode[] bodyNodes = new CAstNode[inits.size() + origStatements.size() + 1];
725           // superCall, inits, ctor body
726           int idx = 0;
727           bodyNodes[idx++] = superCall;
728           for (ASTNode init : inits)
729             bodyNodes[idx++] = visitFieldInitNode(init, context);
730           for (int i = 0; i < origStatements.size(); i++)
731             bodyNodes[idx++] = origStatements.get(i);
732           return makeNode(context, fFactory, n.getBody(), CAstNode.BLOCK_STMT, bodyNodes);
733         }
734       }

737        * Make a "fake" function (it doesn't exist in source code but it does in bytecode) for covariant return types.
738        * 
739        * @param overriding Declaration of the overriding method.
740        * @param overridden Binding of the overridden method, in a a superclass or implemented interface.
741        * @param oldContext
742        * @return
743        */
744       private CAstEntity makeSyntheticCovariantRedirect(MethodDeclaration overriding, IMethodBinding overridingBinding,
745           IMethodBinding overridden, WalkContext oldContext) {
746         // SuperClass foo(A, B, C...)
747         // SubClass foo(A,B,C...)
748         //
749         // add a method exactly like overridden that calls overriding
750     
751         final Map<CAstNode, CAstEntity> memberEntities = new LinkedHashMap<CAstNode, CAstEntity>();
752         final MethodContext context = new MethodContext(oldContext, memberEntities);
753     
754         CAstNode calltarget;
755         if ((overridingBinding.getModifiers() & Modifier.STATIC) == 0)
756           calltarget = makeNode(context, fFactory, null, CAstNode.SUPER);
757         else
758           calltarget = makeNode(context, fFactory, null, CAstNode.VOID);
759     
760         ITypeBinding paramTypes[] = overridden.getParameterTypes();
761     
762         ArrayList<CAstNode> arguments = new ArrayList<CAstNode>();
763         int i = 0;
764         for (Object o : overriding.parameters()) {
765           SingleVariableDeclaration svd = (SingleVariableDeclaration) o;
766           CAstNode varNode = makeNode(context, fFactory, null, CAstNode.VAR, fFactory.makeConstant(svd.getName().getIdentifier()));
767           ITypeBinding fromType = JDT2CAstUtils.getErasedType(paramTypes[i], ast);
768           ITypeBinding toType = JDT2CAstUtils.getErasedType(overridingBinding.getParameterTypes()[i], ast);
769           if (fromType.equals(toType)) {
770             arguments.add(varNode);
771           } else {
772             arguments.add(createCast(null, varNode, fromType, toType, context));
773           }
774           i++;
775         }
776         CAstNode callnode = createMethodInvocation(null, overridingBinding, calltarget, arguments, context);
777         CAstNode mdast = makeNode(context, fFactory, null, CAstNode.LOCAL_SCOPE, makeNode(context, fFactory, null, CAstNode.BLOCK_STMT,
778             makeNode(context, fFactory, null, CAstNode.RETURN, callnode)));
779     
780         // make parameters to new synthetic method
781         // use RETURN TYPE of overridden, everything else from overriding (including parameter names)
782         ArrayList<CAstType> paramCAstTypes = new ArrayList<CAstType>(overridden.getParameterTypes().length);
783         for (ITypeBinding paramType : overridden.getParameterTypes())
784           paramCAstTypes.add(fTypeDict.getCAstTypeFor(paramType));
785         return new ProcedureEntity(mdast, overriding, overridingBinding.getDeclaringClass(), memberEntities, context, paramCAstTypes,
786             overridden.getReturnType(), null);
787       }

789       /**
790        * @param inits Instance intializers & field initializers. Only used if method is a constructor, in which case the initializers
791        *          will be inserted in.
792        */
793       private CAstEntity visit(MethodDeclaration n, ITypeBinding classBinding, WalkContext oldContext, ArrayList<ASTNode> inits) {
794     
795         // pass in memberEntities to the context, later visit(New) etc. may add classes
796         final Map<CAstNode, CAstEntity> memberEntities = new LinkedHashMap<CAstNode, CAstEntity>();
797         final MethodContext context = new MethodContext(oldContext, memberEntities); // LEFTOUT: in polyglot there is a
798         // class context in between method and
799         // root
800     
801         CAstNode mdast;
802     
803         if (n.isConstructor())
804           mdast = createConstructorBody(n, classBinding, context, inits);
805         else if ((n.getModifiers() & Modifier.ABSTRACT) != 0) // abstract
806           mdast = null;
807         else if (n.getBody() == null || n.getBody().statements().size() == 0) // empty
808           mdast = makeNode(context, fFactory, n, CAstNode.RETURN);
809         else
810           mdast = visitNode(n.getBody(), context);
811         // Polyglot comment: Presumably the MethodContext's parent is a ClassContext,
812         // and he has the list of initializers. Hopefully the following
813         // will glue that stuff in the right place in any constructor body.
814         
815         Set<CAstAnnotation> annotations = null;
816         if (n.resolveBinding() != null) {
817         	annotations = handleAnnotations(n.resolveBinding());
818         }
819         
820         return new ProcedureEntity(mdast, n, classBinding, memberEntities, context, annotations);
821       }

823       private Set<CAstAnnotation> handleAnnotations(IBinding binding) {
824         IAnnotationBinding[] annotations = binding.getAnnotations();
825         
826         if(annotations == null || annotations.length == 0) {
827         	return null;
828         }
829         
830         Set<CAstAnnotation> castAnnotations = HashSetFactory.make();
831         for(IAnnotationBinding annotation : annotations) {
832         	ITypeBinding annotationTypeBinding = annotation.getAnnotationType();
833         	final CAstType annotationType = fTypeDict.getCAstTypeFor(annotationTypeBinding);
834         	final Map<String,Object> args = HashMapFactory.make();
835         	for(IMemberValuePairBinding mvpb : annotation.getAllMemberValuePairs()) {
836         		String name = mvpb.getName();
837         		Object value = mvpb.getValue();
838         		args.put(name, value);
839         	}
840         	castAnnotations.add(new CAstAnnotation() {
841     			@Override
842     			public CAstType getType() {
843     				return annotationType;
844     			}
845     			@Override
846     			public Map<String, Object> getArguments() {
847     				return args;
848     			}
849     			@Override
850     			public String toString() {
851     				return annotationType.getName() + args;
852     			}
853         	});
854         }
855         
856         return castAnnotations;
857     }

1122       private CAstNode visitFieldInitNode(ASTNode node, WalkContext context) {
1123         // there are gathered by createClassDeclaration and can only be of two types:
1124         if (node instanceof Initializer) {
1125           return visitNode(((Initializer) node).getBody(), context);
1126         } else if (node instanceof VariableDeclarationFragment) {
1127           VariableDeclarationFragment f = (VariableDeclarationFragment) node; // this is guaranteed to have an initializer
1128

1130           // Type targetType = f.memberInstance().container();
1131           // Type fieldType = f.type().type();
1132           FieldReference fieldRef = fIdentityMapper.getFieldRef(f.resolveBinding());
1133           // We use null to indicate an OBJECT_REF to a static field, as the
1134           // FieldReference doesn't
1135           // hold enough info to determine this. In this case, (unlike field ref)
1136           // we don't have a
1137           // target expr to evaluate.
1138           boolean isStatic = ((f.resolveBinding().getModifiers() & Modifier.STATIC) != 0);
1139           CAstNode thisNode = isStatic ? makeNode(context, fFactory, null, CAstNode.VOID) : makeNode(context, fFactory, f,
1140               CAstNode.THIS);
1141           CAstNode lhsNode = makeNode(context, fFactory, f, CAstNode.OBJECT_REF, thisNode, fFactory.makeConstant(fieldRef));
1142     
1143           Expression init = f.getInitializer();
1144           CAstNode rhsNode = visitNode(init, context);
1145           CAstNode assNode = makeNode(context, fFactory, f, CAstNode.ASSIGN, lhsNode, rhsNode);
1146     
1147           return assNode; // their naming, not mine
1148     
1149         } else if (node instanceof EnumConstantDeclaration) {
1150           return createEnumConstantDeclarationInit((EnumConstantDeclaration) node, context);
1151         } else {
1152           Assertions.UNREACHABLE("invalid init node gathered by createClassDeclaration");
1153           return null;
1154         }

1352       private CAstNode createClassInstanceCreation(ASTNode nn, List/* Expression or CAstNode */arguments, IMethodBinding ctorBinding,
1353           Expression qual, AnonymousClassDeclaration anonDecl, WalkContext context) {
1354         // a new instruction is actually two things: a NEW object and a CALL to a constructor
1355         CAstNode newNode;
1356         CAstNode callNode;
1357         final String tmpName = "ctor temp";
1358         // this name is an illegal Java identifier. this var will hold the new object so we can call the constructor on it
1359     
1360         // GENERICS getMethodDeclaration()
1361         ctorBinding = ctorBinding.getMethodDeclaration(); // unlike polyglot, this will
1362         // point to a default
1363         // constructor in the anon class
1364         MethodReference ctorRef = fIdentityMapper.getMethodRef(ctorBinding);
1365     
1366         // ////////////// PART I: make the NEW expression ///////////////////////////////////////////////////////
1367     
1368         ITypeBinding newType = ctorBinding.getDeclaringClass();
1369         TypeReference newTypeRef = fIdentityMapper.getTypeRef(newType);
1370     
1371         // new nodes with an explicit enclosing argument, e.g. "outer.new Inner()". They are mostly treated the same, except
1372         // in JavaCAst2IRTranslator.doNewObject
1373         CAstNode qualNode = null;
1374     
1375         if (qual == null && newType.getDeclaringClass() != null && ((newType.getModifiers() & Modifier.STATIC) == 0)
1376             && !newType.isLocal()) {
1377           // "new X()" expanded into "new this.X()" or "new MyClass.this.X"
1378           // check isLocal because anonymous classes and local classes are not included.
1379           ITypeBinding plainThisType = JDT2CAstUtils.getDeclaringClassOfNode(nn); // type of "this"
1380           ITypeBinding implicitThisType = findClosestEnclosingClassSubclassOf(plainThisType, newType.getDeclaringClass(), ((newType
1381               .getModifiers() & Modifier.PRIVATE) != 0));
1382           if (implicitThisType.isEqualTo(plainThisType))
1383             qualNode = makeNode(context, fFactory, nn, CAstNode.THIS); // "new this.X()"
1384           else
1385             qualNode = makeNode(context, fFactory, nn, CAstNode.THIS, fFactory.makeConstant(fIdentityMapper
1386                 .getTypeRef(implicitThisType))); // "new Bla.this.X()"
1387         } else if (qual != null)
1388           qualNode = visitNode(qual, context);
1389     
1390         if (qualNode != null)
1391           newNode = makeNode(context, fFactory, nn, CAstNode.NEW_ENCLOSING, fFactory.makeConstant(newTypeRef), qualNode);
1392         else
1393           newNode = makeNode(context, fFactory, nn, CAstNode.NEW, fFactory.makeConstant(newTypeRef));
1394     
1395         ITypeBinding[] newExceptions = new ITypeBinding[] { NoClassDefFoundError, ExceptionInInitializerError, OutOfMemoryError };
1396         context.cfg().map(newNode, newNode);
1397         for (ITypeBinding exp : newExceptions) {
1398           for (Pair<ITypeBinding, Object> catchTarget : context.getCatchTargets(exp)) {
1399             context.cfg().add(newNode, catchTarget.snd, catchTarget.fst);
1400           }
1401         }
1402     
1403         // ANONYMOUS CLASSES
1404         // ctor already points to right place, so should type ref, so all we have to do is make the entity
1405         if (anonDecl != null)
1406           context.addScopedEntity(newNode, visit(anonDecl, context));
1407     
1408         // ////////////// PART II: make the CALL expression ///////////////////////////////////////////////////////
1409         // setup args & handle exceptions
1410     
1411         CAstNode[] argNodes = new CAstNode[arguments.size() + 2];
1412         int idx = 0; // args: [ this, callsiteref, <actual args> ]
1413     
1414         // arg 0: this
1415         argNodes[idx++] = makeNode(context, fFactory, nn, CAstNode.VAR, fFactory.makeConstant(tmpName), fFactory.makeConstant(fTypeDict.getCAstTypeFor(newType)));
1416         // contains output from newNode (see part III)
1417     
1418         // arg 1: call site ref (WHY?)
1419         int dummyPC = 0; // Just want to wrap the kind of call; the "rear end" won't care about anything else
1420         CallSiteReference callSiteRef = CallSiteReference.make(dummyPC, ctorRef, IInvokeInstruction.Dispatch.SPECIAL);
1421         argNodes[idx++] = fFactory.makeConstant(callSiteRef);
1422     
1423         // rest of args
1424         for (Iterator iter = arguments.iterator(); iter.hasNext();) {
1425           Object arg = iter.next();
1426           argNodes[idx++] = (arg instanceof CAstNode) ? ((CAstNode) arg) : visitNode((Expression) arg, context);
1427         }
1428         callNode = makeNode(context, fFactory, nn, CAstNode.CALL, argNodes);
1429         context.cfg().map(nn, callNode);
1430     
1431         handleThrowsFromCall(ctorBinding, nn, context);
1432     
1433         // PART III: make a node with both NEW and CALL
1434         // Make a LOCAL_SCOPE with a BLOCK_EXPR node child which does three things:
1435         // 1) declare a temporary variable and assign the new object to it (LOCAL_SCOPE is needed to chain this new variable
1436         // to this block)
1437         // 2) CALL the constructor on the new variable
1438         // 3) access this temporary variable. Since the value of the block is the last thing in the block, the resultant
1439         // value will be the variable
1440         return makeNode(context, fFactory, nn, CAstNode.LOCAL_SCOPE, makeNode(context, fFactory, nn, CAstNode.BLOCK_EXPR, makeNode(
1441             context, fFactory, nn, CAstNode.DECL_STMT, fFactory.makeConstant(new InternalCAstSymbol(tmpName, fTypeDict.getCAstTypeFor(newType), true)), newNode),
1442             callNode, makeNode(context, fFactory, nn, CAstNode.VAR, fFactory.makeConstant(tmpName))));
1443       }

1499           CAstNode target;
1500           if (n.getExpression() != null) {
1501             target = visitNode(n.getExpression(), context);
1502           } else {
1503             ITypeBinding typeOfThis = JDT2CAstUtils.getDeclaringClassOfNode(n);

1505             boolean methodIsPrivate = (binding.getModifiers() & Modifier.PRIVATE) != 0;
1506     
1507             // how could it be in the subtype and private? this only happens the supertype is also an
1508             // enclosing type. in that case the variable refers to the field in the enclosing instance.
1509             // NOTE: method may be defined in MyClass's superclass, but we still want to expand
1510             // this into MyClass, so we have to find the enclosing class which defines this function.
1511     
1512             ITypeBinding implicitThisClass = findClosestEnclosingClassSubclassOf(typeOfThis, binding.getDeclaringClass(),
1513                 methodIsPrivate);

1515               // "foo = 5" -> "this.foo = 5": expand into THIS + class
1516               target = makeNode(context, fFactory, n, CAstNode.THIS);
1517             else
1518               // "foo = 5" -> "MyClass.this.foo = 5" -- inner class: expand into THIS + class
1519               target = makeNode(context, fFactory, n, CAstNode.THIS, fFactory.makeConstant(fIdentityMapper
1520                   .getTypeRef(implicitThisClass)));

1523           }

1537       private CAstNode createMethodInvocation(ASTNode pos, IMethodBinding methodBinding, CAstNode target,
1538           List/* Expression or CAstNode */arguments, WalkContext context) {
1539         // MethodMethodInstance methodInstance = n.methodInstance();
1540         boolean isStatic = (methodBinding.getModifiers() & Modifier.STATIC) != 0;
1541         ITypeBinding methodOwner = methodBinding.getDeclaringClass();
1542     
1543         if (!(methodOwner.isInterface() || methodOwner.isClass() || methodOwner.isEnum())) {
1544           assert false : "owner " + methodOwner + " of " + methodBinding + " is not a class";
1545         }
1546     
1547         // POPULATE PARAMETERS
1548         // this (or void for static), method reference, rest of args
1549         int nFormals = methodBinding.getParameterTypes().length;
1550         CAstNode[] children = new CAstNode[2 + nFormals];
1551     
1552         // this (or void for static)
1553         assert target != null : "no receiver for " + methodBinding;
1554         children[0] = target;
1555     
1556         // method reference
1557         // unlike polyglot, expression will never be super here. this is handled in SuperMethodInvocation.
1558         IInvokeInstruction.IDispatch dispatchType;
1559         if (isStatic)
1560           dispatchType = IInvokeInstruction.Dispatch.STATIC;
1561         else if (methodOwner.isInterface())
1562           dispatchType = IInvokeInstruction.Dispatch.INTERFACE;
1563         else if ((methodBinding.getModifiers() & Modifier.PRIVATE) != 0 || target.getKind() == CAstNode.SUPER)
1564           // only one possibility, not a virtual call (I guess?)
1565           dispatchType = IInvokeInstruction.Dispatch.SPECIAL;
1566         else
1567           dispatchType = IInvokeInstruction.Dispatch.VIRTUAL;
1568         // pass 0 for dummyPC: Just want to wrap the kind of call; the "rear end" won't care about anything else...
1569         CallSiteReference callSiteRef = CallSiteReference.make(0, fIdentityMapper.getMethodRef(methodBinding), dispatchType);
1570     
1571         children[1] = fFactory.makeConstant(callSiteRef);
1572     
1573         populateArguments(children, methodBinding, arguments, context);
1574     
1575         Object fakeCfgMap = new Object();
1576     
1577         handleThrowsFromCall(methodBinding, fakeCfgMap, context);
1578     
1579         CAstNode result = makeNode(context, fFactory, pos, CAstNode.CALL, children);
1580         context.cfg().map(fakeCfgMap, result);
1581         return result;
1582       }

1584       /**
1585        * Populate children, starting at index 2, for the invocation of methodBinding. If varargs are used this function will collapse
1586        * the proper arguments into an array.
1587        * 
1588        * If the number of actuals equals the number of formals and the function is varargs, we have to check the type of the last
1589        * argument to see if we should "box" it in an array. If the arguments[arguments.length-1] is not an Expression, we cannot get the
1590        * type, so we do not box it. (Making covariant varargs functions require this behavior)
1591        * 
1592        * @param children
1593        * @param methodBinding
1594        * @param arguments
1595        * @param context
1596        */
1597       private void populateArguments(CAstNode[] children, IMethodBinding methodBinding, List/* CAstNode or Expression */arguments,
1598           WalkContext context) {
1599         int nFormals = methodBinding.getParameterTypes().length;
1600         assert children.length == nFormals + 2;
1601         int nActuals = arguments.size();
1602     
1603         ITypeBinding lastArgType = null;
1604         if (nActuals > 0 && arguments.get(nActuals - 1) instanceof Expression)
1605           lastArgType = ((Expression) arguments.get(nActuals - 1)).resolveTypeBinding();
1606         // if the # of actuals equals the # of formals, AND the function is varargs, we have to check
1607         // to see if the lastArgType is subtype compatible with the type of last parameter (which will be an array).
1608         // If it is, we pass this array in directly. Otherwise this it is wrapped in an array init.
1609         // Example: 'void foo(int... x)' can be run via 'foo(5)' or 'foo(new int[] { 5, 6 })' -- both have one argument so we must check
1610         // the type
1611     
1612         if (nActuals == nFormals
1613             && (!methodBinding.isVarargs() || lastArgType == null || lastArgType
1614                 .isSubTypeCompatible(methodBinding.getParameterTypes()[nFormals - 1]))) {
1615           int i = 2;
1616           for (Object arg : arguments)
1617             children[i++] = (arg instanceof CAstNode) ? ((CAstNode) arg) : visitNode((Expression) arg, context);
1618         } else {
1619           assert nActuals >= (nFormals - 1) && methodBinding.isVarargs() : "Invalid number of parameters for constructor call";
1620           for (int i = 0; i < nFormals - 1; i++) {
1621             Object arg = arguments.get(i);
1622             children[i + 2] = (arg instanceof CAstNode) ? ((CAstNode) arg) : visitNode((Expression) arg, context);
1623           }
1624     
1625           CAstNode subargs[] = new CAstNode[nActuals - nFormals + 2];
1626           // nodes for args and one extra for NEW expression
1627           TypeReference newTypeRef = fIdentityMapper.getTypeRef(methodBinding.getParameterTypes()[nFormals - 1]);
1628           subargs[0] = makeNode(context, fFactory, null, CAstNode.NEW, fFactory.makeConstant(newTypeRef), fFactory
1629               .makeConstant(subargs.length - 1));
1630           for (int j = 1; j < subargs.length; j++) {
1631             Object arg = arguments.get(j + nFormals - 2);
1632             subargs[j] = (arg instanceof CAstNode) ? ((CAstNode) arg) : visitNode((Expression) arg, context);
1633           }
1634           children[nFormals + 1] = makeNode(context, fFactory, (ASTNode) null, CAstNode.ARRAY_LITERAL, subargs);
1635         }
1636       }

1698         // String real_oneheyya = (((returnObjectWithSideEffects().y))+="hey")+"ya"; // this is going to be a MAJOR pain...

1724             .getName().getIdentifier(), field.resolveFieldBinding(), left, new AssignmentContext(context));

1727             .getName().getIdentifier(), field.resolveFieldBinding(), left, context);

1730             .getTypesVariablesBase(field.resolveFieldBinding().getVariableDeclaration().getType(), ast);

1738                 infixop, realtype, left.getStartPosition(), left.getLength(), castedObref, right, context))));

1834       /**
1835        * Sees if a field defined in owningTypeRef is contained & accessible to a type of typeOfThis. That is, if owningTypeRef ==
1836        * typeOfThis or typeOfThis is a subtype and isPrivate is false. If this is not that case, looks in the enclosing class of
1837        * typeOfThis and tries again, and its enclosing class, ...
1838        * 
1839        * Essentially if we have a field/method referenced only by name and we know its type (owningTypeRef), this function will return
1840        * owningTypeRef or the subtype that the field is accessed thru, for expanding "f = 5" into "TheClass.this.f = 5".
1841        * 
1842        * @param typeOfThis
1843        * @param owningTypeRef
1844        * @param isPrivate
1845        * @return
1846        */
1847       private ITypeBinding findClosestEnclosingClassSubclassOf(ITypeBinding typeOfThis, ITypeBinding owningType, boolean isPrivate) {
1848         // GENERICS
1849     //    if (owningType.isParameterizedType())
1850     //      owningType = owningType.getTypeDeclaration();
1851     //    if (typeOfThis.isParameterizedType())
1852     //      typeOfThis = typeOfThis.getTypeDeclaration();
1853     //    // typeOfThis.getTypeDeclaration()
1854         owningType = owningType.getErasure();
1855     
1856         ITypeBinding current = typeOfThis;
1857         while (current != null) {
1858           current = current.getErasure();
1859           // Walk the hierarchy rather than using isSubTypeCompatible to handle
1860           // generics -- we need to perform erasure of super types
1861           boolean isInSubtype = false;//current.isSubTypeCompatible(owningType);
1862           ITypeBinding supertp = current;
1863           while (supertp != null){

1865             // Use isSubTypeCompatible even though we are manually walking type hierarchy --
1866             // that way interfaces are handled without us having to do it manually.
1867             if (supertp.isSubTypeCompatible(owningType)){
1868               isInSubtype = true;
1869               break;
1870             }
1871             supertp = supertp.getSuperclass();
1872           }

1874           // how could it be in the subtype and private? this only happens the supertype is also an
1875           // enclosing type. in that case the variable refers to the field in the enclosing instance.

1878             return current;

1880           current = current.getDeclaringClass();
1881         }

1900       /**
1901        * Used by visit(FieldAccess) and visit(SimpleName) -- implicit "this" / static field access. things from 'this' cannot throw an
1902        * exception. maybe handle this in here as a special case? i don't know... or check if targetNode is THIS, that should even work
1903        * for this.x = 5 and (this).x = 5
1904        * 
1905        * @param target Used to evaluate the field access. In the case of static field accesses, this is included in the first part of a
1906        *          block -- thus it is evaluated for any side effects but thrown away.
1907        * @param fieldName Name of the field.
1908        * @param fieldBinding This is used to determine the field's class.
1909        * @param positioningNode Used only for making a JdtPosition.
1910        * @param context
1911        * @return
1912        */
1913       private CAstNode createFieldAccess(CAstNode targetNode, String fieldName, IVariableBinding possiblyParameterizedBinding,
1914           ASTNode positioningNode, WalkContext context) {
1915     
1916         IVariableBinding fieldBinding = possiblyParameterizedBinding.getVariableDeclaration();
1917     
1918         ITypeBinding targetType = fieldBinding.getDeclaringClass();
1919     
1920         if (targetType == null) { // array
1921           assert fieldName.equals("length") : "null targetType but not aray length access";
1922           return makeNode(context, fFactory, positioningNode, CAstNode.ARRAY_LENGTH, targetNode);
1923         }
1924     
1925         assert fieldBinding.isField() : "Field binding is not a field?!"; // we can probably safely delete this
1926         // check
1927     
1928         // translate JDT field ref to WALA field ref
1929         FieldReference fieldRef = fIdentityMapper.getFieldRef(fieldBinding);
1930     
1931         if ((fieldBinding.getModifiers() & Modifier.STATIC) != 0) {
1932           // JLS says: evaluate the target of the field ref and throw it away.
1933           // Hence the following block expr, whose 2 children are the target
1934           // evaluation
1935           // followed by the OBJECT_REF with a null target child (which the
1936           // "back-end"
1937           // CAst -> IR translator interprets as a static ref).
1938           // TODO: enum constants
1939     
1940           // don't worry about generics (can't declare static fields with type variables)
1941           if (fieldBinding.getConstantValue() != null) {
1942             return makeNode(context, fFactory, positioningNode, CAstNode.BLOCK_EXPR, targetNode, fFactory.makeConstant(fieldBinding
1943                 .getConstantValue()));
1944           } else {
1945             return makeNode(context, fFactory, positioningNode, CAstNode.BLOCK_EXPR, targetNode,
1946                 makeNode(context, fFactory, positioningNode, CAstNode.OBJECT_REF, makeNode(context, fFactory, null, CAstNode.VOID),
1947                     fFactory.makeConstant(fieldRef)));
1948           }
1949         } else {
1950           CAstNode refNode = makeNode(context, fFactory, positioningNode, CAstNode.OBJECT_REF, targetNode, fFactory
1951               .makeConstant(fieldRef));
1952     
1953           if (targetNode.getKind() != CAstNode.THIS) { // this.x will never throw a null pointer exception, because this
1954             // can never be null
1955             Collection excTargets = context.getCatchTargets(fNullPointerExcType);
1956             if (!excTargets.isEmpty()) {
1957               // connect NPE exception edge to relevant catch targets
1958               // (presumably only one)
1959               for (Iterator iterator = excTargets.iterator(); iterator.hasNext();) {
1960                 Pair catchPair = (Pair) iterator.next();
1961                 context.cfg().add(refNode, catchPair.snd, fNullPointerExcType);
1962               }
1963             } else {
1964               // connect exception edge to exit
1965               context.cfg().add(refNode, CAstControlFlowMap.EXCEPTION_TO_EXIT, fNullPointerExcType);
1966             }
1967     
1968             context.cfg().map(refNode, refNode);
1969           }
1970     
1971           if (fieldBinding.getConstantValue() != null) {
1972             // don't have to worry about generics, a constant of generic type can only be null
1973             return makeNode(context, fFactory, positioningNode, CAstNode.BLOCK_EXPR, refNode,
1974             // evaluating 'refNode' can have side effects, so we must still evaluate it!
1975                 fFactory.makeConstant(fieldBinding.getConstantValue()));
1976           } else {
1977             if (fieldBinding.getType().isTypeVariable() && !context.needLValue()) {
1978               // GENERICS: add a cast
1979               ITypeBinding realtype = JDT2CAstUtils.getErasedType(possiblyParameterizedBinding.getType(), ast);
1980               ITypeBinding fromtype = JDT2CAstUtils.getTypesVariablesBase(fieldBinding.getType(), ast);
1981               if (!realtype.isEqualTo(fromtype))
1982                 return createCast(positioningNode, refNode, fromtype, realtype, context);
1983             }
1984             return refNode;
1985           }
1986         }
1987     
1988       }

2057       private CAstNode createInfixExpression(InfixExpression.Operator op, ITypeBinding leftType, int leftStartPosition, int leftLength,
2058           CAstNode leftNode, Expression right, WalkContext context) {
2059         CAstNode rightNode = visitNode(right, context);
2060     
2061         int start = leftStartPosition;
2062         int end = right.getStartPosition() + right.getLength();
2063         JdtPosition pos = makePosition(start, end);
2064         JdtPosition rightPos = makePosition(leftStartPosition, leftStartPosition + leftLength);
2065         JdtPosition leftPos = makePosition(right.getStartPosition(), right.getStartPosition() + right.getLength());
2066     
2067         if (op == InfixExpression.Operator.CONDITIONAL_AND) {
2068           return makeNode(context, fFactory, pos, CAstNode.IF_EXPR, leftNode, rightNode, fFactory.makeConstant(false));
2069         } else if (op == InfixExpression.Operator.CONDITIONAL_OR) {
2070           return makeNode(context, fFactory, pos, CAstNode.IF_EXPR, leftNode, fFactory.makeConstant(true), rightNode);
2071         } else {
2072           ITypeBinding rightType = right.resolveTypeBinding();
2073           if (leftType.isPrimitive() && rightType.isPrimitive()) {
2074             // TODO: boxing
2075             ITypeBinding result = JDT2CAstUtils.promoteTypes(leftType, rightType, ast);
2076     
2077             // cast to proper type
2078             if (!result.isEqualTo(leftType))
2079               leftNode = makeNode(context, fFactory, leftPos, CAstNode.CAST, fFactory.makeConstant(fTypeDict.getCAstTypeFor(result)),
2080                   leftNode, fFactory.makeConstant(fTypeDict.getCAstTypeFor(leftType)));
2081             if (!result.isEqualTo(rightType))
2082               rightNode = makeNode(context, fFactory, rightPos, CAstNode.CAST, fFactory.makeConstant(fTypeDict.getCAstTypeFor(result)),
2083                   rightNode, fFactory.makeConstant(fTypeDict.getCAstTypeFor(rightType)));
2084     
2085             CAstNode opNode = makeNode(context, fFactory, pos, CAstNode.BINARY_EXPR, JDT2CAstUtils.mapBinaryOpcode(op), leftNode,
2086                 rightNode);
2087     
2088             // divide by zero exception implicitly thrown
2089             if (JDT2CAstUtils.isLongOrLess(leftType)
2090                 && JDT2CAstUtils.isLongOrLess(rightType)
2091                 && (JDT2CAstUtils.mapBinaryOpcode(op) == CAstOperator.OP_DIV || JDT2CAstUtils.mapBinaryOpcode(op) == CAstOperator.OP_MOD)) {
2092               Collection excTargets = context.getCatchTargets(fDivByZeroExcType);
2093               if (!excTargets.isEmpty()) {
2094                 for (Iterator iterator = excTargets.iterator(); iterator.hasNext();) {
2095                   Pair catchPair = (Pair) iterator.next();
2096                   context.cfg().add(op, catchPair.snd, fDivByZeroExcType);
2097                 }
2098               } else {
2099                 context.cfg().add(op, CAstControlFlowMap.EXCEPTION_TO_EXIT, fDivByZeroExcType);
2100               }
2101             }
2102     
2103             return opNode;
2104     
2105           } else {
2106             return makeNode(context, fFactory, pos, CAstNode.BINARY_EXPR, JDT2CAstUtils.mapBinaryOpcode(op), leftNode, rightNode);
2107           }
2108         }
2109       }

2379       private CAstNode visit(SwitchStatement n, WalkContext context) {
2380         ASTNode breakTarget = makeBreakOrContinueTarget(n, "breakLabel" + n.getStartPosition());
2381         CAstNode breakAst = visitNode(breakTarget, context);
2382         String loopLabel = (String) context.getLabelMap().get(n); // set by labeled statement (if there is one before this
2383         // switch statement)
2384         WalkContext childContext = new BreakContext(context, loopLabel, breakTarget);
2385         Expression cond = n.getExpression();
2386         List/* <Statement> */cases = n.statements();
2387     
2388         // First compute the control flow edges for the various case labels
2389         for (int i = 0; i < cases.size(); i++) {
2390           Statement se = (Statement) cases.get(i);
2391           if (se instanceof SwitchCase) {
2392             SwitchCase c = (SwitchCase) se;
2393     
2394             if (c.isDefault())
2395               context.cfg().add(n, c, CAstControlFlowMap.SWITCH_DEFAULT);
2396             else
2397               context.cfg().add(n, c, getSwitchCaseConstant(c, context));
2398             // if we don't do this, we may not get a constant but a
2399             // block expression or something else
2400           }
2401         }
2402     
2403         ArrayList<CAstNode> caseNodes = new ArrayList<CAstNode>();
2404     
2405         // polyglot bundles all statements in between two statements into a block.
2406         // this is temporary place to hold current bundle of nodes.
2407         ArrayList<CAstNode> currentBlock = new ArrayList<CAstNode>();
2408     
2409         // Now produce the CAst representation for each case
2410         for (Object o : cases) {
2411           Statement s = (Statement) o;
2412           if (s instanceof SwitchCase) {
2413             if (!currentBlock.isEmpty()) {
2414               // bundle up statements before this case
2415               CAstNode stmtNodes[] = currentBlock.toArray(new CAstNode[currentBlock.size()]);
2416               // make position from start of first statement to end of last statement
2417               JdtPosition positionOfAll = makePosition(childContext.pos().getPosition(stmtNodes[0]).getFirstOffset(), childContext
2418                   .pos().getPosition(stmtNodes[stmtNodes.length - 1]).getLastOffset());
2419               caseNodes.add(makeNode(childContext, fFactory, positionOfAll, CAstNode.BLOCK_STMT, stmtNodes));
2420               currentBlock.clear();
2421             }
2422             caseNodes.add(visitNode(s, childContext));
2423           } else {
2424             visitNodeOrNodes(s, childContext, currentBlock);
2425           }
2426         }
2427         if (!currentBlock.isEmpty()) {
2428           // bundle up statements before this case
2429           CAstNode stmtNodes[] = currentBlock.toArray(new CAstNode[currentBlock.size()]);
2430           // make position from start of first statement to end of last statement
2431           JdtPosition positionOfAll = makePosition(childContext.pos().getPosition(stmtNodes[0]).getFirstOffset(), childContext.pos()
2432               .getPosition(stmtNodes[stmtNodes.length - 1]).getLastOffset());
2433           caseNodes.add(makeNode(childContext, fFactory, positionOfAll, CAstNode.BLOCK_STMT, stmtNodes));
2434         }
2435     
2436         // Now produce the switch stmt itself
2437         CAstNode switchAst = makeNode(context, fFactory, n, CAstNode.SWITCH, visitNode(cond, context), makeNode(context, fFactory, n,
2438             CAstNode.BLOCK_STMT, caseNodes.toArray(new CAstNode[caseNodes.size()])));
2439     
2440         context.cfg().map(n, switchAst);
2441     
2442         // Finally, wrap the entire switch in a block so that we have a
2443         // well-defined place to 'break' to.
2444         return makeNode(context, fFactory, n, CAstNode.BLOCK_STMT, switchAst, breakAst);
2445     
2446       }

3150

3152             CatchClause c = (CatchClause) catchIter.next();

3159         @Override
3160         public Collection<Pair<ITypeBinding, Object>> getCatchTargets(ITypeBinding label) {
3161           // Look for all matching targets for this thrown type:
3162           // if supertpe match, then return only matches at this catch
3163           // if subtype match, then matches here and parent matches
3164           Collection<Pair<ITypeBinding, Object>> catchNodes = new ArrayList<Pair<ITypeBinding, Object>>();
3165     
3166           for (Iterator<Pair<ITypeBinding, Object>> iter = fCatchNodes.iterator(); iter.hasNext();) {
3167             Pair<ITypeBinding, Object> p = (Pair<ITypeBinding, Object>) iter.next();
3168             ITypeBinding catchType = (ITypeBinding) p.fst;
3169     
3170             // catchType here should NEVER be FakeExceptionTypeBinary, because these can only be thrown (not caught) by
3171             // "1/0", implicit null pointer exceptions, etc.
3172             assert !(catchNodes instanceof FakeExceptionTypeBinding) : "catchNodes instanceof FakeExceptionTypeBinary!";
3173     
3174             if (label.isSubTypeCompatible(catchType) || label.isEqualTo(catchType)) {
3175               catchNodes.add(p);
