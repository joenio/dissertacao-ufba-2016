125       /**
126        * Get all the nodes in CG with name funName. If funName is of the form
127        * <code>"ctor:nm"</code>, return nodes corresponding to constructor function
128        * for <code>nm</code>. If funName is of the form <code>"suffix:nm"</code>,
129        * return nodes corresponding to functions whose names end with
130        * <code>nm</code>. Otherwise, return nodes for functions whose name matches
131        * funName exactly.
132        */
133       public static Collection<CGNode> getNodes(CallGraph CG, String funName) {
134         boolean suffix = funName.startsWith("suffix:");
135         
136         if (suffix) {
137           Set<CGNode> nodes = new HashSet<CGNode>();
138           String tail = funName.substring(7);
139           for (CGNode n : CG) {
140             if (n.getMethod().getReference().getDeclaringClass().getName().toString().endsWith(tail)) {
141               nodes.add(n);
142             }
143           }
144           return nodes;
145         }
146         
147         MethodReference MR = getMethodReference(funName);
148     
149         return CG.getNodes(MR);
150       }

152       public static MethodReference getMethodReference(String funName) {
153         boolean ctor = funName.startsWith("ctor:");
154         MethodReference MR;
155         if (ctor) {
156           TypeReference TR = TypeReference.findOrCreate(JavaScriptTypes.jsLoader, TypeName.string2TypeName("L" + funName.substring(5)));
157           MR = JavaScriptMethods.makeCtorReference(TR);
158         } else {
159           TypeReference TR = TypeReference.findOrCreate(JavaScriptTypes.jsLoader, TypeName.string2TypeName("L" + funName));
160           MR = AstMethodReference.fnReference(TR);
161         }
162         return MR;
163       }

165       /**
166        * @param cha
167        * @param cl
168        * @param fileName
169        * @param url
170        * @return The set of class names that where defined in the CHA as a result
171        *         loading process.
172        * @throws IOException
173        */
174       public static Set<String> loadAdditionalFile(IClassHierarchy cha, JavaScriptLoader cl, String fileName, URL url)
175           throws IOException {
176         try {
177           SourceURLModule M = new SourceURLModule(url);
178           TranslatorToCAst toCAst = getTranslatorFactory().make(new CAstImpl(), M);
179           final Set<String> names = new HashSet<String>();
180           JSAstTranslator toIR = new JSAstTranslator(cl) {
181             @Override
182             protected void defineFunction(CAstEntity N, WalkContext definingContext, AbstractCFG cfg, SymbolTable symtab,
183                 boolean hasCatchBlock, Map<IBasicBlock,TypeReference[]> caughtTypes, boolean hasMonitorOp, AstLexicalInformation LI,
184                 DebuggingInformation debugInfo) {
185               String fnName = "L" + composeEntityName(definingContext, N);
186               names.add(fnName);
187               super.defineFunction(N, definingContext, cfg, symtab, hasCatchBlock, caughtTypes, hasMonitorOp, LI, debugInfo);
188             }
189     
190             @Override
191             protected void leaveFunctionStmt(CAstNode n, WalkContext c, CAstVisitor<WalkContext> visitor) {
192               CAstEntity fn = (CAstEntity) n.getChild(0).getValue();
193               Scope cs = c.currentScope();
194               if (!cs.contains(fn.getName())
195                   || cs.lookup(fn.getName()).getDefiningScope().getEntity().getKind() == CAstEntity.SCRIPT_ENTITY) {
196                 int result = processFunctionExpr(n, c);
197                 assignValue(n, c, cs.lookup(fn.getName()), fn.getName(), result);
198               } else {
199                 super.leaveFunctionStmt(n, c, visitor);
200               }
201             }
202           };
203           CAstEntity tree;
204           try {
205             tree = toCAst.translateToCAst();
206             if (DEBUG) {
207               CAstPrinter.printTo(tree, new PrintWriter(System.err));
208             }
209             toIR.translate(tree, M);
210             for (String name : names) {
211               IClass fcls = cl.lookupClass(name, cha);
212               cha.addClass(fcls);
213             }
214             return names;
215           } catch (Error e) {
216             return Collections.emptySet();
217           }
218          } catch (RuntimeException e) {
219           return Collections.emptySet();
220         }
221       }

224         // TODO this is very incomplete  --MS
225         if (v instanceof String) {
226           return (String)v;
227         } else if (v instanceof Double) {

230             result = Long.toString(Math.round((Double)v));
231           }
232           return result;
233         } else if (v instanceof Boolean) {
234           if (((Boolean)v).booleanValue()) {
235             return "true";
236           } else {
237             return "false";
238           }
239         } else {
240           return null;
241         }
