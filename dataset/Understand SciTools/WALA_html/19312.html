11     package com.ibm.wala.ipa.slicer;
12     
13     import com.ibm.wala.ipa.callgraph.CGNode;
14     import com.ibm.wala.ssa.SSAAbstractInvokeInstruction;
15     
16     /**
17      * A {@link Statement} representing an actual parameter
18      */
19     public class ParamCaller extends StatementWithInstructionIndex implements ValueNumberCarrier {
20       /**
21        * Value number of the actual parameter
22        */
23       protected final int valueNumber;
24     
25       public ParamCaller(CGNode node, int callIndex, int valueNumber) {
26         super(node, callIndex);
27         this.valueNumber = valueNumber;
28       }
29     
30       @Override
31       public Kind getKind() {
32         return Kind.PARAM_CALLER;
33       }
34     
35       @Override
36       public SSAAbstractInvokeInstruction getInstruction() {
37         return (SSAAbstractInvokeInstruction)super.getInstruction();
38       }
39     
40       @Override
41       public String toString() {
42         return super.toString() +  " v" + getValueNumber();
43       }
44     
45       @Override
46       public int getValueNumber() {
47         return valueNumber;
48       }
49     
50       @Override
51       public int hashCode() {
52         final int prime = 31;
53         int result = super.hashCode();
54         result = prime * result + valueNumber;
55         return result;
56       }
57     
58       @Override
59       public boolean equals(Object obj) {
60         if (this == obj)
61           return true;
62         if (!super.equals(obj))
63           return false;
64         if (getClass() != obj.getClass())
65           return false;
66         final ParamCaller other = (ParamCaller) obj;
67         if (valueNumber != other.valueNumber)
68           return false;
69         return true;
70       }
71     
72     
73     }
