20       public static class ReferenceToken {
21         private final byte kind;
22         private final String className;
23         private final String elementName;
24         private final String descriptor;
25         
26         public ReferenceToken(byte kind, String className, String elementName, String descriptor) {
27           this.kind = kind;
28           this.className = className;
29           this.elementName = elementName;
30           this.descriptor = descriptor;
31         }
32       
33         public byte getKind() {
34           return kind;
35         }
36     
37         public String getClassName() {
38           return className;
39         }
40     
41         public String getElementName() {
42           return elementName;
43         }
44     
45         public String getDescriptor() {
46           return descriptor;
47         }
48     
49         @Override
50         public int hashCode() {
51           final int prime = 31;
52           int result = 1;
53           result = prime * result + ((className == null) ? 0 : className.hashCode());
54           result = prime * result + ((descriptor == null) ? 0 : descriptor.hashCode());
55           result = prime * result + ((elementName == null) ? 0 : elementName.hashCode());
56           return result;
57         }
58       
59         @Override
60         public boolean equals(Object obj) {
61           if (this == obj)
62             return true;
63           if (obj == null)
64             return false;
65           if (getClass() != obj.getClass())
66             return false;
67           ReferenceToken other = (ReferenceToken) obj;
68           if (kind != other.kind) {
69             return false;
70           }
71           if (className == null) {
72             if (other.className != null)
73               return false;
74           } else if (!className.equals(other.className))
75             return false;
76           if (descriptor == null) {
77             if (other.descriptor != null)
78               return false;
79           } else if (!descriptor.equals(other.descriptor))
80             return false;
81           if (elementName == null) {
82             if (other.elementName != null)
83               return false;
84           } else if (!elementName.equals(other.elementName))
85             return false;
86           return true;
87         }
88       }

101       private BootstrapMethodsReader getBootstrapReader() throws InvalidClassFileException {
102         if (invokeDynamicBootstraps == null) { 
103           ClassReader thisClass = new ClassReader(bytes);
104           AttrIterator attrs = new AttrIterator();
105           thisClass.initClassAttributeIterator(attrs);
106           for (; attrs.isValid(); attrs.advance()) {
107             if (attrs.getName().equals("BootstrapMethods")) {
108               invokeDynamicBootstraps = new BootstrapMethodsReader(attrs);
109               break;
110             }
111           }
112           assert invokeDynamicBootstraps != null;
113         }
114         
115         return invokeDynamicBootstraps;
116       }

191       /**
192        * @return the name of the Class at constant pool item i, in JVM format (e.g., java/lang/Object)
193        */
194       public String getCPClass(int i) throws InvalidClassFileException, IllegalArgumentException {
195         if (i < 1 || i >= cpItems.length) {
196           throw new IllegalArgumentException("Constant pool item #" + i + " out of range");
197         }
198         int offset = cpOffsets[i];
199         if (offset == 0 || getByte(offset) != CONSTANT_Class) {
200           throw new IllegalArgumentException("Constant pool item #" + i + " is not a Class");
201         }
202         String s = cpItems[i];
203         if (s == null) {
204           try {
205             s = getCPUtf8(getUShort(offset + 1));
206           } catch (IllegalArgumentException ex) {
207             throw new InvalidClassFileException(offset, "Invalid class name at constant pool item #" + i + ": " + ex.getMessage());
208           }
209           cpItems[i] = s;
210         }
211         return s;
212       }

544       private InvalidClassFileException invalidUtf8(int item, int offset) {
545         return new InvalidClassFileException(offset, "Constant pool item #" + item + " starting at " + cpOffsets[item]
546             + ", is an invalid Java Utf8 string (byte is " + getByte(offset) + ")");
547       }

549       /**
550        * @return the value of the Utf8 string at constant pool item i
551        */
552       public String getCPUtf8(int i) throws InvalidClassFileException, IllegalArgumentException {
553         if (i < 1 || i >= cpItems.length) {
554           throw new IllegalArgumentException("Constant pool item #" + i + " out of range");
555         }
556         int offset = cpOffsets[i];
557         if (offset == 0 || getByte(offset) != CONSTANT_Utf8) {
558           throw new IllegalArgumentException("Constant pool item #" + i + " is not a Utf8");
559         }
560         String s = cpItems[i];
561         if (s == null) {
562           int count = getUShort(offset + 1);
563           int end = count + offset + 3;
564           StringBuffer buf = new StringBuffer(count);
565           offset += 3;
566           while (offset < end) {
567             byte x = getByte(offset);
568             if ((x & 0x80) == 0) {
569               if (x == 0) {
570                 throw invalidUtf8(i, offset);
571               }
572               buf.append((char) x);
573               offset++;
574             } else if ((x & 0xE0) == 0xC0) {
575               if (offset + 1 >= end) {
576                 throw invalidUtf8(i, offset);
577               }
578               byte y = getByte(offset + 1);
579               if ((y & 0xC0) != 0x80) {
580                 throw invalidUtf8(i, offset);
581               }
582               buf.append((char) (((x & 0x1F) << 6) + (y & 0x3F)));
583               offset += 2;
584             } else if ((x & 0xF0) == 0xE0) {
585               if (offset + 2 >= end) {
586                 throw invalidUtf8(i, offset);
587               }
588               byte y = getByte(offset + 1);
589               byte z = getByte(offset + 2);
590               if ((y & 0xC0) != 0x80 || (z & 0xC0) != 0x80) {
591                 throw invalidUtf8(i, offset);
592               }
593               buf.append((char) (((x & 0x0F) << 12) + ((y & 0x3F) << 6) + (z & 0x3F)));
594               offset += 3;
595             } else {
596               throw invalidUtf8(i, offset);
597             }
598           }
599           // s = buf.toString().intern(); // removed intern() call --MS
600           s = buf.toString();
601           cpItems[i] = s;
602         }
603         return s;
604       }

606       private void parseConstantPool(int offset, int itemCount) throws InvalidClassFileException {
607         cpOffsets = new int[itemCount + 1];
608         cpItems = new String[itemCount];
609         for (int i = 1; i < itemCount; i++) {
610           cpOffsets[i] = offset;
611           byte tag = getByte(offset);
612           int itemLen;
613           switch (tag) {
614           case CONSTANT_String:
615           case CONSTANT_Class:
616             itemLen = 2;
617             break;
618           case CONSTANT_NameAndType:
619           case CONSTANT_MethodRef:
620           case CONSTANT_FieldRef:
621           case CONSTANT_InterfaceMethodRef:
622           case CONSTANT_Integer:
623           case CONSTANT_Float:
624             itemLen = 4;
625             break;
626           case CONSTANT_Long:
627           case CONSTANT_Double:
628             itemLen = 8;
629             i++; // ick
630             break;
631           case CONSTANT_Utf8:
632             itemLen = 2 + getUShort(offset + 1);
633             break;
634           case CONSTANT_MethodHandle:
635             itemLen = 3;
636             break;
637           case CONSTANT_MethodType:
638             itemLen = 2;
639             break;
640           case CONSTANT_InvokeDynamic:
641             itemLen = 4;
642             break;
643           default:
644             throw new InvalidClassFileException(offset, "unknown constant pool entry type" + tag);
645           }
646           checkLength(offset, itemLen);
647           offset += itemLen + 1;
648         }
649         cpOffsets[itemCount] = offset;
650       }

656       private int getUShort(int i) {
657         return ((bytes[i] & 0xFF) << 8) + (bytes[i + 1] & 0xFF);
658       }

664       private int getInt(int i) {
665         return (bytes[i] << 24) + ((bytes[i + 1] & 0xFF) << 16) + ((bytes[i + 2] & 0xFF) << 8) + (bytes[i + 3] & 0xFF);
666       }

668       private long getLong(int i) {
669         return ((long) getInt(i) << 32) + (getInt(i + 4) & 0xFFFFFFFFL);
670       }
671     
672       private float getFloat(int i) {
673         return Float.intBitsToFloat(getInt(i));
674       }

676       private double getDouble(int i) {
677         return Double.longBitsToDouble(getLong(i));
