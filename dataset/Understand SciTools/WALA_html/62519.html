40     /**
41      * This is a demo class.
42      * 
43      * Class files are taken as input arguments (or if there are none, from standard input). The methods in those files are
44      * instrumented: we insert a System.err.println() at ever method call, and a System.err.println() at every method entry.
45      * 
46      * In Unix, I run it like this: java -cp ~/dev/shrike/shrike com.ibm.wala.shrikeBT.shrikeCT.tools.Bench test.jar -o output.jar
47      * 
48      * The instrumented classes are placed in the directory "output" under the current directory. Disassembled code is written to the
49      * file "report" under the current directory.
50      */
51     public class Mangler {
52       private static OfflineInstrumenter instrumenter;
53     
54       private static final boolean verify = true;
55     
56       private static final boolean disasm = true;
57     
58       public static void main(String[] args) throws Exception {
59         for (int i = 0; i < 1; i++) {
60           instrumenter = new OfflineInstrumenter(true);
61     
62           Writer w = new BufferedWriter(new FileWriter("report", false));
63     
64           args = instrumenter.parseStandardArgs(args);
65           int seed;
66           try {
67             seed = Integer.parseInt(args[0]);
68           } catch (NumberFormatException ex) {
69             System.err.println("Invalid number: " + args[0]);
70             w.close();
71             return;
72           }
73     
74           Random r = new Random(seed);
75           instrumenter.setPassUnmodifiedClasses(true);
76           instrumenter.beginTraversal();
77           instrumenter.setOutputJar(new File("output.jar"));
78           ClassInstrumenter ci;
79           while ((ci = instrumenter.nextClass()) != null) {
80             doClass(ci, w, r);
81           }
82           instrumenter.close();
83         }
84       }
85     
86       private static void doClass(final ClassInstrumenter ci, Writer w, final Random r) throws Exception {
87         final String className = ci.getReader().getName();
88         w.write("Class: " + className + "\n");
89         w.flush();
90     
91         for (int m = 0; m < ci.getReader().getMethodCount(); m++) {
92           MethodData d = ci.visitMethod(m);
93     
94           // d could be null, e.g., if the method is abstract or native
95           if (d != null) {
96             w.write("Instrumenting " + ci.getReader().getMethodName(m) + " " + ci.getReader().getMethodType(m) + ":\n");
97             w.flush();
98     
99             if (disasm) {
100               w.write("Initial ShrikeBT code:\n");
101               (new Disassembler(d)).disassembleTo(w);
102               w.flush();
103             }
104     
105             if (verify) {
106               Verifier v = new Verifier(d);
107               v.verify();
108             }
109     
110             final int passes = r.nextInt(4) + 1;
111     
112             for (int i = 0; i < passes; i++) {
113               final boolean doGet = true; // r.nextBoolean();
114               final boolean doPut = true; // r.nextBoolean();
115               final boolean doArrayStore = true; // r.nextBoolean();
116               final int tmpInt = LocalAllocator.allocate(d, "I");
117               final int tmpAny = LocalAllocator.allocate(d);
118     
119               final MethodEditor me = new MethodEditor(d);
120               me.beginPass();
121     
122               me.visitInstructions(new MethodEditor.Visitor() {
123                 @Override
124                 public void visitGet(IGetInstruction instruction) {
125                   if (doGet && !instruction.isStatic()) {
126                     insertBefore(new MethodEditor.Patch() {
127                       @Override
128                       public void emitTo(Output w) {
129                         w.emit(DupInstruction.make(0));
130                       }
131                     });
132                     insertAfter(new MethodEditor.Patch() {
133                       @Override
134                       public void emitTo(Output w) {
135                         w.emit(SwapInstruction.make());
136                         w.emit(Util.makePut(Slots.class, "o"));
137                       }
138                     });
139                   }
140                 }
141     
142                 @Override
143                 public void visitPut(IPutInstruction instruction) {
144                   if (doPut && !instruction.isStatic()) {
145                     insertBefore(new MethodEditor.Patch() {
146                       @Override
147                       public void emitTo(Output w) {
148                         w.emit(SwapInstruction.make());
149                         w.emit(DupInstruction.make(1));
150                         w.emit(SwapInstruction.make());
151                       }
152                     });
153                     insertAfter(new MethodEditor.Patch() {
154                       @Override
155                       public void emitTo(Output w) {
156                         w.emit(Util.makePut(Slots.class, "o"));
157                       }
158                     });
159                   }
160                 }
161     
162                 @Override
163                 public void visitArrayStore(final IArrayStoreInstruction instruction) {
164                   if (doArrayStore) {
165                     final int label = me.allocateLabel();
166                     insertBefore(new MethodEditor.Patch() {
167                       @Override
168                       public void emitTo(Output w) {
169                         String t = Util.getStackType(instruction.getType());
170                         w.emit(StoreInstruction.make(t, tmpAny));
171                         w.emit(StoreInstruction.make(Constants.TYPE_int, tmpInt));
172                         w.emit(DupInstruction.make(0));
173                         w.emit(LoadInstruction.make(Constants.TYPE_int, tmpInt));
174                         w.emit(LoadInstruction.make(t, tmpAny));
175                         if (t.equals(Constants.TYPE_int)) {
176                           w.emit(DupInstruction.make(0));
177                           w.emit(ConstantInstruction.make(0));
178                           w.emit(ConditionalBranchInstruction.make(t, ConditionalBranchInstruction.Operator.EQ, label));
179                           w.emit(DupInstruction.make(0));
180                           w.emit(Util.makePut(Slots.class, "i"));
181                           w.emitLabel(label);
182                         }
183                       }
184                     });
185                     insertAfter(new MethodEditor.Patch() {
186                       @Override
187                       public void emitTo(Output w) {
188                         w.emit(Util.makePut(Slots.class, "o"));
189                         w.emit(LoadInstruction.make(Constants.TYPE_int, tmpInt));
190                         w.emit(Util.makePut(Slots.class, "i"));
191                       }
192                     });
193                   }
194                 }
195               });
196     
197               // this updates the data d
198               me.applyPatches();
199             }
200     
201             if (disasm) {
202               w.write("Final ShrikeBT code:\n");
203               (new Disassembler(d)).disassembleTo(w);
204               w.flush();
205             }
206           }
207         }
208     
209         if (ci.isChanged()) {
210           ClassWriter cw = ci.emitClass();
211           instrumenter.outputModifiedClass(ci, cw);
212         }
213       }
214     }
