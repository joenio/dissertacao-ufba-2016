11     package com.ibm.wala.util.intset;
12     
13     /**
14      * utilities for manipulating values at the bit-level.
15      */
16     public class Bits {
17     
18       // there's no reason to instantiate this class.
19       private Bits() {
20       }
21     
22       /**
23        * Return the lower 8 bits (as an int) of an int
24        */
25       public static int lower8(int value) {
26         return (value & 0xff);
27       }
28     
29       /**
30        * Return the lower 16 bits (as an int) of an int
31        */
32       public static int lower16(int value) {
33         return (value & 0xffff);
34       }
35     
36       /**
37        * Return the upper 16 bits (as an int) of an int
38        */
39       public static int upper16(int value) {
40         return value >>> 16;
41       }
42     
43       /**
44        * Return the upper 24 bits (as an int) of an int
45        */
46       public static int upper24(int value) {
47         return value >>> 8;
48       }
49     
50       /**
51        * Return the lower 32 bits (as an int) of a long
52        */
53       public static int lower32(long value) {
54         return (int) value;
55       }
56     
57       /**
58        * Return the upper 32 bits (as an int) of a long
59        */
60       public static int upper32(long value) {
61         return (int) (value >>> 32);
62       }
63     
64       /**
65        * Does an int literal val fit in bits bits?
66        */
67       public static boolean fits(int val, int bits) {
68         val = val >> bits - 1;
69         return (val == 0 || val == -1);
70       }
71     
72       /**
73        * Return the number of ones in the binary representation of an integer. Hank Warren's Hacker's Delight algorithm
74        */
75       public static int populationCount(int value) {
76         int result = ((value & 0xAAAAAAAA) >>> 1) + (value & 0x55555555);
77         result = ((result & 0xCCCCCCCC) >>> 2) + (result & 0x33333333);
78         result = ((result & 0xF0F0F0F0) >>> 4) + (result & 0x0F0F0F0F);
79         result = ((result & 0xFF00FF00) >>> 8) + (result & 0x00FF00FF);
80         result = ((result & 0xFFFF0000) >>> 16) + (result & 0x0000FFFF);
81         return result;
82       }
83     }
