22     import com.ibm.wala.ipa.cha.IClassHierarchy;
23     import com.ibm.wala.util.intset.IntSet;
24     
25     /**
26      * Default object to control context-insensitive context selection, This includes reflection logic.
27      */
28     public class DefaultContextSelector implements ContextSelector {
29     
30       private final ContextSelector delegate;
31     
32       public DefaultContextSelector(AnalysisOptions options, IClassHierarchy cha) {
33         if (options == null) {
34           throw new IllegalArgumentException("null options");
35         }
36         ContextInsensitiveSelector ci = new ContextInsensitiveSelector();
37         ContextSelector r = ReflectionContextSelector.createReflectionContextSelector(options);
38         ContextSelector s = new DelegatingContextSelector(r, ci);
39         delegate = new DelegatingContextSelector(new CloneContextSelector(cha), s);
40       }
41     
42       @Override
43       public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] receiver) {
44         if (caller == null) {
45           throw new IllegalArgumentException("null caller");
46         }
47         return delegate.getCalleeTarget(caller, site, callee, receiver);
48       }
49     
50       @Override
51       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
52         return delegate.getRelevantParameters(caller, site);
53       }
54     
55     }
