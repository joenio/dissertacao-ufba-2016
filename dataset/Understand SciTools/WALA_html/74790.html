11     package com.ibm.wala.dataflow.graph;
12     
13     import com.ibm.wala.fixpoint.IVariable;
14     import com.ibm.wala.util.graph.Graph;
15     
16     /**
17      * A dataflow framework in the style of Kildall, POPL 73
18      * This represents a dataflow problem induced over a graph.
19      * 
20      * @param <T> type of nodes in the graph
21      */
22     @SuppressWarnings("rawtypes")
23     public interface IKilldallFramework<T,V extends IVariable> {
24     
25       /**
26        * @return the flow graph which induces this dataflow problem
27        */
28       public Graph<T> getFlowGraph();
29       
30       /**
31        * @return an object which provides the flow function for each node in the graph
32        */
33       public ITransferFunctionProvider<T,V> getTransferFunctionProvider();
34     }
