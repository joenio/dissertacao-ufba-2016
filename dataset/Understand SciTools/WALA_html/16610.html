11     package com.ibm.wala.cast.ipa.callgraph;

13     import java.util.Map;
14     
15     import com.ibm.wala.classLoader.CallSiteReference;
16     import com.ibm.wala.classLoader.IMethod;
17     import com.ibm.wala.ipa.callgraph.CGNode;
18     import com.ibm.wala.ipa.callgraph.Context;

20     import com.ibm.wala.ipa.callgraph.propagation.InstanceKey;
21     import com.ibm.wala.util.intset.IntSet;
22     import com.ibm.wala.util.strings.Atom;
23     
24     /**
25      * A ContextSelector implementation adapted to work for analysis across
26      * multiple languages.  This context selector delegates to one of several
27      * child selectors based on the language of the code body for which a 
28      * context is being selected.  
29      *
30      *  This provides a convenient way to integrate multiple, language-specific
31      * specialized context policies---such as the ones used for clone() in
32      * Java and runtime primitives in JavaScript.
33      *
34      * @author Julian Dolby (dolby@us.ibm.com)
35      */
36     public class CrossLanguageContextSelector implements ContextSelector {
37     
38       private final Map languageSelectors;
39     
40       public CrossLanguageContextSelector(Map languageSelectors) {
41         this.languageSelectors = languageSelectors;
42       }
43     
44       private static Atom getLanguage(CallSiteReference site) {
45         return site
46     	.getDeclaredTarget()
47     	.getDeclaringClass()
48     	.getClassLoader()
49     	.getLanguage();
50       }
51     
52       private ContextSelector getSelector(CallSiteReference site) {
53         return (ContextSelector)languageSelectors.get(getLanguage(site));
54       }
55     
56       @Override
57       public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] receiver) {
58         return getSelector(site).getCalleeTarget(caller, site, callee, receiver);
59       }
60     
61       @Override
62       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
63         return getSelector(site).getRelevantParameters(caller, site);
64       }
65     }
