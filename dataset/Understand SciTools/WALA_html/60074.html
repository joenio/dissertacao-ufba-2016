45     /**
46      * Understands {@link com.ibm.wala.analysis.reflection.GetMethodContext}.

52       /**
53        * TODO
54        *  MH: Maybe hard-code those in {@link com.ibm.wala.types.MethodReference}?
55        */
56       public final static MethodReference GET_METHOD = MethodReference.findOrCreate(TypeReference.JavaLangClass, "getMethod",
57           "(Ljava/lang/String;[Ljava/lang/Class;)Ljava/lang/reflect/Method;");
58     
59       /**
60        * TODO

62        */
63       public final static MethodReference GET_DECLARED_METHOD = MethodReference.findOrCreate(TypeReference.JavaLangClass,
64           "getDeclaredMethod", "(Ljava/lang/String;[Ljava/lang/Class;)Ljava/lang/reflect/Method;");

68       /**
69        * @see com.ibm.wala.ipa.callgraph.propagation.SSAContextInterpreter#getIR(com.ibm.wala.ipa.callgraph.CGNode)
70        */
71       @Override
72       public IR getIR(CGNode node) {
73         if (node == null) {
74           throw new IllegalArgumentException("node is null");
75         }
76         assert understands(node);
77         if (DEBUG) {
78           System.err.println("generating IR for " + node);
79         }
80         IMethod method = node.getMethod();
81         GetMethodContext context = (GetMethodContext) node.getContext();
82         Map<Integer,ConstantValue> constants = HashMapFactory.make();
83         if (method.getReference().equals(GET_METHOD)) {
84           Atom name = Atom.findOrCreateAsciiAtom(context.getName());
85           SSAInstruction instrs[] = makeGetMethodStatements(context,constants,name);
86           return new SyntheticIR(method, context, new InducedCFG(instrs, method, context), instrs, SSAOptions.defaultOptions(), constants);
87         }
88         if (method.getReference().equals(GET_DECLARED_METHOD)) {
89           Atom name = Atom.findOrCreateAsciiAtom(context.getName());
90           SSAInstruction instrs[] = makeGetDeclaredMethodStatements(context,constants,name);
91           return new SyntheticIR(method, context, new InducedCFG(instrs, method, context), instrs, SSAOptions.defaultOptions(), constants);
92         }
93         Assertions.UNREACHABLE("Unexpected method " + node);
94         return null;
95       }

106       /**
107        * @see com.ibm.wala.ipa.callgraph.propagation.rta.RTAContextInterpreter#understands(com.ibm.wala.ipa.callgraph.CGNode)
108        */
109       @Override
110       public boolean understands(CGNode node) {
111         if (node == null) {
112           throw new IllegalArgumentException("node is null");
113         }
114         if (!(node.getContext() instanceof GetMethodContext)) {
115           return false;
116         }
117         MethodReference mRef = node.getMethod().getReference();
118         return mRef.equals(GET_METHOD) || mRef.equals(GET_DECLARED_METHOD);
119       }

121       @Override
122       public Iterator<NewSiteReference> iterateNewSites(CGNode node) {
123         if (node == null) {
124           throw new IllegalArgumentException("node is null");
125         }
126         assert understands(node);
127         GetMethodContext context = (GetMethodContext) node.getContext();
128         TypeReference tr = context.getType().getTypeReference();
129         if (tr != null) {
130           return new NonNullSingletonIterator<NewSiteReference>(NewSiteReference.make(0, tr));
131         }
132         return EmptyIterator.instance();
133       }

141       /**
142        * Get all non-constructor, non-class-initializer methods declared by a class
143        * if their name is equal to the specified name.
144        * @param cls the class
145        * @param name the name
146        */
147       private Collection<IMethod> getDeclaredNormalMethods(IClass cls,Atom name) {
148         Collection<IMethod> result = HashSetFactory.make();
149         for (IMethod m : cls.getDeclaredMethods()) {
150           if (!m.isInit() && !m.isClinit() && m.getSelector().getName().equals(name)) {
151             result.add(m);
152           }
153         }
154         return result;
155       }
156     
157       /**
158        * Get all non-constructor, non-class-initializer methods declared by a class
159        * and all its superclasses if their name is equal to the specified name.
160        * @param cls the class
161        * @param name the name
162        */
163       private Collection<IMethod> getAllNormalPublicMethods(IClass cls,Atom name) {
164         Collection<IMethod> result = HashSetFactory.make();
165         Collection<IMethod> allMethods = null;
166         allMethods = cls.getAllMethods();
167         for (IMethod m : allMethods) {
168           if (!m.isInit() && !m.isClinit() && m.isPublic() && m.getSelector().getName().equals(name)) {
169             result.add(m);
170           }
171         }
172         return result;
173       }

175       /**
176        * Create statements for methods like getMethod() and getDeclaredMethod(),
177        * which return a single method. This creates a return statement for each
178        * possible return value, each of which is a {@link ConstantValue} for an
179        * {@link IMethod}.
180        * @param returnValues the possible return values for this method
181        * @return the statements
182        */
183       private SSAInstruction[] getParticularMethodStatements
184           (
185             MethodReference ref,
186             Collection<IMethod> returnValues,
187             GetMethodContext context,
188             Map<Integer, ConstantValue> constants
189            ) {
190         ArrayList<SSAInstruction> statements = new ArrayList<SSAInstruction>();
191         int nextLocal = ref.getNumberOfParameters() + 2;
192         IClass cls = context.getType().getType();
193         SSAInstructionFactory insts = context.getType().getType().getClassLoader().getInstructionFactory();
194         if (cls != null) {
195           for (IMethod m : returnValues) {
196             int c = nextLocal++;
197             constants.put(c, new ConstantValue(m));
198             SSAReturnInstruction R = insts.ReturnInstruction(statements.size(), c, false);
199             statements.add(R);
200           }
201         } else {
202           // SJF: This is incorrect. TODO: fix and enable.
203           // SSAThrowInstruction t = insts.ThrowInstruction(retValue);
204           // statements.add(t);
205         }
206         SSAInstruction[] result = new SSAInstruction[statements.size()];
207         Iterator<SSAInstruction> it = statements.iterator();
208         for (int i = 0; i < result.length; i++) {
209           result[i] = it.next();
210         }
211         return result;
212       }
213     
214       private SSAInstruction[] makeGetMethodStatements
215             (
216               GetMethodContext context,
217               Map<Integer,ConstantValue> constants,
218               Atom name
219             ) {
220         IClass cls = context.getType().getType();
221         if (cls == null) {
222           return getParticularMethodStatements(GET_METHOD, null, context, constants);
223         } else {
224           return getParticularMethodStatements(GET_METHOD, getAllNormalPublicMethods(cls,name), context, constants);
225         }
226       }
227     
228       /**
229        * Create statements for {@link Class#getDeclaredMethod(String, Class...)}.
230        */
231       private SSAInstruction[] makeGetDeclaredMethodStatements(GetMethodContext context, Map<Integer, ConstantValue> constants,Atom name) {
232         IClass cls = context.getType().getType();
233         if (cls == null) {
234           return getParticularMethodStatements(GET_DECLARED_METHOD, null, context, constants);
235         } else {
236           return getParticularMethodStatements(GET_DECLARED_METHOD, getDeclaredNormalMethods(cls,name), context, constants);
237         }
238       }
