12     package com.ibm.wala.cfg.exc.intra;
13     
14     import com.ibm.wala.ssa.SSAAbstractInvokeInstruction;
15     
16     /**
17      * Provides a way for the nullpointer analysis to decide whether or not a called method
18      * may throw an exception.
19      * 
20      * @author Juergen Graf <graf@kit.edu>
21      *
22      */
23     public abstract class MethodState {
24     
25       public abstract boolean throwsException(SSAAbstractInvokeInstruction node);
26       
27       public static final MethodState DEFAULT = new MethodState() {
28         
29         @Override
30         public boolean throwsException(SSAAbstractInvokeInstruction node) {
31           // per default assume that every call may throw an exception
32           return true;
33         }
34       };
35     }
