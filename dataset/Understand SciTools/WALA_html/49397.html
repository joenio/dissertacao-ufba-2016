11     package com.ibm.wala.cast.ipa.callgraph;
12     
13     import java.util.Iterator;
14     import java.util.Set;
15     
16     import com.ibm.wala.cast.ipa.callgraph.AstCallGraph.ScriptFakeRoot;
17     import com.ibm.wala.classLoader.CallSiteReference;
18     import com.ibm.wala.classLoader.IClass;
19     import com.ibm.wala.classLoader.IMethod;
20     import com.ibm.wala.ipa.callgraph.Entrypoint;
21     import com.ibm.wala.ipa.callgraph.impl.AbstractRootMethod;
22     import com.ibm.wala.ipa.cha.IClassHierarchy;
23     import com.ibm.wala.ssa.SSAAbstractInvokeInstruction;
24     import com.ibm.wala.types.TypeReference;
25     import com.ibm.wala.util.collections.HashSetFactory;
26     
27     public abstract class ScriptEntryPoints implements Iterable<Entrypoint> {
28     
29       private final IClassHierarchy cha;
30     
31       private final IClass scriptType;
32     
33       private class ScriptEntryPoint extends Entrypoint {
34         ScriptEntryPoint(IMethod scriptCodeBody) {
35           super(scriptCodeBody);
36         }
37           
38         @Override
39         public CallSiteReference makeSite(int programCounter) {
40           return makeScriptSite(getMethod(), programCounter);
41         }
42     
43         @Override
44         public TypeReference[] getParameterTypes(int i) {
45           assert i == 0;
46           if (getMethod().isStatic()) {
47     	return new TypeReference[0];
48           } else {
49     	return new TypeReference[] { getMethod().getDeclaringClass().getReference() };
50           }
51         }
52     
53         @Override
54         public int getNumberOfParameters() {
55           return getMethod().isStatic()? 0: 1;
56         }
57     
58           
59         @Override
60         public SSAAbstractInvokeInstruction addCall(AbstractRootMethod m){
61           CallSiteReference site = makeSite(0);
62     
63           if (site == null) {
64             return null;
65           }
66     
67           int functionVn = getMethod().isStatic()? -1: makeArgument(m, 0);
68           int paramVns[] = new int[Math.min(0, getNumberOfParameters() - 1)];
69           for (int j = 0; j < paramVns.length; j++) {
70             paramVns[j] = makeArgument(m, j + 1);
71           }
72     
73           return ((ScriptFakeRoot) m).addDirectCall(functionVn, paramVns, site);
74         }
75       }
76     
77       public ScriptEntryPoints(IClassHierarchy cha, IClass scriptType) {
78         this.cha = cha;
79         this.scriptType = scriptType;
80       }
81     
82       protected abstract CallSiteReference makeScriptSite(IMethod m, int pc);
83     
84       protected boolean keep(IMethod method) {
85         return true;
86       }
87       
88       @Override
89       public Iterator<Entrypoint> iterator() {
90         Set<Entrypoint> ES = HashSetFactory.make();
91         Iterator<IClass> classes = scriptType.getClassLoader().iterateAllClasses();
92         while (classes.hasNext()) {
93           IClass cls = classes.next();
94           if (cha.isSubclassOf(cls, scriptType) && !cls.isAbstract()) {
95             for (IMethod method : cls.getDeclaredMethods()) {
96               if (keep(method)) {
97                 ES.add(new ScriptEntryPoint(method));
98               }
99             }
100           }
101         }
102     
103         return ES.iterator();
104       }
105     
106       public Entrypoint make(String scriptName) {
107         IClass cls = cha.lookupClass(TypeReference.findOrCreate(scriptType.getClassLoader().getReference(), scriptName));
108         assert cls != null && cha.isSubclassOf(cls, scriptType) && !cls.isAbstract() : String.valueOf(cls) + " for " + scriptName;
109         for (IMethod method : cls.getDeclaredMethods()) {
110           if (keep(method)) {
111             return new ScriptEntryPoint(method);
112           }
113         }
114         
115         assert false;
116         return null;
117       }
118     }
