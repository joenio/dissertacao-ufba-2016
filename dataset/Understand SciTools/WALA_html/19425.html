67     /**
68      *  Read in an extracted AndroidManifest.xml.
69      *
70      *  The file has to be in the extracted (human readable) XML-Format. You can extract it using the program
71      *  `apktool`.
72      *
73      *  Tags and Attributes not known by the Parser are skipped over.
74      *
75      *  To add a Tag to the parsed ones:
76      *      You have to extend the enum Tags: All Tags on the path to the Tag in question have to be in the enum.
77      *      Eventually you will have to adapt the ParserItem of the Parent-Tags and add it to their allowed Sub-Tags. 
78      *
79      * To add an Attribute to the handled ones:
80      *      You'll have to extend the Enum Attrs if the Attribute-Name is not yet present there. Then add the 
81      *      Attribute to the relevant Attributes of it's containing Tag.
82      *
83      *      You will be able to access it using attributesHistory.get(Attr).peek()
84      *
85      * TODO:
86      * @todo    Handle Info in the DATA-Tag correctly!
87      * @since   2013-10-13
88      * @author  Tobias Blaschke <code@tobiasblaschke.de>
89      */
90     public class AndroidManifestXMLReader {

92          *  This logs low-level parsing.

96          *  parser generates all objects using it.
97          */

100         public AndroidManifestXMLReader(File xmlFile) throws IOException {
101             if (xmlFile == null) {
102                 throw new IllegalArgumentException("xmlFile may not be null");
103             }
104             try {
105                 readXML(new FileInputStream(xmlFile));
106             } catch (Exception e) {
107                 e.printStackTrace();
108                 throw new IllegalStateException("Exception was thrown");
109             }
110         }

124         private void readXML(InputStream xml) throws SAXException, IOException, ParserConfigurationException {
125             assert (xml != null) : "xmlFile may not be null";
126     
127             final SAXHandler handler = new SAXHandler();
128             final SAXParserFactory factory = SAXParserFactory.newInstance();
129             factory.newSAXParser().parse(new InputSource(xml), handler);

382         /**
383          *  Handling of a Tag.
384          *
385          *  Does (if not overridden) all the needed Push- and Pop-Operations. Items may however choose to leave some
386          *  stuff on the Stack. In this case this data has to be popped by the handling parent, or the Tag may only
387          *  occur once or bad things will happen.
388          *
389          *  _CAUTION_: This will be instantiated by an Enum, so if you write local Fields you will get surprising 
390          *  results! You should mark all of them as final to be sure.
391          */
392         private static abstract class ParserItem {
393             protected Tag self;
394             /**
395              *  Set the Tag this ParserItem-Instance is an Handler for.
396              *
397              *  This may only be set once!
398              */
399             public void setSelf(Tag self) {
400                 if (this.self != null) {
401                     throw new IllegalStateException("Self can only be set once!");
402                 }
403                 this.self = self;
404             }
405             
406             public ParserItem() {
407             }
408             
409             /**
410              *  Remember attributes to the tag. 
411              *
412              *  The read attributes will be pushed to the attributesHistory.
413              *
414              *  Leave Parser-Stack alone! This is called by SAXHandler only!
415              */    
416             public void enter(Attributes saxAttrs) {
417                 for (Attr relevant : self.getRelevantAttributes()) {
418                     String attr = saxAttrs.getValue(relevant.getName());
419                     if (attr == null) {
420                         attr = saxAttrs.getValue("android:" + relevant.getName());
421                     }
422     
423                     attributesHistory.get(relevant).push(attr);
424                     logger.debug("Pushing '{}' for {} in {}", attr, relevant, self);
425                     // if there is no such value in saxAttrs it returns null 
426                 }
427             }
428             /**
429              *  Remove all Attributes generated by self and self itself.
430              *
431              *  This is called by the consuming ParserItem.
432              */
433             public void popAttributes() {
434                  for (Attr relevant : self.getRelevantAttributes()) {
435                      try {
436                         logger.debug("Popping {} of value {} in {}", relevant, attributesHistory.get(relevant).peek(), self);
437                         attributesHistory.get(relevant).pop();
438                      }  catch (java.util.EmptyStackException e) {
439                         System.err.println(self + " failed to pop " + relevant);
440                         throw e;
441                      }
442                  }
443                  if (attributesHistory.containsKey(self) && attributesHistory.get(self) != null &&
444                          (! attributesHistory.get(self).isEmpty())) {
445                      try {
446                          attributesHistory.get(self).pop();
447                      } catch (java.util.EmptyStackException e) {
448                         System.err.println("The Stack for " + self + " was Empty when trying to pop");
449                         throw e;
450                      }
451                  }
452             }
453     
454             /**
455              *  Consume sub-items on the stack.
456              *
457              *  Do this by popping them, but leave self on the stack!
458              *  For each Item popped call its popAttributes()!
459              */
460             public void leave() {
461                 while (parserStack.peek() != self) {
462                     final Set<Tag> allowedSubTags = self.getAllowedSubTags();
463                     Tag subTag = parserStack.pop();
464                     if (allowedSubTags.contains(subTag)) {
465                         if (subTag.getHandler() == null) {
466                             throw new IllegalArgumentException("The SubTag " + subTag.toString() + " has no handler!");
467                         }
468                         subTag.getHandler().popAttributes(); // hmmm....
469     
470                         logger.debug("New Stack: {}", parserStack);
471                         //parserStack.pop();
472                     } else {
473                         throw new IllegalStateException(subTag + " is not allowed as sub-tag of " + self + " in Context:\n\t" + parserStack);
474                     }
475                 }
476             }
477         }

529         /**
530          *  Read the specification of an Intent from AndroidManifest.
531          *
532          *  @todo   Handle the URI
533          */
534         private static class IntentItem extends ParserItem {
535             public IntentItem() {
536                 super();
537             }
538             @Override
539             public void leave() {
540                 Set<Tag> allowedTags = EnumSet.copyOf(self.getAllowedSubTags());
541                 Set<String> urls = new HashSet<String>();
542                 Set<String> names = new HashSet<String>();
543                 while (parserStack.peek() != self) {
544                     Tag current = parserStack.pop();
545                     if (allowedTags.contains(current)) {
546                         if (current == Tag.ACTION) {
547                             Object oName = attributesHistory.get(Attr.NAME).peek();
548                             if (oName == null) {
549                                  throw new IllegalStateException("The currently parsed Action did not leave the required 'name' Attribute" +
550                                          " on the Stack! Attributes-Stack for name is: " + attributesHistory.get(Attr.NAME));
551                             } else if (oName instanceof String) {
552                                 names.add((String) oName);
553                             } else {
554                                 throw new IllegalStateException("Unexpected Attribute type for name: " + oName.getClass().toString());
555                             }
556                         } else if (current == Tag.DATA) {
557                             Object oUrl = attributesHistory.get(Attr.SCHEME).peek();
558                             if (oUrl == null) {
559                                 // TODO
560                             } else if (oUrl instanceof String) {
561                                 urls.add((String) oUrl);
562                             } else {
563                                 throw new IllegalStateException("Unexpected Attribute type for name: " + oUrl.getClass().toString());
564                             }
565                         } else {
566                             throw new IllegalStateException("Error in parser implementation");
567                         }
568                         current.getHandler().popAttributes();
569                     } else {
570                         throw new IllegalStateException("In INTENT: Tag " + current + " not allowed in Context " + parserStack + "\n\t"+
571                                 "Allowed Tags: " + allowedTags);
572                     }
573                 }
574     
575                 // Pushing intent...
576                 final String pack;
577                 if ((attributesHistory.get(Attr.PACKAGE) != null ) && (!(attributesHistory.get(Attr.PACKAGE).isEmpty()))) {
578                     pack = (String) attributesHistory.get(Attr.PACKAGE).peek();
579                 } else {
580                     logger.warn("Empty Package {}", attributesHistory.get(Attr.PACKAGE).peek());
581                     pack = null;
582                 }
583     
584                 if (!names.isEmpty()) {
585                     for (String name : names) {
586                         if (urls.isEmpty()) urls.add(null);
587                         for (String url : urls) {
588                             logger.info("New Intent ({}, {})", name, url);
589                             final Intent intent = AndroidSettingFactory.intent(name, url);
590                             attributesHistory.get(self).push(intent);
591                         }
592                 }
593                 } else {
594                     throw new IllegalStateException("Error in parser implementation! The required attribute 'name' which should have been " +
595                             "defined in ACTION could not be retrieved. This should have been thrown before as it is a required attribute for " +
596                             "ACTION");
597                 }
598             }
599         }

601         private static class ComponentItem extends ParserItem {
602             public ComponentItem() {
603                 super();
604             }
605             @Override
606              public void leave() {
607                 final Set<Tag> allowedTags = self.getAllowedSubTags();
608                 final Set<Intent> overrideTargets = new HashSet<Intent>(); 
609     
610                 while (parserStack.peek() != self) {
611                     Tag current = parserStack.pop();
612                     if (allowedTags.contains(current)) {
613                         if (current == Tag.INTENT) {
614                             Object oIntent = attributesHistory.get(Tag.INTENT).peek();
615                             if (oIntent == null) {
616                                  throw new IllegalStateException("The currently parsed Intent did not push a Valid intent to the " +
617                                          "Stack! Attributes-Stack for name is: " + attributesHistory.get(Attr.NAME));
618                             } else if (oIntent instanceof Intent) {
619                                 overrideTargets.add( (Intent) oIntent );
620                             } else {
621                                 throw new IllegalStateException("Unexpected Attribute type for Intent: " + oIntent.getClass().toString());
622                             }
623                         } else {
624                             throw new IllegalStateException("Error in parser implementation");
625                         }
626                         current.getHandler().popAttributes();
627                     } else {
628                         throw new IllegalStateException("In " + self + ": Tag " + current + " not allowed in Context " + parserStack + "\n\t"+
629                                 "Allowed Tags: " + allowedTags);
630                     }
631                 }
632     
633                 // Generating Intent for this...
634                 final String pack;
635                 if ((attributesHistory.get(Attr.PACKAGE) != null ) && (!(attributesHistory.get(Attr.PACKAGE).isEmpty()))) {
636                     pack = (String) attributesHistory.get(Attr.PACKAGE).peek();
637                 } else {
638                     logger.warn("Empty Package {}", attributesHistory.get(Attr.PACKAGE).peek());
639                     pack = null;
640                 }
641                 final String name = (String) attributesHistory.get(Attr.NAME).peek(); // TODO: Verify type!
642                 final Intent intent = AndroidSettingFactory.intent(pack, name, null);
643     
644                 logger.info("\tRegister: {}", intent);
645                 AndroidEntryPointManager.MANAGER.registerIntent(intent);
646                 for (Intent ovr: overrideTargets) {
647                     logger.info("\tOverride: {} --> {}", ovr, intent);
648                     AndroidEntryPointManager.MANAGER.setOverride(ovr, intent);
649                 }
650             }
651         }

653         private class SAXHandler extends DefaultHandler {
654             private int unimportantDepth = 0;
655     
656             public SAXHandler() {
657                 super();
658                 parserStack.push(Tag.ROOT);
659             }
660     
661             @Override
662             public void startElement(String uri, String name, String qName, Attributes attrs) {
663                 Tag tag = Tag.fromString(qName);
664                 if ((tag == Tag.UNIMPORTANT) || (unimportantDepth > 0)) {
665                     unimportantDepth++;
666                 } else {
667                     logger.debug("Handling {} made from {}", tag, qName);
668                     
669                     final ParserItem handler = tag.getHandler();
670                     if (handler != null) {
671                         handler.enter(attrs);
672                     }
673                     parserStack.push(tag);
674     
675                 }
676             }
677     
678             @Override
679             public void endElement(String uri, String localName, String qName) {
680                 if (unimportantDepth > 0) {
681                     unimportantDepth--;
682                 } else {
683                     final Tag tag = Tag.fromString(qName);
684                     final ParserItem handler = tag.getHandler();
685                     if (handler != null) {
686                         handler.leave();
687                     }
688                 }
689             }
690         }
