58     public class BinaryOperation extends Instruction {
59     
60         public static enum OpID {CMPL_FLOAT,CMPG_FLOAT,
61             CMPL_DOUBLE,CMPG_DOUBLE,
62             CMPL_LONG,CMPG_LONG,
63             CMPL_INT,CMPG_INT,
64             ADD_INT,SUB_INT,MUL_INT,DIV_INT,REM_INT,AND_INT,OR_INT,XOR_INT,SHL_INT,SHR_INT,USHR_INT,
65             ADD_LONG,SUB_LONG,MUL_LONG,DIV_LONG,REM_LONG,AND_LONG,OR_LONG,XOR_LONG,SHL_LONG,SHR_LONG,USHR_LONG,
66             ADD_FLOAT,SUB_FLOAT,MUL_FLOAT,DIV_FLOAT,REM_FLOAT,
67             ADD_DOUBLE,SUB_DOUBLE,MUL_DOUBLE,DIV_DOUBLE,REM_DOUBLE};
68     
69         public final OpID op;
70         public final int oper1;
71         public final int oper2;
72         public final int destination;
73     
74         public BinaryOperation(int pc, OpID op, int destination, int oper1, int oper2, Opcode opcode, DexIMethod method) {
75             super(pc, opcode, method);
76             this.op = op;
77             this.destination = destination;
78             this.oper1 = oper1;
79             this.oper2 = oper2;
80         }
81     
82         @Override
83         public void visit(Visitor visitor) {
84             visitor.visitBinaryOperation(this);
85         }
86     
87         public IBinaryOpInstruction.IOperator getOperator()
88         {
89             switch(op)
90             {
91             case CMPL_FLOAT:
92                 return CAstBinaryOp.LT;
93             case CMPG_FLOAT:
94                 return CAstBinaryOp.GT;
95             case CMPL_DOUBLE:
96                 return CAstBinaryOp.LT;
97             case CMPG_DOUBLE:
98                 return CAstBinaryOp.GT;
99             case CMPL_LONG:
100                 return CAstBinaryOp.LT;
101             case CMPG_LONG:
102                 return CAstBinaryOp.GT;
103             case CMPL_INT:
104                 return CAstBinaryOp.LT;
105             case CMPG_INT:
106                 return CAstBinaryOp.GT;
107             case ADD_INT:
108                 return IBinaryOpInstruction.Operator.ADD;
109             case SUB_INT:
110                 return IBinaryOpInstruction.Operator.SUB;
111             case MUL_INT:
112                 return IBinaryOpInstruction.Operator.MUL;
113             case DIV_INT:
114                 return IBinaryOpInstruction.Operator.DIV;
115             case REM_INT:
116                 return IBinaryOpInstruction.Operator.REM;
117             case AND_INT:
118                 return IBinaryOpInstruction.Operator.AND;
119             case OR_INT:
120                 return IBinaryOpInstruction.Operator.OR;
121             case XOR_INT:
122                 return IBinaryOpInstruction.Operator.XOR;
123             case SHL_INT:
124                 return IShiftInstruction.Operator.SHL;
125             case SHR_INT:
126                 return IShiftInstruction.Operator.SHR;
127             case USHR_INT:
128                 return IShiftInstruction.Operator.USHR;
129             case ADD_LONG:
130                 return IBinaryOpInstruction.Operator.ADD;
131             case SUB_LONG:
132                 return IBinaryOpInstruction.Operator.SUB;
133             case MUL_LONG:
134                 return IBinaryOpInstruction.Operator.MUL;
135             case DIV_LONG:
136                 return IBinaryOpInstruction.Operator.DIV;
137             case REM_LONG:
138                 return IBinaryOpInstruction.Operator.REM;
139             case AND_LONG:
140                 return IBinaryOpInstruction.Operator.AND;
141             case OR_LONG:
142                 return IBinaryOpInstruction.Operator.OR;
143             case XOR_LONG:
144                 return IBinaryOpInstruction.Operator.XOR;
145             case SHL_LONG:
146                 return IShiftInstruction.Operator.SHL;
147             case SHR_LONG:
148                 return IShiftInstruction.Operator.SHR;
149             case USHR_LONG:
150                 return IShiftInstruction.Operator.USHR;
151             case ADD_FLOAT:
152                 return IBinaryOpInstruction.Operator.ADD;
153             case SUB_FLOAT:
154                 return IBinaryOpInstruction.Operator.SUB;
155             case MUL_FLOAT:
156                 return IBinaryOpInstruction.Operator.MUL;
157             case DIV_FLOAT:
158                 return IBinaryOpInstruction.Operator.DIV;
159             case REM_FLOAT:
160                 return IBinaryOpInstruction.Operator.REM;
161             case ADD_DOUBLE:
162                 return IBinaryOpInstruction.Operator.ADD;
163             case SUB_DOUBLE:
164                 return IBinaryOpInstruction.Operator.SUB;
165             case MUL_DOUBLE:
166                 return IBinaryOpInstruction.Operator.MUL;
167             case DIV_DOUBLE:
168                 return IBinaryOpInstruction.Operator.DIV;
169             case REM_DOUBLE:
170                 return IBinaryOpInstruction.Operator.REM;
171             }
172             return null;
173         }
174         public boolean isFloat()
175         {
176             switch(op)
177             {
178             case CMPL_FLOAT:
179             case CMPG_FLOAT:
180             case CMPL_DOUBLE:
181             case CMPG_DOUBLE:
182             case ADD_FLOAT:
183             case SUB_FLOAT:
184             case MUL_FLOAT:
185             case DIV_FLOAT:
186             case REM_FLOAT:
187             case ADD_DOUBLE:
188             case SUB_DOUBLE:
189             case MUL_DOUBLE:
190             case DIV_DOUBLE:
191             case REM_DOUBLE:
192                 return true;
193             default:
194                 return false;
195             }
196         }
