39     /**
40      * Flexible class to create {@link InstanceKey}s depending on various policies ranging from class-based (i.e. 0-CFA) to
41      * allocation-site-based (0-1-CFA variants).
42      */
43     public class ZeroXInstanceKeys implements InstanceKeyFactory {

48           JavaLangStringBufferName);

52       public final static TypeReference JavaLangStringBuilder = TypeReference.findOrCreate(ClassLoaderReference.Primordial,

60       /**
61        * The NONE policy is not allocation-site based
62        */
63       public static final int NONE = 0;
64     
65       /**
66        * An ALLOCATIONS - based policy distinguishes instances by allocation site. Otherwise, the policy distinguishes instances by
67        * type.
68        */
69       public static final int ALLOCATIONS = 1;
70     
71       /**
72        * A policy variant where String and StringBuffers are NOT disambiguated according to allocation site.
73        */
74       public static final int SMUSH_STRINGS = 2;
75     
76       /**
77        * A policy variant where {@link Throwable} instances are NOT disambiguated according to allocation site.
78        * 
79        */
80       public static final int SMUSH_THROWABLES = 4;
81     
82       /**
83        * A policy variant where if a type T has only primitive instance fields, then instances of type T are NOT disambiguated by
84        * allocation site.
85        */
86       public static final int SMUSH_PRIMITIVE_HOLDERS = 8;
87     
88       /**
89        * This variant counts the N, number of allocation sites of a particular type T in each method. If N > SMUSH_LIMIT, then these N
90        * allocation sites are NOT distinguished ... instead there is a single abstract allocation site for <N,T>
91        * 
92        * Probably the best choice in many cases.
93        */
94       public static final int SMUSH_MANY = 16;
95     
96       /**
97        * Should we use constant-specific keys?
98        */
99       public static final int CONSTANT_SPECIFIC = 32;
100     
101       /**
102        * When using smushing, how many sites in a node will be kept distinct before smushing?
103        */
104       private final int SMUSH_LIMIT = 25;
105     
106       /**
107        * The policy choice for instance disambiguation
108        */
109       private final int policy;

111       /**
112        * A delegate object to create class-based abstract instances

114       private final ClassBasedInstanceKeys classBased;

116       /**

121       /**
122        * A delegate object to create "abstract allocation site" - based abstract instances

124       private final SmushedAllocationSiteInstanceKeys smushed;

139       protected final Map<CGNode, Set<IClass>> smushMap = HashMapFactory.make();

141       public ZeroXInstanceKeys(AnalysisOptions options, IClassHierarchy cha, RTAContextInterpreter contextInterpreter, int policy) {
142         if (options == null) {
143           throw new IllegalArgumentException("null options");
144         }
145         this.policy = policy;
146         if (disambiguateConstants()) {
147           // this is an ugly hack. TODO: clean it all up.
148           options.setUseConstantSpecificKeys(true);
149         }
150         classBased = new ClassBasedInstanceKeys(options, cha);
151         siteBased = new AllocationSiteInNodeFactory(options, cha);
152         smushed = new SmushedAllocationSiteInstanceKeys(options, cha);
153         this.cha = cha;
154         this.contextInterpreter = contextInterpreter;
155       }
156     
157       /**
158        * @return true iff the policy smushes some allocation sites
159        */
160       private boolean smushMany() {
161         return (policy & SMUSH_MANY) > 0;
162       }
163     
164       private boolean allocationPolicy() {
165         return (policy & ALLOCATIONS) > 0;
166       }
167     
168       private boolean smushStrings() {
169         return (policy & SMUSH_STRINGS) > 0;
170       }
171     
172       public boolean smushThrowables() {
173         return (policy & SMUSH_THROWABLES) > 0;
174       }
175     
176       private boolean smushPrimHolders() {
177         return (policy & SMUSH_PRIMITIVE_HOLDERS) > 0;
178       }
179     
180       public boolean disambiguateConstants() {
181         return (policy & CONSTANT_SPECIFIC) > 0;
182       }

184       @Override
185       public InstanceKey getInstanceKeyForAllocation(CGNode node, NewSiteReference allocation) {
186         if (allocation == null) {
187           throw new IllegalArgumentException("allocation is null");
188         }
189         TypeReference t = allocation.getDeclaredType();
190         IClass C = cha.lookupClass(t);
191     
192         if (C != null && isInteresting(C)) {
193           if (smushMany()) {
194             if (exceedsSmushLimit(C, node)) {
195               return smushed.getInstanceKeyForAllocation(node, allocation);
196             } else {
197               return siteBased.getInstanceKeyForAllocation(node, allocation);
198             }
199           } else {
200             return siteBased.getInstanceKeyForAllocation(node, allocation);
201           }
202         } else {
203           return classBased.getInstanceKeyForAllocation(node, allocation);
204         }
205       }
206     
207       /**
208        * side effect: populates the smush map.
209        * 
210        * @return true iff the node contains too many allocation sites of type c
211        */
212       private boolean exceedsSmushLimit(IClass c, CGNode node) {
213         Set<IClass> s = smushMap.get(node);
214         if (s == null) {
215           Map<IClass, Integer> count = countAllocsByType(node);
216           HashSet<IClass> smushees = HashSetFactory.make(5);
217           for (Iterator<Map.Entry<IClass, Integer>> it = count.entrySet().iterator(); it.hasNext();) {
218             Map.Entry<IClass, Integer> e = it.next();
219             Integer i = e.getValue();
220             if (i.intValue() > SMUSH_LIMIT) {
221               smushees.add(e.getKey());
222             }
223           }
224           s = smushees.isEmpty() ? Collections.<IClass> emptySet() : smushees;
225           smushMap.put(node, s);
226         }
227         return s.contains(c);
228       }
229     
230       /**
231        * @return Map: IClass -> Integer, the number of allocation sites for each type.
232        */
233       private Map<IClass, Integer> countAllocsByType(CGNode node) {
234         Map<IClass, Integer> count = HashMapFactory.make();
235         for (Iterator it = contextInterpreter.iterateNewSites(node); it.hasNext();) {
236           NewSiteReference n = (NewSiteReference) it.next();
237           IClass alloc = cha.lookupClass(n.getDeclaredType());
238           if (alloc != null) {
239             Integer old = count.get(alloc);
240             if (old == null) {
241               count.put(alloc, new Integer(1));
242             } else {
243               count.put(alloc, new Integer(old.intValue() + 1));
244             }
245           }
246         }
247         return count;
248       }

250       @Override
251       public InstanceKey getInstanceKeyForMultiNewArray(CGNode node, NewSiteReference allocation, int dim) {
252         if (allocationPolicy()) {
253           return siteBased.getInstanceKeyForMultiNewArray(node, allocation, dim);
254         } else {
255           return classBased.getInstanceKeyForMultiNewArray(node, allocation, dim);
256         }

259       @Override
260       public <T> InstanceKey getInstanceKeyForConstant(TypeReference type, T S) {
261         if (type == null) {
262           throw new IllegalArgumentException("null type");
263         }
264         if (disambiguateConstants() || isReflectiveType(type)) {
265           return new ConstantKey<T>(S, getClassHierarchy().lookupClass(type));
266         } else {
267           return classBased.getInstanceKeyForConstant(type, S);
268         }
269       }

271       private boolean isReflectiveType(TypeReference type) {
272         return type.equals(TypeReference.JavaLangReflectConstructor) || type.equals(TypeReference.JavaLangReflectMethod);
273       }

289       /**
290        * A class is "interesting" iff we distinguish instances of the class
291        */
292       public boolean isInteresting(IClass C) {
293         if (!allocationPolicy()) {
294           return false;
295         } else {
296           if (smushStrings() && isStringish(C)) {
297             return false;
298           } else if (smushThrowables() && (isThrowable(C) || isStackTraceElement(C))) {
299             return false;
300           } else if (smushPrimHolders() && allFieldsArePrimitive(C)) {
301             return false;
302           }
303           return true;
304         }
305       }
306     
307       public static boolean isStringish(IClass C) {
308         if (C == null) {
309           throw new IllegalArgumentException("C is null");
310         }
311         return C.getReference().equals(TypeReference.JavaLangString) || C.getReference().equals(JavaLangStringBuffer)
312             || C.getReference().equals(JavaLangStringBuilder) || C.getReference().equals(JavaLangAbstractStringBuilder);
313       }
314     
315       public static boolean isThrowable(IClass c) {
316         if (c == null) {
317           throw new IllegalArgumentException("null c");
318         }
319         return c.getClassHierarchy().isSubclassOf(c, c.getClassHierarchy().lookupClass(TypeReference.JavaLangThrowable));
320       }
321     
322     
323       public static boolean isStackTraceElement(IClass c) {
324         if (c == null) {
325           throw new IllegalArgumentException("C is null");
326         }
327         return c.getReference().equals(TypeReference.JavaLangStackTraceElement);
328       }
329     
330     
331       private boolean allFieldsArePrimitive(IClass c) {
332         if (c.isArrayClass()) {
333           TypeReference t = c.getReference().getArrayElementType();
334           return t.isPrimitiveType();
335         } else {
336           if (c.getReference().equals(TypeReference.JavaLangObject)) {
337             return true;
338           } else {
339             for (Iterator<IField> it = c.getDeclaredInstanceFields().iterator(); it.hasNext();) {
340               IField f = it.next();
341               if (f.getReference().getFieldType().isReferenceType()) {
342                 return false;
343               }
344             }
345             return allFieldsArePrimitive(c.getSuperclass());
346           }
347         }
348       }
