17     import java.io.File;
18     import java.util.Collection;
19     import java.util.Collections;
20     
21     import com.ibm.wala.cast.tree.CAstNode;
22     import com.ibm.wala.cast.tree.CAstQualifier;
23     import com.ibm.wala.cast.tree.CAstType;
24     
25     public class AbstractScriptEntity extends AbstractCodeEntity {
26       private final File file;
27     
28       public AbstractScriptEntity(File file, CAstType type) {
29         super(type);
30         this.file = file;
31       }
32     
33       public AbstractScriptEntity(String file, CAstType type) {
34         this(new File(file), type);
35       }
36     
37       @Override
38       public int getKind() {
39         return SCRIPT_ENTITY;
40       }
41     
42       protected File getFile() {
43         return file;
44       }
45     
46       @Override
47       public String getName() {
48         return "script " + file.getName();
49       }
50     
51       @Override
52       public String toString() {
53         return "script " + file.getName();
54       }
55     
56       @Override
57       public String[] getArgumentNames() {
58         return new String[] { "script object" };
59       }
60     
61       @Override
62       public CAstNode[] getArgumentDefaults() {
63         return new CAstNode[0];
64       }
65     
66       @Override
67       public int getArgumentCount() {
68         return 1;
69       }
70     
71       @Override
72       public Collection<CAstQualifier> getQualifiers() {
73         return Collections.emptySet();
74       }
75     
76       public String getFileName() {
77         return file.getAbsolutePath();
78       }
79     }
