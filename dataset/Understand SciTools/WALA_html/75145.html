33     /**
34      * This class represents class loaders that introduce classes that do not exist
35      * in the actual application being analyzed. They may be abstract summaries of
36      * unanalyzed library code, wrappers that encode J2EE specialized behavior or
37      * other invented classes.
38      * 
39      * The intention is that there be (at most) one such classloader in a given
40      * class hierarchy, and that it be referenced using the "Synthetic" classloader
41      * reference. Furthermore, it is required that this synthetic loader be a child
42      * loader of the Primordial, Extension and Application loaders.
43      * 
44      * This special classloader has some interactions with the hierarchy for, while
45      * the classes it loads are normal-seeming IClass objects, unlike the other
46      * loaders, its set of classes is not fixed, causing special cases in code that
47      * caches hierarchy data. Also note that this causes the getNumberfClasses and
48      * iterateAllClasses methods to behave differently for those of other
49      * classloaders.
50      * 
51      * Code that wants to introduce synthetic classes uses the registerClass method,
52      * giving it an Atom which is the class name, and an IClass which is the class
53      * to load. Since the synthetic loader musat be a child of the others, it would
54      * be very bad to use an existing name for a new synthetic class.
55      * 
56      * Class lookup works just as for any other classloader.
57      * 
58      * @author Julian Dolby (dolby@us.ibm.com)
59      * 
60      */
61     public class BypassSyntheticClassLoader implements IClassLoader {
62     
63       private final ClassLoaderReference me;
64     
65       private final IClassLoader parent;
66     
67       private final IClassHierarchy cha;
68     
69       private final HashMap<TypeName, IClass> syntheticClasses = HashMapFactory.make();
70     
71       
72       /**
73        * Don't change my signature!  ClassLoaderFactoryImpl calls me by reflection! yuck.
74        * 
75        * @param me the name of this class loader
76        * @param parent its parent
77        * @param exclusions classes to ignore
78        * @param cha governing class hierarchy
79        */
80       public BypassSyntheticClassLoader(ClassLoaderReference me, IClassLoader parent, SetOfClasses exclusions, IClassHierarchy cha) {
81         if (cha == null) {
82           throw new IllegalArgumentException("null cha");
83         }
84         this.me = me;
85         this.cha = cha;
86         this.parent = parent;
87       }
88       
89       @Override
90       public String toString() {
91         return me.getName().toString();
92       }
93     
94       @Override
95       public IClass lookupClass(TypeName className) {
96         IClass pc = parent.lookupClass(className);
97         if (pc == null) {
98           IClass c = syntheticClasses.get(className);
99           return c;
100         } else {
101           return pc;
102         }
103       }
104     
105       /**
106        * Register the existence of a new synthetic class
107        */
108       public void registerClass(TypeName className, IClass theClass) {
109         cha.addClass(theClass);
110         syntheticClasses.put(className, theClass);
111       }
112     
113       /**
114        * Return the ClassLoaderReference for this class loader.
115        */
116       @Override
117       public ClassLoaderReference getReference() {
118         return me;
119       }
120     
121       /**
122        * @return an Iterator of all classes loaded by this loader
123        */
124       @Override
125       public Iterator<IClass> iterateAllClasses() {
126         return syntheticClasses.values().iterator();
127       }
128     
129       /**
130        * @return the number of classes in scope to be loaded by this loader
131        */
132       @Override
133       public int getNumberOfClasses() {
134         return syntheticClasses.size();
135       }
136     
137       /**
138        * @return the unique name that identifies this class loader.
139        */
140       @Override
141       public Atom getName() {
142         return me.getName();
143       }
144     
145       /**
146        * @return the unique name that identifies the programming language
147        *  from which this class loader loads code.
148        */
149       @Override
150       public Language getLanguage() {
151         return Language.JAVA;
152       }
153     
154       /*
155        * @see com.ibm.wala.classLoader.IClassLoader#getNumberOfMethods()
156        */
157       @Override
158       public int getNumberOfMethods() {
159         // TODO Auto-generated method stub
160         return 0;
161       }
162     
163       /*
164        * @see com.ibm.wala.classLoader.IClassLoader#getSourceFileName(com.ibm.wala.classLoader.IClass)
165        */
166       @Override
167       public String getSourceFileName(IClass klass) {
168         return null;
169       }
170     
171       /**
172        * @see com.ibm.wala.classLoader.IClassLoader#getParent()
173        */
174       @Override
175       public IClassLoader getParent() {
176         return parent;
177       }
178     
179       @Override
180       public void init(List<Module> modules) throws IOException {
181       }
182     
183       /*
184        * @see com.ibm.wala.classLoader.IClassLoader#removeAll(java.util.Collection)
185        */
186       @Override
187       public void removeAll(Collection<IClass> toRemove) {
188         if (toRemove == null) {
189           throw new IllegalArgumentException("toRemove is null");
190         }
191         for (Iterator<IClass> it = toRemove.iterator(); it.hasNext();) {
192           IClass klass = it.next();
193           syntheticClasses.remove(klass.getName());
194         }
195       }
196     
197       @Override
198       public Reader getSource(IClass klass) {
199         return null;
200       }
201     
202       @Override
203       public SSAInstructionFactory getInstructionFactory() {
204         return getLanguage().instructionFactory();
205       }
206     
207       @Override
208       public Reader getSource(IMethod method, int offset) {
209         return null;
210       }
211     
212       @Override
213       public String getSourceFileName(IMethod method, int offset) {
214         return null;
215       }
216     }
