44     public class CFGSanitizerTest extends WalaTestCase {
45     
46       /**
47        * check that for all synthetic methods coming from the native specifications, the exit block is not disconnected from the rest of
48        * the sanitized graph
49        * 
50        * @throws IOException
51        * @throws IllegalArgumentException
52        * @throws WalaException
53        */
54       @Test
55       public void testSyntheticEdgeToExit() throws IOException, IllegalArgumentException, WalaException {
56         AnalysisScope scope = AnalysisScopeReader.makePrimordialScope((new FileProvider()).getFile(CallGraphTestUtil.REGRESSION_EXCLUSIONS));
57     
58         ClassHierarchy cha = ClassHierarchy.make(scope);
59         ClassLoader cl = CFGSanitizerTest.class.getClassLoader();
60         InputStream s = cl.getResourceAsStream("natives.xml");
61         XMLMethodSummaryReader summary = new XMLMethodSummaryReader(s, scope);
62         AnalysisOptions options = new AnalysisOptions(scope, null);
63         Map<MethodReference, MethodSummary> summaries = summary.getSummaries();
64         for (MethodReference mr : summaries.keySet()) {
65           IMethod m = cha.resolveMethod(mr);
66           if (m == null) {
67             continue;
68           }
69           System.out.println(m.getSignature());
70           MethodSummary methodSummary = summaries.get(mr);
71           SummarizedMethod summMethod = new SummarizedMethod(mr, methodSummary, m.getDeclaringClass());
72           IR ir = summMethod.makeIR(Everywhere.EVERYWHERE, options.getSSAOptions());
73           System.out.println(ir);
74           Graph<ISSABasicBlock> graph = CFGSanitizer.sanitize(ir, cha);
75           System.out.println(graph);
76           BasicBlock exit = ir.getControlFlowGraph().exit();
77           if (!exit.equals(ir.getControlFlowGraph().entry())) {
78             Assert.assertTrue(graph.getPredNodeCount(exit) > 0);
79           }
80         }
81       }
