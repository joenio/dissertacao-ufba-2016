45       /*
46        * Use the passed-in SDG
47        */
48       public static Collection<Statement> computeBackwardSlice(SDG sdg, Collection<Statement> ss) throws IllegalArgumentException,
49           CancelException {
50         return computeSlice(sdg, ss, true);
51       }
52     
53       /**
54        * @param ss a collection of statements of interest
55        * @throws CancelException
56        */
57       public static Collection<Statement> computeSlice(SDG sdg, Collection<Statement> ss, boolean backward) throws CancelException {
58         return new AstJavaSlicer().slice(sdg, ss, backward);
59       }

61       public static Set<Statement> gatherStatements(CallGraph CG, Collection<CGNode> partialRoots, Predicate<SSAInstruction> filter) {
62         Set<Statement> result = new HashSet<Statement>();
63         for (Iterator<CGNode> ns = DFS.getReachableNodes(CG, partialRoots).iterator(); ns.hasNext();) {
64           CGNode n = ns.next();
65           IR nir = n.getIR();
66           if (nir != null) {
67     	SSAInstruction insts[] = nir.getInstructions();
68     	for (int i = 0; i < insts.length; i++) {
69               if (filter.test(insts[i])) {
70                 result.add(new NormalStatement(n, i));
71     	  }
72     	}
73           }
74         }
75     
76         return result;
77       }
78     
79       public static Set<Statement> gatherAssertions(CallGraph CG, Collection<CGNode> partialRoots) {
80         return gatherStatements(CG, partialRoots, new Predicate<SSAInstruction>() {
81           @Override public boolean test(SSAInstruction o) {
82             return o instanceof AstAssertInstruction;
83           }
84         });
85       }
86     
87       public static Set<Statement> gatherMonitors(CallGraph CG, Collection<CGNode> partialRoots) {
88         return gatherStatements(CG, partialRoots, new Predicate<SSAInstruction>() {
89           @Override public boolean test(SSAInstruction o) {
90             return o instanceof SSAMonitorInstruction;
91           }
92         });
93       }
94     
95       public static Set<Statement> gatherWrites(CallGraph CG, Collection<CGNode> partialRoots) {
96         return gatherStatements(CG, partialRoots, new Predicate<SSAInstruction>() {
97           @Override public boolean test(SSAInstruction o) {
98             return (o instanceof SSAPutInstruction) || (o instanceof SSAArrayStoreInstruction);
99           }
100         });
101       }
102     
103       public static Set<Statement> gatherReads(CallGraph CG, Collection<CGNode> partialRoots) {
104         return gatherStatements(CG, partialRoots, new Predicate<SSAInstruction>() {
105           @Override public boolean test(SSAInstruction o) {
106             return (o instanceof SSAGetInstruction) || (o instanceof SSAArrayLoadInstruction);
107           }
108         });
109       }
110     
111       public static Pair<Collection<Statement>, SDG> computeAssertionSlice(CallGraph CG, PointerAnalysis<InstanceKey> pa,
112           Collection<CGNode> partialRoots, boolean multiThreadedCode) throws IllegalArgumentException, CancelException {
113         CallGraph pcg = PartialCallGraph.make(CG, new LinkedHashSet<CGNode>(partialRoots));
114         SDG sdg = new SDG(pcg, pa, new AstJavaModRef(), DataDependenceOptions.FULL, ControlDependenceOptions.FULL);
115         //System.err.println(("SDG:\n" + sdg));
116         Set<Statement> stmts = gatherAssertions(CG, partialRoots);
117         if (multiThreadedCode) {
118           // Grab anything that has "side effects" under JMM
119           stmts.addAll(gatherReads(CG, partialRoots));
120           stmts.addAll(gatherWrites(CG, partialRoots));
121           stmts.addAll(gatherMonitors(CG, partialRoots));
122         }
123         return Pair.make(AstJavaSlicer.computeBackwardSlice(sdg, stmts), sdg);
124       }
