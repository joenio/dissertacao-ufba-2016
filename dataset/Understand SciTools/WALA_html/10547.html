11     package com.ibm.wala.util.strings;
12     
13     /**
14      * A read-only byte array.
15      */
16     public final class ImmutableByteArray {
17     
18       // allow "friends" in this package to access
19       final byte[] b;
20     
21       public ImmutableByteArray(byte[] b) {
22         if (b == null) {
23           throw new IllegalArgumentException("b is null");
24         }
25         this.b = b;
26       }
27     
28       public ImmutableByteArray(byte[] b, int start, int length) {
29         if (b == null) {
30           throw new IllegalArgumentException("b is null");
31         }
32         if (start < 0) {
33           throw new IllegalArgumentException("invalid start: " + start);
34         }
35         if (length < 0) {
36           throw new IllegalArgumentException("null length");
37         }
38         this.b = new byte[length];
39         try {
40           System.arraycopy(b, start, this.b, 0, length);
41         } catch (ArrayIndexOutOfBoundsException e) {
42           throw new IllegalArgumentException("illegal parameters " + b.length + " " + start + " " + length);
43         }
44       }
45     
46       public int length() {
47         return b.length;
48       }
49     
50       public byte get(int i) throws IllegalArgumentException {
51         if (i < 0 || i >= b.length) {
52           throw new IllegalArgumentException("index out of bounds " + b.length + " " + i);
53         }
54         return b[i];
55       }
56     
57       public byte[] substring(int i, int length) {
58         if (length < 0) {
59           throw new IllegalArgumentException("illegal length: " + length);
60         }
61         if (i < 0) {
62           throw new IllegalArgumentException("illegal i: " + i);
63         }
64         byte[] result = new byte[length];
65         try {
66           System.arraycopy(b, i, result, 0, length);
67         } catch (ArrayIndexOutOfBoundsException e) {
68           throw new IllegalArgumentException("Invalid combination: " + i + " " + length);
69         }
70         return result;
71       }
72     
73       public static ImmutableByteArray concat(byte b, ImmutableByteArray b1) {
74         if (b1 == null) {
75           throw new IllegalArgumentException("b1 is null");
76         }
77         byte[] arr = new byte[b1.length() + 1];
78         arr[0] = b;
79         System.arraycopy(b1.b, 0, arr, 1, b1.b.length);
80         return new ImmutableByteArray(arr);
81       }
82     
83       @Override
84       public String toString() {
85         return new String(b);
86       }
87     
88       public static ImmutableByteArray make(String s) {
89         return new ImmutableByteArray(UTF8Convert.toUTF8(s));
90       }
91     
92     }
