11     package com.ibm.wala.types.generics;
12     
13     import com.ibm.wala.types.TypeReference;
14     
15     public class BaseType extends TypeSignature {
16     
17       private final TypeReference t;
18       
19       final static BaseType BOOLEAN = new BaseType(TypeReference.BooleanName.toString(),TypeReference.Boolean);
20       final static BaseType BYTE     = new BaseType(TypeReference.ByteName.toString(),TypeReference.Byte);
21       final static BaseType SHORT = new BaseType(TypeReference.ShortName.toString(),TypeReference.Short);
22       final static BaseType INT = new BaseType(TypeReference.IntName.toString(),TypeReference.Int);
23       final static BaseType LONG = new BaseType(TypeReference.LongName.toString(),TypeReference.Long);
24       final static BaseType FLOAT = new BaseType(TypeReference.FloatName.toString(),TypeReference.Float);
25       final static BaseType DOUBLE  = new BaseType(TypeReference.DoubleName.toString(),TypeReference.Double);
26       final static BaseType CHAR = new BaseType(TypeReference.CharName.toString(),TypeReference.Char);
27       final static BaseType VOID = new BaseType(TypeReference.VoidName.toString(),TypeReference.Void);
28       
29       private BaseType(String s, TypeReference t) {
30         super(s);
31         this.t = t;
32       }
33     
34       @Override
35       public boolean isClassTypeSignature() {
36         return false;
37       }
38     
39       @Override
40       public boolean isTypeVariable() {
41         return false;
42       }
43     
44       public TypeReference getType() {
45         return t;
46       }
47     
48       @Override
49       public boolean isArrayTypeSignature() {
50         return false;
51       }
52     
53       @Override
54       public boolean isBaseType() {
55         return true;
56       }
57     
58     }
