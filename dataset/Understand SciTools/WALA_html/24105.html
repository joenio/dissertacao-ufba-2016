12     package com.ibm.wala.util.collections;
13     
14     import java.util.Collection;
15     import com.ibm.wala.util.Predicate;
16     
17     /**
18      * A filter defined by set membership
19      */
20     public class CollectionFilter<T> extends Predicate<T> {
21     
22       private final Collection<? extends T> S;
23     
24       public CollectionFilter(Collection<? extends T> S) {
25         if (S == null) {
26           throw new IllegalArgumentException("null S");
27         }
28         this.S = S;
29       }
30     
31       /*
32        * @see com.ibm.wala.util.Filter#accepts(java.lang.Object)
33        */
34       @Override public boolean test(T o) {
35         return S.contains(o);
36       }
37     
38     }
