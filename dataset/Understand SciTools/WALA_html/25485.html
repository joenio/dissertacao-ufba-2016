49       /**
50        * Perform needed bookkeeping when a new class is discovered.
51        */
52       @Override
53       protected void updateSetsForNewClass(IClass klass, InstanceKey iKey, CGNode node, NewSiteReference n) {
54     
55         // set up the selector map to record each method that class implements
56         registerImplementedMethods(klass, iKey);
57     
58         for (Iterator ifaces = klass.getAllImplementedInterfaces().iterator(); ifaces.hasNext();) {
59           IClass c = (IClass) ifaces.next();
60           registerImplementedMethods(c, iKey);
61         }
62         klass = klass.getSuperclass();
63         while (klass != null) {
64           registerImplementedMethods(klass, iKey);
65           klass = klass.getSuperclass();
66         }
67       }
68     
69       /**
70        * Record state for each method implemented by iKey.
71        */
72       private void registerImplementedMethods(IClass declarer, InstanceKey iKey) {
73         if (DEBUG) {
74           System.err.println(("registerImplementedMethods: " + declarer + " " + iKey));
75         }
76         for (Iterator it = declarer.getDeclaredMethods().iterator(); it.hasNext();) {
77           IMethod M = (IMethod) it.next();
78           Selector selector = M.getReference().getSelector();
79           PointerKey sKey = getKeyForSelector(selector);
80           if (DEBUG) {
81             System.err.println(("Add constraint: " + selector + " U= " + iKey.getConcreteType()));
82           }
83           system.newConstraint(sKey, iKey);
84         }
85       }

96       /**
97        * An operator to fire when we discover a potential new callee for a virtual or interface call site.
98        * 
99        * This operator will create a new callee context and constraints if necessary.
100        * 
101        * N.B: This implementation assumes that the calling context depends solely on the dataflow information computed for the receiver.
102        * TODO: generalize this to have other forms of context selection, such as CPA-style algorithms.
103        */
104       private final class DispatchOperator extends UnaryOperator<PointsToSetVariable> {
105         private final CallSiteReference site;
106     
107         private final ExplicitCallGraph.ExplicitNode caller;
108     
109         DispatchOperator(CallSiteReference site, ExplicitNode caller) {
110           this.site = site;
111           this.caller = caller;
112         }
113     
114         /**
115          * The set of classes that have already been processed.
116          */
117         final private MutableIntSet previousReceivers = IntSetUtil.getDefaultIntSetFactory().make();
118     
119         @Override
120         public byte evaluate(PointsToSetVariable lhs, PointsToSetVariable rhs) {
121           IntSetVariable receivers = rhs;
122     
123           // compute the set of pointers that were not previously handled
124           IntSet value = receivers.getValue();
125           if (value == null) {
126             // this constraint was put on the work list, probably by initialization,
127             // even though the right-hand-side is empty.
128             // TODO: be more careful about what goes on the worklist to
129             // avoid this.
130             if (DEBUG) {
131               System.err.println("EVAL dispatch with value null");
132             }
133             return NOT_CHANGED;
134           }
135           if (DEBUG) {
136             String S = "EVAL dispatch to " + caller + ":" + site;
137             System.err.println(S);
138             if (DEBUG_LEVEL >= 2) {
139               System.err.println(("receivers: " + value));
140             }
141           }
142           // TODO: cache this!!!
143           IClass recvClass = getClassHierarchy().lookupClass(site.getDeclaredTarget().getDeclaringClass());
144           if (recvClass == null) {
145             return NOT_CHANGED;
146           }
147           value = filterForClass(value, recvClass);
148           if (DEBUG_LEVEL >= 2) {
149             System.err.println(("filtered value: " + value));
150           }
151           IntSetAction action = new IntSetAction() {
152             @Override
153             public void act(int ptr) {
154               if (DEBUG) {
155                 System.err.println(("    dispatch to ptr " + ptr));
156               }
157               InstanceKey iKey = system.getInstanceKey(ptr);
158     
159               CGNode target = getTargetForCall(caller, site, iKey.getConcreteType(), new InstanceKey[]{iKey});
160               if (target == null) {
161                 // This indicates an error; I sure hope getTargetForCall
162                 // raised a warning about this!
163                 if (DEBUG) {
164                   System.err.println(("Warning: null target for call " + site + " " + iKey));
165                 }
166                 return;
167               }
168               if (clone2Assign) {
169                 if (target.getMethod().getReference().equals(CloneInterpreter.CLONE)) {
170                   // (mostly) ignore a call to clone: it won't affect the
171                   // solution, but we should probably at least have a call
172                   // edge
173                   caller.addTarget(site, target);
174                   return;
175                 }
176               }
177     
178               IntSet targets = getCallGraph().getPossibleTargetNumbers(caller, site);
179               if (targets != null && targets.contains(target.getGraphNodeId())) {
180                 // do nothing; we've previously discovered and handled this
181                 // receiver for this call site.
182                 return;
183               }
184     
185               // process the newly discovered target for this call
186               processResolvedCall(caller, site, target);
187     
188               if (!haveAlreadyVisited(target)) {
189                 markDiscovered(target);
190               }
191             }
192           };
193     
194           value.foreachExcluding(previousReceivers, action);
195     
196           // update the set of receivers previously considered
197           previousReceivers.copySet(value);
198     
199           return NOT_CHANGED;
200         }
201     
202         @Override
203         public String toString() {
204           return "Dispatch";
205         }
206     
207         @Override
208         public int hashCode() {
209           return caller.hashCode() + 8707 * site.hashCode();
210         }
211     
212         @Override
213         public boolean equals(Object o) {
214           if (o instanceof DispatchOperator) {
215             DispatchOperator other = (DispatchOperator) o;
216             return caller.equals(other.caller) && site.equals(other.site);
217           } else {
218             return false;
219           }
220         }
221       }
