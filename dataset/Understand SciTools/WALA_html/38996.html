11     package com.ibm.wala.ipa.callgraph.propagation.cfa;
12     
13     import com.ibm.wala.classLoader.IField;
14     import com.ibm.wala.ipa.callgraph.CGNode;
15     import com.ibm.wala.ipa.callgraph.propagation.ArrayContentsKey;
16     import com.ibm.wala.ipa.callgraph.propagation.FilteredPointerKey;
17     import com.ibm.wala.ipa.callgraph.propagation.InstanceFieldKey;
18     import com.ibm.wala.ipa.callgraph.propagation.InstanceKey;
19     import com.ibm.wala.ipa.callgraph.propagation.LocalPointerKey;
20     import com.ibm.wala.ipa.callgraph.propagation.LocalPointerKeyWithFilter;
21     import com.ibm.wala.ipa.callgraph.propagation.PointerKey;
22     import com.ibm.wala.ipa.callgraph.propagation.PointerKeyFactory;
23     import com.ibm.wala.ipa.callgraph.propagation.ReturnValueKey;
24     import com.ibm.wala.ipa.callgraph.propagation.StaticFieldKey;
25     
26     /**
27      * Default implementation of {@link PointerKeyFactory}
28      */
29     public class DefaultPointerKeyFactory implements PointerKeyFactory {
30     
31       public DefaultPointerKeyFactory() {
32       }
33     
34       @Override
35       public PointerKey getPointerKeyForLocal(CGNode node, int valueNumber) {
36         if (valueNumber <= 0) {
37           throw new IllegalArgumentException("illegal value number: " + valueNumber + " in " + node);
38         }
39         return new LocalPointerKey(node, valueNumber);
40       }
41     
42       @Override
43       public FilteredPointerKey getFilteredPointerKeyForLocal(CGNode node, int valueNumber, FilteredPointerKey.TypeFilter filter) {
44         if (filter == null) {
45           throw new IllegalArgumentException("null filter");
46         }
47         assert valueNumber > 0 : "illegal value number: " + valueNumber + " in " + node;
48         // TODO: add type filters!
49         return new LocalPointerKeyWithFilter(node, valueNumber, filter);
50       }
51     
52       @Override
53       public PointerKey getPointerKeyForReturnValue(CGNode node) {
54         return new ReturnValueKey(node);
55       }
56     
57       @Override
58       public PointerKey getPointerKeyForExceptionalReturnValue(CGNode node) {
59         return new ExceptionReturnValueKey(node);
60       }
61     
62       @Override
63       public PointerKey getPointerKeyForStaticField(IField f) {
64         if (f == null) {
65           throw new IllegalArgumentException("null f");
66         }
67         return new StaticFieldKey(f);
68       }
69     
70       @Override
71       public PointerKey getPointerKeyForInstanceField(InstanceKey I, IField field) {
72         if (field == null) {
73           throw new IllegalArgumentException("field is null");
74         }
75         return new InstanceFieldKey(I, field);
76       }
77     
78       @Override
79       public PointerKey getPointerKeyForArrayContents(InstanceKey I) {
80         return new ArrayContentsKey(I);
81       }
82     }
