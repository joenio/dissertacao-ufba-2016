11     package com.ibm.wala.util.intset;
12     
13     /**
14      * a simple implementation of int vector that can be tuned to control space usage
15      */
16     public class TunedSimpleIntVector extends SimpleIntVector {
17     
18       private final int initialSize;
19     
20       private final float expansion;
21     
22       TunedSimpleIntVector(int defaultValue, int initialSize, float expansion) {
23         super(defaultValue, initialSize);
24         this.initialSize = initialSize;
25         this.expansion = expansion;
26       }
27     
28       @Override
29       float getGrowthFactor() {
30         return expansion;
31       }
32     
33       @Override
34       int getInitialSize() {
35         return initialSize;
36       }
37     
38     }
