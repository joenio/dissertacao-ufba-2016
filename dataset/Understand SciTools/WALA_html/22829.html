33     /**
34      * A call graph implementation where some edges are delegated to other
35      * call sites, since they are guaranteed to be the same.
36      */
37     public class DelegatingExplicitCallGraph extends ExplicitCallGraph {
38     
39       /**
40        * delegateR(x,y) means that for at least one site, node number y delegates to
41        * node number x.
42        */
43       private final IBinaryNaturalRelation delegateR = new BasicNaturalRelation();
44     
45       /**
46        * @param cha
47        * @param options
48        */
49       public DelegatingExplicitCallGraph(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache) {
50         super(cha, options, cache);
51       }
52     
53       /**
54        * In this implementation, super.targets is a mapping from call site ->
55        * Object, where Object is a
56        * <ul>
57        * A Mapping from call site -> Object, where Object is a
58        * <li>CGNode if we've discovered exactly one target for the site
59        * <li> or an IntSet of node numbers if we've discovered more than one target
60        * for the site.
61        * <li> a CallSite if we're delegating these edges to another node
62        * </ul>
63        */
64       public class DelegatingCGNode extends ExplicitNode {
65     
66         protected DelegatingCGNode(IMethod method, Context C) {
67           super(method, C);
68         }
69     
70         /*
71          * @see com.ibm.wala.ipa.callgraph.impl.ExplicitCallGraph.ExplicitNode#getAllTargetNumbers()
72          */
73         @Override
74         public MutableSharedBitVectorIntSet getAllTargetNumbers() {
75           MutableSharedBitVectorIntSet result = new MutableSharedBitVectorIntSet(super.getAllTargetNumbers());
76           for (Iterator it = targets.iterator(); it.hasNext();) {
77             Object n = it.next();
78             if (n instanceof CallSite) {
79               ExplicitNode delegate = (ExplicitNode) ((CallSite) n).getNode();
80               IntSet s = DelegatingExplicitCallGraph.this.getPossibleTargetNumbers(delegate, ((CallSite) n).getSite());
81               if (s != null) {
82                 result.addAll(s);
83               }
84             }
85           }
86           return result;
87         }
88     
89         /*
90          * @see com.ibm.wala.ipa.callgraph.CGNode#getPossibleTargets(com.ibm.wala.classLoader.CallSiteReference)
91          */
92         @Override
93         public Set<CGNode> getPossibleTargets(CallSiteReference site) {
94           Object result = targets.get(site.getProgramCounter());
95           if (result != null && result instanceof CallSite) {
96             CallSite p = (CallSite) result;
97             CGNode n = p.getNode();
98             CallSiteReference s = p.getSite();
99             return DelegatingExplicitCallGraph.this.getPossibleTargets(n, s);
100           } else {
101             return super.getPossibleTargets(site);
102           }
103         }
104     
105         @Override
106         public IntSet getPossibleTargetNumbers(CallSiteReference site) {
107           Object t = targets.get(site.getProgramCounter());
108           if (t != null && t instanceof CallSite) {
109             CallSite p = (CallSite) t;
110             DelegatingCGNode n = (DelegatingCGNode) p.getNode();
111             CallSiteReference s = p.getSite();
112             return n.getPossibleTargetNumbers(s);
113           } else {
114             return super.getPossibleTargetNumbers(site);
115           }
116         }
117     
118         private boolean hasTarget(int y) {
119           if (super.getAllTargetNumbers().contains(y)) {
120             return true;
121           } else {
122             for (Iterator it = targets.iterator(); it.hasNext();) {
123               Object n = it.next();
124               if (n instanceof CallSite) {
125                 ExplicitNode delegate = (ExplicitNode) ((CallSite) n).getNode();
126                 IntSet s = DelegatingExplicitCallGraph.this.getPossibleTargetNumbers(delegate,((CallSite) n).getSite());
127                 if (s!= null && s.contains(y)) {
128                   return true;
129                 }
130               }
131             }
132           }
133           return false;
134         }
135     
136         @Override
137         public int getNumberOfTargets(CallSiteReference site) {
138           Object result = targets.get(site.getProgramCounter());
139           if (result != null && result instanceof CallSite) {
140             CallSite p = (CallSite) result;
141             CGNode n = p.getNode();
142             CallSiteReference s = p.getSite();
143             return DelegatingExplicitCallGraph.this.getNumberOfTargets(n,s);
144           } else {
145             return super.getNumberOfTargets(site);
146           }
147         }
148     
149         public void delegate(CallSiteReference site, CGNode delegateNode, CallSiteReference delegateSite) {
150           CallSite d = new CallSite(delegateSite, delegateNode);
151           targets.set(site.getProgramCounter(), d);
152           int y = getCallGraph().getNumber(this);
153           int x = getCallGraph().getNumber(delegateNode);
154           delegateR.add(x, y);
155         }
156       }
157     
158       /*
159        * @see com.ibm.wala.ipa.callgraph.impl.ExplicitCallGraph#makeNode(com.ibm.wala.classLoader.IMethod,
160        *      com.ibm.wala.ipa.callgraph.Context)
161        */
162       @Override
163       protected ExplicitNode makeNode(IMethod method, Context context) {
164         return new DelegatingCGNode(method, context);
165       }
166     
167       private class DelegatingEdgeManager extends ExplicitEdgeManager {
168     
169         /*
170          * @see com.ibm.wala.ipa.callgraph.impl.ExplicitCallGraph.ExplicitEdgeManager#addEdge(java.lang.Object,
171          *      java.lang.Object)
172          */
173         @Override
174         public void addEdge(CGNode src, CGNode dst) {
175           // we assume that this is called from ExplicitNode.addTarget().
176           // so we only have to track the inverse edge.
177           // this structure is pretty fragile .. we only explicitly represent the
178           // edges when NOT delegating.
179           // see getPredNodeNumbers() below which recovers.
180           super.addEdge(src, dst);
181         }
182     
183         /*
184          * @see com.ibm.wala.ipa.callgraph.impl.ExplicitCallGraph.ExplicitEdgeManager#removeAllIncidentEdges(java.lang.Object)
185          */
186         @Override
187         public void removeAllIncidentEdges(CGNode node) {
188           Assertions.UNREACHABLE();
189         }
190     
191         /*
192          * @see com.ibm.wala.ipa.callgraph.impl.ExplicitCallGraph.ExplicitEdgeManager#removeIncomingEdges(java.lang.Object)
193          */
194         @Override
195         public void removeIncomingEdges(CGNode node) {
196           Assertions.UNREACHABLE();
197         }
198     
199         /*
200          * @see com.ibm.wala.ipa.callgraph.impl.ExplicitCallGraph.ExplicitEdgeManager#removeOutgoingEdges(java.lang.Object)
201          */
202         @Override
203         public void removeOutgoingEdges(CGNode node) {
204           Assertions.UNREACHABLE();
205         }
206     
207         /*
208          * @see com.ibm.wala.ipa.callgraph.impl.ExplicitCallGraph.ExplicitEdgeManager#hasEdge(java.lang.Object,
209          *      java.lang.Object)
210          */
211         @Override
212         public boolean hasEdge(CGNode src, CGNode dst) {
213           if (super.hasEdge(src, dst)) {
214             return true;
215           } else {
216             DelegatingCGNode s = (DelegatingCGNode) src;
217             int y = getNumber(dst);
218             return s.hasTarget(y);
219           }
220         }
221     
222         /*
223          * @see com.ibm.wala.ipa.callgraph.impl.ExplicitCallGraph.ExplicitEdgeManager#getPredNodeCount(java.lang.Object)
224          */
225         @Override
226         public int getPredNodeCount(CGNode N) {
227           IntSet s = getPredNodeNumbers(N);
228           return s == null ? 0 : s.size();
229         }
230     
231         /*
232          * @see com.ibm.wala.ipa.callgraph.impl.ExplicitCallGraph.ExplicitEdgeManager#getPredNodeNumbers(java.lang.Object)
233          */
234         @Override
235         public IntSet getPredNodeNumbers(CGNode node) {
236           IntSet superR = super.getPredNodeNumbers(node);
237           if (superR == null) {
238             return null;
239           } else {
240             MutableSparseIntSet result = MutableSparseIntSet.make(superR);
241             BitVectorIntSet allPossiblePreds = new BitVectorIntSet(superR);
242             for (IntIterator it = superR.intIterator(); it.hasNext();) {
243               int x = it.next();
244               IntSet ySet = delegateR.getRelated(x);
245               if (ySet != null) {
246                 allPossiblePreds.addAll(ySet);
247               }
248             }
249             for (IntIterator it = allPossiblePreds.intIterator(); it.hasNext();) {
250               int y = it.next();
251               DelegatingCGNode yNode = (DelegatingCGNode) getNode(y);
252               if (hasEdge(yNode, node)) {
253                 result.add(y);
254               }
255             }
256             return result;
257           }
258         }
259       }
260     
261       /*
262        * @see com.ibm.wala.ipa.callgraph.impl.ExplicitCallGraph#makeEdgeManger()
263        */
264       @Override
265       protected ExplicitEdgeManager makeEdgeManger() {
266         return new DelegatingEdgeManager();
267       };
