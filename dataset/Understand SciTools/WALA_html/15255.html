14     package com.ibm.wala.cast.java.translator;
15     
16     import java.io.PrintWriter;
17     
18     import com.ibm.wala.cast.java.loader.JavaSourceLoaderImpl;
19     import com.ibm.wala.cast.tree.CAst;
20     import com.ibm.wala.cast.tree.CAstEntity;
21     import com.ibm.wala.cast.tree.impl.CAstImpl;
22     import com.ibm.wala.cast.tree.rewrite.CAstRewriter;
23     import com.ibm.wala.cast.tree.rewrite.CAstRewriterFactory;
24     import com.ibm.wala.cast.util.CAstPrinter;
25     import com.ibm.wala.classLoader.ModuleEntry;
26     
27     public class Java2IRTranslator {
28       private final boolean DEBUG;
29     
30       protected final JavaSourceLoaderImpl fLoader;
31     
32       CAstRewriterFactory castRewriterFactory = null;
33     
34       public Java2IRTranslator(JavaSourceLoaderImpl srcLoader) {
35         this(srcLoader, false);
36       }
37     
38       public Java2IRTranslator(JavaSourceLoaderImpl srcLoader, boolean debug) {
39         this(srcLoader, null, debug);
40       }
41     
42       public Java2IRTranslator(JavaSourceLoaderImpl srcLoader,
43           CAstRewriterFactory castRewriterFactory) {
44         this(srcLoader, castRewriterFactory, false);
45       }
46     
47       public Java2IRTranslator(JavaSourceLoaderImpl srcLoader,
48           CAstRewriterFactory castRewriterFactory, boolean debug) {
49         DEBUG = debug;
50         fLoader = srcLoader;
51         this.castRewriterFactory = castRewriterFactory;
52       }
53     
54       public void translate(ModuleEntry module, CAstEntity ce) {
55         if (DEBUG) {
56           PrintWriter printWriter = new PrintWriter(System.out);
57           CAstPrinter.printTo(ce, printWriter);
58           printWriter.flush();
59         }
60     
61         if (castRewriterFactory != null) {
62           CAst cast = new CAstImpl();
63           CAstRewriter rw = castRewriterFactory.createCAstRewriter(cast);
64           ce = rw.rewrite(ce);
65           if (DEBUG) {
66             PrintWriter printWriter = new PrintWriter(System.out);
67             CAstPrinter.printTo(ce, printWriter);
68             printWriter.flush();
69           }
70         }
71     
72         new JavaCAst2IRTranslator(module, ce, fLoader).translate();
73       }
74     }
