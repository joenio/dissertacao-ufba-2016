77       public static Set<ModuleEntry> getJavaScriptCodeFromProject(String project) throws IOException, CoreException {
78         IJavaScriptProject p = JavaScriptHeadlessUtil.getJavaScriptProjectFromWorkspace(project);
79         JSCallGraphUtil.setTranslatorFactory(new CAstRhinoTranslatorFactory());
80         AnalysisScope s = JavaScriptEclipseProjectPath.make(p, Collections.<Pair<String,Plugin>>emptySet()).toAnalysisScope(new CAstAnalysisScope(JSCallGraphUtil.makeLoaders(), Collections.singleton(JavaScriptLoader.JS)));
81     
82         List<Module> modules = s.getModules(JavaScriptTypes.jsLoader);
83         Set<ModuleEntry> mes = HashSetFactory.make();
84         for(Module m : modules) {
85           for(Iterator<? extends ModuleEntry> mm = m.getEntries(); mm.hasNext(); ) {
86             mes.add(mm.next());
87           }
88         }
89         return mes;
90       }

92       public static CGInfo buildJSDTCallGraph(Set<ModuleEntry> mes) {
93         final CGInfo info = new CGInfo();
94         HeadlessUtil.parseModules(mes, new EclipseCompiler<IJavaScriptUnit, JavaScriptUnit>() {
95           @Override
96           public IJavaScriptUnit getCompilationUnit(IFile file) {
97             return JavaScriptCore.createCompilationUnitFrom(file);
98           }
99           @Override
100           public Parser<IJavaScriptUnit, JavaScriptUnit> getParser() {
101             return new Parser<IJavaScriptUnit, JavaScriptUnit>() {
102               IJavaScriptProject project;
103     
104               @Override
105               public void setProject(IProject project) {
106                 this.project = JavaScriptCore.create(project);
107               }
108               
109               @Override
110               public void processASTs(Map<IJavaScriptUnit, EclipseSourceFileModule> files, Function<Object[], Boolean> errors) {
111                 final ASTVisitor visitor = new ASTVisitor() {
112                   private final CallHierarchy ch = CallHierarchy.getDefault();
113                   @Override
114                   public boolean visit(FunctionDeclaration node) {
115                     try {
116                     if (node.resolveBinding() != null) {
117                       IJavaScriptElement elt = node.resolveBinding().getJavaElement();
118                       if (elt instanceof IFunction) 
119                         try {
120                           MethodWrapper mw = ch.getCallerRoot((IFunction) elt);
121                           MethodWrapper calls[] = mw.getCalls(new NullProgressMonitor());
122                           if (calls != null && calls.length > 0) {
123                             System.err.println("calls: for " + elt);
124                             for(MethodWrapper call : calls) {
125                               System.err.println("calls: " + call.getMember());
126                               if (! info.cg.containsNode((IFunction)elt)) {
127                                 info.cg.addNode((IFunction)elt);
128                               }
129                               if (! info.cg.containsNode(call.getMember())) {
130                                 info.cg.addNode(call.getMember());
131                               }
132                               info.cg.addEdge(call.getMember(), (IFunction)elt);
133                             }
134                           }
135                         } catch (Throwable e) {
136                           // Eclipse does whatever it wants, and we ignore stuff :)
137                         }
138                       }
139                     } catch (RuntimeException e) {
140                       
141                     }
142                     // TODO Auto-generated method stub
143                     return super.visit(node);
144                   }
145     
146                   @Override
147                   public boolean visit(FunctionRef node) {
148                     System.err.println(node.resolveBinding().getJavaElement());
149                     // TODO Auto-generated method stub
150                     return super.visit(node);
151                   }
152     
153                   @Override
154                   public boolean visit(FunctionInvocation node) {
155                     info.calls.add(node);
156                     return super.visit(node);
157                   }
158                   
159                 };
160            
161                 if (useCreateASTs) {
162                   ASTParser parser = ASTParser.newParser(AST.JLS3);
163                   parser.setProject(project);
164                   parser.setResolveBindings(true);
165                   parser.createASTs(files.keySet().toArray(new IJavaScriptUnit[files.size()]), new String[0], new ASTRequestor() {
166                     @Override
167                     public void acceptAST(IJavaScriptUnit source, JavaScriptUnit ast) {
168                       ast.accept(visitor);
169                     }
170                   }, null);
171                 } else {
172                   for (Map.Entry<IJavaScriptUnit, EclipseSourceFileModule> f : files.entrySet()) {
173                     ASTParser parser = ASTParser.newParser(AST.JLS3);
174                     parser.setProject(project);
175                     parser.setResolveBindings(true);
176                     parser.setSource(f.getKey());
177                     try {
178                       ASTNode ast = parser.createAST(null);
179                       ast.accept(visitor);
180                     } catch (Throwable e) {
181                       System.err.println("trouble with " + f.getValue() + ": " + e.getMessage());
182                     }
183                   }
184                 }
185               }
186             };
187           }
188         });
189         return info;
190       }
