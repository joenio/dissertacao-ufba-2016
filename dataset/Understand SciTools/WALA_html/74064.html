36     /**
37      * Optimistic call graph builder that propagates inter-procedural data flow iteratively as
38      * call edges are discovered. Slower, but potentially more sound than {@link PessimisticCallGraphBuilder}.
39      * 
40      * @author mschaefer
41      *
42      */
43     public class OptimisticCallgraphBuilder extends FieldBasedCallGraphBuilder {
44     	/** The maximum number of iterations to perform. */
45     	public int ITERATION_CUTOFF = Integer.MAX_VALUE;
46     	
47     	private final boolean handleCallApply;
48     	
49     	public OptimisticCallgraphBuilder(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache, boolean supportFullPointerAnalysis) {
50     		super(cha, options, cache, supportFullPointerAnalysis);
51     		handleCallApply = options instanceof JSAnalysisOptions && ((JSAnalysisOptions)options).handleCallApply();
52     	}
53     
54     	@Override
55     	public FlowGraph buildFlowGraph(IProgressMonitor monitor) throws CancelException {
56     	   FlowGraph flowgraph = flowGraphFactory();
57     		
58     		// keep track of which call edges we already know about
59     		Set<Pair<CallVertex, FuncVertex>> knownEdges = HashSetFactory.make();
60     		
61     		// flag for fixpoint iteration
62     		boolean changed = true;
63     		int iter = 0;
64     		
65     		while(iter++ < ITERATION_CUTOFF && changed) {
66     			MonitorUtil.throwExceptionIfCanceled(monitor);
67     			changed = false;
68     			
69     			// extract all call edges from the flow graph
70     			Set<Pair<CallVertex, FuncVertex>> newEdges = this.extractCallGraphEdges(flowgraph, monitor);
71     			
72     			for(Pair<CallVertex, FuncVertex> edge : newEdges) {
73     				MonitorUtil.throwExceptionIfCanceled(monitor);
74     				
75     				// set changed to true if this is a new edge
76     				boolean newEdge = knownEdges.add(edge);
77     				changed = changed || newEdge;
78     				
79     				if(newEdge) {
80     					// handle it
81     					addEdge(flowgraph, edge.fst, edge.snd, monitor);
82     				
83     					// special handling of invocations of Function.prototype.call
84     					// TODO: since we've just added some edges to the flow graph, its transitive closure will be
85     					//       recomputed here, which is slow and unnecessary
86     					if(handleCallApply && 
87     					    (edge.snd.getFullName().equals("Lprologue.js/Function_prototype_call") ||
88     					     edge.snd.getFullName().equals("Lprologue.js/Function_prototype_apply"))) {
89     						addReflectiveCallEdge(flowgraph, edge.fst, monitor);
90     					}
91     				}
92     			}
93     		}
94     		
95     		return flowgraph;
96     	}
97     
98     	// add flow corresponding to a new call edge
99     	private void addEdge(FlowGraph flowgraph, CallVertex c, FuncVertex callee, IProgressMonitor monitor) throws CancelException {
100     	  VertexFactory factory = flowgraph.getVertexFactory();
101     	  JavaScriptInvoke invk = c.getInstruction();
102     	  FuncVertex caller = c.getCaller();
103     
104     	  int offset = 0;
105         if (invk.getDeclaredTarget().getSelector().equals(JavaScriptMethods.ctorReference.getSelector())) {
106           offset = 1;
107         }
108         
109         for(int i=0;i<invk.getNumberOfParameters();++i) {
110           // only flow receiver into 'this' if invk is, in fact, a method call
111           flowgraph.addEdge(factory.makeVarVertex(caller, invk.getUse(i)), factory.makeArgVertex(callee));
112           //if(i != 1 || !invk.getDeclaredTarget().getSelector().equals(AstMethodReference.fnSelector))
113             flowgraph.addEdge(factory.makeVarVertex(caller, invk.getUse(i)), factory.makeParamVertex(callee, i+offset));
114         }
115     
116     	  // flow from return vertex to result vertex
117     	  flowgraph.addEdge(factory.makeRetVertex(callee), factory.makeVarVertex(caller, invk.getDef()));			
118     	}
119     	
120     	// add data flow corresponding to a reflective invocation via Function.prototype.call
121     	// NB: for f.call(...), f will _not_ appear as a call target, but the appropriate argument and return data flow will be set up
122     	private void addReflectiveCallEdge(FlowGraph flowgraph, CallVertex c, IProgressMonitor monitor) throws CancelException {
123     	  VertexFactory factory = flowgraph.getVertexFactory();
124     	  FuncVertex caller = c.getCaller();
125     	  JavaScriptInvoke invk = c.getInstruction();
126     
127     	  VarVertex receiverVertex = factory.makeVarVertex(caller, invk.getUse(1));
128     	  OrdinalSet<FuncVertex> realCallees = flowgraph.getReachingSet(receiverVertex, monitor);
129     	  System.err.println("callees " + realCallees + " for " + caller);
130     	  for(FuncVertex realCallee: realCallees) {
131     	    // flow from arguments to parameters
132     	    for(int i=2;i<invk.getNumberOfParameters();++i)
133     	      flowgraph.addEdge(factory.makeVarVertex(caller, invk.getUse(i)), factory.makeParamVertex(realCallee, i-1));
134     
135     	    // flow from return vertex to result vertex
136     	    flowgraph.addEdge(factory.makeRetVertex(realCallee), factory.makeVarVertex(caller, invk.getDef()));
137     	  }
138     	}
