14

22     
23     public class AstLoopUnwinder
24       extends CAstRewriter<CAstRewriter.RewriteContext<AstLoopUnwinder.UnwindKey>,AstLoopUnwinder.UnwindKey>
25     {
26     
27       public static class UnwindKey implements CAstRewriter.CopyKey<UnwindKey> {
28         private int iteration;
29         private UnwindKey rest;
30     			  
31         private UnwindKey(int iteration, UnwindKey rest) {
32           this.rest = rest;
33           this.iteration = iteration;
34         }
35     			  
36         @Override
37         public int hashCode() {
38           return iteration * (rest == null? 1: rest.hashCode());
39         }
40     		
41         @Override
42         public UnwindKey parent() {
43           return rest;
44         }
45     
46         @Override
47         public boolean equals(Object o) {
48           return (o instanceof UnwindKey) &&
49     	((UnwindKey)o).iteration == iteration &&
50     	( rest==null? ((UnwindKey)o).rest == null: rest.equals(((UnwindKey)o).rest) );
51         }
52     
53         @Override
54         public String toString() {
55           return "#"+iteration+ ((rest==null)?"":rest.toString());
56         }
57       }
58     
59       // private static final boolean DEBUG = false;
60     
61       private final int unwindFactor;
62     	
63       public AstLoopUnwinder(CAst Ast, boolean recursive) {
64         this(Ast, recursive, 3);
65       }
66     	  
67       public AstLoopUnwinder(CAst Ast, boolean recursive, int unwindFactor) {
68         super(Ast, recursive, new RootContext());
69         this.unwindFactor = unwindFactor;
70       }
71     	  
72       public CAstEntity translate(CAstEntity original) {
73         return rewrite(original);
74       }
75     
76       private static class RootContext implements RewriteContext<UnwindKey> {
77     		  
78         @Override
79         public UnwindKey key() {
80           return null;
81         }
82       }
83     	  	  
84       private class LoopContext implements RewriteContext<UnwindKey> {
85         private final CAstRewriter.RewriteContext<UnwindKey> parent;
86         private final int iteration;
87             
88         private LoopContext(int iteration, RewriteContext<UnwindKey> parent) {
89           this.iteration = iteration;
90           this.parent = parent;
91         }
92             
93         @Override
94         public UnwindKey key() {
95           return new UnwindKey(iteration, parent.key());
96         }
97           
98       }
99     	  
100       @Override
101       protected CAstNode flowOutTo(Map nodeMap, 
102     			       CAstNode oldSource,
103     			       Object label,
104     			       CAstNode oldTarget,
105     			       CAstControlFlowMap orig, 
106     			       CAstSourcePositionMap src) 
107       {
108         assert oldTarget == CAstControlFlowMap.EXCEPTION_TO_EXIT;
109         return oldTarget;
110       }
111     
112       @Override
113       protected CAstNode copyNodes(CAstNode n, final CAstControlFlowMap cfg, RewriteContext<UnwindKey> c, Map<Pair<CAstNode,UnwindKey>,CAstNode> nodeMap) {
114         if (n instanceof CAstOperator) {
115           return n;
116         } else if (n.getValue() != null) {
117           return Ast.makeConstant( n.getValue() );
118         } else if (n.getKind() == CAstNode.LOOP) {
119           CAstNode test = n.getChild(0);
120           CAstNode body = n.getChild(1);
121           
122           int count = unwindFactor;
123           RewriteContext<UnwindKey> lc = new LoopContext(count, c);
124           CAstNode code = 
125             Ast.makeNode(CAstNode.ASSERT,
126     	  Ast.makeNode(CAstNode.UNARY_EXPR,
127     	    CAstOperator.OP_NOT,
128                 copyNodes(test, cfg, lc, nodeMap)),
129               Ast.makeConstant(false));
130           while (count-- > 0) {
131     	lc = new LoopContext(count, c);
132     	code = Ast.makeNode(CAstNode.IF_STMT, 
133     			    copyNodes(test, cfg, lc, nodeMap),
134     			    Ast.makeNode(CAstNode.BLOCK_STMT,
135     					 copyNodes(body, cfg, lc, nodeMap),
136     					 code));
137           }
138     			  
139           return code;
140         } else {
141     	CAstNode[] newChildren = new CAstNode[ n.getChildCount() ];
142     	for(int i = 0; i < newChildren.length; i++) 
143     	  newChildren[i] = copyNodes(n.getChild(i), cfg, c, nodeMap);
144     			  
145     	CAstNode newN = Ast.makeNode(n.getKind(), newChildren);
146     			  			
147     	nodeMap.put(Pair.make(n, c.key()), newN);
148     	
149     	return newN; 
150         }
151       }	  
152     }
