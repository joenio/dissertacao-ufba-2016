18     package com.ibm.wala.sourcepos;
19     
20     import java.io.DataInputStream;
21     import java.io.IOException;
22     import java.util.LinkedList;
23     
24     /**
25      * This class represents the CharacterRangeTable attribute.
26      * 
27      * @see CRTData
28      * @see CRTFlags
29      * @author Siegfried Weber
30      * @author Juergen Graf <juergen.graf@gmail.com>
31      */
32     public final class CRTable extends PositionsAttribute {
33     
34       /** Stores the attribute name of this attribute */
35       public static final String ATTRIBUTE_NAME = "CharacterRangeTable";
36     
37       private static final String WARN_CRT_ENTRIES_CONTRADICTORY = "CRT entries %1$s and %2$s are contradictory.";
38     
39       private static final String ERR_NO_CRT_ENTRY = "No CRT entry found for program counter %1$s.";
40     
41       /** Stores the CharacterRangeTable data */
42       private CRTData[] crt;
43     
44       /**
45        * Creates a new instance of CRTable.
46        * 
47        * @param data
48        *          the byte array containing the attribute
49        * @throws IOException
50        *           An IOException is thrown if the attribute can't be read.
51        */
52       public CRTable(byte[] data) throws IOException {
53         super(data);
54         if (Debug.PRINT_CHARACTER_RANGE_TABLE) {
55           Debug.info(this.toString());
56         }
57       }
58     
59       @Override
60       protected final void readData(DataInputStream in) throws IOException {
61         assert in != null;
62         short crt_length = in.readShort();
63         crt = new CRTData[crt_length];
64         for (int i = 0; i < crt_length; i++) {
65           short pc_start_index = in.readShort();
66           short pc_end_index = in.readShort();
67           int source_start_position = in.readInt();
68           int source_end_position = in.readInt();
69           short flags = in.readShort();
70           try {
71             crt[i] = new CRTData(pc_start_index, pc_end_index, source_start_position, source_end_position, flags);
72           } catch (InvalidCRTDataException e) {
73             LinkedList<Object> l = e.getData();
74             if (l == null)
75               l = new LinkedList<Object>();
76             l.addFirst(i);
77             Debug.warn(e.getMessage(), l.toArray());
78           }
79         }
80       }
81     
82       /**
83        * Returns the source positions for the given index in the code array of the
84        * code attribute.
85        * 
86        * @param pc
87        *          the index in the code array of the code attribute
88        * @return the most precise source position range
89        */
90       public final Range getSourceInfo(int pc) {
91         CRTData sourceInfo = null;
92         int sourceInfoIndex = 0;
93         for (int i = 0; i < crt.length; i++) {
94           if ((crt[i] != null) && (crt[i].isInRange(pc))) {
95             if ((sourceInfo != null) && !sourceInfo.matches(crt[i]))
96               Debug.warn(WARN_CRT_ENTRIES_CONTRADICTORY, new Object[] { sourceInfoIndex, i });
97             if ((sourceInfo == null) || crt[i].isMorePrecise(sourceInfo)) {
98               sourceInfo = crt[i];
99               sourceInfoIndex = i;
100             }
101           }
102         }
103         if (sourceInfo == null) {
104           Debug.error(ERR_NO_CRT_ENTRY, pc);
105           try {
106             short short_pc = (short) (pc & 0xFFFF);
107             sourceInfo = new CRTData(short_pc, short_pc, 0, 0, CRTFlags.CRT_SOURCE_INFO);
108           } catch (InvalidCRTDataException e) {
109             assert false;
110           }
111         }
112         return sourceInfo.getSourceInfo();
113       }
114     
115       @Override
116       public String toString() {
117         if (crt == null) {
118           return "<undefined>";
119         }
120     
121         StringBuilder sb = new StringBuilder();
122     
123         for (int i = 0; i < crt.length; i++) {
124           sb.append(i + " -> ");
125           sb.append(crt[i] == null ? "<null>" : crt[i]);
126           sb.append("\n");
127         }
128     
129         return sb.toString();
130       }
131     
132     }
