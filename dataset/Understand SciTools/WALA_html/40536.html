11     package com.ibm.wala.cast.ir.ssa;
12     
13     import com.ibm.wala.classLoader.CallSiteReference;
14     import com.ibm.wala.ssa.SSAInstruction;
15     import com.ibm.wala.ssa.SSAInstructionFactory;
16     
17     /**
18      *  This abstract instruction extends the abstract invoke with
19      * functionality to support invocations with a fixed number of
20      * arguments---the only case in some languages and a common case even
21      * in scripting languages.
22      *
23      * 
24      * @author Julian Dolby (dolby@us.ibm.com)
25      */
26     public abstract class FixedParametersInvokeInstruction
27         extends MultiReturnValueInvokeInstruction 
28     {
29     
30       /**
31        * The value numbers of the arguments passed to the call.  For non-static methods,
32        * params[0] == this.  If params == null, this should be a static method with
33        * no parameters.
34        */
35       private final int[] params;
36     
37       public FixedParametersInvokeInstruction(int iindex, int results[], int[] params, int exception, CallSiteReference site) {
38         super(iindex, results, exception, site);
39         this.params = params;
40       }
41     
42       public FixedParametersInvokeInstruction(int iindex, int result, int[] params, int exception, CallSiteReference site) {
43         this(iindex, new int[]{result}, params, exception, site);
44       }
45     
46       /**
47        * Constructor InvokeInstruction. This case for void return values
48        * @param params
49        * @param exception
50        * @param site
51        */
52       public FixedParametersInvokeInstruction(int iindex, int[] params, int exception, CallSiteReference site) {
53         this(iindex, null, params, exception, site);
54       }
55     
56       protected abstract SSAInstruction copyInstruction(SSAInstructionFactory insts, int result[], int[] params, int exception);
57     
58       @Override
59       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
60         int newParams[] = params;
61     
62         if (uses != null) {
63           int i = 0;
64     
65           newParams = new int[params.length];
66           for (int j = 0; j < newParams.length; j++)
67             newParams[j] = uses[i++];
68         }
69     
70         int newLvals[] = null;
71         if (getNumberOfReturnValues() > 0) {
72           newLvals = new int[results.length];
73           System.arraycopy(results, 0, newLvals, 0, results.length);
74         }
75         int newExp = exception;
76     
77         if (defs != null) {
78           int i = 0;
79           if (getNumberOfReturnValues() > 0) {
80             newLvals[0] = defs[i++];
81           }
82           newExp = defs[i++];
83           for (int j = 1; j < getNumberOfReturnValues(); j++) {
84             newLvals[j] = defs[i++];
85           }
86         }
87     
88         return copyInstruction(insts, newLvals, newParams, newExp);
89       }
90     
91       @Override
92       public int getNumberOfParameters() {
93         if (params == null) {
94           return 0;
95         } else {
96           return params.length;
97         }
98       }
99     
100       
101       @Override
102       public void visit(IVisitor v) {
103         // TODO Auto-generated method stub
104         assert false;
105         
106       }
107     
108       @Override
109       public int getNumberOfUses() {
110         return getNumberOfParameters();
111       }
112     
113       @Override
114       public int hashCode() {
115         // TODO Auto-generated method stub
116         assert false;
117         return 0;
118       }
119     
120       @Override
121       public int getUse(int j) {
122         if (j < getNumberOfParameters())
123           return params[j];
124         else {
125           return super.getUse(j);
126         }
127       }
128     
129     }
