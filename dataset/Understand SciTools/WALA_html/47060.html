56     /**
57      * Class for building intra-procedural flow graphs for a given class hierarchy.
58      * 
59      * @author mschaefer
60      */
61     public class FlowGraphBuilder {
62     	private final IClassHierarchy cha;
63     	private final AnalysisCache cache;
64     	private final boolean supportFullPointerAnalysis;
65     	
66     	public FlowGraphBuilder(IClassHierarchy cha, AnalysisCache cache, boolean supportPointerAnalysis) {
67     		this.cha = cha;
68     		this.cache = cache;
69     		this.supportFullPointerAnalysis = supportPointerAnalysis;
70     	}
71     	
72     	/**
73     	 * This is the main entry point of the flow graph builder.
74     	 * 
75     	 * <p>
76     	 * It creates a new, empty flow graph, adds nodes for a small number of special primitive
77     	 * functions such as <code>Object</code> and <code>Function</code> and sets up flow
78     	 * edges to make them flow into the corresponding global variables. Then it iterates over
79     	 * all functions in the class hierarchy and all their IR instructions, and adds the
80     	 * flow edges induced by these instructions.
81     	 * </p>
82     	 * 
83     	 * @return the completed flow graph
84     	 */
85     	public FlowGraph buildFlowGraph() {
86     		FlowGraph flowgraph = new FlowGraph();
87     		
88     		addPrimitives(flowgraph);
89     		
90     		visitProgram(flowgraph);
91     					
92     		return flowgraph;
93     	}
94     
95       public void visitProgram(FlowGraph flowgraph) {
96         for(IClass klass : cha) {
97     			for(IMethod method : klass.getDeclaredMethods()) {
98     				if(method.getDescriptor().equals(AstMethodReference.fnDesc)) {
99     		      visitFunction(flowgraph, method);
100     				}
101     			}
102     		}
103       }
104     
105       public void visitFunction(FlowGraph flowgraph, IMethod method) {
106         {
107           if (method.toString().contains("ctor") && method.toString().contains("dollar_init")) {
108             System.err.println("found it");
109           }
110         	IR ir = cache.getIR(method);
111         	FlowGraphSSAVisitor visitor = new FlowGraphSSAVisitor(ir, flowgraph);
112     
113         	// first visit normal instructions
114         	SSAInstruction[] normalInstructions = ir.getInstructions();
115         	for(int i=0;i<normalInstructions.length;++i)
116         		if(normalInstructions[i] != null) {
117         			visitor.instructionIndex  = i;
118         			normalInstructions[i].visit(visitor);
119         		}
120         	
121         	// now visit phis and catches
122         	visitor.instructionIndex = -1;
123         	for(Iterator<? extends SSAInstruction> iter=ir.iteratePhis();iter.hasNext();)
124         		iter.next().visit(visitor);
125         	
126         	for(Iterator<SSAInstruction> iter=ir.iterateCatchInstructions();iter.hasNext();)
127         		iter.next().visit(visitor);
128         }
129       }

131     	// primitive functions that are treated specially
132     	private static String[] primitiveFunctions = { "Object", "Function", "Array", "StringObject", "NumberObject", "BooleanObject", "RegExp" };
133     	
134     	/**
135     	 * Add flows from the special primitive functions to the corresponding global variables.
136     	 * 
137     	 * @param flowgraph the flow graph under construction
138     	 */
139     	private void addPrimitives(FlowGraph flowgraph) {
140     		VertexFactory factory = flowgraph.getVertexFactory();
141     		for(String pf : primitiveFunctions) {
142     			TypeReference typeref = TypeReference.findOrCreate(JavaScriptTypes.jsLoader, "L" + pf);
143     			IClass klass = cha.lookupClass(typeref);
144     			String prop = pf.endsWith("Object")? pf.substring(0, pf.length() - 6): pf;
145     			flowgraph.addEdge(factory.makeFuncVertex(klass), factory.makePropVertex(prop));
146     		}
147     	}
148     	
149     	/**
150     	 * Visitor class that does the heavy lifting (such as it is) of flow graph construction, adding flow graph
151     	 * edges for every instruction in the method IR.
152     	 * 
153     	 * <p>
154     	 * The only slightly tricky thing are assignments to exposed variables inside their defining function. In
155     	 * the IR, they initially appear as normal SSA variable assignments, without any indication of their lexical
156     	 * nature. The normal call graph construction logic does something convoluted to fix this up later when
157     	 * an actual lexical access is encountered.
158     	 * </p>
159     	 * 
160     	 * <p>
161     	 * We use a much simpler approach. Whenever we see an assignment <code>v<sub>i</sub> = e</code>, we ask the
162     	 * enclosing function whether <code>v<sub>i</sub></code> is an exposed variable. If it is, we determine its
163     	 * source-level names <code>x<sub>1</sub>, x<sub>2</sub>, ..., x<sub>n</sub></code>, and then add edges
164     	 * corresponding to lexical writes of <code>v<sub>i</sub></code> into all the <code>x<sub>j</sub></code>.
165     	 * </p>
166     	 * 
167     	 * @author mschaefer
168     	 */
169     	private class FlowGraphSSAVisitor extends JSMethodInstructionVisitor {
170     		// index of the instruction currently visited; -1 if the instruction isn't a normal instruction
171     		public int instructionIndex = -1;
172     		
173     		// flow graph being built
174     		private final FlowGraph flowgraph;
175     		
176     		// vertex factory to use for constructing new vertices
177     		private final VertexFactory factory;
178     		
179     		// lexical information about the current function
180     		private final LexicalInformation lexicalInfo;
181     		
182     		// the set of SSA variables in the current function that are accessed by nested functions
183     		private final IntSet exposedVars;
184     		
185     		// the IR of the current function
186     		private final IR ir;
187     		
188     		// the function vertex corresponding to the current function
189     		private final FuncVertex func;
190     		
191     		public FlowGraphSSAVisitor(IR ir, FlowGraph flowgraph) {
192     			super(ir.getMethod(), ir.getSymbolTable(), cache.getDefUse(ir));
193     			this.ir = ir;
194     			this.flowgraph = flowgraph;
195     			this.factory = flowgraph.getVertexFactory();
196     			this.func = factory.makeFuncVertex(ir.getMethod().getDeclaringClass());
197     			if(method instanceof AstMethod) {
198     				this.lexicalInfo = ((AstMethod)method).lexicalInfo();
199     				this.exposedVars = lexicalInfo.getAllExposedUses();
200     			} else {
201     				this.lexicalInfo = null;
202     				this.exposedVars = EmptyIntSet.instance;
203     			}
204     		}
205     		
206     		// add extra flow from v_def to every lexical variable it may correspond to at source-level
207     		private void handleLexicalDef(int def) {
208     			assert def != -1;
209     			if(instructionIndex != -1 && exposedVars.contains(def)) {
210     				VarVertex v = factory.makeVarVertex(func, def);
211     				for(String localName : ir.getLocalNames(instructionIndex, def))
212     					flowgraph.addEdge(v, factory.makeLexicalAccessVertex(lexicalInfo.getScopingName(), localName));
213     			}
214     		}
215     		
216     		@Override
217     		public void visitPhi(SSAPhiInstruction phi) {
218     			int n = phi.getNumberOfUses();
219     			VarVertex w = factory.makeVarVertex(func, phi.getDef());
220     			for(int i=0;i<n;++i) {
221     				VarVertex v = factory.makeVarVertex(func, phi.getUse(i));
222     				flowgraph.addEdge(v, w);
223     			}
224     		}
225     		
226     		@Override
227     		public void visitPrototypeLookup(PrototypeLookup proto) {
228     			// treat it simply as an assignment
229     			flowgraph.addEdge(factory.makeVarVertex(func, proto.getUse(0)),
230     							          factory.makeVarVertex(func, proto.getDef()));
231     			handleLexicalDef(proto.getDef());
232     		}
233     	
234     		private void visitPut(int val, String propName) {
235           Vertex v = factory.makeVarVertex(func, val),
236               w = factory.makePropVertex(propName);
237           flowgraph.addEdge(v, w);		  
238     		}

240     		@Override
241     		public void visitPut(SSAPutInstruction put) {
242     			visitPut(put.getVal(), put.getDeclaredField().getName().toString());
243     		}
244     		
245     		@Override
246         public void visitSetPrototype(SetPrototype instruction) {
247     		  visitPut(instruction.getUse(1), "prototype");
248     		}
249     
250         @Override
251     		public void visitAstGlobalWrite(AstGlobalWrite instruction) {
252           String propName = instruction.getDeclaredField().getName().toString();
253     
254           // hack to account for global variables
255           assert propName.startsWith("global ");
256           propName = propName.substring("global ".length());
257     
258           visitPut(instruction.getVal(), propName);
259     		}
260     		
261     		@Override
262     		public void visitJavaScriptPropertyWrite(JavaScriptPropertyWrite pw) {
263     			int p = pw.getMemberRef();
264     			if(symtab.isConstant(p)) {
265     				String pn = JSCallGraphUtil.simulateToStringForPropertyNames(symtab.getConstantValue(p));
266     				
267     				Vertex v = factory.makeVarVertex(func, pw.getValue()),
268     				       w = factory.makePropVertex(pn);
269     				flowgraph.addEdge(v, w);
270     			}
271     		}
272     		
273     		@Override
274     		public void visitAstLexicalWrite(AstLexicalWrite lw) {
275     			for(Access acc : lw.getAccesses()) {
276     				Vertex v = factory.makeVarVertex(func, acc.valueNumber),
277     					   w = factory.makeLexicalAccessVertex(acc.variableDefiner, acc.variableName);
278     				flowgraph.addEdge(v, w);
279     			}
280     		}
281     		
282     		@Override
283     		public void visitGet(SSAGetInstruction get) {
284     			String propName = get.getDeclaredField().getName().toString();
285     			if(propName.startsWith("global "))
286     				propName = propName.substring("global ".length());
287     
288     			Vertex v = factory.makePropVertex(propName),
289     				   w = factory.makeVarVertex(func, get.getDef());
290     			flowgraph.addEdge(v, w);
291     			handleLexicalDef(get.getDef());
292     		}
293     		
294     		@Override
295     		public void visitAstGlobalRead(AstGlobalRead instruction) {
296     		  if (supportFullPointerAnalysis && instruction.getGlobalName().endsWith(JSSSAPropagationCallGraphBuilder.GLOBAL_OBJ_VAR_NAME)) {
297     		    Vertex lval = factory.makeVarVertex(func, instruction.getDef());
298     		    flowgraph.addEdge(factory.global(), lval);
299     		  } else {
300     		    visitGet(instruction);
301     		  }
302     		}
303     		
304     		@Override
305     		public void visitJavaScriptPropertyRead(JavaScriptPropertyRead pr) {
306     			int p = pr.getMemberRef();
307     			if(symtab.isConstant(p)) {
308     				String pn = JSCallGraphUtil.simulateToStringForPropertyNames(symtab.getConstantValue(p));
309     				Vertex v = factory.makePropVertex(pn),
310     				       w = factory.makeVarVertex(func, pr.getDef());
311     				flowgraph.addEdge(v, w);
312     			}
313     			
314     			IntSet argVns = Util.getArgumentsArrayVns(ir, du);
315     			if (argVns.contains(pr.getObjectRef())) {
316     			  Vertex v = factory.makeArgVertex(func),
317                 w = factory.makeVarVertex(func, pr.getDef());
318            flowgraph.addEdge(v, w);
319     			}
320     			
321     			handleLexicalDef(pr.getDef());
322     		}
323     		
324     		@Override
325     		public void visitAstLexicalRead(AstLexicalRead lr) {
326     			for(Access acc : lr.getAccesses()) {
327     				Vertex v = factory.makeLexicalAccessVertex(acc.variableDefiner, acc.variableName),
328     					   w = factory.makeVarVertex(func, acc.valueNumber);
329     				flowgraph.addEdge(v, w);
330     				handleLexicalDef(acc.valueNumber);
331     			}
332     		}
333     		
334     		@Override
335     		public void visitReturn(SSAReturnInstruction ret) {
336     			Vertex v = factory.makeVarVertex(func, ret.getResult()),
337     				   w = factory.makeRetVertex(func);
338     			flowgraph.addEdge(v, w);
339     		}
340     		
341     		@Override
342     		public void visitThrow(SSAThrowInstruction thr) {
343     			Vertex v = factory.makeVarVertex(func, thr.getException()),
344     				   w = factory.makeUnknownVertex();
345     			flowgraph.addEdge(v, w);
346     		}
347     		
348     		@Override
349     		public void visitGetCaughtException(SSAGetCaughtExceptionInstruction katch) {
350     			Vertex v = factory.makeUnknownVertex(),
351     				   w = factory.makeVarVertex(func, katch.getDef());
352     			flowgraph.addEdge(v, w);
353     		}

355     		@Override
356     		public void visitJavaScriptInvoke(JavaScriptInvoke invk) {
357     			flowgraph.addEdge(factory.makeUnknownVertex(),
358     							  factory.makeVarVertex(func, invk.getException()));
359     			
360     			// check whether this invoke corresponds to a function expression/declaration
361           // flow callee variable into callee vertex
362     			if(invk.getDeclaredTarget().equals(JavaScriptMethods.ctorReference)) {
363     
364     			  flowgraph.addEdge(factory.makeVarVertex(func, invk.getFunction()),
365     			      factory.makeCallVertex(func, invk));
366     			  
367     			  if(isFunctionConstructorInvoke(invk)) {
368     			    // second parameter is function name
369     			    String fn_name = symtab.getStringValue(invk.getUse(1));
370     
371     			    // find the function being defined here
372     			    IClass klass = cha.lookupClass(TypeReference.findOrCreate(JavaScriptTypes.jsLoader, fn_name));
373     			    if (klass == null) {
374     			      System.err.println("cannot find " + fn_name + " at " +  ((AstMethod)ir.getMethod()).getSourcePosition(ir.getCallInstructionIndices(invk.getCallSite()).intIterator().next()));
375     			      return;
376     			    }
377     			    
378     			    IMethod fn = klass.getMethod(AstMethodReference.fnSelector);
379     			    FuncVertex fnVertex = factory.makeFuncVertex(klass);
380     			      
381     			    // function flows into its own v1 variable 
382     			    flowgraph.addEdge(fnVertex, factory.makeVarVertex(fnVertex, 1));
383     
384     			    // flow parameters into local variables
385     			    for(int i=1;i<fn.getNumberOfParameters();++i)
386     			      flowgraph.addEdge(factory.makeParamVertex(fnVertex, i), factory.makeVarVertex(fnVertex, i+1));
387     
388     			    // flow function into result variable
389     			    flowgraph.addEdge(fnVertex, factory.makeVarVertex(func, invk.getDef()));
390     
391     			  } else if (supportFullPointerAnalysis) {
392     			    
393     			    CreationSiteVertex cs = factory.makeCreationSiteVertex(method, invk.iindex, JavaScriptTypes.Object);
394     			    
395     			    // flow creation site into result of new call
396     	        flowgraph.addEdge(cs, factory.makeVarVertex(func, invk.getDef())); 
397     	        
398     	        // also passed as 'this' to constructor
399     	        if (invk.getNumberOfParameters() > 1) {
400     	          flowgraph.addEdge(cs, factory.makeVarVertex(func, invk.getUse(0)));
401     	        }
402     			  }
403     
404     			} else {
405     				// check whether it is a method call
406     				if(invk.getDeclaredTarget().equals(JavaScriptMethods.dispatchReference)) {
407     					// we only handle method calls with constant names
408     		      if(symtab.isConstant(invk.getFunction())) {
409     		        String pn = JSCallGraphUtil.simulateToStringForPropertyNames(symtab.getConstantValue(invk.getFunction()));
410     						// flow callee property into callee vertex
411     						flowgraph.addEdge(factory.makePropVertex(pn), factory.makeCallVertex(func, invk));
412     		      }
413     				} else {
414     					// this case is simpler: just flow callee variable into callee vertex
415     					flowgraph.addEdge(factory.makeVarVertex(func, invk.getFunction()),
416     									  factory.makeCallVertex(func, invk));
417     				}
418     			}
419     			handleLexicalDef(invk.getDef());
420     		}
421     
422         @Override
423         public void visitNew(SSANewInstruction invk) {
424           if (supportFullPointerAnalysis) {
425             // special case for supporting full pointer analysis
426             // some core objects in the prologue (and the arguments array objects) get created with 'new'
427             CreationSiteVertex cs = factory.makeCreationSiteVertex(method, invk.iindex, invk.getConcreteType());
428           
429             // flow creation site into result of new call
430             flowgraph.addEdge(cs, factory.makeVarVertex(func, invk.getDef()));  
431           } 
432         }
433     	}
