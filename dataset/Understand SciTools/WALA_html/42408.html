43     /**
44      * A precise interprocedural tabulation solver.
45      * <p>
46      * See Reps, Horwitz, Sagiv POPL 95.
47      * <p>
48      * This version differs in some ways from the POPL algorithm. In particular ...
49      * <ul>
50      * <li>to support exceptional control flow ... there may be several return sites for each call site.
51      * <li>it supports an optional merge operator, useful for non-IFDS problems and widening.
52      * <li>it stores summary edges at each callee instead of at each call site.
53      * </ul>
54      * <p>
55      * 
56      * @param <T> type of node in the supergraph
57      * @param <P> type of a procedure (like a box in an RSM)
58      * @param <F> type of factoids propagated when solving this problem
59      */
60     public class TabulationSolver<T, P, F> {

197       /**
198        * Solve the dataflow problem.
199        * 
200        * @return a representation of the result
201        */
202       public TabulationResult<T, P, F> solve() throws CancelException {
203     
204         try {
205           initialize();
206           forwardTabulateSLRPs();
207           Result r = new Result();
208           return r;
209         } catch (CancelException e) {
210           // store a partially-tabulated result in the thrown exception.
211           Result r = new Result();
212           throw new TabulationCancelException(e, r);
213         } catch (CancelRuntimeException e) {
214           // store a partially-tabulated result in the thrown exception.
215           Result r = new Result();
216           throw new TabulationCancelException(e, r);
217         }
218       }

239       /**
240        * See POPL 95 paper for this algorithm, Figure 3
241        * 
242        * @throws CancelException
243        */
244       private void forwardTabulateSLRPs() throws CancelException {
245         assert curPathEdge == null : "curPathEdge should not be non-null here";
246         if (worklist == null) {
247           worklist = makeWorklist();
248         }
249         while (worklist.size() > 0) {
250           MonitorUtil.throwExceptionIfCanceled(progressMonitor);
251           if (verbose) {
252             performVerboseAction();
253           }
254           if (PERIODIC_WIPE_SOFT_CACHES) {
255             tendToSoftCaches();
256           }
257     
258           final PathEdge<T> edge = popFromWorkList();
259           if (DEBUG_LEVEL > 0) {
260             System.err.println("TABULATE " + edge);
261           }
262           curPathEdge = edge;
263           int j = merge(edge.entry, edge.d1, edge.target, edge.d2);
264           if (j == -1 && DEBUG_LEVEL > 0) {
265             System.err.println("merge -1: DROPPING");
266           }
267           if (j != -1) {
268             if (j != edge.d2) {
269               // this means that we don't want to push the edge. instead,
270               // we'll push the merged fact. a little tricky, but i think should
271               // work.
272               if (DEBUG_LEVEL > 0) {
273                 System.err.println("propagating merged fact " + j);
274               }
275               propagate(edge.entry, edge.d1, edge.target, j);
276             } else {
277               if (supergraph.isCall(edge.target)) {
278                 // [13]
279                 processCall(edge);
280               } else if (supergraph.isExit(edge.target)) {
281                 // [21]
282                 processExit(edge);
283               } else {
284                 // [33]
285                 processNormal(edge);
286               }
287             }
288           }
289         }
290         curPathEdge = null;
291       }

293       /**
294        * For some reason (either a bug in our code that defeats soft references, or a bad policy in the GC), leaving soft reference
295        * caches to clear themselves out doesn't work. Help it out.

297        * It's unfortunate that this method exits.
298        */
299       protected void tendToSoftCaches() {
300         wipeCount++;
301         if (wipeCount > WIPE_SOFT_CACHE_INTERVAL) {
302           wipeCount = 0;
303           ReferenceCleanser.clearSoftCaches();
304         }
305       }

324       /**
325        * Handle lines [33-37] of the algorithm
326        * 
327        * @param edge
328        */
329       private void processNormal(final PathEdge<T> edge) {
330         if (DEBUG_LEVEL > 0) {
331           System.err.println("process normal: " + edge);
332         }
333         for (Iterator<? extends T> it = supergraph.getSuccNodes(edge.target); it.hasNext();) {
334           final T m = it.next();
335           if (DEBUG_LEVEL > 0) {
336             System.err.println("normal successor: " + m);
337           }
338           IUnaryFlowFunction f = flowFunctionMap.getNormalFlowFunction(edge.target, m);
339           IntSet D3 = computeFlow(edge.d2, f);
340           if (DEBUG_LEVEL > 0) {
341             System.err.println(" reached: " + D3);
342           }
343           if (D3 != null) {
344             D3.foreach(new IntSetAction() {
345               @Override
346               public void act(int d3) {
347                 propagate(edge.entry, edge.d1, m, d3);
348               }
349             });
350           }
351         }
352       }

354       /**
355        * Handle lines [21 - 32] of the algorithm, propagating information from an exit node.
356        * 
357        * Note that we've changed the way we record summary edges. Summary edges are now associated with a callee (s_p,exit), where the
358        * original algorithm used a call, return pair in the caller.
359        */
360       protected void processExit(final PathEdge<T> edge) {
361         if (DEBUG_LEVEL > 0) {
362           System.err.println("process exit: " + edge);
363         }
364     
365         final LocalSummaryEdges summaries = findOrCreateLocalSummaryEdges(supergraph.getProcOf(edge.target));
366         int s_p_n = supergraph.getLocalBlockNumber(edge.entry);
367         int x = supergraph.getLocalBlockNumber(edge.target);
368         if (!summaries.contains(s_p_n, x, edge.d1, edge.d2)) {
369           summaries.insertSummaryEdge(s_p_n, x, edge.d1, edge.d2);
370         }
371         assert curSummaryEdge == null : "curSummaryEdge should be null here";
372         curSummaryEdge = edge;
373     
374         final CallFlowEdges callFlow = findOrCreateCallFlowEdges(edge.entry);
375     
376         // [22] for each c /in callers(p)
377         IntSet callFlowSourceNodes = callFlow.getCallFlowSourceNodes(edge.d1);
378         if (callFlowSourceNodes != null) {
379           for (IntIterator it = callFlowSourceNodes.intIterator(); it.hasNext();) {
380             // [23] for each d4 s.t. <c,d4> -> <s_p,d1> occurred earlier
381             int globalC = it.next();
382             final IntSet D4 = callFlow.getCallFlowSources(globalC, edge.d1);
383     
384             // [23] for each d5 s.t. <e_p,d2> -> <returnSite(c),d5> ...
385             propagateToReturnSites(edge, supergraph.getNode(globalC), D4);
386           }
387         }
388         curSummaryEdge = null;
389       }

391       /**
392        * Propagate information for an "exit" edge to the appropriate return sites
393        * 
394        * [23] for each d5 s.t. <s_p,d2> -> <returnSite(c),d5> ..
395        * 
396        * @param edge the edge being processed
397        * @param succ numbers of the nodes that are successors of edge.n (the return block in the callee) in the call graph.
398        * @param c a call site of edge.s_p
399        * @param D4 set of d1 s.t. <c, d1> -> <edge.s_p, edge.d2> was recorded as call flow
400        */
401       private void propagateToReturnSites(final PathEdge<T> edge, final T c, final IntSet D4) {
402         P proc = supergraph.getProcOf(c);
403         final T[] entries = supergraph.getEntriesForProcedure(proc);
404     
405         // we iterate over each potential return site;
406         // we might have multiple return sites due to exceptions
407         // note that we might have different summary edges for each
408         // potential return site, and different flow functions from this
409         // exit block to each return site.
410         for (Iterator<? extends T> retSites = supergraph.getReturnSites(c, supergraph.getProcOf(edge.target)); retSites.hasNext();) {
411           final T retSite = retSites.next();
412           if (DEBUG_LEVEL > 1) {
413             System.err.println("candidate return site: " + retSite + " " + supergraph.getNumber(retSite));
414           }
415           // note: since we might have multiple exit nodes for the callee, (to handle exceptional returns)
416           // not every return site might be valid for this exit node (edge.n).
417           // so, we'll filter the logic by checking that we only process reachable return sites.
418           // the supergraph carries the information regarding the legal successors
419           // of the exit node
420           if (!supergraph.hasEdge(edge.target, retSite)) {
421             continue;
422           }
423           if (DEBUG_LEVEL > 1) {
424             System.err.println("feasible return site: " + retSite);
425           }
426           final IFlowFunction retf = flowFunctionMap.getReturnFlowFunction(c, edge.target, retSite);
427           if (retf instanceof IBinaryReturnFlowFunction) {
428             propagateToReturnSiteWithBinaryFlowFunction(edge, c, D4, entries, retSite, retf);
429           } else {
430             final IntSet D5 = computeFlow(edge.d2, (IUnaryFlowFunction) retf);
431             if (DEBUG_LEVEL > 1) {
432               System.err.println("D4" + D4);
433               System.err.println("D5 " + D5);
434             }
435             IntSetAction action = new IntSetAction() {
436               @Override
437               public void act(final int d4) {
438                 propToReturnSite(c, entries, retSite, d4, D5);
439               }
440             };
441             D4.foreach(action);
442           }
443         }
444       }

446       /**
447        * Propagate information for an "exit" edge to a caller return site
448        * 
449        * [23] for each d5 s.t. <s_p,d2> -> <returnSite(c),d5> ..
450        * 
451        * @param edge the edge being processed
452        * @param c a call site of edge.s_p
453        * @param D4 set of d1 s.t. <c, d1> -> <edge.s_p, edge.d2> was recorded as call flow
454        * @param entries the blocks in the supergraph that are entries for the procedure of c
455        * @param retSite the return site being propagated to
456        * @param retf the flow function
457        */
458       private void propagateToReturnSiteWithBinaryFlowFunction(final PathEdge edge, final T c, final IntSet D4, final T[] entries,
459           final T retSite, final IFlowFunction retf) {
460         D4.foreach(new IntSetAction() {
461           @Override
462           public void act(final int d4) {
463             final IntSet D5 = computeBinaryFlow(d4, edge.d2, (IBinaryReturnFlowFunction) retf);
464             propToReturnSite(c, entries, retSite, d4, D5);
465           }
466     
467         });
468       }

470       /**
471        * Propagate information to a particular return site.
472        * 
473        * @param c the corresponding call site
474        * @param entries entry nodes in the caller
475        * @param retSite the return site
476        * @param d4 a fact s.t. <c, d4> -> <callee, d2> was recorded as call flow and <callee, d2> is the source of the summary edge
477        *          being applied
478        * @param D5 facts to propagate to return site
479        */
480       private void propToReturnSite(final T c, final T[] entries, final T retSite, final int d4, final IntSet D5) {
481         if (D5 != null) {
482           D5.foreach(new IntSetAction() {
483             @Override
484             public void act(final int d5) {
485               // [26 - 28]
486               // note that we've modified the algorithm here to account
487               // for potential
488               // multiple entry nodes. Instead of propagating the new
489               // summary edge
490               // with respect to one s_profOf(c), we have to propagate
491               // for each
492               // potential entry node s_p /in s_procof(c)
493               for (int i = 0; i < entries.length; i++) {
494                 final T s_p = entries[i];
495                 if (DEBUG_LEVEL > 1) {
496                   System.err.println(" do entry " + s_p);
497                 }
498                 IntSet D3 = getInversePathEdges(s_p, c, d4);
499                 if (DEBUG_LEVEL > 1) {
500                   System.err.println("D3" + D3);
501                 }
502                 if (D3 != null) {
503                   D3.foreach(new IntSetAction() {
504                     @Override
505                     public void act(int d3) {
506                       // set curPathEdge to be consistent with its setting in processCall() when applying a summary edge
507                       curPathEdge = PathEdge.createPathEdge(s_p, d3, c, d4);
508                       propagate(s_p, d3, retSite, d5);
509                     }
510                   });
511                 }
512               }
513             }
514           });
515         }
516       }

518       /**
519        * @param s_p
520        * @param n
521        * @param d2 note that s_p must be an entry for procof(n)
522        * @return set of d1 s.t. <s_p, d1> -> <n, d2> is a path edge, or null if none found
523        */
524       protected IntSet getInversePathEdges(T s_p, T n, int d2) {
525         int number = supergraph.getLocalBlockNumber(n);
526         LocalPathEdges lp = pathEdges.get(s_p);
527         if (lp == null) {
528           return null;
529         }
530         return lp.getInverse(number, d2);
531       }

533       /**
534        * Handle lines [14 - 19] of the algorithm, propagating information into and across a call site.
535        */
536       protected void processCall(final PathEdge<T> edge) {
537         if (DEBUG_LEVEL > 0) {
538           System.err.println("process call: " + edge);
539         }
540     
541         // c:= number of the call node
542         final int c = supergraph.getNumber(edge.target);
543     
544         Collection<T> allReturnSites = HashSetFactory.make();
545         // populate allReturnSites with return sites for missing calls.
546         for (Iterator<? extends T> it = supergraph.getReturnSites(edge.target, null); it.hasNext();) {
547           allReturnSites.add(it.next());
548         }
549         // [14 - 16]
550         boolean hasCallee = false;
551         for (Iterator<? extends T> it = supergraph.getCalledNodes(edge.target); it.hasNext();) {
552           hasCallee = true;
553           final T callee = it.next();
554           processParticularCallee(edge, c, allReturnSites, callee);
555         }
556         // special logic: in backwards problems, a "call" node can have
557         // "normal" successors as well. deal with these.
558         for (Iterator<? extends T> it = supergraph.getNormalSuccessors(edge.target); it.hasNext();) {
559           final T m = it.next();
560           if (DEBUG_LEVEL > 0) {
561             System.err.println("normal successor: " + m);
562           }
563           IUnaryFlowFunction f = flowFunctionMap.getNormalFlowFunction(edge.target, m);
564           IntSet D3 = computeFlow(edge.d2, f);
565           if (DEBUG_LEVEL > 0) {
566             System.err.println("normal successor reached: " + D3);
567           }
568           if (D3 != null) {
569             D3.foreach(new IntSetAction() {
570               @Override
571               public void act(int d3) {
572                 propagate(edge.entry, edge.d1, m, d3);
573               }
574             });
575           }
576         }
577     
578         // [17 - 19]
579         // we modify this to handle each return site individually
580         for (final T returnSite : allReturnSites) {
581           if (DEBUG_LEVEL > 0) {
582             System.err.println(" process return site: " + returnSite);
583           }
584           IUnaryFlowFunction f = null;
585           if (hasCallee) {
586             f = flowFunctionMap.getCallToReturnFlowFunction(edge.target, returnSite);
587           } else {
588             f = flowFunctionMap.getCallNoneToReturnFlowFunction(edge.target, returnSite);
589           }
590           IntSet reached = computeFlow(edge.d2, f);
591           if (DEBUG_LEVEL > 0) {
592             System.err.println("reached: " + reached);
593           }
594           if (reached != null) {
595             reached.foreach(new IntSetAction() {
596               @Override
597               public void act(int x) {
598                 assert x >= 0;
599                 assert edge.d1 >= 0;
600                 propagate(edge.entry, edge.d1, returnSite, x);
601               }
602             });
603           }
604         }
605       }

607       /**
608        * handle a particular callee for some call node.
609        * 
610        * @param edge the path edge being processed
611        * @param callNodeNum the number of the call node in the supergraph
612        * @param allReturnSites a set collecting return sites for the call. This set is mutated with the return sites for this callee.
613        * @param calleeEntry the entry node of the callee in question
614        */
615       protected void processParticularCallee(final PathEdge<T> edge, final int callNodeNum, Collection<T> allReturnSites, final T calleeEntry) {
616         if (DEBUG_LEVEL > 0) {
617           System.err.println(" process callee: " + calleeEntry);
618         }
619         // reached := {d1} that reach the callee
620         MutableSparseIntSet reached = MutableSparseIntSet.makeEmpty();
621         final Collection<T> returnSitesForCallee = Iterator2Collection.toSet(supergraph.getReturnSites(edge.target, supergraph
622             .getProcOf(calleeEntry)));
623         allReturnSites.addAll(returnSitesForCallee);
624         // we modify this to handle each return site individually. Some types of problems
625         // compute different flow functions for each return site.
626         for (final T returnSite : returnSitesForCallee) {
627           IUnaryFlowFunction f = flowFunctionMap.getCallFlowFunction(edge.target, calleeEntry, returnSite);
628           IntSet r = computeFlow(edge.d2, f);
629           if (r != null) {
630             reached.addAll(r);
631           }
632         }
633         // in some problems, we also want to consider flow into a callee that can never flow out
634         // via a return. in this case, the return site is null.
635         IUnaryFlowFunction f = flowFunctionMap.getCallFlowFunction(edge.target, calleeEntry, null);
636         IntSet r = computeFlow(edge.d2, f);
637         if (r != null) {
638           reached.addAll(r);
639         }
640         if (DEBUG_LEVEL > 0) {
641           System.err.println(" reached: " + reached);
642         }
643         if (reached != null) {
644           final LocalSummaryEdges summaries = summaryEdges.get(supergraph.getProcOf(calleeEntry));
645           final CallFlowEdges callFlow = findOrCreateCallFlowEdges(calleeEntry);
646           final int s_p_num = supergraph.getLocalBlockNumber(calleeEntry);
647     
648           reached.foreach(new IntSetAction() {
649             @Override
650             public void act(final int d1) {
651               // we get reuse if we _don't_ propagate a new fact to the callee entry
652               final boolean gotReuse = !propagate(calleeEntry, d1, calleeEntry, d1);
653               recordCall(edge.target, calleeEntry, d1, gotReuse);
654               // cache the fact that we've flowed <c, d2> -> <callee, d1> by a
655               // call flow
656               callFlow.addCallEdge(callNodeNum, edge.d2, d1);
657               // handle summary edges now as well. this is different from the PoPL
658               // 95 paper.
659               if (summaries != null) {
660                 // for each exit from the callee
661                 P p = supergraph.getProcOf(calleeEntry);
662                 T[] exits = supergraph.getExitsForProcedure(p);
663                 for (int e = 0; e < exits.length; e++) {
664                   final T exit = exits[e];
665                   if (DEBUG_LEVEL > 0) {
666                     assert supergraph.containsNode(exit);
667                   }
668                   int x_num = supergraph.getLocalBlockNumber(exit);
669                   // reachedBySummary := {d2} s.t. <callee,d1> -> <exit,d2>
670                   // was recorded as a summary edge
671                   IntSet reachedBySummary = summaries.getSummaryEdges(s_p_num, x_num, d1);
672                   if (reachedBySummary != null) {
673                     for (final T returnSite : returnSitesForCallee) {
674                       // if "exit" is a valid exit from the callee to the return
675                       // site being processed
676                       if (supergraph.hasEdge(exit, returnSite)) {
677                         final IFlowFunction retf = flowFunctionMap.getReturnFlowFunction(edge.target, exit, returnSite);
678                         reachedBySummary.foreach(new IntSetAction() {
679                           @Override
680                           public void act(int d2) {
681                             assert curSummaryEdge == null : "curSummaryEdge should be null here";
682                             curSummaryEdge = PathEdge.createPathEdge(calleeEntry, d1, exit, d2);
683                             if (retf instanceof IBinaryReturnFlowFunction) {
684                               final IntSet D5 = computeBinaryFlow(edge.d2, d2, (IBinaryReturnFlowFunction) retf);
685                               if (D5 != null) {
686                                 D5.foreach(new IntSetAction() {
687                                   @Override
688                                   public void act(int d5) {
689                                     propagate(edge.entry, edge.d1, returnSite, d5);
690                                   }
691                                 });
692                               }
693                             } else {
694                               final IntSet D5 = computeFlow(d2, (IUnaryFlowFunction) retf);
695                               if (D5 != null) {
696                                 D5.foreach(new IntSetAction() {
697                                   @Override
698                                   public void act(int d5) {
699                                     propagate(edge.entry, edge.d1, returnSite, d5);
700                                   }
701                                 });
702                               }
703                             }
704                             curSummaryEdge = null;
705                           }
706                         });
707                       }
708                     }
709                   }
710                 }
711               }
712             }
713           });
714         }
715       }

777       /**
778        * Propagate the fact <s_p,i> -> <n, j> has arisen as a path edge. Returns <code>true</code> iff the path edge was not previously
779        * observed.
780        * 
781        * @param s_p entry block
782        * @param i dataflow fact on entry
783        * @param n reached block
784        * @param j dataflow fact reached
785        */
786       protected boolean propagate(T s_p, int i, T n, int j) {
787         int number = supergraph.getLocalBlockNumber(n);
788         if (number < 0) {
789           System.err.println("BOOM " + n);
790           supergraph.getLocalBlockNumber(n);
791         }
792         assert number >= 0;
793     
794         LocalPathEdges pLocal = findOrCreateLocalPathEdges(s_p);
795     
796         assert j >= 0;
797     
798         if (!pLocal.contains(i, number, j)) {
799           if (DEBUG_LEVEL > 0) {
800             System.err.println("propagate " + s_p + "  " + i + " " + number + " " + j);
801           }
802           pLocal.addPathEdge(i, number, j);
803           addToWorkList(s_p, i, n, j);
804           return true;
805         }
806         return false;
807       }

813       /**
814        * Merging: suppose we're doing propagate <s_p,i> -> <n,j> but we already have path edges <s_p,i> -> <n, x>, <s_p,i> -> <n,y>, and
815        * <s_p,i> -><n, z>.
816        * 
817        * let \alpha be the merge function. then instead of <s_p,i> -> <n,j>, we propagate <s_p,i> -> <n, \alpha(j,x,y,z) > !!!
818        * 
819        * return -1 if no fact should be propagated
820        */
821       private int merge(T s_p, int i, T n, int j) {
822         assert j >= 0;
823         IMergeFunction alpha = problem.getMergeFunction();
824         if (alpha != null) {
825           LocalPathEdges lp = pathEdges.get(s_p);
826           IntSet preExistFacts = lp.getReachable(supergraph.getLocalBlockNumber(n), i);
827           if (preExistFacts == null) {
828             return j;
829           } else {
830             int size = preExistFacts.size();
831             if ((size == 0) || ((size == 1) && preExistFacts.contains(j))) {
832               return j;
833             } else {
834               int result = alpha.merge(preExistFacts, j);
835               return result;
836             }
837           }
838         } else {
839           return j;
840         }
841       }

884       /**
885        * get the bitvector of facts that hold at the entry to a given node
886        * 
887        * @return IntSet representing the bitvector
888        */
889       public IntSet getResult(T node) {
890         P proc = supergraph.getProcOf(node);
891         int n = supergraph.getLocalBlockNumber(node);
892         T[] entries = supergraph.getEntriesForProcedure(proc);
893         MutableIntSet result = MutableSparseIntSet.makeEmpty();
894     
895         Set<T> allEntries = HashSetFactory.make(Arrays.asList(entries));
896         Set<PathEdge<T>> pSeeds = seeds.get(proc);
897         if (pSeeds != null) {
898         	for (PathEdge<T> seed : pSeeds) {
899         		allEntries.add(seed.entry);
900         	}
901         }    
902     
903         for (T entry : allEntries){
904         	LocalPathEdges lp = pathEdges.get(entry);
905         	if (lp != null) {
906         		result.addAll(lp.getReachable(n));
907         	}
908         }    	
909         
910         return result;
911       }

925         @Override
926         public String toString() {
927     
928           StringBuffer result = new StringBuffer();
929           TreeMap<Object, TreeSet<T>> map = new TreeMap<Object, TreeSet<T>>(ToStringComparator.instance());
930     
931           Comparator<Object> c = new Comparator<Object>() {
932             @Override
933             public int compare(Object o1, Object o2) {
934               if (!(o1 instanceof IBasicBlock)) {
935                 return -1;
936               }
937               IBasicBlock bb1 = (IBasicBlock) o1;
938               IBasicBlock bb2 = (IBasicBlock) o2;
939               return bb1.getNumber() - bb2.getNumber();
940             }
941           };
942           for (Iterator<? extends T> it = supergraph.iterator(); it.hasNext();) {
943             T n = it.next();
944             P proc = supergraph.getProcOf(n);
945             TreeSet<T> s = map.get(proc);
946             if (s == null) {
947               s = new TreeSet<T>(c);
948               map.put(proc, s);
949             }
950             s.add(n);
951           }
952     
953           for (Iterator<Map.Entry<Object, TreeSet<T>>> it = map.entrySet().iterator(); it.hasNext();) {
954             Map.Entry<Object, TreeSet<T>> e = it.next();
955             Set<T> s = e.getValue();
956             for (Iterator<T> it2 = s.iterator(); it2.hasNext();) {
957               T o = it2.next();
958               result.append(o + " : " + getResult(o) + "\n");
959             }
960           }
961           return result.toString();
962         }
