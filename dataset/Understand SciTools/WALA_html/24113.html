147       protected void updateSideEffects(PointsToSetVariable p, PointsToSetVariable rep) {
148         Set<UnarySideEffect> set = fixedSetMap.get(p);
149         if (set != null) {
150           for (Iterator it = set.iterator(); it.hasNext();) {
151             UnarySideEffect s = (UnarySideEffect) it.next();
152             s.replaceFixedSet(rep);
153           }
154           Set<UnarySideEffect> s2 = MapUtil.findOrCreateSet(fixedSetMap, rep);
155           s2.addAll(set);
156           fixedSetMap.remove(p);
157         }
158       }

252       /**
253        * If key is unified, returns the representative
254        * 
255        * @param key
256        * @return the dataflow variable that tracks the points-to set for key
257        */
258       public PointsToSetVariable findOrCreatePointsToSet(PointerKey key) {
259     
260         if (key == null) {
261           throw new IllegalArgumentException("null key");
262         }
263     
264         if (pointsToMap.isImplicit(key)) {
265           System.err.println("Did not expect to findOrCreatePointsToSet for implicitly represented PointerKey");
266           System.err.println(key);
267           Assertions.UNREACHABLE();
268         }
269         PointsToSetVariable result = pointsToMap.getPointsToSet(key);
270         if (result == null) {
271           result = new PointsToSetVariable(key);
272           pointsToMap.put(key, result);
273         } else {
274           // check that the filter for this variable remains unique
275           if (!pointsToMap.isUnified(key) && key instanceof FilteredPointerKey) {
276             PointerKey pk = result.getPointerKey();
277             if (!(pk instanceof FilteredPointerKey)) {
278               // add a filter for all future evaluations.
279               // this is tricky, but the logic is OK .. any constraints that need
280               // the filter will see it ...
281               // CALLERS MUST BE EXTRA CAREFUL WHEN DEALING WITH UNIFICATION!
282               result.setPointerKey(key);
283               pk = key;
284             }
285             FilteredPointerKey fpk = (FilteredPointerKey) pk;
286             if (fpk == null) {
287               Assertions.UNREACHABLE("fpk is null");
288             }
289             if (key == null) {
290               Assertions.UNREACHABLE("key is null");
291             }
292             if (fpk.getTypeFilter() == null) {
293               Assertions.UNREACHABLE("fpk.getTypeFilter() is null");
294             }
295             if (!fpk.getTypeFilter().equals(((FilteredPointerKey) key).getTypeFilter())) {
296               Assertions.UNREACHABLE("Cannot use filter " + ((FilteredPointerKey) key).getTypeFilter() + " for " + key
297                   + ": previously created different filter " + fpk.getTypeFilter());
298             }
299           }
300         }
301         return result;
302       }

316        * NB: this is idempotent ... if the given constraint exists, it will not be added to the system; however, this will be more
317        * expensive since it must check if the constraint pre-exits.
318        * 
319        * @return true iff the system changes
320        */
321       public boolean newConstraint(PointerKey lhs, UnaryOperator<PointsToSetVariable> op, PointerKey rhs) {
322         if (lhs == null) {
323           throw new IllegalArgumentException("null lhs");
324         }
325         if (op == null) {
326           throw new IllegalArgumentException("op null");
327         }
328         if (rhs == null) {
329           throw new IllegalArgumentException("rhs null");
330         }
331         if (DEBUG) {
332           System.err.println("Add constraint A: " + lhs + " " + op + " " + rhs);
333         }
334         PointsToSetVariable L = findOrCreatePointsToSet(lhs);
335         PointsToSetVariable R = findOrCreatePointsToSet(rhs);
336         if (op instanceof FilterOperator) {
337           // we do not want to revert the lhs to pre-transitive form;
338           // we instead want to check in the outer loop of the pre-transitive
339           // solver if the value of L changes.
340           pointsToMap.recordTransitiveRoot(L.getPointerKey());
341           if (!(L.getPointerKey() instanceof FilteredPointerKey)) {
342             Assertions.UNREACHABLE("expected filtered lhs " + L.getPointerKey() + " " + L.getPointerKey().getClass() + " " + lhs + " "
343                 + lhs.getClass());
344           }
345         }
346         return newStatement(L, op, R, true, true);
347       }

408       /**
409        * @return true iff the system changes
410        */
411       public boolean newConstraint(PointerKey lhs, InstanceKey value) {
412         if (DEBUG) {
413           System.err.println("Add constraint B: " + lhs + " U= " + value);
414         }
415         pointsToMap.recordTransitiveRoot(lhs);
416     
417         // we don't actually add a constraint.
418         // instead, we immediately add the value to the points-to set.
419         // This works since the solver is monotonic with TOP = {}
420         PointsToSetVariable L = findOrCreatePointsToSet(lhs);
421         int index = findOrCreateIndexForInstanceKey(value);
422         if (L.contains(index)) {
423           // a no-op
424           return false;
425         } else {
426           L.add(index);
427     
428           // also register that we have an instanceKey for the klass
429           assert value.getConcreteType() != null;
430     
431           if (!value.getConcreteType().getReference().equals(TypeReference.JavaLangObject)) {
432             registerInstanceOfClass(value.getConcreteType(), index);
433           }
434     
435           // we'd better update the worklist appropriately
436           // if graphNodeId == -1, then there are no equations that use this
437           // variable.
438           if (L.getGraphNodeId() > -1) {
439             changedVariable(L);
440           }
441           return true;
442         }
443     
444       }

446       /**
447        * Record that we have a new instanceKey for a given declared type.
448        */
449       private void registerInstanceOfClass(IClass klass, int index) {
450     
451         if (DEBUG) {
452           System.err.println("registerInstanceOfClass " + klass + " " + index);
453         }
454     
455         assert !klass.getReference().equals(TypeReference.JavaLangObject);
456     
457         try {
458           IClass T = klass;
459           registerInstanceWithAllSuperclasses(index, T);
460           registerInstanceWithAllInterfaces(klass, index);
461     
462           if (klass.isArrayClass()) {
463             ArrayClass aClass = (ArrayClass) klass;
464             int dim = aClass.getDimensionality();
465             registerMultiDimArraysForArrayOfObjectTypes(dim, index, aClass);
466     
467             IClass elementClass = aClass.getInnermostElementClass();
468             if (elementClass != null) {
469               registerArrayInstanceWithAllSuperclassesOfElement(index, elementClass, dim);
470               registerArrayInstanceWithAllInterfacesOfElement(index, elementClass, dim);
471             }
472           }
473         } catch (ClassHierarchyException e) {
474           Warnings.add(ClassHierarchyWarning.create(e.getMessage()));
475         }
476       }

478       private int registerMultiDimArraysForArrayOfObjectTypes(int dim, int index, ArrayClass aClass) {
479     
480         for (int i = 1; i < dim; i++) {
481           TypeReference jlo = makeArray(TypeReference.JavaLangObject, i);
482           IClass jloClass = null;
483           jloClass = aClass.getClassLoader().lookupClass(jlo.getName());
484           MutableIntSet set = findOrCreateSparseSetForClass(jloClass);
485           set.add(index);
486         }
487         return dim;
488       }

503         }

506       private TypeReference makeArray(TypeReference element, int dim) {
507         TypeReference iArrayRef = element;
508         for (int i = 0; i < dim; i++) {
509           iArrayRef = TypeReference.findOrCreateArrayOf(iArrayRef);
510         }
511         return iArrayRef;
512       }

514       private void registerArrayInstanceWithAllSuperclassesOfElement(int index, IClass elementClass, int dim)
515           throws ClassHierarchyException {
516         IClass T;
517         // register the array with each supertype of the element class
518         T = elementClass.getSuperclass();
519         while (T != null) {
520           TypeReference tArrayRef = makeArray(T.getReference(), dim);
521           IClass tArrayClass = null;
522           tArrayClass = T.getClassLoader().lookupClass(tArrayRef.getName());
523           MutableIntSet set = findOrCreateSparseSetForClass(tArrayClass);
524           set.add(index);
525           if (DEBUG) {
526             System.err.println("dense filter for class " + tArrayClass + " " + set);
527           }
528           T = T.getSuperclass();
529         }
530       }

618       @Override
619       public void performVerboseAction() {
620         super.performVerboseAction();
621         if (DEBUG_MEMORY) {
622           DEBUG_MEM_COUNTER++;
623           if (DEBUG_MEM_COUNTER % DEBUG_MEM_INTERVAL == 0) {
624             DEBUG_MEM_COUNTER = 0;
625             ReferenceCleanser.clearSoftCaches();
626     
627             System.err.println(flowGraph.spaceReport());
628     
629             System.err.println("Analyze leaks..");
630             HeapTracer.traceHeap(Collections.singleton(this), true);
631             System.err.println("done analyzing leaks");
632           }
633         }
634         if (getFixedPointSystem() instanceof VerboseAction) {
635           ((VerboseAction) getFixedPointSystem()).performVerboseAction();
636         }
637         if (!workList.isEmpty()) {
638           AbstractStatement s = workList.takeStatement();
639           System.err.println(printRHSInstances(s));
640           workList.insertStatement(s);
641           System.err.println("CGNodes: " + cg.getNumberOfNodes());
642         }
643     
644       }

646       private String printRHSInstances(AbstractStatement s) {
647         if (s instanceof UnaryStatement) {
648           UnaryStatement u = (UnaryStatement) s;
649           PointsToSetVariable rhs = (PointsToSetVariable) u.getRightHandSide();
650           IntSet value = rhs.getValue();
651           final int[] topFive = new int[5];
652           value.foreach(new IntSetAction() {
653             @Override
654             public void act(int x) {
655               for (int i = 0; i < 4; i++) {
656                 topFive[i] = topFive[i + 1];
657               }
658               topFive[4] = x;
659             }
660           });
661           StringBuffer result = new StringBuffer();
662           for (int i = 0; i < 5; i++) {
663             int p = topFive[i];
664             if (p != 0) {
665               InstanceKey ik = getInstanceKey(p);
666               result.append(p).append("  ").append(ik).append("\n");
667             }
668           }
669           return result.toString();
670         } else {
671           return s.getClass().toString();
672         }
673       }

770       /**
771        * Unify the points-to-sets for the variables identified by the set s
772        * 
773        * @param s numbers of points-to-set variables
774        * @throws IllegalArgumentException if s is null
775        */
776       public void unify(IntSet s) {
777         if (s == null) {
778           throw new IllegalArgumentException("s is null");
779         }
780         // cache the variables represented
781         HashSet<PointsToSetVariable> cache = HashSetFactory.make(s.size());
782         for (IntIterator it = s.intIterator(); it.hasNext();) {
783           int i = it.next();
784           cache.add(pointsToMap.getPointsToSet(i));
785         }
786     
787         // unify the variables
788         pointsToMap.unify(s);
789         int rep = pointsToMap.getRepresentative(s.intIterator().next());
790     
791         // clean up the equations
792         updateEquationsForUnification(cache, rep);
793     
794         // special logic to clean up side effects
795         updateSideEffectsForUnification(cache, rep);
796       }
797     
798       /**
799        * Update side effect after unification
800        * 
801        * @param s set of PointsToSetVariables that have been unified
802        * @param rep number of the representative variable for the unified set.
803        */
804       private void updateSideEffectsForUnification(HashSet<PointsToSetVariable> s, int rep) {
805         PointsToSetVariable pRef = pointsToMap.getPointsToSet(rep);
806         for (Iterator<PointsToSetVariable> it = s.iterator(); it.hasNext();) {
807           PointsToSetVariable p = it.next();
808           updateSideEffects(p, pRef);
809         }
810       }

812       /**
813        * Update equation def/uses after unification
814        * 
815        * @param s set of PointsToSetVariables that have been unified
816        * @param rep number of the representative variable for the unified set.
817        */
818       @SuppressWarnings("unchecked")
819       private void updateEquationsForUnification(HashSet<PointsToSetVariable> s, int rep) {
820         PointsToSetVariable pRef = pointsToMap.getPointsToSet(rep);
821         for (Iterator<PointsToSetVariable> it = s.iterator(); it.hasNext();) {
822           PointsToSetVariable p = it.next();
823     
824           if (p != pRef) {
825             // pRef is the representative for p.
826             // be careful: cache the defs before mucking with the underlying system
827             for (Iterator d = Iterator2Collection.toSet(getStatementsThatDef(p)).iterator(); d.hasNext();) {
828               AbstractStatement as = (AbstractStatement) d.next();
829     
830               if (as instanceof AssignEquation) {
831                 AssignEquation assign = (AssignEquation) as;
832                 PointsToSetVariable rhs = assign.getRightHandSide();
833                 int rhsRep = pointsToMap.getRepresentative(pointsToMap.getIndex(rhs.getPointerKey()));
834                 if (rhsRep == rep) {
835                   flowGraph.removeStatement(as);
836                 } else {
837                   replaceLHS(pRef, p, as);
838                 }
839               } else {
840                 replaceLHS(pRef, p, as);
841               }
842             }
843             // be careful: cache the defs before mucking with the underlying system
844             for (Iterator u = Iterator2Collection.toSet(getStatementsThatUse(p)).iterator(); u.hasNext();) {
845               AbstractStatement as = (AbstractStatement) u.next();
846               if (as instanceof AssignEquation) {
847                 AssignEquation assign = (AssignEquation) as;
848                 PointsToSetVariable lhs = assign.getLHS();
849                 int lhsRep = pointsToMap.getRepresentative(pointsToMap.getIndex(lhs.getPointerKey()));
850                 if (lhsRep == rep) {
851                   flowGraph.removeStatement(as);
852                 } else {
853                   replaceRHS(pRef, p, as);
854                 }
855               } else {
856                 replaceRHS(pRef, p, as);
857               }
858             }
859             if (flowGraph.getNumberOfStatementsThatDef(p) == 0 && flowGraph.getNumberOfStatementsThatUse(p) == 0) {
860               flowGraph.removeVariable(p);
861             }
862           }
863         }
864       }

866       /**
867        * replace all occurrences of p on the rhs of a statement with pRef
868        * 
869        * @param as a statement that uses p in it's right-hand side
870        */
871       private void replaceRHS(PointsToSetVariable pRef, PointsToSetVariable p,
872           AbstractStatement<PointsToSetVariable, AbstractOperator<PointsToSetVariable>> as) {
873         if (as instanceof UnaryStatement) {
874           assert ((UnaryStatement) as).getRightHandSide() == p;
875           newStatement(as.getLHS(), (UnaryOperator<PointsToSetVariable>) as.getOperator(), pRef, false, false);
876         } else {
877           IVariable[] rhs = as.getRHS();
878           PointsToSetVariable[] newRHS = new PointsToSetVariable[rhs.length];
879           for (int i = 0; i < rhs.length; i++) {
880             if (rhs[i].equals(p)) {
881               newRHS[i] = pRef;
882             } else {
883               newRHS[i] = (PointsToSetVariable) rhs[i];
884             }
885           }
886           newStatement(as.getLHS(), as.getOperator(), newRHS, false, false);
887         }
888         flowGraph.removeStatement(as);
889       }

891       /**
892        * replace all occurences of p on the lhs of a statement with pRef
893        * 
894        * @param as a statement that defs p
895        */
896       private void replaceLHS(PointsToSetVariable pRef, PointsToSetVariable p,
897           AbstractStatement<PointsToSetVariable, AbstractOperator<PointsToSetVariable>> as) {
898         assert as.getLHS() == p;
899         if (as instanceof UnaryStatement) {
900           newStatement(pRef, (UnaryOperator<PointsToSetVariable>) as.getOperator(), (PointsToSetVariable) ((UnaryStatement) as)
901               .getRightHandSide(), false, false);
902         } else {
903           newStatement(pRef, as.getOperator(), as.getRHS(), false, false);
904         }
905         flowGraph.removeStatement(as);
906       }
