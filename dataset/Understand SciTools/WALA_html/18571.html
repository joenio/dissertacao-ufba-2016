61     /**
62      *  Helper for building the Descriptor of a model.
63      *
64      *  Parameters used in a model can be either marked as CREATE or REUSE. This information is derived
65      *  from the IInstantiationBehavior.
66      *
67      *  This class only handles parameters marked as REUSE: These will be parameters to the function 
68      *  representing the model itself. On all uses of a variable named REUSE the same Instance (optionally
69      *  altered using Phi) will be used.
70      *
71      *  ReuseParameters collects all those parameters and builds the Descriptor of the later model.
72      *  
73      *  Also ReuseParameters may be queried how to access these parameters the use of ParameterAccessor
74      *  is the better way to get them.
75      *
76      *  @see    com.ibm.wala.dalvik.ipa.callgraph.androidModel.parameters.IInstantiationBehavior
77      *  @see    com.ibm.wala.util.ssa.ParameterAccessor 
78      *  
79      *  @author Tobias Blaschke <code@tobiasblaschke.de>
80      *  @since  2013-11-02
81      */
82     public class ReuseParameters {
83          public static class ReuseParameter extends Parameter {
84             protected ReuseParameter(final int number, final String name, final TypeReference type, final MethodReference mRef, 
85                     final int descriptorOffset) {
86                 super(number, name, type, ParamerterDisposition.PARAM, BasedOn.IMETHOD, mRef, descriptorOffset);
87             }
88         }

108         /**
109          *  Searches the given entrypoints for those parameters.
110          *
111          *  A call to this function resets the internal knowledge of REUSE-Parameters. So in order to
112          *  get a union of these parameters a union of the given entrypoints has to be built.
113          *
114          *  @param  entrypoints The entrypoints to consider in the search.
115          */
116         public void collectParameters(final Iterable<? extends Entrypoint> entrypoints) {
117     //        int paramsToModel = firstParamSSA();
118             this.reuseParameters = new ArrayList<TypeName>();
119     
120             for (final Entrypoint ep : entrypoints) {
121                 final int paramCount = ep.getNumberOfParameters();
122     
123                 for (int i = 0; i < paramCount; ++i) {
124                     { // determine paramType
125                         final TypeReference[] types = ep.getParameterTypes(i);
126                         if (types.length < 1) {
127                             throw new IllegalStateException("The Etrypoint " + ep + " did not return any types for its " + i + "th parameter");
128                         }
129                         
130                         // Assert the rest of the types have the same name
131                         for (int j = 0; j < types.length; ++j) {
132                             final TypeName paramType = types[j].getName();
133     
134                             if (isReuse(paramType, ALL_TARGETS)) {
135                                 if (! reuseParameters.contains(paramType)) {    // XXX: Why not use a Set?
136                                     reuseParameters.add(paramType);
137                                     
138                                 }
139                             }
140                         }
141                     }
142                 }
143             }
144         }

146         /**
147          *  Get the ssa-number for a parameter to an IMethod.
148          *
149          *  @see    com.ibm.wala.util.ssa.ParameterAccessor
150          */
151         private int ssaFor(IMethod inCallTo, int paramNo) {
152             assert (paramNo >= 0);
153             assert (paramNo < inCallTo.getNumberOfParameters());
154     
155             if (inCallTo.isStatic()) {
156                 return paramNo + 1;
157             } else {
158                 return paramNo + 1; // TODO 2 or 1?
159             }
160         }

162         /**
163          *  Get the first paramNo of a given type.
164          *
165          *  @see    com.ibm.wala.util.ssa.ParameterAccessor
166          */
167         private int firstOf(TypeName type, IMethod inCallTo) {
168             for (int i = 0; i < inCallTo.getNumberOfParameters(); ++i) {
169                 if (inCallTo.getParameterType(i).getName().equals(type)) {
170                     return i;
171                 }
172             }
173             throw new IllegalArgumentException(type.toString() + " is not a parameter to " + inCallTo);
174         }

176         /**
177          *  Is the parameter REUSE in a call from forModel to inCallTo.
178          *
179          *  The 'forModel' was set in the constructor. Even so a parameter occurs in the descriptor it
180          *  does not have to be REUSE for all calls.
181          *
182          *  The result of this method may vary over time :/
183          *
184          *  @param  param       The parameter in question of being reuse
185          *  @param  inCallTo    The callee to query the REUSEness for
186          */
187         public boolean isReuse(TypeName param, IMethod inCallTo) {  // TODO: Use IInstantiationBehavior.getBehavior(TypeName param, IMethod inCallTo)
188             final TypeName asParameterTo;
189             final MethodReference inCall;
190             /*final*/ String withName;
191     
192             if ((inCallTo != null) && (inCallTo != ALL_TARGETS)) {
193                 final int bcIndex = 0; // The PC to get the variable name from
194                 final int localNumber = ssaFor(inCallTo, firstOf(param, inCallTo));
195                 try {
196                     withName = inCallTo.getLocalVariableName (bcIndex, localNumber);
197                 } catch (UnsupportedOperationException e) {
198                     // DexIMethod doesn't implement this :(
199                     
200                     withName = null;
201                 }
202                 asParameterTo = inCallTo.getDeclaringClass().getName();
203                 inCall = inCallTo.getReference();
204                 /*{ // DEBUG
205                     System.out.println("isReuse: ");
206                     System.out.println("\tparam = \t\t" + param);
207                     System.out.println("\tasParameterTo =\t" + asParameterTo);
208                     System.out.println("\tinCall =\t" + inCall);
209                     System.out.println("\twithName =\t" + withName);
210                 } // */
211             } else {
212                 withName = null;
213                 asParameterTo = null;
214                 inCall = null;
215             }
216     
217             final InstanceBehavior beh = this.instanceBehavior.getBehavior(param, asParameterTo, inCall, withName);
218     
219             return (beh == InstanceBehavior.REUSE);
220         }

222         /**
223          *  Generate the descriptor to use for the model.
224          *
225          *  @param returnType the return type of the later function
226          */
227         private Descriptor toDescriptor(TypeName returnType) { // Keep private!
228             final TypeName[] aTypes = reuseParameters.toArray(new TypeName[reuseParameters.size()]);
229     
230             return Descriptor.findOrCreate(aTypes, returnType); 
231         }

233         public MethodReference toMethodReference(final AndroidModelParameterManager pm) {
234             final TypeReference clazz = this.forModel.getDeclaringClass().getReference();
235             final Atom name = this.forModel.getName();
236             final TypeName returnType = this.forModel.getReturnType();
237     
238             final Descriptor descr = toDescriptor(returnType);
239             final MethodReference mRef = MethodReference.findOrCreate(clazz, name, descr);
240     
241             // TODO: Build Parameters and register them
242             if (pm != null) {
243                 int paramSSA = 1;
244                 if (! this.forModel.isStatic()) {
245                     paramSSA = 2;
246                 }
247                 final int descriptorOffset;
248                 if (this.forModel.isStatic()) {
249                     descriptorOffset = 0;  // TODO Verify!
250                 } else {
251                     descriptorOffset = -1;  // TODO Verify!
252                 }
253     
254                 for (final TypeName param : reuseParameters) {
255                     final String tName = null;   // TODO
256                     final TypeReference tRef = TypeReference.find (ClassLoaderReference.Primordial, param); // TODO: Loaders!
257                     final ReuseParameter rp = new ReuseParameter(paramSSA, tName, tRef, mRef, descriptorOffset );
258                     pm.setAllocation(rp);
259                     //pm.setAllocation(tRef, paramSSA); // TODO: Old-school call
260                     
261     
262                     paramSSA++;
263                 }
264             }
265     
266             return mRef;
267         }
