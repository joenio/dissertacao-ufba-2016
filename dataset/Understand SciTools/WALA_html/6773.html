11     package com.ibm.wala.dataflow.IFDS;
12     
13     import com.ibm.wala.util.intset.IntSet;
14     import com.ibm.wala.util.intset.SparseIntSet;
15     
16     /**
17      * A function which gens a vector of outgoing dataflow facts. 
18      */
19     public class VectorGenFlowFunction implements IReversibleFlowFunction {
20     
21       private final IntSet gen;
22       
23       /**
24        * @param gen
25        *          the intset of facts which are gen'ned by this flow function. gen
26        *          <em>must</em> contain 0.
27        */
28       private VectorGenFlowFunction(IntSet gen) {
29         this.gen = gen;
30         assert gen.contains(0);
31       }
32     
33       @Override
34       public IntSet getTargets(int i) {
35         return (i == 0) ? gen : gen.contains(i) ? null : SparseIntSet.singleton(i);
36       }
37     
38       @Override
39       public IntSet getSources(int i) {
40         return (gen.contains(i)) ? SparseIntSet.singleton(0) : SparseIntSet.singleton(i);
41       }
42     
43       /**
44        * @param gen the intset of facts which should be gen'ed by a function
45        * @return an instance of a flow function which gens these facts
46        */
47       public static VectorGenFlowFunction make(IntSet gen) {
48         if (gen == null) {
49           throw new IllegalArgumentException("null gen");
50         }
51         return new VectorGenFlowFunction(gen);
52       }
53     
54       @Override
55       public String toString() {
56         return "VectorGen: " + gen;
57       }
58     }
