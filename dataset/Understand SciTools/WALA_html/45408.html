62       static class UnknownAttributeException extends Exception {
63         private static final long serialVersionUID = 8845177787110364793L;
64     
65         UnknownAttributeException(String t) {
66           super("Attribute '" + t + "' not understood");
67         }
68       }

70       public static void main(String[] args) throws Exception {
71         if (args == null || args.length == 0) {
72           System.err.println(USAGE);
73           System.exit(1);
74         }
75     
76         for (int i = 0; i < args.length - 1; i++) {
77           if (args[i] == null) {
78             throw new IllegalArgumentException("args[" + i + "] is null");
79           }
80           if (args[i].equals("-c")) {
81             copyright = "\u00A9 Copyright " + args[i + 1];
82             String[] newArgs = new String[args.length - 2];
83             System.arraycopy(args, 0, newArgs, 0, i);
84             System.arraycopy(args, i + 2, newArgs, i, newArgs.length - i);
85             args = newArgs;
86             break;
87           }
88         }
89         if (copyright == null) {
90           System.err.println(USAGE);
91           System.exit(1);
92         }
93     
94         final ArrayList<ZipEntry> entries = new ArrayList<ZipEntry>();
95     
96         instrumenter = new OfflineInstrumenter(true);
97         instrumenter.setManifestBuilder(new OfflineInstrumenter.ManifestBuilder() {
98           @Override
99           public void addEntry(ZipEntry ze) {
100             entries.add(ze);
101           }
102         });
103         instrumenter.parseStandardArgs(args);
104         instrumenter.setJARComment(copyright);
105         instrumenter.beginTraversal();
106         ClassInstrumenter ci;
107         CopyWriter cw = new CopyWriter();
108         while ((ci = instrumenter.nextClass()) != null) {
109           try {
110             cw.doClass(ci);
111           } catch (UnknownAttributeException ex) {
112             System.err.println(ex.getMessage() + " in " + instrumenter.getLastClassResourceName());
113           }
114         }
115     
116         instrumenter.writeUnmodifiedClasses();
117     
118         Writer w = new OutputStreamWriter(instrumenter.addOutputJarEntry(new ZipEntry("IBM-Copyright")));
119         w.write(copyright + "\n");
120         for (Iterator<ZipEntry> iter = entries.iterator(); iter.hasNext();) {
121           ZipEntry ze = iter.next();
122           w.write("  " + ze.getName() + "\n");
123         }
124         w.write(copyright + "\n");
125         w.flush();
126         instrumenter.endOutputJarEntry();
127     
128         instrumenter.close();
129       }

131       private int transformCPIndex(int i) {
132         if (i == replace) {
133           return replaceWith;
134         } else {
135           return i;
136         }
137       }

139       private Element transformAttribute(ClassReader cr, int m, ClassWriter w, AttrIterator iter) throws InvalidClassFileException,
140           UnknownAttributeException, InvalidBytecodeException {
141         String name = iter.getName();
142     
143         boolean needTransform = true;
144         if (name.equals("Synthetic") || name.equals("Deprecated") || name.equals("LineNumberTable")) {
145           needTransform = false;
146         }
147     
148         int offset = iter.getRawOffset();
149         int end = offset + iter.getRawSize();
150     
151         if (needTransform) {
152           needTransform = false;
153           for (int i = offset; i + 1 < end; i++) {
154             if (cr.getUShort(i) == replace) {
155               break;
156             }
157           }
158         }
159     
160         if (!needTransform) {
161           return new ClassWriter.RawElement(cr.getBytes(), offset, end - offset);
162         }
163     
164         if (name.equals("Code")) {
165           CodeReader r = new CodeReader(iter);
166           CTDecoder decoder = new CTDecoder(r);
167           decoder.decode();
168           MethodData md = new MethodData(decoder, cr.getMethodAccessFlags(m), CTDecoder.convertClassToType(cr.getName()), cr
169               .getMethodName(m), cr.getMethodType(m));
170           CTCompiler compiler = CTCompiler.make(w, md);
171           compiler.compile();
172           if (compiler.getAuxiliaryMethods().length > 0)
173             throw new Error("Where did this auxiliary method come from?");
174           Compiler.Output out = compiler.getOutput();
175           CodeWriter cw = new CodeWriter(w);
176           cw.setMaxLocals(out.getMaxLocals());
177           cw.setMaxStack(out.getMaxStack());
178           cw.setCode(out.getCode());
179           cw.setRawHandlers(out.getRawHandlers());
180           ClassReader.AttrIterator iterator = new ClassReader.AttrIterator();
181           r.initAttributeIterator(iterator);
182           cw.setAttributes(collectAttributes(cr, m, w, iterator));
183           return cw;
184         } else if (name.equals("ConstantValue")) {
185           ConstantValueReader r = new ConstantValueReader(iter);
186           ConstantValueWriter cw = new ConstantValueWriter(w);
187           cw.setValueCPIndex(transformCPIndex(r.getValueCPIndex()));
188           return cw;
189         } else if (name.equals("SourceFile")) {
190           SourceFileReader r = new SourceFileReader(iter);
191           SourceFileWriter cw = new SourceFileWriter(w);
192           cw.setSourceFileCPIndex(transformCPIndex(r.getSourceFileCPIndex()));
193           return cw;
194         } else if (name.equals("LocalVariableTableReader")) {
195           LocalVariableTableReader lr = new LocalVariableTableReader(iter);
196           LocalVariableTableWriter lw = new LocalVariableTableWriter(w);
197           int[] table = lr.getRawTable();
198           for (int i = 0; i < table.length; i += 5) {
199             table[i + 2] = transformCPIndex(table[i + 2]);
200             table[i + 3] = transformCPIndex(table[i + 3]);
201           }
202           lw.setRawTable(table);
203           return lw;
204         } else if (name.equals("Exceptions")) {
205           ExceptionsReader lr = new ExceptionsReader(iter);
206           ExceptionsWriter lw = new ExceptionsWriter(w);
207           int[] table = lr.getRawTable();
208           for (int i = 0; i < table.length; i++) {
209             table[i] = transformCPIndex(table[i]);
210           }
211           lw.setRawTable(table);
212           return lw;
213         } else if (name.equals("InnerClasses")) {
214           InnerClassesReader lr = new InnerClassesReader(iter);
215           InnerClassesWriter lw = new InnerClassesWriter(w);
216           int[] table = lr.getRawTable();
217           for (int i = 0; i < table.length; i += 4) {
218             table[i] = transformCPIndex(table[i]);
219             table[i + 1] = transformCPIndex(table[i + 1]);
220             table[i + 2] = transformCPIndex(table[i + 2]);
221           }
222           lw.setRawTable(table);
223           return lw;
224         }
225     
226         throw new UnknownAttributeException(name);
227       }

229       private Element[] collectAttributes(ClassReader cr, int m, ClassWriter w, AttrIterator iter) throws InvalidClassFileException,
230           UnknownAttributeException, InvalidBytecodeException {
231         Element[] elems = new Element[iter.getRemainingAttributesCount()];
232         for (int i = 0; i < elems.length; i++) {
233           elems[i] = transformAttribute(cr, m, w, iter);
234           iter.advance();
235         }
236         return elems;
237       }

239       private int copyEntry(ConstantPoolParser cp, ClassWriter w, int i) throws InvalidClassFileException {
240         byte t = cp.getItemType(i);
241         switch (t) {
242         case ClassReader.CONSTANT_String:
243           return w.addCPString(cp.getCPString(i));
244         case ClassReader.CONSTANT_Class:
245           return w.addCPClass(cp.getCPClass(i));
246         case ClassReader.CONSTANT_FieldRef:
247           return w.addCPFieldRef(cp.getCPRefClass(i), cp.getCPRefName(i), cp.getCPRefType(i));
248         case ClassReader.CONSTANT_InterfaceMethodRef:
249           return w.addCPInterfaceMethodRef(cp.getCPRefClass(i), cp.getCPRefName(i), cp.getCPRefType(i));
250         case ClassReader.CONSTANT_MethodRef:
251           return w.addCPMethodRef(cp.getCPRefClass(i), cp.getCPRefName(i), cp.getCPRefType(i));
252         case ClassReader.CONSTANT_NameAndType:
253           return w.addCPNAT(cp.getCPNATName(i), cp.getCPNATType(i));
254         case ClassReader.CONSTANT_Integer:
255           return w.addCPInt(cp.getCPInt(i));
256         case ClassReader.CONSTANT_Float:
257           return w.addCPFloat(cp.getCPFloat(i));
258         case ClassReader.CONSTANT_Long:
259           return w.addCPLong(cp.getCPLong(i));
260         case ClassReader.CONSTANT_Double:
261           return w.addCPDouble(cp.getCPDouble(i));
262         case ClassReader.CONSTANT_Utf8:
263           return w.addCPUtf8(cp.getCPUtf8(i));
264         }
265         return -1;
266       }

268       private void doClass(final ClassInstrumenter ci) throws Exception {
269         /*
270          * Our basic strategy is to make the first element of the constant pool be the copyright string (as a UTF8 constant pool item).
271          * This requires us to parse and emit any class data which might refer to that constant pool item (#1). We will assume that any
272          * attribute which refers to that constant pool item must contain the byte sequence '00 01', so we can just copy over any
273          * attributes which don't contain that byte sequence. If we detect an unknown attribute type containing the sequence '00 01',
274          * then we will abort.
275          */
276         ClassReader cr = ci.getReader();
277         ClassWriter w = new ClassWriter();
278     
279         // Make sure that when we're moving over the old constant pool,
280         // anytime we add a new entry it really is added and we don't just
281         // reuse an existing entry.
282         w.setForceAddCPEntries(true);
283     
284         // Make the first string in the constant pool be the copyright string
285         int r = w.addCPUtf8(copyright);
286         if (r != 1)
287           throw new Error("Invalid constant pool index: " + r);
288     
289         // Now add the rest of the CP entries
290         ConstantPoolParser cp = cr.getCP();
291         int CPCount = cp.getItemCount();
292     
293         if (1 < CPCount) {
294           switch (cp.getItemType(1)) {
295           case ClassReader.CONSTANT_Long:
296           case ClassReader.CONSTANT_Double:
297             // item 1 is a double-word item, so the next real item is at 3
298             // to make sure item 3 is allocated at index 3, we'll need to
299             // insert a dummy entry at index 2
300             r = w.addCPUtf8("");
301             if (r != 2)
302               throw new Error("Invalid constant pool index for dummy: " + r);
303             break;
304           }
305         }
306         for (int i = 2; i < CPCount; i++) {
307           r = copyEntry(cp, w, i);
308           if (r != -1 && r != i)
309             throw new Error("Invalid constant pool index allocated: " + r + ", expected " + i);
310         }
311         w.setForceAddCPEntries(false);
312     
313         // add CP entry we replaced
314         replaceWith = copyEntry(cp, w, 1);
315         replace = 1;
316     
317         // emit class
318         w.setMajorVersion(cr.getMajorVersion());
319         w.setMinorVersion(cr.getMinorVersion());
320         w.setAccessFlags(cr.getAccessFlags());
321         w.setName(cr.getName());
322         w.setSuperName(cr.getSuperName());
323         w.setInterfaceNames(cr.getInterfaceNames());
324     
325         ClassReader.AttrIterator iter = new ClassReader.AttrIterator();
326     
327         int fieldCount = cr.getFieldCount();
328         for (int i = 0; i < fieldCount; i++) {
329           cr.initFieldAttributeIterator(i, iter);
330           w.addField(cr.getFieldAccessFlags(i), cr.getFieldName(i), cr.getFieldType(i), collectAttributes(cr, i, w, iter));
331         }
332     
333         int methodCount = cr.getMethodCount();
334         for (int i = 0; i < methodCount; i++) {
335           cr.initMethodAttributeIterator(i, iter);
336           w.addMethod(cr.getMethodAccessFlags(i), cr.getMethodName(i), cr.getMethodType(i), collectAttributes(cr, i, w, iter));
337         }
338     
339         cr.initClassAttributeIterator(iter);
340         for (; iter.isValid(); iter.advance()) {
341           w.addClassAttribute(transformAttribute(cr, 0, w, iter));
342         }
343     
344         instrumenter.outputModifiedClass(ci, w);
345       }
