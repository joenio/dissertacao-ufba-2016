342       private CAstNode makeCall(CAstNode fun, CAstNode thisptr, CAstNode args[], WalkContext context, String callee) {
343         int children = (args == null)? 0 : args.length;
344     
345         // children of CAst CALL node are the expression that evaluates to the
346         // function, followed by a name (either STANDARD_CALL_FN_NAME or
347         // CTOR_CALL_FN_NAME), followed by the actual
348         // parameters
349         int nargs = (thisptr == null) ? children + 2 : children + 3;
350         int i = 0;
351         CAstNode arguments[] = new CAstNode[nargs];
352         arguments[i++] = fun;
353         // assert callee.equals(STANDARD_CALL_FN_NAME) || callee.equals(CTOR_CALL_FN_NAME);
354         arguments[i++] = Ast.makeConstant(callee);
355         if (thisptr != null)
356           arguments[i++] = thisptr;
357         if (args != null) {
358         	for (CAstNode arg : args) {
359         		arguments[i++] = arg;
360         	}
361         }
362         
363         CAstNode call = Ast.makeNode(CAstNode.CALL, arguments);
364     
365         context.cfg().map(call, call);
366         if (context.getCatchTarget() != null) {
367           context.cfg().add(call, context.getCatchTarget(), null);
368         }
369     
370         return call;
371       }

509       private CAstEntity walkEntity(final AstNode n, List<CAstNode> body, String name, WalkContext child) {
510         CAstNode[] stmts = body.toArray(new CAstNode[body.size()]);
511     
512         // add variable / constant / function declarations, if any
513         if (!child.getNameDecls().isEmpty()) {
514           // new first statement will be a block declaring all names.
515           CAstNode[] newStmts = new CAstNode[stmts.length + 1];
516     
517           if (child.getNameDecls().size() == 1) {
518             newStmts[0] = child.getNameDecls().iterator().next();
519           } else {
520             newStmts[0] = Ast.makeNode(CAstNode.BLOCK_STMT, child.getNameDecls().toArray(new CAstNode[child.getNameDecls().size()]));
521           }
522           System.arraycopy(stmts, 0, newStmts, 1, stmts.length);
523     
524           stmts = newStmts;
525         }
526     
527         final CAstNode ast = noteSourcePosition(child, Ast.makeNode(CAstNode.BLOCK_STMT, stmts), n);
528         final CAstControlFlowMap map = child.cfg();
529         final CAstSourcePositionMap pos = child.pos();
530     
531         // not sure if we need this copy --MS
532         final Map<CAstNode, Collection<CAstEntity>> subs = HashMapFactory.make(child.getScopedEntities());
533         
534         return new ScriptOrFnEntity(n, subs, ast, map, pos, name);
535       }

553     	  if (context.pos().getPosition(n) == null && !(n.getKind()==CAstNode.FUNCTION_EXPR || n.getKind()==CAstNode.FUNCTION_STMT)) {
554     	      context.pos().setPosition(n, p);
555     	      for(int i = 0; i < n.getChildCount(); i++) {
556     	    	  pushSourcePosition(context, n.getChild(i), p);

558     	  }
