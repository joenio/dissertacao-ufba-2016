39     public class LoadFileTargetSelector implements MethodTargetSelector {
40     
41       private final MethodTargetSelector base;
42     
43       private final JSSSAPropagationCallGraphBuilder builder;
44       
45       private final TypeReference loadFileRef = TypeReference.findOrCreate(JavaScriptTypes.jsLoader, TypeName.string2TypeName("Lprologue.js/loadFile"));
46       
47       private final MethodReference loadFileFunRef = AstMethodReference.fnReference(loadFileRef);
48       
49       private final HashSet<URL> loadedFiles = HashSetFactory.make();
50     
51       @Override
52       public IMethod getCalleeTarget(CGNode caller, CallSiteReference site, IClass receiver) {
53         IMethod target = base.getCalleeTarget(caller, site, receiver);
54         if (target != null && target.getReference().equals(loadFileFunRef)) {
55           
56           Set<String> names = new HashSet<String>();
57           SSAInstruction call = caller.getIR().getInstructions()[caller.getIR().getCallInstructionIndices(site).intIterator().next()];
58           LocalPointerKey fileNameV = new LocalPointerKey(caller, call.getUse(1));
59           OrdinalSet<InstanceKey> ptrs = builder.getPointerAnalysis().getPointsToSet(fileNameV);
60           for(InstanceKey k : ptrs) {
61             if (k instanceof ConstantKey) {
62               Object v = ((ConstantKey)k).getValue();
63               if (v instanceof String) {
64                 names.add((String)v);
65               }
66             }
67           }
68           
69           if (names.size() == 1) {
70             String str = names.iterator().next();
71             try {
72               JavaScriptLoader cl = (JavaScriptLoader) builder.getClassHierarchy().getLoader(JavaScriptTypes.jsLoader);
73               URL url = new URL(builder.getBaseURL(), str);
74               if(!loadedFiles.contains(url)) {
75                 // try to open the input stream for the URL.  if it fails, we'll get an IOException and fall through to default case
76                 InputStream inputStream = url.openConnection().getInputStream();
77                 inputStream.close();
78                 JSCallGraphUtil.loadAdditionalFile(builder.getClassHierarchy() , cl, str, url);
79                 loadedFiles.add(url);
80                 IClass script = builder.getClassHierarchy().lookupClass(TypeReference.findOrCreate(cl.getReference(), "L" + url.getFile()));
81                 return script.getMethod(JavaScriptMethods.fnSelector);
82               }
83             } catch (MalformedURLException e1) {
84               // do nothing, fall through and return 'target'
85             } catch (IOException e) {
86               // do nothing, fall through and return 'target'
87             } catch (RuntimeException e) {
88               // do nothing, fall through and return 'target'
89             }
90           }
91         }
92         
93         return target;
94       }
95     
96       public LoadFileTargetSelector(MethodTargetSelector base, JSSSAPropagationCallGraphBuilder builder) {
97         super();
98         this.base = base;
99         this.builder = builder;
100       }
101     
102     }
