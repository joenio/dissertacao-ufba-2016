11     package com.ibm.wala.shrikeCT;
12     
13     /**
14      * This class reads Exceptions attributes.
15      */
16     public final class ExceptionsReader extends AttributeReader {
17       /**
18        * Build a reader for the attribute 'iter'.
19        */
20       public ExceptionsReader(ClassReader.AttrIterator iter) throws InvalidClassFileException {
21         super(iter, "Exceptions");
22     
23         checkSize(attr, 8);
24         int count = cr.getUShort(attr + 6);
25         checkSizeEquals(attr + 8, 2 * count);
26       }
27     
28       /**
29        * @return the indices of the constant pool items for the exceptions
30        */
31       public int[] getRawTable() {
32         int count = cr.getUShort(attr + 6);
33         int[] r = new int[count];
34         for (int i = 0; i < r.length; i++) {
35           r[i] = cr.getUShort(attr + 8 + i * 2);
36         }
37         return r;
38       }
39     
40       /**
41        * @return the classes of exceptions that can be thrown by the method
42        */
43       public String[] getClasses() throws InvalidClassFileException {
44         int count = cr.getUShort(attr + 6);
45         String[] r = new String[count];
46         ConstantPoolParser cp = cr.getCP();
47         for (int i = 0; i < r.length; i++) {
48           r[i] = cp.getCPClass(cr.getUShort(attr + 8 + i * 2));
49         }
50         return r;
51       }
52     }
