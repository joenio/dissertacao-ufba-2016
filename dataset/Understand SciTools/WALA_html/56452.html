11     package com.ibm.wala.dataflow.IFDS;
12     
13     import com.ibm.wala.util.intset.SparseIntSet;
14     
15     /**
16      * A binary flow function corresponding to a return statements combining information from the call site and the exit site.
17      * 
18      * This function should be pairwise distributive for use with the Tabulation algorithm.
19      * 
20      * SJF: I have made this extend IFlowFunction to minimize damage to the extant class hierarchy. But calling super.getTargets() will
21      * be a problem, so be very careful in how you implement and use this. The Tabulation solver will do the right thing.
22      */
23     public interface IBinaryReturnFlowFunction extends IFlowFunction {
24     
25       /**
26        * @param call_d factoid of the caller at the call site
27        * @param exit_d factoid of the callee at the exit site
28        * @return set of ret_d such that (<call_d, exit_d>, ret_d) is an edge in this distributive function's graph representation, or
29        *         null if there are none
30        */
31       public SparseIntSet getTargets(int call_d, int exit_d);
32     }
