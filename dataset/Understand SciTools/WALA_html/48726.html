11     package com.ibm.wala.shrikeBT.shrikeCT;
12     
13     import java.util.Random;
14     
15     import com.ibm.wala.shrikeBT.Compiler;
16     import com.ibm.wala.shrikeBT.MethodData;
17     import com.ibm.wala.shrikeCT.ClassWriter;
18     import com.ibm.wala.shrikeCT.ConstantPoolParser.ReferenceToken;
19     
20     /**
21      * This class lets you compile ShrikeBT intermediate code into real Java bytecodes using ShrikeCT.
22      */
23     final public class CTCompiler extends Compiler {
24       final private ClassWriter cw;
25     
26       final private Random random = new Random();
27     
28       /**
29        * Compile 'md' into the class given by 'cw'.
30        */
31       private CTCompiler(ClassWriter cw, MethodData md) {
32         super(md);
33         this.cw = cw;
34       }
35     
36       @Override
37       protected int allocateConstantPoolInteger(int v) {
38         return cw.addCPInt(v);
39       }
40     
41       @Override
42       protected int allocateConstantPoolFloat(float v) {
43         return cw.addCPFloat(v);
44       }
45     
46       @Override
47       protected int allocateConstantPoolLong(long v) {
48         return cw.addCPLong(v);
49       }
50     
51       @Override
52       protected int allocateConstantPoolDouble(double v) {
53         return cw.addCPDouble(v);
54       }
55     
56       @Override
57       protected int allocateConstantPoolString(String v) {
58         return cw.addCPString(v);
59       }
60     
61       @Override
62       protected int allocateConstantPoolClassType(String c) {
63         return cw.addCPClass(convertTypeToClass(c));
64       }
65     
66       @Override
67       protected int allocateConstantPoolMethodType(String c) {
68         return cw.addCPMethodType(c);
69       }
70     
71       @Override
72       protected int allocateConstantPoolMethodHandle(ReferenceToken c) {
73         return cw.addCPMethodHandle(c);
74       }
75     
76       /**
77        * Convert a JVM type to the internal JVM class name (e.g., Ljava/lang/Object; to java/lang/Object)
78        * 
79        * @throws IllegalArgumentException if s is null
80        */
81       public static String convertTypeToClass(String s) {
82         if (s == null) {
83           throw new IllegalArgumentException("s is null");
84         }
85         if (s.length() > 0 && s.charAt(0) == 'L') {
86           return s.substring(1, s.length() - 1);
87         } else {
88           return s;
89         }
90       }
91     
92       public static CTCompiler make(ClassWriter cw, MethodData md) {
93         if (md == null) {
94           throw new IllegalArgumentException("md is null");
95         }
96         return new CTCompiler(cw, md);
97       }
98     
99       @Override
100       protected int allocateConstantPoolField(String c, String name, String type) {
101         return cw.addCPFieldRef(convertTypeToClass(c), name, type);
102       }
103     
104       @Override
105       protected int allocateConstantPoolMethod(String c, String name, String sig) {
106         return cw.addCPMethodRef(convertTypeToClass(c), name, sig);
107       }
108     
109       @Override
110       protected int allocateConstantPoolInterfaceMethod(String c, String name, String sig) {
111         return cw.addCPInterfaceMethodRef(convertTypeToClass(c), name, sig);
112       }
113     
114       @Override
115       protected String createHelperMethod(boolean isStatic, String sig) {
116         long r = Math.abs(random.nextLong());
117         String name = "_helper_" + r;
118     
119         return name;
120       }
121     }
