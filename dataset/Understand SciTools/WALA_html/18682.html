73     	public static void loadActivityModel() {
74     		if (ActivityModelM == null) {
75     			DexFile activityModelDF;
76     			ClassDefItem activityModelCDI = null;
77     			try {
78     				FileProvider fp = new FileProvider();
79     				File apkFile = fp.getFile("models/ActivityModel.apk", ActivityModelMethod.class.getClassLoader());
80     				activityModelDF = new DexFile(apkFile);
81     						/* new DexFile(new File(
82     						ActivityModelMethod.class.getClassLoader()
83     								.getResource("models/ActivityModel.apk")
84     								.toURI())); */
85     			} catch (Exception e) {
86     				throw new IllegalArgumentException(e);
87     			}
88     			@SuppressWarnings("unchecked")
89     			Section<ClassDefItem> cldeff = activityModelDF.getSectionForType(TYPE_CLASS_DEF_ITEM);
90     			for (ClassDefItem cdefitems : cldeff.getItems()) {
91     				if (cdefitems.getClassType().getTypeDescriptor()
92     						.equals("Lactivity/model/ActivityModelActivity;")) {
93     					activityModelCDI = cdefitems;
94     				}
95     			}
96     			assert (activityModelCDI != null);
97     			// final EncodedMethod[] virtualMethods =
98     			// ActivityModelCDI.getClassData().getVirtualMethods();
99     			for (EncodedMethod virtualMethod : activityModelCDI.getClassData()
100     					.getVirtualMethods()) {
101     				if (virtualMethod.method.getMethodName().getStringValue()
102     						.equals("ActivityModel"))
103     					ActivityModelM = virtualMethod;
104     			}
105     			assert (ActivityModelM != null);
106     		}
107     	}
108     
109     	public static EncodedMethod getActivityModel() {
110     		if (ActivityModelM == null) {
111     			loadActivityModel();
112     		}
113     		return ActivityModelM;
114     	}
