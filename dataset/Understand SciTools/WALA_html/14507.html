15     /**
16      * Simple utilities for Eclipse progress monitors
17      */
18     public class MonitorUtil {
19       
20       /**
21        * Use this interface to decouple core utilities from the Eclipse layer
22        */
23       public interface IProgressMonitor {
24     
25         /** Constant indicating an unknown amount of work. */
26         public static final int UNKNOWN = -1;
27     
28         void beginTask(String task, int totalWork);
29     
30     /** BEGIN Custom change: subtasks and canceling */
31         void subTask(String subTask);
32     
33         void cancel();
34     
35     /** END Custom change: subtasks and canceling */
36         boolean isCanceled();
37     
38         void done();
39     
40         void worked(int units);
41         
42         String getCancelMessage();
43       }
44     
45       public static void beginTask(IProgressMonitor monitor, String task, int totalWork) throws CancelException {
46         if (monitor != null) {
47           monitor.beginTask(task, totalWork);
48           if (monitor.isCanceled()) {
49             throw CancelException.make("cancelled in " + task);
50           }
51         }
52       }
53     
54       public static void done(IProgressMonitor monitor) throws CancelException {
55         if (monitor != null) {
56           monitor.done();
57           if (monitor.isCanceled()) {
58             throw CancelException.make("cancelled in " + monitor.toString());
59           }
60         }
61       }
62     
63       public static void worked(IProgressMonitor monitor, int units) throws CancelException {
64         if (monitor != null) {
65           monitor.worked(units);
66           if (monitor.isCanceled()) {
67             throw CancelException.make("cancelled in " + monitor.toString());
68           }
69         }
70       }
71     
72       public static void throwExceptionIfCanceled(IProgressMonitor progressMonitor) throws CancelException {
73         if (progressMonitor != null) {
74           if (progressMonitor.isCanceled()) {
75             throw CancelException.make(progressMonitor.getCancelMessage());
76           }
77         }
78       }
79     /** BEGIN Custom change: more on subtasks */
80     
81       public static void subTask(IProgressMonitor progressMonitor, String subTask) throws CancelException {
82         if (progressMonitor != null) {
83           progressMonitor.subTask(subTask);
84           if (progressMonitor.isCanceled()) {
85             throw CancelException.make("cancelled in " + subTask);
86           }
87         }
88       }
89     
90       public static boolean isCanceled(IProgressMonitor progressMonitor) {
91         if (progressMonitor == null) {
92           return false;
93         } else {
94           return progressMonitor.isCanceled();
95         }
96       }
97       
98       public static void cancel(IProgressMonitor progress) {
99         if (progress != null) {
100           progress.cancel();
101         }
102       }
103     /** END Custom change: more on subtasks */
