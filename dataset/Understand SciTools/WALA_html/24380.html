36     /**
37      * Call graph builder for building pessimistic call graphs, where inter-procedural flows are not
38      * tracked except in the trivial case of local calls. This builder is fast, but in general less
39      * sound than {@link OptimisticCallgraphBuilder}.
40      * 
41      * @author mschaefer
42      *
43      */
44     public class PessimisticCallGraphBuilder extends FieldBasedCallGraphBuilder {
45     	public PessimisticCallGraphBuilder(IClassHierarchy cha,	AnalysisOptions options, AnalysisCache cache, boolean supportFullPointerAnalysis) {
46     		super(cha, options, cache, supportFullPointerAnalysis);
47     	}
48     
49     	@Override
50     	public FlowGraph buildFlowGraph(IProgressMonitor monitor) {
51     	  FlowGraph flowgraph = flowGraphFactory();
52     	  resolveLocalCalls(flowgraph);
53     		return flowgraph;
54     	}
55     
56     	protected boolean filterFunction(IMethod function) {
57     	  return function.getDescriptor().equals(AstMethodReference.fnDesc);
58     	}
59     	
60     	// add inter-procedural flow for local calls
61     	private void resolveLocalCalls(FlowGraph flowgraph) {
62     		for(IClass klass : cha) {
63     			for(IMethod method : klass.getDeclaredMethods()) {
64     				if (filterFunction(method)) {
65     					IR ir = cache.getIR(method);
66     					ir.visitAllInstructions(new LocalCallSSAVisitor(method, ir.getSymbolTable(), cache.getDefUse(ir), flowgraph));
67     				}
68     			}
69     		}
70     	}
71     
72     	/**
73     	 * This visitor looks for calls where the callee can be determined locally by a def-use graph, and adds
74     	 * inter-procedural edges.
75     	 * 
76     	 * @author mschaefer
77     	 *
78     	 */
79     	private class LocalCallSSAVisitor extends JSMethodInstructionVisitor {
80     		private final FlowGraph flowgraph;
81     		private final VertexFactory factory;
82     		private final FuncVertex caller;
83     		
84     		public LocalCallSSAVisitor(IMethod method, SymbolTable symtab, DefUse du, FlowGraph flowgraph) {
85     			super(method, symtab, du);
86     			this.flowgraph = flowgraph;
87     			this.factory = flowgraph.getVertexFactory();
88     			this.caller = this.factory.makeFuncVertex(method.getDeclaringClass());
89     		}
90     		
91     		@Override
92     		public void visitJavaScriptInvoke(JavaScriptInvoke invk) {
93     			// check whether this instruction corresponds to a function expression/declaration
94     			if(isFunctionConstructorInvoke(invk)) {
95     				int defn = invk.getDef();
96     				
97     				// the name of the function
98     				String fnName = symtab.getStringValue(invk.getUse(1));
99     				IClass fnClass = cha.lookupClass(TypeReference.findOrCreate(JavaScriptTypes.jsLoader, fnName));
100             if (fnClass == null) {
101               System.err.println("cannot find " + fnName + " at " +  ((AstMethod)method).getSourcePosition());
102               return;
103             }
104     				IMethod fn = fnClass.getMethod(AstMethodReference.fnSelector);
105     				FuncVertex callee = factory.makeFuncVertex(fnClass);
106     				
107     				// look at all uses
108     				for(Iterator<SSAInstruction> uses = du.getUses(defn);uses.hasNext();) {
109     					SSAInstruction use = uses.next();
110     					
111     					// check whether this is a local call
112     					if(use instanceof JavaScriptInvoke && ((JavaScriptInvoke)use).getFunction() == defn) {
113     						JavaScriptInvoke use_invk = (JavaScriptInvoke)use;
114     						
115     						// yes, so add edges from arguments to parameters...
116     						for(int i=2;i<use_invk.getNumberOfParameters();++i)
117     							flowgraph.addEdge(factory.makeVarVertex(caller, use_invk.getUse(i)), factory.makeParamVertex(callee, i));
118     						
119     						// ...and from return to result
120     						flowgraph.addEdge(factory.makeRetVertex(callee), factory.makeVarVertex(caller, use.getDef()));
121     						
122     						// note: local calls are never qualified, so there is no flow into the receiver vertex
123     					} else {
124     						// no, it's a more complicated use, so add flows from/to unknown
125     						for(int i=1;i<fn.getNumberOfParameters();++i)
126     							flowgraph.addEdge(factory.makeUnknownVertex(), factory.makeParamVertex(callee, i));
127     						flowgraph.addEdge(factory.makeRetVertex(callee), factory.makeUnknownVertex());
128     					}
129     				}
130     			} else {
131     				// this is a genuine function call; find out where the function came from
132     				SSAInstruction def = du.getDef(invk.getFunction());
133     				
134     				// if it's not a local call, add flows from/to unknown
135     				if(!(def instanceof JavaScriptInvoke) || !isFunctionConstructorInvoke((JavaScriptInvoke)def)) {
136     					for(int i=1;i<invk.getNumberOfParameters();++i)
137     						flowgraph.addEdge(factory.makeVarVertex(caller, invk.getUse(i)), factory.makeUnknownVertex());
138     					flowgraph.addEdge(factory.makeUnknownVertex(), factory.makeVarVertex(caller, invk.getDef()));
139     				}
140     			}
141     		}
142     	}
143     }
