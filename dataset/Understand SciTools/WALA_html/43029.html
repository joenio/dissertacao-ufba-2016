11     package com.ibm.wala.cast.java.ssa;
12     
13     import java.util.Collection;
14     
15     import com.ibm.wala.classLoader.JavaLanguage;
16     import com.ibm.wala.classLoader.NewSiteReference;
17     import com.ibm.wala.ssa.SSAInstruction;
18     import com.ibm.wala.ssa.SSAInstructionFactory;
19     import com.ibm.wala.ssa.SSANewInstruction;
20     import com.ibm.wala.types.TypeReference;
21     
22     // A new instruction with an explicit outer class, i.e. "Inner inner = outer.new Inner();"
23     public class AstJavaNewEnclosingInstruction extends SSANewInstruction {
24     
25       int enclosing;
26       
27       @Override
28       public int getNumberOfUses() {
29         return 1;
30       }
31       
32       @Override
33       public int getUse(int i) {
34         assert i == 0;
35         return enclosing;
36       }
37       
38       public AstJavaNewEnclosingInstruction(int iindex, int result, NewSiteReference site, int enclosing) throws IllegalArgumentException {
39         super(iindex, result, site);
40         this.enclosing = enclosing;
41       }
42       
43       public int getEnclosing() {
44         return this.enclosing;
45       }
46       
47       @Override
48       public String toString() {
49         return super.toString() + " ENCLOSING v" + enclosing;
50       }
51     
52       @Override
53       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
54         return ((AstJavaInstructionFactory)insts).JavaNewEnclosingInstruction(iindex, defs==null? getDef(0): defs[0], getNewSite(), uses==null? enclosing: uses[0]);
55       }
56       
57       @Override
58       public Collection<TypeReference> getExceptionTypes() {
59         return JavaLanguage.getNewScalarExceptions();
60       }
61     
62     }
