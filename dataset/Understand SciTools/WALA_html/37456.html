31     /**
32      * Simple extension to JUnit test case.
33      */
34     public abstract class WalaTestCase {
35     
36       final private static boolean ANALYZE_LEAKS = false;
37     
38       public static boolean useShortProfile() {
39         String profile = System.getProperty("com.ibm.wala.junit.profile", "long");
40         if (profile.equals("short")) {
41           return true;
42         } else {
43           return false;
44         }
45       }
46     
47       public static boolean analyzingJar() {
48         return "true".equals(System.getProperty("com.ibm.wala.junit.analyzingJar"));
49       }
50       /*
51        * @see junit.framework.TestCase#setUp()
52        */
53       @Before
54       public void setUp() throws Exception {
55       }
56     
57       /*
58        * @see junit.framework.TestCase#tearDown()
59        */
60       @After
61       public void tearDown() throws Exception {
62         Warnings.clear();
63         if (ANALYZE_LEAKS) {
64           HeapTracer.analyzeLeaks();
65         }
66       }
67     
68       protected AnalysisCache makeAnalysisCache() {
69         return new AnalysisCache();
70       }
71     
72       public static IClassHierarchy makeCHA() throws IOException, ClassHierarchyException {
73         AnalysisScope scope = AnalysisScopeReader.readJavaScope(TestConstants.WALA_TESTDATA,
74             (new FileProvider()).getFile(CallGraphTestUtil.REGRESSION_EXCLUSIONS), AnnotationTest.class.getClassLoader());
75         return ClassHierarchy.make(scope);    
76       }
77     
78       /**
79        * Utility function: each DetoxTestCase subclass can have a main() method that calls this, to create a test suite consisting of
80        * just this test. Useful when investigating a single failing test.
81        */
82       protected static void justThisTest(Class<?> testClass) {
83         JUnitCore.runClasses(testClass);
84       }
85     
86     }
