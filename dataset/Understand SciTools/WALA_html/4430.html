80       /**
81        * Usage: PDFSlice -appJar [jar file name] -mainClass [main class] -srcCaller [method name] -srcCallee [method name] -dd [data
82        * dependence options] -cd [control dependence options] -dir [forward|backward]
83        * 
84        * <ul>
85        * <li>"jar file name" should be something like "c:/temp/testdata/java_cup.jar"
86        * <li>"main class" should beshould be something like "c:/temp/testdata/java_cup.jar"
87        * <li>"method name" should be the name of a method. This takes a slice from the statement that calls "srcCallee" from "srcCaller"
88        * <li>"data dependence options" can be one of "-full", "-no_base_ptrs", "-no_base_no_heap", "-no_heap",
89        * "-no_base_no_heap_no_cast", or "-none".
90        * </ul>
91        * 
92        * @throws CancelException
93        * @throws IllegalArgumentException
94        * @throws IOException
95        * 
96        * @see com.ibm.wala.ipa.slicer.Slicer.DataDependenceOptions <li>"control dependence options" can be "-full" or "-none" <li>the
97        *      -dir argument tells whether to compute a forwards or backwards slice. </ul>
98        * 
99        */
100       public static void main(String[] args) throws WalaException, IllegalArgumentException, CancelException, IOException {
101         run(args);

104       /**
105        * see {@link #main(String[])} for command-line arguments
106        * 
107        * @throws CancelException
108        * @throws IllegalArgumentException
109        * @throws IOException
110        */
111       public static Process run(String[] args) throws WalaException, IllegalArgumentException, CancelException, IOException {
112         // parse the command-line into a Properties object
113         Properties p = CommandLine.parse(args);
114         // validate that the command-line has the expected format
115         validateCommandLine(p);
116     
117         // run the applications
118         return run(p.getProperty("appJar"), p.getProperty("mainClass"), p.getProperty("srcCaller"), p.getProperty("srcCallee"),
119             goBackward(p), PDFSDG.getDataDependenceOptions(p), PDFSDG.getControlDependenceOptions(p));
120       }

122       /**
123        * Should the slice be a backwards slice?
124        */
125       private static boolean goBackward(Properties p) {
126         return !p.getProperty("dir", "backward").equals("forward");
127       }

129       /**
130        * Compute a slice from a call statements, dot it, and fire off the PDF viewer to visualize the result
131        * 
132        * @param appJar should be something like "c:/temp/testdata/java_cup.jar"
133        * @param mainClass should be something like "c:/temp/testdata/java_cup.jar"
134        * @param srcCaller name of the method containing the statement of interest
135        * @param srcCallee name of the method called by the statement of interest
136        * @param goBackward do a backward slice?
137        * @param dOptions options controlling data dependence
138        * @param cOptions options controlling control dependence
139        * @return a Process running the PDF viewer to visualize the dot'ted representation of the slice
140        * @throws CancelException
141        * @throws IllegalArgumentException
142        */
143       public static Process run(String appJar, String mainClass, String srcCaller, String srcCallee, boolean goBackward,
144           DataDependenceOptions dOptions, ControlDependenceOptions cOptions) throws IllegalArgumentException, CancelException,
145           IOException {
146         try {
147           // create an analysis scope representing the appJar as a J2SE application
148           AnalysisScope scope = AnalysisScopeReader.makeJavaBinaryAnalysisScope(appJar, (new FileProvider())
149               .getFile(CallGraphTestUtil.REGRESSION_EXCLUSIONS));
150     
151           // build a class hierarchy, call graph, and system dependence graph
152           ClassHierarchy cha = ClassHierarchy.make(scope);
153           Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha, mainClass);
154           AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
155           CallGraphBuilder builder = Util.makeVanillaZeroOneCFABuilder(options, new AnalysisCache(), cha, scope);
156           // CallGraphBuilder builder = Util.makeZeroOneCFABuilder(options, new
157           // AnalysisCache(), cha, scope);
158           CallGraph cg = builder.makeCallGraph(options, null);
159           SDG sdg = new SDG(cg, builder.getPointerAnalysis(), dOptions, cOptions);
160     
161           // find the call statement of interest
162           CGNode callerNode = SlicerTest.findMethod(cg, srcCaller);
163           Statement s = SlicerTest.findCallTo(callerNode, srcCallee);
164           System.err.println("Statement: " + s);
165     
166           // compute the slice as a collection of statements
167           Collection<Statement> slice = null;
168           if (goBackward) {
169             slice = Slicer.computeBackwardSlice(s, cg, builder.getPointerAnalysis(), dOptions, cOptions);
170           } else {
171             // for forward slices ... we actually slice from the return value of
172             // calls.
173             s = getReturnStatementForCall(s);
174             slice = Slicer.computeForwardSlice(s, cg, builder.getPointerAnalysis(), dOptions, cOptions);
175           }
176           SlicerTest.dumpSlice(slice);
177     
178           // create a view of the SDG restricted to nodes in the slice
179           Graph<Statement> g = pruneSDG(sdg, slice);
180     
181           sanityCheck(slice, g);
182     
183           // load Properties from standard WALA and the WALA examples project
184           Properties p = null;
185           try {
186             p = WalaExamplesProperties.loadProperties();
187             p.putAll(WalaProperties.loadProperties());
188           } catch (WalaException e) {
189             e.printStackTrace();
190             Assertions.UNREACHABLE();
191           }
192           // create a dot representation.
193           String psFile = p.getProperty(WalaProperties.OUTPUT_DIR) + File.separatorChar + PDF_FILE;
194           String dotExe = p.getProperty(WalaExamplesProperties.DOT_EXE);
195           DotUtil.dotify(g, makeNodeDecorator(), PDFTypeHierarchy.DOT_FILE, psFile, dotExe);
196     
197           // fire off the PDF viewer
198           String gvExe = p.getProperty(WalaExamplesProperties.PDFVIEW_EXE);
199           return PDFViewUtil.launchPDFView(psFile, gvExe);
200     
201         } catch (WalaException e) {
202           // something bad happened.
203           e.printStackTrace();
204           return null;
205         }
206       }

208       /**
209        * check that g is a well-formed graph, and that it contains exactly the number of nodes in the slice
210        */
211       private static void sanityCheck(Collection<Statement> slice, Graph<Statement> g) {
212         try {
213           GraphIntegrity.check(g);
214         } catch (UnsoundGraphException e1) {
215           e1.printStackTrace();
216           Assertions.UNREACHABLE();
217         }
218         Assertions.productionAssertion(g.getNumberOfNodes() == slice.size(), "panic " + g.getNumberOfNodes() + " " + slice.size());
219       }

221       /**
222        * If s is a call statement, return the statement representing the normal return from s
223        */
224       public static Statement getReturnStatementForCall(Statement s) {
225         if (s.getKind() == Kind.NORMAL) {
226           NormalStatement n = (NormalStatement) s;
227           SSAInstruction st = n.getInstruction();
228           if (st instanceof SSAInvokeInstruction) {
229             SSAAbstractInvokeInstruction call = (SSAAbstractInvokeInstruction) st;
230             if (call.getCallSite().getDeclaredTarget().getReturnType().equals(TypeReference.Void)) {
231               throw new IllegalArgumentException("this driver computes forward slices from the return value of calls.\n" + ""
232                   + "Method " + call.getCallSite().getDeclaredTarget().getSignature() + " returns void.");
233             }
234             return new NormalReturnCaller(s.getNode(), n.getInstructionIndex());
235           } else {
236             return s;
237           }
238         } else {
239           return s;
240         }
241       }

243       /**
244        * return a view of the sdg restricted to the statements in the slice
245        */
246       public static Graph<Statement> pruneSDG(SDG sdg, final Collection<Statement> slice) {
247         Predicate<Statement> f = new Predicate<Statement>() {
248           @Override public boolean test(Statement o) {
249             return slice.contains(o);
250           }
251         };
252         return GraphSlicer.prune(sdg, f);
253       }

255       /**
256        * @return a NodeDecorator that decorates statements in a slice for a dot-ted representation
257        */
258       public static NodeDecorator<Statement> makeNodeDecorator() {
259         return new NodeDecorator<Statement>() {
260           @Override
261           public String getLabel(Statement s) throws WalaException {
262             switch (s.getKind()) {
263             case HEAP_PARAM_CALLEE:
264             case HEAP_PARAM_CALLER:
265             case HEAP_RET_CALLEE:
266             case HEAP_RET_CALLER:
267               HeapStatement h = (HeapStatement) s;
268               return s.getKind() + "\\n" + h.getNode() + "\\n" + h.getLocation();
269             case NORMAL:
270               NormalStatement n = (NormalStatement) s;
271               return n.getInstruction() + "\\n" + n.getNode().getMethod().getSignature();
272             case PARAM_CALLEE:
273               ParamCallee paramCallee = (ParamCallee) s;
274               return s.getKind() + " " + paramCallee.getValueNumber() + "\\n" + s.getNode().getMethod().getName();
275             case PARAM_CALLER:
276               ParamCaller paramCaller = (ParamCaller) s;
277               return s.getKind() + " " + paramCaller.getValueNumber() + "\\n" + s.getNode().getMethod().getName() + "\\n"
278                   + paramCaller.getInstruction().getCallSite().getDeclaredTarget().getName();
279             case EXC_RET_CALLEE:
280             case EXC_RET_CALLER:
281             case NORMAL_RET_CALLEE:
282             case NORMAL_RET_CALLER:
283             case PHI:
284             default:
285               return s.toString();
286             }
287           }
288     
289         };
290       }

292       /**
293        * Validate that the command-line arguments obey the expected usage.
294        * 
295        * Usage:
296        * <ul>
297        * <li>args[0] : "-appJar"
298        * <li>args[1] : something like "c:/temp/testdata/java_cup.jar"
299        * <li>args[2] : "-mainClass"
300        * <li>args[3] : something like "Lslice/TestRecursion" *
301        * <li>args[4] : "-srcCallee"
302        * <li>args[5] : something like "print" *
303        * <li>args[4] : "-srcCaller"
304        * <li>args[5] : something like "main"
305        * </ul>
306        * 
307        * @throws UnsupportedOperationException if command-line is malformed.
308        */
309       static void validateCommandLine(Properties p) {
310         if (p.get("appJar") == null) {
311           throw new UnsupportedOperationException("expected command-line to include -appJar");
312         }
313         if (p.get("mainClass") == null) {
314           throw new UnsupportedOperationException("expected command-line to include -mainClass");
315         }
316         if (p.get("srcCallee") == null) {
317           throw new UnsupportedOperationException("expected command-line to include -srcCallee");
318         }
319         if (p.get("srcCaller") == null) {
320           throw new UnsupportedOperationException("expected command-line to include -srcCaller");
321         }
322       }
