21     /**
22      * This object will analyze a method in a context and return information needed for RTA.
23      */
24     public interface RTAContextInterpreter extends CHAContextInterpreter {
25     
26       /**
27        * @return an Iterator of the types that may be allocated by a given method in a given context.
28        */
29       public abstract Iterator<NewSiteReference> iterateNewSites(CGNode node);
30     
31       /**
32        * @return iterator of FieldReference
33        */
34       public Iterator<FieldReference> iterateFieldsRead(CGNode node);
35     
36       /**
37        * @return iterator of FieldReference
38        */
39       public Iterator<FieldReference> iterateFieldsWritten(CGNode node);
40     
41       /**
42        * record that the "factory" method of a node should be interpreted to allocate a particular class.
43        * 
44        * TODO: this is a little ugly, is there a better place to move this?
45        * 
46        * @param node
47        * @param klass
48        * @return true iff a NEW type was recorded, false if the type was previously recorded.
49        */
50       public boolean recordFactoryType(CGNode node, IClass klass);
51     }
