11     package com.ibm.wala.ipa.callgraph.propagation;
12     
13     import com.ibm.wala.classLoader.IField;
14     import com.ibm.wala.ipa.callgraph.CGNode;
15     
16     /**
17      * An object that abstracts how to model pointers in the heap.
18      */
19     public interface PointerKeyFactory {
20       /**
21        * @return the PointerKey that acts as a representative for the class of pointers that includes the local variable identified by
22        *         the value number parameter.
23        * 
24        */
25       PointerKey getPointerKeyForLocal(CGNode node, int valueNumber);
26     
27       /**
28        * @return the PointerKey that acts as a representative for the class of pointers that includes the local variable identified by
29        *         the value number parameter.
30        * 
31        */
32       FilteredPointerKey getFilteredPointerKeyForLocal(CGNode node, int valueNumber, FilteredPointerKey.TypeFilter filter);
33     
34       /**
35        * @param node
36        * @return the PointerKey that acts as a representative for the class of pointers that includes the return value for a node
37        */
38       PointerKey getPointerKeyForReturnValue(CGNode node);
39     
40       /**
41        * @return the PointerKey that acts as a representative for the class of pointers that includes the exceptional return value
42        */
43       PointerKey getPointerKeyForExceptionalReturnValue(CGNode node);
44     
45       /**
46        * @return the PointerKey that acts as a representative for the class of pointers that includes the contents of the static field
47        */
48       PointerKey getPointerKeyForStaticField(IField f);
49     
50       /**
51        * @return the PointerKey that acts as a representation for the class of pointers that includes the given instance field.
52        */
53       PointerKey getPointerKeyForInstanceField(InstanceKey I, IField field);
54     
55       /**
56        * TODO: expand this API to differentiate between different array indices
57        * 
58        * @param I an InstanceKey representing an abstract array
59        * @return the PointerKey that acts as a representation for the class of pointers that includes the given array contents.
60        */
61       PointerKey getPointerKeyForArrayContents(InstanceKey I);
62     }
