47       /**
48        * spawn a process to view a WALA IR
49        * 
50        * @return a handle to the pdf viewer process
51        * @throws IllegalArgumentException if ir is null
52        */
53       public static Process ghostviewIR(IClassHierarchy cha, IR ir, String pdfFile, String dotFile, String dotExe, String pdfViewExe,
54           NodeDecorator<ISSABasicBlock> annotations) throws WalaException {
55     
56         if (ir == null) {
57           throw new IllegalArgumentException("ir is null");
58         }
59         Graph<ISSABasicBlock> g = ir.getControlFlowGraph();
60     
61         NodeDecorator<ISSABasicBlock> labels = makeIRDecorator(ir);
62         if (annotations != null) {
63           labels = new ConcatenatingNodeDecorator<ISSABasicBlock>(annotations, labels);
64         }
65     
66         g = CFGSanitizer.sanitize(ir, cha);
67     
68         DotUtil.<ISSABasicBlock>dotify(g,labels,dotFile,pdfFile,dotExe);
69     
70         return launchPDFView(pdfFile, pdfViewExe);
71       }

73       public static NodeDecorator<ISSABasicBlock> makeIRDecorator(IR ir) {
74         if (ir == null) {
75           throw new IllegalArgumentException("ir is null");
76         }
77         final HashMap<ISSABasicBlock,String> labelMap = HashMapFactory.make();
78         for (Iterator it = ir.getControlFlowGraph().iterator(); it.hasNext();) {
79           SSACFG.BasicBlock bb = (SSACFG.BasicBlock) it.next();
80           labelMap.put(bb, getNodeLabel(ir, bb));
81         }
82         NodeDecorator<ISSABasicBlock> labels = new NodeDecorator<ISSABasicBlock>() {
83           @Override
84           public String getLabel(ISSABasicBlock bb) {
85             return labelMap.get(bb);
86           }
87         };
88         return labels;
89       }

91       /**
92        * A node decorator which concatenates the labels from two other node decorators
93        * @param <T> the type of the node
94        */
95       private final static class ConcatenatingNodeDecorator<T> implements NodeDecorator<T> {
96     
97         private final NodeDecorator<T> A;
98     
99         private final NodeDecorator<T> B;
100     
101         ConcatenatingNodeDecorator(NodeDecorator<T> A, NodeDecorator<T> B) {
102           this.A = A;
103           this.B = B;
104         }
105     
106         @Override
107         public String getLabel(T n) throws WalaException {
108           return A.getLabel(n) + B.getLabel(n);
109         }
110     
111       }

113       private static String getNodeLabel(IR ir, BasicBlock bb) {
114         StringBuffer result = new StringBuffer();
115     
116         int start = bb.getFirstInstructionIndex();
117         int end = bb.getLastInstructionIndex();
118         result.append("BB").append(bb.getNumber());
119         if (bb.isEntryBlock()) {
120           result.append(" (en)\\n");
121         } else if (bb.isExitBlock()) {
122           result.append(" (ex)\\n");
123         }
124         if (bb instanceof ExceptionHandlerBasicBlock) {
125           result.append("<Handler>");
126         }
127         result.append("\\n");
128         for (Iterator it = bb.iteratePhis(); it.hasNext();) {
129           SSAPhiInstruction phi = (SSAPhiInstruction) it.next();
130           if (phi != null) {
131             result.append("           " + phi.toString(ir.getSymbolTable())).append("\\l");
132           }
133         }
134         if (bb instanceof ExceptionHandlerBasicBlock) {
135           ExceptionHandlerBasicBlock ebb = (ExceptionHandlerBasicBlock) bb;
136           SSAGetCaughtExceptionInstruction s = ebb.getCatchInstruction();
137           if (s != null) {
138             result.append("           " + s.toString(ir.getSymbolTable())).append("\\l");
139           } else {
140             result.append("           " + " No catch instruction. Unreachable?\\l");
141           }
142         }
143         SSAInstruction[] instructions = ir.getInstructions();
144         for (int j = start; j <= end; j++) {
145           if (instructions[j] != null) {
146             StringBuffer x = new StringBuffer(j + "   " + instructions[j].toString(ir.getSymbolTable()));
147             StringStuff.padWithSpaces(x, 35);
148             result.append(x);
149             result.append("\\l");
150           }
151         }
152         for (Iterator it = bb.iteratePis(); it.hasNext();) {
153           SSAPiInstruction pi = (SSAPiInstruction) it.next();
154           if (pi != null) {
155             result.append("           " + pi.toString(ir.getSymbolTable())).append("\\l");
156           }
157         }
158         return result.toString();
159       }

161       /**
162        * Launch a process to view a PDF file
163        */
164       public static Process launchPDFView(String pdfFile, String gvExe) throws WalaException {
165         // set up a viewer for the ps file.
166         if (gvExe == null) {
167           throw new IllegalArgumentException("null gvExe");
168         }
169         if (pdfFile == null) {
170           throw new IllegalArgumentException("null psFile");
171         }
172         final PDFViewLauncher gv = new PDFViewLauncher();
173         gv.setGvExe(gvExe);
174         gv.setPDFFile(pdfFile);
175         gv.run();
176         if (gv.getProcess() == null) {
177           throw new WalaException(" problem spawning process ");
178         }
179         return gv.getProcess();
180       }
