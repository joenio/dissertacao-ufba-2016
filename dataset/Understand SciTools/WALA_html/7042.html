11     package com.ibm.wala.util.collections;
12     
13     import java.util.NoSuchElementException;
14     
15     /**
16      * Simple Heap data structure.
17      */
18     public abstract class Heap<T> {
19     
20       /**
21        * @return true iff elt1 is considered < elt2
22        */
23       abstract protected boolean compareElements(T elt1, T elt2);
24     
25       private int numberOfElements;
26     
27       private T[] backingStore;
28     
29       /**
30        * @return number of elements in this heap
31        */
32       public int size() {
33         return numberOfElements;
34       }
35     
36       @SuppressWarnings("unchecked")
37       public Heap(int initialCapacity) {
38         numberOfElements = 0;
39         backingStore = (T[])new Object[initialCapacity];
40       }
41     
42       /**
43        * @return true iff this heap is non-empty
44        */
45       final public boolean isEmpty() {
46         return numberOfElements == 0;
47       }
48     
49     
50       public void insert(T elt) {
51         ensureCapacity(numberOfElements + 1);
52         bubbleUp(elt, numberOfElements);
53         numberOfElements++;
54       }
55     
56       /**
57        * @return the first object in the priority queue
58        */
59       public T take() throws NoSuchElementException {
60         if (numberOfElements == 0) {
61           throw new NoSuchElementException();
62         }
63         T result = backingStore[0];
64         removeElement(0);
65         return result;
66       }
67     
68       private static int heapParent(int index) {
69         return (index - 1) / 2;
70       }
71     
72       private static int heapLeftChild(int index) {
73         return index * 2 + 1;
74       }
75     
76       private static int heapRightChild(int index) {
77         return index * 2 + 2;
78       }
79     
80     
81       @SuppressWarnings("unchecked")
82       final private void ensureCapacity(int min) {
83         if (backingStore.length < min) {
84           T newStore[] = (T[])new Object[2 * min];
85           System.arraycopy(backingStore, 0, newStore, 0, backingStore.length);
86           backingStore = newStore;
87         }
88       }
89     
90       /**
91        * SJF: I know this is horribly uglified ... I've attempted to make things as
92        * easy as possible on the JIT, since this is performance critical.
93        * 
94        * @param index
95        */
96       final private void removeElement(int index) {
97         int ne = numberOfElements;
98         T[] bs = backingStore;
99         while (true) {
100           int leftIndex = heapLeftChild(index);
101           if (leftIndex < ne) {
102             int rightIndex = heapRightChild(index);
103             if (rightIndex < ne) {
104               T leftObject = bs[leftIndex];
105               T rightObject = bs[rightIndex];
106               if (compareElements(leftObject, rightObject)) {
107                 bs[index] = leftObject;
108                 index = leftIndex;
109               } else {
110                 bs[index] = rightObject;
111                 index = rightIndex;
112               }
113             } else {
114               bs[index] = bs[leftIndex];
115               index = leftIndex;
116             }
117             // manual tail recursion elimination here
118           } else {
119             numberOfElements--;
120             ne = numberOfElements;
121             if (index != ne) {
122               bubbleUp(bs[ne], index);
123             }
124             return;
125           }
126         }
127       }
128     
129       /**
130        * SJF: I know this is uglified ... I've attempted to make things as easy as
131        * possible on the JIT, since this is performance critical.
132        * 
133        * @param elt
134        * @param index
135        */
136       final private void bubbleUp(T elt, int index) {
137         T[] bs = backingStore;
138         while (true) {
139           if (index == 0) {
140             bs[index] = elt;
141             return;
142           }
143           int hpIndex = heapParent(index);
144           T parent = bs[hpIndex];
145           if (compareElements(parent, elt)) {
146             bs[index] = elt;
147             return;
148           } else {
149             bs[index] = parent;
150             // manual tail recursion elimination
151             index = hpIndex;
152           }
153         }
154       }
155       
156       @Override
157       public String toString() {
158         StringBuffer s = new StringBuffer();
159         s.append("[");
160         for (int i = 0; i < size(); i++) {
161           if (backingStore[i] != null) {
162             if (i > 0)
163               s.append(",");
164             s.append(backingStore[i].toString());
165           }
166         }
167         s.append("]");
168         return s.toString();
169       }
170     }
