41     /**
42      * A CallGraph implementation adapted to work for graphs that contain code
43      * entities from multiple languages, and hence multiple specialized forms of IR.
44      * The root node delegates to one of several language-specific root nodes,
45      * allowing each language to use its own specialized IR constructs for entry
46      * points.
47      * 
48      * @author Julian Dolby (dolby@us.ibm.com)
49      */
50     public class CrossLanguageCallGraph extends AstCallGraph {
51     
52       public CrossLanguageCallGraph(TargetLanguageSelector<AbstractRootMethod, CrossLanguageCallGraph> roots, IClassHierarchy cha,
53           AnalysisOptions options, AnalysisCache cache) {
54         super(cha, options, cache);
55         this.roots = roots;
56       }
57     
58       private final TargetLanguageSelector<AbstractRootMethod, CrossLanguageCallGraph> roots;
59     
60       private final Set<CGNode> languageRootNodes = HashSetFactory.make();
61     
62       private final Map<Atom,IMethod> languageRoots = HashMapFactory.make();
63     
64       @SuppressWarnings("deprecation")
65       public AbstractRootMethod getLanguageRoot(Atom language){
66         if (!languageRoots.containsKey(language)) {
67           AbstractRootMethod languageRoot = roots.get(language, this);
68     
69           CGNode languageRootNode = null;
70           try {
71             languageRootNode = findOrCreateNode(languageRoot, Everywhere.EVERYWHERE);
72           } catch (CancelException e) {
73             e.printStackTrace();
74             Assertions.UNREACHABLE();
75           }
76     
77           languageRootNodes.add(languageRootNode);
78     
79           CallSiteReference site = CallSiteReference.make(1, languageRoot.getReference(), IInvokeInstruction.Dispatch.STATIC);
80     
81           CGNode fakeRootNode = getFakeRootNode();
82           CrossLanguageFakeRoot fakeRootMethod = (CrossLanguageFakeRoot) fakeRootNode.getMethod();
83     
84           site = fakeRootMethod.addInvocationInternal(new int[0], site).getCallSite();
85     
86           fakeRootNode.addTarget(site, languageRootNode);
87     
88           languageRoots.put(language, languageRoot);
89         }
90     
91         return (AbstractRootMethod) languageRoots.get(language);
92       }
93     
94       public class CrossLanguageFakeRoot extends ScriptFakeRoot {
95     
96         public CrossLanguageFakeRoot(IClass declaringClass, IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache) {
97           super(FakeRootMethod.rootMethod, declaringClass, cha, options, cache);
98         }
99     
100         public CrossLanguageFakeRoot(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache) {
101           super(FakeRootMethod.rootMethod, cha, options, cache);
102         }
103     
104         public int addPhi(TypeReference type, int[] values) {
105           Atom language = type.getClassLoader().getLanguage();
106           AbstractRootMethod root = getLanguageRoot(language);
107           return root.addPhi(values);
108         }
109     
110         @Override
111         public int addGetInstance(FieldReference ref, int object) {
112           TypeReference type = ref.getDeclaringClass();
113           Atom language = type.getClassLoader().getLanguage();
114           AbstractRootMethod root = getLanguageRoot(language);
115           return root.addGetInstance(ref, object);
116         }
117     
118         @Override
119         public int addGetStatic(FieldReference ref) {
120           TypeReference type = ref.getDeclaringClass();
121           Atom language = type.getClassLoader().getLanguage();
122           AbstractRootMethod root = getLanguageRoot(language);
123           return root.addGetStatic(ref);
124         }
125     
126         @Override
127         public int addCheckcast(TypeReference[] type, int rv, boolean isPEI) {
128           Atom language = type[0].getClassLoader().getLanguage();
129           AbstractRootMethod root = getLanguageRoot(language);
130           return root.addCheckcast(type, rv, isPEI);
131         }
132     
133         @Override
134         public SSANewInstruction addAllocation(TypeReference type) {
135           Atom language = type.getClassLoader().getLanguage();
136           AbstractRootMethod root = getLanguageRoot(language);
137           return root.addAllocation(type);
138         }
139     
140         @Override
141         public SSAInvokeInstruction addInvocation(int[] params, CallSiteReference site) {
142           TypeReference type = site.getDeclaredTarget().getDeclaringClass();
143           Atom language = type.getClassLoader().getLanguage();
144           AbstractRootMethod root = getLanguageRoot(language);
145           return root.addInvocation(params, site);
146         }
147     
148         public SSAInvokeInstruction addInvocationInternal(int[] params, CallSiteReference site) {
149           return super.addInvocation(params, site);
150         }
151     
152         @Override
153         public AstLexicalRead addGlobalRead(TypeReference type, String name) {
154           Atom language = type.getClassLoader().getLanguage();
155           AbstractRootMethod root = getLanguageRoot(language);
156           return ((AstFakeRoot) root).addGlobalRead(type, name);
157         }
158     
159         @Override
160         public SSAAbstractInvokeInstruction addDirectCall(int functionVn, int[] argVns, CallSiteReference callSite) {
161           TypeReference type = callSite.getDeclaredTarget().getDeclaringClass();
162           Atom language = type.getClassLoader().getLanguage();
163           AbstractRootMethod root = getLanguageRoot(language);
164           return ((ScriptFakeRoot) root).addDirectCall(functionVn, argVns, callSite);
165         }
166       }
167     
168       Iterator getLanguageRoots() {
169         return languageRootNodes.iterator();
170       }
171     
172       @Override
173       protected CGNode makeFakeRootNode() throws CancelException {
174         return findOrCreateNode(new CrossLanguageFakeRoot(cha, options, getAnalysisCache()), Everywhere.EVERYWHERE);
175       }
176     }
