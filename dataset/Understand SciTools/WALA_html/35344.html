29     /**
30      * A function vertex represents a function object (or, more precisely, all function objects
31      * arising from a single function expression or declaration).
32      * 
33      * @author mschaefer
34      */
35     public class FuncVertex extends Vertex implements ObjectVertex {
36     	// the IClass representing this function in the class hierarchy
37     	protected final IClass klass;
38     
39     	FuncVertex(IClass method) {
40     		this.klass = method;
41     	}
42     		
43     	public String getFullName() {
44     		return klass.getName().toString();
45     	}
46     
47     	@Override
48     	public <T> T accept(VertexVisitor<T> visitor) {
49     		return visitor.visitFuncVertex(this);
50     	}
51     
52     	@Override
53     	public String toString() {
54     		String methodName = klass.getName().toString();
55         return "Func(" + methodName.substring(methodName.lastIndexOf('/')+1) + ")";
56     	}
57     
58       @Override
59       public Iterator<Pair<CGNode, NewSiteReference>> getCreationSites(CallGraph CG) {
60         MethodReference ctorRef = JavaScriptMethods.makeCtorReference(JavaScriptTypes.Function);
61         Set<CGNode> f = CG.getNodes(ctorRef);
62         CGNode ctor = null;
63         for(CGNode n : f) {
64           JavaScriptConstructor c = (JavaScriptConstructor) n.getMethod();
65           if (c.constructedType().equals(klass)) {
66             ctor = n;
67             break;
68           }
69         }
70         
71         // built in objects
72         if (ctor == null) {
73           return EmptyIterator.instance();
74         }
75         
76         Iterator<CGNode> callers = CG.getPredNodes(ctor);
77         CGNode caller = callers.next();
78         assert !callers.hasNext();
79         
80         Iterator<CallSiteReference> sites = CG.getPossibleSites(caller, ctor);
81         CallSiteReference site = sites.next();
82         assert !sites.hasNext() : caller + " --> " + ctor + " @ " + site + " and " + sites.next() + "\n" + caller.getIR();
83         
84         return NonNullSingletonIterator.make(Pair.make(caller, NewSiteReference.make(site.getProgramCounter(), klass.getReference())));
85       }
86     
87       @Override
88       public IClass getConcreteType() {
89         return klass;
90       }
91     }
