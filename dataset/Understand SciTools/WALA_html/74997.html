15     public class CreationSiteVertex extends Vertex implements ObjectVertex {
16       private final IMethod node;
17       private final int instructionIndex;
18       private final TypeReference createdType;
19      
20       public CreationSiteVertex(IMethod node, int instructionIndex, TypeReference createdType) {
21         super();
22         this.node = node;
23         this.instructionIndex = instructionIndex;
24         this.createdType = createdType;
25       }
26       
27       @Override
28       public IClass getConcreteType() {
29         return node.getClassHierarchy().lookupClass(createdType);
30       }
31     
32       public IMethod getMethod() {
33         return node;
34       }
35       
36       @Override
37       public Iterator<Pair<CGNode, NewSiteReference>> getCreationSites(CallGraph CG) {
38         CGNode cgn = CG.getNode(node, Everywhere.EVERYWHERE);
39         assert cgn != null : node;     
40         return NonNullSingletonIterator.make(Pair.make(cgn, NewSiteReference.make(instructionIndex, createdType)));
41       }
42     
43       @Override
44       public <T> T accept(VertexVisitor<T> visitor) {
45         return visitor.visitCreationSiteVertex(this);
46       }
