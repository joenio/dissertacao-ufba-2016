11     package com.ibm.wala.cast.ir.ssa;
12     
13     import java.util.Collection;
14     import java.util.Collections;
15     
16     import com.ibm.wala.ssa.SSAAbstractBinaryInstruction;
17     import com.ibm.wala.ssa.SSAInstruction;
18     import com.ibm.wala.ssa.SSAInstructionFactory;

20     import com.ibm.wala.types.TypeReference;

22     /**
23      * This instruction represents iterating through the properties of its receiver object. The use represents an object,
24      * and the l-value represents one of a sequence of property names, suitable for use with the appropriate
25      * AbstractReflectiveGet sub-class.
26      * 
27      * Iterating across the fields or properties of a given object is a common idiom in scripting languages, which is why
28      * the IR has first-class support for it.
29      * 
30      * @author Julian Dolby (dolby@us.ibm.com)
31      */
32     public class EachElementGetInstruction extends SSAAbstractBinaryInstruction {
33     
34       public EachElementGetInstruction(int iindex, int lValue, int objectRef, int previousProp) {
35         super(iindex, lValue, objectRef, previousProp);
36       }
37     
38       @Override
39       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
40         return ((AstInstructionFactory)insts).EachElementGetInstruction(iindex, (defs == null) ? getDef(0) : defs[0], (uses == null) ? getUse(0) : uses[0], (uses == null) ? getUse(1) : uses[1]);
41       }
42     
43       @Override
44       public String toString(SymbolTable symbolTable) {
45         return getValueString(symbolTable, getDef(0)) + " = a property name of " + getValueString(symbolTable, getUse(0));
46       }
47     
48       @Override
49       public void visit(IVisitor v) {
50         ((AstInstructionVisitor) v).visitEachElementGet(this);
51       }
52     
53       @Override
54       public Collection<TypeReference> getExceptionTypes() {
55         return Collections.emptySet();
56       }
57     
58       @Override
59       public boolean isFallThrough() {
60         return true;
61       }
62     }
