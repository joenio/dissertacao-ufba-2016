11     package com.ibm.wala.cast.js.ipa.callgraph.correlations.extraction;
12     
13     import com.ibm.wala.cast.tree.CAstNode;
14     
15     /**
16      * A {@link NodePos} for a node that labels a CFG edge; currently only seems to occur with 'switch'
17      * statements.
18      * 
19      * @author mschaefer
20      *
21      */
22     public class LabelPos extends NodePos {
23       private final CAstNode parent;
24       private final NodePos parent_pos;
25     
26       public LabelPos(CAstNode node, NodePos parent_pos) {
27         this.parent = node;
28         this.parent_pos = parent_pos;
29       }
30     
31       public CAstNode getParent() {
32         return parent;
33       }
34     
35       public NodePos getParentPos() {
36         return parent_pos;
37       }
38     
39       @Override
40       public <A> A accept(PosSwitch<A> ps) {
41         return ps.caseLabelPos(this);
42       }
43     
44     }
