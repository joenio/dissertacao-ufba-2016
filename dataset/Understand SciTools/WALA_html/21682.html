47     	static Collection<Annotation> getAnnotations(Collection<AnnotationItem> as, ClassLoaderReference clr) {
48     		Collection<Annotation> result = HashSetFactory.make();
49     		for(AnnotationItem a : as) {
50     			result.add(getAnnotation(a, clr));
51     		}
52     		return result;
53     	}

55     	static Annotation getAnnotation(AnnotationItem a, ClassLoaderReference clr) {
56     		return getAnnotation(a.getEncodedAnnotation(), clr);
57     	}

59     	static Annotation getAnnotation(AnnotationEncodedSubValue ea, ClassLoaderReference clr) {
60     		Map<String,ElementValue> values = HashMapFactory.make();
61     		TypeReference at = getTypeRef(ea.annotationType, clr);
62     
63     		for(int i = 0; i < ea.names.length; i++) {
64     			String name = ea.names[i].getStringValue();
65     			EncodedValue v = ea.values[i];
66     			ElementValue value = getValue(clr, v);
67     			values.put(name, value);
68     		}
69     		
70     		return Annotation.makeWithNamed(at, values);
71     	}

73     	static ElementValue getValue(ClassLoaderReference clr, EncodedValue v) {
74     		switch (v.getValueType()) {
75     		case VALUE_ANNOTATION:
76     			Annotation a = getAnnotation((AnnotationEncodedSubValue)v, clr);
77     			return new AnnotationAttribute(a.getType().getName().toString() +";", a.getNamedArguments());
78     			
79     		case VALUE_ARRAY:
80     			EncodedValue[] vs = ((ArrayEncodedValue)v).values;
81     			ElementValue rs[] = new ElementValue[ vs.length ];
82     			for(int idx = 0; idx < vs.length; idx++) {
83     				rs[idx] = getValue(clr, vs[idx]);
84     			}
85     			return new ArrayElementValue(rs);
86     			
87     		case VALUE_BOOLEAN:
88     			Boolean bl = ((BooleanEncodedValue)v).value;
89     			return new ConstantElementValue(bl);
90     			
91     		case VALUE_BYTE:
92     			Byte bt = ((ByteEncodedValue)v).value;
93     			return new ConstantElementValue(bt);
94     			
95     		case VALUE_CHAR:
96     			Character c = ((CharEncodedValue)v).value;
97     			return new ConstantElementValue(c);
98     			
99     		case VALUE_DOUBLE:
100     			Double d = ((DoubleEncodedValue)v).value;
101     			return new ConstantElementValue(d);
102     			
103     		case VALUE_ENUM:
104     			FieldIdItem o = ((EnumEncodedValue)v).value;
105     			return new EnumElementValue(o.getFieldType().getTypeDescriptor(), o.getFieldName().getStringValue());
106     			
107     		case VALUE_FIELD:
108     			o = v.getValueType()==ValueType.VALUE_ENUM? ((EnumEncodedValue)v).value: ((FieldEncodedValue)v).value;
109     			String fieldName = o.getFieldName().getStringValue();
110     			TypeReference ft = getTypeRef(o.getFieldType(), clr);
111     			TypeReference ct = getTypeRef(o.getContainingClass(), clr);
112     			return new ConstantElementValue(FieldReference.findOrCreate(ct, Atom.findOrCreateUnicodeAtom(fieldName), ft));
113     						
114     		case VALUE_FLOAT:
115     			Float f = ((FloatEncodedValue)v).value;
116     			return new ConstantElementValue(f);
117     			
118     		case VALUE_INT:
119     			Integer iv = ((IntEncodedValue)v).value;
120     			return new ConstantElementValue(iv);
121     			
122     		case VALUE_LONG:
123     			Long l = ((LongEncodedValue)v).value;
124     			return new ConstantElementValue(l);
125     			
126     		case VALUE_METHOD:
127     			MethodIdItem m = ((MethodEncodedValue)v).value;
128     			ct = getTypeRef(m.getContainingClass(), clr);
129     			String methodName = m.getMethodName().getStringValue();
130     			String methodSig = m.getPrototype().getPrototypeString();
131     			return new ConstantElementValue(MethodReference.findOrCreate(ct, Atom.findOrCreateUnicodeAtom(methodName), Descriptor.findOrCreateUTF8(methodSig)));
132     			
133     		case VALUE_NULL:
134     			return new ConstantElementValue(null);
135     			
136     		case VALUE_SHORT:
137     			Short s = ((ShortEncodedValue)v).value;
138     			return new ConstantElementValue(s);
139     
140     		case VALUE_STRING:
141     			String str = ((StringEncodedValue)v).value.getStringValue();
142     			return new ConstantElementValue(str);
143     			
144     		case VALUE_TYPE:
145     			TypeIdItem t = ((TypeEncodedValue)v).value;
146     			return new ConstantElementValue(getTypeName(t) + ";");
147     			
148     		default:
149     			assert false : v;
150     			return null;
151     		}
152     	}
153     	
154     	static TypeReference getTypeRef(TypeIdItem type, ClassLoaderReference clr) {
155     		return TypeReference.findOrCreate(clr, getTypeName(type));
156     	}
157     	
158     	static TypeName getTypeName(TypeIdItem fieldType) {
159     		ImmutableByteArray fieldTypeArray = ImmutableByteArray.make(fieldType.getTypeDescriptor());
160     	    TypeName T = null;
161     	    if (fieldTypeArray.get(fieldTypeArray.length() - 1) == ';') {
162     	        T = TypeName.findOrCreate(fieldTypeArray, 0, fieldTypeArray.length() - 1);
163     	    } else {
164     	        T = TypeName.findOrCreate(fieldTypeArray);
165     	    }
166     		return T;
167     	}
