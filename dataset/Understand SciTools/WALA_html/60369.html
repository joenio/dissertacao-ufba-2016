76         public Context getCalleeTarget(CGNode caller, CallSiteReference site,
77                 IMethod callee, InstanceKey[] receivers) {
78             if(callee.getSignature().equals("java.lang.StringBuilder.toString()Ljava/lang/String;") ||
79                     callee.getSignature().equals("java.lang.StringBuilder.append(Ljava/lang/String;)Ljava/lang/StringBuilder;") ||
80                     callee.getSignature().equals("java.lang.String.valueOf(Ljava/lang/Object;)Ljava/lang/String;") ||
81                     callee.getSignature().equals("java.lang.String.toString()Ljava/lang/String;") ||
82                     callee.getSignature().equals("android.net.Uri.parse(Ljava/lang/String;)Landroid/net/Uri;") ||
83                     callee.getSignature().equals("android.net.Uri.withAppendedPath(Landroid/net/Uri;Ljava/lang/String;)Landroid/net/Uri;") ||
84     //                callee.getSignature().equals("android.net.Uri$StringUri.buildUpon()Landroid/net/Uri$Builder;") ||
85     //                callee.getSignature().equals("android.net.Uri$OpaqueUri.buildUpon()Landroid/net/Uri$Builder;") ||
86                     callee.getSignature().equals("android.net.Uri$Builder.build()Landroid/net/Uri;")
87                     )
88             {
89     //            System.out.println("Adding context to "+callee.getSignature());
90     //            for (int i = 0; i < receivers.length; i++)
91     //            	System.out.println("\t#"+i+" "+receivers[i]);
92                 if(receivers[0] instanceof NormalAllocationInNode)
93                 {
94     //                System.out.println("\t\tNormalAllocationInNode "+callee.getSignature());
95                     if(((NormalAllocationInNode)receivers[0]).getSite().getDeclaredType().getClassLoader().equals(ClassLoaderReference.Application))
96                         // create a context based on the site and the receiver
97                         return new CallerSiteContextPair(caller,site,new ReceiverInstanceContext(receivers[0]));
98     //                if (callee.getSignature().equals("android.net.Uri$StringUri.buildUpon()Landroid/net/Uri$Builder;") ||
99     //                		callee.getSignature().equals("android.net.Uri$OpaqueUri.buildUpon()Landroid/net/Uri$Builder;") ||
100     //                		callee.getSignature().equals("android.net.Uri$Builder.build()Landroid/net/Uri;"))
101     //                	return new CallerSiteContextPair(caller,site,new ReceiverInstanceContext(receivers[0]));
102                     if (callee.getSignature().equals("android.net.Uri$Builder.build()Landroid/net/Uri;"))
103                     	return new CallerSiteContextPair(caller,site,new ReceiverInstanceContext(receivers[0]));
104                         //return new CallerSiteContext(caller,site);                	
105                 }
106                 else if(callee.getSignature().equals("java.lang.String.valueOf(Ljava/lang/Object;)Ljava/lang/String;") ||
107                         callee.getSignature().equals("java.lang.String.toString()Ljava/lang/String;") ||
108                         callee.getSignature().equals("android.net.Uri.parse(Ljava/lang/String;)Landroid/net/Uri;") ||
109                         callee.getSignature().equals("android.net.Uri.withAppendedPath(Landroid/net/Uri;Ljava/lang/String;)Landroid/net/Uri;")
110                         )
111                 {
112                     return new CallerSiteContext(caller,site);
113                 }
114             }
115             else //if(callee.getSignature().contains("value"))
116             {
117     //          System.out.println("Found signature: "+callee.getSignature());
118             }
119             if(!caller.getContext().equals(Everywhere.EVERYWHERE))
120             {
121                 //System.out.println("Call to "+callee+" from caller "+caller+" in context "+caller.getContext());
122             }
123             return super.getCalleeTarget(caller, site, callee, receivers);
124         }
