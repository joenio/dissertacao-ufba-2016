11     package com.ibm.wala.shrikeBT;
12     
13     /**
14      * This class represents put and putstatic instructions.
15      */
16     public class PutInstruction extends Instruction implements IPutInstruction {
17       protected String type;
18     
19       protected String classType;
20     
21       protected String fieldName;
22     
23       PutInstruction(short opcode, String type, String classType, String fieldName) {
24         super(opcode);
25         this.type = type;
26         this.classType = classType;
27         this.fieldName = fieldName;
28       }
29     
30       ConstantPoolReader getLazyConstantPool() {
31         return null;
32       }
33     
34       final static class Lazy extends PutInstruction {
35         final private ConstantPoolReader cp;
36     
37         final private int index;
38     
39         Lazy(short opcode, ConstantPoolReader cp, int index) {
40           super(opcode, null, null, null);
41           this.index = index;
42           this.cp = cp;
43         }
44     
45         @Override
46         ConstantPoolReader getLazyConstantPool() {
47           return cp;
48         }
49     
50         int getCPIndex() {
51           return index;
52         }
53     
54         @Override
55         public String getClassType() {
56           if (classType == null) {
57             classType = cp.getConstantPoolMemberClassType(index);
58           }
59           return classType;
60         }
61     
62         @Override
63         public String getFieldName() {
64           if (fieldName == null) {
65             fieldName = cp.getConstantPoolMemberName(index);
66           }
67           return fieldName;
68         }
69     
70         @Override
71         public String getFieldType() {
72           if (type == null) {
73             type = cp.getConstantPoolMemberType(index);
74           }
75           return type;
76         }
77       }
78     
79       static PutInstruction make(ConstantPoolReader cp, int index, boolean isStatic) {
80         return new Lazy(isStatic ? OP_putstatic : OP_putfield, cp, index);
81       }
82     
83       public static PutInstruction make(String type, String className, String fieldName, boolean isStatic) {
84         if (type == null) {
85           throw new IllegalArgumentException("type must not be null");
86         }
87         if (className == null) {
88           throw new IllegalArgumentException("className must not be null");
89         }
90         if (fieldName == null) {
91           throw new IllegalArgumentException("fieldName must not be null");
92         }
93         return new PutInstruction(isStatic ? OP_putstatic : OP_putfield, type, className, fieldName);
94       }
95     
96       @Override
97       final public boolean equals(Object o) {
98         if (o instanceof PutInstruction) {
99           PutInstruction i = (PutInstruction) o;
100           return i.getFieldType().equals(getFieldType()) && i.getClassType().equals(getClassType())
101               && i.getFieldName().equals(getFieldName()) && i.opcode == opcode;
102         } else {
103           return false;
104         }
105       }
106     
107       @Override
108       public String getClassType() {
109         return classType;
110       }
111     
112       @Override
113       public String getFieldType() {
114         return type;
115       }
116     
117       @Override
118       public String getFieldName() {
119         return fieldName;
120       }
121     
122       @Override
123       final public boolean isStatic() {
124         return opcode == OP_putstatic;
125       }
126     
127       @Override
128       final public int hashCode() {
129         return getClassType().hashCode() + 9011 * getClassType().hashCode() + 317 * getFieldName().hashCode() + opcode;
130       }
131     
132       @Override
133       final public int getPoppedCount() {
134         return isStatic() ? 1 : 2;
135       }
136     
137       @Override
138       final public String toString() {
139         return "Put(" + getFieldType() + "," + (isStatic() ? "STATIC" : "NONSTATIC") + "," + getClassType() + "," + getFieldName()
140             + ")";
141       }
142     
143       @Override
144       final public void visit(IInstruction.Visitor v) throws NullPointerException {
145         v.visitPut(this);
146       }
147     
148       @Override
149       public boolean isPEI() {
150         return !isStatic();
151       }
152     }
