41     public class LambdaTest extends WalaTestCase {

43       @Test public void testSortingExample() throws ClassHierarchyException, IllegalArgumentException, CancelException, IOException {
44     
45         AnalysisScope scope = CallGraphTestUtil.makeJ2SEAnalysisScope(TestConstants.WALA_TESTDATA, CallGraphTestUtil.REGRESSION_EXCLUSIONS);
46         ClassHierarchy cha = ClassHierarchy.make(scope);
47         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha,
48             "Llambda/SortingExample");
49         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
50     
51         CallGraph cg = CallGraphTestUtil.buildZeroCFA(options, new AnalysisCache(), cha, scope, false);
52         
53         // Find compareTo
54         TypeReference str = TypeReference.findOrCreate(ClassLoaderReference.Primordial, "Ljava/lang/String");
55         MethodReference ct = MethodReference.findOrCreate(str, Atom.findOrCreateUnicodeAtom("compareTo"), Descriptor.findOrCreateUTF8("(Ljava/lang/String;)I"));
56         Set<CGNode> ctnodes = cg.getNodes(ct);
57        
58         checkCompareToCalls(cg, ctnodes, "id1", 1);
59         checkCompareToCalls(cg, ctnodes, "id2", 1);
60         checkCompareToCalls(cg, ctnodes, "id3", 2);
61         checkCompareToCalls(cg, ctnodes, "id4", 1);
62       }
63     
64       protected void checkCompareToCalls(CallGraph cg, Set<CGNode> ctnodes, String x, int expected) {
65         // Find node corresponding to id1
66         TypeReference tid1 = TypeReference.findOrCreate(ClassLoaderReference.Application, "Llambda/SortingExample");
67         MethodReference mid1 = MethodReference.findOrCreate(tid1, x, "(I)I");
68         Assert.assertTrue("expect " + x + " node", cg.getNodes(mid1).iterator().hasNext());
69         CGNode id1node = cg.getNodes(mid1).iterator().next();
70         
71         // caller of id1 is dynamic from sortForward, and has 1 compareTo
72         CGNode sfnode = cg.getPredNodes(id1node).next();
73         int count = 0;
74         for(Iterator<CallSiteReference> sites = sfnode.iterateCallSites(); sites.hasNext(); ) {
75          if (ctnodes.containsAll(cg.getPossibleTargets(sfnode, sites.next()))) {
76            count++;
77          }
78         }
79         Assert.assertEquals("expected one call to compareTo", expected, count);
80         System.err.println("found " + count + " compareTo calls in " + sfnode);
81       }
82     }
