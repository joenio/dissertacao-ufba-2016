20     /**
21      * Interface for an SDG (loosely defined here as a graph of {@link Statement}s. 
22      * This interface implies that the underlying graph is computed lazily on demand.
23      */
24     public interface ISDG extends NumberedGraph<Statement>, IClassHierarchyDweller {
25     
26       /**
27        * {@link ControlDependenceOptions} used to construct this graph.
28        */
29       ControlDependenceOptions getCOptions();
30     
31       /**
32        * Get the program dependence graph constructed for a particular node.
33        */
34       PDG getPDG(CGNode node);
35     
36       /**
37        * Iterate over the nodes which have been discovered so far, but do <em>NOT</em> eagerly construct the entire graph.
38        */
39       Iterator<? extends Statement> iterateLazyNodes();
40     
41     }
