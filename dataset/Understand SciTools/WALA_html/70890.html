32       /**
33        * create a Properties object representing the properties set by the command
34        * line args. if args[i] is "-foo" and args[i+1] is "bar", then the result
35        * will define a property with key "foo" and value "bar"
36        */
37       public static Properties parseCommandLine(String[] cmdLine) {
38         if (cmdLine == null) {
39           throw new IllegalArgumentException("null cmdLine");
40         }
41         if (cmdLine.length == 0) {
42           throw new IllegalArgumentException("cmdLine must have at least one parameter");
43         }
44         Properties p = null;
45         assert cmdLine[0].equals("-pdelaunch");
46         String[] x = new String[cmdLine.length - 1];
47         System.arraycopy(cmdLine, 1, x, 0, x.length);
48         try {
49           p = CommandLine.parse(x);
50         } catch (IllegalArgumentException e) {
51           e.printStackTrace();
52           System.err.println("Length " + x.length);
53           for (String s : x) {
54             System.err.println(s);
55           }
56           Assertions.UNREACHABLE();
57         }
58         return p;
59       }

61       protected static <X> X getProjectFromWorkspace(Function<IProject, X> pred) {
62         IWorkspaceRoot workspaceRoot = ResourcesPlugin.getWorkspace().getRoot();
63         IPath workspaceRootPath = workspaceRoot.getLocation();
64         System.out.println("workspace: " + workspaceRootPath.toOSString());
65       
66         for (IProject p : workspaceRoot.getProjects()) {
67           X result = pred.apply(p);
68           if (result != null) {
69             return result;
70           }
71         }
72         Assertions.UNREACHABLE();
73         return null;
74       }

82       public interface Parser<Unit, CompiledUnit> {
83         void setProject(IProject project);
84         
85         void processASTs(Map<Unit,EclipseSourceFileModule> files, Function<Object[], Boolean> errors);
86       }
87         
88       public static <Unit, CompiledUnit> void parseModules(Set<ModuleEntry> modules, EclipseCompiler<Unit, CompiledUnit> compiler) {
89         // sort files into projects
90         Map<IProject, Map<Unit,EclipseSourceFileModule>> projectsFiles = new HashMap<IProject, Map<Unit,EclipseSourceFileModule>>();
91         for (ModuleEntry m : modules) {
92           if (m instanceof EclipseSourceFileModule) {
93             EclipseSourceFileModule entry = (EclipseSourceFileModule) m;
94             IProject proj = entry.getIFile().getProject();
95             if (!projectsFiles.containsKey(proj)) {
96               projectsFiles.put(proj, new HashMap<Unit,EclipseSourceFileModule>());
97             }
98             projectsFiles.get(proj).put(compiler.getCompilationUnit(entry.getIFile()), entry);
99           }
100         }
101     
102       final Parser<Unit,CompiledUnit> parser = compiler.getParser();
103     
104       for (final Map.Entry<IProject,Map<Unit,EclipseSourceFileModule>> proj : projectsFiles.entrySet()) {
105         parser.setProject(proj.getKey());
106         parser.processASTs(proj.getValue(), new Function<Object[],Boolean>() {
107           @Override
108           public Boolean apply(Object[] problems) {
109             int length = problems.length;
110             if (length > 0) {
111               StringBuffer buffer = new StringBuffer();
112               for (int i = 0; i < length; i++) {
113                 buffer.append(problems[i].toString());
114                 buffer.append('\n');
115               }
116               if (length != 0) {
117                 System.err.println(buffer.toString());
118                 return true;
119               }
120             }
121             return false;
122          }    
123         });
124       }
125       }
