25     /**
26      * generates instructions to simulate the semantics of JS constructor invocations
27      *
28      */
29     public class JavaScriptConstructTargetSelector implements MethodTargetSelector {
30       private final MethodTargetSelector base;
31     
32       private final JavaScriptConstructorFunctions constructors;
33       
34       public JavaScriptConstructTargetSelector(IClassHierarchy cha, MethodTargetSelector base) {
35         this.constructors = new JavaScriptConstructorFunctions(cha);
36         this.base = base;
37       }
38     
39       public JavaScriptConstructTargetSelector(JavaScriptConstructorFunctions constructors, MethodTargetSelector base) {
40         this.constructors = constructors;
41         this.base = base;
42       }
43     
44       @Override
45       public IMethod getCalleeTarget(CGNode caller, CallSiteReference site, IClass receiver) {
46         if (site.getDeclaredTarget().equals(JavaScriptMethods.ctorReference)) {
47           IR callerIR = caller.getIR();
48           SSAAbstractInvokeInstruction callStmts[] = callerIR.getCalls(site);
49           assert callStmts.length == 1;
50           int nargs = callStmts[0].getNumberOfParameters();
51           return constructors.findOrCreateConstructorMethod(callerIR, callStmts[0], receiver, nargs - 1);
52         } else {
53           return base.getCalleeTarget(caller, site, receiver);
54         }
55       }
56     
57       public boolean mightReturnSyntheticMethod(CGNode caller, CallSiteReference site) {
58         return true;
59       }
60     
61       public boolean mightReturnSyntheticMethod(MethodReference declaredTarget) {
62         return true;
63       }
64     }
