40     import com.ibm.wala.classLoader.IField;
41     import com.ibm.wala.demandpa.alg.statemachine.StateMachine;
42     import com.ibm.wala.demandpa.flowgraph.IFlowLabel;
43     import com.ibm.wala.ipa.callgraph.propagation.PointerKey;
44     
45     /**
46      * Interface for specifying a policy for refinement of field access handling.
47      * 
48      * @author Manu Sridharan
49      * 
50      */
51     public interface FieldRefinePolicy {
52     
53       /**
54        * 
55        * @param field the accessed field
56        * @param basePtr the base pointer of the access
57        * @param val
58        * @param state
59        * @return <code>true</code> if match edges for the field access should be refined. Otherwise, <code>false</code>
60        *         is returned, indicating that the field can be handled with match edges.
61        */
62       public boolean shouldRefine(IField field, PointerKey basePtr, PointerKey val, IFlowLabel label, StateMachine.State state);
63     
64       /**
65        * 
66        * @return <code>true</code> if more refinement can be done, and hence another pass can be attempted;
67        *         <code>false</code> otherwise
68        */
69       public boolean nextPass();
70     
71     }
