11     public class MiniaturList {
12     	MiniaturList next;
13     
14     	int data;
15     
16     	public MiniaturList remove(int elt) {
17     		MiniaturList xp = null;
18     		MiniaturList head = this;
19     		MiniaturList x = head;
20     
21     		while (x != null) {
22     			if (x.data == elt) {
23     				if (xp == null)
24     					head = x.next;
25     				else
26     					xp.next = x.next;
27     				break;
28     			}
29     			xp = x;
30     			x = x.next;
31     		}
32     		return head;
33     	}
34     
35     	public static MiniaturList cons(int elt, MiniaturList l) {
36     		MiniaturList ret = new MiniaturList();
37     		ret.data = elt;
38     		ret.next = l;
39     		return ret;
40     	}
41     
42     	public boolean contains(int elt) {
43     		MiniaturList head = this;
44     		while (head != null) {
45     			if (head.data == elt)
46     				return true;
47     			head = head.next;
48     		}
49     		return false;
50     	}
51     
52     	public static void main(String[] args) {
53     		MiniaturList l1 = cons(1, cons(2, cons(3, cons(2, null))));
54     		MiniaturList l2 = cons(5, null);
55     
56     		l1 = l1.remove(2);
57     		//assert 2 !in l1.*next.data
58     
59     		System.out.println(l1.contains(3));
60     		System.out.println(l2.contains(6));
61     
62     	}
63     }
