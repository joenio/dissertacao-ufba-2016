12     package com.ibm.wala.ssa;
13     
14     import com.ibm.wala.types.FieldReference;
15     import com.ibm.wala.types.TypeReference;
16     
17     /**
18      * Abstract base class for {@link SSAGetInstruction} and {@link SSAPutInstruction}.
19      */
20     public abstract class SSAFieldAccessInstruction extends SSAInstruction {
21     
22       private final FieldReference field;
23     
24       private final int ref;
25     
26       protected SSAFieldAccessInstruction(int iindex, FieldReference field, int ref) throws IllegalArgumentException {
27         super(iindex);
28         this.field = field;
29         this.ref = ref;
30         if (field == null) {
31           throw new IllegalArgumentException("field cannot be null");
32         }
33       }
34     
35       public TypeReference getDeclaredFieldType() {
36         return field.getFieldType();
37       }
38     
39       public FieldReference getDeclaredField() {
40         return field;
41       }
42     
43       public int getRef() {
44         return ref;
45       }
46     
47       public boolean isStatic() {
48         return ref == -1;
49       }
50     
51       @Override
52       public boolean isPEI() {
53         return !isStatic();
54       }
55     }
