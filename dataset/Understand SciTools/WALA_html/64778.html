11     package com.ibm.wala.util.collections;
12     
13     import com.ibm.wala.util.Predicate;
14     
15     /**
16      * intersection of two filters
17      */
18     public class Filtersection<T> extends Predicate<T> {
19     
20       final private Predicate<T> a;
21       final private Predicate<T> b;
22     
23       public Filtersection(Predicate<T> a, Predicate<T> b) {
24         this.a = a;
25         this.b = b;
26         if (a == null) {
27           throw new IllegalArgumentException("null a");
28         }
29         if (b == null) {
30           throw new IllegalArgumentException("null b");
31         }
32       }
33     
34       @Override
35       public boolean test(T o) {
36         return a.test(o) && b.test(o);
37       }
38     
39     }
