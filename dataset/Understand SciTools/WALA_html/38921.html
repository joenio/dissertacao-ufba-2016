11     package com.ibm.wala.cast.js.translator;
12     
13     import java.io.IOException;
14     import java.util.LinkedList;
15     import java.util.List;
16     
17     import com.ibm.wala.cast.ir.translator.TranslatorToCAst;
18     import com.ibm.wala.cast.tree.CAstEntity;
19     import com.ibm.wala.cast.tree.impl.CAstImpl;
20     import com.ibm.wala.cast.tree.rewrite.CAstRewriterFactory;
21     import com.ibm.wala.cast.tree.rewrite.CAstRewriter.CopyKey;
22     import com.ibm.wala.cast.tree.rewrite.CAstRewriter.RewriteContext;
23     import com.ibm.wala.classLoader.SourceFileModule;
24     import com.ibm.wala.classLoader.SourceModule;
25     
26     public class CAstRhinoTranslator implements TranslatorToCAst {
27       private final List<CAstRewriterFactory> rewriters = new LinkedList<CAstRewriterFactory>();
28       private final SourceModule M;
29       private final boolean replicateForDoLoops;
30         
31       public CAstRhinoTranslator(SourceModule M, boolean replicateForDoLoops) {
32         this.M = M;
33         this.replicateForDoLoops = replicateForDoLoops;
34        }
35     
36       @Override
37       public <C extends RewriteContext<K>, K extends CopyKey<K>> void addRewriter(CAstRewriterFactory<C, K> factory, boolean prepend) {
38         if(prepend)
39           rewriters.add(0, factory);
40         else
41           rewriters.add(factory);
42       }  
43     
44       @Override
45       public CAstEntity translateToCAst() throws IOException, Error {
46         String N;
47         if (M instanceof SourceFileModule) {
48           N = ((SourceFileModule) M).getClassName();
49         } else {
50           N = M.getName();
51         }
52     
53         CAstImpl Ast = new CAstImpl();
54         CAstEntity entity = new RhinoToAstTranslator(Ast, M, N, replicateForDoLoops).translateToCAst();
55         for(CAstRewriterFactory rwf : rewriters)
56           entity = rwf.createCAstRewriter(Ast).rewrite(entity);
57         return entity;
58       }
59     
60     }
