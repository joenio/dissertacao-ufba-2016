11     package com.ibm.wala.shrikeBT;
12     
13     /**
14      * This class represents comparisons between floats, longs and doubles.
15      */
16     final public class ComparisonInstruction extends Instruction implements IComparisonInstruction {
17       protected ComparisonInstruction(short opcode) {
18         super(opcode);
19       }
20     
21       private final static ComparisonInstruction preallocatedLCMP = new ComparisonInstruction(OP_lcmp);
22     
23       private final static ComparisonInstruction[] preallocatedFloatingCompares = preallocateFloatingCompares();
24     
25       private static ComparisonInstruction[] preallocateFloatingCompares() {
26         ComparisonInstruction[] r = new ComparisonInstruction[OP_dcmpg - OP_fcmpl + 1];
27         for (short i = OP_fcmpl; i <= OP_dcmpg; i++) {
28           r[i - OP_fcmpl] = new ComparisonInstruction(i);
29         }
30         return r;
31       }
32     
33       public static ComparisonInstruction make(String type, Operator operator) throws IllegalArgumentException {
34         int t = Util.getTypeIndex(type);
35         switch (t) {
36         case TYPE_long_index:
37           if (operator != Operator.CMP) {
38             throw new IllegalArgumentException("Operator " + operator + " is not a valid comparison operator for longs");
39           } else {
40             return preallocatedLCMP;
41           }
42         case TYPE_float_index:
43         case TYPE_double_index:
44           if (operator == Operator.CMP) {
45             throw new IllegalArgumentException("Operator " + operator + " is not a valid comparison operator for floating point values");
46           } else {
47             return preallocatedFloatingCompares[(operator.ordinal() - Operator.CMPL.ordinal()) + (t - TYPE_float_index) * 2];
48           }
49         default:
50           throw new IllegalArgumentException("Type " + type + " cannot be compared");
51         }
52       }
53     
54       @Override
55       public boolean equals(Object o) {
56         if (o instanceof ComparisonInstruction) {
57           ComparisonInstruction i = (ComparisonInstruction) o;
58           return i.opcode == opcode;
59         } else {
60           return false;
61         }
62       }
63     
64       /**
65        * @return OPR_cmp (for long), OPR_cmpl, or OPR_cmpg (for float and double)
66        */
67       @Override
68       public Operator getOperator() {
69         switch (opcode) {
70         case OP_lcmp:
71           return Operator.CMP;
72         case OP_fcmpl:
73         case OP_dcmpl:
74           return Operator.CMPL;
75         case OP_dcmpg:
76         case OP_fcmpg:
77           return Operator.CMPG;
78         default:
79           throw new Error("Unknown opcode");
80         }
81       }
82     
83       @Override
84       public String getType() {
85         switch (opcode) {
86         case OP_lcmp:
87           return TYPE_long;
88         case OP_fcmpg:
89         case OP_fcmpl:
90           return TYPE_float;
91         case OP_dcmpl:
92         case OP_dcmpg:
93           return TYPE_double;
94         default:
95           throw new Error("Unknown opcode");
96         }
97       }
98     
99       @Override
100       public int hashCode() {
101         return opcode + 1391901;
102       }
103     
104       @Override
105       public int getPoppedCount() {
106         return 2;
107       }
108     
109       @Override
110       public String getPushedType(String[] types) {
111         return Constants.TYPE_boolean;
112       }
113     
114       @Override
115       public byte getPushedWordSize() {
116         return 1;
117       }
118     
119       @Override
120       public void visit(IInstruction.Visitor v) throws NullPointerException {
121         v.visitComparison(this);
122       }
123     
124       @Override
125       public String toString() {
126         return "Comparison(" + getType() + "," + getOperator() + ")";
127       }
128     
129       @Override
130       public boolean isPEI() {
131         return false;
132       }
133     }
