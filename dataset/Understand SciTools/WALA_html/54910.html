26     /**
27      * A modifiable control flow graph.
28      * 
29      * @author Juergen Graf <graf@kit.edu>
30      *
31      */
32     public class MutableCFG<X, T extends IBasicBlock<X>> extends SparseNumberedGraph<T> implements ControlFlowGraph<X, T> {
33     
34       private final ControlFlowGraph<X, T> orig;
35       
36       private MutableCFG(final ControlFlowGraph<X, T> orig) {
37         this.orig = orig;
38       }
39       
40       public static <I, T extends IBasicBlock<I>> MutableCFG<I, T> copyFrom(ControlFlowGraph<I, T> cfg) {
41         MutableCFG<I, T> mutable = new MutableCFG<I, T>(cfg);
42         
43         for (T node : cfg) {
44           mutable.addNode(node);
45         }
46         
47         for (T node : cfg) {
48           for (T succ : cfg.getNormalSuccessors(node)) {
49             mutable.addEdge(node, succ);
50           }
51     
52           for (T succ : cfg.getExceptionalSuccessors(node)) {
53             mutable.addEdge(node, succ);
54           }
55         }
56         
57         return mutable;
58       }
59     
60       @Override
61       public T entry() {
62         return orig.entry();
63       }
64     
65       @Override
66       public T exit() {
67         return orig.exit();
68       }
69     
70       // slow
71       @Override
72       public BitVector getCatchBlocks() {
73         final BitVector bvOrig = orig.getCatchBlocks();
74         final BitVector bvThis = new BitVector();
75         
76         for (final T block : this) {
77           bvThis.set(block.getNumber());
78         }
79         
80         bvThis.and(bvOrig);
81         
82         return bvThis;
83       }
84     
85       @Override
86       public T getBlockForInstruction(int index) {
87         final T block =  orig.getBlockForInstruction(index);
88         
89         return (containsNode(block) ? block : null);
90       }
91     
92       @Override
93       public X[] getInstructions() {
94         return orig.getInstructions();
95       }
96     
97       @Override
98       public int getProgramCounter(int index) {
99         return orig.getProgramCounter(index);
100       }
101     
102       @Override
103       public IMethod getMethod() {
104         return orig.getMethod();
105       }
106     
107       @Override
108       public List<T> getExceptionalSuccessors(T b) {
109         final List<T> origSucc = orig.getExceptionalSuccessors(b);
110         final IntSet allSuccs = this.getSuccNodeNumbers(b);
111         final List<T> thisSuccs = new LinkedList<T>();
112         
113         for (final T block : origSucc) {
114           if (allSuccs.contains(block.getNumber())) {
115             thisSuccs.add(block);
116           }
117         }
118         
119         return thisSuccs;
120       }
121     
122       @Override
123       public Collection<T> getNormalSuccessors(T b) {
124         final List<T> excSuccs = getExceptionalSuccessors(b);
125         final List<T> thisSuccs = new LinkedList<T>();
126         
127         final Iterator<T> succs = getSuccNodes(b);
128         while (succs.hasNext()) {
129           final T succ = succs.next();
130           if (!excSuccs.contains(succ)) {
131             thisSuccs.add(succ);
132           }
133         }
134         
135         return thisSuccs;
136       }
137     
138       @Override
139       public Collection<T> getExceptionalPredecessors(T b) {
140         final Collection<T> origPreds = orig.getExceptionalPredecessors(b);
141         final IntSet allPreds = this.getPredNodeNumbers(b);
142         final List<T> thisPreds = new LinkedList<T>();
143         
144         for (final T block : origPreds) {
145           if (allPreds.contains(block.getNumber())) {
146             thisPreds.add(block);
147           }
148         }
149         
150         return thisPreds;
151       }
152     
153       @Override
154       public Collection<T> getNormalPredecessors(T b) {
155         final Collection<T> excPreds = getExceptionalPredecessors(b);
156         final List<T> thisPreds = new LinkedList<T>();
157         
158         final Iterator<T> preds = getPredNodes(b);
159         while (preds.hasNext()) {
160           final T pred = preds.next();
161           if (!excPreds.contains(pred)) {
162             thisPreds.add(pred);
163           }
164         }
165         
166         return thisPreds;
167       }
168       
169     }
