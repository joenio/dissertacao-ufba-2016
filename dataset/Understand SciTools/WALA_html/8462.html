23     /**
24      * A basic context selector that ignores context.
25      */
26     public class ContextInsensitiveSelector implements ContextSelector {

28       @Override
29       public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] receiver) {
30         return Everywhere.EVERYWHERE;
31       }
32     
33       @Override

35         return EmptyIntSet.instance;
36       }

38     }
