31       protected void verifyCFGAssertions(CallGraph CG, Object[][] assertionData) {
32         for(Object[] dat : assertionData) {
33           String function = (String) dat[0];

36             SSACFG cfg = N.getIR().getControlFlowGraph();

38               SSACFG.BasicBlock bb = cfg.getNode(i);

40               for (int j = 0; j < edges[i].length; j++) {
41                 Assert.assertTrue(cfg.hasEdge(bb, cfg.getNode(edges[i][j])));
42               }
43             }
44           }

52             if (N.getMethod() instanceof AstMethod) {
53               AstMethod M = (AstMethod) N.getMethod();
54               SSAInstruction[] insts = N.getIR().getInstructions();
55               insts: for(int i = 0; i < insts.length; i++) {
56                 SSAInstruction inst = insts[i];
57                 if (inst != null) {
58                   Position pos = M.getSourcePosition(i);
59                   if (pos != null) {
60                     String fileName = pos.getURL().toString();
61                     if (fileName.lastIndexOf('/') >= 0) {
62                       fileName = fileName.substring(fileName.lastIndexOf('/')+1);
63                     }
64                     for(int j = 0; j < assertionData.length; j++) {
65                       String file = (String) assertionData[j][1];
66                       if (file.indexOf('/') >= 0) {

128       protected void verifyGraphAssertions(CallGraph CG, Object[][] assertionData) {
129         // System.err.println(CG);
130     
131         if (assertionData == null) {
132           return;
133         }
134         
135         for (int i = 0; i < assertionData.length; i++) {
136     
137           check_target: for (int j = 0; j < ((String[]) assertionData[i][1]).length; j++) {
138             Iterator srcs = (assertionData[i][0] instanceof String) ? getNodes(CG, (String) assertionData[i][0]).iterator()
139                 : new NonNullSingletonIterator<CGNode>(CG.getFakeRootNode());
140     
141             Assert.assertTrue("cannot find " + assertionData[i][0], srcs.hasNext());
142     
143             boolean checkAbsence = false;
144             String targetName = ((String[]) assertionData[i][1])[j];
145             if (targetName.startsWith("!")) {
146               checkAbsence = true;
147               targetName = targetName.substring(1);
148             }
149     
150             while (srcs.hasNext()) {
151               CGNode src = (CGNode) srcs.next();
152               for (Iterator sites = src.iterateCallSites(); sites.hasNext();) {
153                 CallSiteReference sr = (CallSiteReference) sites.next();
154                
155                 Iterator dsts = getNodes(CG, targetName).iterator();
156                 if (! checkAbsence) {
157                   Assert.assertTrue("cannot find " + targetName, dsts.hasNext());
158                 }
159                 
160                 while (dsts.hasNext()) {
161                   CGNode dst = (CGNode) dsts.next();
162                   for (Iterator tos = CG.getPossibleTargets(src, sr).iterator(); tos.hasNext();) {
163                     if (tos.next().equals(dst)) {
164                       if (checkAbsence) {
165                         System.err.println(("found unexpected " + src + " --> " + dst + " at " + sr));
166                         Assert.assertTrue("found edge " + assertionData[i][0] + " ---> " + targetName, false);
167                       } else {
168                         System.err.println(("found expected " + src + " --> " + dst + " at " + sr));
169                         continue check_target;
170                       }
171                     }
172                   }
173                 }
174               }
175             }
176     
177             System.err.println("cannot find edge " + assertionData[i][0] + " ---> " + targetName);
178             Assert.assertTrue("cannot find edge " + assertionData[i][0] + " ---> " + targetName, checkAbsence);
179           }
180         }
181       }

184       /**
185        * Verifies that none of the nodes that match the source description has an edge to any of the nodes that match the destination
186        * description. (Used for checking for false connections in the callgraph)
187        * 
188        * @param CG
189        * @param sourceDescription
190        * @param destDescription
191        */
192       protected void verifyNoEdges(CallGraph CG, String sourceDescription, String destDescription) {
193         Collection sources = getNodes(CG, sourceDescription);
194         Collection dests = getNodes(CG, destDescription);
195         for (Object source : sources) {
196           for (Object dest : dests) {
197             for (Iterator<CGNode> i = CG.getSuccNodes((CGNode) source); i.hasNext();) {
198               if (i.next().equals(dest)) {
199                 Assert.fail("Found a link from " + source + " to " + dest);
200               }
201             }
202           }
203         }
204       }
