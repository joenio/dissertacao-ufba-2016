13     import java.io.BufferedWriter;
14     import java.io.OutputStreamWriter;
15     import java.io.Writer;
16     import java.util.HashMap;

19     import com.ibm.wala.shrikeBT.Constants;

25     /**
26      * @author roca
27      */
28     public class InterfaceAnalyzer {
29       final static class TypeStats {
30         int totalOccurrences;
31     
32         int methodOccurrences;
33     
34         int publicMethodOccurrences;
35     
36         int foreignPublicMethodOccurrences;
37     
38         int lastMUID;
39       }
40     
41       final static HashMap<String, TypeStats> typeStats = new HashMap<String, TypeStats>();
42     
43       public static void main(String[] args) throws Exception {
44         OfflineInstrumenter instrumenter = new OfflineInstrumenter(true);
45     
46         Writer w = new BufferedWriter(new OutputStreamWriter(System.out));
47     
48         args = instrumenter.parseStandardArgs(args);
49     
50         instrumenter.beginTraversal();
51         ClassInstrumenter ci;
52         while ((ci = instrumenter.nextClass()) != null) {
53           doClass(ci.getReader());
54         }
55         instrumenter.close();
56     
57         w.write("Type\t# Total\t# Method\t# Public Method\t# Public Method as Foreign\n");
58         for (Iterator<String> i = typeStats.keySet().iterator(); i.hasNext();) {
59           String k = i.next();
60           TypeStats t = typeStats.get(k);
61           w.write(k + "\t" + t.totalOccurrences + "\t" + t.methodOccurrences + "\t" + t.publicMethodOccurrences + "\t"
62               + t.foreignPublicMethodOccurrences + "\n");
63         }
64         w.close();
65       }
66     
67       static int methodUID = 0;
68     
69       /**
70        * @param reader
71        */
72       private static void doClass(ClassReader reader) throws Exception {
73         if ((reader.getAccessFlags() & Constants.ACC_INTERFACE) != 0 && (reader.getAccessFlags() & Constants.ACC_PUBLIC) != 0) {
74           String cType = Util.makeType(reader.getName());
75           for (int m = 0; m < reader.getMethodCount(); m++) {
76             String sig = reader.getMethodType(m);
77             String[] params = Util.getParamsTypes(null, sig);
78             int flags = reader.getMethodAccessFlags(m);
79             int mUID = methodUID++;
80             for (int p = 0; p < params.length; p++) {
81               doType(flags, params[p], cType, mUID);
82             }
83             doType(flags, Util.getReturnType(sig), cType, mUID);
84           }
85         }
86       }
87     
88       private static void doType(int flags, String type, String containerType, int mUID) {
89         TypeStats t = typeStats.get(type);
90         if (t == null) {
91           t = new TypeStats();
92           typeStats.put(type, t);
93         }
94         t.totalOccurrences++;
95         if (t.lastMUID != mUID) {
96           t.methodOccurrences++;
97           if ((flags & Constants.ACC_PUBLIC) != 0) {
98             t.publicMethodOccurrences++;
99             String elemType = type;
100             while (Util.isArrayType(elemType)) {
101               elemType = elemType.substring(1);
102             }
103             if (!Util.isPrimitiveType(elemType) && !packagePart(elemType, 2).equals(packagePart(containerType, 2))) {
104               t.foreignPublicMethodOccurrences++;
105             }
106           }
107         }
108         t.lastMUID = mUID;
109       }
110     
111       private static String packagePart(String t, int count) {
112         String c = Util.makeClass(t);
113         int lastDot = -1;
114         for (int i = 0; i < count; i++) {
115           int dot = c.indexOf('.', lastDot + 1);
116           if (dot < 0) {
117             return c;
118           }
119           lastDot = dot;
120         }
121         return c.substring(0, lastDot);
122       }
123     }
