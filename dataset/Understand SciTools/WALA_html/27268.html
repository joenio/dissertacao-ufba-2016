36       /**
37        * Perform a DFS starting with a particular node and return the set of all nodes visited.
38        * 
39        * @param C collection of nodes to start from
40        * @param filter only traverse nodes that need this filter
41        * @throws IllegalArgumentException if C is null
42        */
43       @SuppressWarnings("serial")
44       public static <T> Collection<T> getReachableNodes(final Graph<T> G, Collection<? extends T> C, @SuppressWarnings("rawtypes") final Predicate filter) {
45         if (C == null) {
46           throw new IllegalArgumentException("C is null");
47         }
48         Iterator<T> dfs = new SlowDFSFinishTimeIterator<T>(G, C.iterator()) {
49     
50           @Override
51           protected Iterator<T> getConnected(T n) {
52             return new FilterIterator<T>(G.getSuccNodes(n), filter);
53           }
54         };
55         return Iterator2Collection.toSet(dfs);
56       }

58       /**
59        * Perform a DFS starting with a particular node set and return the set of all nodes visited.
60        * 
61        * @param G the graph containing n
62        * @return Set
63        * @throws IllegalArgumentException if C is null
64        */
65       public static <T> Set<T> getReachableNodes(Graph<T> G, Collection<? extends T> C) {
66         if (C == null) {
67           throw new IllegalArgumentException("C is null");
68         }
69         HashSet<T> result = HashSetFactory.make();
70         Iterator<T> dfs = iterateFinishTime(G, C.iterator());
71         while (dfs.hasNext()) {
72           result.add(dfs.next());
73         }
74         return result;
75       }

97       /**
98        * Perform a DFS of a graph starting with a specified node and return a sorted list of nodes. The nodes are sorted by depth first
99        * order.
100        * 
101        * @param G a graph
102        * @param n the initial node
103        * @return a sorted set of nodes in the graph in depth first order
104        */
105       public static <T> SortedSet<T> sortByDepthFirstOrder(Graph<T> G, T n) {
106         Map<T, Integer> order = HashMapFactory.make();
107         TreeSet<T> result = new TreeSet<T>(new DFSComparator<T>(order));
108     
109         Iterator<T> dfs = iterateFinishTime(G, new NonNullSingletonIterator<T>(n));
110         int i = 0;
111         while (dfs.hasNext()) {
112           T nxt = dfs.next();
113           order.put(nxt, new Integer(i++));
114           result.add(nxt);
115         }
116         return result;
117       }

119       /**
120        * Comparator class to order the nodes in the DFS according to the depth first order
121        */
122       static class DFSComparator<T> implements Comparator<T> {
123         final private Map<T, Integer> order;
124     
125         DFSComparator(Map<T, Integer> order) {
126           this.order = order;
127         }
128     
129         @Override
130         public int compare(T o1, T o2) {
131           // throws an exception if either argument is not a Node object
132           if (o1 == o2) {
133             return 0;
134           }
135           Integer t1 = order.get(o1);
136           Integer t2 = order.get(o2);
137           // throws an exception if either node has not been ordered
138           return (t1.intValue() - t2.intValue());
139         }
140       }

142       /**
143        * @param G
144        * @return iterator of nodes of G in order of DFS discover time
145        */
146       public static <T> DFSDiscoverTimeIterator<T> iterateDiscoverTime(Graph<T> G) {
147         if (G instanceof NumberedGraph) {
148           return new NumberedDFSDiscoverTimeIterator<T>((NumberedGraph<T>) G);
149         } else {
150           return new SlowDFSDiscoverTimeIterator<T>(G);
151         }
152       }
