38     /**
39      * A cheap, context-insensitive slicer based on reachability over a custom SDG.
40      * 
41      * This is a prototype implementation; not tuned.
42      * 
43      * Currently supports backward slices only.
44      * 
45      * TODO: Introduce a slicer interface common between this and the CS slicer. TODO: This hasn't been tested much. Need regression
46      * tests.
47      */
48     public class CISlicer {
49     
50       /**
51        * the dependence graph used for context-insensitive slicing
52        */
53       private final Graph<Statement> depGraph;
54     
55       public CISlicer(CallGraph cg, PointerAnalysis<InstanceKey> pa, DataDependenceOptions dOptions, ControlDependenceOptions cOptions) {
56         this(cg, pa, ModRef.make(), dOptions, cOptions);
57       }
58     
59       public CISlicer(CallGraph cg, PointerAnalysis<InstanceKey> pa, ModRef modRef, DataDependenceOptions dOptions, ControlDependenceOptions cOptions)
60           throws IllegalArgumentException {
61         if (dOptions == null) {
62           throw new IllegalArgumentException("dOptions == null");
63         }
64         if (dOptions.equals(DataDependenceOptions.NO_BASE_PTRS) || dOptions.equals(DataDependenceOptions.FULL)) {
65           throw new IllegalArgumentException("Heap data dependences requested in CISlicer!");
66         }
67     
68         SDG sdg = new SDG(cg, pa, modRef, dOptions, cOptions, null);
69     
70         Map<Statement, Set<PointerKey>> mod = scanForMod(sdg, pa, modRef);
71         Map<Statement, Set<PointerKey>> ref = scanForRef(sdg, pa, modRef);
72     
73         depGraph = GraphInverter.invert(new CISDG(sdg, mod, ref));
74     
75       }
76     
77       public CISlicer(final SDG sdg, final PointerAnalysis<InstanceKey> pa, final ModRef modRef) {
78         Map<Statement, Set<PointerKey>> mod = scanForMod(sdg, pa, modRef);
79         Map<Statement, Set<PointerKey>> ref = scanForRef(sdg, pa, modRef);
80     
81         depGraph = GraphInverter.invert(new CISDG(sdg, mod, ref));
82       }
83     
84       public Collection<Statement> computeBackwardThinSlice(Statement seed) {
85         Collection<Statement> slice = DFS.getReachableNodes(depGraph, Collections.singleton(seed));
86         return slice;
87       }
88     
89       public Collection<Statement> computeBackwardThinSlice(Collection<Statement> seeds) {
90         Collection<Statement> slice = DFS.getReachableNodes(depGraph, seeds);
91         return slice;
92       }

94       /**
95        * Compute the set of pointer keys each statement mods
96        */
97       public static Map<Statement, Set<PointerKey>> scanForMod(SDG sdg, PointerAnalysis<InstanceKey> pa) {
98         return scanForMod(sdg, pa, false, ModRef.make());
99       }
100     
101       /**
102        * Compute the set of pointer keys each statement refs
103        */
104       public static Map<Statement, Set<PointerKey>> scanForRef(SDG sdg, PointerAnalysis<InstanceKey> pa) {
105         if (sdg == null) {
106           throw new IllegalArgumentException("null sdg");
107         }
108         return scanForRef(sdg, pa, ModRef.make());
109       }
110     
111       /**
112        * Compute the set of pointer keys each statement mods
113        */
114       public static Map<Statement, Set<PointerKey>> scanForMod(SDG sdg, PointerAnalysis<InstanceKey> pa, ModRef modRef) {
115         return scanForMod(sdg, pa, false, modRef);
116       }
117     
118       /**
119        * Compute the set of pointer keys each statement mods. Be careful to avoid eager PDG construction here! That means .. don't
120        * iterate over SDG statements!
121        */
122       public static Map<Statement, Set<PointerKey>> scanForMod(SDG sdg, PointerAnalysis<InstanceKey> pa, boolean ignoreAllocHeapDefs, ModRef modRef) {
123         if (pa == null) {
124           throw new IllegalArgumentException("null pa");
125         }
126         ExtendedHeapModel h = new DelegatingExtendedHeapModel(pa.getHeapModel());
127         Map<Statement, Set<PointerKey>> result = HashMapFactory.make();
128         for (CGNode n : sdg.getCallGraph()) {
129           IR ir = n.getIR();
130           if (ir != null) {
131             for (int i = 0; i < ir.getInstructions().length; i++) {
132               SSAInstruction st = ir.getInstructions()[i];
133               if (st != null) {
134                 Set<PointerKey> mod = modRef.getMod(n, h, pa, st, null, ignoreAllocHeapDefs);
135                 if (!mod.isEmpty()) {
136                   NormalStatement normal = new NormalStatement(n, i);
137                   result.put(normal, mod);
138                 }
139               }
140             }
141           }
142         }
143         return result;
144       }
145     
146       /**
147        * Compute the set of PointerKeys each statement refs.Be careful to avoid eager PDG construction here! That means .. don't iterate
148        * over SDG statements!
149        */
150       public static Map<Statement, Set<PointerKey>> scanForRef(SDG sdg, PointerAnalysis<InstanceKey> pa, ModRef modRef) {
151         if (pa == null) {
152           throw new IllegalArgumentException("null pa");
153         }
154         ExtendedHeapModel h = new DelegatingExtendedHeapModel(pa.getHeapModel());
155         Map<Statement, Set<PointerKey>> result = HashMapFactory.make();
156         for (CGNode n : sdg.getCallGraph()) {
157           IR ir = n.getIR();
158           if (ir != null) {
159             for (int i = 0; i < ir.getInstructions().length; i++) {
160               SSAInstruction st = ir.getInstructions()[i];
161               if (st != null) {
162                 Set<PointerKey> mod = modRef.getRef(n, h, pa, st, null);
163                 if (!mod.isEmpty()) {
164                   NormalStatement normal = new NormalStatement(n, i);
165                   result.put(normal, mod);
166                 }
167               }
168             }
169           }
170         }
171         return result;
172       }
