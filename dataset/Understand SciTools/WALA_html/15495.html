66       /**
67        * @param args
68        *          -appJar [jar file name] -sig [method signature] The "jar file
69        *          name" should be something like "c:/temp/testdata/java_cup.jar" The
70        *          signature should be something like "java_cup.lexer.advance()V"
71        * @throws IOException 
72        */
73       public static Process run(String[] args) throws IOException {
74         validateCommandLine(args);
75         return run(args[1], args[3]);
76       }
77     
78       /**
79        * @param appJar
80        *          should be something like "c:/temp/testdata/java_cup.jar"
81        * @param methodSig
82        *          should be something like "java_cup.lexer.advance()V"
83        * @throws IOException 
84        */
85       public static Process run(String appJar, String methodSig) throws IOException {
86         try {
87           if (PDFCallGraph.isDirectory(appJar)) {
88             appJar = PDFCallGraph.findJarFiles(new String[] { appJar });
89           }
90           AnalysisScope scope = AnalysisScopeReader.makeJavaBinaryAnalysisScope(appJar, (new FileProvider()).getFile(CallGraphTestUtil.REGRESSION_EXCLUSIONS));
91     
92           ClassHierarchy cha = ClassHierarchy.make(scope);
93     
94           MethodReference mr = StringStuff.makeMethodReference(methodSig);
95     
96           IMethod m = cha.resolveMethod(mr);
97           if (m == null) {
98             System.err.println("could not resolve " + mr);
99             throw new RuntimeException();
100           }
101           AnalysisOptions options = new AnalysisOptions();
102           options.getSSAOptions().setPiNodePolicy(SSAOptions.getAllBuiltInPiNodes());
103           AnalysisCache cache = new AnalysisCache();
104           IR ir = cache.getSSACache().findOrCreateIR(m, Everywhere.EVERYWHERE, options.getSSAOptions() );
105     
106           if (ir == null) {
107             Assertions.UNREACHABLE("Null IR for " + m);
108           }
109     
110           System.err.println(ir.toString());
111           ControlDependenceGraph<SSAInstruction, ISSABasicBlock> cdg = new ControlDependenceGraph<SSAInstruction, ISSABasicBlock>(ir.getControlFlowGraph());
112     
113           Properties wp = null;
114           try {
115             wp = WalaProperties.loadProperties();
116             wp.putAll(WalaExamplesProperties.loadProperties());
117           } catch (WalaException e) {
118             e.printStackTrace();
119             Assertions.UNREACHABLE();
120           }
121           String psFile = wp.getProperty(WalaProperties.OUTPUT_DIR) + File.separatorChar + PDFControlDependenceGraph.PDF_FILE;
122           String dotFile = wp.getProperty(WalaProperties.OUTPUT_DIR) + File.separatorChar
123               + PDFTypeHierarchy.DOT_FILE;
124           String dotExe = wp.getProperty(WalaExamplesProperties.DOT_EXE);
125           String gvExe = wp.getProperty(WalaExamplesProperties.PDFVIEW_EXE);
126           
127           DotUtil.<ISSABasicBlock>dotify(cdg, PDFViewUtil.makeIRDecorator(ir), dotFile, psFile, dotExe);
128     
129           return PDFViewUtil.launchPDFView(psFile, gvExe);
130     
131         } catch (WalaException e) {
132           // TODO Auto-generated catch block
133           e.printStackTrace();
134           return null;
135         }
136       }

138       /**
139        * Validate that the command-line arguments obey the expected usage.
140        * 
141        * Usage:
142        * <ul>
143        * <li> args[0] : "-appJar"
144        * <li> args[1] : something like "c:/temp/testdata/java_cup.jar"
145        * <li> args[2] : "-sig"
146        * <li> args[3] : a method signature like "java_cup.lexer.advance()V" </ul?
147        * 
148        * @param args
149        * @throws UnsupportedOperationException
150        *           if command-line is malformed.
151        */
152       static void validateCommandLine(String[] args) {
153         if (args.length != 4) {
154           throw new UnsupportedOperationException("must have at exactly 4 command-line arguments");
155         }
156         if (!args[0].equals("-appJar")) {
157           throw new UnsupportedOperationException("invalid command-line, args[0] should be -appJar, but is " + args[0]);
158         }
159         if (!args[2].equals("-sig")) {
160           throw new UnsupportedOperationException("invalid command-line, args[2] should be -sig, but is " + args[0]);
161         }
162       }
