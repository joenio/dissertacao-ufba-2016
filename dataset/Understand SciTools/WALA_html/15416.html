11     package com.ibm.wala.classLoader;
12     
13     import com.ibm.wala.ipa.callgraph.ContextItem;
14     import com.ibm.wala.shrikeCT.InvalidClassFileException;
15     import com.ibm.wala.types.Descriptor;
16     import com.ibm.wala.types.MethodReference;
17     import com.ibm.wala.types.Selector;
18     import com.ibm.wala.types.TypeReference;
19     
20     /**
21      * Basic interface for an object that represents a single Java method for analysis purposes.
22      */
23     public interface IMethod extends IMember, ContextItem {
24     
25       /**
26        * Is this method synchronized?
27        */
28       boolean isSynchronized();
29     
30       /**
31        * Is this method a class initializer?
32        */
33       boolean isClinit();
34     
35       /**
36        * Is this method an object initializer?
37        */
38       boolean isInit();
39     
40       /**
41        * Is this method native?
42        */
43       boolean isNative();
44     
45       /**
46        * Did someone synthesize this method? (As opposed to reading it from a class file)
47        */
48       boolean isSynthetic();
49     
50       /**
51        * Is this method abstract?
52        */
53       boolean isAbstract();
54     
55       /**
56        * Is this method private?
57        */
58       boolean isPrivate();
59     
60       /**
61        * Is this method protected?
62        */
63       boolean isProtected();
64     
65       /**
66        * Is this method public?
67        */
68       boolean isPublic();
69     
70       /**
71        * Is this method final?
72        */
73       boolean isFinal();
74       
75       /**
76        * Is this method a bridge method?  See JLS 3rd Edition 15.12.4.5
77        */
78       boolean isBridge();
79     
80       /**
81        * @return canonical MethodReference corresponding to this method
82        */
83       MethodReference getReference();
84     
85       /**
86        * @return true iff this method has at least one exception handler
87        */
88       boolean hasExceptionHandler();
89     
90       /**
91        * By convention, for a non-static method, getParameterType(0) is the this pointer
92        */
93       TypeReference getParameterType(int i);
94     
95       /**
96        * @return the name of the return type for this method
97        */
98       TypeReference getReturnType();
99     
100       /**
101        * Method getNumberOfParameters. This result includes the "this" pointer if applicable
102        */
103       int getNumberOfParameters();
104     
105       /**
106        * @return an array of the exception types declared by the throws clause for this method, or null if there are none
107        * @throws InvalidClassFileException
108        */
109       TypeReference[] getDeclaredExceptions() throws InvalidClassFileException, UnsupportedOperationException;
110     
111       /**
112        * @return the source line number corresponding to a particular bytecode index, or -1 if the information is not available.
113        */
114       int getLineNumber(int bcIndex);
115     /** BEGIN Custom change: precise positions */
116       
117       public interface SourcePosition extends Comparable {
118         int getFirstLine();
119         int getLastLine();
120         int getFirstCol();
121         int getLastCol();
122         int getFirstOffset();
123         int getLastOffset(); 
124       }
125       
126       SourcePosition getSourcePosition(int instructionIndex) throws InvalidClassFileException;
127     
128       SourcePosition getParameterSourcePosition(int paramNum) throws InvalidClassFileException;
129     /** END Custom change: precise positions */
130       
131       /**
132        * @return the (source code) name of the local variable of a given number at the specified program counter, or null if the
133        *         information is not available.
134        */
135       String getLocalVariableName(int bcIndex, int localNumber);
136     
137       /**
138        * something like: com.foo.bar.createLargeOrder(IILjava.lang.String;SLjava.sql.Date;)Ljava.lang.Integer;
139        */
140       public String getSignature();
141     
142       /**
143        * something like: foo(Ljava/langString;)Ljava/lang/Class;
144        */
145       public Selector getSelector();
146     
147       /**
148        * something like: (IILjava.lang.String;SLjava.sql.Date;)Ljava.lang.Integer;
149        */
150       Descriptor getDescriptor();
151     
152       /**
153        * @return true iff the local variable table information for this method is available
154        */
155       boolean hasLocalVariableTable();
156     }
