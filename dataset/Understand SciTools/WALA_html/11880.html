105     /**
106      *  The model to be executed at application start.
107      *
108      *  This method models the lifecycle of an Android Application. By doing so it calls all AndroidEntryPoints
109      *  set in the AnalysisOptions.
110      *  <p>
111      *  Between the calls to the AndroidEntryPoints special behavior is inserted. You can change that behavior
112      *  by implementing an AbstractAndroidModel or set one of the existing ones in the AnalysisOptions.
113      *
114      *  Additionally care of how types are instantiated is taken. You can change this behavior by setting the
115      *  IInstanciationBehavior in the AnalysisOptions.
116      *  
117      *  Smaller Models exist: 
118      *      * MiniModel calls all components of a specific type (for example all Activities)
119      *      * MicroModel calls a single specific component
120      *      * ExternalModel doesn't call anything but fiddles with the data on its own
121      *
122      *  All these Models are added to a synthetic AndroidModelClass.
123      *
124      *  @author Tobias Blaschke <code@tobiasblaschke.de>
125      */

167         /**
168          *  Generates the model on a sub-set of Entrypoints. 
169          *
170          *  Asks {@link #selectEntryPoint(AndroidEntryPoint)} for each EntryPoint known to the AndroidEntryPointManager,
171          *  if the EntryPoint should be included in the model. Then calls {@link #build(Atom, Iterable<? extends Entrypoint>)}
172          *  on these.
173          *
174          *  @param  name    The name the generated method will be known as
175          */
176         protected void build(Atom name) throws CancelException {
177             final List<AndroidEntryPoint> restrictedEntries = new ArrayList<AndroidEntryPoint>();
178     
179             for (AndroidEntryPoint ep: AndroidEntryPointManager.ENTRIES) {
180                 if (selectEntryPoint(ep)) {
181                     restrictedEntries.add(ep);
182                 }
183             }
184             build(name, restrictedEntries);
185         }

203         /**
204          *  Generate the SummarizedMethod for the model (in this.model).
205          *
206          *  The actual generated model depends on the on the properties of this overloaded class. Most 
207          *  generated methods should reside in the AndroidModelClass and take AndroidComponents as well
208          *  as some parameters (these marked REUSE) to the EntryPoints of the components.
209          *
210          *  Use {@link #getMethod()} to retrieve the method generated here or getMethodAs to get a version
211          *  which is wrapped to another signature.
212          *
213          *  @param  name            The name the generated method will be known as
214          *  @param  entrypoints     The functions to call additionally to boot-code and XXX
215          */
216         protected void build(Atom name, Collection<? extends AndroidEntryPoint> entrypoints) throws CancelException {
217     
218             // register
219             this.klass = cha.lookupClass(AndroidModelClass.ANDROID_MODEL_CLASS);
220            
221             if (this.klass == null) {
222                 // add to cha
223                 
224                 this.klass = AndroidModelClass.getInstance(cha);
225                 cha.addClass(this.klass);
226             }
227     
228             this.reuseParameters = new ReuseParameters(this.instanceBehavior, this);
229     //        this.instructionFactory = new JavaInstructionFactory(); // TODO: TSIF
230            
231             // Complete the signature of the method
232             reuseParameters.collectParameters(entrypoints);
233             this.mRef = reuseParameters.toMethodReference(null);
234             this.modelAcc = new ParameterAccessor(this.mRef, (! isStatic()));
235             this.paramManager = new SSAValueManager(modelAcc);
236     
237             final Selector selector = this.mRef.getSelector();
238             final AndroidModelClass mClass = AndroidModelClass.getInstance(cha);
239             if (mClass.containsMethod(selector)) {
240                 
241                 assert (mClass.getMethod(selector) instanceof SummarizedMethod);
242                 this.model = (SummarizedMethod) mClass.getMethod(selector);
243                 return;
244             }
245             this.body = new VolatileMethodSummary(new MethodSummary(this.mRef));
246             this.body.setStatic(true);
247     
248             this.labelSpecial = AndroidEntryPointManager.MANAGER.makeModelBehavior(this.body, new TypeSafeInstructionFactory(cha),
249                     this.paramManager, entrypoints);
250     
251             this.monitor = AndroidEntryPointManager.MANAGER.getProgressMonitor();
252             this.maxProgress = entrypoints.size();
253     
254             AndroidModel.doBoot &= AndroidEntryPointManager.MANAGER.getDoBootSequence();
255     
256             // BUILD
257             this.monitor.beginTask("Building " + name, this.maxProgress);
258             populate(entrypoints);
259     
260     
261             assert (cha.lookupClass(AndroidModelClass.ANDROID_MODEL_CLASS) != null) : "Adding the class failed!";
262     
263             if (this.klass == null) {
264                 throw new IllegalStateException("Could not find ANDROID_MODEL_CLASS in cha.");
265             }
266     
267             this.body.setLocalNames(this.paramManager.makeLocalNames());
268             this.model = new SummarizedMethodWithNames(this.mRef, this.body, this.klass) {
269                 @Override
270                 public TypeReference getParameterType (int i) {
271                     IClassHierarchy cha = getClassHierarchy();
272                     TypeReference tRef = super.getParameterType(i);
273     
274                     if (tRef.isClassType()) {
275                         if (cha.lookupClass(tRef) != null) {
276                             return tRef;
277                         } else {
278                             for (IClass c : cha) {
279                                 if (c.getName().toString().equals(tRef.getName().toString())) {
280                                     return c.getReference();
281                                 }
282                             }
283                         }
284                         return tRef;
285                         //throw new IllegalStateException("Error looking up " + tRef);
286                     } else {
287                         return tRef;
288                     }
289                 }
290             };
291         
292             this.built = true;
293         }

335         /**
336          *  Add Instructions to the model.
337          *
338          *  {@link #build(Iterable<?extends Entrypoint>)} prepares the MethodSummary, then calls populate() to
339          *  add the instructions, then finishes the model. Populate is only an extra function to shorten build(),
340          *  calling it doesn't make sense in an other context.
341          */
342         private void populate(Iterable<? extends AndroidEntryPoint> entrypoints) throws CancelException {
343             assert (! built) : "You can only build once";
344             int currentProgress = 0;
345     
346             final TypeSafeInstructionFactory tsif = new TypeSafeInstructionFactory(this.cha);
347             final Instantiator instantiator = new Instantiator (this.body, tsif, this.paramManager, this.cha, this.mRef, this.scope);
348             boolean enteredASection = false;
349             //
350             //  Add preparing code to the model
351             //
352             if (AndroidModel.doBoot) {
353     //            final Set<Parameter> allActivities = new HashSet<Parameter>(modelAcc.allExtend(AndroidTypes.ActivityName, getClassHierarchy()));
354                 //assert(allActivities.size() > 0) : "There are no Activities in the Model"; // XXX
355     //            final IntentStarters.StartInfo toolInfo = IntentStarters.StartInfo.makeContextFree(null);
356     //            final AndroidStartComponentTool tool = new AndroidStartComponentTool(this.cha, this.mRef, toolInfo.getFlags(),
357     //                    /* caller */ null, tsif, modelAcc, this.paramManager, this.body, /* self */ null, toolInfo, /* callerNd */ null);
358                 final SSAValue application;
359                 {
360                     final SSAValue tmpApp = modelAcc.firstExtends(AndroidTypes.ApplicationName, this.cha);
361                     if (tmpApp != null) {
362                         application = tmpApp;
363                     } else {
364                         // Generate a real one?
365                         
366                         application = paramManager.getUnmanaged(AndroidTypes.Application, "app");
367                         this.body.addConstant(application.getNumber(), new ConstantValue(null));
368                         application.setAssigned();
369                     }
370                 }
371                 final SSAValue nullIntent;
372                 {
373                     nullIntent = paramManager.getUnmanaged(AndroidTypes.Intent, "nullIntent");
374                     this.body.addConstant(nullIntent.getNumber(), new ConstantValue(null));
375                     nullIntent.setAssigned();
376                 }
377                 final SSAValue nullBinder;
378                 {
379                     nullBinder = paramManager.getUnmanaged(AndroidTypes.IBinder, "nullBinder");
380                     this.body.addConstant(nullBinder.getNumber(), new ConstantValue(null));
381                     nullBinder.setAssigned();
382                 }
383     
384                 
385                 {
386                     final AndroidBoot boot = new AndroidBoot(null); 
387                     boot.addBootCode(tsif, null, paramManager, this.body);
388                     //tool.attachActivities(allActivities, application, boot.getMainThread(), /* Should be application context TODO */
389                     //        boot.getPackageContext(), nullBinder, nullIntent); 
390                 }
391     
392                 // TODO: Assign context to the other components
393             }
394             
395             
396     
397             for (final AndroidEntryPoint ep : entrypoints) {
398                 this.monitor.subTask(ep.getMethod().getReference().getSignature() );
399                 
400                 if (! selectEntryPoint(ep)) {
401                     assert(false): "The ep should not reach here!";
402                     
403                     currentProgress++;
404                     continue;
405                 }
406     
407                 //
408                 //  Is special handling to be inserted?
409                 //
410                 if (this.labelSpecial.hadSectionSwitch(ep.order)) {
411                     
412                     this.labelSpecial.enter(ep.getSection(), body.getNextProgramCounter());
413                     enteredASection = true;
414                 }
415     
416                 //
417                 //  Collect arguments to ep
418                 //  if there are multiple paramses call the entrypoint multiple times
419                 //
420                 List<List<SSAValue>> paramses = new ArrayList<List<SSAValue>>(1);
421                 {
422                     final List<Integer> mutliTypePositions = new ArrayList<Integer>();
423                     { // Add single-type parameters and collect positions for multi-type
424                         final List<SSAValue> params = new ArrayList<SSAValue>(ep.getNumberOfParameters());
425                         paramses.add(params);
426     
427                         for (int i = 0; i < ep.getNumberOfParameters(); ++i) {
428                             if (ep.getParameterTypes(i).length != 1) {
429                                  mutliTypePositions.add(i);
430                                 params.add(null); // will get set later
431                             } else {
432                                 for (final TypeReference type : ep.getParameterTypes(i)) {
433                                     if (this.instanceBehavior.getBehavior(type.getName(), ep.getMethod(), null) == InstanceBehavior.REUSE) {
434                                         params.add(this.paramManager.getCurrent(new TypeKey(type.getName())));
435                                     } else if (type.isPrimitiveType()) {
436                                         params.add(paramManager.getUnmanaged(type, "p"));
437                                     } else {
438                                         // It is an CREATE parameter
439                                         final boolean asManaged = false;
440                                         final VariableKey key = null;   // auto-generates UniqueKey
441                                         final Set<SSAValue> seen = null; 
442                                         params.add(instantiator.createInstance(type, asManaged, key, seen));
443                                     }
444                                 }
445                             }
446                         }
447                     }
448     
449                     // Now for the mutliTypePositions: we'll build the Cartesian product for these
450                     for (int positionInMutliTypePosition = 0; positionInMutliTypePosition < mutliTypePositions.size(); ++positionInMutliTypePosition) {
451                         final Integer multiTypePosition = mutliTypePositions.get(positionInMutliTypePosition);
452                         final TypeReference[] typesOnPosition = ep.getParameterTypes(multiTypePosition);
453                         final int typeCountOnPosition = typesOnPosition.length;
454     
455                         { // Extend the list size to hold the product
456                             final List<List<SSAValue>> new_paramses = new ArrayList<List<SSAValue>>(paramses.size() * typeCountOnPosition);
457     
458                             for (int i = 0; i < typeCountOnPosition; ++i) {
459                                 //new_paramses.addAll(paramses); *grrr* JVM! You could copy at least null - but noooo...
460                                 for (final List<SSAValue> params : paramses) {
461                                     final List<SSAValue> new_params = new ArrayList<SSAValue>(params.size());
462                                     new_params.addAll(params);
463                                     new_paramses.add(new_params);
464                                 }
465                             }
466                             paramses = new_paramses;
467                         }
468     
469                         { // set the current multiTypePosition
470                             for (int i = 0; i < paramses.size(); ++i) {
471                                 final List<SSAValue> params = paramses.get(i);
472                                 assert(params.get(multiTypePosition) == null) : "Expected null, got " + params.get(multiTypePosition) + " iter " + i;
473     
474                                 // XXX: This could be faster, but well...
475                                 final TypeReference type = typesOnPosition[(i * (positionInMutliTypePosition + 1)) % typeCountOnPosition];
476     
477                                 if (this.instanceBehavior.getBehavior(type.getName(), ep.getMethod(), null) == InstanceBehavior.REUSE) {
478                                     params.set(multiTypePosition, this.paramManager.getCurrent(new TypeKey(type.getName())));
479                                 } else if (type.isPrimitiveType()) {
480                                     params.set(multiTypePosition, paramManager.getUnmanaged(type, "p"));
481                                 } else {
482                                     // It is an CREATE parameter
483                                     final boolean asManaged = false;
484                                     final VariableKey key = null;   // auto-generates UniqueKey
485                                     final Set<SSAValue> seen = null; 
486                                     params.set(multiTypePosition, instantiator.createInstance(type, asManaged, key, seen));
487                                 }
488                             }
489                         }
490                     }
491                 }
492     
493                 /*{ // DEBUG
494                     if (paramses.size() > 1) {
495                         System.out.println("\n\nParamses on " + ep.getMethod().getSignature() + ":");
496                         for (final List<SSAValue> params : paramses) {
497                             System.out.println("\t" + params);
498                         }
499                     }
500                 } // */
501     
502                 //
503                 //  Insert the call optionally handling its return value
504                 //
505                 for (final List<SSAValue> params : paramses) {
506     
507                     final int callPC = body.getNextProgramCounter();
508                     final CallSiteReference site = ep.makeSite(callPC);
509                     final SSAAbstractInvokeInstruction invokation;
510                     final SSAValue exception = paramManager.getException();  // will hold the exception object of ep
511     
512                     if (ep.getMethod().getReturnType().equals(TypeReference.Void)) {
513                         invokation = tsif.InvokeInstruction(callPC, params, exception, site);
514                         this.body.addStatement(invokation);
515                     } else {
516                         //  Check if we have to mix in the return value of this ep using a Phi
517                         final TypeReference returnType = ep.getMethod().getReturnType();
518                         final TypeKey returnKey = new TypeKey(returnType.getName());
519     
520                         if (this.paramManager.isSeen(returnKey)) {
521                             // if it's seen it most likely is a REUSE-Type. However probably it makes sense for 
522                             // other types too so we don't test on isReuse.
523                             
524     
525                             final SSAValue oldValue = this.paramManager.getCurrent(returnKey);
526                             this.paramManager.invalidate(returnKey);
527                             final SSAValue returnValue = paramManager.getUnallocated(returnType, returnKey);
528     
529                             invokation = tsif.InvokeInstruction(callPC, returnValue, params, exception, site); 
530                             this.body.addStatement(invokation);
531                             this.paramManager.setAllocation(returnValue, invokation);
532     
533                             // ... and Phi things together ...
534                             this.paramManager.invalidate(returnKey);
535                             final SSAValue newValue = this.paramManager.getFree(returnType, returnKey);
536                             final int phiPC = body.getNextProgramCounter();
537                             final List<SSAValue> toPhi = new ArrayList<SSAValue>(2);
538                             toPhi.add(oldValue);
539                             toPhi.add(returnValue);
540                             final SSAPhiInstruction phi = tsif.PhiInstruction(phiPC, newValue, toPhi);
541                             this.body.addStatement(phi);
542                             this.paramManager.setPhi(newValue, phi);
543                         } else {
544                             // Just throw away the return value
545                             final SSAValue returnValue = paramManager.getUnmanaged(returnType, new SSAValue.UniqueKey());
546                             invokation = tsif.InvokeInstruction(callPC, returnValue, params, exception, site);
547                             this.body.addStatement(invokation);
548                         }
549                     }
550                 }
551     
552                 this.monitor.worked(++currentProgress);
553                 MonitorUtil.throwExceptionIfCanceled(this.monitor); 
554             }
555     
556             
557             //  Close all sections by "jumping over" the remaining labels
558             if (enteredASection) {
559                 labelSpecial.finish(body.getNextProgramCounter());
560             }
561             this.monitor.done();
562         }

564         /**
565          *  Get method of the Model in an other Signature.
566          *
567          *  Generates a new Method that wraps the model so it can be called using the given Signature. 
568          *  Flags control the behavior of that wrapper.
569          *
570          *  Arguments to the wrapping function are "connected through" to the model based on their type only,
571          *  so if there are multiple Arguments of the same type this may yield to unexpected connections.
572          *

846         /**
847          *  Creates an "encapsulated" version of the model.
848          *
849          *  The generated method will take no parameters. New instances for REUSE-Parameters will
850          *  be created.
851          *
852          *  This variant is useful for the start of an analysis.
853          */
854         public IMethod getMethodEncap() throws CancelException {
855             final MethodReference asMethod;
856             {
857                 final TypeReference clazz = AndroidModelClass.ANDROID_MODEL_CLASS;
858                 final Atom methodName = Atom.concat(this.getName(), Atom.findOrCreateAsciiAtom("Encap"));
859                 //final TypeName returnType = this.getReturnType();
860                 final TypeName returnType = TypeReference.VoidName;
861                 final Descriptor descr = Descriptor.findOrCreate(new TypeName[]{}, returnType);
862                 asMethod = MethodReference.findOrCreate(clazz, methodName, descr);
863             }
864     
865             final AndroidModelClass mClass = AndroidModelClass.getInstance(cha);
866     
867             if (mClass.containsMethod(asMethod.getSelector())) {
868                 // There's already an encap for this method
869                 return mClass.getMethod(asMethod.getSelector());
870             }
871     
872             final VolatileMethodSummary encap = new VolatileMethodSummary(new MethodSummary(asMethod));
873             encap.setStatic(true);
874             final TypeSafeInstructionFactory instructionFactory = new TypeSafeInstructionFactory(getClassHierarchy());
875             final ParameterAccessor acc = new ParameterAccessor(asMethod, /* hasImplicitThis: */ false);
876             final SSAValueManager pm = new SSAValueManager(acc);
877             pm.breadCrumb = "Encap: " + this.getClass().toString();
878     
879             final SummarizedMethod model = getMethod();
880     
881             final List<SSAValue> params = new ArrayList<SSAValue>();
882             { // Collect Params
883                 final FlatInstantiator instantiator = new FlatInstantiator(encap, instructionFactory, pm, this.cha, asMethod, this.scope);
884     
885                 for (int i = 0; i < model.getNumberOfParameters(); ++i) {
886                     final TypeReference argT = model.getParameterType(i);
887                     final SSAValue arg;
888     
889                     if  ( ( AndroidEntryPointManager.MANAGER.doFlatComponents()) &&  
890                                 (AndroidComponent.isAndroidComponent(argT, cha)) ) { 
891                         // Get / Put filed in AndroidModelClass for Android-Components
892                         final Atom fdName = argT.getName().getClassName();
893     
894                         if (mClass.getField(fdName) != null) {
895                             final IField field = mClass.getField(fdName);
896                             final int instPC = encap.getNextProgramCounter();
897                             arg = pm.getUnallocated(argT, new SSAValue.WeaklyNamedKey(argT.getName(), 
898                                         "got" + fdName.toString()));
899                             final SSAInstruction getInst = instructionFactory.GetInstruction(instPC, arg, field.getReference());
900                             encap.addStatement(getInst);
901                             pm.setAllocation(arg, getInst);
902                         } else {
903                             arg = instantiator.createInstance(argT, false, null, null);
904     
905                             mClass.putField(fdName, argT);
906                             final int instPC = encap.getNextProgramCounter();
907                             final FieldReference fdRef = FieldReference.findOrCreate(mClass.getReference(), fdName, argT);
908                             final SSAInstruction putInst = instructionFactory.PutInstruction(instPC, arg, fdRef);
909                             encap.addStatement(putInst);
910                         }
911                     } else {
912                         final boolean managed = false;
913                         final SSAValue.VariableKey key = new SSAValue.TypeKey(argT.getName());
914                         arg = instantiator.createInstance (argT, managed, key, null);
915                     }
916                     params.add(arg);
917                 }
918             }
919     
920             { // Call the model
921                 final int callPC = encap.getNextProgramCounter();
922                 final CallSiteReference site = CallSiteReference.make(callPC, model.getReference(), IInvokeInstruction.Dispatch.STATIC);
923                 final SSAValue exception = pm.getException();
924             
925                 final SSAAbstractInvokeInstruction invokation = instructionFactory.InvokeInstruction(callPC, params, exception, site);
926                 encap.addStatement(invokation);
927             }
928     
929             encap.setLocalNames(pm.makeLocalNames());
930             final SummarizedMethod method = new SummarizedMethodWithNames(asMethod, encap, mClass);
931             mClass.addMethod(method);
932     
933             return method;
934         }
