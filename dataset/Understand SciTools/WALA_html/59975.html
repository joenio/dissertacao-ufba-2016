12     package com.ibm.wala.cast.js.ipa.callgraph.correlations;
13     
14     import java.util.Set;
15     
16     import com.ibm.wala.cast.ir.ssa.AbstractReflectiveGet;
17     import com.ibm.wala.cast.tree.CAstSourcePositionMap.Position;
18     import com.ibm.wala.ssa.SSAAbstractInvokeInstruction;
19     
20     /**
21      * An escape correlation conservatively captures inter-procedural correlated pairs: for a dynamic property
22      * read <i>r</i> of the form <code>e[p]</code>, if both the result of <i>r</i> and the value of <code>p</code>
23      * flow into a function call <i>c</i>, we consider <i>r</i> and <i>c</i> to be a correlated pair to account
24      * for the fact that the function called by <i>c</i> may perform a write of property <code>p</code>. 
25      * 
26      * @author mschaefer
27      *
28      */
29     public class EscapeCorrelation extends Correlation {
30     	private final AbstractReflectiveGet get;
31     	private final SSAAbstractInvokeInstruction invoke;
32     	
33     	public EscapeCorrelation(AbstractReflectiveGet get, SSAAbstractInvokeInstruction invoke,
34     	                         String indexName, Set<String> flownThroughLocals) {
35     	  super(indexName, flownThroughLocals);
36     		this.get = get;
37     		this.invoke = invoke;
38     	}
39     	
40     	@Override
41     	public Position getStartPosition(SSASourcePositionMap positions) {
42     		return positions.getPosition(get);
43     	}
44     	
45     	@Override
46     	public Position getEndPosition(SSASourcePositionMap positions) {
47     	  return positions.getPosition(invoke);
48     	}
49     	
50     	public int getNumberOfArguments() {
51     	  return invoke.getNumberOfParameters() - 2; // deduct one for the function object, one for the receiver
52     	}
53     	
54     	@Override
55     	public String pp(SSASourcePositionMap positions) {
56     		return get + "@" + positions.getPosition(get) + " [" + getIndexName() + "] ->? " + invoke + "@" + positions.getPosition(invoke);
57     	}
58     	
59     	@Override
60     	public <T> T accept(CorrelationVisitor<T> visitor) {
61     	  return visitor.visitEscapeCorrelation(this);
62     	}
63     }
