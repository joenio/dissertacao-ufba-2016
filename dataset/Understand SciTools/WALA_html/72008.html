38     /**
39      * Generate IR to model Function.call()
40      * 
41      * @see <a
42      *      href="https://developer.mozilla.org/en/JavaScript/Reference/Global_Objects/Function/Call">MDN
43      *      Function.call() docs</a>
44      * 
45      * @author manu
46      * 
47      */
48     public class JavaScriptFunctionDotCallTargetSelector implements MethodTargetSelector {
49       /*
50        * Call graph imprecision often leads to spurious invocations of Function.prototype.call; two common
51        * patterns are invocations of "new" on Function.prototype.call (which in reality would lead to a
52        * type error), and self-applications of Function.prototype.call.
53        * 
54        * While neither of these situations is a priori impossible, they are most likely due to analysis
55        * imprecision. If this flag is set to true, we emit a warning when seeing them. 
56        */
57       public static boolean WARN_ABOUT_IMPRECISE_CALLGRAPH = true;
58       
59       public static final boolean DEBUG_SYNTHETIC_CALL_METHODS = false;
60     
61       private static final TypeName CALL_TYPE_NAME = TypeName.findOrCreate("Lprologue.js/Function_prototype_call");
62       private final MethodTargetSelector base;
63     
64       public JavaScriptFunctionDotCallTargetSelector(MethodTargetSelector base) {
65         this.base = base;
66     
67       }
68     
69       /*
70        * (non-Javadoc)
71        * 
72        * @see
73        * com.ibm.wala.ipa.callgraph.MethodTargetSelector#getCalleeTarget(com.ibm
74        * .wala.ipa.callgraph.CGNode, com.ibm.wala.classLoader.CallSiteReference,
75        * com.ibm.wala.classLoader.IClass)
76        */
77       @Override
78       public IMethod getCalleeTarget(CGNode caller, CallSiteReference site, IClass receiver) {
79         IMethod method = receiver.getMethod(AstMethodReference.fnSelector);
80         if (method != null) {
81           TypeName tn = method.getReference().getDeclaringClass().getName();
82           if (tn.equals(CALL_TYPE_NAME)) {
83             /* invoking Function.prototype.call as a constructor results in a TypeError
84              * see ECMA-262 5.1, 15: "None of the built-in functions described in this clause that 
85              *   are not constructors shall implement the [[Construct]] internal method unless otherwise 
86              *   specified" */
87             if(!site.getDeclaredTarget().equals(JavaScriptMethods.ctorReference)) {
88               IMethod target = getFunctionCallTarget(caller, site, receiver);
89               if(target != null)
90                 return target;
91             }
92             // if we get here, we either saw an invocation of "call" as a constructor, or an invocation 
93             // without receiver object; in either case, this is likely due to bad call graph info
94             if(WARN_ABOUT_IMPRECISE_CALLGRAPH)
95               warnAboutImpreciseCallGraph(caller, site);
96           }
97         }
98         return base.getCalleeTarget(caller, site, receiver);
99       }
100     
101       protected void warnAboutImpreciseCallGraph(CGNode caller, CallSiteReference site) {
102         IntIterator indices = caller.getIR().getCallInstructionIndices(site).intIterator();
103         IMethod callerMethod = caller.getMethod();
104         Position pos = null;
105         if(indices.hasNext() && callerMethod instanceof AstMethod) {
106           pos = ((AstMethod)callerMethod).getSourcePosition(indices.next());
107         }
108         System.err.println("Detected improbable call to Function.prototype.call " +
109             (pos == null ? "in function " + caller : "at position " + pos) +
110             "; this is likely caused by call graph imprecision.");
111       }
112       
113       private static final boolean SEPARATE_SYNTHETIC_METHOD_PER_SITE = false;
114     
115       /**
116        * cache synthetic method for each arity of Function.call() invocation
117        */
118       private final Map<Object, JavaScriptSummarizedFunction> callModels = HashMapFactory.make();
119     
120       /**
121        * generate a synthetic method modeling the invocation of Function.call() at
122        * the site
123        * 
124        * @param caller
125        * @param site
126        * @param receiver
127        * @return
128        */
129       private IMethod getFunctionCallTarget(CGNode caller, CallSiteReference site, IClass receiver) {
130         int nargs = getNumberOfArgsPassed(caller, site);
131         if(nargs < 2)
132           return null;
133         String key = getKey(nargs, caller, site);
134         if (callModels.containsKey(key)) {
135           return callModels.get(key);
136         }
137         JSInstructionFactory insts = (JSInstructionFactory) receiver.getClassLoader().getInstructionFactory();
138         MethodReference ref = genSyntheticMethodRef(receiver, nargs, key);
139         JavaScriptSummary S = new JavaScriptSummary(ref, nargs);
140         
141         if(WARN_ABOUT_IMPRECISE_CALLGRAPH && caller.getMethod().getName().toString().contains(SYNTHETIC_CALL_METHOD_PREFIX))
142           warnAboutImpreciseCallGraph(caller, site);
143         
144         // print information about where the method was created if desired 
145         if(DEBUG_SYNTHETIC_CALL_METHODS) {
146           IMethod method = caller.getMethod();
147           if(method instanceof AstMethod) {
148             int line = ((AstMethod)method).getLineNumber(caller.getIR().getCallInstructionIndices(site).intIterator().next());
149             System.err.println("creating " + ref.getName() + " at line " + line + " in " + caller);
150           } else {
151             System.err.println("creating " + ref.getName() + " in " + method.getName());
152           }
153         }
154     
155         // generate invocation instruction for the real method being invoked
156         int resultVal = nargs + 2;
157         CallSiteReference cs = new JSCallSiteReference(S.getNextProgramCounter());
158         int[] params = new int[nargs - 2];
159         for (int i = 0; i < params.length; i++) {
160           // add 3 to skip v1 (which points to Function.call() itself) and v2 (the
161           // real function being invoked)
162           params[i] = i + 3;
163         }
164         // function being invoked is in v2
165         S.addStatement(insts.Invoke(S.getNumberOfStatements(), 2, resultVal, params, resultVal + 1, cs));
166         S.getNextProgramCounter();
167     
168         S.addStatement(insts.ReturnInstruction(S.getNumberOfStatements(), resultVal, false));
169         S.getNextProgramCounter();
170     
171         JavaScriptSummarizedFunction t = new JavaScriptSummarizedFunction(ref, S, receiver);
172         callModels.put(key, t);
173         return t;
174       }
175     
176       public static final String SYNTHETIC_CALL_METHOD_PREFIX = "$$ call_";
177     
178       private MethodReference genSyntheticMethodRef(IClass receiver, int nargs, String key) {
179         Atom atom = Atom.findOrCreateUnicodeAtom(SYNTHETIC_CALL_METHOD_PREFIX + key);
180         Descriptor desc = Descriptor.findOrCreateUTF8(JavaScriptLoader.JS, "()LRoot;");
181         MethodReference ref = MethodReference.findOrCreate(receiver.getReference(), atom, desc);
182         return ref;
183       }
184     
185       private String getKey(int nargs, CGNode caller, CallSiteReference site) {
186         if (SEPARATE_SYNTHETIC_METHOD_PER_SITE) {
187           return JSCallGraphUtil.getShortName(caller) + "_" + caller.getGraphNodeId() + "_" + site.getProgramCounter();
188         } else {
189           return ""+nargs;
190         }
191       }
192     
193       private int getNumberOfArgsPassed(CGNode caller, CallSiteReference site) {
194         IR callerIR = caller.getIR();
195         SSAAbstractInvokeInstruction callStmts[] = callerIR.getCalls(site);
196         assert callStmts.length == 1;
197         int nargs = callStmts[0].getNumberOfParameters();
198         return nargs;
199       }
