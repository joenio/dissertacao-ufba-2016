41     /**
42      * Driver that constructs a call graph for an application specified via a scope file.  
43      * Useful for getting some code to copy-paste.    
44      */
45     public class ScopeFileCallGraph {
46     
47       /**
48        * Usage: ScopeFileCallGraph -scopeFile file_path [-entryClass class_name |
49        * -mainClass class_name]
50        * 
51        * If given -mainClass, uses main() method of class_name as entrypoint. If
52        * given -entryClass, uses all public methods of class_name.
53        * 
54        * @throws IOException
55        * @throws ClassHierarchyException
56        * @throws CallGraphBuilderCancelException
57        * @throws IllegalArgumentException
58        */
59       public static void main(String[] args) throws IOException, ClassHierarchyException, IllegalArgumentException,
60           CallGraphBuilderCancelException {
61         long start = System.currentTimeMillis();
62         Properties p = CommandLine.parse(args);
63         String scopeFile = p.getProperty("scopeFile");
64         String entryClass = p.getProperty("entryClass");
65         String mainClass = p.getProperty("mainClass");
66         if (mainClass != null && entryClass != null) {
67           throw new IllegalArgumentException("only specify one of mainClass or entryClass");
68         }
69         // use exclusions to eliminate certain library packages
70         File exclusionsFile = null;
71         AnalysisScope scope = AnalysisScopeReader.readJavaScope(scopeFile, exclusionsFile, ScopeFileCallGraph.class.getClassLoader());
72         IClassHierarchy cha = ClassHierarchy.make(scope);
73         System.out.println(cha.getNumberOfClasses() + " classes");
74         System.out.println(Warnings.asString());
75         Warnings.clear();
76         AnalysisOptions options = new AnalysisOptions();
77         Iterable<Entrypoint> entrypoints = entryClass != null ? makePublicEntrypoints(scope, cha, entryClass) : Util.makeMainEntrypoints(scope, cha, mainClass);
78         options.setEntrypoints(entrypoints);
79         // you can dial down reflection handling if you like
80     //    options.setReflectionOptions(ReflectionOptions.NONE);
81         AnalysisCache cache = new AnalysisCache();
82         // other builders can be constructed with different Util methods
83         CallGraphBuilder builder = Util.makeZeroOneContainerCFABuilder(options, cache, cha, scope);
84     //    CallGraphBuilder builder = Util.makeNCFABuilder(2, options, cache, cha, scope);
85     //    CallGraphBuilder builder = Util.makeVanillaNCFABuilder(2, options, cache, cha, scope);
86         System.out.println("building call graph...");
87         CallGraph cg = builder.makeCallGraph(options, null);
88         long end = System.currentTimeMillis();
89         System.out.println("done");
90         System.out.println("took " + (end-start) + "ms");
91         System.out.println(CallGraphStats.getStats(cg));
92       }
93     
94       private static Iterable<Entrypoint> makePublicEntrypoints(AnalysisScope scope, IClassHierarchy cha, String entryClass) {
95         Collection<Entrypoint> result = new ArrayList<Entrypoint>();
96         IClass klass = cha.lookupClass(TypeReference.findOrCreate(ClassLoaderReference.Application,
97             StringStuff.deployment2CanonicalTypeString(entryClass)));
98         for (IMethod m : klass.getDeclaredMethods()) {
99           if (m.isPublic()) {
100             result.add(new DefaultEntrypoint(m, cha));
101           }
102         }
103         return result;
104       }
105     }
