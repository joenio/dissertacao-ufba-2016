35     /**
36      * a view of a portion of a call graph.
37      */
38     public class PartialCallGraph extends DelegatingGraph<CGNode> implements CallGraph {
39     
40       protected final CallGraph cg;
41     
42       protected final Collection<CGNode> partialRoots;
43     
44       protected PartialCallGraph(CallGraph cg, Collection<CGNode> partialRoots, Graph<CGNode> partialGraph) {
45         super(partialGraph);
46         this.cg = cg;
47         this.partialRoots = partialRoots;
48       }
49     
50       /**
51        * @param cg the original call graph
52        * @param partialRoots roots of the new, partial graph
53        * @param nodes set of nodes that will be included in the new, partial call graph
54        */
55       public static PartialCallGraph make(final CallGraph cg, final Collection<CGNode> partialRoots, final Collection<CGNode> nodes) {
56         Graph<CGNode> partialGraph = GraphSlicer.prune(cg, new Predicate<CGNode>() {
57           @Override public boolean test(CGNode o) {
58             return nodes.contains(o);
59           }
60         });
61     
62         return new PartialCallGraph(cg, partialRoots, partialGraph);
63       }
64     
65       /**
66        * @param cg the original call graph
67        * @param partialRoots roots of the new, partial graph
68        * the result contains only nodes reachable from the partialRoots in the original call graph.
69        */
70       public static PartialCallGraph make(CallGraph cg, Collection<CGNode> partialRoots) {
71         final Set<CGNode> nodes = DFS.getReachableNodes(cg, partialRoots);
72         Graph<CGNode> partialGraph = GraphSlicer.prune(cg, new Predicate<CGNode>() {
73           @Override public boolean test(CGNode o) {
74             return nodes.contains(o);
75           }
76         });
77     
78         return new PartialCallGraph(cg, partialRoots, partialGraph);
79       }
80     
81       @Override
82       public CGNode getFakeRootNode() throws UnsupportedOperationException {
83         throw new UnsupportedOperationException();
84       }
85     
86       @Override
87       public CGNode getFakeWorldClinitNode() throws UnsupportedOperationException {
88         throw new UnsupportedOperationException();
89       }
90     
91       @Override
92       public Collection<CGNode> getEntrypointNodes() {
93         return partialRoots;
94       }
95     
96       @Override
97       public CGNode getNode(IMethod method, Context C) {
98         CGNode x = cg.getNode(method, C);
99         if (x == null) {
100           return null;
101         }
102         return (containsNode(x) ? x : null);
103       }
104     
105       @Override
106       public Set<CGNode> getNodes(MethodReference m) {
107         Set<CGNode> result = HashSetFactory.make();
108         for (Iterator xs = cg.getNodes(m).iterator(); xs.hasNext();) {
109           CGNode x = (CGNode) xs.next();
110           if (containsNode(x)) {
111             result.add(x);
112           }
113         }
114     
115         return result;
116       }
117     
118       @Override
119       public IClassHierarchy getClassHierarchy() {
120         return cg.getClassHierarchy();
121       }
122     
123       @Override
124       public Iterator<CGNode> iterateNodes(IntSet nodes) {
125         return new FilterIterator<CGNode>(cg.iterateNodes(nodes), new Predicate() {
126           @Override public boolean test(Object o) {
127             return containsNode((CGNode) o);
128           }
129         });
130       }
131     
132       @Override
133       public int getMaxNumber() {
134         return cg.getMaxNumber();
135       }
136     
137       @Override
138       public CGNode getNode(int index) {
139         CGNode n = cg.getNode(index);
140         return (containsNode(n) ? n : null);
141       }
142     
143       @Override
144       public int getNumber(CGNode n) {
145         return (containsNode(n) ? cg.getNumber(n) : -1);
146       }
147     
148       @Override
149       public IntSet getSuccNodeNumbers(CGNode node) {
150         assert containsNode(node);
151         MutableIntSet x = IntSetUtil.make();
152         for (Iterator ns = getSuccNodes(node); ns.hasNext();) {
153           CGNode succ = (CGNode) ns.next();
154           if (containsNode(succ)) {
155             x.add(getNumber(succ));
156           }
157         }
158     
159         return x;
160       }
161     
162       @Override
163       public IntSet getPredNodeNumbers(CGNode node) {
164         assert containsNode(node);
165         MutableIntSet x = IntSetUtil.make();
166         for (Iterator ns = getPredNodes(node); ns.hasNext();) {
167           CGNode pred = (CGNode) ns.next();
168           if (containsNode(pred)) {
169             x.add(getNumber(pred));
170           }
171         }
172     
173         return x;
174       }
175     
176       @Override
177       public int getNumberOfTargets(CGNode node, CallSiteReference site) {
178         return (containsNode(node) ? getPossibleTargets(node, site).size() : -1);
179       }
180     
181       @Override
182       public Iterator<CallSiteReference> getPossibleSites(CGNode src, CGNode target) {
183         return ((containsNode(src) && containsNode(target)) ? cg.getPossibleSites(src, target) : null);
184       }
185     
186       @Override
187       public Set<CGNode> getPossibleTargets(CGNode node, CallSiteReference site) {
188         if (!containsNode(node)) {
189           return null;
190         }
191         Set<CGNode> result = HashSetFactory.make();
192         for (Iterator ns = cg.getPossibleTargets(node, site).iterator(); ns.hasNext();) {
193           CGNode target = (CGNode) ns.next();
194           if (containsNode(target)) {
195             result.add(target);
196           }
197         }
198     
199         return result;
200       }
201     }
