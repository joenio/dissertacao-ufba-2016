11     package com.ibm.wala.dataflow.IFDS;
12     
13     import java.util.Collection;
14     
15     import com.ibm.wala.util.intset.IntSet;
16     
17     /**
18      * The solution of a tabulation problem: a mapping from supergraph node -> bit vector representing the dataflow facts that hold at
19      * the entry to the supergraph node.
20      * 
21      * @param <T> type of node in the supergraph
22      * @param <P> type of a procedure, like a box in an RSM
23      * @param <F> type of factoids propagated when solving this problem
24      */
25     public interface TabulationResult<T, P, F> {
26       /**
27        * get the bitvector of facts that hold at IN for a given node in the supergraph.
28        * 
29        * @param node a node in the supergraph
30        * @return SparseIntSet efficiently representing the bitvector
31        */
32       public IntSet getResult(T node);
33     
34       /**
35        * @return the governing IFDS problem
36        */
37       public TabulationProblem<T, P, F> getProblem();
38     
39       /**
40        * @return the set of supergraph nodes for which any fact is reached
41        */
42       public Collection<T> getSupergraphNodesReached();
43     
44       /**
45        * @param n1
46        * @param d1
47        * @param n2
48        * @return set of d2 s.t. (n1,d1) -> (n2,d2) is recorded as a summary edge, or null if none found
49        */
50       public IntSet getSummaryTargets(T n1, int d1, T n2);
51     
52       /**
53        * @return the set of all {@link PathEdge}s that were used as seeds during the tabulation.
54        */
55       public Collection<PathEdge<T>> getSeeds();
56     
57     }
