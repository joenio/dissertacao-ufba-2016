142     	public CLISCanDroidOptions(String[] args, boolean reqArgs) {
143     		try {
144     			line = parser.parse(options, args);
145     		} catch (ParseException exp) {
146     			System.err.println("Unexpected exception: " + exp.getMessage());
147     			System.err.println("Usage: " + USAGE);
148     			System.exit(0);
149     		}
150     
151     		if (hasOption("help")) {
152     			HelpFormatter formatter = new HelpFormatter();
153     			formatter.printHelp(USAGE, options);
154     			System.exit(0);
155     		}
156     
157     		// handle verbosity
158     		// parse this arg as a Logback level, then set the root logger level
159     		// appropriately
160     
161     		if (!hasOption(ANDROID_LIB)) {
162     			System.err.println("Please specify an android library");
163     			System.exit(0);
164     		}
165     
166     		classpath = processClasspath(reqArgs);
167     		filename = processFilename();
168     		androidLib = processURIArg(getOption(ANDROID_LIB));
169     		summariesFile = processURIArg(getOption(SUMMARIES_FILE));
170     		reflectionOptions = processReflectionOptions();
171     
172     		if (reqArgs
173     				&& !(filename.endsWith(".apk") || filename.endsWith(".jar"))) {
174     			System.err.println("Usage: " + USAGE);
175     			System.exit(0);
176     		}
177     	}
178     
179     	private URI processURIArg(String arg) {
180     		if (arg == null) {
181     			return null;
182     		} else {
183     			return new File(arg).toURI();
184     		}
185     	}
186     
187     	private URI processClasspath(boolean reqArgs) {
188     		// getArgs() returns all args that are not recognized;
189     		String[] myargs = line.getArgs();
190     		if ((myargs.length != 1 || !(myargs[0].endsWith(".apk") || myargs[0]
191     				.endsWith(".jar"))) && reqArgs) {
192     			System.err.println("Usage: " + USAGE);
193     			System.exit(0);
194     		}
195     		return processURIArg(myargs[0]);
196     	}
197     
198     	private String processFilename() {
199     		if (classpath == null)
200     			return null;
201     		return new File(classpath).getName();
202     	}
203     
204     	private ReflectionOptions processReflectionOptions() {
205     		final String reflection = getOption(REFLECTION);
206     		if (reflection == null) {
207     			return ReflectionOptions.NONE;
208     		} else {
209     			return ReflectionOptions.valueOf(reflection);
210     		}
211     	}

218     		return line.getOptionValue(s);
