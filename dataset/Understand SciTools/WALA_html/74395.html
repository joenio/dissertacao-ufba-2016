11     package com.ibm.wala.viz.viewer;
12     
13     import java.awt.BorderLayout;
14     import java.io.BufferedReader;
15     import java.io.IOException;
16     import java.io.InputStreamReader;
17     import java.net.URL;
18     
19     import javax.swing.DefaultListModel;
20     import javax.swing.JList;
21     import javax.swing.JPanel;
22     import javax.swing.JScrollPane;
23     import javax.swing.JTextField;
24     
25     public class SourceViewer extends JPanel{
26       private URL sourceURL;
27       private JTextField sourceCodeLocation;
28       private DefaultListModel sourceCodeLinesList = new DefaultListModel();
29       private JList sourceCodeLines;
30     
31       public SourceViewer() {
32         super(new BorderLayout());
33         sourceURL = null;
34         sourceCodeLines = new JList(sourceCodeLinesList);
35         sourceCodeLocation = new JTextField("Source code");
36         this.add(sourceCodeLocation, BorderLayout.PAGE_START);
37         this.add(new JScrollPane(sourceCodeLines), BorderLayout.CENTER);
38       }
39     
40       public void setSource(URL url) {
41         setSource(url, IrViewer.NA);
42       }
43     
44       public void setSource(URL url, int sourceLine) {
45         boolean succsess = loadSource(url);
46         if (succsess){
47           sourceCodeLocation.setText("Source code: " + url);
48           if (sourceLine != IrViewer.NA){
49             sourceCodeLines.ensureIndexIsVisible(sourceLine-1);
50             sourceCodeLines.setSelectedIndex(sourceLine-1);
51             sourceCodeLines.validate();
52           }
53         } else {
54           sourceCodeLocation.setText("Error loading source code from: " + url);
55         }
56       }
57       
58       private boolean loadSource(URL url) {
59         if (url == null) {
60           if (sourceURL != null) {
61             // easing the current code.
62             sourceCodeLinesList.clear();
63           }
64           return false; // nothing to load
65         } else {
66           if (url.equals(sourceURL)) {
67             return true; // already loaded
68           } else {
69             sourceCodeLinesList.clear();
70             try {
71               BufferedReader br = new BufferedReader(new InputStreamReader(url.openStream()));
72               String line;
73               while ((line = br.readLine()) != null) {
74                 sourceCodeLinesList.addElement(line.replaceAll("\t", "   "));
75               }
76               br.close();
77               return true;
78             } catch (IOException e) {
79               System.err.println("Could not load source at " + url);
80               return false;
81             }
82           }
83         }
84       }
85     
86       public void removeSelection() {
87         int curSelectedIndex = sourceCodeLines.getSelectedIndex();
88         sourceCodeLines.removeSelectionInterval(curSelectedIndex, curSelectedIndex);
89       }
90     
91       public void removeSource() {
92         sourceURL = null;
93         sourceCodeLocation.setText("Source code");
94         sourceCodeLinesList.clear();
95         sourceCodeLines.validate();
96       }
97     
98     
99     }
