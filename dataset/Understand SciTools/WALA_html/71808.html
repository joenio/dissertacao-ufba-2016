33     public class BootstrapDumper {
34       final private PrintWriter w;
35     
36       /**
37        * Get ready to print a class to the given output stream.
38        */
39       public BootstrapDumper(PrintWriter w) {
40         this.w = w;
41       }
42     
43       public static void main(String[] args) throws Exception {
44         OfflineInstrumenter oi = new OfflineInstrumenter(true);
45         String[] classpathEntries = oi.parseStandardArgs(args);
46         
47         PrintWriter w = new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out)));
48     
49         BootstrapDumper p = new BootstrapDumper(w);
50     
51         URL[] urls = new URL[ classpathEntries.length-1 ];
52         for(int i = 1; i < classpathEntries.length; i++) {
53           System.err.println(classpathEntries[i]);
54           File f = new File(classpathEntries[i]);
55           assert f.exists();
56           urls[i-1] = f.toURI().toURL(); 
57         }
58         URLClassLoader image = URLClassLoader.newInstance(urls, BootstrapDumper.class.getClassLoader().getParent());
59         
60         System.err.println(image);
61         
62         ClassInstrumenter ci;
63         oi.beginTraversal();
64         while ((ci = oi.nextClass()) != null) {
65           try {
66             p.doClass(image, ci.getReader());
67           } finally {
68             w.flush();
69           }
70         }
71     
72         oi.close();
73       }

75       private void dumpAttributes(Class cl, ClassReader cr, int i, ClassReader.AttrIterator attrs) throws InvalidClassFileException,
76           InvalidBytecodeException, IOException, ClassNotFoundException, NoSuchMethodException, SecurityException, IllegalAccessException, IllegalArgumentException, InvocationTargetException, NoSuchFieldException {
77         for (; attrs.isValid(); attrs.advance()) {
78           String name = attrs.getName();
79           if (name.equals("Code")) {
80             CodeReader code = new CodeReader(attrs);
81     
82             CTDecoder decoder = new CTDecoder(code);
83             decoder.decode();
84             IInstruction[] insts = decoder.getInstructions();
85             for(IInstruction inst : insts) {
86               if (inst instanceof InvokeDynamicInstruction) {
87                 CallSite target = ((InvokeDynamicInstruction)inst).bootstrap(cl);
88                 w.println(target.dynamicInvoker());
89                 w.println(target.getTarget());
90                 /*
91                  * only in Java 8.  Uncomment when we mandate Java 8.
92                 try {
93                   w.println(MethodHandles.reflectAs(Method.class, target.dynamicInvoker()));
94                 } catch (Throwable e) {
95                   System.out.println(e);
96                 }
97                 */
98               }
99             }
100           }
101         }
102       }

106       /**
107        * Print a class.
108        * @throws InvocationTargetException 
109        * @throws IllegalAccessException 
110        * @throws SecurityException 
111        * @throws NoSuchMethodException 
112        * @throws ClassNotFoundException 
113        * 
114        * @throws IllegalArgumentException if cr is null
115        * @throws NoSuchFieldException 
116        */
117       public void doClass(ClassLoader image, final ClassReader cr) throws InvalidClassFileException, InvalidBytecodeException, IOException, ClassNotFoundException, NoSuchMethodException, SecurityException, IllegalAccessException, IllegalArgumentException, InvocationTargetException, NoSuchFieldException {
118         if (cr == null) {
119           throw new IllegalArgumentException("cr is null");
120         }
121     
122         ClassReader.AttrIterator attrs = new ClassReader.AttrIterator();
123         cr.initClassAttributeIterator(attrs);
124         int methodCount = cr.getMethodCount();
125         
126         for (int i = 0; i < methodCount; i++) {
127           cr.initMethodAttributeIterator(i, attrs);
128           dumpAttributes(Class.forName(cr.getName().replace('/', '.'), false, image), cr, i, attrs);
129         }
130       }
