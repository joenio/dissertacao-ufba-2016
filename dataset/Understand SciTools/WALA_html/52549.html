29     /**
30      * Represents a set of {@link IFixedPointStatement}s to be solved by a {@link IFixedPointSolver}
31      * 
32      * <p>
33      * Implementation Note:
34      * 
35      * The set of steps and variables is internally represented as a graph. Each step and each variable is a node in the graph. If a
36      * step produces a variable that is used by another step, the graph has a directed edge from the producer to the consumer.
37      * Fixed-point iteration proceeds in a topological order according to these edges.
38      */
39     @SuppressWarnings("rawtypes")
40     public abstract class AbstractFixedPointSolver<T extends IVariable<?>> implements IFixedPointSolver<T>, FixedPointConstants,
41         VerboseAction {

123       /**
124        * Solve the set of dataflow graph.
125        * <p>
126        * PRECONDITION: graph is set up
127        * 
128        * @return true iff the evaluation of some equation caused a change in the value of some variable.
129        */
130       @Override
131       @SuppressWarnings("unchecked")
132       public boolean solve(IProgressMonitor monitor) throws CancelException {
133     
134         boolean globalChange = false;
135     
136         if (firstSolve) {
137           initForFirstSolve();
138         }
139     
140         while (!workList.isEmpty()) {
141           MonitorUtil.throwExceptionIfCanceled(monitor);
142           orderStatements();
143     
144           // duplicate insertion detection
145           AbstractStatement s = workList.takeStatement();
146     
147           if (DEBUG) {
148             System.err.println(("Before evaluation " + s));
149           }
150           byte code = s.evaluate();
151           if (verbose) {
152             nEvaluated++;
153             if (nEvaluated % getVerboseInterval() == 0) {
154               performVerboseAction();
155             }
156             if (nEvaluated % getPeriodicMaintainInterval() == 0) {
157               periodicMaintenance();
158             }
159     
160           }
161           if (DEBUG) {
162             System.err.println(("After evaluation  " + s + " " + isChanged(code)));
163           }
164           if (isChanged(code)) {
165             globalChange = true;
166             updateWorkList(s);
167           }
168           if (isFixed(code)) {
169             removeStatement(s);
170           }
171         }
172         return globalChange;
173       }

175       @Override
176       public void performVerboseAction() {
177         System.err.println("Evaluated " + nEvaluated);
178         System.err.println("Created   " + nCreated);
179         System.err.println("Worklist  " + workList.size());
180         if (MORE_VERBOSE) {
181           if (!workList.isEmpty()) {
182             AbstractStatement s = workList.takeStatement();
183             System.err.println("Peek      " + lineBreak(s.toString(), 132));
184             if (s instanceof VerboseAction) {
185               ((VerboseAction) s).performVerboseAction();
186             }
187             workList.insertStatement(s);
188           }
189         }
190       }

192       public static String lineBreak(String string, int wrap) {
193         if (string == null) {
194           throw new IllegalArgumentException("string is null");
195         }
196         if (string.length() > wrap) {
197           StringBuffer result = new StringBuffer();
198           int start = 0;
199           while (start < string.length()) {
200             int end = Math.min(start + wrap, string.length());
201             result.append(string.substring(start, end));
202             result.append("\n  ");
203             start = end;
204           }
205           return result.toString();
206         } else {
207           return string;
208         }
209       }

260       /**
261        * Add a step with zero operands on the right-hand side.
262        * 
263        * TODO: this is a little odd, in that this equation will never fire unless explicitly added to a work list. I think in most cases
264        * we shouldn't be creating this nullary form.
265        * 
266        * @param lhs the variable set by this equation
267        * @param operator the step operator
268        * @throws IllegalArgumentException if lhs is null
269        */
270       public boolean newStatement(final T lhs, final NullaryOperator<T> operator, final boolean toWorkList, final boolean eager) {
271         if (lhs == null) {
272           throw new IllegalArgumentException("lhs is null");
273         }
274         // add to the list of graph
275         lhs.setOrderNumber(nextOrderNumber++);
276         final NullaryStatement<T> s = new BasicNullaryStatement<T>(lhs, operator);
277         if (getFixedPointSystem().containsStatement(s)) {
278           return false;
279         }
280         nCreated++;
281         getFixedPointSystem().addStatement(s);
282         incorporateNewStatement(toWorkList, eager, s);
283         topologicalCounter++;
284         return true;
285       }

287       @SuppressWarnings("unchecked")
288       private void incorporateNewStatement(boolean toWorkList, boolean eager, AbstractStatement s) {
289         if (eager) {
290           byte code = s.evaluate();
291           if (verbose) {
292             nEvaluated++;
293             if (nEvaluated % getVerboseInterval() == 0) {
294               performVerboseAction();
295             }
296             if (nEvaluated % getPeriodicMaintainInterval() == 0) {
297               periodicMaintenance();
298             }
299           }
300           if (isChanged(code)) {
301             updateWorkList(s);
302           }
303           if (isFixed(code)) {
304             removeStatement(s);
305           }
306         } else if (toWorkList) {
307           addToWorkList(s);
308         }
309       }

311       /**
312        * Add a step with one operand on the right-hand side.
313        * 
314        * @param lhs the lattice variable set by this equation
315        * @param operator the step's operator
316        * @param rhs first operand on the rhs
317        * @return true iff the system changes
318        * @throws IllegalArgumentException if operator is null
319        */
320       public boolean newStatement(T lhs, UnaryOperator<T> operator, T rhs, boolean toWorkList, boolean eager) {
321         if (operator == null) {
322           throw new IllegalArgumentException("operator is null");
323         }
324         // add to the list of graph
325         UnaryStatement<T> s = operator.makeEquation(lhs, rhs);
326         if (getFixedPointSystem().containsStatement(s)) {
327           return false;
328         }
329         if (lhs != null) {
330           lhs.setOrderNumber(nextOrderNumber++);
331         }
332         nCreated++;
333         getFixedPointSystem().addStatement(s);
334         incorporateNewStatement(toWorkList, eager, s);
335         topologicalCounter++;
336         return true;
337       }

402         }

404         lhs.setOrderNumber(nextOrderNumber++);
405         GeneralStatement<T> s = new Statement(lhs, operator, op1, op2, op3);

451       /**
452        * Update the worklist, assuming that a particular equation has been re-evaluated
453        * 
454        * @param s the equation that has been re-evaluated.
455        */
456       private void updateWorkList(AbstractStatement<T, ?> s) {
457         // find each equation which uses this lattice cell, and
458         // add it to the work list
459         T v = s.getLHS();
460         if (v == null) {
461           return;
462         }
463         changedVariable(v);
464       }

466       /**
467        * Number the graph in topological order.
468        */
469       private void orderStatementsInternal() {
470         if (verbose) {
471           if (nEvaluated > 0) {
472             System.err.println("Reorder " + nEvaluated + " " + nCreated);
473           }
474         }
475         reorder();
476         if (verbose) {
477           if (nEvaluated > 0) {
478             System.err.println("Reorder finished " + nEvaluated + " " + nCreated);
479           }
480         }
481         topologicalCounter = 0;
482         evaluationsAtLastOrdering = nEvaluated;
483       }
484     
485       /**
486        * 
487        */
488       public void orderStatements() {
489     
490         if (nextOrderNumber > minSizeForTopSort) {
491           if (((double) topologicalCounter / (double) nextOrderNumber) > topologicalGrowthFactor) {
492             orderStatementsInternal();
493             return;
494           }
495         }
496     
497         if ((nEvaluated - evaluationsAtLastOrdering) > maxEvalBetweenTopo) {
498           orderStatementsInternal();
499           return;
500         }
501       }
502     
503       /**
504        * Re-order the step definitions.
505        */
506       private void reorder() {
507         // drain the worklist
508         LinkedList<AbstractStatement> temp = new LinkedList<AbstractStatement>();
509         while (!workList.isEmpty()) {
510           AbstractStatement eq = workList.takeStatement();
511           temp.add(eq);
512         }
513         workList = new Worklist();
514     
515         // compute new ordering
516         getFixedPointSystem().reorder();
517     
518         // re-populate worklist
519         for (Iterator<AbstractStatement> it = temp.iterator(); it.hasNext();) {
520           AbstractStatement s = it.next();
521           workList.insertStatement(s);
522         }
523       }
