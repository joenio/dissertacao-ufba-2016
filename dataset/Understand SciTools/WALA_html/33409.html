11     package com.ibm.wala.util.graph;
12     
13     import java.util.Iterator;
14     import java.util.Map;
15     
16     import com.ibm.wala.util.functions.Function;
17     
18     public class GXL {
19     
20       public interface EntityTypes<T> {
21         String type(T entity);
22         String type(Graph<T> entity);
23         String type(T from, T to);
24       }
25       
26       public static <T> String toGXL(Graph<T> G, 
27           EntityTypes<T> types,
28           String graphId, 
29           Function<T,String> nodeIds, 
30           Function<T,Map<String,String>> nodeProperties) 
31       {
32         StringBuffer sb = new StringBuffer();
33         
34         sb.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
35         sb.append("<!DOCTYPE gxl SYSTEM \"http://www.gupro.de/GXL/gxl-1.0.dtd\">\n");
36         sb.append("<gxl xmlns:xlink=\"http://www.w3.org/1999/xlink\">\n");
37         sb.append("  <graph id=\"" + graphId + "\" edgemode=\"directed\" hypergraph=\"false\">\n");
38         sb.append("    <type xlink:href=\"" + types.type(G) + "\"/>\n");
39         
40         for(T n : G) {
41           sb.append("    <node id=\"" + nodeIds.apply(n) + "\">\n");
42           sb.append("      <type xlink:href=\"" + types.type(n) + "\"/>\n");
43           Map<String,String> props = nodeProperties.apply(n);
44           if (props != null) {
45             for(Map.Entry<String,String> e : props.entrySet()) {
46               sb.append("      <attr name=\"" + e.getKey() + "\">\n");
47               if (e.getValue() != null) {
48                 sb.append("        <string>" + e.getValue() + "</string>\n");
49               } else {
50                 sb.append("        <string/>\n");            
51               }
52               sb.append("      </attr>\n");
53             }
54           }
55           sb.append("    </node>\n");
56         }
57         
58         for(T n : G) {
59           Iterator<T> ss = G.getSuccNodes(n);
60           while (ss.hasNext()) {
61             T s = ss.next();
62             sb.append("    <edge from=\"" + nodeIds.apply(n) + "\" to=\"" + nodeIds.apply(s) + "\">\n");
63             sb.append("      <type xlink:href=\"" + types.type(n, s) + "\"/>\n");
64             
65             sb.append("    </edge>\n");
66           }
67         }
68         
69         sb.append("  </graph>\n");
70         sb.append("</gxl>\n");
71         
72         return sb.toString();
73       }
74     }
