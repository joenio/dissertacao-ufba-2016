24       
25       private static Graph<String> createGraph(String edges) {
26         Graph<String> g = SlowSparseNumberedGraph.make();
27         for(int i = 0; i < edges.length(); i+= 2) {
28           String from = edges.substring(i, i+1);
29           if (! g.containsNode(from)) {
30             g.addNode(from);
31           }
32           
33           String to = edges.substring(i+1, i+2);
34           if (! g.containsNode(to)) {
35             g.addNode(to);
36           }
37           
38           g.addEdge(from, to);
39         }
40         return g;
41       }
42     
43       private static DFSAllPathsFinder<String> makeFinder(Graph<String> g, String start, final String end) {
44         return new DFSAllPathsFinder<String>(g, start, new Predicate<String>() {
45           @Override public boolean test(String o) {
46             return end.equals(o);
47           }
48         });
49       }
50     
51       private static void checkPaths(DFSAllPathsFinder<String> paths, int expectedCount) {
52         int count = 0;
53         List<String> path;
54         while ((path = paths.find()) != null) {  
55           System.err.println(path);
56           count++;
57         }
58         Assert.assertEquals(expectedCount, count);
59       }

64       public void testPaths1() {
65         Graph<String> g = createGraph(edges1);
66         DFSAllPathsFinder<String> paths = makeFinder(g, "A", "L");

68       }
