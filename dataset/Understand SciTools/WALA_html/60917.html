136       /**
137        * Return a set of {@link IClass} that holds all superclasses of klass
138        * 
139        * @param klass class in question
140        * @return Set the result set
141        */
142       private Set<IClass> computeSuperclasses(IClass klass) {
143         if (DEBUG) {
144           System.err.println("computeSuperclasses: " + klass);
145         }
146     
147         Set<IClass> result = HashSetFactory.make(3);
148     
149         klass = klass.getSuperclass();
150     
151         while (klass != null) {
152           if (DEBUG) {
153             System.err.println("got superclass " + klass);
154           }
155           boolean added = result.add(klass);
156           if (!added) {
157             // oops.  we have A is a sub-class of B and B is a sub-class of A.  blow up.
158             throw new IllegalStateException("cycle in the extends relation for class " + klass);
159           }
160           klass = klass.getSuperclass();
161           if (klass != null && klass.getReference().getName().equals(rootTypeRef.getName())) {
162             if (!klass.getReference().getClassLoader().equals(rootTypeRef.getClassLoader())) {
163               throw new IllegalStateException("class " + klass + " is invalid, unexpected classloader");
164             }
165           }
166         }
167         return result;
168       }

180       private ClassHierarchy(AnalysisScope scope, ClassLoaderFactory factory, Collection<Language> languages,
181           IProgressMonitor progressMonitor) throws ClassHierarchyException, IllegalArgumentException {
182         // now is a good time to clear the warnings globally.
183         // TODO: think of a better way to guard against warning leaks.
184         Warnings.clear();
185     
186         if (factory == null) {
187           throw new IllegalArgumentException();
188         }
189         if (scope.getLanguages().size() == 0) {
190           throw new IllegalArgumentException("AnalysisScope must contain at least 1 language");
191         }
192         this.scope = scope;
193         this.factory = factory;
194         Set<Atom> langNames = HashSetFactory.make();
195         for (Language lang : languages) {
196           this.languages.add(lang);
197           this.languages.addAll(lang.getDerivedLanguages());
198           langNames.add(lang.getName());
199         }
200         for (Language lang : this.languages) {
201           if (lang.getRootType() != null && lang.getRootType() != this.rootTypeRef) {
202             if (this.rootTypeRef != null) {
203               throw new IllegalArgumentException("AnalysisScope must have only 1 root type: " + lang.getRootType() + ", " + rootTypeRef);
204             } else {
205               this.rootTypeRef = lang.getRootType();
206             }
207           }
208         }
209         try {
210           int numLoaders = 0;
211           for (ClassLoaderReference ref : scope.getLoaders()) {
212             if (langNames.contains(ref.getLanguage())) {
213               numLoaders++;
214             }
215           }
216     
217           loaders = new IClassLoader[numLoaders];
218           int idx = 0;
219     
220           if (progressMonitor != null) {
221             progressMonitor.beginTask("Build Class Hierarchy", (numLoaders) * 2 - 1);
222           }
223           for (ClassLoaderReference ref : scope.getLoaders()) {
224             if (progressMonitor != null) {
225               if (progressMonitor.isCanceled()) {
226                 throw new CancelCHAConstructionException();
227               }
228             }
229     
230             if (langNames.contains(ref.getLanguage())) {
231               IClassLoader icl = factory.getLoader(ref, this, scope);
232               loaders[idx++] = icl;
233     
234               if (progressMonitor != null) {
235                 progressMonitor.worked(idx);
236               }
237             }
238           }
239     
240           for (IClassLoader icl : loaders) {
241             if (progressMonitor != null) {
242               progressMonitor.subTask("From " + icl.getName().toString());
243             }
244             addAllClasses(icl, progressMonitor);
245     
246             if (progressMonitor != null) {
247               progressMonitor.worked(idx++);
248             }
249           }
250     
251         } catch (IOException e) {
252           throw new ClassHierarchyException("factory.getLoader failed " + e);
253         } finally {
254           if (progressMonitor != null) {
255             progressMonitor.done(); // In case an exception is thrown.
256           }
257         }
258     
259         if (root == null) {
260           throw new ClassHierarchyException("failed to load root " + rootTypeRef + " of class hierarchy");
261         }
262     
263         // perform numbering for subclass tests.
264         numberTree();
265         ReferenceCleanser.registerClassHierarchy(this);
266       }

268       /**
269        * Add all classes in a class loader to the hierarchy.
270        */
271       private void addAllClasses(IClassLoader loader, IProgressMonitor progressMonitor) throws CancelCHAConstructionException {
272         if (DEBUG) {
273           System.err.println(("Add all classes from loader " + loader));
274         }
275         Collection<IClass> toRemove = HashSetFactory.make();
276         for (Iterator<IClass> it = loader.iterateAllClasses(); it.hasNext();) {
277           if (progressMonitor != null) {
278             if (progressMonitor.isCanceled()) {
279               throw new CancelCHAConstructionException();
280             }
281           }
282           IClass klass = it.next();
283           boolean added = addClass(klass);
284           if (!added) {
285             toRemove.add(klass);
286           }
287         }
288         loader.removeAll(toRemove);
289     
290       }

292       /**
293        * @return true if the add succeeded; false if it failed for some reason
294        * @throws IllegalArgumentException if klass is null
295        */
296       @Override
297       public boolean addClass(IClass klass) {
298         if (klass == null) {
299           throw new IllegalArgumentException("klass is null");
300         }
301         if (klass.getReference().getName().equals(rootTypeRef.getName())) {
302           if (!klass.getReference().getClassLoader().equals(rootTypeRef.getClassLoader())) {
303             throw new IllegalArgumentException("class " + klass + " is invalid, unexpected classloader");
304           }
305         }
306         if (DEBUG) {
307           System.err.println(("Attempt to add class " + klass));
308         }
309         Set<IClass> loadedSuperclasses;
310         Collection loadedSuperInterfaces;
311         try {
312           loadedSuperclasses = computeSuperclasses(klass);
313           loadedSuperInterfaces = klass.getAllImplementedInterfaces();
314         } catch (Exception e) {
315           // a little cleanup
316           if (klass instanceof ShrikeClass) {
317             if (DEBUG) {
318               System.err.println(("Exception.  Clearing " + klass));
319             }
320           }
321           Warnings.add(ClassExclusion.create(klass.getReference(), e.getMessage()));
322           return false;
323         }
324         Node node = findOrCreateNode(klass);
325     
326         if (klass.getReference().equals(this.rootTypeRef)) {
327           // there is only one root
328           assert root == null;
329           root = node;
330         }
331     
332         Set workingSuperclasses = HashSetFactory.make(loadedSuperclasses);
333         while (node != null) {
334           IClass c = node.getJavaClass();
335           IClass superclass = null;
336           superclass = c.getSuperclass();
337           if (superclass != null) {
338             workingSuperclasses.remove(superclass);
339             Node supernode = findOrCreateNode(superclass);
340             if (DEBUG) {
341               System.err.println(("addChild " + node.getJavaClass() + " to " + supernode.getJavaClass()));
342             }
343             supernode.addChild(node);
344             if (supernode.getJavaClass().getReference().equals(rootTypeRef)) {
345               node = null;
346             } else {
347               node = supernode;
348             }
349           } else {
350             node = null;
351           }
352         }
353     
354         if (loadedSuperInterfaces != null) {
355           for (Iterator it3 = loadedSuperInterfaces.iterator(); it3.hasNext();) {
356             final IClass iface = (IClass) it3.next();
357             try {
358               // make sure we'll be able to load the interface!
359               computeSuperclasses(iface);
360             } catch (IllegalStateException e) {
361               Warnings.add(ClassExclusion.create(iface.getReference(), e.getMessage()));
362               continue;
363             }
364             if (!iface.isInterface()) {
365               Warnings.add(new Warning() {
366                 
367                 @Override
368                 public String getMsg() {
369                   return "class implements non-interface " + iface.getReference() + " as an interface";
370                 }
371               });
372               continue;
373             }
374             recordImplements(klass, iface);
375           }
376         }
377         return true;
378       }

388       /**
389        * Find the possible targets of a call to a method reference. Note that if the reference is to an instance initialization method,
390        * we assume the method was called with invokespecial rather than invokevirtual.
391        * 
392        * @param ref method reference
393        * @return the set of IMethods that this call can resolve to.
394        * @throws IllegalArgumentException if ref is null
395        */
396       @Override
397       public Set<IMethod> getPossibleTargets(MethodReference ref) {
398         if (ref == null) {
399           throw new IllegalArgumentException("ref is null");
400         }
401         IClassLoader loader;
402         try {
403           loader = factory.getLoader(ref.getDeclaringClass().getClassLoader(), this, scope);
404         } catch (IOException e) {
405           throw new UnimplementedError("factory.getLoader failed " + e);
406         }
407         IClass declaredClass;
408         declaredClass = loader.lookupClass(ref.getDeclaringClass().getName());
409         if (declaredClass == null) {
410           return Collections.emptySet();
411         }
412         Set<IMethod> targets = HashSetFactory.make();
413         targets.addAll(findOrCreateTargetSet(declaredClass, ref));
414         return (targets);
415       }

417       /**
418        * Find the possible targets of a call to a method reference
419        * 
420        * @param ref method reference
421        * @return the set of IMethods that this call can resolve to.
422        */
423       @SuppressWarnings("unchecked")
424       private Set<IMethod> findOrCreateTargetSet(IClass declaredClass, MethodReference ref) {
425         Map<MethodReference, Set<IMethod>> classCache = (Map<MethodReference, Set<IMethod>>) CacheReference.get(targetCache
426             .get(declaredClass));
427         if (classCache == null) {
428           classCache = HashMapFactory.make(3);
429           targetCache.put(declaredClass, CacheReference.make(classCache));
430         }
431         Set<IMethod> result = classCache.get(ref);
432         if (result == null) {
433           result = getPossibleTargets(declaredClass, ref);
434           classCache.put(ref, result);
435         }
436         return result;
437       }

439       /**
440        * Find the possible receivers of a call to a method reference
441        * 
442        * @param ref method reference
443        * @return the set of IMethods that this call can resolve to.
444        */
445       @Override
446       public Set<IMethod> getPossibleTargets(IClass declaredClass, MethodReference ref) {
447     
448         if (ref.getName().equals(MethodReference.initAtom)) {
449           // for an object init method, use the method alone as a possible target,
450           // rather than inspecting subclasses
451           IMethod resolvedMethod = resolveMethod(ref);
452           assert resolvedMethod != null;
453           return Collections.singleton(resolvedMethod);
454         }
455         if (declaredClass.isInterface()) {
456           HashSet<IMethod> result = HashSetFactory.make(3);
457           Set impls = implementors.get(declaredClass);
458           if (impls == null) {
459             // give up and return no receivers
460             return Collections.emptySet();
461           }
462           for (Iterator it = impls.iterator(); it.hasNext();) {
463             IClass klass = (IClass) it.next();
464             if (!klass.isInterface() && !klass.isAbstract()) {
465               result.addAll(computeTargetsNotInterface(ref, klass));
466             }
467           }
468           return result;
469         } else {
470           return computeTargetsNotInterface(ref, declaredClass);
471         }
472     
473       }

475       /**
476        * Get the targets for a method ref invoked on a class klass. The klass had better not be an interface.
477        * 
478        * @param ref method to invoke
479        * @param klass declaringClass of receiver
480        * @return Set the set of method implementations that might receive the message
481        */
482       private Set<IMethod> computeTargetsNotInterface(MethodReference ref, IClass klass) {
483     
484         Node n = findNode(klass);
485         HashSet<IMethod> result = HashSetFactory.make(3);
486         // if n is null, then for some reason this class is excluded
487         // from the analysis. Return a result of no targets.
488         if (n == null)
489           return result;
490     
491         Selector selector = ref.getSelector();
492     
493         // try to resolve the method by walking UP the class hierarchy
494         IMethod resolved = resolveMethod(klass, selector);
495     
496         if (resolved != null) {
497           result.add(resolved);
498         }
499     
500         // find any receivers that override the method with inheritance
501         result.addAll(computeOverriders(n, selector));
502     
503         return result;
504       }

507        * Return the unique receiver of an invocation of method on an object of type m.getDeclaredClass

509        * @param m
510        * @return IMethod, or null if no appropriate receiver is found.

519         if (receiver == null) {
520           return null;
521         }
522         Selector selector = m.getSelector();
523         return resolveMethod(receiver, selector);

558       /**
559        * Return the unique target of an invocation of method on an object of type declaringClass
560        * 
561        * @param receiverClass type of receiver
562        * @param selector method signature
563        * @return Method resolved method abstraction
564        * @throws IllegalArgumentException if receiverClass is null
565        */
566       @Override
567       public IMethod resolveMethod(IClass receiverClass, Selector selector) {
568         if (receiverClass == null) {
569           throw new IllegalArgumentException("receiverClass is null");
570         }
571         IMethod result = findMethod(receiverClass, selector);
572         if (result != null) {
573           return result;
574         } else {
575           IClass superclass = null;
576           superclass = receiverClass.getSuperclass();
577           if (superclass == null) {
578             if (DEBUG) {
579               System.err.println(("resolveMethod(" + selector + ") failed: method not found"));
580             }
581             return null;
582           } else {
583             if (DEBUG) {
584               System.err.println(("Attempt to resolve for " + receiverClass + " in superclass: " + superclass + " " + selector));
585             }
586             return resolveMethod(superclass, selector);
587           }
588         }
589       }

602       /**
603        * Get the set of subclasses of a class that provide implementations of a method
604        * 
605        * @param node abstraction of class in question
606        * @param selector method signature
607        * @return Set set of IMethods that override the method
608        */
609       private Set<IMethod> computeOverriders(Node node, Selector selector) {
610         HashSet<IMethod> result = HashSetFactory.make(3);
611         for (Iterator<Node> it = node.getChildren(); it.hasNext();) {
612     
613           Node child = it.next();
614           IMethod m = findMethod(child.getJavaClass(), selector);
615           if (m != null) {
616             result.add(m);
617           }
618           result.addAll(computeOverriders(child, selector));
619         }
620         return result;
621       }

643       private void recursiveStringify(Node n, StringBuffer buffer) {
644         buffer.append(n.toString()).append("\n");

651       /**
652        * Number the class hierarchy tree to support efficient subclass tests. After numbering the tree, n1 is a child of n2 iff n2.left
653        * <= n1.left ^ n1.left <= n2.right. Described as "relative numbering" by Vitek, Horspool, and Krall, OOPSLA 97
654        * 
655        * TODO: this implementation is recursive; un-recursify if needed
656        */
657       private int nextNumber = 1;
658     
659       private void numberTree() {
660         assert root != null;
661         visitForNumbering(root);
662       }
663     
664       private void visitForNumbering(Node N) {
665         N.left = nextNumber++;
666         for (Iterator<Node> it = N.children.iterator(); it.hasNext();) {
667           Node C = it.next();
668           visitForNumbering(C);
669         }
670         N.right = nextNumber++;
671       }

673       /**
674        * internal representation of a node in the class hiearachy, representing one java class.
675        */
676       private static final class Node {
677     
678         private final IClass klass;
679     
680         final private Set<Node> children = HashSetFactory.make(3);
681     
682         // the following two fields are used for efficient subclass tests.
683         // After numbering the tree, n1 is a child of n2 iff
684         // n2.left <= n1.left ^ n1.left <= n2.right.
685         // Described as "relative numbering" by Vitek, Horspool, and Krall, OOPSLA
686         // 97
687         private int left = -1;
688     
689         private int right = -1;
690     
691         Node(IClass klass) {
692           this.klass = klass;
693         }
694     
695         boolean isInterface() {
696           return klass.isInterface();

741       /**
742        * @throws IllegalArgumentException if A is null
743        */
744       @Override
745       public IClass getLeastCommonSuperclass(IClass a, IClass b) {
746         assert (a.getClassLoader().getLanguage().equals(b.getClassLoader().getLanguage()));
747         Language lang = a.getClassLoader().getLanguage();
748         TypeReference tempA = a.getReference();
749         if (a.equals(b)) {
750           return a;
751         } else if (tempA.equals(TypeReference.Null)) {
752           return b;
753         } else if (b.getReference().equals(TypeReference.Null)) {
754           return a;
755         } else if (b.getReference().equals(lang.getRootType())) {
756           return b;
757         } else {
758           Node n = map.get(b.getReference());
759           if (n == null) {
760             assert n != null : "null n for " + b;
761           }
762           Set<IClass> superB;
763           try {
764             superB = getSuperclasses(b);
765           } catch (ClassHierarchyException e1) {
766             e1.printStackTrace();
767             Assertions.UNREACHABLE();
768             superB = null;
769           }
770           IClass aa = a;
771           while (aa != null) {
772             if (b.equals(aa) || superB.contains(aa)) {
773               return aa;
774             }
775             aa = aa.getSuperclass();
776           }
777           Set<IClass> superA;
778           try {
779             superA = getSuperclasses(a);
780           } catch (ClassHierarchyException e1) {
781             e1.printStackTrace();
782             Assertions.UNREACHABLE();
783             superA = null;
784           }
785           Assertions.UNREACHABLE("getLeastCommonSuperclass " + tempA + " " + b + ": " + superA + ", " + superB);
786           return null;
787         }
788       }

790       private Set<IClass> getSuperclasses(IClass c) throws ClassHierarchyException {
791         HashSet<IClass> result = HashSetFactory.make(3);
792         while (c.getSuperclass() != null) {
793           result.add(c.getSuperclass());
794           c = c.getSuperclass();
795         }
796         return result;
797       }

801        * com.ibm.wala.types.TypeReference)

803       @Override
804       public TypeReference getLeastCommonSuperclass(TypeReference a, TypeReference b) {
805         if (a == null) {
806           throw new IllegalArgumentException("a is null");
807         }
808         if (a.equals(b))
809           return a;
810         IClass aClass = lookupClass(a);
811         IClass bClass = lookupClass(b);
812         if (aClass == null || bClass == null) {
813           // One of the classes is not in scope. Give up.
814           if (aClass != null) {
815             return aClass.getClassLoader().getLanguage().getRootType();
816           } else if (bClass != null) {
817             return bClass.getClassLoader().getLanguage().getRootType();
818           } else {
819             return getRootClass().getReference();
820           }
821         }
822         return getLeastCommonSuperclass(aClass, bClass).getReference();
823       }

847       private IClass lookupClassRecursive(TypeReference a) {
848     /** END Custom change: remember unresolved classes */
849         ClassLoaderReference loader = a.getClassLoader();
850     
851         ClassLoaderReference parent = loader.getParent();
852         // first delegate lookup to the parent loader.
853         if (parent != null) {
854           TypeReference p = TypeReference.findOrCreate(parent, a.getName());
855           IClass c = lookupClassRecursive(p);
856           if (c != null) {
857             return c;
858           }
859         }
860     
861         // lookup in the parent failed. lookup based on the declared loader of a.
862         if (a.isArrayType()) {
863           TypeReference elt = a.getInnermostElementType();
864           if (elt.isPrimitiveType()) {
865             // look it up with the primordial loader.
866             return getRootClass().getClassLoader().lookupClass(a.getName());
867           } else {
868             IClass c = lookupClassRecursive(elt);
869             if (c == null) {
870               // can't load the element class, so give up.
871               return null;
872             } else {
873               // we know it comes from c's class loader.
874               return c.getClassLoader().lookupClass(a.getName());
875             }
876           }
877         } else {
878           Node n = map.get(a);
879           if (n != null) {
880             return n.klass;
881           } else {
882             return null;
883           }
884         }
885       }

887       private boolean slowIsSubclass(IClass sub, IClass sup) {
888         if (sub == sup) {
889           return true;
890         } else {
891           IClass parent = sub.getSuperclass();
892           if (parent == null) {
893             return false;
894           } else {
895             return slowIsSubclass(parent, sup);
896           }
897         }

900       /**
901        * Is c a subclass of T?
902        * 
903        * @throws IllegalArgumentException if c is null
904        */
905       @Override
906       public boolean isSubclassOf(IClass c, IClass t) {
907         if (c == null) {
908           throw new IllegalArgumentException("c is null");
909         }
910         assert t != null : "null T";
911     
912         if (c.isArrayClass()) {
913           if (t.getReference() == TypeReference.JavaLangObject) {
914             return true;
915           } else if (t.getReference().isArrayType()) {
916             TypeReference elementType = t.getReference().getArrayElementType();
917             if (elementType.isPrimitiveType()) {
918               return elementType.equals(c.getReference().getArrayElementType());
919             } else {
920               IClass elementKlass = lookupClass(elementType);
921               if (elementKlass == null) {
922                 // uh oh.
923                 Warnings.add(ClassHierarchyWarning.create("could not find " + elementType));
924                 return false;
925               }
926               IClass ce = ((ArrayClass) c).getElementClass();
927               if (ce == null) {
928                 return false;
929               }
930               if (elementKlass.isInterface()) {
931                 return implementsInterface(ce, elementKlass);
932               } else {
933                 return isSubclassOf(ce, elementKlass);
934               }
935             }
936           } else {
937             return false;
938           }
939         } else {
940           if (t.getReference().isArrayType()) {
941             return false;
942           }
943           if (c.getReference().equals(t.getReference())) {
944             return true;
945           }
946           Node n1 = map.get(c.getReference());
947           if (n1 == null) {
948             // some wacky case, like a FakeRootClass
949             return false;
950           }
951           Node n2 = map.get(t.getReference());
952           if (n2 == null) {
953             // some wacky case, like a FakeRootClass
954             return false;
955           }
956           if (n1.left == -1) {
957             return slowIsSubclass(c, t);
958           } else if (n2.left == -1) {
959             return slowIsSubclass(c, t);
960           } else {
961             return (n2.left <= n1.left) && (n1.left <= n2.right);
962           }
963         }
964       }

966       /**
967        * Does c implement i?
968        * 
969        * @return true iff i is an interface and c is a class that implements i, or c is an interface that extends i.
970        */
971       @Override
972       public boolean implementsInterface(IClass c, IClass i) {
973         if (i == null) {
974           throw new IllegalArgumentException("Cannot ask implementsInterface with i == null");
975         }
976         if (c == null) {
977           throw new IllegalArgumentException("Cannot ask implementsInterface with c == null");
978         }
979         if (!i.isInterface()) {
980           return false;
981         }
982         if (c.equals(i)) {
983           return true;
984         }
985         if (c.isArrayClass()) {
986           // arrays implement Cloneable and Serializable
987           return i.equals(lookupClass(TypeReference.JavaLangCloneable)) || i.equals(lookupClass(TypeReference.JavaIoSerializable));
988         }
989         Set impls = implementors.get(i);
990         if (impls != null && impls.contains(c)) {
991           return true;
992         }
993         return false;
994       }

1173       private Collection<IClass> getImmediateArraySubclasses(ArrayClass klass) {
1174         IClass innermost = getInnermostTypeOfArrayClass(klass);
1175         if (innermost == null) {
1176           return Collections.emptySet();
1177         }
1178         Collection<IClass> innermostSubclasses = getImmediateSubclasses(innermost);
1179         int dim = klass.getDimensionality();
1180         Collection<IClass> result = HashSetFactory.make();
1181         for (IClass k : innermostSubclasses) {
1182           TypeReference ref = k.getReference();
1183           for (int i = 0; i < dim; i++) {
1184             ref = ref.getArrayTypeForElementType();
1185           }
1186           result.add(lookupClass(ref));
1187         }
1188         return result;
