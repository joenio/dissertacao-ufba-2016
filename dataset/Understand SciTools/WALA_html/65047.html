11     package com.ibm.wala.ssa;
12     
13     import com.ibm.wala.types.FieldReference;
14     
15     /**
16      * A putfield or putstatic instruction
17      */
18     public abstract class SSAPutInstruction extends SSAFieldAccessInstruction {
19     
20       private final int val;
21     
22       protected SSAPutInstruction(int iindex, int ref, int val, FieldReference field) {
23         super(iindex, field, ref);
24         this.val = val;
25       }
26     
27       protected SSAPutInstruction(int iindex, int val, FieldReference field) {
28         super(iindex, field, -1);
29         this.val = val;
30       }
31     
32       @Override
33       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
34         if (isStatic())
35           return insts.PutInstruction(iindex, uses == null ? val : uses[0], getDeclaredField());
36         else
37           return insts.PutInstruction(iindex, uses == null ? getRef() : uses[0], uses == null ? val : uses[1], getDeclaredField());
38       }
39     
40       @Override
41       public String toString(SymbolTable symbolTable) {
42         if (isStatic()) {
43           return "putstatic " + getDeclaredField() + " = " + getValueString(symbolTable, val);
44         } else {
45           return "putfield " + getValueString(symbolTable, getRef()) + "." + getDeclaredField() + " = " + getValueString(symbolTable, val);
46         }
47       }
48     
49       /**
50        * @see com.ibm.wala.ssa.SSAInstruction#visit(IVisitor)
51        * @throws IllegalArgumentException if v is null
52        */
53       @Override
54       public void visit(IVisitor v) {
55         if (v == null) {
56           throw new IllegalArgumentException("v is null");
57         }
58         v.visitPut(this);
59       }
60     
61       /**
62        * @see com.ibm.wala.ssa.SSAInstruction#getNumberOfUses()
63        */
64       @Override
65       public int getNumberOfUses() {
66         return isStatic() ? 1 : 2;
67       }
68     
69       /**
70        * @see com.ibm.wala.ssa.SSAInstruction#getUse(int)
71        */
72       @Override
73       public int getUse(int j) {
74         assert j == 0 || (!isStatic() && j == 1);
75         return (j == 0 && !isStatic()) ? getRef() : val;
76       }
77     
78       public int getVal() {
79         return val;
80       }
81     
82       @Override
83       public int hashCode() {
84         return val * 9929 ^ 2063;
85       }
86     
87       /*
88        * @see com.ibm.wala.ssa.Instruction#isFallThrough()
89        */
90       @Override
91       public boolean isFallThrough() {
92         return true;
93       }
94     }
