19       /**
20        * @param urlFound the link as appear
21        * @param context the URL in which the link appeared
22        * @return
23        * @throws MalformedURLException
24        */
25       public static URL relativeToAbsoluteUrl(String urlFound, URL context) throws MalformedURLException {
26         urlFound = urlFound.replace("\\", "/").toLowerCase();
27         
28         URL absoluteUrl;
29         if (!isAbsoluteUrl(urlFound)) {
30           if (urlFound.startsWith("//")) {
31             //create URL taking only the protocol from the context
32             String origHostAndPath = urlFound.substring(2);// removing "//"
33             String host;
34             String path;
35             int indexOf = origHostAndPath.indexOf("/");
36             if (indexOf > 0) {
37               host = origHostAndPath.substring(0, indexOf);
38               path = origHostAndPath.substring(indexOf);
39             } else {
40               host = origHostAndPath;
41               path = "";
42             }
43             absoluteUrl = new URL(context.getProtocol(), host, path);
44           } else if (urlFound.startsWith("/")) {
45             //create URL taking the protocol and the host from the context
46             absoluteUrl = new URL(context.getProtocol(), context.getHost(), urlFound);
47           } else {
48             //"concat" URL to context
49             int backDir = 0; // removing directories due to "../" 
50             while(urlFound.startsWith("../")){
51               urlFound = urlFound.substring(3);
52               backDir++;
53             }
54             StringBuilder contextPath = new StringBuilder();
55             String path = context.getPath().replace("\\", "/");
56             boolean isContextDirectory = path.endsWith("/"); 
57             String[] split = path.split("/");
58             // we are also removing last element in case of a directory
59             int rightTrimFromPath = (isContextDirectory ? 0 : 1) + backDir; 
60     
61             for (int i = 0; i < split.length - rightTrimFromPath; i++) {
62               contextPath.append(split[i]);
63               contextPath.append("/");
64             }
65             absoluteUrl = new URL(context.getProtocol(), context.getHost(), contextPath.toString() + urlFound);
66           }
67         } else{
68           absoluteUrl = new URL(urlFound);
69         }
70         return absoluteUrl;
71       }
72     
73       private static boolean isAbsoluteUrl(String orig) {
74         return orig.startsWith("http");
75       }
