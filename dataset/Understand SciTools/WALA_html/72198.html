74       private Collection<Statement> slice(String file, DataDependenceOptions data, ControlDependenceOptions ctrl) throws IOException, WalaException, CancelException {
75         JSCFABuilder B = JSCallGraphBuilderUtil.makeScriptCGBuilder("tests", file);
76         CallGraph CG = B.makeCallGraph(B.getOptions());
77     
78         final Collection<Statement> ss = findTargetStatement(CG);
79      
80         SDG sdg = new SDG(CG, B.getPointerAnalysis(), new JavaScriptModRef(), data, ctrl);
81         Collection<Statement> result = Slicer.computeBackwardSlice(sdg, ss);
82         return result;
83       }
84     
85       private Collection<Statement> findTargetStatement(CallGraph CG) {
86         final Collection<Statement> ss = HashSetFactory.make();
87         for(CGNode n : getNodes(CG, "suffix:_slice_target_fn")) {
88           for(Iterator<CGNode> callers = CG.getPredNodes(n); callers.hasNext(); ) {
89             final CGNode caller = callers.next();
90             for(Iterator<CallSiteReference> sites = CG.getPossibleSites(caller, n); sites.hasNext(); ) {
91               caller.getIR().getCallInstructionIndices(sites.next()).foreach(new IntSetAction() {
92                 @Override
93                 public void act(int x) {
94                   ss.add(new NormalStatement(caller, x));
95                 }
96               });
97             }
98           }
99         }
100         return ss;
101       }
