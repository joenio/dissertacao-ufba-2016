29     /**
30      * Trivial method-level escape analysis.
31      * 
32      * An instance does not escape from method m if the following hold:
33      * <ol>
34      * <li>the instance is only ever pointed to by locals (it is never stored in the heap)
35      * <li>the method m does NOT return (either normally or exceptionally) a pointer to the instance
36      * </ol>
37      */
38     public class TrivialMethodEscape implements IMethodEscapeAnalysis, INodeEscapeAnalysis {
39     
40       /**
41        * Heap graph representation of pointer analysis
42        */
43       private final HeapGraph hg;
44     
45       /**
46        * Governing call graph
47        */
48       private final CallGraph cg;
49     
50       /**
51        * @param hg Heap graph representation of pointer analysis
52        * @param cg governing call graph
53        */
54       public TrivialMethodEscape(CallGraph cg, HeapGraph hg) {
55         this.hg = hg;
56         this.cg = cg;
57       }
58     
59       @Override
60       public boolean mayEscape(MethodReference allocMethod, int allocPC, MethodReference m) throws WalaException {
61     
62         if (allocMethod == null) {
63           throw new IllegalArgumentException("null allocMethod");
64         }
65         // nodes:= set of call graph nodes representing method m
66         Set nodes = cg.getNodes(m);
67         if (nodes.size() == 0) {
68           throw new WalaException("could not find call graph node for method " + m);
69         }
70     
71         // allocN := set of call graph nodes representing method allocMethod
72         Set allocN = cg.getNodes(allocMethod);
73         if (allocN.size() == 0) {
74           throw new WalaException("could not find call graph node for allocation method " + allocMethod);
75         }
76         return mayEscape(allocN, allocPC, nodes);
77       }
78     
79       @Override
80       public boolean mayEscape(CGNode allocNode, int allocPC, CGNode node) throws WalaException {
81         return mayEscape(Collections.singleton(allocNode), allocPC, Collections.singleton(node));
82       }
83     
84       /**
85        * @param allocN Set<CGNode> representing the allocation site.
86        * @param allocPC
87        * @param nodes Set<CGNode>, the nodes of interest
88        * @return true iff some instance allocated at a site N \in &lt;allocN, allocPC> might escape from some activation of a node m \in
89        *         { nodes }
90        * @throws WalaException
91        */
92       private boolean mayEscape(Set allocN, int allocPC, Set nodes) throws WalaException {
93         Set<InstanceKey> instances = HashSetFactory.make();
94         // instances := set of instance key allocated at &lt;allocMethod, allocPC>
95         for (Iterator it = allocN.iterator(); it.hasNext();) {
96           CGNode n = (CGNode) it.next();
97           NewSiteReference site = findAlloc(n, allocPC);
98           InstanceKey ik = hg.getHeapModel().getInstanceKeyForAllocation(n, site);
99           if (ik == null) {
100             throw new WalaException("could not get instance key at site " + site + " in " + n);
101           }
102           instances.add(ik);
103         }
104     
105         for (Iterator<InstanceKey> it = instances.iterator(); it.hasNext();) {
106           InstanceKey ik = it.next();
107           for (Iterator it2 = hg.getPredNodes(ik); it2.hasNext();) {
108             PointerKey p = (PointerKey) it2.next();
109             if (!(p instanceof AbstractLocalPointerKey)) {
110               // a pointer from the heap. give up.
111               return true;
112             } else {
113               if (p instanceof ReturnValueKey) {
114                 ReturnValueKey rk = (ReturnValueKey) p;
115                 if (nodes.contains(rk.getNode())) {
116                   // some node representing method m returns the instance to its
117                   // caller
118                   return true;
119                 }
120               }
121             }
122           }
123         }
124     
125         // if we get here, it may not escape
126         return false;
127       }
128     
129       /**
130        * @param n a call graph node
131        * @param allocPC a bytecode index corresponding to an allocation
132        * @return the NewSiteReference for the allocation
133        * @throws WalaException
134        */
135       static NewSiteReference findAlloc(CGNode n, int allocPC) throws WalaException {
136         if (n == null) {
137           throw new IllegalArgumentException("null n");
138         }
139         for (Iterator it = n.iterateNewSites(); it.hasNext();) {
140           NewSiteReference site = (NewSiteReference) it.next();
141           if (site.getProgramCounter() == allocPC) {
142             return site;
143           }
144         }
145         throw new WalaException("Failed to find an allocation at pc " + allocPC + " in node " + n);
146       }
147     
148     }
