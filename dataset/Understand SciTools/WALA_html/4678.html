11     package com.ibm.wala.ipa.callgraph.propagation;
12     
13     import com.ibm.wala.classLoader.NewSiteReference;
14     import com.ibm.wala.classLoader.ProgramCounter;
15     import com.ibm.wala.ipa.callgraph.CGNode;
16     import com.ibm.wala.types.TypeReference;
17     
18     /**
19      * An object that abstracts how to model instances in the heap.
20      */
21     public interface InstanceKeyFactory {
22       /**
23        * @return the instance key that represents a particular allocation
24        */
25       public abstract InstanceKey getInstanceKeyForAllocation(CGNode node, NewSiteReference allocation);
26     
27       /**
28        * @return the instance key that represents the array allocated as the dim_th
29        *         dimension at a particular allocation
30        */
31       public abstract InstanceKey getInstanceKeyForMultiNewArray(CGNode node, NewSiteReference allocation, int dim);
32     
33       /**
34        * @return the instance key that represents a constant with value S, when considered as a particular type
35        */
36       public abstract <T> InstanceKey getInstanceKeyForConstant(TypeReference type, T S);
37     
38       /**
39        * @param node
40        * @param instr
41        * @param type
42        * @return the instance key that represents the exception of type _type_
43        *         thrown by a particular PEI.
44        */
45       public abstract InstanceKey getInstanceKeyForPEI(CGNode node, ProgramCounter instr, TypeReference type);
46     
47       /**
48        * @param objType TODO
49        * @return the instance key that represents the metadata object obj
50        */
51       public abstract InstanceKey getInstanceKeyForMetadataObject(Object obj, TypeReference objType);
52     
53     }
