76     public class JDTSourceModuleTranslator implements SourceModuleTranslator {
77       protected boolean dump;
78       protected JDTSourceLoaderImpl sourceLoader;

80       public JDTSourceModuleTranslator(AnalysisScope scope, JDTSourceLoaderImpl sourceLoader) {
81         this(scope, sourceLoader, false);

84       public JDTSourceModuleTranslator(AnalysisScope scope, JDTSourceLoaderImpl sourceLoader, boolean dump) {
85         computeClassPath(scope);
86         this.sourceLoader = sourceLoader;
87         this.dump = dump;
88       }
89     
90       private void computeClassPath(AnalysisScope scope) {
91         StringBuffer buf = new StringBuffer();
92     
93         ClassLoaderReference cl = scope.getApplicationLoader();
94     
95         while (cl != null) {
96           List<Module> modules = scope.getModules(cl);
97     
98           for (Iterator<Module> iter = modules.iterator(); iter.hasNext();) {
99             Module m = iter.next();
100     
101             if (buf.length() > 0)
102               buf.append(File.pathSeparator);
103             if (m instanceof JarFileModule) {
104               JarFileModule jarFileModule = (JarFileModule) m;
105     
106               buf.append(jarFileModule.getAbsolutePath());
107             } else if (m instanceof DirectoryTreeModule) {
108               DirectoryTreeModule directoryTreeModule = (DirectoryTreeModule) m;
109     
110               buf.append(directoryTreeModule.getPath());
111             } else
112               Assertions.UNREACHABLE("Module entry is neither jar file nor directory");
113           }
114           cl = cl.getParent();
115         }
116       }

118       /*
119        * Project -> AST code from org.eclipse.jdt.core.tests.performance

122       @Override
123       public void loadAllSources(Set<ModuleEntry> modules) {
124         // TODO: we might need one AST (-> "Object" class) for all files.
125         // TODO: group by project and send 'em in
126     
127         // sort files into projects
128         Map<IProject, Map<ICompilationUnit,EclipseSourceFileModule>> projectsFiles = new HashMap<IProject, Map<ICompilationUnit,EclipseSourceFileModule>>();
129         for (ModuleEntry m : modules) {
130           assert m instanceof EclipseSourceFileModule : "Expecing EclipseSourceFileModule, not " + m.getClass();
131           EclipseSourceFileModule entry = (EclipseSourceFileModule) m;
132           IProject proj = entry.getIFile().getProject();
133           if (!projectsFiles.containsKey(proj)) {
134             projectsFiles.put(proj, new HashMap<ICompilationUnit,EclipseSourceFileModule>());
135           }
136           projectsFiles.get(proj).put(JavaCore.createCompilationUnitFrom(entry.getIFile()), entry);
137         }
138     
139         final ASTParser parser = ASTParser.newParser(AST.JLS4);
140      
141         for (final Map.Entry<IProject,Map<ICompilationUnit,EclipseSourceFileModule>> proj : projectsFiles.entrySet()) {
142           parser.setProject(JavaCore.create(proj.getKey()));
143           parser.setResolveBindings(true);
144      
145           Set<ICompilationUnit> units = proj.getValue().keySet();
146           parser.createASTs(units.toArray(new ICompilationUnit[units.size()]), new String[0], new ASTRequestor() {
147             @Override
148             public void acceptAST(ICompilationUnit source, CompilationUnit ast) {
149     
150               try {
151                 JDTJava2CAstTranslator jdt2cast = makeCAstTranslator(ast, proj.getValue().get(source).getIFile(), source.getUnderlyingResource().getLocation().toOSString());
152                 final Java2IRTranslator java2ir = makeIRTranslator();
153                 java2ir.translate(proj.getValue().get(source), jdt2cast.translateToCAst());
154               } catch (JavaModelException e) {
155                 e.printStackTrace();
156               }
157     
158               if (! "true".equals(System.getProperty("wala.jdt.quiet"))) {
159                 IProblem[] problems = ast.getProblems();
160                 int length = problems.length;
161                 if (length > 0) {
162                   StringBuffer buffer = new StringBuffer();
163                   for (int i = 0; i < length; i++) {
164                     buffer.append(problems[i].getMessage());
165                     buffer.append('\n');
166                   }
167                   if (length != 0)
168                     System.err.println("Unexpected problems in " + source.getElementName() + buffer.toString());
169                 }
170               }
171             }
172           }, null);
173     
174         }
175       }
176     
177       protected Java2IRTranslator makeIRTranslator() {
178         return new Java2IRTranslator(sourceLoader);
179       }
180     
181       protected JDTJava2CAstTranslator makeCAstTranslator(CompilationUnit cu, IFile sourceFile, String fullPath) {
182         return new JDTJava2CAstTranslator(sourceLoader, cu, sourceFile, fullPath, false, dump);
183       }
