13     public class Helper {

16       
17       public Helper() {

19         this.b = new Object();
20         System.out.println("Helper constructor with no parameter invoked");
21       }
22       
23       public Helper(Integer x) {
24         a = b = null;

27       }
28       
29       public Helper(Object a) {

32         System.out.println("Helper constructor with one parameter invoked");
33       }
34

36         this.a = a;
37         this.b = b;

39       }
40       
41       public void m(Object a, Object b, Object c) {
42         System.out.println("m method invoked");
43       }
44       public void n(Object a, Object b) {
45         System.out.println("n method invoked");
46       }
47       public void o(Object a, Object b) {
48         System.out.println("o method invoked");
49       }
50       public static void s(Object a, Object b) {
51         System.out.println("s method invoked");
52       }
53       
54       public static void t(Integer x) {
55         x.toString();
56       }
57       
58       public void u(Integer x) {
59         x.toString();
