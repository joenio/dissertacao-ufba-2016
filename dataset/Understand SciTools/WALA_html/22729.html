21     /**
22      * This class decodes Java bytecodes into ShrikeBT code using a ShrikeCT class reader.
23      */
24     final public class CTDecoder extends Decoder {
25     
26       /**
27        * Decode the code resource 'r'.
28        * 
29        * @throws NullPointerException if r is null
30        */
31       public CTDecoder(CodeReader r) throws NullPointerException {
32         this(r, makeConstantPoolReader(r.getClassReader()));
33       }
34     
35       /**
36        * Decode the code resource 'r' using the predeclared constant pool reader 'cpr' (obtained by makeConstantPoolReader below).
37        * 
38        * @throws NullPointerException if r is null
39        */
40       public CTDecoder(CodeReader r, ConstantPoolReader cpr) throws NullPointerException {
41         super(r.getBytecode(), r.getRawHandlers(), cpr);
42       }
43     
44       /**
45        * Convert the internal JVM class name to a JVM type name (e.g., java/lang/Object to Ljava/lang/Object;).
46        * 
47        * @throws IllegalArgumentException if s is null
48        */
49       public static String convertClassToType(String s) {
50         if (s == null) {
51           throw new IllegalArgumentException("s is null");
52         }
53         if (s.length() > 0 && s.charAt(0) != '[') {
54           return "L" + s + ";";
55         } else {
56           return s;
57         }
58       }
59     
60       /**
61        * Build a ConstantPoolReader implementation to read the constant pool from 'cr'.
62        */
63       public static ConstantPoolReader makeConstantPoolReader(ClassReader cr) throws IllegalArgumentException {
64         if (cr == null) {
65           throw new IllegalArgumentException("illegal null cr");
66         }
67         return new CPReader(cr.getCP());
68       }
69     
70       final static class CPReader extends ConstantPoolReader {

77         @Override
78         public int getConstantPoolItemType(int index) {
79           return cp.getItemType(index);
80         }
