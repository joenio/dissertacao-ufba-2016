41     /**
42      * Test that the SSA-numbering of variables in the IR is deterministic.
43      * 
44      * Introduced 05-AUG-03; the default implementation of hashCode was being
45      * invoked. Object.hashCode is a source of random numbers and has no place in a
46      * deterministic program.
47      */
48     public class TypeInferenceTest extends WalaTestCase {
49     
50       private static final ClassLoader MY_CLASSLOADER = TypeInferenceTest.class.getClassLoader();
51     
52       private static AnalysisScope scope;
53     
54       private static ClassHierarchy cha;
55     
56       private static AnalysisOptions options;
57     
58       private static AnalysisCache cache;
59     
60       public static void main(String[] args) {
61         justThisTest(TypeInferenceTest.class);
62       }
63     
64       @BeforeClass
65       public static void beforeClass() throws Exception {
66     
67         scope = AnalysisScopeReader.readJavaScope(TestConstants.WALA_TESTDATA, (new FileProvider()).getFile("J2SEClassHierarchyExclusions.txt"), MY_CLASSLOADER);
68     
69         options = new AnalysisOptions(scope, null);
70         cache = new AnalysisCache();
71         ClassLoaderFactory factory = new ClassLoaderFactoryImpl(scope.getExclusions());
72     
73         try {
74           cha = ClassHierarchy.make(scope, factory);
75         } catch (ClassHierarchyException e) {
76           throw new Exception();
77         }
78       }
79     
80       @AfterClass
81       public static void afterClass() throws Exception {
82         Warnings.clear();
83         scope = null;
84         cha = null;
85         options = null;
86         cache = null;
87       }
88     
89       @Test public void test1() {
90         MethodReference method = scope.findMethod(AnalysisScope.APPLICATION, "LtypeInference/TI", Atom.findOrCreateUnicodeAtom("foo"),
91             new ImmutableByteArray(UTF8Convert.toUTF8("()V")));
92         Assert.assertNotNull("method not found", method);
93         IMethod imethod = cha.resolveMethod(method);
94         Assert.assertNotNull("imethod not found", imethod);
95         IR ir = cache.getIRFactory().makeIR(imethod, Everywhere.EVERYWHERE, options.getSSAOptions());
96         System.out.println(ir);
97     
98         TypeInference ti = TypeInference.make(ir, false);
99         for (int i = 1; i <= ir.getSymbolTable().getMaxValueNumber(); i++) {
100           System.err.println(i + " " + ti.getType(i));
101         }
102       }
103     
104       @Test public void test2() {
105         MethodReference method = scope.findMethod(AnalysisScope.APPLICATION, "LtypeInference/TI", Atom.findOrCreateUnicodeAtom("bar"),
106             new ImmutableByteArray(UTF8Convert.toUTF8("(I)V")));
107         Assert.assertNotNull("method not found", method);
108         IMethod imethod = cha.resolveMethod(method);
109         Assert.assertNotNull("imethod not found", imethod);
110         IR ir = cache.getIRFactory().makeIR(imethod, Everywhere.EVERYWHERE, options.getSSAOptions());
111         System.out.println(ir);
112     
113         TypeInference ti = TypeInference.make(ir, true);
114         Assert.assertNotNull("null type abstraction for parameter", ti.getType(2));
115       }
116     
117       @Test public void test3() {
118         MethodReference method = scope.findMethod(AnalysisScope.APPLICATION, "LtypeInference/TI", Atom.findOrCreateUnicodeAtom("inferInt"),
119             new ImmutableByteArray(UTF8Convert.toUTF8("()V")));
120         Assert.assertNotNull("method not found", method);
121         IMethod imethod = cha.resolveMethod(method);
122         Assert.assertNotNull("imethod not found", imethod);
123         IR ir = cache.getIRFactory().makeIR(imethod, Everywhere.EVERYWHERE, options.getSSAOptions());
124         System.out.println(ir);
125     
126         TypeInference ti = TypeInference.make(ir, true);
127         TypeAbstraction type = ti.getType(7);
128         Assert.assertNotNull("null type abstraction", type);
129         Assert.assertTrue("inferred wrong type", type.toString().equals("int"));
130       }
131     
132       @Test public void test4() {
133         MethodReference method = scope.findMethod(AnalysisScope.APPLICATION, "LtypeInference/TI", Atom.findOrCreateUnicodeAtom("useCast"),
134             new ImmutableByteArray(UTF8Convert.toUTF8("(Ljava/lang/Object;)V")));
135         Assert.assertNotNull("method not found", method);
136         IMethod imethod = cha.resolveMethod(method);
137         Assert.assertNotNull("imethod not found", imethod);
138         IR ir = cache.getIRFactory().makeIR(imethod, Everywhere.EVERYWHERE, options.getSSAOptions());
139         System.out.println(ir);
140     
141         TypeInference ti = TypeInference.make(ir, false);
142         TypeAbstraction type = ti.getType(4);
143         Assert.assertNotNull("null type abstraction", type);
144         Assert.assertTrue("inferred wrong type " + type, type instanceof ConeType && ((ConeType)type).getTypeReference().getName().toString().equals("Ljava/lang/String"));
145       }
