17     /**
18      * This class is a container for a bunch of information that we might know about a method. It's here for convenience so users can
19      * just pass one MethodData around instead of passing around an array of instructions, an array of exception handler lists, etc.
20      * 
21      * It also provides a place to hang annotations. We provide a table mapping abstract "keys" to annotation objects. We also provide
22      * an invalidation protocol so that the annotation objects can be notified of code changes and react appropriately. This is useful
23      * for caching analysis results for a method.
24      */
25     public final class MethodData {
26       final private HashMap<Object, Results> map = new HashMap<Object, Results>();
27     
28       final private int access;
29     
30       final private String classType;
31     
32       final private String name;
33     
34       final private String signature;
35     
36       private IInstruction[] instructions;
37     
38       private ExceptionHandler[][] handlers;
39     
40       private int[] instructionsToBytecodes;
41     
42       private boolean hasChanged = false;
43     
44       /**
45        * Create information for a method, with no exception handlers and a dummy mapping of instructions to original bytecodes. 
46        * @param access the access flags
47        * @param classType the class in which the method is defined, in JVM type format (e.g., Ljava/lang/Object;)
48        * @param name the method name
49        * @param signature the method signature, in JVM type format (e.g., (ILjava/lang/Object;)V)
50        * @param instructions the instructions making up the method
51        */
52       public static MethodData makeWithDefaultHandlersAndInstToBytecodes(int access, String classType, String name, String signature, IInstruction[] instructions) {
53         ExceptionHandler[][] handlers = new ExceptionHandler[instructions.length][];
54         Arrays.fill(handlers, new ExceptionHandler[0]);
55         int[] i2b = new int[instructions.length];
56         for (int i = 0; i < i2b.length; i++) {
57           i2b[i] = i;
58         }
59         return new MethodData(access, classType, name, signature, instructions, handlers, i2b);    
60       }
61       
62       /**
63        * Gather the information for a method "from scratch".
64        * 
65        * @param access the access flags
66        * @param classType the class in which the method is defined, in JVM type format (e.g., Ljava/lang/Object;)
67        * @param name the method name
68        * @param signature the method signature, in JVM type format (e.g., (ILjava/lang/Object;)V)
69        * @param instructions the instructions making up the method
70        * @param handlers a list of exception handlers for each instruction
71        * @param instructionsToBytecodes a map stating, for each instruction, the offset of the original bytecode instruction(s) giving
72        *          rise to this instruction
73        */
74       public MethodData(int access, String classType, String name, String signature, IInstruction[] instructions,
75           ExceptionHandler[][] handlers, int[] instructionsToBytecodes) {
76         this.classType = classType;
77         this.access = access;
78         this.name = name;
79         this.signature = signature;
80         this.instructions = instructions;
81         this.handlers = handlers;
82         this.instructionsToBytecodes = instructionsToBytecodes;
83     
84         if (instructions == null) {
85           throw new IllegalArgumentException("Instruction array cannot be null");
86         }
87         if (handlers == null) {
88           throw new IllegalArgumentException("Handler array cannot be null");
89         }
90         if (instructionsToBytecodes == null) {
91           throw new IllegalArgumentException("InstructionToBytecodes array cannot be null");
92         }
93         if (instructions.length != handlers.length) {
94           throw new IllegalArgumentException("Handlers array must be the same length as the instructions");
95         }
96         if (instructions.length != instructionsToBytecodes.length) {
97           throw new IllegalArgumentException("Bytecode map array must be the same length as the instructions");
98         }
99       }
100     
101       /**
102        * Gather the information for a method after it has been decoded.
103        * 
104        * @param d the decoder which has decoded the method
105        * @param access the access flags
106        * @param classType the class in which the method is defined, in JVM type format (e.g., Ljava/lang/Object;)
107        * @param name the method name
108        * @param signature the method signature, in JVM type format (e.g., (ILjava/lang/Object;)V)
109        * @throws NullPointerException if d is null
110        */
111       public MethodData(Decoder d, int access, String classType, String name, String signature) throws NullPointerException {
112         this(access, classType, name, signature, d.getInstructions(), d.getHandlers(), d.getInstructionsToBytecodes());
113       }

116         hasChanged = true;
117       }

124       }

129       public String getName() {

172         return instructions;

178       public int[] getInstructionsToBytecodes() {
179         return instructionsToBytecodes;

182       /**
183        * Annotation objects implement this Results interface. The Results interface is used to notify an annotation that the method code
184        * has been updated.
185        */
186       public static interface Results {
187         /**
188          * This method is called just before the code for a method changes. The existing instructions, handlers, etc can be read from
189          * the current info.
190          * 
191          * @param info the method data this annotation is attached to
192          * @param newInstructions the instructions the method will change to
193          * @param newHandlers the handler lists the method will change to
194          * @param newInstructionMap the instructions-to-bytecodes map the method will change to
195          * @return true to remove the object from the info set, for example because the annotation is now invalid
196          */
197         public boolean notifyUpdate(MethodData info, IInstruction[] newInstructions, ExceptionHandler[][] newHandlers,
198             int[] newInstructionMap);
199       }
200     
201       /**
202        * Get the annotation for the given key.
203        * 
204        * @return the annotation or null if there isn't one
205        */
206       public Results getInfo(Object key) {
207         return map.get(key);
208       }
209     
210       /**
211        * Set the annotation for the given key.
212        */
213       public void putInfo(Object key, Results value) {
214         map.put(key, value);
215       }
216     
217       void update(IInstruction[] instructions, ExceptionHandler[][] handlers, int[] newInstructionMap, int[] instructionsToBytecodes) {
218         for (Iterator<Object> i = map.keySet().iterator(); i.hasNext();) {
219           Object key = i.next();
220           Results r = map.get(key);
221           if (r.notifyUpdate(this, instructions, handlers, newInstructionMap)) {
222             i.remove();
223           }
224         }
225     
226         this.instructions = instructions;
227         this.handlers = handlers;
228         this.instructionsToBytecodes = instructionsToBytecodes;
229         hasChanged = true;
230       }
231     
232       /**
233        * @return true iff the code has been updated at least once
234        */
235       public boolean getHasChanged() {
236         return hasChanged;
237       }
238     
239       @Override
240       public String toString() {
241         return getClassType() + "." + getName() + getSignature();
242       }
243     }
