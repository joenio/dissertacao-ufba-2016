11     package com.ibm.wala.viz.viewer;
12     
13     import java.awt.Component;
14     
15     import javax.swing.ImageIcon;
16     import javax.swing.JTree;
17     import javax.swing.tree.DefaultMutableTreeNode;
18     import javax.swing.tree.DefaultTreeCellRenderer;
19     import javax.swing.tree.TreeCellRenderer;
20     
21     import com.ibm.wala.ipa.callgraph.propagation.InstanceKey;
22     import com.ibm.wala.ipa.callgraph.propagation.PointerKey;
23     
24     /**
25      * Renderer for the heap tree. Gives non default icons for pointer keys and instance keys.
26      * @author yinnonh
27      *
28      */
29     class DualTreeCellRenderer implements TreeCellRenderer {
30     
31       private final DefaultTreeCellRenderer defaultTreeCellRenderer;
32       private final DefaultTreeCellRenderer ikTreeCellRenderer;
33       private final DefaultTreeCellRenderer pkTreeCellRenderer;
34     
35       public DualTreeCellRenderer() {
36         defaultTreeCellRenderer = new DefaultTreeCellRenderer();
37         
38         
39         ikTreeCellRenderer = new DefaultTreeCellRenderer();
40         ikTreeCellRenderer.setOpenIcon(createImageIcon("images/ik_open.png"));
41         ikTreeCellRenderer.setClosedIcon(createImageIcon("images/ik_closed.png"));
42         ikTreeCellRenderer.setLeafIcon(createImageIcon("images/ik_leaf.png"));
43     
44         pkTreeCellRenderer = new DefaultTreeCellRenderer();
45         pkTreeCellRenderer.setOpenIcon(createImageIcon("images/pk_open.png"));
46         pkTreeCellRenderer.setClosedIcon(createImageIcon("images/pk_closed.png"));
47         pkTreeCellRenderer.setLeafIcon(createImageIcon("images/pk_leaf.png"));
48       }
49       
50       @Override
51       public Component getTreeCellRendererComponent(JTree tree, Object value, boolean selected, boolean expanded, boolean leaf,
52           int row, boolean hasFocus) {
53         TreeCellRenderer delegate = getTreeCellRenderer(value);
54         return delegate.getTreeCellRendererComponent(tree, value, selected, expanded, leaf, row, hasFocus);
55       }
56     
57       private TreeCellRenderer getTreeCellRenderer(Object value) {
58         if (value instanceof DefaultMutableTreeNode){
59           value = ((DefaultMutableTreeNode) value).getUserObject();
60         }
61         if (value instanceof PointerKey){
62           return pkTreeCellRenderer;
63         } else if (value instanceof InstanceKey){
64           return ikTreeCellRenderer;
65         } else {
66           return defaultTreeCellRenderer;
67         }
68       }
69     
70       /** Returns an ImageIcon, or null if the path was invalid. */
71       protected ImageIcon createImageIcon(String path) {
72           java.net.URL imgURL = DualTreeCellRenderer.class.getResource(path);
73           if (imgURL != null) {
74               return new ImageIcon(imgURL);
75           } else {
76               System.err.println("Couldn't find file: " + path);
77               return null;
78           }
79       }
80     
81     }
