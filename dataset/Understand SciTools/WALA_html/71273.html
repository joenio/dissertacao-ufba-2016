63       public static Collection<CAstQualifier> mapModifiersToQualifiers(int modifiers, boolean isInterface, boolean isAnnotation) {
64         Set<CAstQualifier> quals = new LinkedHashSet<CAstQualifier>();
65     
66         if (isInterface)
67           quals.add(CAstQualifier.INTERFACE);
68     
69         if (isAnnotation)
70           quals.add(CAstQualifier.ANNOTATION);
71         
72         if ((modifiers & Modifier.ABSTRACT) != 0)
73           quals.add(CAstQualifier.ABSTRACT);
74         if ((modifiers & Modifier.FINAL) != 0)
75           quals.add(CAstQualifier.FINAL);
76         if ((modifiers & Modifier.NATIVE) != 0)
77           quals.add(CAstQualifier.NATIVE);
78         // if (flags.isPackage()) quals.add(CAstQualifier.PACKAGE);
79         if ((modifiers & Modifier.PRIVATE) != 0)
80           quals.add(CAstQualifier.PRIVATE);
81         if ((modifiers & Modifier.PROTECTED) != 0)
82           quals.add(CAstQualifier.PROTECTED);
83         if ((modifiers & Modifier.PUBLIC) != 0)
84           quals.add(CAstQualifier.PUBLIC);
85         if ((modifiers & Modifier.STATIC) != 0)
86           quals.add(CAstQualifier.STATIC);
87         if ((modifiers & Modifier.STRICTFP) != 0)
88           quals.add(CAstQualifier.STRICTFP);
89         if ((modifiers & Modifier.SYNCHRONIZED) != 0)
90           quals.add(CAstQualifier.SYNCHRONIZED);
91         if ((modifiers & Modifier.TRANSIENT) != 0)
92           quals.add(CAstQualifier.TRANSIENT);
93         if ((modifiers & Modifier.VOLATILE) != 0)
94           quals.add(CAstQualifier.VOLATILE);
95     
96         return quals;
97       }

99       public static CAstOperator mapAssignOperator(Operator op) {
100         if (op == Assignment.Operator.PLUS_ASSIGN)
101           return CAstOperator.OP_ADD;
102         else if (op == Assignment.Operator.BIT_AND_ASSIGN)
103           return CAstOperator.OP_BIT_AND;
104         else if (op == Assignment.Operator.BIT_OR_ASSIGN)
105           return CAstOperator.OP_BIT_OR;
106         else if (op == Assignment.Operator.BIT_XOR_ASSIGN)
107           return CAstOperator.OP_BIT_XOR;
108         else if (op == Assignment.Operator.DIVIDE_ASSIGN)
109           return CAstOperator.OP_DIV;
110         else if (op == Assignment.Operator.REMAINDER_ASSIGN)
111           return CAstOperator.OP_MOD;
112         else if (op == Assignment.Operator.TIMES_ASSIGN)
113           return CAstOperator.OP_MUL;
114         else if (op == Assignment.Operator.LEFT_SHIFT_ASSIGN)
115           return CAstOperator.OP_LSH;
116         else if (op == Assignment.Operator.RIGHT_SHIFT_SIGNED_ASSIGN)
117           return CAstOperator.OP_RSH;
118         else if (op == Assignment.Operator.MINUS_ASSIGN)
119           return CAstOperator.OP_SUB;
120         else if (op == Assignment.Operator.RIGHT_SHIFT_UNSIGNED_ASSIGN)
121           return CAstOperator.OP_URSH;
122         Assertions.UNREACHABLE("Unknown assignment operator");
123         return null;
124       }

126       protected static CAstOperator mapBinaryOpcode(InfixExpression.Operator operator) {
127         if (operator == InfixExpression.Operator.PLUS)
128           return CAstOperator.OP_ADD;
129         // separate bitwise and logical AND / OR ? '&' / '|' ?
130         if (operator == InfixExpression.Operator.AND)
131           return CAstOperator.OP_BIT_AND;
132         if (operator == InfixExpression.Operator.OR)
133           return CAstOperator.OP_BIT_OR;
134         if (operator == InfixExpression.Operator.XOR)
135           return CAstOperator.OP_BIT_XOR;
136     
137         // TODO: shouldn't get here (conditional and handled differently); however should separate bitwise & logical '&' / '|', maybe.
138         if (operator == InfixExpression.Operator.CONDITIONAL_AND)
139           return CAstOperator.OP_REL_AND;
140         if (operator == InfixExpression.Operator.CONDITIONAL_OR)
141           return CAstOperator.OP_REL_OR;
142     
143         if (operator == InfixExpression.Operator.DIVIDE)
144           return CAstOperator.OP_DIV;
145         if (operator == InfixExpression.Operator.EQUALS)
146           return CAstOperator.OP_EQ;
147         if (operator == InfixExpression.Operator.GREATER_EQUALS)
148           return CAstOperator.OP_GE;
149         if (operator == InfixExpression.Operator.GREATER)
150           return CAstOperator.OP_GT;
151         if (operator == InfixExpression.Operator.LESS_EQUALS)
152           return CAstOperator.OP_LE;
153         if (operator == InfixExpression.Operator.LESS)
154           return CAstOperator.OP_LT;
155         if (operator == InfixExpression.Operator.REMAINDER)
156           return CAstOperator.OP_MOD;
157         if (operator == InfixExpression.Operator.TIMES)
158           return CAstOperator.OP_MUL;
159         if (operator == InfixExpression.Operator.NOT_EQUALS)
160           return CAstOperator.OP_NE;
161         if (operator == InfixExpression.Operator.LEFT_SHIFT)
162           return CAstOperator.OP_LSH;
163         if (operator == InfixExpression.Operator.RIGHT_SHIFT_SIGNED)
164           return CAstOperator.OP_RSH;
165         if (operator == InfixExpression.Operator.MINUS)
166           return CAstOperator.OP_SUB;
167         if (operator == InfixExpression.Operator.RIGHT_SHIFT_UNSIGNED)
168           return CAstOperator.OP_URSH;
169         Assertions.UNREACHABLE("Java2CAstTranslator.JavaTranslatingVisitorImpl.mapBinaryOpcode(): unrecognized binary operator.");
170         return null;
171       }

173       /**
174        * Returns true if type is char, byte, short, int, or long. Return false otherwise (including boolean!)
175        * 
176        * @param type
177        * @return
178        */
179       public static boolean isLongOrLess(ITypeBinding type) {
180         String t = type.getBinaryName();
181         return t.equals("C") || t.equals("B") || t.equals("S") || t.equals("I") || t.equals("J");
182       }
183     
184       /**
185        * If isLongOrLess(type), returns Integer(0). If a float or double, returns Double(0.0) Otherwise (including boolean), returns
186        * CAstSymbol.NULL_DEFAULT_VALUE.
187        * 
188        * @param type
189        * @return
190        */
191       public static Object defaultValueForType(ITypeBinding type) {
192         if (isLongOrLess(type))
193           return new Integer(0);
194         else if (type.getBinaryName().equals("D") || type.getBinaryName().equals("F"))
195           return new Double(0.0);
196         else
197           return CAstSymbol.NULL_DEFAULT_VALUE;
198       }

200       public static ITypeBinding promoteTypes(ITypeBinding t1, ITypeBinding t2, AST ast) {
201         // JLS 5.6.2
202         ITypeBinding doble = ast.resolveWellKnownType("double");
203         if (t1.equals(doble) || t2.equals(doble))
204           return doble;
205         ITypeBinding flotando = ast.resolveWellKnownType("float");
206         if (t1.equals(flotando) || t2.equals(flotando))
207           return flotando;
208         ITypeBinding largo = ast.resolveWellKnownType("long");
209         if (t1.equals(largo) || t2.equals(largo))
210           return largo;
211         return ast.resolveWellKnownType("int");
212       }

214       public static ITypeBinding getDeclaringClassOfNode(ASTNode n) {
215         ASTNode current = n;
216         while (current != null) {
217           if (current instanceof TypeDeclaration)
218             return ((TypeDeclaration) current).resolveBinding();
219           else if (current instanceof AnonymousClassDeclaration)
220             return ((AnonymousClassDeclaration) current).resolveBinding();
221           else if (current instanceof EnumDeclaration)
222             return ((EnumDeclaration) current).resolveBinding();
223           current = current.getParent();
224         }
225         Assertions.UNREACHABLE("Couldn't find declaring class of node");
226         return null;
227       }

229       static String anonTypeName(ITypeBinding ct) {
230         String binName = ct.getBinaryName();
231         String dollarSignNumber = binName.substring(binName.indexOf('$'));
232         return "<anonymous subclass of " + ct.getSuperclass().getBinaryName() + ">" + dollarSignNumber;
233       }

243         if (returnType.isTypeVariable() || returnType.isCapture())
244           return getTypesVariablesBase(returnType, ast);
245         return returnType.getTypeDeclaration(); // Things like "Collection<? extends Bla>" are parameterized types...

248       public static ITypeBinding getTypesVariablesBase(ITypeBinding returnType, AST ast) {
249         assert returnType.isTypeVariable() || returnType.isCapture();
250         if (returnType.getTypeBounds().length > 0)
251           return returnType.getTypeBounds()[0]; // TODO: why is there more than one bound?
252         else
253           return ast.resolveWellKnownType("java.lang.Object");
254       }

256       public static InfixExpression.Operator mapAssignOperatorToInfixOperator(Assignment.Operator op) {
257         if (op == Assignment.Operator.PLUS_ASSIGN)
258           return InfixExpression.Operator.PLUS;
259         else if (op == Assignment.Operator.BIT_AND_ASSIGN)
260           return InfixExpression.Operator.AND;
261         else if (op == Assignment.Operator.BIT_OR_ASSIGN)
262           return InfixExpression.Operator.OR;
263         else if (op == Assignment.Operator.BIT_XOR_ASSIGN)
264           return InfixExpression.Operator.XOR;
265         else if (op == Assignment.Operator.DIVIDE_ASSIGN)
266           return InfixExpression.Operator.DIVIDE;
267         else if (op == Assignment.Operator.REMAINDER_ASSIGN)
268           return InfixExpression.Operator.REMAINDER;
269         else if (op == Assignment.Operator.TIMES_ASSIGN)
270           return InfixExpression.Operator.TIMES;
271         else if (op == Assignment.Operator.LEFT_SHIFT_ASSIGN)
272           return InfixExpression.Operator.LEFT_SHIFT;
273         else if (op == Assignment.Operator.RIGHT_SHIFT_SIGNED_ASSIGN)
274           return InfixExpression.Operator.RIGHT_SHIFT_SIGNED;
275         else if (op == Assignment.Operator.MINUS_ASSIGN)
276           return InfixExpression.Operator.MINUS;
277         else if (op == Assignment.Operator.RIGHT_SHIFT_UNSIGNED_ASSIGN)
278           return InfixExpression.Operator.RIGHT_SHIFT_UNSIGNED;
279         Assertions.UNREACHABLE("Unknown assignment operator");
280         return null;
281       }

283       private static void getMethodInClassOrSuperclass(IMethodBinding met, ITypeBinding klass, boolean superclassonly,
284           HashMap<ITypeBinding, IMethodBinding> overridden) {
285         if (!superclassonly) {
286           for (IMethodBinding ourmet : klass.getDeclaredMethods())
287             if (met.overrides(ourmet)) {
288               overridden.put(ourmet.getMethodDeclaration().getReturnType(), ourmet.getMethodDeclaration());
289               break; // there can only be one per class so don't bother looking for more
290             }
291         }
292     
293         for (ITypeBinding iface : klass.getInterfaces())
294           getMethodInClassOrSuperclass(met, iface, false, overridden);
295     
296         ITypeBinding superclass = klass.getSuperclass();
297         if (superclass != null)
298           getMethodInClassOrSuperclass(met, superclass, false, overridden);
299       }

301       public static Collection<IMethodBinding> getOverriddenMethod(IMethodBinding met) {
302         HashMap<ITypeBinding, IMethodBinding> overridden = new HashMap<ITypeBinding, IMethodBinding>();
303         if (met == null)
304           return null;
305         getMethodInClassOrSuperclass(met, met.getDeclaringClass(), true, overridden);
306         if (overridden.size() == 0)
307           return null;
308         return overridden.values();
309       }

311       public static boolean sameErasedSignatureAndReturnType(IMethodBinding met1, IMethodBinding met2) {
312         if (!met1.getReturnType().getErasure().isEqualTo(met2.getReturnType().getErasure()))
313           return false;
314     
315         ITypeBinding[] params1 = met1.getParameterTypes();
316         ITypeBinding[] params2 = met2.getParameterTypes();
317         if (params1.length != params2.length)
318           return false;
319     
320         for (int i = 0; i < params1.length; i++)
321           if (!params1[i].getErasure().isEqualTo(params2[i].getErasure()))
322             return false;
323     
324         return true;
325       }
