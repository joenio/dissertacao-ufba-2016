106       /**
107        * @return a graph with the expected structure
108        */
109       private static Graph<String> buildGraph() {
110         Graph<String> G = SlowSparseNumberedGraph.make();
111         for (int i = 0; i < nodeNames.length(); i++) {
112           String n = nodeNames.substring(i, i + 1);
113           G.addNode(n);
114           nodes[i] = n;
115         }
116         G.addEdge(nodes[0], nodes[1]);
117         G.addEdge(nodes[1], nodes[2]);
118         G.addEdge(nodes[1], nodes[3]);
119         G.addEdge(nodes[2], nodes[4]);
120         G.addEdge(nodes[3], nodes[4]);
121         G.addEdge(nodes[4], nodes[5]);
122         return G;
123       }
124     
125       /**
126        * Solve the dataflow system and return the result as a string
127        * @throws CancelException 
128        */
129       private static String solveNodeOnly(Graph<String> G) throws CancelException {
130         final OrdinalSetMapping<String> values = new MutableMapping<String>(nodes);
131         ITransferFunctionProvider<String, BitVectorVariable> functions = new ITransferFunctionProvider<String, BitVectorVariable>() {
132     
133           @Override
134           public UnaryOperator<BitVectorVariable> getNodeTransferFunction(String node) {
135             return new BitVectorUnionConstant(values.getMappedIndex(node));
136           }
137     
138           @Override
139           public boolean hasNodeTransferFunctions() {
140             return true;
141           }
142     
143           @Override
144           public UnaryOperator<BitVectorVariable> getEdgeTransferFunction(String from, String to) {
145             Assertions.UNREACHABLE();
146             return null;
147           }
148     
149           @Override
150           public boolean hasEdgeTransferFunctions() {
151             return false;
152           }
153     
154           @Override
155           public AbstractMeetOperator<BitVectorVariable> getMeetOperator() {
156             return BitVectorUnion.instance();
157           }
158     
159         };
160     
161         BitVectorFramework<String,String> F = new BitVectorFramework<String,String>(G, functions, values);
162         BitVectorSolver<String> s = new BitVectorSolver<String>(F);
163         s.solve(null);
164         return result2String(s);
165       }
166     
167       private static String solveNodeEdge(Graph<String> G) throws CancelException {
168         final OrdinalSetMapping<String> values = new MutableMapping<String>(nodes);
169         ITransferFunctionProvider<String, BitVectorVariable> functions = new ITransferFunctionProvider<String, BitVectorVariable>() {
170     
171           @Override
172           public UnaryOperator<BitVectorVariable> getNodeTransferFunction(String node) {
173             return new BitVectorUnionConstant(values.getMappedIndex(node));
174           }
175     
176           @Override
177           public boolean hasNodeTransferFunctions() {
178             return true;
179           }
180     
181           @Override
182           public UnaryOperator<BitVectorVariable> getEdgeTransferFunction(String from, String to) {
183             if (from == nodes[1] && to == nodes[3])
184               return new BitVectorFilter(zero());
185             else if (from == nodes[1] && to == nodes[2])
186               return new BitVectorFilter(one());
187             else {
188               return BitVectorIdentity.instance();
189             }
190           }
191     
192           @Override
193           public boolean hasEdgeTransferFunctions() {
194             return true;
195           }
196     
197           @Override
198           public AbstractMeetOperator<BitVectorVariable> getMeetOperator() {
199             return BitVectorUnion.instance();
200           }
201     
202         };
203     
204         BitVectorFramework<String,String> F = new BitVectorFramework<String,String>(G, functions, values);
205         BitVectorSolver<String> s = new BitVectorSolver<String>(F);
206         s.solve(null);
207         return result2String(s);
208       }
209     
210       public static String result2String(BitVectorSolver<String> solver) {
211         StringBuffer result = new StringBuffer("------\n");
212         for (int i = 0; i < nodes.length; i++) {
213           String n = nodes[i];
214           BitVectorVariable varI = solver.getOut(n);
215           String s = varI.toString();
216           result.append("Node " + n + "(" + i + ") = " + s + "\n");
217         }
218         return result.toString();
219       }
