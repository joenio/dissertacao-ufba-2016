11     package com.ibm.wala.shrikeBT;
12     
13     /**
14      * Instructions are immutable objects. It is always legal to take a reference to an instruction and use it in any other context. You
15      * never need to copy instructions. It is often a good idea to keep references to frequently used instructions cached in static
16      * fields.
17      * 
18      * To generate an Instruction, locate the class for the instruction you want to generate, and invoke the appropriate "make" static
19      * method on that class. The Util class has convenience methods for creating some of the more complex instructions using reflection
20      * to fill in some of the needed information (e.g., makeGet, makePut, makeInvoke).
21      * 
22      * We simplify the bytecode instruction set a bit using some preprocessing and postprocessing:
23      * 
24      * There is no 'iinc' instruction. 'iinc' instructions are expanded to 'iload; bipush; iadd; istore' during decoding and reassembled
25      * during compilation.
26      * 
27      * There are no 'jsr' or 'ret' instructions. Bytecode subroutines are expanded inline during decoding.
28      * 
29      * There are no 'ifeq', 'ifne', 'ifgt', 'ifge', 'iflt', 'ifle' instructions. These instructions are expanded to 'bipush 0; if_icmp'
30      * during decoding and reassembled during compilation.
31      * 
32      * There are no 'ifnull' or 'ifnonnull' instructions. These instructions are expanded to 'aconst_null; if_acmp' during decoding and
33      * reassembled during compilation.
34      * 
35      * All Java values, including longs and doubles, occupy just one word on the stack. Places where the JVM assumes differently are
36      * fixed up during compilation. However, longs and double still take up two local variable slots (this usually doesn't matter to
37      * instrumentation).
38      * 
39      * Control transfer instructions refer to target instructions using integers. These integers are usually indices into an array of
40      * instructions.
41      */
42     public abstract class Instruction implements Constants, Cloneable, IInstruction {
43       public final static int[] noInstructions = new int[0];
44     
45       /**
46        * Ensure that only this package can subclass Instruction.
47        */
48       Instruction(short opcode) {
49         this.opcode = opcode;
50       }
51     
52       final short opcode;
53     
54       /**
55        * @return true if the instruction can "fall through" to the following instruction
56        */
57       @Override
58       public boolean isFallThrough() {
59         return true;
60       }
61     
62       /**
63        * @return an array containing the labels this instruction can branch to (not including the following instruction if this
64        *         instruction 'falls through')
65        */
66       @Override
67       public int[] getBranchTargets() {
68         return noInstructions;
69       }
70     
71       /**
72        * @return an Instruction equivalent to this one but with any branch labels updated by looking them up in the targetMap array
73        */
74       @Override
75       public IInstruction redirectTargets(int[] targetMap) {
76         return this;
77       }
78     
79       /**
80        * @return the number of values this instruction pops off the working stack
81        */
82       @Override
83       public int getPoppedCount() {
84         return 0;
85       }
86     
87       /**
88        * @return the opcode selected for this instruction, or -1 if we don't know it yet
89        */
90       public final short getOpcode() {
91         return opcode;
92       }
93     
94       /**
95        * Computes the type of data pushed onto the stack, or null if none is pushed.
96        * 
97        * @param poppedTypesToCheck the types of the data popped off the stack by this instruction; if poppedTypes is null, then we don't
98        *          know the incoming stack types and the result of this method may be less accurate
99        */
100       @Override
101       public String getPushedType(String[] poppedTypesToCheck) {
102         return null;
103       }
104     
105       /**
106        * @return the JVM word size of the value this instruction pushes onto the stack, or 0 if this instruction doesn't push anything
107        *         onto the stack.
108        */
109       @Override
110       public byte getPushedWordSize() {
111         return 0;
112       }
113     
114       /**
115        * Apply a Visitor to this instruction. We invoke the appropriate Visitor method according to the type of this instruction.
116        */
117       @Override
118       public abstract void visit(IInstruction.Visitor v);
119     
120       /**
121        * Subclasses must implement toString.
122        */
123       @Override
124       public abstract String toString();
125     
126       /**
127        * We're immutable so there's no need to clone any Instruction object.
128        */
129       @Override
130       final public Object clone() {
131         return this;
132       }
133     }
