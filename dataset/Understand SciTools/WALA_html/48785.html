152       private static void doTests(AnalysisScope scope, final ClassHierarchy cha, AnalysisOptions options) throws IllegalArgumentException, CancelException {
153         final SSAPropagationCallGraphBuilder builder = Util.makeVanillaZeroOneCFABuilder(options, new AnalysisCache(), cha, scope);
154         final CallGraph oldCG = builder.makeCallGraph(options,null);
155         final PointerAnalysis<InstanceKey> pa = builder.getPointerAnalysis();
156     
157         // now, run our analysis
158         // build an RTA call graph
159         CallGraphBuilder rtaBuilder = Util.makeRTABuilder(options, new AnalysisCache(), cha, scope);
160         final CallGraph cg = rtaBuilder.makeCallGraph(options, null);
161         // System.err.println(cg.toString());
162     
163         MemoryAccessMap fam = new SimpleMemoryAccessMap(cg, rtaBuilder.getPointerAnalysis().getHeapModel(), false);
164     
165         final IDemandPointerAnalysis dmp = makeDemandPointerAnalysis(options, cha, scope, cg, fam);
166     
167         final class Helper {
168           void checkPointersInMethod(CGNode node) {
169             // TODO remove this hack
170             if (node.getMethod().getReference().toString().indexOf("clone()Ljava/lang/Object;") != -1) {
171               System.err.println(("SKIPPING " + node));
172               return;
173             }
174             CGNode oldNode = CallGraphMapUtil.mapCGNode(node, cg, oldCG);
175             if (oldNode == null) {
176               return;
177             }
178             System.err.println(("METHOD " + node));
179             IR ir = node.getIR();
180             TypeInference ti = TypeInference.make(ir, false);
181             for (int i = 1; i <= ir.getSymbolTable().getMaxValueNumber(); i++) {
182               TypeAbstraction t = ti.getType(i);
183               if (t != null) {
184                 final HeapModel heapModel = dmp.getHeapModel();
185                 LocalPointerKey pk = (LocalPointerKey) heapModel.getPointerKeyForLocal(node, i);
186                 LocalPointerKey oldPk = (LocalPointerKey) CallGraphMapUtil.mapPointerKey(pk, cg, oldCG, heapModel);
187                 Collection<InstanceKey> p2set = dmp.getPointsTo(pk);
188                 OrdinalSet<InstanceKey> otherP2Set = pa.getPointsToSet(oldPk);
189                 System.err.println(("OLD POINTS-TO " + otherP2Set));
190                 for (InstanceKey key : otherP2Set) {
191                   if (knownBug(key)) {
192                     continue;
193                   }
194                   InstanceKey newKey = CallGraphMapUtil.mapInstKey(key, oldCG, cg, heapModel);
195                   if (!p2set.contains(newKey)) {
196                     System.err.println("BADNESS");
197                     System.err.println(("pointer key " + pk));
198                     System.err.println(("missing " + newKey));
199                     Assertions.UNREACHABLE();
200                   }
201                 }
202               }
203             }
204           }
205     
206         }
207         Helper h = new Helper();
208         for (Iterator<? extends CGNode> nodeIter = cg.iterator(); nodeIter.hasNext();) {
209           CGNode node = nodeIter.next();
210           h.checkPointersInMethod(node);
211         }
212       }
213     
214       private static boolean knownBug(InstanceKey key) {
215         // if (key instanceof MultiNewArrayAllocationSiteKey) {
216         // return true;
217         // }
218         if (key instanceof InstanceKeyWithNode) {
219           CGNode node = ((InstanceKeyWithNode) key).getNode();
220           MethodReference methodRef = node.getMethod().getReference();
221           if (methodRef.toString().equals("< Primordial, Ljava/lang/Object, clone()Ljava/lang/Object; >")) {
222             return true;
223           }
224         }
225         return false;
226       }

228       private static IDemandPointerAnalysis makeDemandPointerAnalysis(AnalysisOptions options, ClassHierarchy cha, AnalysisScope scope,
229           CallGraph cg, MemoryAccessMap fam) {
230         SSAPropagationCallGraphBuilder builder = Util.makeVanillaZeroOneCFABuilder(options, new AnalysisCache(), cha, scope);
231         // return new TestNewGraphPointsTo(cg, builder, fam, cha, warnings);
232         DemandRefinementPointsTo fullDemandPointsTo = DemandRefinementPointsTo.makeWithDefaultFlowGraph(cg, builder, fam, cha, options, new DummyStateMachine.Factory<IFlowLabel>());
233         // fullDemandPointsTo.setOnTheFly(true);
234         // fullDemandPointsTo.setRefineFields(true);
235         return fullDemandPointsTo;
236       }
