11     package com.ibm.wala.ipa.callgraph.pruned;
12     
13     import com.ibm.wala.ipa.callgraph.AnalysisScope;
14     import com.ibm.wala.ipa.callgraph.CGNode;
15     
16     /**
17      * Keeps a given CGNode if it stems from application code
18      * @author Martin Mohr
19      *
20      */
21     public class ApplicationLoaderPolicy implements PruningPolicy {
22       
23       public static final ApplicationLoaderPolicy INSTANCE = new ApplicationLoaderPolicy();
24       
25       private ApplicationLoaderPolicy() {
26         
27       }
28       
29       @Override
30       public boolean check(CGNode n) {
31         return n.getMethod().getDeclaringClass().getClassLoader().getName().equals(AnalysisScope.APPLICATION);
32       }
33       
34     }
