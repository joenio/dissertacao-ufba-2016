26     import com.ibm.wala.ipa.cha.ClassHierarchyException;
27     import com.ibm.wala.shrikeCT.InvalidClassFileException;
28     import com.ibm.wala.types.ClassLoaderReference;
29     import com.ibm.wala.types.TypeName;

32     import com.ibm.wala.util.io.FileProvider;
33     
34     /**
35      * Test stuff with inner classes
36      */
37     public class InnerClassesTest extends WalaTestCase {
38     
39       private static final ClassLoader MY_CLASSLOADER = InnerClassesTest.class.getClassLoader();
40     
41       private static AnalysisScope scope;
42     
43       private static ClassHierarchy cha;
44     
45       public static void main(String[] args) {
46         justThisTest(InnerClassesTest.class);
47       }
48     
49       @BeforeClass
50       public static void beforeClass() throws Exception {
51         scope = AnalysisScopeReader.readJavaScope(TestConstants.WALA_TESTDATA, (new FileProvider()).getFile("J2SEClassHierarchyExclusions.txt"),
52             MY_CLASSLOADER);
53     
54         ClassLoaderFactory factory = new ClassLoaderFactoryImpl(scope.getExclusions());
55     
56         try {
57           cha = ClassHierarchy.make(scope, factory);
58         } catch (ClassHierarchyException e) {
59           throw new Exception();
60         }
61       }
62     
63       @AfterClass
64       public static void afterClass() throws Exception {
65         scope = null;
66         cha = null;
67       }
68     
69       @Test public void test1() throws InvalidClassFileException {
70         TypeReference t = TypeReference.findOrCreate(ClassLoaderReference.Application, TypeName
71             .string2TypeName("Linner/TestStaticInner"));
72         IClass klass = cha.lookupClass(t);
73         assert klass != null;
74         ShrikeClass s = (ShrikeClass)klass;
75         Assert.assertFalse(s.isInnerClass());
76       }
77       
78       @Test public void test2() throws InvalidClassFileException {
79         TypeReference t = TypeReference.findOrCreate(ClassLoaderReference.Application, TypeName
80             .string2TypeName("Linner/TestStaticInner$A"));
81         IClass klass = cha.lookupClass(t);
82         assert klass != null;
83         ShrikeClass s = (ShrikeClass)klass;
84         Assert.assertTrue(s.isInnerClass());
85         Assert.assertTrue(s.isStaticInnerClass());
86       }
87       
88       @Test public void test3() throws InvalidClassFileException {
89         TypeReference t = TypeReference.findOrCreate(ClassLoaderReference.Application, TypeName
90             .string2TypeName("Linner/TestInner$A"));
91         IClass klass = cha.lookupClass(t);
92         assert klass != null;
93         ShrikeClass s = (ShrikeClass)klass;
94         Assert.assertTrue(s.isInnerClass());
95         Assert.assertFalse(s.isStaticInnerClass());
96       }
97     
98     }
