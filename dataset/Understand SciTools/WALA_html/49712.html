11     package com.ibm.wala.cast.ir.ssa;
12     
13     import java.util.Collection;
14     import java.util.Collections;
15     
16     import com.ibm.wala.ssa.SSAInstruction;
17     import com.ibm.wala.ssa.SSAInstructionFactory;
18     import com.ibm.wala.ssa.SymbolTable;
19     import com.ibm.wala.types.FieldReference;
20     import com.ibm.wala.types.TypeReference;
21     import com.ibm.wala.util.debug.Assertions;
22     
23     /**
24      * IR instruction to check whether a field is defined on some object. The field
25      * is represented either by a {@link FieldReference} or by a local value number.  
26      */
27     public class AstIsDefinedInstruction extends SSAInstruction {
28       /**
29        * name of the field.  If non-null, fieldVal should be -1.
30        */
31       private final FieldReference fieldRef;
32     
33       /**
34        * value number holding the field string.  If non-negative,
35        * fieldRef should be null.
36        */
37       private final int fieldVal;
38     
39       /**
40        * the base pointer
41        */
42       private final int rval;
43     
44       /**
45        * gets 1 if the field is defined, 0 otherwise.
46        */
47       private final int lval;
48     
49       /**
50        * This constructor should only be used from {@link SSAInstruction#copyForSSA(SSAInstructionFactory, int[], int[])}
51        */
52       public AstIsDefinedInstruction(int iindex, int lval, int rval, int fieldVal, FieldReference fieldRef) {
53         super(iindex);
54         this.lval = lval;
55         this.rval = rval;
56         this.fieldVal = fieldVal;
57         this.fieldRef = fieldRef;
58       }
59     
60       public AstIsDefinedInstruction(int iindex, int lval, int rval, FieldReference fieldRef) {
61         super(iindex);
62         this.lval = lval;
63         this.rval = rval;
64         this.fieldVal = -1;
65         this.fieldRef = fieldRef;
66       }
67     
68       public AstIsDefinedInstruction(int iindex, int lval, int rval, int fieldVal) {
69         super(iindex);
70         this.lval = lval;
71         this.rval = rval;
72         this.fieldVal = fieldVal;
73         this.fieldRef = null;
74       }
75     
76       public AstIsDefinedInstruction(int iindex, int lval, int rval) {
77         super(iindex);
78         this.lval = lval;
79         this.rval = rval;
80         this.fieldVal = -1;
81         this.fieldRef = null;
82       }
83     
84       @Override
85       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
86         assert fieldVal == -1 || fieldRef == null;
87     
88         return ((AstInstructionFactory) insts).IsDefinedInstruction(iindex, (defs == null) ? lval : defs[0], (uses == null) ? rval : uses[0],
89             (uses == null || fieldVal == -1) ? fieldVal : uses[1], fieldRef);
90       }
91     
92       @Override
93       public String toString(SymbolTable symbolTable) {
94         if (fieldVal == -1 && fieldRef == null) {
95           return getValueString(symbolTable, lval) + " = isDefined(" + getValueString(symbolTable, rval) + ")";
96         } else if (fieldVal == -1) {
97           return getValueString(symbolTable, lval) + " = isDefined(" + getValueString(symbolTable, rval) + "," + fieldRef.getName()
98               + ")";
99         } else if (fieldRef == null) {
100           return getValueString(symbolTable, lval) + " = isDefined(" + getValueString(symbolTable, rval) + ","
101               + getValueString(symbolTable, fieldVal) + ")";
102         } else {
103           Assertions.UNREACHABLE();
104           return null;
105         }
106       }
107     
108       @Override
109       public void visit(IVisitor v) {
110         ((AstInstructionVisitor) v).visitIsDefined(this);
111       }
112     
113       @Override
114       public Collection<TypeReference> getExceptionTypes() {
115         return Collections.emptySet();
116       }
117     
118       @Override
119       public boolean hasDef() {
120         return true;
121       }
122     
123       @Override
124       public int getDef() {
125         return lval;
126       }
127     
128       @Override
129       public int getDef(int i) {
130         assert i == 0;
131     
132         return lval;
133       }
134     
135       @Override
136       public int getNumberOfDefs() {
137         return 1;
138       }
139     
140       @Override
141       public int getNumberOfUses() {
142         return (fieldVal == -1) ? 1 : 2;
143       }
144     
145       @Override
146       public int getUse(int j) {
147         if (j == 0) {
148           return rval;
149         } else if (j == 1 && fieldVal != -1) {
150           return fieldVal;
151         } else {
152           Assertions.UNREACHABLE();
153           return -1;
154         }
155       }
156     
157       @Override
158       public boolean isFallThrough() {
159         return true;
160       }
161     
162       @Override
163       public int hashCode() {
164         return 3077 * fieldVal * rval;
165       }
166     
167       public FieldReference getFieldRef() {
168         return fieldRef;
169       }
170     }
