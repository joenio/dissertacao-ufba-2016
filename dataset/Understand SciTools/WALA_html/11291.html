124       public Pair<JSCallGraph, PointerAnalysis<ObjectVertex>> getFieldBasedCallGraph(String scriptName) throws CancelException {
125         Set<Entrypoint> eps= HashSetFactory.make();
126         eps.add(JSCallGraphUtil.makeScriptRoots(getClassHierarchy()).make(scriptName));
127         eps.add(JSCallGraphUtil.makeScriptRoots(getClassHierarchy()).make("Lprologue.js"));
128         return getFieldBasedCallGraph(eps);
129       }
130       
131       private String getScriptName(AstMethod m) {
132         
133         // we want the original including file, since that will be the "script"
134         Position p = m.getSourcePosition();
135         while (p instanceof IncludedPosition) {
136           p = ((IncludedPosition)p).getIncludePosition();
137         }
138         
139         String fileName = p.getURL().getFile();
140         return fileName.substring(fileName.lastIndexOf('/') + 1);    
141       }
142       
143       protected Pair<JSCallGraph, PointerAnalysis<ObjectVertex>> getFieldBasedCallGraph(Iterable<Entrypoint> roots) throws CancelException {
144         final Set<String> scripts = HashSetFactory.make();
145         for(Entrypoint e : roots) {
146           String scriptName = getScriptName(((AstMethod)e.getMethod()));
147           scripts.add(scriptName);
148         }
149      
150         final Function<IMethod, Boolean> filter = new Function<IMethod, Boolean>() {
151           @Override
152           public Boolean apply(IMethod object) {
153             if (object instanceof AstMethod) {
154                return scripts.contains(getScriptName((AstMethod)object));
155             } else {
156               return true;
157             }
158           }
159         };
160     
161         AnalysisOptions options = getDefaultOptions(roots);
162         if (builderType.equals(BuilderType.OPTIMISTIC)) {
163           ((JSAnalysisOptions)options).setHandleCallApply(false);
164         }
165     
166         FieldBasedCallGraphBuilder builder = 
167             builderType.equals(BuilderType.PESSIMISTIC)? 
168                 new PessimisticCallGraphBuilder(getClassHierarchy(), options, makeDefaultCache(), false) {
169                   @Override
170                   protected FlowGraph flowGraphFactory() {
171                     FlowGraphBuilder b = new FilteredFlowGraphBuilder(cha, cache, true, filter);
172                     return b.buildFlowGraph();
173                   }
174                   @Override
175                   protected boolean filterFunction(IMethod function) {
176                      return super.filterFunction(function) && filter.apply(function);
177                   }     
178                 }      
179                 : new OptimisticCallgraphBuilder(getClassHierarchy(), options, makeDefaultCache(), true) {
180                   @Override
181                   protected FlowGraph flowGraphFactory() {
182                     FlowGraphBuilder b = new FilteredFlowGraphBuilder(cha, cache, true, filter);
183                     return b.buildFlowGraph();
184                   }  
185                 };
186         
187         return builder.buildCallGraph(roots, new NullProgressMonitor());
188       }
