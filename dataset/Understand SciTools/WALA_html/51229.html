11     package com.ibm.wala.shrikeCT;
12     
13     /**
14      * This class builds serializable ConstantValue attributes. These attributes are associated with final fields.
15      * 
16      * After constructing a ConstantValueWriter, you must call setValueCPIndex.
17      */
18     public final class ConstantValueWriter extends ClassWriter.Element {
19       final private int attrID;
20     
21       private int index = -1;
22     
23       final private ClassWriter w;
24     
25       /**
26        * Build an empty writer.
27        * 
28        * @throws IllegalArgumentException if w is null
29        */
30       public ConstantValueWriter(ClassWriter w) {
31         if (w == null) {
32           throw new IllegalArgumentException("w is null");
33         }
34         this.w = w;
35         attrID = w.addCPUtf8("ConstantValue");
36       }
37     
38       /**
39        * Build an writer for a 'long' constant value.
40        */
41       public ConstantValueWriter(ClassWriter w, long v) {
42         this(w);
43         setLong(v);
44       }
45     
46       /**
47        * Build an writer for an 'int' constant value.
48        */
49       public ConstantValueWriter(ClassWriter w, int v) {
50         this(w);
51         setInt(v);
52       }
53     
54       /**
55        * Build an writer for a 'float' constant value.
56        */
57       public ConstantValueWriter(ClassWriter w, float v) {
58         this(w);
59         setFloat(v);
60       }
61     
62       /**
63        * Build an writer for a 'double' constant value.
64        */
65       public ConstantValueWriter(ClassWriter w, double v) {
66         this(w);
67         setDouble(v);
68       }
69     
70       /**
71        * Build an writer for a 'String' constant value.
72        */
73       public ConstantValueWriter(ClassWriter w, String v) {
74         this(w);
75         setString(v);
76       }
77     
78       private void verify() {
79         if (index < 0) {
80           throw new IllegalArgumentException("The value's constant pool index is not set");
81         }
82       }
83     
84       @Override
85       public int getSize() {
86         verify();
87         return 8;
88       }
89     
90       @Override
91       public int copyInto(byte[] buf, int offset) throws IllegalArgumentException {
92         verify();
93         ClassWriter.setUShort(buf, offset, attrID);
94         ClassWriter.setInt(buf, offset + 2, 2);
95         ClassWriter.setUShort(buf, offset + 6, index);
96         return offset + 8;
97       }
98     
99       /**
100        * Set the constant value to a long.
101        */
102       public void setLong(long value) {
103         this.index = w.addCPLong(value);
104       }
105     
106       /**
107        * Set the constant value to a double.
108        */
109       public void setDouble(double value) {
110         this.index = w.addCPDouble(value);
111       }
112     
113       /**
114        * Set the constant value to an int.
115        */
116       public void setInt(int value) {
117         this.index = w.addCPInt(value);
118       }
119     
120       /**
121        * Set the constant value to a float.
122        */
123       public void setFloat(float value) {
124         this.index = w.addCPFloat(value);
125       }
126     
127       /**
128        * Set the constant value to a String.
129        */
130       public void setString(String value) {
131         if (value == null) {
132           throw new IllegalArgumentException("null value");
133         }
134         this.index = w.addCPString(value);
135       }
136     
137       /**
138        * Set the index of the constant pool item holding the constant value.
139        */
140       public void setValueCPIndex(int index) throws IllegalArgumentException {
141         if (index < 1 || index > 0xFFFF) {
142           throw new IllegalArgumentException("Invalid CP index: " + index);
143         }
144     
145         this.index = index;
146       }
147     }
