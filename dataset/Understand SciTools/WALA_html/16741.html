11     package com.ibm.wala.dataflow.graph;
12     
13     import com.ibm.wala.fixpoint.BitVectorVariable;
14     import com.ibm.wala.fixpoint.UnaryOperator;
15     import com.ibm.wala.util.intset.BitVector;
16     
17     
18     /**
19      * Operator OUT = IN U v
20      */
21     public class BitVectorOr extends UnaryOperator<BitVectorVariable> {
22       
23       private final BitVector v;
24       public BitVectorOr(BitVector v) {
25         if (v == null) {
26           throw new IllegalArgumentException("null v");
27         }
28         this.v = v;
29       }
30       
31     
32       @Override
33       public byte evaluate(BitVectorVariable lhs, BitVectorVariable rhs) throws IllegalArgumentException {
34         if (lhs == null) {
35           throw new IllegalArgumentException("lhs == null");
36         }
37         if (rhs == null) {
38           throw new IllegalArgumentException("rhs == null");
39         }
40         BitVectorVariable U = new BitVectorVariable();
41         U.copyState(lhs);
42         U.addAll(rhs);
43         U.addAll(v);
44         if (!lhs.sameValue(U)) {
45           lhs.copyState(U);
46           return CHANGED;
47         } else {
48           return NOT_CHANGED;
49         }
50       }
51     
52       @Override
53       public String toString() {
54         return "U " + v;
55       }
56     
57       @Override
58       public int hashCode() {
59         return 9901 * v.hashCode();
60       }
61     
62       @Override
63       public boolean equals(Object o) {
64         if (o instanceof BitVectorOr) {
65           BitVectorOr other = (BitVectorOr)o;
66           return v.equals(other.v);
67         } else {
68           return false;
69         }
70       }
71     }
