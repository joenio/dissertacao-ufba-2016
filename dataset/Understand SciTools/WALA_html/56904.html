11     package com.ibm.wala.util.graph.traverse;
12     
13     import java.util.Iterator;
14     import java.util.NoSuchElementException;
15     import java.util.Set;
16     
17     import com.ibm.wala.util.collections.HashSetFactory;
18     import com.ibm.wala.util.collections.ReverseIterator;
19     import com.ibm.wala.util.graph.Graph;
20     import com.ibm.wala.util.graph.impl.GraphInverter;
21     
22     /**
23      * This class computes strongly connected components for a Graph (or a subset of
24      * it). It does not store the SCCs in any lookaside structure, but rather simply
25      * generates an enumeration of them. See Cormen, Leiserson, Rivest Ch. 23 Sec. 5
26      */
27     public class SCCIterator<T> implements Iterator<Set<T>> {
28       /**
29        * The second DFS (the reverse one) needed while computing SCCs
30        */
31       final private DFSFinishTimeIterator<T> rev;
32     
33       /**
34        * Construct an enumeration across the SCCs of a given graph.
35        * 
36        * @param G
37        *          The graph over which to construct SCCs
38        * @throws NullPointerException  if G is null
39        */
40       public SCCIterator(Graph<T> G) throws NullPointerException {
41         this(G, G == null ? null : G.iterator());
42       }
43     
44       /**
45        * Construct an enumeration of the SCCs of the subset of a given graph
46        * determined by starting at a given set of nodes.
47        */
48       public SCCIterator(Graph<T> G, Iterator<T> nodes) {
49         if (G == null) {
50           throw new IllegalArgumentException("G cannot be null");
51         }
52         Iterator<T> reverseFinishTime = ReverseIterator.reverse(DFS.iterateFinishTime(G, nodes));
53     
54         rev = DFS.iterateFinishTime(GraphInverter.invert(G), reverseFinishTime);
55       }
56     
57       /**
58        * Determine whether there are any more SCCs remaining in this enumeration.
59        */
60       @Override
61       public boolean hasNext() {
62         return rev.hasNext();
63       }
64     
65       /**
66        * Find the next SCC in this enumeration
67        */
68       @Override
69       public Set<T> next() throws NoSuchElementException {
70         Set<T> currentSCC = HashSetFactory.make();
71     
72         T v = rev.next();
73         currentSCC.add(v);
74     
75         while (rev.hasNext() && !rev.isEmpty()) {
76           currentSCC.add(rev.next());
77         }
78     
79         return currentSCC;
80       }
81     
82     
83       @Override
84       public void remove() throws UnsupportedOperationException {
85         throw new UnsupportedOperationException();
86       }
87     
88     }
