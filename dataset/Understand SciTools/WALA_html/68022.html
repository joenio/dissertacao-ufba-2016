11     package com.ibm.wala.util.graph.impl;
12     
13     import java.util.Iterator;
14     
15     import com.ibm.wala.util.debug.Assertions;
16     import com.ibm.wala.util.graph.INodeWithNumber;
17     import com.ibm.wala.util.graph.NumberedNodeManager;
18     import com.ibm.wala.util.intset.IntSet;
19     
20     /**
21      * Basic implementation of a numbered graph -- this implementation relies on nodes that carry numbers and edges.
22      * 
23      * The management of node numbers is a bit fragile, but designed this way for efficiency. Use this class with care.
24      */
25     public class DelegatingNumberedNodeManager<T extends INodeWithNumber> implements NumberedNodeManager<T> {
26     
27       private final double BUFFER_FACTOR = 1.5;
28     
29       private INodeWithNumber[] nodes = new INodeWithNumber[20];
30     
31       private int maxNumber = -1;
32     
33       private int numberOfNodes = 0;
34     
35       /*
36        * @see com.ibm.wala.util.graph.NumberedGraph#getNumber(com.ibm.wala.util.graph.Node)
37        */
38       @Override
39       public int getNumber(T N) {
40         if (N == null) {
41           throw new IllegalArgumentException("N is null");
42         }
43         INodeWithNumber n = N;
44         return n.getGraphNodeId();
45       }
46     
47       @Override
48       @SuppressWarnings("unchecked")
49       public T getNode(int number) {
50         try {
51           return (T) nodes[number];
52         } catch (ArrayIndexOutOfBoundsException e) {
53           throw new IllegalArgumentException("Invalid number " + number);
54         }
55       }
56     
57       /*
58        * @see com.ibm.wala.util.graph.NumberedGraph#getMaxNumber()
59        */
60       @Override
61       public int getMaxNumber() {
62         return maxNumber;
63       }
64     
65       /*
66        * @see com.ibm.wala.util.graph.Graph#iterateNodes()
67        */
68       @Override
69       public Iterator<T> iterator() {
70         final INodeWithNumber[] arr = nodes;
71         return new Iterator<T>() {
72           int next = -1;
73           {
74             advance();
75           }
76     
77           void advance() {
78             for (int i = next + 1; i < arr.length; i++) {
79               if (arr[i] != null) {
80                 next = i;
81                 return;
82               }
83             }
84             next = -1;
85           }
86     
87           @Override
88           public boolean hasNext() {
89             return next != -1;
90           }
91     
92           @Override
93           @SuppressWarnings("unchecked")
94           public T next() {
95             if (hasNext()) {
96               int r = next;
97               advance();
98               return (T) arr[r];
99             } else {
100               return null;
101             }
102           }
103     
104           @Override
105           public void remove() {
106             Assertions.UNREACHABLE();
107           }
108         };
109       }
110     
111       /*
112        * @see com.ibm.wala.util.graph.Graph#getNumberOfNodes()
113        */
114       @Override
115       public int getNumberOfNodes() {
116         return numberOfNodes;
117       }
118     
119       /**
120        * If N.getNumber() == -1, then set N.number and insert this node in the graph. Use with extreme care.
121        * 
122        * @see com.ibm.wala.util.graph.NodeManager#addNode(java.lang.Object)
123        * @throws IllegalArgumentException if n is null
124        */
125       @Override
126       public void addNode(T n) {
127         if (n == null) {
128           throw new IllegalArgumentException("n is null");
129         }
130         INodeWithNumber N = n;
131         int number = N.getGraphNodeId();
132         if (number == -1) {
133           maxNumber++;
134           N.setGraphNodeId(maxNumber);
135           number = maxNumber;
136         } else {
137           if (number > maxNumber) {
138             maxNumber = number;
139           }
140         }
141         ensureCapacity(number);
142         if (nodes[number] != null && nodes[number] != N) {
143           Assertions.UNREACHABLE("number: " + number + " N: " + N + " nodes[number]: " + nodes[number]);
144         }
145         nodes[number] = N;
146         numberOfNodes++;
147       }
148     
149       /**
150        * @param number
151        */
152       private void ensureCapacity(int number) {
153         if (nodes.length < number + 1) {
154           int newLength = (int) ((number + 1) * BUFFER_FACTOR);
155           INodeWithNumber[] old = nodes;
156           nodes = new INodeWithNumber[newLength];
157           System.arraycopy(old, 0, nodes, 0, old.length);
158         }
159       }
160     
161       /*
162        * @see com.ibm.wala.util.graph.NodeManager#remove(com.ibm.wala.util.graph.Node)
163        */
164       @Override
165       public void removeNode(T n) {
166         if (n == null) {
167           throw new IllegalArgumentException("n is null");
168         }
169         INodeWithNumber N = n;
170         int number = N.getGraphNodeId();
171         if (number == -1) {
172           throw new IllegalArgumentException("Cannot remove node, not in graph");
173         }
174         if (nodes[number] != null) {
175           nodes[number] = null;
176           numberOfNodes--;
177         }
178       }
179     
180       @Override
181       public String toString() {
182         StringBuffer result = new StringBuffer("Nodes:\n");
183         for (int i = 0; i <= maxNumber; i++) {
184           result.append(i).append(" ");
185           if (nodes[i] != null) {
186             result.append(nodes[i].toString());
187           }
188           result.append("\n");
189         }
190         return result.toString();
191       }
192     
193       /*
194        * @see com.ibm.wala.util.graph.NodeManager#containsNode(com.ibm.wala.util.graph.Node)
195        */
196       @Override
197       public boolean containsNode(T n) {
198         if (n == null) {
199           throw new IllegalArgumentException("n is null");
200         }
201         INodeWithNumber N = n;
202         int number = N.getGraphNodeId();
203         if (number == -1) {
204           return false;
205         }
206         if (number >= nodes.length) {
207           throw new IllegalArgumentException(
208               "node already has a graph node id, but is not registered there in this graph (number too big)\n"
209                   + "this graph implementation is fragile and won't support this kind of test\n" + n.getClass() + " : " + n);
210         }
211         if (nodes[number] != N) {
212           throw new IllegalArgumentException("node already has a graph node id, but is not registered there in this graph\n"
213               + "this graph implementation is fragile and won't support this kind of test\n" + n.getClass() + " : " + n);
214         }
215         return true;
216         // return (nodes[number] == N);
217       }
218     
219       /*
220        * @see com.ibm.wala.util.graph.NumberedNodeManager#iterateNodes(com.ibm.wala.util.intset.IntSet)
221        */
222       @Override
223       public Iterator<T> iterateNodes(IntSet s) {
224         return new NumberedNodeIterator<T>(s, this);
225       }
226     
227     }
