11     package com.ibm.wala.demandpa.flowgraph;
12     
13     import java.util.Iterator;
14     import java.util.Set;
15     
16     import com.ibm.wala.classLoader.CallSiteReference;
17     import com.ibm.wala.classLoader.IField;
18     import com.ibm.wala.demandpa.flowgraph.IFlowLabel.IFlowLabelVisitor;
19     import com.ibm.wala.ipa.callgraph.CGNode;
20     import com.ibm.wala.ipa.callgraph.propagation.LocalPointerKey;
21     import com.ibm.wala.ipa.callgraph.propagation.PointerKey;
22     import com.ibm.wala.ipa.callgraph.propagation.StaticFieldKey;
23     import com.ibm.wala.ipa.callgraph.propagation.cfa.CallerSiteContext;
24     import com.ibm.wala.ssa.SSAAbstractInvokeInstruction;
25     import com.ibm.wala.util.graph.labeled.LabeledGraph;
26     
27     public interface IFlowGraph extends LabeledGraph<Object, IFlowLabel> {
28     
29       /**
30        * Apply a visitor to the successors of some node.
31        */
32       public abstract void visitSuccs(Object node, IFlowLabelVisitor v);
33     
34       /**
35        * Apply a visitor to the predecessors of some node.
36        */
37       public abstract void visitPreds(Object node, IFlowLabelVisitor v);
38     
39       /**
40        * add representation of flow for a node, if not already present
41        * 
42        * @throws IllegalArgumentException if node == null
43        */
44       public abstract void addSubgraphForNode(CGNode node) throws IllegalArgumentException;
45     
46       public abstract boolean hasSubgraphForNode(CGNode node);
47     
48       /**
49        * @param pk
50        * @return <code>true</code> iff <code>pk</code> is a formal parameter
51        */
52       public abstract boolean isParam(LocalPointerKey pk);
53     
54       /**
55        * @param pk
56        * @return the {@link SSAInvokeInstruction}s passing some pointer as a parameter
57        */
58       public abstract Iterator<SSAAbstractInvokeInstruction> getInstrsPassingParam(LocalPointerKey pk);
59     
60       /**
61        * get the {@link SSAInvokeInstruction} whose return value is assigned to a pointer key.
62        * 
63        * @return the instruction, or <code>null</code> if no return value is assigned to pk
64        */
65       public abstract SSAAbstractInvokeInstruction getInstrReturningTo(LocalPointerKey pk);
66     
67       /**
68        * @param sfk the static field
69        * @return all the variables whose values are written to sfk
70        * @throws IllegalArgumentException if sfk == null
71        */
72       public abstract Iterator<? extends Object> getWritesToStaticField(StaticFieldKey sfk) throws IllegalArgumentException;
73     
74       /**
75        * @param sfk the static field
76        * @return all the variables that get the value of sfk
77        * @throws IllegalArgumentException if sfk == null
78        */
79       public abstract Iterator<? extends Object> getReadsOfStaticField(StaticFieldKey sfk) throws IllegalArgumentException;
80     
81       public abstract Iterator<PointerKey> getWritesToInstanceField(PointerKey pk, IField f);
82     
83       public abstract Iterator<PointerKey> getReadsOfInstanceField(PointerKey pk, IField f);
84     
85       /**
86        * 
87        * @param formalPk a {@link PointerKey} representing either a formal parameter or return value
88        * @return the {@link CallerSiteContext}s representing pointer callers of <code>formalPk</code>'s method
89        */
90       public abstract Set<CallerSiteContext> getPotentialCallers(PointerKey formalPk);
91     
92       /**
93        * get the callees that should be considered at a particular call site
94        * 
95        * @param caller the caller
96        * @param site the call site
97        * @param actualPk a {@link LocalPointerKey} corresponding to the actual parameter or return value of interest. This may be used
98        *          to filter out certain callees.
99        * @return the callees of interest
100        */
101       public abstract Set<CGNode> getPossibleTargets(CGNode caller, CallSiteReference site, LocalPointerKey actualPk);
102     
103     }
