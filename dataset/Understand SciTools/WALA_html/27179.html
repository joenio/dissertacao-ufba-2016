11     package com.ibm.wala.util.graph.impl;
12     
13     import com.ibm.wala.util.graph.AbstractNumberedGraph;
14     import com.ibm.wala.util.graph.INodeWithNumberedEdges;
15     import com.ibm.wala.util.graph.NumberedEdgeManager;
16     import com.ibm.wala.util.graph.NumberedNodeManager;
17     
18     /**
19      * Basic functionality for a graph that delegates node and edge management, and
20      * tracks node numbers
21      */
22     public class DelegatingNumberedGraph<T extends INodeWithNumberedEdges> extends AbstractNumberedGraph<T> {
23     
24       final private DelegatingNumberedNodeManager<T> nodeManager = new DelegatingNumberedNodeManager<T>();
25     
26       final private DelegatingNumberedEdgeManager<T> edgeManager = new DelegatingNumberedEdgeManager<T>(nodeManager);
27     
28       /*
29        * @see com.ibm.wala.util.graph.AbstractGraph#getNodeManager()
30        */
31       @Override
32       protected NumberedNodeManager<T> getNodeManager() {
33         return nodeManager;
34       }
35     
36       /*
37        * @see com.ibm.wala.util.graph.AbstractGraph#getEdgeManager()
38        */
39       @Override
40       protected NumberedEdgeManager<T> getEdgeManager() {
41         return edgeManager;
42       }
43     }
