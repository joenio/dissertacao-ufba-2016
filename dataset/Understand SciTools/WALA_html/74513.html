39     package com.ibm.wala.util.collections;
40     
41     import java.util.Arrays;
42     import java.util.EmptyStackException;
43     import java.util.Iterator;
44     
45     /**
46      * An immutable stack of objects. The {@link #push(Object)} and {@link #pop()}
47      * operations create new stacks.
48      * 
49      * @param <T>
50      */
51     public class ImmutableStack<T> implements Iterable<T> {
52     
53     	private static final ImmutableStack<Object> EMPTY = new ImmutableStack<Object>(
54     			new Object[0]);
55     
56     	private static final int MAX_SIZE = Integer.MAX_VALUE;
57     
58     	public static int getMaxSize() {
59     		return MAX_SIZE;
60     	}
61     
62     	@SuppressWarnings("unchecked")
63     	public static final <T> ImmutableStack<T> emptyStack() {
64     		return (ImmutableStack<T>) EMPTY;
65     	}
66     
67     	final private T[] entries;
68     
69     	private final int cachedHashCode;
70     
71     	protected ImmutableStack(T[] entries) {
72     		this.entries = entries;
73     		this.cachedHashCode = Arrays.hashCode(entries);
74     	}
75     
76     	@SuppressWarnings("rawtypes")
77     	@Override
78     	public boolean equals(Object o) {
79     		if (this == o)
80     			return true;
81     		if (o != null && o instanceof ImmutableStack) {
82     			ImmutableStack other = (ImmutableStack) o;
83     			return Arrays.equals(entries, other.entries);
84     		}
85     		return false;
86     	}
87     
88     	@Override
89     	public int hashCode() {
90     		return cachedHashCode;
91     		// return Util.hashArray(this.entries);
92     	}
93     
94     	@SuppressWarnings("unused")
95     	public ImmutableStack<T> push(T entry) {
96     		if (entry == null) {
97     			throw new IllegalArgumentException("null entry");
98     		}
99     		if (MAX_SIZE == 0) {
100     			return emptyStack();
101     		}
102     		int size = entries.length + 1;
103     		T[] tmpEntries = null;
104     		if (size <= MAX_SIZE) {
105     			tmpEntries = makeInternalArray(size);
106     			System.arraycopy(entries, 0, tmpEntries, 0, entries.length);
107     			tmpEntries[size - 1] = entry;
108     		} else {
109     			tmpEntries = makeInternalArray(MAX_SIZE);
110     			System.arraycopy(entries, 1, tmpEntries, 0, entries.length - 1);
111     			tmpEntries[MAX_SIZE - 1] = entry;
112     
113     		}
114     		return makeStack(tmpEntries);
115     	}
116     
117     	@SuppressWarnings("unchecked")
118     	protected T[] makeInternalArray(int size) {
119     		return (T[]) new Object[size];
120     	}
121     
122     	protected ImmutableStack<T> makeStack(T[] tmpEntries) {
123     		return new ImmutableStack<T>(tmpEntries);
124     	}
125     
126     	/**
127     	 * @return the element on the top of the stack
128     	 * @throws EmptyStackException
129     	 *             if stack is empty
130     	 */
131     	public T peek() {
132     		if (entries.length == 0) {
133     			throw new EmptyStackException();
134     		}
135     		return entries[entries.length - 1];
136     	}
137     
138     	/**
139     	 * @throws EmptyStackException
140     	 *             if stack is empty
141     	 */
142     	public ImmutableStack<T> pop() {
143     		if (entries.length == 0) {
144     			throw new EmptyStackException();
145     		}
146     		int size = entries.length - 1;
147     		T[] tmpEntries = makeInternalArray(size);
148     		System.arraycopy(entries, 0, tmpEntries, 0, size);
149     		return makeStack(tmpEntries);
150     	}
151     
152     	public boolean isEmpty() {
153     		return entries.length == 0;
154     	}
155     
156     	public int size() {
157     		return entries.length;
158     	}
159     
160     	public T get(int i) {
161     		try {
162     			return entries[i];
163     		} catch (ArrayIndexOutOfBoundsException e) {
164     			throw new IllegalArgumentException("invalid i: " + i);
165     		}
166     	}
167     
168     	@Override
169     	public String toString() {
170     		String objArrayToString = Arrays.toString(entries);
171     		assert entries.length <= MAX_SIZE : objArrayToString;
172     		return objArrayToString;
173     	}
174     
175     	public boolean contains(T entry) {
176     		if (entry == null) {
177     			return false;
178     		}
179     		for (int i = 0; i < entries.length; i++) {
180     			if (entries[i] != null && entries[i].equals(entry))
181     				return true;
182     		}
183     		return false;
184     	}
185     
186     	/**
187     	 * @return <code>true</code> iff other.size() = k, k <= this.size(), and the
188     	 *         top k elements of this equal other
189     	 * @throws IllegalArgumentException
190     	 *             if other == null
191     	 */
192     	public boolean topMatches(ImmutableStack<T> other)
193     			throws IllegalArgumentException {
194     		if (other == null) {
195     			throw new IllegalArgumentException("other == null");
196     		}
197     		if (other.size() > size()) {
198     			return false;
199     		}
200     		for (int i = other.size() - 1, j = this.size() - 1; i >= 0; i--, j--) {
201     			if (!other.get(i).equals(get(j)))
202     				return false;
203     		}
204     		return true;
205     	}
206     
207     	@SuppressWarnings("unchecked")
208     	public ImmutableStack<T> reverse() {
209     		T[] tmpEntries = (T[]) new Object[entries.length];
210     		for (int i = entries.length - 1, j = 0; i >= 0; i--, j++) {
211     			tmpEntries[j] = entries[i];
212     		}
213     		return new ImmutableStack<T>(tmpEntries);
214     	}
215     
216     	@SuppressWarnings("unchecked")
217     	public ImmutableStack<T> popAll(ImmutableStack<T> other) {
218     		if (!topMatches(other)) {
219     			throw new IllegalArgumentException("top does not match");
220     		}
221     		int size = entries.length - other.entries.length;
222     		T[] tmpEntries = (T[]) new Object[size];
223     		System.arraycopy(entries, 0, tmpEntries, 0, size);
224     		return new ImmutableStack<T>(tmpEntries);
225     	}
226     
227     	@SuppressWarnings("unchecked")
228     	public ImmutableStack<T> pushAll(ImmutableStack<T> other) {
229     		if (other == null) {
230     			throw new IllegalArgumentException("null other");
231     		}
232     		int size = entries.length + other.entries.length;
233     		T[] tmpEntries = null;
234     		if (size <= MAX_SIZE) {
235     			tmpEntries = (T[]) new Object[size];
236     			System.arraycopy(entries, 0, tmpEntries, 0, entries.length);
237     			System.arraycopy(other.entries, 0, tmpEntries, entries.length,
238     					other.entries.length);
239     		} else {
240     			tmpEntries = (T[]) new Object[MAX_SIZE];
241     			// other has size at most MAX_SIZE
242     			// must keep all in other
243     			// top MAX_SIZE - other.size from this
244     			int numFromThis = MAX_SIZE - other.entries.length;
245     			System.arraycopy(entries, entries.length - numFromThis, tmpEntries,
246     					0, numFromThis);
247     			System.arraycopy(other.entries, 0, tmpEntries, numFromThis,
248     					other.entries.length);
249     		}
250     		return new ImmutableStack<T>(tmpEntries);
251     	}
252     
253     	@Override
254       public Iterator<T> iterator() {
255     		if (entries.length == 0) {
256     			return EmptyIterator.instance();
257     		}
258     		return new ArrayIterator<T>(entries);
259     	}
260     
261     	/**
262     	 * return a new stack with the top replaced with t
263     	 * 
264     	 * @throws EmptyStackException
265     	 */
266     	public ImmutableStack<T> replaceTop(T t) {
267     		if (isEmpty()) {
268     			throw new EmptyStackException();
269     		}
270     		int size = entries.length;
271     		T[] tmpEntries = makeInternalArray(size);
272     		System.arraycopy(entries, 0, tmpEntries, 0, entries.length - 1);
273     		tmpEntries[size - 1] = t;
274     		return makeStack(tmpEntries);
275     	}
276     }
