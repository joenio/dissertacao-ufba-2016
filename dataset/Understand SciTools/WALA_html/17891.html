11     package com.ibm.wala.classLoader;
12     
13     import java.io.File;
14     import java.io.InputStreamReader;
15     import java.io.Reader;
16     import java.net.MalformedURLException;
17     import java.net.URL;
18     
19     import com.ibm.wala.util.io.FileSuffixes;
20     
21     /**
22      * A {@link Module} which is a wrapper around a source file
23      */
24     public class SourceFileModule extends FileModule implements Module, ModuleEntry, SourceModule {
25     
26       private final String fileName;
27       public SourceFileModule(File f, String fileName, Module container) {
28         super(f, container);
29         this.fileName = fileName;
30       }
31     
32       public SourceFileModule(File f, SourceFileModule clonedFrom) {
33         super(f, clonedFrom.getContainer());
34         this.fileName = clonedFrom.fileName;
35       }
36     
37       @Override
38       public String toString() {
39         return "SourceFileModule:" + getFile().toString();
40       }
41     
42       /*
43        * @see com.ibm.wala.classLoader.ModuleEntry#isClassFile()
44        */
45       @Override
46       public boolean isClassFile() {
47         return false;
48       }
49     
50       /*
51        * @see com.ibm.wala.classLoader.ModuleEntry#getClassName()
52        */
53       @Override
54       public String getClassName() {
55         return FileSuffixes.stripSuffix(fileName).replace(File.separator.charAt(0), '/');
56       }
57     
58       /*
59        * @see com.ibm.wala.classLoader.ModuleEntry#isSourceFile()
60        */
61       @Override
62       public boolean isSourceFile() {
63         return true;
64       }
65     
66       @Override
67       public Reader getInputReader() {
68         return new InputStreamReader(getInputStream());
69       }
70      
71       @Override
72       public URL getURL() {
73         try {
74           return getFile().toURI().toURL();
75         } catch (MalformedURLException e) {
76           throw new Error("error making URL for " + getFile());
77         }
78       }
79     }
