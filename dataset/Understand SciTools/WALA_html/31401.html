98       /**
99        * @param s a stream containing a table in text format, whitespace delimited
100        * @throws IOException
101        * @throws IllegalArgumentException if s is null
102        */
103       public static StringTable readFromStream(InputStream s, Character commentToken, Character delimiter) throws IOException {
104         if (s == null) {
105           throw new IllegalArgumentException("s is null");
106         }
107         StringTable result = new StringTable();
108     
109         LineNumberReader reader = new LineNumberReader(new InputStreamReader(s));
110     
111         // LineNumberReader reader = new LineNumberReader(new
112         // InputStreamReader(new FileInputStream(f)));
113         String line = readNextNonCommentLine(reader, commentToken);
114         if (line == null) {
115           throw new IOException("first line expected to be column headings");
116         }
117         result.populateColumnHeadings(line, delimiter);
118     
119         line = readNextNonCommentLine(reader, commentToken);
120         int row = 0;
121         while (line != null) {
122           result.populateRow(row, line, delimiter);
123           line = readNextNonCommentLine(reader, commentToken);
124           row++;
125         }
126     
127         return result;
128       }
129     
130       public static String readNextNonCommentLine(LineNumberReader reader, Character commentToken) throws IOException {
131         if (reader == null) {
132           throw new IllegalArgumentException("reader is null");
133         }
134         String line = reader.readLine();
135         while (line != null && isCommented(line, commentToken)) {
136           line = reader.readLine();
137         }
138         return line;
139       }
140     
141       private static boolean isCommented(String line, Character commentToken) {
142         if (line.length() == 0) {
143           return true;
144         }
145         if (commentToken == null) {
146           return false;
147         }
148         return line.charAt(0) == commentToken;
149       }
150     
151       private void populateRow(int row, String line, Character delimiter) {
152         StringTokenizer st = delimiter == null ? new StringTokenizer(line) : new StringTokenizer(line, delimiter.toString());
153         int nColumns = st.countTokens();
154         Assertions.productionAssertion(nColumns == getNumberOfColumns(), "expected " + getNumberOfColumns() + " got " + nColumns
155             + " row " + row + " " + line.length() + " " + line);
156         SimpleVector<String> r = new SimpleVector<String>();
157         rows.add(row, r);
158         for (int i = 0; i < nColumns; i++) {
159           r.set(i, (String) st.nextElement());
160         }
161       }
162     
163       /**
164        * @param line a whitespace-delimited string of column names
165        */
166       private void populateColumnHeadings(String line, Character delimiter) {
167         StringTokenizer st = delimiter == null ? new StringTokenizer(line) : new StringTokenizer(line, delimiter.toString());
168         int nColumns = st.countTokens();
169         for (int i = 0; i < nColumns; i++) {
170           columnHeadings.set(i, (String) st.nextElement());
171         }
172       }
173     
174       @Override
175       public Object clone() throws CloneNotSupportedException {
176         StringTable result = new StringTable(this);
177         for (int i = 0; i < getNumberOfRows(); i++) {
178           result.addRow(row2Map(i));
179         }
180         return result;
181       }
