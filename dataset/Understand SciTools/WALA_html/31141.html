34     public class JSTypeInference extends AstTypeInference {
35     
36       public JSTypeInference(IR ir, IClassHierarchy cha) {
37         super(ir, cha, new PointType(cha.lookupClass(JavaScriptTypes.Boolean)), true);
38       }
39     
40       @Override
41       protected void initialize() {
42         class JSTypeOperatorFactory extends AstTypeOperatorFactory implements com.ibm.wala.cast.js.ssa.JSInstructionVisitor {
43           @Override
44           public void visitJavaScriptInvoke(JavaScriptInvoke inst) {
45             result = new DeclaredTypeOperator(new ConeType(cha.getRootClass()));
46           }
47     
48           @Override
49           public void visitJavaScriptPropertyRead(JavaScriptPropertyRead inst) {
50             result = new DeclaredTypeOperator(new ConeType(cha.getRootClass()));
51           }
52     
53           @Override
54           public void visitTypeOf(JavaScriptTypeOfInstruction inst) {
55             result = new DeclaredTypeOperator(new PointType(cha.lookupClass(JavaScriptTypes.String)));
56           }
57     
58           @Override
59           public void visitJavaScriptInstanceOf(JavaScriptInstanceOf inst) {
60             result = new DeclaredTypeOperator(new PointType(cha.lookupClass(JavaScriptTypes.Boolean)));
61           }
62     
63           @Override
64           public void visitJavaScriptPropertyWrite(JavaScriptPropertyWrite inst) {
65           }
66     
67           @Override
68           public void visitCheckRef(JavaScriptCheckReference instruction) {
69           }
70     
71           @Override
72           public void visitWithRegion(JavaScriptWithRegion instruction) {
73           }
74     
75           @Override
76           public void visitSetPrototype(SetPrototype instruction) {
77           }
78     
79           @Override
80           public void visitPrototypeLookup(PrototypeLookup instruction) {
81             result = new DeclaredTypeOperator(new ConeType(cha.getRootClass()));
82           }
83         }
84         ;
85     
86         class JSTypeVarFactory extends TypeVarFactory {
87     
88           private TypeAbstraction make(TypeReference typeRef) {
89             return new PointType(cha.lookupClass(typeRef));
90           }
91     
92           @Override
93           public IVariable makeVariable(int vn) {
94             if (ir.getSymbolTable().isStringConstant(vn)) {
95               return new TypeVariable(make(JavaScriptTypes.String));
96             } else if (ir.getSymbolTable().isBooleanConstant(vn)) {
97               return new TypeVariable(make(JavaScriptTypes.Boolean));
98             } else if (ir.getSymbolTable().isNullConstant(vn)) {
99               return new TypeVariable(make(JavaScriptTypes.Null));
100             } else if (ir.getSymbolTable().isNumberConstant(vn)) {
101               return new TypeVariable(make(JavaScriptTypes.Number));
102             } else {
103               return super.makeVariable(vn);
104             }
105           }
106         }
107         ;
108     
109         init(ir, new JSTypeVarFactory(), new JSTypeOperatorFactory());
110       }
111     
112       @Override
113       public TypeAbstraction getConstantType(int valueNumber) {
114         SymbolTable st = ir.getSymbolTable();
115         if (st.isStringConstant(valueNumber)) {
116           return new PointType(cha.lookupClass(JavaScriptTypes.String));
117         } else if (st.isBooleanConstant(valueNumber)) {
118           return new PointType(cha.lookupClass(JavaScriptTypes.Boolean));
119         } else if (st.isNullConstant(valueNumber)) {
120           return new PointType(cha.lookupClass(JavaScriptTypes.Null));
121         } else {
122           return new PointType(cha.lookupClass(JavaScriptTypes.Number));
123         }
124       }
125     }
