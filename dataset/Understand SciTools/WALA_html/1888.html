38     package com.ibm.wala.demandpa.alg.refinepolicy;
39     
40     import java.util.regex.Matcher;
41     import java.util.regex.Pattern;
42     
43     import com.ibm.wala.classLoader.IClass;
44     import com.ibm.wala.classLoader.IField;
45     import com.ibm.wala.demandpa.alg.statemachine.StateMachine;
46     import com.ibm.wala.demandpa.flowgraph.IFlowLabel;
47     import com.ibm.wala.demandpa.util.ArrayContents;
48     import com.ibm.wala.ipa.callgraph.propagation.PointerKey;
49     import com.ibm.wala.ipa.cha.IClassHierarchy;
50     import com.ibm.wala.types.ClassLoaderReference;
51     import com.ibm.wala.types.TypeReference;
52     
53     /**
54      * Manually annotated policy for refining field accesses.
55      */
56     public class ManualFieldPolicy implements FieldRefinePolicy {
57     
58       protected final Pattern refinePattern;// =
59     
60       // Pattern.compile("Lca/mcgill/sable/util|Ljava/util|Lpolyglot/util/TypedList");
61     
62       private static final int NUM_DECISIONS_TO_TRACK = 10;
63     
64       final private boolean[] decisions = new boolean[NUM_DECISIONS_TO_TRACK];
65     
66       private int curDecision;
67     
68       final private IClass[] encounteredClasses = new IClass[NUM_DECISIONS_TO_TRACK];
69     
70       @Override
71       public boolean shouldRefine(IField field, PointerKey basePtr, PointerKey val, IFlowLabel label, StateMachine.State state) {
72         if (field == null) {
73           throw new IllegalArgumentException("null field");
74         }
75         if (field == ArrayContents.v()) {
76           return true;
77         } else {
78           final IClass declaringClass = field.getDeclaringClass();
79           final Matcher m = refinePattern.matcher(declaringClass.toString());
80           final boolean foundPattern = m.find();
81           recordDecision(declaringClass, foundPattern);
82           return foundPattern;
83         }
84       }
85     
86       private void recordDecision(final IClass declaringClass, final boolean foundPattern) {
87         if (curDecision < NUM_DECISIONS_TO_TRACK) {
88           IClass topMostClass = removeInner(declaringClass);
89           if (notSuperOfAnyEncountered(topMostClass)) {
90             encounteredClasses[curDecision] = topMostClass;
91             decisions[curDecision] = foundPattern;
92             curDecision++;
93           }
94         }
95       }
96     
97       private boolean notSuperOfAnyEncountered(IClass klass) {
98         for (int i = 0; i < curDecision; i++) {
99           if (cha.isAssignableFrom(klass, encounteredClasses[i])) {
100             return false;
101           }
102         }
103         return true;
104       }
105     
106       private final IClassHierarchy cha;
107     
108       /**
109        * @return the top-level {@link IClass} where klass is declared, or klass itself if klass is top-level
110        */
111       private IClass removeInner(IClass klass) {
112         ClassLoaderReference cl = klass.getClassLoader().getReference();
113         String klassStr = klass.getName().toString();
114         int dollarIndex = klassStr.indexOf('$');
115         if (dollarIndex == -1) {
116           return klass;
117         } else {
118           String topMostName = klassStr.substring(0, dollarIndex);
119           IClass topMostClass = cha.lookupClass(TypeReference.findOrCreate(cl, topMostName));
120           assert topMostClass != null;
121           return topMostClass;
122         }
123       }
124     
125       /**
126        * @param refinePattern a pattern for detecting which match edges to refine. If the <em>declaring class</em> of the field related
127        *          to the match edge matches the pattern, the match edge will be refined. For example, the pattern
128        *          <code>Pattern.compile("Ljava/util")</code> will cause all fields of classes in the <code>java.util</code> package to
129        *          be refined.
130        */
131       public ManualFieldPolicy(IClassHierarchy cha, Pattern refinePattern) {
132         this.cha = cha;
133         this.refinePattern = refinePattern;
134       }
135     
136       @Override
137       public boolean nextPass() {
138         return false;
139       }
140     
141       /**
142        * 
143        * 
144        * @return a String representation of the decisions made by this
145        */
146       public String getHistory() {
147         StringBuilder ret = new StringBuilder();
148         for (int i = 0; i < curDecision; i++) {
149           if (decisions[i]) {
150             ret.append("refined ");
151           } else {
152             ret.append("skipped ");
153           }
154           ret.append(encounteredClasses[i]);
155           if (i + 1 < curDecision) {
156             ret.append(", ");
157           }
158         }
159         return ret.toString();
160       }
161     
162     }
