34     import com.ibm.wala.ipa.cha.ClassHierarchy;
35     import com.ibm.wala.ipa.cha.ClassHierarchyException;
36     import com.ibm.wala.ipa.cha.IClassHierarchy;
37     import com.ibm.wala.util.debug.Assertions;
38     
39     public class JavaScriptAnalysisEngine extends AbstractAnalysisEngine {
40     
41       protected JavaScriptLoaderFactory loaderFactory;
42     
43       protected JavaScriptTranslatorFactory translatorFactory;
44     
45        public JavaScriptAnalysisEngine() {
46       }
47     
48       @Override
49       public void buildAnalysisScope() {
50         try {
51           loaderFactory = new JavaScriptLoaderFactory(translatorFactory);
52     
53           SourceModule[] files = moduleFiles.toArray(new SourceModule[moduleFiles.size()]);
54     
55           scope = new CAstAnalysisScope(files, loaderFactory, Collections.singleton(JavaScriptLoader.JS));
56         } catch (IOException e) {
57           Assertions.UNREACHABLE(e.toString());
58         }
59       }
60     
61       @Override
62       public IClassHierarchy buildClassHierarchy() {
63         try {
64           return ClassHierarchy.make(getScope(), loaderFactory, JavaScriptLoader.JS);
65         } catch (ClassHierarchyException e) {
66           Assertions.UNREACHABLE(e.toString());
67           return null;
68         }
69       }
70     
71       public void setTranslatorFactory(JavaScriptTranslatorFactory factory) {
72         this.translatorFactory = factory;
73       }
74     
75       @Override
76       public void setJ2SELibraries(JarFile[] libs) {
77         Assertions.UNREACHABLE("Illegal to call setJ2SELibraries");
78       }
79     
80       @Override
81       public void setJ2SELibraries(Module[] libs) {
82         Assertions.UNREACHABLE("Illegal to call setJ2SELibraries");
83       }
84     
85       @Override
86       protected Iterable<Entrypoint> makeDefaultEntrypoints(AnalysisScope scope, IClassHierarchy cha) {
87         return new JavaScriptEntryPoints(cha, cha.getLoader(JavaScriptTypes.jsLoader));
88       }
89     
90       @Override
91       public AnalysisCache makeDefaultCache() {
92         return new AnalysisCache(AstIRFactory.makeDefaultFactory());
93       }
94     
95       @Override
96       public AnalysisOptions getDefaultOptions(Iterable<Entrypoint> roots) {
97         final AnalysisOptions options = new AnalysisOptions(scope, roots);
98     
99         options.setUseConstantSpecificKeys(true);
100     
101         options.setUseStacksForLexicalScoping(true);
102     
103         return options;
104       }
105     
106       @Override
107       protected CallGraphBuilder getCallGraphBuilder(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache) {
108         return new ZeroCFABuilderFactory().make((JSAnalysisOptions) options, cache, cha, scope, false);
109       }
110     }
