67     public class StaticSpecs implements ISpecs {
68     
69     	private final ClassHierarchy cha;
70     	private final String methodSignature;
71         private final Collection<IField> fields;
72     
73     	public StaticSpecs(ClassHierarchy cha, String methodSignature) {
74     		this.cha = cha;
75     		this.methodSignature = methodSignature;
76     		this.fields = collectFields();
77     	}
78     
79     	/* (non-Javadoc)
80     	 * @see org.scandroid.spec.ISpecs#getEntrypointSpecs()
81     	 */
82     	@Override
83     	public MethodNamePattern[] getEntrypointSpecs() {
84     		return new MethodNamePattern[0];
85     	}
86     
87     	/* (non-Javadoc)
88     	 * @see org.scandroid.spec.ISpecs#getSourceSpecs()
89     	 */
90     	@Override
91     	public SourceSpec[] getSourceSpecs() {
92     //		List<SourceSpec> specs = Lists.newArrayList();
93     //
94     //		for (IField field : fields) {
95     //			specs.add(new StaticFieldSourceSpec(field));
96     //		}
97     //
98     //		return specs.toArray(new SourceSpec[] {});
99     		return new SourceSpec[] {};
100     	}
101     
102     	/**
103     	 * @return
104     	 */
105     	private List<IField> collectFields() {
106     		List<IField> fields = new ArrayList<IField>();
107     		Iterator<IClass> itr = cha.iterator();
108     		while (itr.hasNext()) {
109     			IClass cls = itr.next();
110     			for (IField field : cls.getAllStaticFields()) {
111     				if (field.getFieldTypeReference().isReferenceType()) {
112     					fields.add(field);
113     				}
114     			}
115     		}
116     		return fields;
117     	}
118     
119     	/* (non-Javadoc)
120     	 * @see org.scandroid.spec.ISpecs#getSinkSpecs()
121     	 */
122     	@Override
123     	public SinkSpec[] getSinkSpecs() {		
124     		List<SinkSpec> specs = new ArrayList<SinkSpec>();
125     		Collection<IMethod> methods = cha.getPossibleTargets(StringStuff.makeMethodReference(methodSignature));
126     		for (IField field : fields) {
127     			if (!field.isFinal()) {
128     				for (IMethod method : methods) {
129     					specs.add(new StaticFieldSinkSpec(field, method));
130     				}
131     			}
132     		}
133     		return specs.toArray(new SinkSpec[] {});
134     	}
