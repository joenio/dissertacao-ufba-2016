47     /**
48      * An abstract superclass of various {@link SSAContextInterpreter}s that deal with reflection methods.
49      */
50     public abstract class AbstractReflectionInterpreter implements SSAContextInterpreter {
51     
52       protected static final boolean DEBUG = false;
53     
54       protected final static int CONE_BOUND = 10;
55     
56       protected int indexLocal = 100;
57     
58       protected final Map<TypeReference, Integer> typeIndexMap = HashMapFactory.make();
59     
60       /**
61        * Governing analysis options
62        */
63       protected AnalysisOptions options;
64     
65       /**
66        * cache of analysis information
67        */
68       protected AnalysisCache cache;

71       protected int getLocalForType(TypeReference T) {
72         Integer I = typeIndexMap.get(T);
73         if (I == null) {
74           typeIndexMap.put(T, I = new Integer(indexLocal += 2));
75         }
76         return I.intValue();
77       }
78     
79       protected int getExceptionsForType(TypeReference T) {
80         return getLocalForType(T) + 1;
81       }
82     
83       protected int getCallSiteForType(TypeReference T) {
84         return getLocalForType(T);
85       }
86     
87       protected int getNewSiteForType(TypeReference T) {
88         return getLocalForType(T) + 1;
89       }
90     
91       /**
92        * @param type
93        * @return a TypeAbstraction object representing this type. We just use ConeTypes by default, since we don't propagate
94        *         information allowing us to distinguish between points and cones yet.
95        */
96       protected TypeAbstraction typeRef2TypeAbstraction(IClassHierarchy cha, TypeReference type) {
97         IClass klass = cha.lookupClass(type);
98         if (klass != null) {
99           return new ConeType(klass);
100         }
101         Assertions.UNREACHABLE(type.toString());
102         return null;
103       }

169       protected class SpecializedMethod extends SyntheticMethod {
170     
171         /**
172          * Set of types that we have already inserted an allocation for.
173          */
174         protected final HashSet<TypeReference> typesAllocated = HashSetFactory.make(5);
175     
176         /**
177          * List of synthetic allocation statements we model for this specialized instance
178          */
179         final protected ArrayList<SSAInstruction> allocations = new ArrayList<SSAInstruction>();
180     
181         /**
182          * List of synthetic invoke instructions we model for this specialized instance.
183          */
184         final protected ArrayList<SSAInstruction> calls = new ArrayList<SSAInstruction>();
185     
186         /**
187          * List of all instructions
188          */
189         protected final ArrayList<SSAInstruction> allInstructions = new ArrayList<SSAInstruction>();
190     
191         private final SSAInstructionFactory insts = declaringClass.getClassLoader().getInstructionFactory();
192         
193         public SpecializedMethod(MethodReference method, IClass declaringClass, boolean isStatic, boolean isFactory) {
194           super(method, declaringClass, isStatic, isFactory);
195         }
196     
197         public SpecializedMethod(IMethod method, IClass declaringClass, boolean isStatic, boolean isFactory) {
198           super(method, declaringClass, isStatic, isFactory);
199         }
200     
201         /**
202          * @param T type allocated by the instruction.   
203          */
204         protected void addInstruction(final TypeReference T, SSAInstruction instr, boolean isAllocation) {
205           if (isAllocation) {
206             if (typesAllocated.contains(T)) {
207               return;
208             } else {
209               typesAllocated.add(T);
210             }
211           }
212     
213           allInstructions.add(instr);
214           if (isAllocation) {
215             allocations.add(instr);
216           }
217         }
218     
219         /**
220          * @param t type of object to allocate
221          * @return value number of the newly allocated object
222          */
223         protected int addStatementsForConcreteSimpleType(final TypeReference t) {
224           // assert we haven't allocated this type already.
225           assert !typesAllocated.contains(t);
226           if (DEBUG) {
227             System.err.println(("addStatementsForConcreteType: " + t));
228           }
229           NewSiteReference ref = NewSiteReference.make(getNewSiteForType(t), t);
230           int alloc = getLocalForType(t);
231     
232           if (t.isArrayType()) {
233             // for now, just allocate an array of size 1 in each dimension.
234             int dims = 0;
235             int dim = t.getDerivedMask();
236             if ((dim&ElementMask) == PrimitiveMask) {
237               dim >>= 2;
238             }
239             while ((dim&ElementMask) == ArrayMask) {
240               dims++;
241               dim >>=2;
242             }
243             
244             int[] extents = new int[dims];
245             Arrays.fill(extents, 1);
246             SSANewInstruction a = insts.NewInstruction(allInstructions.size(), alloc, ref, extents);
247             addInstruction(t, a, true);
248           } else {
249             SSANewInstruction a = insts.NewInstruction(allInstructions.size(), alloc, ref);
250             addInstruction(t, a, true);
251             addCtorInvokeInstruction(t, alloc);
252           }
253     
254           SSAReturnInstruction r = insts.ReturnInstruction(allInstructions.size(), alloc, false);
255           addInstruction(t, r, false);
256           return alloc;
257         }
258     
259         /**
260          * Add an instruction to invoke the default constructor on the object of value number alloc of type t.
261          */
262         protected void addCtorInvokeInstruction(final TypeReference t, int alloc) {
263           MethodReference init = MethodReference.findOrCreate(t, MethodReference.initAtom, MethodReference.defaultInitDesc);
264           CallSiteReference site = CallSiteReference.make(getCallSiteForType(t), init, IInvokeInstruction.Dispatch.SPECIAL);
265           int[] params = new int[1];
266           params[0] = alloc;
267           int exc = getExceptionsForType(t);
268           SSAInvokeInstruction s = insts.InvokeInstruction(allInstructions.size(), params, exc, site, null);
269           calls.add(s);
270           allInstructions.add(s);
271         }
272       }
