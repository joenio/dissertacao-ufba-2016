11     package com.ibm.wala.fixedpoint.impl;
12     
13     import com.ibm.wala.fixpoint.AbstractStatement;
14     import com.ibm.wala.fixpoint.IVariable;
15     
16     /**
17      * Represents a single step, restricted to a nullary
18      * operator.
19      */
20     @SuppressWarnings("rawtypes")
21     public abstract class NullaryStatement<T extends IVariable> extends AbstractStatement<T, NullaryOperator<T>> {
22     
23       /**
24        * The operands
25        */
26       final protected T lhs;
27     
28       /** 
29        * Evaluate this equation, setting a new value for the
30        * left-hand side. 
31        * 
32        * @return true if the lhs value changed. false otherwise
33        */
34       @Override
35       public byte evaluate() {
36         NullaryOperator<T> op = getOperator();
37         return op.evaluate(lhs);
38       }
39     
40       /** 
41        * Return the left-hand side of this equation.
42        * 
43        * @return the lattice cell this equation computes
44        */
45       @Override
46       public T getLHS() {
47         return lhs;
48       }
49     
50       /** 
51        * Does this equation contain an appearance of a given cell?
52        * @param cell the cell in question
53        * @return true or false
54        */
55       @Override
56       public boolean hasVariable(T cell) {
57         return lhs == cell;
58       }
59     
60       /** 
61        * Constructor for case of one operand on the right-hand side.
62        *
63        * @param lhs the lattice cell set by this equation
64        */
65       protected NullaryStatement(T lhs) {
66         super();
67         this.lhs = lhs;
68       }
69     
70       @Override
71       public boolean equals(Object o) {
72         if (o instanceof NullaryStatement) {
73           NullaryStatement other = (NullaryStatement) o;
74     
75           if (!getOperator().equals(other.getOperator())) {
76             return false;
77           }
78           if (lhs == null) {
79             if (other.lhs != null) {
80               return false;
81             }
82           } else {
83             if (other.lhs == null) {
84               return false;
85             }
86             if (!lhs.equals(other.lhs)) {
87               return false;
88             }
89           }
90           return true;
91         } else {
92           return false;
93         }
94       }
95     
96       @Override
97       public int hashCode() {
98         int result = getOperator().hashCode() * 1381;
99         if (lhs != null) {
100           result += 1399 * lhs.hashCode();
101         }
102         return result;
103       }
104       
105       @Override
106       public T[] getRHS() throws UnsupportedOperationException {
107         throw new UnsupportedOperationException();
108       }
109     }
