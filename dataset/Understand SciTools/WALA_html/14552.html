11     package com.ibm.wala.util.ref;
12     
13     import java.lang.ref.SoftReference;
14     import java.lang.ref.WeakReference;
15     
16     import com.ibm.wala.util.debug.Assertions;
17     
18     /**
19      * A factory for References ... useful for debugging.
20      */
21     public final class CacheReference {
22     
23       private final static byte SOFT = 0;
24     
25       private final static byte WEAK = 1;
26     
27       private final static byte HARD = 2;
28     
29       // should be SOFT except during debugging.
30       private final static byte choice = SOFT;
31     
32       public final static Object make(final Object referent) {
33     
34         switch (choice) {
35         case SOFT:
36           return new SoftReference<Object>(referent);
37         case WEAK:
38           return new WeakReference<Object>(referent);
39         case HARD:
40           return referent;
41         default:
42           Assertions.UNREACHABLE();
43           return null;
44         }
45       }
46     
47       public final static Object get(final Object reference) throws IllegalArgumentException {
48     
49         if (reference == null) {
50           return null;
51         }
52         switch (choice) {
53         case SOFT:
54           if (!(reference instanceof java.lang.ref.SoftReference)) {
55             throw new IllegalArgumentException("not ( reference instanceof java.lang.ref.SoftReference ) ");
56           }
57           return ((SoftReference) reference).get();
58         case WEAK:
59           return ((WeakReference) reference).get();
60         case HARD:
61           return reference;
62         default:
63           Assertions.UNREACHABLE();
64           return null;
65         }
66       }
67     
68     }
