11     package com.ibm.wala.cast.loader;
12     
13     import java.util.HashMap;
14     import java.util.Map;
15     
16     import com.ibm.wala.cast.tree.CAstSourcePositionMap;
17     import com.ibm.wala.classLoader.IClassLoader;
18     import com.ibm.wala.classLoader.IField;
19     import com.ibm.wala.classLoader.IMethod;
20     import com.ibm.wala.types.Selector;
21     import com.ibm.wala.types.TypeName;
22     import com.ibm.wala.types.TypeReference;
23     import com.ibm.wala.util.strings.Atom;
24     
25     public abstract class AstDynamicPropertyClass extends AstClass {
26       private final TypeReference defaultDescriptor;
27     
28       protected AstDynamicPropertyClass(CAstSourcePositionMap.Position sourcePosition, TypeName typeName, IClassLoader loader,
29           short modifiers, Map<Selector, IMethod> declaredMethods, TypeReference defaultDescriptor) {
30         super(sourcePosition, typeName, loader, modifiers, new HashMap<Atom, IField>(), declaredMethods);
31         this.defaultDescriptor = defaultDescriptor;
32       }
33     
34       @Override
35       public IField getField(final Atom name) {
36         if (declaredFields.containsKey(name)) {
37           return declaredFields.get(name);
38         } else if (getSuperclass() != null) {
39           return getSuperclass().getField(name);
40         } else {
41           final boolean isStatic = isStaticField(name);
42           declaredFields.put(name, new AstDynamicField(isStatic, this, name, defaultDescriptor));
43     
44           return declaredFields.get(name);
45         }
46     
47       }
48     
49       protected boolean isStaticField(Atom name) {
50         return name.toString().startsWith("global ");
51       }
52     
53     }
