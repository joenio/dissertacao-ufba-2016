12     package com.ibm.wala.classLoader;
13     
14     import java.io.ByteArrayInputStream;
15     import java.io.ByteArrayOutputStream;
16     import java.io.IOException;
17     import java.io.InputStream;
18     import java.util.HashMap;
19     import java.util.Iterator;
20     import java.util.jar.JarInputStream;
21     import java.util.zip.ZipEntry;
22     
23     import com.ibm.wala.util.collections.HashMapFactory;
24     import com.ibm.wala.util.debug.Assertions;
25     import com.ibm.wala.util.io.FileSuffixes;
26     import com.ibm.wala.util.warnings.Warning;

29     /**
30      * Read in a jar file from an input stream. Most parts are copied from the NestedJarFileModule class
31      * and adapted to work with an input stream. 
32      * @author Juergen Graf <juergen.graf@gmail.com>
33      */
34     public class JarStreamModule implements Module {
35     
36       private static final boolean DEBUG = false;
37     
38       private final JarInputStream stream;
39     
40       /**
41        * For efficiency, we cache the byte[] holding each ZipEntry's contents; this will help avoid multiple unzipping TODO: use a soft
42        * reference?
43        */
44       private HashMap<String, byte[]> cache = null;
45     
46       public JarStreamModule(JarInputStream stream) {
47         if (stream == null) {
48           throw new IllegalArgumentException("null stream");
49         }
50         
51         this.stream = stream;
52       }
53     
54       public InputStream getInputStream(String name) {
55         populateCache();
56         byte[] b = cache.get(name);
57         return new ByteArrayInputStream(b);
58       }
59     
60       private void populateCache() {
61         if (cache != null) {
62           return;
63         }
64         cache = HashMapFactory.make();
65         try {
66           for (ZipEntry z = stream.getNextEntry(); z != null; z = stream.getNextEntry()) {
67             final String name = z.getName();
68             if (DEBUG) {
69               System.err.println(("got entry: " + name));
70             }
71             if (FileSuffixes.isClassFile(name) || FileSuffixes.isSourceFile(name)) {
72               ByteArrayOutputStream out = new ByteArrayOutputStream();
73               byte[] temp = new byte[1024];
74               int n = stream.read(temp);
75               while (n != -1) {
76                 out.write(temp, 0, n);
77                 n = stream.read(temp);
78               }
79               byte[] bb = out.toByteArray();
80               cache.put(name, bb);
81             }
82           }
83         } catch (IOException e) {
84           // just go with what we have
85           Warnings.add(new Warning() {
86     
87             @Override
88             public String getMsg() {
89               return "could not read contents of jar input stream.";
90             }
91           });
92         }
93     
94       }
95     
96       protected long getEntrySize(String name) {
97         populateCache();
98         byte[] b = cache.get(name);
99         return b.length;
100       }
101     
102       /*
103        * @see com.ibm.wala.classLoader.Module#getEntries()
104        */
105       public Iterator<ModuleEntry> getEntries() {
106         populateCache();
107         final Iterator<String> it = cache.keySet().iterator();
108         return new Iterator<ModuleEntry>() {
109           String next = null;
110           {
111             advance();
112           }
113     
114           private void advance() {
115             if (it.hasNext()) {
116               next = it.next();
117             } else {
118               next = null;
119             }
120           }
121     
122           public boolean hasNext() {
123             return next != null;
124           }
125     
126           public ModuleEntry next() {
127             ModuleEntry result = new Entry(next);
128             advance();
129             return result;
130           }
131     
132           public void remove() {
133             Assertions.UNREACHABLE();
134           }
135         };
136       }
137     
138       /**
139        * @author sfink an entry in a nested jar file.
140        */
141       private class Entry implements ModuleEntry {
142     
143         private final String name;
144     
145         Entry(String name) {
146           this.name = name;
147         }
148     
149         /*
150          * @see com.ibm.wala.classLoader.ModuleEntry#getName()
151          */
152         public String getName() {
153           return name;
154         }
155     
156         /*
157          * @see com.ibm.wala.classLoader.ModuleEntry#isClassFile()
158          */
159         public boolean isClassFile() {
160           return FileSuffixes.isClassFile(getName());
161         }
162     
163         /*
164          * @see com.ibm.wala.classLoader.ModuleEntry#getInputStream()
165          */
166         public InputStream getInputStream() {
167           return JarStreamModule.this.getInputStream(name);
168         }
169     
170         /*
171          * @see com.ibm.wala.classLoader.ModuleEntry#isModuleFile()
172          */
173         public boolean isModuleFile() {
174           return false;
175         }
176     
177         /*
178          * @see com.ibm.wala.classLoader.ModuleEntry#asModule()
179          */
180         public Module asModule() {
181           Assertions.UNREACHABLE();
182           return null;
183         }
184     
185         @Override
186         public String toString() {
187           return "nested entry: " + name;
188         }
189     
190         /*
191          * @see com.ibm.wala.classLoader.ModuleEntry#getClassName()
192          */
193         public String getClassName() {
194           return FileSuffixes.stripSuffix(getName());
195         }
196     
197         /*
198          * @see com.ibm.wala.classLoader.ModuleEntry#isSourceFile()
199          */
200         public boolean isSourceFile() {
201           return FileSuffixes.isSourceFile(getName());
202         }
203     
204         @Override
205         public int hashCode() {
206           final int prime = 31;
207           int result = 1;
208           result = prime * result + getOuterType().hashCode();
209           result = prime * result + ((name == null) ? 0 : name.hashCode());
210           return result;
211         }
212     
213         @Override
214         public boolean equals(Object obj) {
215           if (this == obj)
216             return true;
217           if (obj == null)
218             return false;
219           if (getClass() != obj.getClass())
220             return false;
221           Entry other = (Entry) obj;
222           if (!getOuterType().equals(other.getOuterType()))
223             return false;
224           if (name == null) {
225             if (other.name != null)
226               return false;
227           } else if (!name.equals(other.name))
228             return false;
229           return true;
230         }
231     
232         private JarStreamModule getOuterType() {
233           return JarStreamModule.this;
234         }
235     
236         @Override
237         public Module getContainer() {
238           return JarStreamModule.this;
239         }
240     
241       }
242     
243       @Override
244       public String toString() {
245         return "Jar input stream " + stream.toString();
246       }
247     
248       @Override
249       public int hashCode() {
250         final int prime = 31;
251         int result = 1;
252         result = prime * result + (stream.hashCode());
253         return result;
254       }
255     
256       @Override
257       public boolean equals(Object obj) {
258         if (this == obj)
259           return true;
260         if (obj == null)
261           return false;
262         if (getClass() != obj.getClass())
263           return false;
264         JarStreamModule other = (JarStreamModule) obj;
265         return stream.equals(other.stream);
266       }
