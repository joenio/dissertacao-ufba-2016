52     /**
53      * Determines the target of an Android-Intent.
54      *
55      * An Intent is generated at each Constructor-Call of an Android-Intent. Information to that Constructor
56      * is added as target. 
57      *
58      * If you want to change the target don't change this Object! Instead place an override using the 
59      * AndroidEntryPointManager so no information is lost.
60      *
61      * Upon the creation of an Intent it's target-type almost always points to UNKNOWN. Again change this
62      * using an override.
63      *
64      * This class contains functions that determine the target-type on the Intent itself. They are intended
65      * as fallback only.
66      *
67      * CAUTION: If you inherit from this class - keep hashCodes and equals clashing!
68      *
69      * @see     com.ibm.wala.dalvik.util.AndroidEntryPointManager
70      * @see     com.ibm.wala.dalvik.ipa.callgraph.propagation.cfa.IntentContextSelector
71      * @see     com.ibm.wala.dalvik.ipa.callgraph.propagation.cfa.IntentContextInterpreter
72      *
73      * @author  Tobias Blaschke <code@tobiasblaschke.de>
74      * @since   2013-10-12
75      */
76     public class Intent implements ContextItem, Comparable<Intent> {

81         public enum IntentType {
82             /** The target could not be Identified for sure */
83             UNKNOWN_TARGET,     
84             /** The Class the Intent Action refers to belongs to this App */
85             INTERNAL_TARGET,   
86             /** The Class the Intent Action refers to belongs to an external App */
87             EXTERNAL_TARGET,    
88             /** The Action is a Constant defined on the Android-Reference Manual */
89             STANDARD_ACTION,    
90             SYSTEM_SERVICE,
91             /** So External and maybe internal */
92             BROADCAST,          
93             /** Do not handle intent */
94             IGNORE
95         };

167         /**
168          *  Set the explicit target of the intent.
169          *
170          *  If setAction is called multible times on an Intent it becomes UNBOUND.
171          */
172         public void setActionExplicit(Atom action) {
173             if (action == null) {
174                 throw new IllegalArgumentException("Action may not be null!");
175             }
176     
177             if (this.action == null) {
178                 assert (this.explicit == Explicit.UNSET) : "No Action but Intent is not UNSET - is " + this.explicit;
179                 assert (! immutable) : "Intent was marked immutable - can't change it.";
180                 this.action = action;
181                 this.explicit = Explicit.EXPLICIT;
182             } else if (isExplicit() && (! this.action.equals(action))) {
183                 // We already have the explicit target. Ignore the change.
184                 
185                 unbind();
186             } else if (! isExplicit() ) {
187                 
188                 assert (! immutable) : "Intent was marked immutable - can't change it.";
189                 this.action = action;
190                 this.explicit = Explicit.EXPLICIT;
191                 // TODO: Set type?
192             } else {
193                 // Set to same values - OK
194             }
195         }

199             assert (! immutable) : "Intent was marked immutable - can't change it.";
200             this.action = UNBOUND;
201             this.type = IntentType.UNKNOWN_TARGET;
202             this.explicit = Explicit.MULTI;             // XXX shoulb we do this?
203         }

205         /**
206          *  Set the target of the intent.
207          *
208          *  If setAction is called multible times on an Intent it becomes UNBOUND.
209          */
210         public void setAction(Atom action) {
211             if (this.action == null) {
212                 assert (! immutable) : "Intent was marked immutable - can't change it.";
213                 this.action = action;
214             } else if (isExplicit()) {
215                 // We already have the explicit target. Ignore the change.
216             } else if (! action.equals(this.action)) {
217                 unbind();
218             }
219         }

221         public Atom getAction() {
222             if (this.action == null)  {
223                 assert (! isExplicit()) : "Beeing explicit implies having an action!";
224                 return UNBOUND; 
225             }
226             return this.action;
227         }

229         public IntentType getType() {
230             if (this.type != null) {
231                 return this.type;
232             } else {
233                 if (isSystemService(this)) {
234                     this.type = IntentType.SYSTEM_SERVICE;
235                 } else if (isStandardAction(this)) {
236                     this.type = IntentType.STANDARD_ACTION;
237                 } else if (isInternal(this)) {
238                     this.type = IntentType.INTERNAL_TARGET;
239                 } else if (isExternal(this)) {
240                     this.type = IntentType.EXTERNAL_TARGET;
241                 } else {
242                     this.type = IntentType.UNKNOWN_TARGET;
243                 }
244             }
245             return this.type;
246         }

339         /**
340          *  Is the Intents target internally resolvable.
341          *
342          *  @return if the Intent is associated to a class in the analyzed application.
343          *  @param  strict   if false return unknown target as internal
344          */
345         public boolean isInternal(boolean strict) {
346             IntentType type = getType();    // Asserts type is computed
347             
348             return ((type == IntentType.INTERNAL_TARGET) || ((! strict) && (type == IntentType.UNKNOWN_TARGET)));
349         }

351         /**
352          *  Has the target to be resolved by an external App.
353          *
354          *  The Intent is not associated to a class in this application or it's a Standard
355          *  action defined in the Android Reference Manual.
356          *
357          * @param   strict    if false return unknown target as external
358          */
359         public boolean isExternal(boolean strict) {
360             IntentType type = getType();    // Asserts type is computed
361             
362             return ((type == IntentType.EXTERNAL_TARGET) || (type == IntentType.STANDARD_ACTION) 
363                     || ((! strict) && (type == IntentType.UNKNOWN_TARGET)));
364         }

367          *  Is the Intent one of the System-Defined ones.
368          *
369          *  It's a Standard action defined in the Android Reference Manual. Implies isExternal.
370          *
371          *  @param  strict    if false return unknown target as standard
372          */
373         public boolean isStandard(boolean strict) {
374             IntentType type = getType();    // Asserts type is computed
375             
376             return ((type == IntentType.STANDARD_ACTION) || ((! strict) && (type == IntentType.UNKNOWN_TARGET)));
377         }

380         @Override
381         public String toString() {
382             StringBuffer ret;
383             if ((this.action == null) || (this.action.equals(UNBOUND))) {
384                 return "Unbound Intent";
385             } else if (getType() == IntentType.SYSTEM_SERVICE) {
386                 ret = new StringBuffer("SystemService(");
387             } else {
388                 ret = new StringBuffer("Intent(");
389             }
390             ret.append(this.action.toString());
391             if (uri != null) {
392                 ret.append(", ");
393                 ret.append(this.uri.toString());
394             }
395             ret.append(") of type ");
396             ret.append(getType());
397             return ret.toString();
398         }

433             if (o instanceof Intent) {
434                 Intent other = (Intent) o;
435     
436                 // DO NOT USE TYPE!
437                 if (this.uri != null) {
438                     return ( (this.uri.equals(other.uri)) && equalAction(other) ); // && (this.explicit == other.explicit));
439                 } else {
440                     return ( (other.uri == null) && equalAction(other) ); // && (this.explicit == other.explicit)) ;
441                 }
442             } else {
443                 System.err.println("WARNING: Can't compare Intent to " + o.getClass());
444                 return false;
445             }
446         }
