11     package com.ibm.wala.shrikeCT;
12     
13     /**
14      * This class reads InnerClasses attributes.
15      */
16     public final class InnerClassesReader extends AttributeReader {
17       /**
18        * Build a reader for the attribute 'iter'.
19        */
20       public InnerClassesReader(ClassReader.AttrIterator iter) throws InvalidClassFileException {
21         super(iter, "InnerClasses");
22     
23         checkSize(attr, 8);
24         int count = cr.getUShort(attr + 6);
25         checkSizeEquals(attr + 8, 8 * count);
26       }
27     
28       /**
29        * @return the raw values that make up this attribute
30        */
31       public int[] getRawTable() {
32         int count = cr.getUShort(attr + 6);
33         int[] r = new int[count * 4];
34         for (int i = 0; i < r.length; i++) {
35           r[i] = cr.getUShort(attr + 8 + i * 2);
36         }
37         return r;
38       }
39     
40       /**
41        * @return the names of inner classes this attribute holds information about.
42        */
43       public String[] getInnerClasses() throws InvalidClassFileException {
44         int count = cr.getUShort(attr + 6);
45         String[] r = new String[count];
46         ConstantPoolParser cp = cr.getCP();
47         for (int i = 0; i < r.length; i++) {
48           r[i] = cp.getCPClass(cr.getUShort(attr + 8 + i * 8));
49         }
50         return r;
51       }
52     
53       /**
54        * return the name of the outer class recorded as the enclosing class for a class named s. return null if not found.
55        * 
56        * @throws InvalidClassFileException
57        */
58       public String getOuterClass(String s) throws InvalidClassFileException {
59         String[] inner = getInnerClasses();
60         for (int i = 0; i < inner.length; i++) {
61           if (inner[i].equals(s)) {
62             int x = cr.getUShort(attr + 8 + i * 8 + 2);
63             if (x != 0) {
64               ConstantPoolParser cp = cr.getCP();
65               return cp.getCPClass(cr.getUShort(attr + 8 + i * 8 + 2));
66             }
67           }
68         }
69         return null;
70       }
71     
72       /**
73        * return the mask of flags recorded in the InnerClasses attribute for a class named s. return 0 if not found.
74        * 
75        * @throws InvalidClassFileException
76        */
77       public int getAccessFlags(String s) throws InvalidClassFileException {
78         String[] inner = getInnerClasses();
79         for (int i = 0; i < inner.length; i++) {
80           if (inner[i].equals(s)) {
81             return cr.getUShort(attr + 8 + i * 8 + 6);
82           }
83         }
84         return 0;
85       }
86     
87     }
