12     package com.ibm.wala.cast.tree.pattern;
13     
14     import com.ibm.wala.cast.tree.CAstNode;
15     
16     /**
17      * A node pattern matching any constant. A pattern of this class also stores its last match.
18      * 
19      * @author mschaefer
20      *
21      */
22     public class SomeConstant extends NodeOfKind {
23     	private Object last_match;
24     	
25     	public SomeConstant() {
26     		super(CAstNode.CONSTANT);
27     	}
28     	
29     	@Override
30     	public boolean matches(CAstNode node) {
31     		boolean res = super.matches(node);
32     		if(res)
33     			this.last_match = node.getValue();
34     		return res;
35     	}
36     	
37     	public Object getLastMatch() {
38     		return last_match;
39     	}
40     }
