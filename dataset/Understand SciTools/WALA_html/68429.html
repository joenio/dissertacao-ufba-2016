11     package com.ibm.wala.dataflow.IFDS;
12     
13     /**
14      * A {@link TabulationProblem} with additional support for computing with partially balanced parentheses.
15      * 
16      * @param <T> type of node in the supergraph
17      * @param <P> type of a procedure (like a box in an RSM)
18      * @param <F> type of factoids propagated when solving this problem
19      * 
20      */
21     public interface PartiallyBalancedTabulationProblem<T, P, F> extends TabulationProblem<T, P, F> {
22     
23       @Override
24       public IPartiallyBalancedFlowFunctions<T> getFunctionMap();
25     
26       /**
27        * If n is reached by a partially balanced parenthesis, what is the entry node we should use as the root of the {@link PathEdge}
28        * to n? Note that the result <em>must</em> in fact be an entry node of the procedure containing n.
29        */
30       public T getFakeEntry(T n);
31     }
