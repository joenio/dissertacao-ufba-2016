11     package com.ibm.wala.fixpoint;
12     
13     
14     /**
15      * operator for a step in an iterative solver
16      * 
17      * This is an abstract class and not an interface in order to force subclasses to re-implement equals(), hashCode(), and toString()
18      */
19     @SuppressWarnings("rawtypes")
20     public abstract class AbstractOperator<T extends IVariable> implements FixedPointConstants {
21     
22       /**
23        * Evaluate this equation, setting a new value for the left-hand side.
24        * 
25        * @return a code that indicates: 1) has the lhs value changed? 2) has this equation reached a fixed-point, in that we never have
26        *         to evaluate the equation again, even if rhs operands change?
27        */
28       public abstract byte evaluate(T lhs, T[] rhs);
29     
30       @Override
31       public abstract int hashCode();
32     
33       @Override
34       public abstract boolean equals(Object o);
35     
36       @Override
37       public abstract String toString();
38     }
