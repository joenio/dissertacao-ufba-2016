11     package com.ibm.wala.ipa.cha;
12     
13     import com.ibm.wala.classLoader.IClassLoader;
14     
15     /**
16      * Statistics about a class hierarchy.
17      */
18     public class ClassHierarchyStats {
19     
20       /**
21        * Dump stats about the class hierarchy to stdout.
22        */
23       public static void printStats(IClassHierarchy cha) throws IllegalArgumentException {
24         if (cha == null) {
25           throw new IllegalArgumentException("cha cannot be null");
26         }
27         IClassLoader[] loaders = cha.getLoaders();
28         for (int i = 0; i < loaders.length; i++) {
29           System.out.println("loader: " + loaders[i]);
30           System.out.println("  classes: " + loaders[i].getNumberOfClasses());
31           System.out.println("  methods: " + loaders[i].getNumberOfMethods());
32         }
33       }
34     }
