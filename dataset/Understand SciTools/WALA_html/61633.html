20     import com.ibm.wala.util.debug.UnimplementedError;

22     /**
23      * a debugging aid. This implementation complains if you stick an object in here which appears to use System.identityHashCode(), or
24      * if it detects more than BAD_HC collisions in the Set (possibly indicated a bad hash function)
25      */
26     public class ParanoidHashSet<T> extends LinkedHashSet<T> {
27       public static final long serialVersionUID = 30919839181133333L;
28     
29       /**
30        * A mapping from Integer (hashcode) -> Set of objects
31        */
32       private final Map<Integer, Set<T>> hcFreq;
33     
34       private int nAdded = 0;
35     
36       /**
37        * If a hash set contains more than this number of items with the same hash code, complain.
38        */
39       private final int BAD_HC = 3;
40     
41       /**
42        * @param s
43        * @throws NullPointerException if s is null
44        */
45       public ParanoidHashSet(Collection<T> s) throws NullPointerException {
46         super(s.size());
47         hcFreq = HashMapFactory.make(s.size());
48         for (Iterator<T> it = s.iterator(); it.hasNext();) {
49           add(it.next());
50         }
51       }
52     
53       /**
54        * 
55        */
56       public ParanoidHashSet() {
57         super();
58         hcFreq = HashMapFactory.make();
59       }
60     
61       public ParanoidHashSet(int size) {
62         super(size);
63         hcFreq = HashMapFactory.make(size);
64       }
65     
66       /**
67        * @see java.util.Collection#add(java.lang.Object)
68        * @throws UnimplementedError if there's a bad hash code problem
69        */
70       @Override
71       public boolean add(T arg0) {
72         if (arg0 == null) {
73           throw new IllegalArgumentException("arg0 is null");
74         }
75     
76         ParanoidHashMap.assertOverridesHashCode(arg0);
77         boolean result = super.add(arg0);
78         if (result) {
79           nAdded++;
80           int hc = arg0.hashCode();
81           Set<T> s = hcFreq.get(new Integer(hc));
82           if (s == null) {
83             HashSet<T> h = new LinkedHashSet<T>(1);
84             h.add(arg0);
85             hcFreq.put(new Integer(hc), h);
86           } else {
87             if (s.size() == BAD_HC) {
88               for (Iterator<T> it = s.iterator(); it.hasNext();) {
89                 Object o = it.next();
90                 System.err.println(o + " " + o.hashCode());
91               }
92               assert false : "bad hc " + arg0.getClass() + " " + arg0;
93             } else {
94               s.add(arg0);
95             }
96           }
97         }
98         return result;
99       }
100     }
