177       private void collectInstructionInfo() {
178         final BitSet s = new BitSet(instructions.length);
179         final BitSet localsUsed = new BitSet(32);
180         final BitSet localsWide = new BitSet(32);
181     
182         IInstruction.Visitor visitor = new IInstruction.Visitor() {
183           private void visitTargets(IInstruction instr) {
184             int[] ts = instr.getBranchTargets();
185             for (int k = 0; k < ts.length; k++) {
186               s.set(ts[k]);
187             }
188           }

190           @Override
191           public void visitGoto(GotoInstruction instruction) {
192             visitTargets(instruction);
193           }

195           @Override
196           public void visitLocalStore(IStoreInstruction instruction) {
197             localsUsed.set(instruction.getVarIndex());
198             String t = instruction.getType();
199             if (t.equals(TYPE_long) || t.equals(TYPE_double)) {
200               localsWide.set(instruction.getVarIndex());
201             }
202           }
203     
204           @Override
205           public void visitConditionalBranch(IConditionalBranchInstruction instruction) {
206             visitTargets(instruction);
207           }
208     
209           @Override
210           public void visitSwitch(SwitchInstruction instruction) {
211             visitTargets(instruction);
212           }
213         };
214     
215         for (int i = 0; i < instructions.length; i++) {
216           instructions[i].visit(visitor);
217         }
218     
219         String[] paramTypes = Util.getParamsTypes(isStatic ? null : TYPE_Object, signature);
220         int index = 0;
221         for (int i = 0; i < paramTypes.length; i++) {
222           String t = paramTypes[i];
223           localsUsed.set(index);
224           if (t.equals(TYPE_long) || t.equals(TYPE_double)) {
225             localsWide.set(index);
226             index += 2;
227           } else {
228             index++;
229           }
230         }
231     
232         ExceptionHandler[] lastHS = null;
233         for (int i = 0; i < handlers.length; i++) {
234           ExceptionHandler[] hs = handlers[i];
235           if (hs != lastHS) {
236             for (int j = 0; j < hs.length; j++) {
237               s.set(hs[j].handler);
238             }
239             lastHS = hs;
240           }
241         }
242     
243         this.branchTargets = s;
244     
245         int maxUsed = localsUsed.length();
246         if (maxUsed > 0 && localsWide.get(maxUsed - 1)) {
247           maxUsed++;
248         }
249         this.maxLocals = maxUsed;
250       }

253         code[offset] = (byte) (v >> 24);

268       private boolean inBasicBlock(int i, int n) {
269         if (i + n - 1 >= instructions.length) {
270           return false;
271         }
272     
273         for (int j = i + 1; j < i + n; j++) {
274           if (branchTargets.get(j)) {
275             return false;
276           }
277     
278           if (!Arrays.equals(handlers[j], handlers[i])) {
279             return false;
280           }
281     
282           if (instructionsToBytecodes[j] != instructionsToBytecodes[i]) {
283             return false;
284           }
285         }
286     
287         return true;
288       }

310       private void computeStackWordsAt(int i, int stackLen, byte[] stackWords, boolean[] visited) {
311         while (!visited[i]) {
312           IInstruction instr = instructions[i];
313     
314           if (i > 0 && !instructions[i - 1].isFallThrough()) {
315             byte[] newWords = new byte[stackLen];
316             System.arraycopy(stackWords, 0, newWords, 0, stackLen);
317             this.stackWords[i] = newWords;
318           }
319     
320           visited[i] = true;
321     
322           if (stackLen < instr.getPoppedCount()) {
323             throw new IllegalArgumentException("Stack underflow in intermediate code, at offset " + i);
324           }
325     
326           if (instr instanceof DupInstruction) {
327             DupInstruction d = (DupInstruction) instr;
328             int size = d.getSize();
329             int delta = d.getDelta();
330     
331             System.arraycopy(stackWords, stackLen - size - delta, stackWords, stackLen - delta, delta + size);
332             System.arraycopy(stackWords, stackLen, stackWords, stackLen - size - delta, size);
333             stackLen += size;
334             checkStackWordSize(stackWords, stackLen);
335           } else if (instr instanceof SwapInstruction) {
336             // we may have to emulate this using a dup[2]_x[1,2]
337             // followed by a pop[2]. So update the maxStack to account for the
338             // temporarily larger stack size
339             int words = stackWords[stackLen - 1];
340             for (int j = 0; j < stackLen; j++) {
341               words += stackWords[j];
342             }
343             if (words > maxStack) {
344               maxStack = words;
345             }
346     
347             byte b = stackWords[stackLen - 2];
348             stackWords[stackLen - 2] = stackWords[stackLen - 1];
349             stackWords[stackLen - 1] = b;
350           } else {
351             stackLen -= instr.getPoppedCount();
352     
353             byte w = instr.getPushedWordSize();
354             if (w > 0) {
355               stackWords[stackLen] = w;
356               stackLen++;
357               checkStackWordSize(stackWords, stackLen);
358             }
359           }
360     
361           int[] bt = instr.getBranchTargets();
362           for (int j = 0; j < bt.length; j++) {
363             int t = bt[j];
364             if (t < 0 || t >= visited.length) {
365               throw new IllegalArgumentException("Branch target at offset " + i + " is out of bounds: " + t + " (max " + visited.length
366                   + ")");
367             }
368             if (!visited[t]) {
369               computeStackWordsAt(bt[j], stackLen, stackWords.clone(), visited);
370             }
371           }
372     
373           ExceptionHandler[] hs = handlers[i];
374           for (int j = 0; j < hs.length; j++) {
375             int t = hs[j].handler;
376             if (!visited[t]) {
377               byte[] newWords = stackWords.clone();
378               newWords[0] = 1;
379               computeStackWordsAt(t, 1, newWords, visited);
380             }
381           }
382     
383           if (!instr.isFallThrough()) {
384             return;
385           }
386     
387           i++;
388         }
389       }

477       private boolean outputInstructions(int startInstruction, int endInstruction, int startOffset, boolean farBranches,
478           byte[] initialStack) {
479         instructionsToOffsets = new int[instructions.length];
480         code = makeCodeBuf();
481     
482         ArrayList<Patch> patches = new ArrayList<Patch>();
483     
484         int curOffset = startOffset;
485         final int[] curOffsetRef = new int[1];
486         int stackLen = initialStack == null ? 0 : initialStack.length;
487         final int[] stackLenRef = new int[1];
488         final byte[] stackWords = new byte[maxStack];
489         if (stackLen > 0) {
490           System.arraycopy(initialStack, 0, stackWords, 0, stackLen);
491         }
492         final int[] instrRef = new int[1];
493     
494         IInstruction.Visitor noOpcodeHandler = new IInstruction.Visitor() {
495           @Override
496           public void visitPop(PopInstruction instruction) {
497             int count = instruction.getPoppedCount();
498             int offset = curOffsetRef[0];
499             int stackLen = stackLenRef[0];
500     
501             while (count > 0) {
502               code[offset] = (byte) (stackWords[stackLen - 1] == 1 ? OP_pop : OP_pop2);
503               count--;
504               stackLen--;
505               offset++;
506             }
507     
508             curOffsetRef[0] = offset;
509           }
510     
511           @Override
512           public void visitDup(DupInstruction instruction) {
513             int size = instruction.getSize();
514             int delta = instruction.getDelta();
515             int offset = curOffsetRef[0];
516             int stackLen = stackLenRef[0];
517     
518             int sizeWords = stackWords[stackLen - 1];
519             if (size == 2) {
520               sizeWords += stackWords[stackLen - 2];
521             }
522             int deltaWords = delta == 0 ? 0 : stackWords[stackLen - 1 - size];
523             if (delta == 2) {
524               deltaWords += stackWords[stackLen - 1 - size - 1];
525             }
526             if (sizeWords > 2 || deltaWords > 2) {
527               throw new IllegalArgumentException("Invalid dup size");
528             }
529     
530             code[offset] = (byte) (OP_dup + (3 * (sizeWords - 1)) + deltaWords);
531             offset++;
532             curOffsetRef[0] = offset;
533           }
534     
535           @Override
536           public void visitSwap(SwapInstruction instruction) {
537             int offset = curOffsetRef[0];
538             int stackLen = stackLenRef[0];
539             int topSize = stackWords[stackLen - 1];
540             int nextSize = stackWords[stackLen - 2];
541     
542             if (topSize == 1 && nextSize == 1) {
543               code[offset] = (byte) OP_swap;
544               offset++;
545             } else {
546               code[offset] = (byte) (OP_dup + (3 * (topSize - 1)) + nextSize);
547               code[offset + 1] = (byte) (topSize == 1 ? OP_pop : OP_pop2);
548               offset += 2;
549             }
550     
551             curOffsetRef[0] = offset;
552           }
553         };
554     
555         for (int i = startInstruction; i < endInstruction; i++) {
556           Instruction instr = (Instruction) instructions[i];
557           int opcode = instr.getOpcode();
558           int startI = i;
559     
560           instructionsToOffsets[i] = curOffset;
561     
562           if (opcode != -1) {
563             boolean fallToConditional = false;
564     
565             code[curOffset] = (byte) opcode;
566             curOffset++;
567     
568             switch (opcode) {
569             case OP_iconst_0:
570               if (inBasicBlock(i, 2) && instructions[i + 1] instanceof ConditionalBranchInstruction) {
571                 ConditionalBranchInstruction cbr = (ConditionalBranchInstruction) instructions[i + 1];
572                 if (cbr.getType().equals(TYPE_int)) {
573                   code[curOffset - 1] = (byte) (cbr.getOperator().ordinal() + OP_ifeq);
574                   fallToConditional = true;
575                   i++;
576                   instr = (Instruction) instructions[i];
577                 }
578               }
579               if (!fallToConditional) {
580                 break;
581               }
582             case OP_aconst_null:
583               if (!fallToConditional && inBasicBlock(i, 2) && instructions[i + 1] instanceof ConditionalBranchInstruction) {
584                 ConditionalBranchInstruction cbr = (ConditionalBranchInstruction) instructions[i + 1];
585                 if (cbr.getType().equals(TYPE_Object)) {
586                   code[curOffset - 1] = (byte) (cbr.getOperator().ordinal() + OP_ifnull);
587                   fallToConditional = true;
588                   i++;
589                   instr = (Instruction) instructions[i];
590                 }
591               }
592               if (!fallToConditional) {
593                 break;
594               }
595               // by Xiangyu
596             case OP_ifeq:
597             case OP_ifge:
598             case OP_ifgt:
599             case OP_ifle:
600             case OP_iflt:
601             case OP_ifne: {
602               int targetI = instr.getBranchTargets()[0];
603               boolean invert = false;
604               int iStart = curOffset - 1;
605     
606               if (inBasicBlock(i, 2) && instr.getBranchTargets()[0] == i + 2 && instructions[i + 1] instanceof GotoInstruction) {
607                 invert = true;
608                 targetI = instructions[i + 1].getBranchTargets()[0];
609                 i++;
610               }
611     
612               if (targetI <= i) {
613                 int delta = instructionsToOffsets[targetI] - iStart;
614                 if ((short) delta != delta) {
615                   // emit "if_!XX TMP; goto_w L; TMP:"
616                   invert = !invert;
617                   writeShort(curOffset, 8);
618                   code[curOffset + 2] = (byte) OP_goto_w;
619                   writeInt(curOffset + 3, delta - 3);
620                   curOffset += 7;
621                 } else {
622                   writeShort(curOffset, (short) delta);
623                   curOffset += 2;
624                 }
625               } else {
626                 Patch p;
627                 if (farBranches) {
628                   // emit "if_!XX TMP; goto_w L; TMP:"
629                   invert = !invert;
630                   writeShort(curOffset, 8);
631                   code[curOffset + 2] = (byte) OP_goto_w;
632                   p = new IntPatch(curOffset + 2, curOffset + 3, targetI);
633                   curOffset += 7;
634                 } else {
635                   p = new ShortPatch(iStart, curOffset, targetI);
636                   curOffset += 2;
637                 }
638                 patches.add(p);
639               }
640     
641               if (invert) {
642                 code[iStart] = (byte) (((code[iStart] - OP_ifeq) ^ 1) + OP_ifeq);
643               }
644               break;
645             }
646               // by Xiangyu
647     
648             case OP_if_icmpeq:
649             case OP_if_icmpge:
650             case OP_if_icmpgt:
651             case OP_if_icmple:
652             case OP_if_icmplt:
653             case OP_if_icmpne:
654             case OP_if_acmpeq:
655             case OP_if_acmpne: {
656               int targetI = instr.getBranchTargets()[0];
657               boolean invert = false;
658               int iStart = curOffset - 1;
659     
660               if (inBasicBlock(i, 2) && instr.getBranchTargets()[0] == i + 2 && instructions[i + 1] instanceof GotoInstruction) {
661                 invert = true;
662                 targetI = instructions[i + 1].getBranchTargets()[0];
663                 i++;
664               }
665     
666               if (targetI <= i) {
667                 int delta = instructionsToOffsets[targetI] - iStart;
668                 if ((short) delta != delta) {
669                   // emit "if_!XX TMP; goto_w L; TMP:"
670                   invert = !invert;
671                   writeShort(curOffset, 8);
672                   code[curOffset + 2] = (byte) OP_goto_w;
673                   writeInt(curOffset + 3, delta - 3);
674                   curOffset += 7;
675                 } else {
676                   writeShort(curOffset, (short) delta);
677                   curOffset += 2;
678                 }
679               } else {
680                 Patch p;
681                 if (farBranches) {
682                   // emit "if_!XX TMP; goto_w L; TMP:"
683                   invert = !invert;
684                   writeShort(curOffset, 8);
685                   code[curOffset + 2] = (byte) OP_goto_w;
686                   p = new IntPatch(curOffset + 2, curOffset + 3, targetI);
687                   curOffset += 7;
688                 } else {
689                   p = new ShortPatch(iStart, curOffset, targetI);
690                   curOffset += 2;
691                 }
692                 patches.add(p);
693               }
694     
695               if (invert) {
696                 code[iStart] = (byte) (((code[iStart] - OP_if_icmpeq) ^ 1) + OP_if_icmpeq);
697               }
698               break;
699             }
700             case OP_bipush:
701               writeByte(curOffset, ((ConstantInstruction.ConstInt) instr).getIntValue());
702               curOffset++;
703               break;
704             case OP_sipush:
705               writeShort(curOffset, ((ConstantInstruction.ConstInt) instr).getIntValue());
706               curOffset += 2;
707               break;
708             case OP_ldc_w: {
709               int cpIndex;
710               ConstantInstruction ci = (ConstantInstruction) instr;
711               if (presetConstants != null && ci.getLazyConstantPool() == presetConstants) {
712                 cpIndex = ci.getCPIndex();
713               } else {
714                 String t = instr.getPushedType(null);
715                 if (t.equals(TYPE_int)) {
716                   cpIndex = allocateConstantPoolInteger(((ConstantInstruction.ConstInt) instr).getIntValue());
717                 } else if (t.equals(TYPE_String)) {
718                   cpIndex = allocateConstantPoolString((String) ((ConstantInstruction.ConstString) instr).getValue());
719                 } else if (t.equals(TYPE_Class)) {
720                   cpIndex = allocateConstantPoolClassType(((ClassToken) ((ConstantInstruction.ConstClass) instr).getValue()).getTypeName());
721                 } else if (t.equals(TYPE_MethodType)) {
722                   cpIndex = allocateConstantPoolMethodType(((String) ((ConstantInstruction.ConstMethodType) instr).getValue()));
723                 } else if (t.equals(TYPE_MethodHandle)) {
724                   cpIndex = allocateConstantPoolMethodHandle(((ReferenceToken) ((ConstantInstruction.ConstMethodHandle) instr).getValue()));
725                 } else {
726                   cpIndex = allocateConstantPoolFloat(((ConstantInstruction.ConstFloat) instr).getFloatValue());
727                 }
728               }
729     
730               if (cpIndex < 256) {
731                 code[curOffset - 1] = (byte) OP_ldc;
732                 code[curOffset] = (byte) cpIndex;
733                 curOffset++;
734               } else {
735                 writeShort(curOffset, cpIndex);
736                 curOffset += 2;
737               }
738               break;
739             }
740             case OP_ldc2_w: {
741               int cpIndex;
742               ConstantInstruction ci = (ConstantInstruction) instr;
743               if (presetConstants != null && ci.getLazyConstantPool() == presetConstants) {
744                 cpIndex = ci.getCPIndex();
745               } else {
746                 String t = instr.getPushedType(null);
747                 if (t.equals(TYPE_long)) {
748                   cpIndex = allocateConstantPoolLong(((ConstantInstruction.ConstLong) instr).getLongValue());
749                 } else {
750                   cpIndex = allocateConstantPoolDouble(((ConstantInstruction.ConstDouble) instr).getDoubleValue());
751                 }
752               }
753     
754               writeShort(curOffset, cpIndex);
755               curOffset += 2;
756               break;
757             }
758             case OP_iload_0:
759             case OP_iload_1:
760             case OP_iload_2:
761             case OP_iload_3:
762             case OP_iload: {
763               if (inBasicBlock(i, 4)) {
764                 // try to generate an OP_iinc
765                 if (instructions[i + 1] instanceof ConstantInstruction.ConstInt && instructions[i + 2] instanceof BinaryOpInstruction
766                     && instructions[i + 3] instanceof StoreInstruction) {
767                   LoadInstruction i0 = (LoadInstruction) instr;
768                   ConstantInstruction.ConstInt i1 = (ConstantInstruction.ConstInt) instructions[i + 1];
769                   BinaryOpInstruction i2 = (BinaryOpInstruction) instructions[i + 2];
770                   StoreInstruction i3 = (StoreInstruction) instructions[i + 3];
771     
772                   int c = i1.getIntValue();
773                   int v = i0.getVarIndex();
774                   BinaryOpInstruction.Operator op = i2.getOperator();
775                   if ((short) c == c && i3.getVarIndex() == v && (op == Operator.ADD || op == Operator.SUB)
776                       && i2.getType().equals(TYPE_int) && i3.getType().equals(TYPE_int)) {
777                     if (v < 256 && (byte) c == c) {
778                       code[curOffset - 1] = (byte) OP_iinc;
779                       writeByte(curOffset, v);
780                       writeByte(curOffset + 1, c);
781                       curOffset += 2;
782                     } else {
783                       code[curOffset - 1] = (byte) OP_wide;
784                       code[curOffset] = (byte) OP_iinc;
785                       writeShort(curOffset + 1, v);
786                       writeShort(curOffset + 3, c);
787                       curOffset += 5;
788                     }
789                     instructionsToOffsets[i + 1] = -1;
790                     instructionsToOffsets[i + 2] = -1;
791                     instructionsToOffsets[i + 3] = -1;
792                     i += 3;
793                     break;
794                   }
795                 }
796               }
797               if (opcode != OP_iload) {
798                 break;
799               }
800             }
801             case OP_lload:
802             case OP_fload:
803             case OP_dload:
804             case OP_aload: {
805               int v = ((LoadInstruction) instr).getVarIndex();
806     
807               if (v < 256) {
808                 writeByte(curOffset, v);
809                 curOffset++;
810               } else {
811                 code[curOffset - 1] = (byte) OP_wide;
812                 code[curOffset] = (byte) opcode;
813                 writeShort(curOffset + 1, v);
814                 curOffset += 3;
815               }
816               break;
817             }
818             case OP_istore:
819             case OP_lstore:
820             case OP_fstore:
821             case OP_dstore:
822             case OP_astore: {
823               int v = ((StoreInstruction) instr).getVarIndex();
824     
825               if (v < 256) {
826                 writeByte(curOffset, v);
827                 curOffset++;
828               } else {
829                 code[curOffset - 1] = (byte) OP_wide;
830                 code[curOffset] = (byte) opcode;
831                 writeShort(curOffset + 1, v);
832                 curOffset += 3;
833               }
834               break;
835             }
836             case OP_goto: {
837               int targetI = instr.getBranchTargets()[0];
838               if (targetI <= i) {
839                 int delta = instructionsToOffsets[targetI] - (curOffset - 1);
840                 if ((short) delta != delta) {
841                   code[curOffset - 1] = (byte) OP_goto_w;
842                   writeInt(curOffset, delta);
843                   curOffset += 4;
844                 } else {
845                   writeShort(curOffset, (short) delta);
846                   curOffset += 2;
847                 }
848               } else if (targetI == i + 1) {
849                 // ignore noop gotos
850                 curOffset--;
851               } else {
852                 Patch p;
853                 if (farBranches) {
854                   code[curOffset - 1] = (byte) OP_goto_w;
855                   p = new IntPatch(curOffset - 1, curOffset, instr.getBranchTargets()[0]);
856                   curOffset += 4;
857                 } else {
858                   p = new ShortPatch(curOffset - 1, curOffset, instr.getBranchTargets()[0]);
859                   curOffset += 2;
860                 }
861                 patches.add(p);
862               }
863               break;
864             }
865             case OP_lookupswitch: {
866               int start = curOffset - 1;
867               SwitchInstruction sw = (SwitchInstruction) instr;
868               int[] casesAndLabels = sw.getCasesAndLabels();
869     
870               while ((curOffset & 3) != 0) {
871                 writeByte(curOffset, 0);
872                 curOffset++;
873               }
874     
875               if (curOffset + 4 * casesAndLabels.length + 8 > code.length) {
876                 return false;
877               }
878               insertBranchOffsetInt(patches, start, curOffset, sw.getDefaultLabel());
879               writeInt(curOffset + 4, casesAndLabels.length / 2);
880               curOffset += 8;
881               for (int j = 0; j < casesAndLabels.length; j += 2) {
882                 writeInt(curOffset, casesAndLabels[j]);
883                 insertBranchOffsetInt(patches, start, curOffset + 4, casesAndLabels[j + 1]);
884                 curOffset += 8;
885               }
886               break;
887             }
888             case OP_tableswitch: {
889               int start = curOffset - 1;
890               SwitchInstruction sw = (SwitchInstruction) instr;
891               int[] casesAndLabels = sw.getCasesAndLabels();
892     
893               while ((curOffset & 3) != 0) {
894                 writeByte(curOffset, 0);
895                 curOffset++;
896               }
897               if (curOffset + 2 * casesAndLabels.length + 12 > code.length) {
898                 return false;
899               }
900               insertBranchOffsetInt(patches, start, curOffset, sw.getDefaultLabel());
901               writeInt(curOffset + 4, casesAndLabels[0]);
902               writeInt(curOffset + 8, casesAndLabels[casesAndLabels.length - 2]);
903               curOffset += 12;
904               for (int j = 0; j < casesAndLabels.length; j += 2) {
905                 insertBranchOffsetInt(patches, start, curOffset, casesAndLabels[j + 1]);
906                 curOffset += 4;
907               }
908               break;
909             }
910             case OP_getfield:
911             case OP_getstatic: {
912               GetInstruction g = (GetInstruction) instr;
913               int cpIndex;
914     
915               if (presetConstants != null && presetConstants == g.getLazyConstantPool()) {
916                 cpIndex = ((GetInstruction.Lazy) g).getCPIndex();
917               } else {
918                 cpIndex = allocateConstantPoolField(g.getClassType(), g.getFieldName(), g.getFieldType());
919               }
920               writeShort(curOffset, cpIndex);
921               curOffset += 2;
922               break;
923             }
924             case OP_putfield:
925             case OP_putstatic: {
926               PutInstruction p = (PutInstruction) instr;
927               int cpIndex;
928     
929               if (presetConstants != null && presetConstants == p.getLazyConstantPool()) {
930                 cpIndex = ((PutInstruction.Lazy) p).getCPIndex();
931               } else {
932                 cpIndex = allocateConstantPoolField(p.getClassType(), p.getFieldName(), p.getFieldType());
933               }
934               writeShort(curOffset, cpIndex);
935               curOffset += 2;
936               break;
937             }
938             case OP_invokespecial:
939             case OP_invokestatic:
940             case OP_invokevirtual: {
941               InvokeInstruction inv = (InvokeInstruction) instr;
942               int cpIndex;
943     
944               if (presetConstants != null && presetConstants == inv.getLazyConstantPool()) {
945                 cpIndex = ((InvokeInstruction.Lazy) inv).getCPIndex();
946               } else {
947                 cpIndex = allocateConstantPoolMethod(inv.getClassType(), inv.getMethodName(), inv.getMethodSignature());
948               }
949               writeShort(curOffset, cpIndex);
950               curOffset += 2;
951               break;
952             }
953             case OP_invokedynamic: {
954               InvokeDynamicInstruction inv = (InvokeDynamicInstruction) instr;
955               String sig = inv.getMethodSignature();
956               int cpIndex;
957     
958               if (presetConstants != null && presetConstants == inv.getLazyConstantPool()) {
959                 cpIndex = ((InvokeDynamicInstruction.Lazy) inv).getCPIndex();
960               } else {
961                 cpIndex = allocateConstantPoolInterfaceMethod(inv.getClassType(), inv.getMethodName(), sig);
962               }
963               writeShort(curOffset, cpIndex);
964               code[curOffset + 2] = 0;
965               code[curOffset + 3] = 0;
966               curOffset += 4;
967               break;
968             }
969             case OP_invokeinterface: {
970               InvokeInstruction inv = (InvokeInstruction) instr;
971               String sig = inv.getMethodSignature();
972               int cpIndex;
973     
974               if (presetConstants != null && presetConstants == inv.getLazyConstantPool()) {
975                 cpIndex = ((InvokeInstruction.Lazy) inv).getCPIndex();
976               } else {
977                 cpIndex = allocateConstantPoolInterfaceMethod(inv.getClassType(), inv.getMethodName(), sig);
978               }
979               writeShort(curOffset, cpIndex);
980               code[curOffset + 2] = (byte) (Util.getParamsWordSize(sig) + 1);
981               code[curOffset + 3] = 0;
982               curOffset += 4;
983               break;
984             }
985            case OP_new:
986               writeShort(curOffset, allocateConstantPoolClassType(((NewInstruction) instr).getType()));
987               curOffset += 2;
988               break;
989             case OP_newarray:
990               code[curOffset] = indexedTypes_T[Util.getTypeIndex(((NewInstruction) instr).getType().substring(1))];
991               curOffset++;
992               break;
993             case OP_anewarray:
994               writeShort(curOffset, allocateConstantPoolClassType(((NewInstruction) instr).getType().substring(1)));
995               curOffset += 2;
996               break;
997             case OP_multianewarray: {
998               NewInstruction n = (NewInstruction) instr;
999               writeShort(curOffset, allocateConstantPoolClassType(n.getType()));
1000               code[curOffset + 2] = (byte) n.getArrayBoundsCount();
1001               curOffset += 3;
1002               break;
1003             }
1004             case OP_checkcast:
1005               writeShort(curOffset, allocateConstantPoolClassType(((CheckCastInstruction) instr).getTypes()[0]));
1006               curOffset += 2;
1007               break;
1008             case OP_instanceof:
1009               writeShort(curOffset, allocateConstantPoolClassType(((InstanceofInstruction) instr).getType()));
1010               curOffset += 2;
1011               break;
1012             }
1013           } else {
1014             stackLenRef[0] = stackLen;
1015             curOffsetRef[0] = curOffset;
1016             instrRef[0] = i;
1017             instr.visit(noOpcodeHandler);
1018             curOffset = curOffsetRef[0];
1019             i = instrRef[0];
1020           }
1021     
1022           boolean haveStack = true;
1023     
1024           while (startI <= i) {
1025             instr = (Instruction) instructions[startI];
1026             if (instr.isFallThrough() && haveStack) {
1027               if (stackLen < instr.getPoppedCount()) {
1028                 throw new IllegalArgumentException("Stack underflow in intermediate code, at offset " + startI);
1029               }
1030     
1031               if (instr instanceof DupInstruction) {
1032                 DupInstruction d = (DupInstruction) instr;
1033                 int size = d.getSize();
1034                 int delta = d.getDelta();
1035     
1036                 System.arraycopy(stackWords, stackLen - size - delta, stackWords, stackLen - delta, delta + size);
1037                 System.arraycopy(stackWords, stackLen, stackWords, stackLen - size - delta, size);
1038                 stackLen += size;
1039               } else if (instr instanceof SwapInstruction) {
1040                 byte b = stackWords[stackLen - 1];
1041                 stackWords[stackLen - 1] = stackWords[stackLen - 2];
1042                 stackWords[stackLen - 2] = b;
1043               } else {
1044                 stackLen -= instr.getPoppedCount();
1045     
1046                 byte w = instr.getPushedWordSize();
1047                 if (w > 0) {
1048                   stackWords[stackLen] = w;
1049                   stackLen++;
1050                 }
1051               }
1052             } else {
1053               // No stack, or the instruction doesn't fall through
1054               // try to grab the stack state at the start of the next instruction
1055               if (startI + 1 < endInstruction) {
1056                 byte[] s = this.stackWords[startI + 1];
1057                 // if the next instruction doesn't have stack info (it's not
1058                 // reachable), then just ignore it and remember that we don't have
1059                 // stack info
1060                 if (s == null) {
1061                   haveStack = false;
1062                 } else {
1063                   stackLen = s.length;
1064                   System.arraycopy(s, 0, stackWords, 0, stackLen);
1065                 }
1066               }
1067             }
1068     
1069             startI++;
1070           }
1071     
1072           if (curOffset > code.length - 8) {
1073             return false;
1074           }
1075     
1076           if (!haveStack) {
1077             // skip forward through the unreachable instructions until we find
1078             // an instruction for which we know the stack state
1079             while (i + 1 < endInstruction) {
1080               byte[] s = this.stackWords[i + 1];
1081               if (s != null) {
1082                 stackLen = s.length;
1083                 System.arraycopy(s, 0, stackWords, 0, stackLen);
1084                 break;
1085               }
1086               i++;
1087             }
1088           }
1089         }
1090     
1091         if (applyPatches(patches)) {
1092           byte[] newCode = new byte[curOffset];
1093           System.arraycopy(code, 0, newCode, 0, curOffset);
1094           releaseCodeBuf(code);
1095           code = newCode;
1096         } else {
1097           if (farBranches) {
1098             throw new Error("Failed to apply patches even with farBranches on");
1099           } else {
1100             return outputInstructions(startInstruction, endInstruction, startOffset, true, initialStack);
1101           }
1102         }
1103     
1104         return true;
1105       }

1107       private int[] buildRawHandlers(int start, int end) {
1108         int[] handlerCounts = new int[end - start];
1109         int maxCount = 0;
1110     
1111         for (int i = start; i < end; i++) {
1112           int len = handlers[i].length;
1113           handlerCounts[i - start] = len;
1114           if (len > maxCount) {
1115             maxCount = len;
1116           }
1117         }
1118     
1119         if (maxCount == 0) {
1120           return noRawHandlers;
1121         } else {
1122           ArrayList<int[]> rawHandlerList = new ArrayList<int[]>();
1123     
1124           for (int i = maxCount; i > 0; i--) {
1125             for (int j = start; j < end; j++) {
1126               if (handlerCounts[j - start] == i) {
1127                 int first = j;
1128                 ExceptionHandler h = handlers[j][handlers[j].length - i];
1129     
1130                 do {
1131                   handlerCounts[j - start]--;
1132                   j++;
1133                 } while (j < end && handlerCounts[j - start] == i && handlers[j][handlers[j].length - i].equals(h));
1134     
1135                 if (h.handler >= start && h.handler < end) {
1136                   rawHandlerList.add(new int[] { instructionsToOffsets[first], j < end ? instructionsToOffsets[j] : code.length,
1137                       instructionsToOffsets[h.handler], h.catchClass == null ? 0 : allocateConstantPoolClassType(h.catchClass) });
1138                 }
1139     
1140                 j--;
1141               }
1142             }
1143           }
1144     
1145           int[] rawHandlers = new int[4 * rawHandlerList.size()];
1146           int count = 0;
1147           for (Iterator<int[]> iter = rawHandlerList.iterator(); iter.hasNext();) {
1148             int[] element = iter.next();
1149             System.arraycopy(element, 0, rawHandlers, count, 4);
1150             count += 4;
1151           }
1152           return rawHandlers;
1153         }
1154       }

1190       private void addBackEdge(int from, int to) {
1191         int[] oldEdges = backEdges[from];
1192         if (oldEdges == null) {
1193           backEdges[from] = new int[] { to };
1194         } else if (oldEdges[oldEdges.length - 1] < 0) {
1195           int left = 1;
1196           int right = oldEdges.length - 1;
1197           while (true) {
1198             if (right - left < 2) {
1199               if (oldEdges[left] < 0) {
1200                 break;
1201               } else {
1202                 if (oldEdges[right] >= 0)
1203                   throw new Error("Failed binary search");
1204                 left = right;
1205                 break;
1206               }
1207             } else {
1208               int mid = (left + right) / 2;
1209               if (oldEdges[mid] < 0) {
1210                 right = mid;
1211               } else {
1212                 left = mid + 1;
1213               }
1214             }
1215           }
1216           oldEdges[left] = to;
1217         } else {
1218           int[] newEdges = new int[oldEdges.length * 2];
1219           System.arraycopy(oldEdges, 0, newEdges, 0, oldEdges.length);
1220           newEdges[oldEdges.length] = to;
1221           for (int i = oldEdges.length + 1; i < newEdges.length; i++) {
1222             newEdges[i] = -1;
1223           }
1224           backEdges[from] = newEdges;
1225         }
1226       }

1228       private void addLiveVar(int instruction, int index) {
1229         while (true) {
1230           if (liveLocals[instruction].get(index)) {
1231             break;
1232           }
1233     
1234           IInstruction instr = instructions[instruction];
1235           if (instr instanceof StoreInstruction && ((StoreInstruction) instr).getVarIndex() == index) {
1236             break;
1237           }
1238     
1239           liveLocals[instruction].set(index);
1240           int[] back = backEdges[instruction];
1241           if (back != null) {
1242             for (int i = 0; i < back.length; i++) {
1243               addLiveVar(back[i], index);
1244             }
1245           }
1246     
1247           if (instruction > 0 && instructions[instruction - 1].isFallThrough()) {
1248             instruction--;
1249           } else {
1250             break;
1251           }
1252         }
1253       }

1255       private void makeLiveLocals() {
1256         liveLocals = new BitSet[instructions.length];
1257         backEdges = new int[instructions.length][];
1258     
1259         for (int i = 0; i < instructions.length; i++) {
1260           IInstruction instr = instructions[i];
1261           int[] targets = instr.getBranchTargets();
1262           for (int j = 0; j < targets.length; j++) {
1263             addBackEdge(targets[j], i);
1264           }
1265           ExceptionHandler[] hs = handlers[i];
1266           for (int j = 0; j < hs.length; j++) {
1267             addBackEdge(hs[j].handler, i);
1268           }
1269           liveLocals[i] = new BitSet();
1270         }
1271     
1272         for (int i = 0; i < backEdges.length; i++) {
1273           int[] back = backEdges[i];
1274           if (back != null && back[back.length - 1] < 0) {
1275             int j = back.length;
1276             while (back[j - 1] < 0) {
1277               j--;
1278             }
1279             int[] newBack = new int[j];
1280             System.arraycopy(back, 0, newBack, 0, newBack.length);
1281             backEdges[i] = newBack;
1282           }
1283         }
1284     
1285         for (int i = 0; i < instructions.length; i++) {
1286           IInstruction instr = instructions[i];
1287           if (instr instanceof LoadInstruction) {
1288             addLiveVar(i, ((LoadInstruction) instr).getVarIndex());
1289           }
1290         }
1291       }

1311       private HelperPatch makeHelperPatch(int start, int len, int retVar, int unreadStack, int untouchedStack) {
1312         String retType = retVar >= 0 ? getAndCheckLocalType(start + len, retVar) : "V";
1313     
1314         ArrayList<Instruction> callWrapper = new ArrayList<Instruction>();
1315         int curStackLen = stackTypes[start].length;
1316     
1317         StringBuffer sigBuf = new StringBuffer();
1318         sigBuf.append("(");
1319         // spill needed stack variables to allocated locals;
1320         allocateLocals(curStackLen - unreadStack);
1321         for (int i = curStackLen - 1; i >= unreadStack; i--) {
1322           if (i < untouchedStack) {
1323             callWrapper.add(DupInstruction.make(0));
1324           }
1325           callWrapper.add(StoreInstruction.make(stackTypes[start][i], allocatedLocals + 2 * (i - unreadStack)));
1326         }
1327         // push needed locals
1328         BitSet liveVars = liveLocals[start];
1329         for (int i = 0; i < liveVars.length(); i++) {
1330           if (liveVars.get(i)) {
1331             String t = getAndCheckLocalType(start, i);
1332             sigBuf.append(t);
1333             callWrapper.add(LoadInstruction.make(t, i));
1334             if (Util.getWordSize(t) > 1) {
1335               i++;
1336             }
1337           } else {
1338             // dummy
1339             sigBuf.append("I");
1340             callWrapper.add(ConstantInstruction.make(0));
1341           }
1342         }
1343         // push stack variables
1344         for (int i = unreadStack; i < curStackLen; i++) {
1345           callWrapper.add(LoadInstruction.make(stackTypes[start][i], allocatedLocals + 2 * (i - unreadStack)));
1346           sigBuf.append(stackTypes[start][i]);
1347           if (Util.getWordSize(stackTypes[start][i]) == 2) {
1348             sigBuf.append("I");
1349             callWrapper.add(ConstantInstruction.make(0));
1350           }
1351         }
1352         sigBuf.append(")");
1353         sigBuf.append(retType);
1354         String sig = sigBuf.toString();
1355     
1356         String name = createHelperMethod(true, sig);
1357     
1358         callWrapper.add(InvokeInstruction.make(sig, classType, name, IInvokeInstruction.Dispatch.STATIC));
1359     
1360         int savedMaxStack = maxStack;
1361         maxStack += curStackLen - unreadStack;
1362     
1363         int prefixLength = 4 * (curStackLen - unreadStack);
1364         byte[] initialStack = new byte[curStackLen - unreadStack];
1365         for (int i = 0; i < initialStack.length; i++) {
1366           initialStack[i] = Util.getWordSize(stackTypes[start][unreadStack + i]);
1367         }
1368         if (!outputInstructions(start, start + len, prefixLength, false, initialStack)) {
1369           throw new Error("Helper function is overlarge");
1370         }
1371         byte[] newCode = new byte[code.length + (retVar >= 0 ? 5 : 1)];
1372         for (int i = 0; i < curStackLen - unreadStack; i++) {
1373           int local = allocatedLocals + i * 2;
1374           newCode[i * 4] = (byte) OP_wide;
1375           newCode[i * 4 + 1] = (byte) LoadInstruction.make(stackTypes[start][i + unreadStack], 500).getOpcode();
1376           newCode[i * 4 + 2] = (byte) (local >> 8);
1377           newCode[i * 4 + 3] = (byte) local;
1378         }
1379         System.arraycopy(code, prefixLength, newCode, prefixLength, code.length - prefixLength);
1380         int suffixOffset = code.length;
1381         if (retVar >= 0) {
1382           newCode[suffixOffset] = (byte) OP_wide;
1383           newCode[suffixOffset + 1] = (byte) LoadInstruction.make(retType, 500).getOpcode();
1384           newCode[suffixOffset + 2] = (byte) (retVar >> 8);
1385           newCode[suffixOffset + 3] = (byte) retVar;
1386           newCode[suffixOffset + 4] = (byte) ReturnInstruction.make(retType).getOpcode();
1387           callWrapper.add(StoreInstruction.make(retType, retVar));
1388         } else {
1389           newCode[suffixOffset] = (byte) ReturnInstruction.make(TYPE_void).getOpcode();
1390         }
1391     
1392         if (callWrapper.size() > len) {
1393           return null;
1394         }
1395     
1396         int[] rawHandlers = buildRawHandlers(start, start + len);
1397         int[] bytecodeMap = buildBytecodeMap(start, start + len);
1398     
1399         auxMethods.add(new Output(name, sig, newCode, rawHandlers, bytecodeMap, maxLocals, maxStack, true, null));
1400     
1401         maxStack = savedMaxStack;
1402     
1403         Instruction[] patch = new Instruction[callWrapper.size()];
1404         callWrapper.toArray(patch);
1405     
1406         ExceptionHandler[] startHS = handlers[start];
1407         ArrayList<ExceptionHandler> newHS = new ArrayList<ExceptionHandler>();
1408         for (int i = 0; i < startHS.length; i++) {
1409           int t = startHS[i].handler;
1410           if (t < start || t >= start + len) {
1411             newHS.add(startHS[i]);
1412           }
1413         }
1414         ExceptionHandler[] patchHS = new ExceptionHandler[newHS.size()];
1415         newHS.toArray(patchHS);
1416     
1417         return new HelperPatch(start, len, patch, patchHS);
1418       }

1420       private HelperPatch findBlock(int start, int len) {
1421         while (len > 100) {
1422           // make sure there is at most one entry
1423           int lastInvalid = start - 1;
1424           for (int i = start + 1; i < start + len; i++) {
1425             int[] back = backEdges[i];
1426             boolean outsideBranch = false;
1427             for (int j = 0; back != null && j < back.length; j++) {
1428               if (back[j] < start || back[j] >= start + len) {
1429                 outsideBranch = true;
1430               }
1431             }
1432             if (outsideBranch) {
1433               HelperPatch p = findBlock(lastInvalid + 1, i - lastInvalid - 1);
1434               if (p != null) {
1435                 return p;
1436               }
1437               lastInvalid = i;
1438             }
1439           }
1440           if (lastInvalid >= start) {
1441             return null;
1442           }
1443     
1444           // make sure there is at most one exit (fall through at the end)
1445           if (!instructions[start + len - 1].isFallThrough()) {
1446             len--;
1447             continue;
1448           }
1449           lastInvalid = start - 1;
1450           for (int i = start; i < start + len; i++) {
1451             int[] targets = instructions[i].getBranchTargets();
1452             boolean outsideBranch = false;
1453             if (instructions[i] instanceof ReturnInstruction) {
1454               outsideBranch = true;
1455             }
1456             for (int j = 0; j < targets.length; j++) {
1457               if (targets[j] < start || targets[j] >= start + len) {
1458                 outsideBranch = true;
1459               }
1460             }
1461             if (outsideBranch) {
1462               HelperPatch p = findBlock(lastInvalid + 1, i - lastInvalid - 1);
1463               if (p != null) {
1464                 return p;
1465               }
1466               lastInvalid = i;
1467             }
1468           }
1469           if (lastInvalid >= start) {
1470             return null;
1471           }
1472     
1473           lastInvalid = start - 1;
1474           for (int i = start; i < start + len; i++) {
1475             boolean out = false;
1476             ExceptionHandler[] hs = handlers[i];
1477             for (int j = 0; j < hs.length; j++) {
1478               int h = hs[j].handler;
1479               if (h < start || h >= start + len) {
1480                 out = true;
1481               }
1482             }
1483             int[] targets = instructions[i].getBranchTargets();
1484             for (int j = 0; j < targets.length; j++) {
1485               int t = targets[j];
1486               if (t < start || t >= start + len) {
1487                 out = true;
1488               }
1489             }
1490             if (out) {
1491               HelperPatch p = findBlock(lastInvalid + 1, i - lastInvalid - 1);
1492               if (p != null) {
1493                 return p;
1494               }
1495               lastInvalid = i;
1496             }
1497           }
1498     
1499           if (lastInvalid >= start) {
1500             return null;
1501           }
1502     
1503           if (stackTypes[start] == null) {
1504             while (stackTypes[start] == null && len > 0) {
1505               start++;
1506               len--;
1507             }
1508             continue;
1509           }
1510     
1511           // See how many stack elements at entry are still there, unchanged, at
1512           // exit
1513           int untouchedStack = Integer.MAX_VALUE;
1514           // See how many elements of that part of the stack are never even read
1515           int unreadStack = Integer.MAX_VALUE;
1516           for (int i = start; i < start + len; i++) {
1517             if (stackTypes[i] == null) {
1518               untouchedStack = 0;
1519               unreadStack = 0;
1520               break;
1521             }
1522             int lowWaterMark = stackTypes[i].length - instructions[i].getPoppedCount();
1523             unreadStack = Math.min(unreadStack, lowWaterMark);
1524             if (instructions[i] instanceof DupInstruction) {
1525               // dup instructions don't actually pop off/change the element they
1526               // duplicate
1527               lowWaterMark += instructions[i].getPoppedCount();
1528             }
1529             untouchedStack = Math.min(untouchedStack, lowWaterMark);
1530           }
1531     
1532           if (untouchedStack > unreadStack + 1 || (untouchedStack == unreadStack + 1 && untouchedStack < stackTypes[start].length)) {
1533             // we can only handle 1 read-but-untouched element
1534             start++;
1535             len--;
1536             continue;
1537           }
1538     
1539           // make sure we know the type of all the stack values that must be passed
1540           // in
1541           boolean unknownType = false;
1542           for (int i = unreadStack; i < untouchedStack; i++) {
1543             String t = stackTypes[start][i];
1544             if (t == null || t.equals(TYPE_unknown) || t.equals(TYPE_null)) {
1545               unknownType = true;
1546               break;
1547             }
1548           }
1549           if (unknownType) {
1550             start++;
1551             len--;
1552             continue;
1553           }
1554     
1555           // make sure outgoing stack size is no more than the untouched stack size.
1556           if (stackTypes[start + len] == null || stackTypes[start + len].length > untouchedStack) {
1557             // This is a little conservative. We might be able to stop sooner with a
1558             // valid
1559             // extractable method, because changing 'len' might mean we have more
1560             // untouched stack elements. But we'll do this in a dumb way to avoid
1561             // being caught in some N^2 loop looking for extractable code.
1562             while (len > 0 && (stackTypes[start + len] == null || stackTypes[start + len].length > untouchedStack)) {
1563               len--;
1564             }
1565             continue;
1566           }
1567     
1568           // make sure at most one local is defined and live on exit
1569           BitSet liveAtEnd = liveLocals[start + len];
1570           boolean multipleDefs = false;
1571           int localDefed = -1;
1572           int firstDef = -1;
1573           int secondDef = -1;
1574           for (int i = start; i < start + len; i++) {
1575             IInstruction instr = instructions[i];
1576             if (instr instanceof StoreInstruction) {
1577               int l = ((StoreInstruction) instr).getVarIndex();
1578               if (liveAtEnd.get(l) && l != localDefed) {
1579                 if (localDefed < 0) {
1580                   localDefed = l;
1581                   firstDef = i;
1582                 } else {
1583                   multipleDefs = true;
1584                   secondDef = i;
1585                   break;
1586                 }
1587               }
1588             }
1589           }
1590     
1591           if (multipleDefs) {
1592             HelperPatch p = findBlock(start, secondDef - start);
1593             if (p != null) {
1594               return p;
1595             }
1596             len = (start + len) - (firstDef + 1);
1597             start = firstDef + 1;
1598             continue;
1599           }
1600     
1601           // make sure that the same external handlers are used all the way through
1602           ExceptionHandler[] startHS = handlers[start];
1603           int numOuts = 0;
1604           for (int j = 0; j < startHS.length; j++) {
1605             int t = startHS[j].handler;
1606             if (t < start || t >= start + len) {
1607               numOuts++;
1608             }
1609           }
1610           boolean mismatchedHandlers = false;
1611           int firstMismatch = -1;
1612           for (int i = start + 1; i < start + len; i++) {
1613             ExceptionHandler[] hs = handlers[i];
1614             int matchingOuts = 0;
1615             for (int j = 0; j < hs.length; j++) {
1616               int t = hs[j].handler;
1617               if (t < start || t >= start + len) {
1618                 boolean match = false;
1619                 for (int k = 0; k < startHS.length; k++) {
1620                   if (startHS[k].equals(hs[j])) {
1621                     match = true;
1622                     break;
1623                   }
1624                 }
1625                 if (match) {
1626                   matchingOuts++;
1627                 }
1628               }
1629             }
1630             if (matchingOuts != numOuts) {
1631               firstMismatch = i;
1632               mismatchedHandlers = true;
1633               break;
1634             }
1635           }
1636           if (mismatchedHandlers) {
1637             HelperPatch p = findBlock(start, firstMismatch - start);
1638             if (p != null) {
1639               return p;
1640             }
1641             start = firstMismatch;
1642             continue;
1643           }
1644     
1645           // all conditions satisfied, extract the code
1646           try {
1647             HelperPatch p = makeHelperPatch(start, len, localDefed, unreadStack, untouchedStack);
1648             if (p == null) {
1649               // something went wrong. Probably the code to call the helper ended up
1650               // being
1651               // bigger than the original code we extracted!
1652               return null;
1653             } else {
1654               return p;
1655             }
1656           } catch (IllegalArgumentException ex) {
1657             return null;
1658           }
1659         }
1660     
1661         return null;
1662       }

1664       private void makeHelpers() {
1665         int offset = 0;
1666         ArrayList<HelperPatch> patches = new ArrayList<HelperPatch>();
1667     
1668         while (offset + 5000 < instructions.length) {
1669           HelperPatch p = findBlock(offset, 5000);
1670           if (p != null) {
1671             patches.add(p);
1672             offset = p.start + p.length;
1673           } else {
1674             offset += 500;
1675           }
1676         }
1677     
1678         for (Iterator<HelperPatch> i = patches.iterator(); i.hasNext();) {
1679           HelperPatch p = i.next();
1680     
1681           System.arraycopy(p.code, 0, instructions, p.start, p.code.length);
1682           for (int j = 0; j < p.length; j++) {
1683             int index = j + p.start;
1684             if (j < p.code.length) {
1685               instructions[index] = p.code[j];
1686             } else {
1687               instructions[index] = PopInstruction.make(0); // nop
1688             }
1689             handlers[index] = p.handlers;
1690             instructionsToBytecodes[index] = -1;
1691           }
1692         }
1693       }

1696         Verifier v = new Verifier(isConstructor, isStatic, classType, signature, instructions, handlers, instructionsToBytecodes, null);

1700         try {
1701           v.computeTypes();
1702         } catch (Verifier.FailureException ex) {

1709       /**
1710        * Do the work of generating new bytecodes.
1711        * 
1712        * In pathological cases this could throw an Error, when the code you passed in is too large to fit into a single JVM method and
1713        * Compiler can't find a way to break it up into helper methods. You probably won't encounter this unless you try to make it
1714        * happen :-).
1715        */
1716       final public void compile() {
1717         collectInstructionInfo();
1718     
1719         computeStackWords();
1720     
1721         if (!outputInstructions(0, instructions.length, 0, false, null)) {
1722           allocatedLocals = maxLocals;
1723           makeLiveLocals();
1724           makeTypes();
1725           auxMethods = new ArrayList<Output>();
1726           makeHelpers();
1727     
1728           computeStackWords();
1729           if (!outputInstructions(0, instructions.length, 0, false, null)) {
1730             throw new Error("Input code too large; consider breaking up your code");
1731           }
1732         }
1733         mainMethod = new Output(null, null, code, buildRawHandlers(0, instructions.length), buildBytecodeMap(0, instructions.length),
1734             maxLocals, maxStack, isStatic, instructionsToOffsets);
1735     
1736         instructionsToOffsets = null;
1737         branchTargets = null;
1738         stackWords = null;
1739         code = null;
1740       }
