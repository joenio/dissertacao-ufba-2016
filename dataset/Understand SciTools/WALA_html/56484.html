37     /**
38      * A helper class which can modify a {@link PropagationCallGraphBuilder} to deal with reflective factory methods.
39      */
40     public class ReflectionHandler {
41       private final static boolean VERBOSE = false;
42     
43       private final PropagationCallGraphBuilder builder;
44     
45       public ReflectionHandler(PropagationCallGraphBuilder builder) {
46         this.builder = builder;
47       }
48     
49       /**
50        * update the pointer analysis solver based on flow of reflective factory results to checkcasts
51        * @param monitor 
52        * 
53        * @return true if anything has changed
54        * @throws CancelException
55        * @throws IllegalArgumentException
56        */
57       protected boolean updateForReflection(IProgressMonitor monitor) throws IllegalArgumentException, CancelException {
58     
59         Collection<Statement> returnStatements = computeFactoryReturnStatements();
60         Set<CGNode> changedNodes = HashSetFactory.make();
61         for (Statement st : returnStatements) {
62           if (VERBOSE) {
63             System.err.println("Slice " + st);
64           }
65           Collection<Statement> slice = Slicer.computeForwardSlice(st, builder.callGraph, null, DataDependenceOptions.REFLECTION,
66               ControlDependenceOptions.NONE);
67           if (VERBOSE) {
68             for (Statement x : slice) {
69               System.err.println(" " + x);
70             }
71           }
72           Predicate<Statement> f = new Predicate<Statement>() {
73             @Override
74             public boolean test(Statement s) {
75               if (s.getKind() == Kind.NORMAL) {
76                 return ((NormalStatement) s).getInstruction() instanceof SSACheckCastInstruction;
77               } else {
78                 return false;
79               }
80             }
81           };
82           Collection<Statement> casts = Predicate.filter(slice.iterator(), f);
83           changedNodes.addAll(modifyFactoryInterpreter(st, casts, builder.getContextInterpreter(), builder.getClassHierarchy()));
84         }
85         for (Iterator<CGNode> it = changedNodes.iterator(); it.hasNext();) {
86           builder.addConstraintsFromChangedNode(it.next(), monitor);
87         }
88         return changedNodes.size() > 0;
89     
90       }
91     
92       private Collection<Statement> computeFactoryReturnStatements() {
93         // todo: clean up logic with inheritance, delegation.
94         HashSet<Statement> result = HashSetFactory.make();
95         for (Iterator it = builder.getCallGraph().iterator(); it.hasNext();) {
96           CGNode n = (CGNode) it.next();
97           if (n.getMethod() instanceof SyntheticMethod) {
98             SyntheticMethod m = (SyntheticMethod) n.getMethod();
99             if (m.isFactoryMethod()) {
100               result.add(new NormalReturnCallee(n));
101             }
102           }
103         }
104         return result;
105       }
106     
107       /**
108        * modify the contextInterpreter to account for new interpretations of factory methods.
109        * 
110        * @return set of nodes whose interpretation has changed.
111        */
112       private Set<CGNode> modifyFactoryInterpreter(Statement returnStatement, Collection<Statement> casts,
113           RTAContextInterpreter contextInterpreter, IClassHierarchy cha) {
114         HashSet<CGNode> result = HashSetFactory.make();
115     
116         for (Statement st : casts) {
117           SSACheckCastInstruction c = (SSACheckCastInstruction) ((NormalStatement) st).getInstruction();
118     
119           for (TypeReference type : c.getDeclaredResultTypes()) {
120             IClass klass = cha.lookupClass(type);
121             if (klass != null) {
122               if (contextInterpreter.recordFactoryType(returnStatement.getNode(), klass)) {
123                 result.add(returnStatement.getNode());
124               }
125             }
126           }
127         }
128         return result;
129       }
130     }
