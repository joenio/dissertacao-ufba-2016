12     package com.ibm.wala.cfg.exc.inter;
13     
14     import java.util.Map;
15     
16     import com.ibm.wala.cfg.exc.intra.MethodState;
17     import com.ibm.wala.ipa.callgraph.CGNode;
18     import com.ibm.wala.ipa.callgraph.CallGraph;
19     import com.ibm.wala.ssa.SSAAbstractInvokeInstruction;
20     
21     /**
22      * A MethodState for the interprocedural analysis.
23      * 
24      * This class has been developed as part of a student project "Studienarbeit" by Markus Herhoffer.
25      * It has been adapted and integrated into the WALA project by Juergen Graf.
26      * 
27      * @author Markus Herhoffer <markus.herhoffer@student.kit.edu>
28      * @author Juergen Graf <graf@kit.edu>
29      * 
30      */
31     class InterprocMethodState extends MethodState {
32     
33       private final Map<CGNode, IntraprocAnalysisState> map;
34       private final CGNode method;
35       private final CallGraph cg;
36     
37       InterprocMethodState(final CGNode method, final CallGraph cg, final Map<CGNode, IntraprocAnalysisState> map) {
38         this.map = map;
39         this.method = method;
40         this.cg = cg;
41       }
42     
43       /*
44        * (non-Javadoc)
45        * 
46        * @see edu.kit.ipd.wala.intra.MethodState#throwsException(com.ibm.wala.ipa.callgraph.CGNode)
47        */
48       @Override
49       public boolean throwsException(final SSAAbstractInvokeInstruction node) {
50         for (final CGNode called : cg.getPossibleTargets(method, node.getCallSite())) {
51           final IntraprocAnalysisState info = map.get(called);
52           
53           if (info == null || info.hasExceptions()) {
54             return true;
55           }
56         }
57     
58         return false;
59       }
60     
61     }
