35     /**
36      * This is a simple example WALA application. It's neither efficient nor concise, but is intended to demonstrate some basic
37      * framework concepts.
38      * 
39      * This application builds a type hierarchy visualizes it with an SWT {@link TreeViewer}.
40      */
41     public class SWTTypeHierarchy {
42       // This example takes one command-line argument, so args[1] should be the "-classpath" parameter
43       final static int CLASSPATH_INDEX = 1;
44     
45       /**
46        * Usage: SWTTypeHierarchy -classpath [classpath]
47        */
48       public static void main(String[] args) {
49         // check that the command-line is kosher
50         validateCommandLine(args);
51         run(args[CLASSPATH_INDEX]);
52       }
53     
54       public static ApplicationWindow run(String classpath) {
55     
56         try {
57           AnalysisScope scope = AnalysisScopeReader.makeJavaBinaryAnalysisScope(classpath, (new FileProvider())
58               .getFile(CallGraphTestUtil.REGRESSION_EXCLUSIONS));
59     
60           // invoke WALA to build a class hierarchy
61           ClassHierarchy cha = ClassHierarchy.make(scope);
62     
63           Graph<IClass> g = typeHierarchy2Graph(cha);
64           g = pruneForAppLoader(g);
65     
66           // create and run the viewer
67           final SWTTreeViewer v = new SWTTreeViewer();
68           v.setGraphInput(g);
69           Collection<IClass> roots = InferGraphRoots.inferRoots(g);
70           if (roots.size() < 1) {
71             System.err.println("PANIC: roots.size()=" + roots.size());
72             System.exit(-1);
73           }
74           v.setRootsInput(roots);
75           v.run();
76           return v.getApplicationWindow();
77     
78         } catch (Exception e) {
79           e.printStackTrace();
80           return null;
81         }
82       }
83     
84       /**
85        * Return a view of an {@link IClassHierarchy} as a {@link Graph}, with edges from classes to immediate subtypes
86        */
87       public static Graph<IClass> typeHierarchy2Graph(IClassHierarchy cha) throws WalaException {
88         Graph<IClass> result = SlowSparseNumberedGraph.make();
89         for (IClass c : cha) {
90           result.addNode(c);
91         }
92         for (IClass c : cha) {
93           for (IClass x : cha.getImmediateSubclasses(c)) {
94             result.addEdge(c, x);
95           }
96           if (c.isInterface()) {
97             for (IClass x : cha.getImplementors(c.getReference())) {
98               result.addEdge(c, x);
99             }
100           }
101         }
102         return result;
103       }
104     
105       /**
106        * Restrict g to nodes from the Application loader
107        */
108       static Graph<IClass> pruneForAppLoader(Graph<IClass> g) throws WalaException {
109         Predicate<IClass> f = new Predicate<IClass>() {
110           @Override public boolean test(IClass c) {
111             return (c.getClassLoader().getReference().equals(ClassLoaderReference.Application));
112           }
113         };
114         return pruneGraph(g, f);
115       }
116     
117       /**
118        * Remove from a graph g any nodes that are not accepted by a {@link Predicate}
119        */
120       public static <T> Graph<T> pruneGraph(Graph<T> g, Predicate<T> f) throws WalaException {
121         Collection<T> slice = GraphSlicer.slice(g, f);
122         return GraphSlicer.prune(g, new CollectionFilter<T>(slice));
123       }

125       /**
126        * Validate that the command-line arguments obey the expected usage.
127        * 
128        * Usage: args[0] : "-classpath" args[1] : String, a ";"-delimited class path
129        * 
130        * @throws UnsupportedOperationException if command-line is malformed.
131        */
132       static void validateCommandLine(String[] args) {
133         if (args.length < 2) {
134           throw new UnsupportedOperationException("must have at least 2 command-line arguments");
135         }
136         if (!args[0].equals("-classpath")) {
137           throw new UnsupportedOperationException("invalid command-line, args[0] should be -classpath, but is " + args[0]);
138         }
139       }
