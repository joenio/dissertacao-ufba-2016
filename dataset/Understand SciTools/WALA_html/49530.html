28     /**
29      * Augments the PaPanel with: 1) global pointer keys at the root level. 2) property catalog key for instance keys.
30      * @author yinnonh
31      *
32      */
33     public class JsPaPanel extends PaPanel {
34     
35     	private static final long serialVersionUID = 1L;
36     
37     	private MutableMapping<List<ObjectPropertyCatalogKey>> instanceKeyIdToObjectPropertyCatalogKey = MutableMapping.<List<ObjectPropertyCatalogKey>> make();
38     	private List<AstGlobalPointerKey> globalsPointerKeys = new ArrayList<AstGlobalPointerKey>();
39     
40     	public JsPaPanel(CallGraph cg, PointerAnalysis<InstanceKey> pa) {
41     		super(cg, pa);
42     		initDataStructures(pa);
43     	}
44     
45     	private void initDataStructures(PointerAnalysis<InstanceKey> pa) {
46     		HeapGraph heapGraph = pa.getHeapGraph();
47     		OrdinalSetMapping<InstanceKey> instanceKeyMapping = pa.getInstanceKeyMapping();
48     		for (Object n : heapGraph){
49     			if (heapGraph.getPredNodeCount(n) == 0){
50     				if (n instanceof PointerKey){
51     					if (n instanceof ObjectPropertyCatalogKey){
52     						ObjectPropertyCatalogKey opck = (ObjectPropertyCatalogKey) n;
53     						InstanceKey instanceKey = opck.getObject();
54     						int instanceKeyId = instanceKeyMapping.getMappedIndex(instanceKey);
55     						mapUsingMutableMapping(instanceKeyIdToObjectPropertyCatalogKey, instanceKeyId, opck);
56     					} else if (n instanceof AstGlobalPointerKey){
57     						globalsPointerKeys.add((AstGlobalPointerKey) n);
58     					}
59     				} else {
60     					System.err.println("Non Pointer key root: " + n);
61     				}
62     			}
63     		}
64     	}
65     
66     	@Override
67       protected List<PointerKey> getPointerKeysUnderInstanceKey(InstanceKey ik) {
68     		List<PointerKey> ret = new ArrayList<PointerKey>();
69     		ret.addAll(super.getPointerKeysUnderInstanceKey(ik));
70     		int ikIndex = pa.getInstanceKeyMapping().getMappedIndex(ik);
71     		ret.addAll(nonNullList(instanceKeyIdToObjectPropertyCatalogKey.getMappedObject(ikIndex)));
72     		return ret;
73     	}
74     
75     	private String cgNodesRoot = "CGNodes";
76     	private String globalsRoot = "Globals";
77     	
78     	@Override
79     	protected List<Object> getRootNodes() {
80     		List<Object> ret = new ArrayList<Object>(2);
81     		ret.add(cgNodesRoot);
82     		ret.add(globalsRoot);
83     		return ret;
84     	}
85     
86     	@Override
87     	protected List<Object> getChildrenFor(Object node) {
88     		List<Object> ret = new ArrayList<Object>();
89     		if (node == cgNodesRoot){
90     	    for (int nodeId = 0 ; nodeId < cg.getNumberOfNodes(); nodeId++){
91     	      CGNode cgNode = cg.getNode(nodeId);
92     	      ret.add(cgNode);
93     	    }
94     		} else if (node == globalsRoot){
95     			ret.addAll(globalsPointerKeys);
96     		} else {
97     			ret.addAll(super.getChildrenFor(node));
98     		}
99     		return ret;
100     	}
101     }
