11     package com.ibm.wala.cast.js.callgraph.fieldbased.flowgraph.vertices;
12     
13     import java.util.Collection;
14     import java.util.Map;
15     
16     import com.ibm.wala.cast.js.ssa.JavaScriptInvoke;
17     import com.ibm.wala.classLoader.CallSiteReference;
18     import com.ibm.wala.classLoader.IClass;
19     import com.ibm.wala.classLoader.IMethod;
20     import com.ibm.wala.types.TypeReference;
21     import com.ibm.wala.util.collections.HashMapFactory;
22     import com.ibm.wala.util.collections.Pair;
23     
24     /**
25      * A vertex factory is associated with a flow graph. It manages its vertex set, making sure that
26      * vertices aren't unnecessarily created twice.
27      *  
28      * @author mschaefer
29      *
30      */
31     public class VertexFactory {
32     	private final Map<Pair<FuncVertex, CallSiteReference>, CallVertex> callVertexCache = HashMapFactory.make();
33     	private final Map<IClass, FuncVertex> funcVertexCache = HashMapFactory.make();
34     	private final Map<Pair<FuncVertex, Integer>, ParamVertex> paramVertexCache = HashMapFactory.make();
35     	private final Map<String, PropVertex> propVertexCache = HashMapFactory.make();
36     	private final Map<FuncVertex, RetVertex> retVertexCache = HashMapFactory.make();
37       private final Map<FuncVertex, ArgVertex> argVertexCache = HashMapFactory.make();
38     	private final Map<Pair<FuncVertex, Integer>, VarVertex> varVertexCache = HashMapFactory.make();
39     	private final Map<Pair<String, String>, LexicalVarVertex> lexicalAccessVertexCache = HashMapFactory.make();
40     	private final Map<Pair<IMethod,Integer>, CreationSiteVertex> creationSites = HashMapFactory.make();
41     	
42     	public CallVertex makeCallVertex(FuncVertex func, JavaScriptInvoke invk) {
43     		CallSiteReference site = invk.getCallSite();
44     		Pair<FuncVertex, CallSiteReference> key = Pair.make(func, site);
45     		CallVertex value = callVertexCache.get(key);
46     		if(value == null)
47     			callVertexCache.put(key, value = new CallVertex(func, site, invk));
48     		return value;
49     	}
50     	
51     	public Iterable<CallVertex> getCallVertices() {
52     		return callVertexCache.values();
53     	}
54     
55     	public CreationSiteVertex makeCreationSiteVertex(IMethod method, int instruction, TypeReference createdType) {
56     	  Pair<IMethod, Integer> key = Pair.make(method, instruction);
57         CreationSiteVertex value = creationSites.get(key);
58     	  if (value == null) {
59     	    creationSites.put(key, value = new CreationSiteVertex(method, instruction, createdType));
60     	  }
61     	  return value;
62     	}
63     	
64     	public Collection<CreationSiteVertex> creationSites() {
65     	  return creationSites.values();
66     	}
67     	
68     	public FuncVertex makeFuncVertex(IClass klass) {
69     		FuncVertex value = funcVertexCache.get(klass);
70     		if(value == null)
71     			funcVertexCache.put(klass, value = new FuncVertex(klass));
72     		return value;
73     	}
74     
75     	public Collection<FuncVertex> getFuncVertices() {
76     	  return funcVertexCache.values();
77     	}
78     
79     	public ParamVertex makeParamVertex(FuncVertex func, int index) {
80     		Pair<FuncVertex, Integer> key = Pair.make(func, index);
81     		ParamVertex value = paramVertexCache.get(key);
82     		if(value == null)
83     			paramVertexCache.put(key, value = new ParamVertex(func, index));
84     		return value;
85     	}
86     
87     	public PropVertex makePropVertex(String name) {
88     		PropVertex value = propVertexCache.get(name);
89     		if(value == null)
90     			propVertexCache.put(name, value = new PropVertex(name));
91     		return value;
92     	}
93     	
94     	public Iterable<PropVertex> getPropVertices() {
95     	  return propVertexCache.values();
96     	}
97     
98     	public RetVertex makeRetVertex(FuncVertex func) {
99     		RetVertex value = retVertexCache.get(func);
100     		if(value == null)
101     			retVertexCache.put(func, value = new RetVertex(func));
102     		return value;
103     	}
104     
105     	public ArgVertex makeArgVertex(FuncVertex func) {
106         ArgVertex value = argVertexCache.get(func);
107         if(value == null)
108           argVertexCache.put(func, value = new ArgVertex(func));
109         return value;
110       }
111     
112     	public UnknownVertex makeUnknownVertex() {
113     		return UnknownVertex.INSTANCE;
114     	}
115     
116     	public VarVertex makeVarVertex(FuncVertex func, int valueNumber) {
117     		Pair<FuncVertex, Integer> key = Pair.make(func, valueNumber);
118     		VarVertex value = varVertexCache.get(key);
119     		if(value == null)
120     			varVertexCache.put(key, value = new VarVertex(func, valueNumber));
121     		return value;
122     	}
123     
124     	public LexicalVarVertex makeLexicalAccessVertex(String definer, String name) {
125     		Pair<String, String> key = Pair.make(definer, name);
126     		LexicalVarVertex value = lexicalAccessVertexCache.get(key);
127     		if(value == null)
128     			lexicalAccessVertexCache.put(key, value = new LexicalVarVertex(definer, name));
129     		return value;
130     	}
131     	
132     	private GlobalVertex global = GlobalVertex.instance();
133     	
134     	public GlobalVertex global() {
135     	  return global;
136     	}
137     }
