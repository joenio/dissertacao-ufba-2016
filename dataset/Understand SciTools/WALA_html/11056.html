39     /**
40      * basic abstract class loader implementation
41      *
42      */
43     public abstract class CAstAbstractLoader implements IClassLoader {
44     
45       /**
46        * types loaded by this
47        */
48       protected final Map<TypeName,IClass> types = HashMapFactory.make();
49     
50       protected final IClassHierarchy cha;
51     
52       protected final IClassLoader parent;
53     
54       /**
55        * warnings generated while loading each module
56        */
57       private final Map<ModuleEntry, Set<Warning>> errors = new HashMap<ModuleEntry, Set<Warning>>();
58       
59       public CAstAbstractLoader(IClassHierarchy cha, IClassLoader parent) {
60         this.cha = cha;
61         this.parent = parent;
62       }
63     
64       public CAstAbstractLoader(IClassHierarchy cha) {
65         this(cha, null);
66       }
67     
68       public void addMessage(ModuleEntry module, Set<Warning> message) {
69         if (! errors.containsKey(module)) {
70           errors.put(module, new HashSet<Warning>());
71         }
72         
73         errors.get(module).addAll(message);
74       }
75     
76       public void addMessage(ModuleEntry module, Warning message) {
77         if (! errors.containsKey(module)) {
78           errors.put(module, new HashSet<Warning>());
79         }
80         
81         errors.get(module).add(message);
82       }
83     
84       private Iterator<ModuleEntry> getMessages(final byte severity) {
85         return new MapIterator<Map.Entry<ModuleEntry,Set<Warning>>, ModuleEntry>(new FilterIterator<Map.Entry<ModuleEntry,Set<Warning>>>(errors.entrySet().iterator(), new Predicate<Map.Entry<ModuleEntry,Set<Warning>>>()  {
86           @Override public boolean test(Entry<ModuleEntry, Set<Warning>> o) {
87              for(Warning w : o.getValue()) {
88                if (w.getLevel() == severity) {
89                  return true;
90                }
91              }
92              return false;
93           }
94         }), new Function<Map.Entry<ModuleEntry,Set<Warning>>, ModuleEntry>() {
95           @Override
96           public ModuleEntry apply(Entry<ModuleEntry, Set<Warning>> object) {
97             return object.getKey();
98           }      
99         });
100       }
101       
102       public Iterator<ModuleEntry> getModulesWithParseErrors() {
103          return getMessages(Warning.SEVERE);
104       }
105     
106       public Iterator<ModuleEntry> getModulesWithWarnings() {
107         return getMessages(Warning.MILD);
108       }
109     
110       public Set<Warning> getMessages(ModuleEntry m) {
111         return errors.get(m);
112       }
113       
114       
115       public void clearMessages() {
116         errors.clear();
117       }
118       
119       public IClass lookupClass(String className, IClassHierarchy cha) {
120         assert this.cha == cha;
121         return types.get(TypeName.string2TypeName(className));
122       }
123     
124       @Override
125       public IClass lookupClass(TypeName className) {
126         return types.get(className);
127       }
128     
129       @Override
130       public Iterator<IClass> iterateAllClasses() {
131         return types.values().iterator();
132       }
133     
134       @Override
135       public int getNumberOfClasses() {
136         return types.size();
137       }
138     
139       @Override
140       public Atom getName() {
141         return getReference().getName();
142       }
143     
144       @Override
145       public int getNumberOfMethods() {
146         int i = 0;
147         for (Iterator cls = types.values().iterator(); cls.hasNext();) {
148           for (Iterator ms = ((IClass) cls.next()).getDeclaredMethods().iterator();
149     	   ms.hasNext(); )
150           {
151             i++;
152             ms.next();
153           }
154         }
155     
156         return i;
157       }
158     
159       @Override
160       public String getSourceFileName(IMethod method, int bcOffset) {
161         if (!(method instanceof AstMethod)){
162           return null;
163         }
164         Position pos = ((AstMethod)method).getSourcePosition(bcOffset);
165         if (null == pos){
166           return null;
167         } 
168         return pos.getURL().getFile();
169       }
170       
171       @Override
172       public String getSourceFileName(IClass klass) {
173         return ((AstClass)klass).getSourcePosition().getURL().getFile();
174       }
175       
176       @Override
177       public Reader getSource(IClass klass) {
178         try {
179           return ((AstClass)klass).getSourcePosition().getReader();
180         } catch (IOException e) {
181           return null;
182         }
183       }
184     
185       @Override
186       public Reader getSource(IMethod method, int bcOffset) {
187         try {
188           return ((AstMethod)method).getSourcePosition(bcOffset).getReader();
189         } catch (IOException e) {
190           return null;
191         }
192       }
193     
194       @Override
195       public IClassLoader getParent() {
196         assert parent != null;
197         return parent;
198       }
199     
200       @Override
201       public void removeAll(Collection<IClass> toRemove) {
202         Set<TypeName> keys = HashSetFactory.make();
203     
204         for (Iterator<Map.Entry<TypeName,IClass>> EE = types.entrySet().iterator(); EE.hasNext();) {
205           Map.Entry<TypeName,IClass> E =  EE.next();
206           if (toRemove.contains(E.getValue())) {
207             keys.add(E.getKey());
208           }
209         }
210     
211         for (Iterator KK = keys.iterator(); KK.hasNext();) {
212           types.remove(KK.next());
213         }
214       }
215     
216     }
