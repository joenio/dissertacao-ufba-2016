11     package com.ibm.wala.util.graph.traverse;
12     
13     import java.util.Iterator;
14     import java.util.NoSuchElementException;
15     import java.util.Stack;
16     
17     import com.ibm.wala.util.collections.EmptyIterator;
18     import com.ibm.wala.util.collections.NonNullSingletonIterator;
19     import com.ibm.wala.util.debug.UnimplementedError;
20     import com.ibm.wala.util.graph.NumberedGraph;
21     
22     /**
23      * This class implements depth-first search over a {@link NumberedGraph}, return an enumeration of the nodes of the graph in order of
24      * increasing discover time. This class follows the outNodes of the graph nodes to define the graph, but this behavior can be
25      * changed by overriding the getConnected method.
26      */
27     public abstract class DFSDiscoverTimeIterator<T> extends Stack<T> implements Iterator<T> {
28     
29       /**
30        * an enumeration of all nodes to search from
31        */
32       private Iterator<? extends T> roots;
33     
34       /**
35        * subclass constructors must call this!
36        */
37       protected void init(Iterator<? extends T> nodes) {
38         roots = nodes;
39         assert nodes != null;
40         if (roots.hasNext()) {
41           T n = roots.next();
42           push(n);
43           setPendingChildren(n, getConnected(n));
44         }
45       }
46     
47       /**
48        * subclass constructors must call this!
49        */
50       protected void init(T N) {
51         init(new NonNullSingletonIterator<T>(N));
52       }
53     
54       /**
55        * Return whether there are any more nodes left to enumerate.
56        * 
57        * @return true if there nodes left to enumerate.
58        */
59       @Override
60       public boolean hasNext() {
61         return (!empty());
62       }
63     
64       abstract protected Iterator<? extends T> getPendingChildren(T n);
65     
66       abstract protected void setPendingChildren(T v, Iterator<? extends T> iterator);
67     
68       /**
69        * Find the next graph node in discover time order.
70        * 
71        * @return the next graph node in discover time order.
72        */
73       @Override
74       public T next() throws NoSuchElementException {
75     
76         if (empty()) {
77           throw new NoSuchElementException();
78         }
79     
80         // we always return the top node on the stack.
81         T toReturn = peek();
82     
83         // compute the next node to return.
84         assert getPendingChildren(toReturn) != null;
85         do {
86           T stackTop = peek();
87           for (Iterator<? extends T> it = getPendingChildren(stackTop); it.hasNext();) {
88             T child = it.next();
89             if (getPendingChildren(child) == null) {
90               // found a new child.
91               visitEdge(stackTop, child);
92               setPendingChildren(child, getConnected(child));
93               push(child);
94               return toReturn;
95             }
96           }
97           // the following saves space by allowing the original iterator to be GCed
98           Iterator<T> empty = EmptyIterator.instance();
99           setPendingChildren(stackTop, empty);
100           // didn't find any new children. pop the stack and try again.
101           pop();
102     
103         } while (!empty());
104     
105         // search for the next unvisited root.
106         while (roots.hasNext()) {
107           T nextRoot = roots.next();
108           if (getPendingChildren(nextRoot) == null) {
109             push(nextRoot);
110             setPendingChildren(nextRoot, getConnected(nextRoot));
111             return toReturn;
112           }
113         }
114     
115         return toReturn;
116       }
117     
118       /**
119        * get the out edges of a given node
120        * 
121        * @param n the node of which to get the out edges
122        * @return the out edges
123        * 
124        */
125       abstract protected Iterator<? extends T> getConnected(T n);
126     
127       @Override
128       public void remove() throws UnimplementedError {
129         throw new UnimplementedError();
130       }
131     
132       /**
133        * @param from source of the edge to visit
134        * @param to target of the edge to visit
135        */
136       protected void visitEdge(T from, T to) {
137         // do nothing. subclasses will override.
138       }
139     }
