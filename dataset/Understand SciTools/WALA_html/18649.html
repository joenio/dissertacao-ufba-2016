33     public class ObjectSensitivityContextSelector implements ContextSelector {
34       private final ContextSelector base;
35       
36       public ObjectSensitivityContextSelector(ContextSelector base) {
37         this.base = base;
38       }
39       
40       private final HashMap<MethodReference, Boolean> returnsThis_cache = HashMapFactory.make();
41       
42       private final IRFactory<IMethod> factory = AstIRFactory.makeDefaultFactory();
43     
44       // determine whether the method returns "this"
45       private boolean returnsThis(IMethod method) {
46         MethodReference mref = method.getReference();
47         if(method.getNumberOfParameters() < 1)
48           return false;
49         Boolean b = returnsThis_cache.get(mref);
50         if(b != null)
51           return b;
52         for(SSAInstruction inst : factory.makeIR(method, Everywhere.EVERYWHERE, SSAOptions.defaultOptions()).getInstructions()) {
53           if(inst instanceof SSAReturnInstruction) {
54             SSAReturnInstruction ret = (SSAReturnInstruction)inst;
55             if(ret.getResult() == 2) {
56               returnsThis_cache.put(mref, true);
57               return true;
58             }
59           }
60         }
61         returnsThis_cache.put(mref, false);
62         return false;
63       }
64     
65       @Override
66       public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] arguments) {
67         Context baseContext = base.getCalleeTarget(caller, site, callee, arguments);
68         if(returnsThis(callee)) {
69           if(arguments.length > 1 && arguments[1] != null) {
70             return new ArgumentInstanceContext(baseContext, 1, arguments[1]);
71           }
72         }
73         return baseContext;
74       }
75     
76       @Override
77       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
78         if (caller.getIR().getCalls(site)[0].getNumberOfUses() > 1) {
79           return IntSetUtil.make(new int[]{1}).union(base.getRelevantParameters(caller, site));
80         } else {
81           return base.getRelevantParameters(caller, site);
82         }
83       }
84     
85     }
