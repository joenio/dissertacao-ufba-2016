11     package com.ibm.wala.shrikeBT.shrikeCT.tools;
12     
13     import java.io.BufferedWriter;
14     import java.io.FileWriter;
15     import java.io.Writer;
16     
17     import com.ibm.wala.shrikeBT.shrikeCT.ClassInstrumenter;
18     import com.ibm.wala.shrikeBT.shrikeCT.OfflineInstrumenter;
19     import com.ibm.wala.shrikeCT.ClassReader;
20     import com.ibm.wala.shrikeCT.ConstantPoolParser;
21     
22     /**
23      * This is a demo class.
24      * 
25      * Class files are taken as input arguments (or if there are none, from standard input). We search those class files for all
26      * references to the Java library classes "SoftReference" or "WeakReference". This is just a demo to show how to write a simple tool
27      * like this. Here we're using the OfflineInstrumenter class to manage loading a set of class files and JARs for analysis; we don't
28      * actually modify any code.
29      * 
30      * In Unix, I run it like this: java -cp ~/dev/shrike/shrike com.ibm.wala.shrikeBT.shrikeCT.tools.ClassSearcher test.jar -o
31      * output.jar
32      */
33     public class ClassSearcher {
34       private static OfflineInstrumenter instrumenter;
35     
36       private static int scanned = 0;
37     
38       public static void main(String[] args) throws Exception {
39         instrumenter = new OfflineInstrumenter(true);
40     
41         Writer w = new BufferedWriter(new FileWriter("report", true));
42     
43         instrumenter.parseStandardArgs(args);
44         instrumenter.beginTraversal();
45         ClassInstrumenter ci;
46         while ((ci = instrumenter.nextClass()) != null) {
47           doClass(ci, w, instrumenter.getLastClassResourceName());
48         }
49         instrumenter.close();
50         w.close();
51     
52         System.out.println("Classes scanned: " + scanned);
53       }
54     
55       private static void doClass(final ClassInstrumenter ci, Writer w, String resource) throws Exception {
56         scanned++;
57     
58         String cl1 = "java/lang/ref/WeakReference";
59         String cl2 = "java/lang/ref/SoftReference";
60         ClassReader r = ci.getReader();
61         ConstantPoolParser cp = r.getCP();
62         for (int i = 1; i < cp.getItemCount(); i++) {
63           if (cp.getItemType(i) == ConstantPoolParser.CONSTANT_Class && (cp.getCPClass(i).equals(cl1) || cp.getCPClass(i).equals(cl2))) {
64             w.write(cp.getCPClass(i) + " " + resource + " " + r.getName() + "\n");
65           }
66         }
67       }
68     }
