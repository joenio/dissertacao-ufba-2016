78     /**
79      * Wrapper for the logic (nasty cascaded instanceof tests) necessary to visit a Polyglot AST and dispatch to the appropriate
80      * TranslatingVisitor methods for each AST node type.
81      * 
82      * @author rfuhrer
83      */
84     public class ASTTraverser {
85       protected ASTTraverser() {
86       }
87     
88       public static CAstNode visit(Node n, TranslatingVisitor tv, WalkContext wc) {
89         if (n instanceof MethodDecl) {
90           return tv.visit((MethodDecl) n, (MethodContext) wc);
91         } else if (n instanceof ConstructorDecl) {
92           return tv.visit((ConstructorDecl) n, (MethodContext) wc);
93         } else if (n instanceof FieldDecl) {
94           return tv.visit((FieldDecl) n, (MethodContext) wc);
95         } else if (n instanceof Import) {
96           return tv.visit((Import) n, wc);
97         } else if (n instanceof PackageNode) {
98           return tv.visit((PackageNode) n, wc);
99         } else if (n instanceof CanonicalTypeNode) {
100           return tv.visit((CanonicalTypeNode) n, wc);
101         } else if (n instanceof ArrayTypeNode) {
102           return tv.visit((ArrayTypeNode) n, wc);
103         } else if (n instanceof ArrayInit) {
104           return tv.visit((ArrayInit) n, wc);
105         } else if (n instanceof ArrayAccessAssign) {
106           return tv.visit((ArrayAccessAssign) n, wc);
107         } else if (n instanceof FieldAssign) {
108           return tv.visit((FieldAssign) n, wc);
109         } else if (n instanceof LocalAssign) {
110           return tv.visit((LocalAssign) n, wc);
111         } else if (n instanceof Binary) {
112           return tv.visit((Binary) n, wc);
113         } else if (n instanceof Call) {
114           return tv.visit((Call) n, wc);
115         } else if (n instanceof ConstructorCall) {
116           return tv.visit((ConstructorCall) n, wc);
117         } else if (n instanceof Cast) {
118           return tv.visit((Cast) n, wc);
119         } else if (n instanceof Conditional) {
120           return tv.visit((Conditional) n, wc);
121         } else if (n instanceof Instanceof) {
122           return tv.visit((Instanceof) n, wc);
123         } else if (n instanceof BooleanLit) {
124           return tv.visit((BooleanLit) n, wc);
125         } else if (n instanceof ClassLit) {
126           return tv.visit((ClassLit) n, wc);
127         } else if (n instanceof FloatLit) {
128           return tv.visit((FloatLit) n, wc);
129         } else if (n instanceof NullLit) {
130           return tv.visit((NullLit) n, wc);
131         } else if (n instanceof CharLit) {
132           return tv.visit((CharLit) n, wc);
133         } else if (n instanceof IntLit) {
134           return tv.visit((IntLit) n, wc);
135         } else if (n instanceof StringLit) {
136           return tv.visit((StringLit) n, wc);
137         } else if (n instanceof New) {
138           return tv.visit((New) n, wc);
139         } else if (n instanceof NewArray) {
140           return tv.visit((NewArray) n, wc);
141         } else if (n instanceof Special) {
142           return tv.visit((Special) n, wc);
143         } else if (n instanceof Unary) {
144           return tv.visit((Unary) n, wc);
145         } else if (n instanceof ArrayAccess) {
146           return tv.visit((ArrayAccess) n, wc);
147         } else if (n instanceof Field) {
148           return tv.visit((Field) n, wc);
149         } else if (n instanceof Local) {
150           return tv.visit((Local) n, wc);
151         } else if (n instanceof ClassBody) {
152           return tv.visit((ClassBody) n, wc);
153         } else if (n instanceof ClassDecl) {
154           return tv.visit((ClassDecl) n, wc);
155         } else if (n instanceof Initializer) {
156           return tv.visit((Initializer) n, wc);
157         } else if (n instanceof Assert) {
158           return tv.visit((Assert) n, wc);
159         } else if (n instanceof Branch) {
160           return tv.visit((Branch) n, wc);
161         } else if (n instanceof SwitchBlock) { // must test for this one before Block
162           return tv.visit((SwitchBlock) n, wc);
163         } else if (n instanceof Block) { // must test for this one before Block
164           return tv.visit((Block) n, wc);
165         } else if (n instanceof Catch) {
166           return tv.visit((Catch) n, wc);
167         } else if (n instanceof If) {
168           return tv.visit((If) n, wc);
169         } else if (n instanceof Labeled) {
170           return tv.visit((Labeled) n, wc);
171         } else if (n instanceof LocalClassDecl) {
172           return tv.visit((LocalClassDecl) n, wc);
173         } else if (n instanceof Do) {
174           return tv.visit((Do) n, wc);
175         } else if (n instanceof For) {
176           return tv.visit((For) n, wc);
177         } else if (n instanceof While) {
178           return tv.visit((While) n, wc);
179         } else if (n instanceof Switch) {
180           return tv.visit((Switch) n, wc);
181         } else if (n instanceof Synchronized) {
182           return tv.visit((Synchronized) n, wc);
183         } else if (n instanceof Try) {
184           return tv.visit((Try) n, wc);
185         } else if (n instanceof Empty) {
186           return tv.visit((Empty) n, wc);
187         } else if (n instanceof Eval) {
188           return tv.visit((Eval) n, wc);
189         } else if (n instanceof LocalDecl) {
190           return tv.visit((LocalDecl) n, wc);
191         } else if (n instanceof Return) {
192           return tv.visit((Return) n, wc);
193         } else if (n instanceof Case) {
194           return tv.visit((Case) n, wc);
195         } else if (n instanceof Throw) {
196           return tv.visit((Throw) n, wc);
197         } else if (n instanceof Formal) {
198           return tv.visit((Formal) n, wc);
199         } else {
200           Assertions.UNREACHABLE("Unhandled node " + n + " of type " + n.getClass().getName() + " in ASTTraverser.visit().");
201           return null;
202         }
203       }
204     }
