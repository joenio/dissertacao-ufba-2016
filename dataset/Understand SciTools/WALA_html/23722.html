25     /**
26      * A synthetic method that models the fake root node.
27      */
28     public class FakeRootMethod extends AbstractRootMethod {
29     
30       public static final Atom name = Atom.findOrCreateAsciiAtom("fakeRootMethod");
31     
32       public static final Descriptor descr = Descriptor.findOrCreate(new TypeName[0], TypeReference.VoidName);
33     
34       public static final MethodReference rootMethod = MethodReference.findOrCreate(FakeRootClass.FAKE_ROOT_CLASS, name, descr);
35     
36       public FakeRootMethod(final IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache) {
37         super(rootMethod, cha, options, cache);
38       }
39     
40       /**
41        * @return true iff m is the fake root method.
42        * @throws IllegalArgumentException if m is null
43        */
44       public static boolean isFakeRootMethod(MemberReference m) {
45         if (m == null) {
46           throw new IllegalArgumentException("m is null");
47         }
48         return m.equals(rootMethod);
49       }
50     
51       /**
52        * @return true iff block is a basic block in the fake root method
53        * @throws IllegalArgumentException if block is null
54        */
55       public static boolean isFromFakeRoot(IBasicBlock block) {
56         if (block == null) {
57           throw new IllegalArgumentException("block is null");
58         }
59         IMethod m = block.getMethod();
60         return FakeRootMethod.isFakeRootMethod(m.getReference());
61       }
62     
63       public static MethodReference getRootMethod() {
64         return rootMethod;
65       }
66     
67     }
