11     package com.ibm.wala.ipa.summaries;
12     
13     import com.ibm.wala.classLoader.IClass;
14     import com.ibm.wala.classLoader.SyntheticMethod;
15     import com.ibm.wala.ipa.callgraph.Context;
16     import com.ibm.wala.ipa.callgraph.impl.Everywhere;
17     import com.ibm.wala.ssa.IR;
18     import com.ibm.wala.ssa.SSAInstruction;
19     import com.ibm.wala.ssa.SSAOptions;
20     import com.ibm.wala.types.MethodReference;
21     import com.ibm.wala.types.TypeReference;
22     
23     /**
24      * A {@link SyntheticMethod} representing the semantics encoded in a {@link MethodSummary}
25      */
26     public class SummarizedMethod extends SyntheticMethod {
27       static final boolean DEBUG = false;
28     
29       final private MethodSummary summary;
30     
31       public SummarizedMethod(MethodReference ref, MethodSummary summary, IClass declaringClass) throws NullPointerException {
32         super(ref, declaringClass, summary.isStatic(), summary.isFactory());
33         this.summary = summary;
34         assert declaringClass != null;
35         if (DEBUG) {
36           System.err.println(("SummarizedMethod ctor: " + ref + " " + summary));
37         }
38       }
39     
40       /**
41        * @see com.ibm.wala.classLoader.IMethod#isNative()
42        */
43       @Override
44       public boolean isNative() {
45         return summary.isNative();
46       }
47     
48       /**
49        * @see com.ibm.wala.classLoader.IMethod#isAbstract()
50        */
51       @Override
52       public boolean isAbstract() {
53         return false;
54       }
55     
56       /*
57        * @see com.ibm.wala.classLoader.SyntheticMethod#getPoison()
58        */
59       @Override
60       public String getPoison() {
61         return summary.getPoison();
62       }
63     
64       /*
65        * @see com.ibm.wala.classLoader.SyntheticMethod#getPoisonLevel()
66        */
67       @Override
68       public byte getPoisonLevel() {
69         return summary.getPoisonLevel();
70       }
71     
72       /*
73        * @see com.ibm.wala.classLoader.SyntheticMethod#hasPoison()
74        */
75       @Override
76       public boolean hasPoison() {
77         return summary.hasPoison();
78       }
79     
80       @Override
81       public SSAInstruction[] getStatements(SSAOptions options) {
82         if (DEBUG) {
83           System.err.println(("getStatements: " + this));
84         }
85         return summary.getStatements();
86       }
87     
88       @Override
89       public IR makeIR(Context context, SSAOptions options) {
90         SSAInstruction instrs[] = getStatements(options);
91         return new SyntheticIR(this, Everywhere.EVERYWHERE, makeControlFlowGraph(instrs), instrs, options, summary.getConstants());
92       }
93     
94       /*
95        * @see com.ibm.wala.classLoader.IMethod#getNumberOfParameters()
96        */
97       @Override
98       public int getNumberOfParameters() {
99         return summary.getNumberOfParameters();
100       }
101     
102       /*
103        * @see com.ibm.wala.classLoader.IMethod#isStatic()
104        */
105       @Override
106       public boolean isStatic() {
107         return summary.isStatic();
108       }
109     
110       /*
111        * @see com.ibm.wala.classLoader.IMethod#getParameterType(int)
112        */
113       @Override
114       public TypeReference getParameterType(int i) {
115         return summary.getParameterType(i);
116       }
117     
118     }
