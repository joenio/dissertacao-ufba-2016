11     package com.ibm.wala.ipa.callgraph;
12     
13     import java.util.Collection;
14     import java.util.Iterator;
15     import java.util.Set;
16     
17     import com.ibm.wala.classLoader.CallSiteReference;
18     import com.ibm.wala.classLoader.IMethod;
19     import com.ibm.wala.ipa.cha.IClassHierarchy;
20     import com.ibm.wala.types.MethodReference;
21     import com.ibm.wala.util.graph.NumberedGraph;
22     
23     /**
24      * Basic interface for a call graph, which is a graph of {@link CGNode}
25      */
26     public interface CallGraph extends NumberedGraph<CGNode> {
27     
28       /**
29        * Return the (fake) interprocedural {@link CGNode root node} of the call graph.
30        * 
31        * @return the "fake" root node the call graph
32        */
33       public CGNode getFakeRootNode();
34     
35       CGNode getFakeWorldClinitNode();
36       
37       /**
38        * @return an Iterator of the nodes designated as "root nodes"
39        */
40       public Collection<CGNode> getEntrypointNodes();
41     
42       /**
43        * If you want to get <em> all </em> the nodes corresponding to a particular method, regardless of context, then use
44        * {@link CGNode getNodes}
45        * 
46        * @return the node corresponding a method in a context
47        */
48       public CGNode getNode(IMethod method, Context C);
49     
50       /**
51        * @param m a method reference
52        * @return the set of all nodes in the call graph that represent this method.
53        */
54       public Set<CGNode> getNodes(MethodReference m);
55     
56       /**
57        * @return the governing class hierarchy for this call graph
58        */
59       public IClassHierarchy getClassHierarchy();
60     
61       /**
62        * Return the set of CGNodes that represent possible targets of a particular call site from a particular node
63        */
64       public Set<CGNode> getPossibleTargets(CGNode node, CallSiteReference site);
65     
66       /**
67        * @return the number of nodes that the call site may dispatch to
68        */
69       public int getNumberOfTargets(CGNode node, CallSiteReference site);
70     
71       /**
72        * @return iterator of CallSiteReference, the call sites in a node that might dispatch to the target node.
73        */
74       Iterator<CallSiteReference> getPossibleSites(CGNode src, CGNode target);
75     
76     }
