41     /**
42      * 
43      * This application is a WALA client: it invokes an SWT TreeViewer to visualize
44      * a Call Graph
45      * 
46      * @author sfink
47      */
48     public class SWTCallGraph {
49     
50       private final static boolean CHECK_GRAPH = false;
51     
52       /**
53        * Usage: SWTCallGraph -appJar [jar file name]
54        * 
55        * The "jar file name" should be something like
56        * "c:/temp/testdata/java_cup.jar"
57        * 
58        * If it's a directory, then we'll try to find all jar files under that
59        * directory.
60        * 
61        * @param args
62        * @throws WalaException
63        */
64       public static void main(String[] args) throws WalaException {
65         Properties p = CommandLine.parse(args);
66         PDFCallGraph.validateCommandLine(p);
67         run(p);
68       }
69     
70       /**
71        * @param p
72        *            should contain at least the following properties:
73        *            <ul>
74        *            <li>appJar should be something like
75        *            "c:/temp/testdata/java_cup.jar"
76        *            <li>algorithm (optional) can be one of:
77        *            <ul>
78        *            <li> "ZERO_CFA" (default value)
79        *            <li> "RTA"
80        *            </ul>
81        *            </ul>
82        * 
83        * @throws WalaException
84        */
85       public static ApplicationWindow run(Properties p) throws WalaException {
86     
87         try {
88           String appJar = p.getProperty("appJar");
89           if (PDFCallGraph.isDirectory(appJar)) {
90             appJar = PDFCallGraph.findJarFiles(new String[] { appJar });
91           }
92     
93           String exclusionFile = p.getProperty("exclusions");
94     
95           AnalysisScope scope = AnalysisScopeReader.makeJavaBinaryAnalysisScope(appJar, exclusionFile != null ? new File(exclusionFile)
96               : (new FileProvider()).getFile(CallGraphTestUtil.REGRESSION_EXCLUSIONS));
97     
98           ClassHierarchy cha = ClassHierarchy.make(scope);
99     
100           Iterable<Entrypoint> entrypoints = null;
101           JarFile jar = new JarFile(appJar);
102           if (jar.getManifest() != null) {
103             String mainClass = jar.getManifest().getMainAttributes().getValue("Main-Class");
104             if (mainClass != null) {
105               entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha, "L" + mainClass.replace('.', '/'));
106             }
107           }
108           if (entrypoints == null) {
109             entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha);
110           }
111           
112           AnalysisOptions options = new AnalysisOptions(scope, entrypoints);
113           options.setReflectionOptions(ReflectionOptions.ONE_FLOW_TO_CASTS_NO_METHOD_INVOKE);
114           
115           // //
116           // build the call graph
117           // //
118           com.ibm.wala.ipa.callgraph.CallGraphBuilder builder = Util.makeZeroCFABuilder(options, new AnalysisCache(), cha, scope, null,
119               null);
120           CallGraph cg = builder.makeCallGraph(options,null);
121     
122           System.out.println(CallGraphStats.getStats(cg));
123     
124           if (CHECK_GRAPH) {
125             GraphIntegrity.check(cg);
126           }
127     
128           Properties wp = null;
129           try {
130             wp = WalaProperties.loadProperties();
131             wp.putAll(WalaExamplesProperties.loadProperties());
132           } catch (WalaException e) {
133             e.printStackTrace();
134             Assertions.UNREACHABLE();
135           }
136           String psFile = wp.getProperty(WalaProperties.OUTPUT_DIR) + File.separatorChar + PDFWalaIR.PDF_FILE;
137           String dotFile = wp.getProperty(WalaProperties.OUTPUT_DIR) + File.separatorChar + PDFTypeHierarchy.DOT_FILE;
138           String dotExe = wp.getProperty(WalaExamplesProperties.DOT_EXE);
139           String gvExe = wp.getProperty(WalaExamplesProperties.PDFVIEW_EXE);
140     
141           // create and run the viewer
142           final SWTTreeViewer v = new SWTTreeViewer();
143           v.setGraphInput(cg);
144           v.setRootsInput(InferGraphRoots.inferRoots(cg));
145           v.getPopUpActions().add(new ViewIRAction(v, cg, psFile, dotFile, dotExe, gvExe));
146           v.run();
147           return v.getApplicationWindow();
148     
149         } catch (Exception e) {
150           e.printStackTrace();
151           return null;
152         }
153       }
154     }
