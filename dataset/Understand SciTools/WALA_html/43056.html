45     /**
46      * An Eclipse action that analyzes a Java selection
47      */
48     public abstract class AbstractJavaAnalysisAction implements IObjectActionDelegate, IRunnableWithProgress {
49     
50       /**
51        * The current {@link ISelection} highlighted in the Eclipse workspace
52        */
53       private ISelection currentSelection;
54     
55       public AbstractJavaAnalysisAction() {
56         super();
57       }
58     
59       /*
60        * @see IObjectActionDelegate#setActivePart(IAction, IWorkbenchPart)
61        */
62       @Override
63       public void setActivePart(IAction action, IWorkbenchPart targetPart) {
64       }
65     
66       /**
67        * Compute an analysis scope for the current selection
68        */
69       public static AnalysisScope computeScope(IStructuredSelection selection) throws IOException {
70         return computeScope(selection, EclipseProjectPath.AnalysisScopeType.NO_SOURCE);
71       }
72     
73       /**
74        * Compute an analysis scope for the current selection
75        * 
76        * @param scopeType should analysis use the source files in the Eclipse projects rather than the class files.
77        */
78       public static AnalysisScope computeScope(final IStructuredSelection selection, 
79     		  final EclipseProjectPath.AnalysisScopeType scopeType) throws IOException 
80       {
81         if (selection == null) {
82           throw new IllegalArgumentException("null selection");
83         }
84         final Collection<EclipseProjectPath> projectPaths = new LinkedList<EclipseProjectPath>();
85         Job job = new Job("Compute project paths") {
86     
87           @SuppressWarnings("unchecked")
88           @Override
89           protected IStatus run(IProgressMonitor monitor) {
90             for (Iterator it = selection.iterator(); it.hasNext();) {
91               Object object = it.next();
92               if (object instanceof IJavaElement) {
93                 IJavaElement e = (IJavaElement) object;
94                 IJavaProject jp = e.getJavaProject();
95                 try {
96                   projectPaths.add(JavaEclipseProjectPath.make(jp, scopeType));
97                 } catch (CoreException e1) {
98                   e1.printStackTrace();
99                   // skip and continue
100                 } catch (IOException e2) {
101                   e2.printStackTrace();
102                   return new Status(IStatus.ERROR, "", 0, "", e2);
103                 }
104               } else {
105                 Assertions.UNREACHABLE(object.getClass());
106               }
107             }
108             return Status.OK_STATUS;
109           }
110     
111         };
112         // lock the whole workspace
113         job.setRule(ResourcesPlugin.getWorkspace().getRoot());
114         job.schedule();
115         try {
116           job.join();
117           IStatus result = job.getResult();
118           if (result.getSeverity() == IStatus.ERROR) {
119             Throwable exception = result.getException();
120             if (exception instanceof IOException) {
121               throw (IOException) exception;
122             } else if (exception instanceof RuntimeException) {
123               throw (RuntimeException) exception;
124             }
125           }
126         } catch (InterruptedException e) {
127           e.printStackTrace();
128           assert false;
129         }
130         AnalysisScope scope = mergeProjectPaths(projectPaths);
131         return scope;
132       }
133     
134       /**
135        * compute the java projects represented by the current selection
136        */
137       @SuppressWarnings("unchecked")
138       protected Collection<IJavaProject> computeJavaProjects() {
139         IStructuredSelection selection = (IStructuredSelection) currentSelection;
140         Collection<IJavaProject> projects = HashSetFactory.make();
141         for (Iterator it = selection.iterator(); it.hasNext();) {
142           Object object = it.next();
143           if (object instanceof IJavaElement) {
144             IJavaElement e = (IJavaElement) object;
145             IJavaProject jp = e.getJavaProject();
146             projects.add(jp);
147           } else {
148             Assertions.UNREACHABLE(object.getClass());
149           }
150         }
151         return projects;
152       }
153     
154       /**
155        * create an analysis scope as the union of a bunch of EclipseProjectPath
156        */
157       private static AnalysisScope mergeProjectPaths(Collection<EclipseProjectPath> projectPaths) throws IOException {
158         AnalysisScope scope = AnalysisScope.createJavaAnalysisScope();
159     
160         Collection<Module> seen = HashSetFactory.make();
161         // to avoid duplicates, we first add all application modules, then
162         // extension
163         // modules, then primordial
164         buildScope(ClassLoaderReference.Application, projectPaths, scope, seen);
165         buildScope(ClassLoaderReference.Extension, projectPaths, scope, seen);
166         buildScope(ClassLoaderReference.Primordial, projectPaths, scope, seen);
167         return scope;
168       }
169     
170       /**
171        * Enhance an {@link AnalysisScope} to include in a particular loader, elements from a set of Eclipse projects
172        * 
173        * @param loader the class loader in which new {@link Module}s will live
174        * @param projectPaths Eclipse project paths to add to the analysis scope
175        * @param scope the {@link AnalysisScope} under construction. This will be mutated.
176        * @param seen set of {@link Module}s which have already been seen, and should not be added to the analysis scope
177        */
178       private static void buildScope(ClassLoaderReference loader, Collection<EclipseProjectPath> projectPaths, AnalysisScope scope,
179           Collection<Module> seen) throws IOException {
180         for (EclipseProjectPath path : projectPaths) {
181           AnalysisScope pScope = path.toAnalysisScope((File) null);
182           for (Module m : pScope.getModules(loader)) {
183             if (!seen.contains(m)) {
184               seen.add(m);
185               scope.addToScope(loader, m);
186             }
187           }
188         }
189       }

191       /*
192        * @see IActionDelegate#run(IAction)
193        */
194       @Override
195       public void run(IAction action) {
196         IProgressService progressService = PlatformUI.getWorkbench().getProgressService();
197         try {
198           progressService.busyCursorWhile(this);
199         } catch (InvocationTargetException e) {
200           e.printStackTrace();
201         } catch (InterruptedException e) {
202           e.printStackTrace();
203         }
204     
205       }
206     
207       /*
208        * @see IActionDelegate#selectionChanged(IAction, ISelection)
209        */
210       @Override
211       public void selectionChanged(IAction action, ISelection selection) {
212         currentSelection = selection;
213       }
214     
215       /**
216        * @return The current {@link ISelection} highlighted in the Eclipse workspace
217        */
218       public ISelection getCurrentSelection() {
219         return currentSelection;
220       }
