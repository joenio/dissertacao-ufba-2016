11     package com.ibm.wala.cast.ipa.callgraph;
12     
13     import java.util.Iterator;
14     import java.util.LinkedList;
15     import java.util.List;
16     
17     import com.ibm.wala.classLoader.IClass;
18     import com.ibm.wala.classLoader.IField;
19     import com.ibm.wala.ipa.callgraph.CGNode;
20     import com.ibm.wala.ipa.callgraph.propagation.ConcreteTypeKey;
21     import com.ibm.wala.ipa.callgraph.propagation.ConstantKey;
22     import com.ibm.wala.ipa.callgraph.propagation.FilteredPointerKey;
23     import com.ibm.wala.ipa.callgraph.propagation.InstanceFieldKey;
24     import com.ibm.wala.ipa.callgraph.propagation.InstanceKey;
25     import com.ibm.wala.ipa.callgraph.propagation.PointerKey;
26     import com.ibm.wala.ipa.callgraph.propagation.PointerKeyFactory;
27     import com.ibm.wala.util.collections.NonNullSingletonIterator;
28     import com.ibm.wala.util.strings.Atom;
29     
30     public class DelegatingAstPointerKeys implements AstPointerKeyFactory {
31       private final PointerKeyFactory base;
32     
33       public DelegatingAstPointerKeys(PointerKeyFactory base) {
34         this.base = base;
35       }
36     
37       @Override
38       public PointerKey getPointerKeyForLocal(CGNode node, int valueNumber) {
39         return base.getPointerKeyForLocal(node, valueNumber);
40       }
41     
42       @Override
43       public FilteredPointerKey getFilteredPointerKeyForLocal(CGNode node, int valueNumber, FilteredPointerKey.TypeFilter filter) {
44         return base.getFilteredPointerKeyForLocal(node, valueNumber, filter);
45       }
46     
47       @Override
48       public PointerKey getPointerKeyForReturnValue(CGNode node) {
49         return base.getPointerKeyForReturnValue(node);
50       }
51     
52       @Override
53       public PointerKey getPointerKeyForExceptionalReturnValue(CGNode node) {
54         return base.getPointerKeyForExceptionalReturnValue(node);
55       }
56     
57       @Override
58       public PointerKey getPointerKeyForStaticField(IField f) {
59         return base.getPointerKeyForStaticField(f);
60       }
61     
62       @Override
63       public PointerKey getPointerKeyForObjectCatalog(InstanceKey I) {
64         return new ObjectPropertyCatalogKey(I);
65       }
66     
67       @Override
68       public PointerKey getPointerKeyForInstanceField(InstanceKey I, IField f) {
69         return base.getPointerKeyForInstanceField(I, f);
70       }
71     
72       @Override
73       public PointerKey getPointerKeyForArrayContents(InstanceKey I) {
74         return base.getPointerKeyForArrayContents(I);
75       }
76     
77       @Override
78       public Iterator<PointerKey> getPointerKeysForReflectedFieldWrite(InstanceKey I, InstanceKey F) {
79         List<PointerKey> result = new LinkedList<PointerKey>();
80     
81         if (F instanceof ConstantKey) {
82           PointerKey ifk = getInstanceFieldPointerKeyForConstant(I, (ConstantKey) F);
83           if (ifk != null) {
84             result.add(ifk);
85           }
86         }
87     
88         result.add(ReflectedFieldPointerKey.mapped(new ConcreteTypeKey(getFieldNameType(F)), I));
89     
90         return result.iterator();
91       }
92     
93       /**
94        * get type for F appropriate for use in a field name.
95        * 
96        * @param F
97        * @return
98        */
99       protected IClass getFieldNameType(InstanceKey F) {
100         return F.getConcreteType();
101       }
102     
103       /**
104        * if F is a supported constant representing a field, return the corresponding {@link InstanceFieldKey} for I.  Otherwise, return <code>null</code>.
105        * @param F
106        * @return
107        */
108       protected PointerKey getInstanceFieldPointerKeyForConstant(InstanceKey I, ConstantKey F) {
109         Object v = F.getValue();
110         // FIXME: current only constant string are handled
111         if (v instanceof String) {
112           IField f = I.getConcreteType().getField(Atom.findOrCreateUnicodeAtom((String) v));
113           return getPointerKeyForInstanceField(I, f);
114         }
115         return null;
116       }
117       
118       @Override
119       public Iterator<PointerKey> getPointerKeysForReflectedFieldRead(InstanceKey I, InstanceKey F) {
120         if (F instanceof ConstantKey) {
121           PointerKey ifk = getInstanceFieldPointerKeyForConstant(I, (ConstantKey) F);
122           if (ifk != null) {
123             return new NonNullSingletonIterator<PointerKey>(ifk);
124           }
125         }
126         return new NonNullSingletonIterator<PointerKey>(ReflectedFieldPointerKey.mapped(new ConcreteTypeKey(getFieldNameType(F)), I));
127       }
128     }
