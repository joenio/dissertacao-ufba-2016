42     /**
43      */
44     public abstract class JavaSourceAnalysisEngine extends AbstractAnalysisEngine {
45     
46       /**
47        * Modules which are user-space code
48        */
49       private final Set<Module> userEntries = HashSetFactory.make();
50     
51       /**
52        * Modules which are source code
53        */
54       private final Set<Module> sourceEntries = HashSetFactory.make();
55     
56       /**
57        * Modules which are system or library code TODO: what about extension loader?
58        */
59       private final Set<Module> systemEntries = HashSetFactory.make();
60     
61       public JavaSourceAnalysisEngine() {
62         super();
63       }
64     
65       /**
66        * Adds the given source module to the source loader's module list. Clients
67        * should/may call this method if they don't supply an IJavaProject to the
68        * constructor.
69        */
70       public void addSourceModule(Module M) {
71         sourceEntries.add(M);
72       }
73     
74       /**
75        * Adds the given compiled module to the application loader's module list.
76        * Clients should/may call this method if they don't supply an IJavaProject to
77        * the constructor.
78        */
79       public void addCompiledModule(Module M) {
80         userEntries.add(M);
81       }
82     
83       /**
84        * Adds the given module to the primordial loader's module list. Clients
85        * should/may call this method if they don't supply an IJavaProject to the
86        * constructor.
87        */
88       public void addSystemModule(Module M) {
89         systemEntries.add(M);
90       }
91     
92       @Override
93       protected void addApplicationModulesToScope() {
94         ClassLoaderReference app = scope.getApplicationLoader();
95         for (Module M : userEntries) {
96           scope.addToScope(app, M);
97         }
98     
99         ClassLoaderReference src = ((JavaSourceAnalysisScope) scope).getSourceLoader();
100     
101         for (Module M : sourceEntries) {
102           scope.addToScope(src, M);
103         }
104       }
105     
106       @Override
107       public void buildAnalysisScope() throws IOException {
108         scope = makeSourceAnalysisScope();
109     
110         if (getExclusionsFile() != null) {
111           InputStream is = new File(getExclusionsFile()).exists()? new FileInputStream(getExclusionsFile()): FileProvider.class.getClassLoader().getResourceAsStream(getExclusionsFile());
112           scope.setExclusions(new FileOfClasses(is));
113         }
114     
115         for (Module M : this.systemEntries) {
116           scope.addToScope(scope.getPrimordialLoader(), M);
117         }
118     
119         // add user stuff
120         addApplicationModulesToScope();
121       }
122     
123       protected AnalysisScope makeSourceAnalysisScope() {
124         return new JavaSourceAnalysisScope();
125       }
126     
127       protected abstract ClassLoaderFactory getClassLoaderFactory(SetOfClasses exclusions);
128       
129       @Override
130       public IClassHierarchy buildClassHierarchy() {
131         IClassHierarchy cha = null;
132         ClassLoaderFactory factory = getClassLoaderFactory(scope.getExclusions());
133     
134         try {
135           cha = ClassHierarchy.make(getScope(), factory);
136         } catch (ClassHierarchyException e) {
137           System.err.println("Class Hierarchy construction failed");
138           System.err.println(e.toString());
139           e.printStackTrace();
140         }
141         return cha;
142       }
143     
144       @Override
145       protected Iterable<Entrypoint> makeDefaultEntrypoints(AnalysisScope scope, IClassHierarchy cha) {
146         return Util.makeMainEntrypoints(JavaSourceAnalysisScope.SOURCE, cha);
147       }
148     
149       @Override
150       public AnalysisCache makeDefaultCache() {
151         return new AnalysisCache(AstIRFactory.makeDefaultFactory());
152       }
153     
154       @Override
155       public AnalysisOptions getDefaultOptions(Iterable<Entrypoint> entrypoints) {
156         AnalysisOptions options = new AnalysisOptions(getScope(), entrypoints);
157     
158         SSAOptions ssaOptions = new SSAOptions();
159         ssaOptions.setDefaultValues(new SSAOptions.DefaultValues() {
160           @Override
161           public int getDefaultValue(SymbolTable symtab, int valueNumber) {
162             return symtab.getDefaultValue(valueNumber);
163           }
164         });
165     
166         options.setSSAOptions(ssaOptions);
167     
168         return options;
169       }
170     
171       @Override
172       protected CallGraphBuilder getCallGraphBuilder(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache) {
173         return new ZeroCFABuilderFactory().make(options, cache, cha, scope, false);
174       }
175     }
