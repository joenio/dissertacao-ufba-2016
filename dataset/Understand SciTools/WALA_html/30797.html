108         /**
109          * Compute a mapping from instruction to basic block. Also, compute the blocks that end with a 'normal' return.
110          */
111         private void computeI2BMapping() {
112             instruction2Block = new int[getInstructions().length];
113             for (Iterator<BasicBlock> it = iterator(); it.hasNext();) {
114                 final BasicBlock b = it.next();
115                 for (int j = b.getFirstInstructionIndex(); j <= b.getLastInstructionIndex(); j++) {
116                     instruction2Block[j] = getNumber(b);
117                 }
118             }
119         }

121         /**
122          * Compute outgoing edges in the control flow graph.
123          */
124         private void computeEdges() {
125             for (Iterator<BasicBlock> it = iterator(); it.hasNext();) {
126                 BasicBlock b = it.next();
127                 if (b.equals(exit())) {
128                     continue;
129                 } else if (b.equals(entry())) {
130                     BasicBlock bb0 = getBlockForInstruction(0);
131                     assert bb0 != null;
132                     addNormalEdge(b, bb0);
133                 } else {
134                     b.computeOutgoingEdges();
135                 }
136             }
137         }

139         private void makeBasicBlocks() {
140             ExceptionHandler[][] handlers;
141             try {
142                 handlers = dexMethod.getHandlers();
143             } catch (InvalidClassFileException e) {
144                 e.printStackTrace();
145                 Assertions.UNREACHABLE();
146                 handlers = null;
147             }
148             boolean[] r = new boolean[getInstructions().length];
149             boolean[] catchers = new boolean[getInstructions().length];
150             // we initially start with both the entry and exit block.
151             @SuppressWarnings("unused")
152     		int blockCount = 2;
153     
154             // Compute r so r[i] == true iff instruction i begins a basic block.
155             // While doing so count the number of blocks.
156             r[0] = true;
157             Instruction[] instructions = getInstructions();
158             for (int i = 0; i < instructions.length; i++) {
159                 int[] targets = instructions[i].getBranchTargets();
160     
161                 // if there are any targets, then break the basic block here.
162                 // also break the basic block after a return
163                 if (targets.length > 0 || !instructions[i].isFallThrough()) {
164                     if (i + 1 < instructions.length && !r[i + 1]) {
165                         r[i + 1] = true;
166                         blockCount++;
167                     }
168                 }
169     
170                 for (int j = 0; j < targets.length; j++) {
171                     if (!r[targets[j]]) {
172                         r[targets[j]] = true;
173                         blockCount++;
174                     }
175                 }
176                 if (instructions[i].isPEI()) {
177                     ExceptionHandler[] hs = handlers[i];
178                     // break the basic block here.
179                     if (i + 1 < instructions.length && !r[i + 1]) {
180                         r[i + 1] = true;
181                         blockCount++;
182                     }
183                     if (hs != null && hs.length > 0) {
184                         for (int j = 0; j < hs.length; j++) {
185                             exceptionHandlers.add(hs[j]);
186                             if (!r[hs[j].getHandler()]) {
187                                 // we have not discovered the catch block yet.
188                                 // form a new basic block
189                                 r[hs[j].getHandler()] = true;
190                                 blockCount++;
191                             }
192                             catchers[hs[j].getHandler()] = true;
193                         }
194                     }
195                 }
196             }
197     
198             BasicBlock entry = new BasicBlock(-1);
199             addNode(entry);
200     
201             int j = 1;
202             for (int i = 0; i < r.length; i++) {
203                 if (r[i]) {
204                     BasicBlock b = new BasicBlock(i);
205                     addNode(b);
206                     if (catchers[i]) {
207                         setCatchBlock(j);
208                     }
209                     j++;
210                 }
211             }
212     
213             BasicBlock exit = new BasicBlock(-1);
214             addNode(exit);
215         }

239             private void computeOutgoingEdges() {
240                 if (DEBUG) {
241                     System.err.println("Block " + this + ": computeOutgoingEdges()");
242                 }
243     
244                 Instruction last = getInstructions()[getLastInstructionIndex()];
245                 int[] targets = last.getBranchTargets();
246                 for (int i = 0; i < targets.length; i++) {
247                     BasicBlock b = getBlockForInstruction(targets[i]);
248                     addNormalEdgeTo(b);
249                 }
250                 addExceptionalEdges(last);
251                 if (last.isFallThrough()) {
252                     BasicBlock next = getNode(getNumber() + 1);
253                     addNormalEdgeTo(next);
254                 }
255                 if (last instanceof Return) {
256                     // link each return instruction to the exit block.
257                     BasicBlock exit = exit();
258                     addNormalEdgeTo(exit);
259                 }
260             }

262             /**
263              * Add any exceptional edges generated by the last instruction in a basic block.
264              *
265              * @param last the last instruction in a basic block.
266              */
267             protected void addExceptionalEdges(Instruction last) {
268                 IClassHierarchy cha = getMethod().getClassHierarchy();
269                 if (last.isPEI()) {
270                     Collection<TypeReference> exceptionTypes = null;
271                     boolean goToAllHandlers = false;
272     
273                     ExceptionHandler[] hs = getExceptionHandlers();
274                     if (last instanceof Throw) {
275                         // this class does not have the type information needed
276                         // to determine what the athrow throws. So, add an
277                         // edge to all reachable handlers. Better information can
278                         // be obtained later with SSA type propagation.
279                         // TODO: consider pruning to only the exception types that
280                         // this method either catches or allocates, since these are
281                         // the only types that can flow to an athrow.
282                         goToAllHandlers = true;
283                     } else {
284                         if (hs != null && hs.length > 0) {
285                             IClassLoader loader = getMethod().getDeclaringClass().getClassLoader();
286                             BytecodeLanguage l = (BytecodeLanguage) loader.getLanguage();
287                             //exceptionTypes = l.getImplicitExceptionTypes(last);
288                             exceptionTypes = getImplicitExceptionTypes(last);
289                             if (last instanceof Invoke) {
290                                 Invoke call = (Invoke) last;
291                                 exceptionTypes = HashSetFactory.make(exceptionTypes);
292                                 MethodReference target = MethodReference.findOrCreate(l, loader.getReference(), call.clazzName, call
293                                         .methodName, call.descriptor);
294                                 try {
295                                     exceptionTypes.addAll(l.inferInvokeExceptions(target, cha));
296                                 } catch (InvalidClassFileException e) {
297                                     e.printStackTrace();
298                                     Assertions.UNREACHABLE();
299                                 }
300                             }
301                         }
302                     }
303     
304                     if (hs != null && hs.length > 0) {
305                         // found a handler for this PEI
306     
307                         // create a mutable copy
308                         if (!goToAllHandlers) {
309                             exceptionTypes = HashSetFactory.make(exceptionTypes);
310                         }
311     
312                         for (int j = 0; j < hs.length; j++) {
313                             if (DEBUG) {
314                                 System.err.println(" handler " + hs[j]);
315                             }
316                             BasicBlock b = getBlockForInstruction(hs[j].getHandler());
317                             if (DEBUG) {
318                                 System.err.println(" target " + b);
319                             }
320                             if (goToAllHandlers) {
321                                 // add an edge to the catch block.
322                                 if (DEBUG) {
323                                     System.err.println(" gotoAllHandlers " + b);
324                                 }
325                                 addExceptionalEdgeTo(b);
326                             } else {
327                                 TypeReference caughtException = null;
328                                 if (hs[j].getCatchClass() != null) {
329                                     ClassLoaderReference loader = DexCFG.this.getMethod().getDeclaringClass().getReference().getClassLoader();
330                                     caughtException = ShrikeUtil.makeTypeReference(loader, hs[j].getCatchClass());
331                                     if (DEBUG) {
332                                         System.err.println(" caughtException " + caughtException);
333                                     }
334                                     IClass caughtClass = cha.lookupClass(caughtException);
335                                     if (caughtClass == null) {
336                                         // conservatively add the edge, and raise a warning
337                                         addExceptionalEdgeTo(b);
338                                         Warnings.add(FailedExceptionResolutionWarning.create(caughtException));
339                                         // null out caughtException, to avoid attempting to process it
340                                         caughtException = null;
341                                     }
342                                 } else {
343                                     if (DEBUG) {
344                                         System.err.println(" catchClass() == null");
345                                     }
346                                     // hs[j].getCatchClass() == null.
347                                     // this means that the handler catches all exceptions.
348                                     // add the edge and null out all types
349                                     if (!exceptionTypes.isEmpty()) {
350                                         addExceptionalEdgeTo(b);
351                                         exceptionTypes.clear();
352                                         caughtException = null;
353                                     }
354                                 }
355                                 if (caughtException != null) {
356                                     IClass caughtClass = cha.lookupClass(caughtException);
357                                     // the set "caught" should be the set of exceptions that MUST
358                                     // have been caught by the handlers in scope
359                                     ArrayList<TypeReference> caught = new ArrayList<TypeReference>(exceptionTypes.size());
360                                     // check if we should add an edge to the catch block.
361                                     for (TypeReference t : exceptionTypes) {
362                                         if (t != null) {
363                                             IClass klass = cha.lookupClass(t);
364                                             if (klass == null) {
365                                                 Warnings.add(FailedExceptionResolutionWarning.create(caughtException));
366                                                 // conservatively add an edge
367                                                 addExceptionalEdgeTo(b);
368                                             } else {
369                                                 boolean subtype1 = cha.isSubclassOf(klass, caughtClass);
370                                                 if (subtype1 || cha.isSubclassOf(caughtClass, klass)) {
371                                                     // add the edge and null out the type from the array
372                                                     addExceptionalEdgeTo(b);
373                                                     if (subtype1) {
374                                                         caught.add(t);
375                                                     }
376                                                 }
377                                             }
378                                         }
379                                     }
380                                     exceptionTypes.removeAll(caught);
381                                 }
382                             }
383                         }
384                         // if needed, add an edge to the exit block.
385                         if (exceptionTypes == null || !exceptionTypes.isEmpty()) {
386                             BasicBlock exit = exit();
387                             addExceptionalEdgeTo(exit);
388                         }
389                     } else {
390                         // found no handler for this PEI ... link to the exit block.
391                         BasicBlock exit = exit();
392                         addExceptionalEdgeTo(exit);
393                     }
394                 }
395             }

581             public int getLastInstructionIndex() {
582                 if (this == entry() || this == exit()) {
583                     // these are the special end blocks
584                     return -2;
585                 }
586                 if (getNumber() == (getMaxNumber() - 1)) {
587                     // this is the last non-exit block
588                     return getInstructions().length - 1;
589                 } else {
590                     BasicBlock next = getNode(getNumber() + 1);
591                     return next.getFirstInstructionIndex() - 1;
592                 }
593             }

648         @Override
649         public String toString() {
650             StringBuffer s = new StringBuffer("");
651             BitVector catches = this.getCatchBlocks();
652             for (Iterator<BasicBlock> it = iterator(); it.hasNext();) {
653                 BasicBlock bb = it.next();
654                 s.append("BB").append(getNumber(bb));
655                 if (catches.contains(bb.getNumber())) {
656                 	s.append("<Handler>");
657                 }
658                 s.append("\n");
659                 for (int j = bb.getFirstInstructionIndex(); j <= bb.getLastInstructionIndex(); j++) {
660                     s.append("  ").append(j).append("  ").append(getInstructions()[j]).append("\n");
661                 }
662     
663                 Iterator<BasicBlock> succNodes = getSuccNodes(bb);
664                 while (succNodes.hasNext()) {
665                     s.append("    -> BB").append(getNumber(succNodes.next())).append("\n");
666                 }
667             }
668             return s.toString();
669         }
