18     /**
19      * @author danielk
20      * Data structure representing an HTML tag, with its attributes and content. Used by the HTML parser when calling the callback.
21      */
22     public interface ITag {
23     
24     	/**
25     	 * @return tag's name (e.g., "HEAD" / "HTML" / "FORM")
26     	 */
27     	public String getName();
28     
29     	/**
30     	 * Retrieves a specific attribute 
31     	 * @param name
32     	 * @return null if there is no such attribute
33     	 */
34     	public Pair<String, Position> getAttributeByName(String name);
35     
36     	public Map<String, Pair<String, Position>> getAllAttributes();
37     
38     	/**
39     	 * Returns the starting line number of the tag.
40     	 * @return null if no known
41     	 */
42     	public Position getElementPosition();
43     
44       public Position getContentPosition();
45     }
