11     package com.ibm.wala.shrikeCT;
12     
13     /**
14      * This class helps emit Code elements.
15      * 
16      * After constructing a CodeWriter, at least the max stack, max locals and bytecode bytes must be set before it can be used.
17      */
18     public final class CodeWriter extends ClassWriter.Element {
19       final private int attrID;
20     
21       private int maxLocals = -1;
22     
23       private int maxStack = -1;
24     
25       private byte[] code;
26     
27       private int[] exnHandlers;
28     
29       private ClassWriter.Element[] attributes;
30     
31       /**
32        * Build an empty serializable Code attribute.
33        * 
34        * @throws IllegalArgumentException if w is null
35        */
36       public CodeWriter(ClassWriter w) {
37         if (w == null) {
38           throw new IllegalArgumentException("w is null");
39         }
40         attrID = w.addCPUtf8("Code");
41       }
42     
43       private void verify() {
44         if (maxStack < 0) {
45           throw new IllegalArgumentException("maxStack not set");
46         }
47         if (maxLocals < 0) {
48           throw new IllegalArgumentException("maxLocals not set");
49         }
50         if (code == null) {
51           throw new IllegalArgumentException("No bytecodes set");
52         }
53       }
54     
55       public int getCodeLength() throws IllegalStateException {
56         if (code == null) {
57           throw new IllegalStateException("code not initialized");
58         }
59         return code.length;
60       }
61     
62       @Override
63       public int getSize() throws IllegalArgumentException {
64         verify();
65     
66         int size = 14 + code.length + 2 + (exnHandlers == null ? 0 : exnHandlers.length) * 2 + 2;
67         if (attributes != null) {
68           for (int i = 0; i < attributes.length; i++) {
69             size += attributes[i].getSize();
70           }
71         }
72         return size;
73       }
74     
75       @Override
76       public int copyInto(byte[] buf, int offset) throws IllegalArgumentException {
77         verify();
78     
79         int start = offset;
80         ClassWriter.setUShort(buf, offset, attrID);
81         ClassWriter.setUShort(buf, offset + 6, maxStack);
82         ClassWriter.setUShort(buf, offset + 8, maxLocals);
83         ClassWriter.setInt(buf, offset + 10, code.length);
84         offset += 14;
85         System.arraycopy(code, 0, buf, offset, code.length);
86         offset += code.length;
87         ClassWriter.setUShort(buf, offset, (exnHandlers == null ? 0 : exnHandlers.length) / 4);
88         offset += 2;
89         if (exnHandlers != null) {
90           for (int i = 0; i < exnHandlers.length; i++) {
91             ClassWriter.setUShort(buf, offset, exnHandlers[i]);
92             offset += 2;
93           }
94         }
95     
96         ClassWriter.setUShort(buf, offset, (attributes == null ? 0 : attributes.length));
97         offset += 2;
98         if (attributes != null) {
99           for (int i = 0; i < attributes.length; i++) {
100             offset = attributes[i].copyInto(buf, offset);
101           }
102         }
103         ClassWriter.setInt(buf, start + 2, offset - start - 6);
104         return offset;
105       }
106     
107       /**
108        * Set the bytecodes for this Code attribute.
109        * 
110        * @throws IllegalArgumentException if code is null
111        */
112       public void setCode(byte[] code) throws IllegalArgumentException {
113         if (code == null) {
114           throw new IllegalArgumentException("code is null");
115         }
116         if (code.length > 0xFFFF) {
117           throw new IllegalArgumentException("Code array is too long: " + code.length);
118         }
119         if (code.length == 0) {
120           throw new IllegalArgumentException("Code array is empty");
121         }
122     
123         this.code = code;
124       }
125     
126       /**
127        * Set the raw handler data for this Code attribute.
128        * 
129        * @param exnHandlers a flattened sequence of (startPC, endPC, catchClassIndex, catchPC) tuples
130        * @throws IllegalArgumentException if exnHandlers is null
131        */
132       public void setRawHandlers(int[] exnHandlers) {
133         if (exnHandlers == null) {
134           throw new IllegalArgumentException("exnHandlers is null");
135         }
136         if (exnHandlers.length % 4 != 0) {
137           throw new IllegalArgumentException("Exception handlers array has bad length: " + exnHandlers.length);
138         }
139         if (exnHandlers.length / 4 > 0xFFFF) {
140           throw new IllegalArgumentException("Too many exception handlers: " + exnHandlers.length / 4);
141         }
142         for (int i = 0; i < exnHandlers.length; i++) {
143           int v = exnHandlers[i];
144           if (v < 0 || v > 0xFFFF) {
145             throw new IllegalArgumentException("Invalid exception handler entry at " + i);
146           }
147         }
148     
149         this.exnHandlers = exnHandlers;
150       }
151     
152       /**
153        * Set the maximum number of local variable space used, in words, by this Code.
154        */
155       public void setMaxLocals(int maxLocals) {
156         this.maxLocals = maxLocals;
157       }
158     
159       /**
160        * Set the maximum stack size, in words, in this Code.
161        */
162       public void setMaxStack(int maxStack) {
163         this.maxStack = maxStack;
164       }
165     
166       /**
167        * Set the attributes of this Code.
168        */
169       public void setAttributes(ClassWriter.Element[] attributes) {
170         this.attributes = attributes;
171       }
172     }
