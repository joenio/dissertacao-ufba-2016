11     package com.ibm.wala.ipa.cfg;
12     
13     import java.util.Iterator;
14     
15     import com.ibm.wala.classLoader.IMethod;
16     import com.ibm.wala.ipa.callgraph.CGNode;
17     import com.ibm.wala.ssa.ISSABasicBlock;
18     import com.ibm.wala.ssa.SSAInstruction;

24     /**
25      * A helper class to make the ipcfg work correctly with context-sensitive call graphs.
26      */
27     public final class BasicBlockInContext<T extends ISSABasicBlock> extends NodeWithNumber implements ISSABasicBlock {
28       private final T delegate;
29     
30       private final CGNode node;
31     
32       public BasicBlockInContext(CGNode node, T bb) {
33         if (bb == null) {
34           throw new IllegalArgumentException("null bb");
35         }
36         this.delegate = bb;
37         this.node = node;
38       }
39     
40       /*
41        * @see com.ibm.wala.cfg.IBasicBlock#getFirstInstructionIndex()
42        */
43       @Override
44       public int getFirstInstructionIndex() {
45         return delegate.getFirstInstructionIndex();
46       }
47     
48       /*
49        * @see com.ibm.wala.cfg.IBasicBlock#getLastInstructionIndex()
50        */
51       @Override
52       public int getLastInstructionIndex() {
53         return delegate.getLastInstructionIndex();
54       }
55     
56       /*
57        * @see com.ibm.wala.cfg.IBasicBlock#iterateAllInstructions()
58        */
59       @Override
60       public Iterator<SSAInstruction> iterator() {
61         return delegate.iterator();
62       }
63     
64       /*
65        * @see com.ibm.wala.cfg.IBasicBlock#getMethod()
66        */
67       @Override
68       public IMethod getMethod() {
69         return delegate.getMethod();
70       }
71     
72       /*
73        * @see com.ibm.wala.cfg.IBasicBlock#getNumber()
74        */
75       @Override
76       public int getNumber() {
77         return delegate.getNumber();
78       }
79     
80       /*
81        * @see com.ibm.wala.cfg.IBasicBlock#isCatchBlock()
82        */
83       @Override
84       public boolean isCatchBlock() {
85         return delegate.isCatchBlock();
86       }
87     
88       /*
89        * @see com.ibm.wala.cfg.IBasicBlock#isEntryBlock()
90        */
91       @Override
92       public boolean isEntryBlock() {
93         return delegate.isEntryBlock();
94       }
95     
96       /*
97        * @see com.ibm.wala.cfg.IBasicBlock#isExitBlock()
98        */
99       @Override
100       public boolean isExitBlock() {
101         return delegate.isExitBlock();
102       }
103     
104       @Override
105       public int hashCode() {
106         final int prime = 31;
107         int result = 1;
108         result = prime * result + ((delegate == null) ? 0 : delegate.hashCode());
109         result = prime * result + ((node == null) ? 0 : node.hashCode());
110         return result;
111       }
112     
113       @Override
114       public boolean equals(Object obj) {
115         if (this == obj)
116           return true;
117         if (obj == null)
118           return false;
119         if (getClass() != obj.getClass())
120           return false;
121         final BasicBlockInContext other = (BasicBlockInContext) obj;
122         if (delegate == null) {
123           if (other.delegate != null)
124             return false;
125         } else if (!delegate.equals(other.delegate))
126           return false;
127         if (node == null) {
128           if (other.node != null)
129             return false;
130         } else if (!node.equals(other.node))
131           return false;
132         return true;
133       }
134     
135       public T getDelegate() {
136         return delegate;
137       }
138     
139       public CGNode getNode() {
140         return node;
141       }
142     
143       @Override
144       public String toString() {
145         return delegate.toString();
146       }
147     
148       @Override
149       public Iterator<TypeReference> getCaughtExceptionTypes() {
150         return delegate.getCaughtExceptionTypes();
151       }
152     
153       @Override
154       public SSAInstruction getLastInstruction() {
155         return delegate.getLastInstruction();
156       }
157     
158       @Override
159       public Iterator<SSAPhiInstruction> iteratePhis() {
160         return delegate.iteratePhis();
161       }
162     
163       @Override
164       public Iterator<SSAPiInstruction> iteratePis() {
165         return delegate.iteratePis();
166       }
167     
168     }
