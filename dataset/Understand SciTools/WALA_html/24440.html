45     /**
46      * Class files are taken as input arguments (or if there are none, from standard
47      * input). The methods in those files are instrumented: we insert a
48      * System.err.println() at ever method call, and a System.err.println() at every
49      * method entry.
50      * 
51      * The instrumented classes are placed in the directory "output" under the
52      * current directory. Disassembled code is written to the file "report" under
53      * the current directory.
54      * 
55      * @author CHammer
56      * @author Julian Dolby (dolby@us.ibm.com)
57      * @since 10/18
58      */
59     public class OfflineDynamicCallGraph {
60     	private final static boolean disasm = true;
61     	private final static boolean verify = true;
62     
63     	private static boolean patchExits = true;
64     	
65     	private static Class<?> runtime = Runtime.class;
66     	
67     	private static SetOfClasses filter;
68     	
69     	private static ClassHierarchyStore cha = new ClassHierarchyStore();
70     	
71     	public static void main(String[] args) throws IOException, ClassNotFoundException, InvalidClassFileException, FailureException {
72     	  OfflineInstrumenter instrumenter;
73     	  ClassInstrumenter ci;
74     	  Writer w = new BufferedWriter(new FileWriter("report", false));
75     
76     	   for(int i = 0; i < args.length - 1; i++) {
77     	      if ("--runtime".equals(args[i])) {
78     	        runtime = Class.forName(args[i+1]);
79     	      } else if ("--exclusions".equals(args[i])) {
80     	        filter = new FileOfClasses(new FileInputStream(args[i+1]));
81     	      } else if ("--dont-patch-exits".equals(args[i])) {
82     	        patchExits = false;
83     	      } else if ("--rt-jar".equals(args[i])) {
84     	        System.err.println("using " + args[i+1] + " as stdlib");
85     	        OfflineInstrumenter libReader = new OfflineInstrumenter(true);
86     	        libReader.addInputJar(new File(args[i+1]));
87     	        while ((ci = libReader.nextClass()) != null) {
88     	          CTUtils.addClassToHierarchy(cha, ci.getReader());
89     	        }
90     	      }
91     	    }
92     
93     	  instrumenter = new OfflineInstrumenter(true);
94     	  args = instrumenter.parseStandardArgs(args);
95     				  
96     	  instrumenter.setPassUnmodifiedClasses(true);
97     
98     	  instrumenter.beginTraversal();
99     	  while ((ci = instrumenter.nextClass()) != null) {
100     	    CTUtils.addClassToHierarchy(cha, ci.getReader());
101     	  }
102     
103     	  instrumenter.setClassHierarchyProvider(cha);
104     	  
105     	  instrumenter.beginTraversal();
106     	  while ((ci = instrumenter.nextClass()) != null) {
107     	    ClassWriter cw = doClass(ci, w);
108     	    if (cw != null) {
109     	      instrumenter.outputModifiedClass(ci, cw);
110     	    }
111     	  }
112     	  
113     	  instrumenter.close();
114     	}
115     
116     	static ClassWriter doClass(final ClassInstrumenter ci, Writer w) throws InvalidClassFileException, IOException, FailureException {
117     		final String className = ci.getReader().getName();
118         if (filter != null && filter.contains(className)) {
119           return null;
120         }
121         
122         if (disasm) {
123           w.write("Class: " + className + "\n");
124           w.flush();
125         }
126         
127     		final ClassReader r = ci.getReader();
128     		
129     		for (int m = 0; m < ci.getReader().getMethodCount(); m++) {
130     			final MethodData d = ci.visitMethod(m);
131     
132     			// d could be null, e.g., if the method is abstract or native
133     			if (d != null) {
134     		    if (filter != null && filter.contains(className + "." + ci.getReader().getMethodName(m))) {
135     		      return null;
136     		    }
137     
138     				if (disasm) {
139     	        w.write("Instrumenting " + ci.getReader().getMethodName(m) + " " + ci.getReader().getMethodType(m) + ":\n");
140     					w.write("Initial ShrikeBT code:\n");
141     					(new Disassembler(d)).disassembleTo(w);
142     					w.flush();
143     				}
144     
145     				if (verify) {
146     					Verifier v = new Verifier(d);
147     					// v.setClassHierarchy(cha);
148     					v.verify();
149     				}
150     
151     				final MethodEditor me = new MethodEditor(d);
152     				me.beginPass();
153     			
154     				final String theClass = r.getName();
155     				final String theMethod = r.getMethodName(m).concat(r.getMethodType(m));
156     				final boolean isConstructor = theMethod.contains("<init>");
157     				final boolean nonStatic = !java.lang.reflect.Modifier.isStatic(r.getMethodAccessFlags(m));
158     
159     				me.insertAtStart(new MethodEditor.Patch() {
160     				  @Override
161               public void emitTo(MethodEditor.Output w) {
162                 w.emit(ConstantInstruction.makeString(theClass));
163     				    w.emit(ConstantInstruction.makeString(theMethod));
164     				    if (nonStatic && !isConstructor)
165     				      w.emit(LoadInstruction.make(Constants.TYPE_Object, 0)); //load this
166     				    else
167                   w.emit(Util.makeGet(runtime, "NULL_TAG"));
168     				      // w.emit(ConstantInstruction.make(Constants.TYPE_null, null));
169     				    w.emit(Util.makeInvoke(runtime, "execution", new Class[] {String.class, String.class, Object.class}));
170     				  }
171     				});
172     
173     				if (patchExits) {
174     				  me.addMethodExceptionHandler(null, new MethodEditor.Patch() { 
175     				    @Override
176     				    public void emitTo(Output w) {
177     				      w.emit(ConstantInstruction.makeString(theClass));
178     				      w.emit(ConstantInstruction.makeString(theMethod));
179     				      //if (nonStatic)
180     				      //  w.emit(LoadInstruction.make(Constants.TYPE_Object, 0)); //load this
181     				      //else
182     				      w.emit(Util.makeGet(runtime, "NULL_TAG"));
183     				      // w.emit(ConstantInstruction.make(Constants.TYPE_null, null));
184     				      w.emit(ConstantInstruction.make(1)); // true
185     				      w.emit(Util.makeInvoke(runtime, "termination", new Class[] {String.class, String.class, Object.class, boolean.class}));
186     				      w.emit(ThrowInstruction.make(false));
187     				    }
188     				  });
189     
190     				  me.visitInstructions(new MethodEditor.Visitor() {
191     				    @Override
192     				    public void visitReturn(ReturnInstruction instruction) {
193     				      insertBefore(new MethodEditor.Patch() {
194     				        @Override
195     				        public void emitTo(MethodEditor.Output w) {
196     				          w.emit(ConstantInstruction.makeString(theClass));
197     				          w.emit(ConstantInstruction.makeString(theMethod));
198     				          if (nonStatic)
199     				            w.emit(LoadInstruction.make(Constants.TYPE_Object, 0)); //load this
200     				          else
201     				            w.emit(Util.makeGet(runtime, "NULL_TAG"));
202     				          // w.emit(ConstantInstruction.make(Constants.TYPE, null));
203     				          w.emit(ConstantInstruction.make(0)); // false
204     				          w.emit(Util.makeInvoke(runtime, "termination", new Class[] {String.class, String.class, Object.class, boolean.class}));
205     				        }
206     				      });
207     				    }
208     				  });
209     				}
210     
211     				// this updates the data d
212     				me.applyPatches();
213     
214     				if (disasm) {
215     					w.write("Final ShrikeBT code:\n");
216     					(new Disassembler(d)).disassembleTo(w);
217     					w.flush();
218     				}
219     
220     				if (verify) {
221     					Verifier v = new Verifier(d);
222               // v.setClassHierarchy(cha);
223     					v.verify();
224     				}
225     			}
226     		}
227     
228     		if (ci.isChanged()) {
229     		  ClassWriter cw = new ClassWriter() {
230     		    private final Map<Object, Integer> entries = HashMapFactory.make();
231     		    
232     		    {
233     		      ConstantPoolParser p = r.getCP();
234     		      for(int i = 1; i < p.getItemCount(); i++) {
235     		        switch (p.getItemType(i)) {
236     		        case CONSTANT_Integer:
237     		          entries.put(new Integer(p.getCPInt(i)), i);
238     		          break;
239     		        case CONSTANT_Long:
240                   entries.put(new Long(p.getCPLong(i)), i);
241                   break;
242     		        case CONSTANT_Float:
243                   entries.put(new Float(p.getCPFloat(i)), i);
244                   break;
245     		        case CONSTANT_Double:
246                   entries.put(new Double(p.getCPDouble(i)), i);
247                   break;
248     		        case CONSTANT_Utf8:
249     		          entries.put(p.getCPUtf8(i), i);
250     		          break;
251     		        case CONSTANT_String:
252                   entries.put(new CWStringItem(p.getCPString(i), CONSTANT_String), i);
253                   break;
254     		        case CONSTANT_Class:
255                   entries.put(new CWStringItem(p.getCPClass(i), CONSTANT_Class), i);
256                   break;
257     		        case CONSTANT_MethodHandle:
258     		        case CONSTANT_MethodType:
259     		        case CONSTANT_InvokeDynamic:
260     		        }
261     		      }
262     		    }
263     		    
264     		    private int findExistingEntry(Object o) {
265     		      if (entries.containsKey(o)) {
266     		        return entries.get(o);
267     		      } else {
268     		        return -1;
269     		      }
270     		    }
271     		    
272             @Override
273             protected int addCPEntry(Object o, int size) {
274               int entry = findExistingEntry(o);
275               if (entry != -1) {
276                 return entry;
277               } else {
278                 return super.addCPEntry(o, size);
279               }
280             }
281     		  };
282     			ci.emitClass(cw);
283     			return cw;
284     		
285     		} else {
286     		  return null;
287     		}
288     	}
289     
290     }
