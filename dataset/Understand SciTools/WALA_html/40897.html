29     /**
30      * This is a simple example WALA application.
31      * 
32      * For each method in the standard libraries, it maps the IMethod to the set of fields the method reads.
33      * 
34      * @author sfink
35      */
36     public class GetLoadedFields {
37     
38       private final static ClassLoader MY_CLASSLOADER = GetLoadedFields.class.getClassLoader();
39     
40       /**
41        * Use the 'GetLoadedFields' launcher to run this program with the appropriate classpath
42        * @throws InvalidClassFileException
43        */
44       public static void main(String[] args) throws IOException, ClassHierarchyException, InvalidClassFileException {
45         // build an analysis scope representing the standard libraries, excluding no classes
46         AnalysisScope scope = AnalysisScopeReader.readJavaScope("primordial.txt", null, MY_CLASSLOADER);
47     
48         // build a class hierarchy
49         System.err.print("Build class hierarchy...");
50         IClassHierarchy cha = ClassHierarchy.make(scope);
51         System.err.println("Done");
52     
53         int nMethods = 0;
54         int nFields = 0;
55     
56         Map<IMethod, Collection<FieldReference>> method2Field = HashMapFactory.make();
57         for (IClass c : cha) {
58           for (IMethod m : c.getDeclaredMethods()) {
59             nMethods++;
60             Collection<FieldReference> fields = CodeScanner.getFieldsRead(m);
61             nFields += fields.size();
62             method2Field.put(m, fields);
63           }
64         }
65         
66         System.out.println(nMethods + " methods");
67         System.out.println((float)nFields/(float)nMethods + " fields read per method");
68       }
69     }
