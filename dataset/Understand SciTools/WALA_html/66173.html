215     	public TypeReference[] getDeclaredExceptions()
216     			throws InvalidClassFileException, UnsupportedOperationException {
217     /** BEGIN Custom change: Variable Names in synth. methods */
218     		assert (eMethod.method != null);
219             if (myClass.getClassDefItem().getAnnotations() == null) {
220                 return null;
221             }
222             ArrayList<String> strings = new ArrayList<String>();
223             AnnotationSetItem annotationSet = myClass.getClassDefItem().getAnnotations().getMethodAnnotations(eMethod.method);
224     /** END Custom change: Variable Names in synth. methods */
225     
226     		if (annotationSet != null) {
227     			for (AnnotationItem annotationItem: annotationSet.getAnnotations())
228     			{
229     				if (annotationItem.getEncodedAnnotation().annotationType.getTypeDescriptor().contentEquals("Ldalvik/annotation/Throws;")) {
230     					for (int i = 0; i < annotationItem.getEncodedAnnotation().values.length; i++) {
231     						for (int j = 0; j < ((ArrayEncodedValue)annotationItem.getEncodedAnnotation().values[i]).values.length; j++) {
232     							String tname = ((TypeEncodedValue)((ArrayEncodedValue)annotationItem.getEncodedAnnotation().values[i]).values[j]).value.getTypeDescriptor();
233     							if (tname.endsWith(";"))
234     								tname = tname.substring(0,tname.length()-1);
235     							strings.add(tname);
236     						}
237     					}
238     				}
239     			}
240     		}
241     
242     		if (strings.size() == 0)
243     			return null;
244     
245     		ClassLoaderReference loader = getDeclaringClass().getClassLoader().getReference();
246     
247     		TypeReference[] result = new TypeReference[strings.size()];
248     		for (int i = 0; i < result.length; i++) {
249     			result[i] = TypeReference.findOrCreate(loader, TypeName.findOrCreate(ImmutableByteArray.make(strings.get(i))));
250     		}
251     		return result;
252     	}

3107     	private TypeReference findOutArrayElementType(
3108     			org.jf.dexlib.Code.Instruction[] instrucs, Instruction[] walaInstructions, int instCounter) {
3109     		if (instCounter < 0 || instrucs[instCounter].opcode != Opcode.FILL_ARRAY_DATA) {
3110     			throw new IllegalArgumentException();
3111     		} else if (instCounter == 0) {
3112     			throw new UnsupportedOperationException("fill-array-data as first instruction is not supported!");
3113     		}
3114     		Instruction31t arrayFill = (Instruction31t)instrucs[instCounter];
3115     		int interestingRegister = arrayFill.getRegisterA();
3116     		int curCounter = instCounter - 1;
3117     
3118     		while (curCounter >= 0) {
3119     			org.jf.dexlib.Code.Instruction curInst = instrucs[curCounter];
3120     			// do we have a 'new-array'-instruction, where the destination register coincides with the current interesting register?
3121     			// then we return the element type of that array
3122     			if (curInst.opcode == Opcode.NEW_ARRAY) {
3123     				Instruction22c newArrayInst = (Instruction22c) curInst;
3124     				if (newArrayInst.getRegisterA() == interestingRegister) {
3125     					NewArray newArray = (NewArray) walaInstructions[curCounter];
3126     					return newArray.newSiteRef.getDeclaredType().getArrayElementType();
3127     				}
3128     			} else if (curInst.opcode == Opcode.MOVE_OBJECT || curInst.opcode == Opcode.MOVE_OBJECT_16 || curInst.opcode == Opcode.MOVE_OBJECT_FROM16) {
3129     				TwoRegisterInstruction tri = (TwoRegisterInstruction) curInst;
3130     				int regA = tri.getRegisterA();
3131     				int regB = tri.getRegisterB();
3132     				if (regA == interestingRegister) {
3133     					interestingRegister = regB;
3134     				}
3135     			}
3136     			// all other instructions are ignored
3137     			curCounter--;
3138     		}
3139     
3140     		throw new UnsupportedOperationException("found a fill-array-data instruction without a corresponding new-array instruction. This should not happen!");
3141     	}
