11     package com.ibm.wala.ipa.slicer;
12     
13     import com.ibm.wala.ipa.callgraph.CGNode;
14     
15     /**
16      * A statement that has a corresponding index in the SSA IR
17      */
18     public class NormalStatement extends StatementWithInstructionIndex {
19     
20       public NormalStatement(CGNode node, int instructionIndex) {
21         super(node, instructionIndex);
22       }
23     
24       @Override
25       public Kind getKind() {
26         return Kind.NORMAL;
27       }
28     
29       @Override
30       public String toString() {
31         String name = "";
32         if (getInstruction().hasDef()) {
33           String[] names = getNode().getIR().getLocalNames(getInstructionIndex(), getInstruction().getDef());
34           if (names != null && names.length > 0) {
35             name = "[" + names[0];
36             for (int i = 1; i < names.length; i++) {
37               name = name + ", " + names[i];
38             }
39             name = name + "]: ";
40           }
41         }
42     
43         return "NORMAL " + getNode().getMethod().getName() + ":" + name + getInstruction().toString() + " " + getNode();
44       }
45     
46     }
