56     /**
57      * A context interpreter for java.lang.Object.clone
58      * 
59      * TODO: The current implementation does not model CloneNotSupportedExceptions

61     public class CloneInterpreter implements SSAContextInterpreter {

63       /**
64        * Comment for <code>cloneAtom</code>
65        */
66       public final static Atom cloneAtom = Atom.findOrCreateUnicodeAtom("clone");
67     
68       private final static Descriptor cloneDesc = Descriptor.findOrCreateUTF8("()Ljava/lang/Object;");
69     
70       /**
71        * Comment for <code>CLONE</code>
72        */
73       public final static MethodReference CLONE = MethodReference.findOrCreate(TypeReference.JavaLangObject, cloneAtom, cloneDesc);
74     
75       private final static TypeReference SYNTHETIC_SYSTEM = TypeReference.findOrCreate(ClassLoaderReference.Primordial, TypeName
76           .string2TypeName("Lcom/ibm/wala/model/java/lang/System"));
77     
78       private final static Atom arraycopyAtom = Atom.findOrCreateUnicodeAtom("arraycopy");
79     
80       private final static Descriptor arraycopyDesc = Descriptor.findOrCreateUTF8("(Ljava/lang/Object;Ljava/lang/Object;)V");
81     
82       private final static MethodReference SYNTHETIC_ARRAYCOPY = MethodReference.findOrCreate(SYNTHETIC_SYSTEM, arraycopyAtom,
83           arraycopyDesc);

85       /**
86        * If the type is an array, the program counter of the synthesized call to arraycopy. Doesn't really matter what it is.
87        */
88       private final static int ARRAYCOPY_PC = 3;
89     
90       private final static CallSiteReference ARRAYCOPY_SITE = CallSiteReference.make(ARRAYCOPY_PC, SYNTHETIC_ARRAYCOPY,
91           IInvokeInstruction.Dispatch.STATIC);

93       private final static int NEW_PC = 0;

96        * Mapping from TypeReference -> IR TODO: Soft references?

98       final private Map<TypeReference, IR> IRCache = HashMapFactory.make();

100       private final SSAInstructionFactory insts = Language.JAVA.instructionFactory();

102       @Override
103       public IR getIR(CGNode node) {
104         if (node == null) {
105           throw new IllegalArgumentException("node is null");
106         }
107         assert understands(node);
108         IClass cls = ContextUtil.getConcreteClassFromContext(node.getContext());
109         IR result = IRCache.get(cls.getReference());
110         if (result == null) {
111           result = makeIR(node.getMethod(), node.getContext(), cls);
112           IRCache.put(cls.getReference(), result);
113         }
114         return result;
115       }

117       @Override
118       public int getNumberOfStatements(CGNode node) {
119         assert understands(node);
120         return getIR(node).getInstructions().length;
121       }
122     
123       @Override
124       public boolean understands(CGNode node) {
125         if (node == null) {
126           throw new IllegalArgumentException("node is null");
127         }
128         return (node.getMethod().getReference().equals(CLONE) && ContextUtil.getConcreteClassFromContext(node.getContext()) != null);
129       }

131       @Override
132       public Iterator<NewSiteReference> iterateNewSites(CGNode node) {
133         if (node == null) {
134           throw new IllegalArgumentException("node is null");
135         }
136         assert understands(node);
137         IClass cls = ContextUtil.getConcreteClassFromContext(node.getContext());
138         return new NonNullSingletonIterator<NewSiteReference>(NewSiteReference.make(NEW_PC, cls.getReference()));
139       }
140     
141       @Override
142       public Iterator<CallSiteReference> iterateCallSites(CGNode node) {
143         assert understands(node);
144         return new NonNullSingletonIterator<CallSiteReference>(ARRAYCOPY_SITE);
145       }

147       /**
148        * @return an array of statements that encode the behavior of the clone method for a given type.
149        */
150       private SSAInstruction[] makeStatements(IClass klass) {
151         assert klass != null;
152     
153         ArrayList<SSAInstruction> statements = new ArrayList<SSAInstruction>();
154         // value number 1 is "this".
155         int nextLocal = 2;
156     
157         int retValue = nextLocal++;
158         // value number of the result of the clone()
159         NewSiteReference ref = NewSiteReference.make(NEW_PC, klass.getReference());
160         SSANewInstruction N = null;
161         if (klass.isArrayClass()) {
162           int length = nextLocal++;
163           statements.add(insts.ArrayLengthInstruction(statements.size(), length, 1));
164           int[] sizes = new int[((ArrayClass)klass).getDimensionality()];
165           Arrays.fill(sizes, length);
166           N = insts.NewInstruction(statements.size(), retValue, ref, sizes);
167         } else {
168           N = insts.NewInstruction(statements.size(), retValue, ref);
169         }
170         statements.add(N);
171     
172         int exceptionValue = nextLocal++;
173     
174         if (klass.getReference().isArrayType()) {
175           // generate a synthetic arraycopy from this (v.n. 1) to the clone
176           int[] params = new int[2];
177           params[0] = 1;
178           params[1] = retValue;
179           SSAInvokeInstruction S = insts.InvokeInstruction(statements.size(), params, exceptionValue, ARRAYCOPY_SITE, null);
180           statements.add(S);
181         } else {
182           // copy the fields over, one by one.
183           // TODO:
184           IClass k = klass;
185           while (k != null) {
186             for (Iterator<IField> it = klass.getDeclaredInstanceFields().iterator(); it.hasNext();) {
187               IField f = it.next();
188               int tempValue = nextLocal++;
189               SSAGetInstruction G = insts.GetInstruction(statements.size(), tempValue, 1, f.getReference());
190               statements.add(G);
191               SSAPutInstruction P = insts.PutInstruction(statements.size(), retValue, tempValue, f.getReference());
192               statements.add(P);
193             }
194             k = k.getSuperclass();
195           }
196     
197         }
198     
199         SSAReturnInstruction R = insts.ReturnInstruction(statements.size(), retValue, false);
200         statements.add(R);
201     
202         SSAInstruction[] result = new SSAInstruction[statements.size()];
203         Iterator<SSAInstruction> it = statements.iterator();
204         for (int i = 0; i < result.length; i++) {
205           result[i] = it.next();
206         }
207         return result;
208       }

210       /**
211        * @return an IR that encodes the behavior of the clone method for a given type.
212        */
213       private IR makeIR(IMethod method, Context context, IClass klass) {
214         assert klass != null;
215         SSAInstruction instrs[] = makeStatements(klass);
216         return new SyntheticIR(method, context, new InducedCFG(instrs, method, context), instrs, SSAOptions.defaultOptions(), null);
217       }
