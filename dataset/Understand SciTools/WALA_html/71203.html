33     /**
34      * An {@link InstanceKeyFactory} that returns {@link ScopeMappingInstanceKey}s
35      * as necessary to handle interprocedural lexical scoping (specifically, to
36      * handle closure creation when a function escapes its allocating scope)
37      */
38     abstract public class ScopeMappingInstanceKeys implements InstanceKeyFactory {
39     
40       /**
41        * does base require a scope mapping key? Typically, true if base is allocated
42        * in a nested lexical scope, to handle the case of base being a function that
43        * performs closure accesses
44        */
45       protected abstract boolean needsScopeMappingKey(InstanceKey base);
46     
47       protected final PropagationCallGraphBuilder builder;
48     
49       private final InstanceKeyFactory basic;
50     
51       /**
52        * An {@link InstanceKey} carrying information about which {@link CGNode}s
53        * represent lexical parents of the allocating {@link CGNode}.
54        * 
55        * The fact that we discover at most one {@link CGNode} per lexical parent
56        * relies on the following property: in a call graph, the contexts used for a
57        * nested function can be finer than those used for the containing function,
58        * but _not_ coarser. This ensures that there is at most one CGNode
59        * corresponding to a lexical parent (e.g., we don't get two clones of
60        * function f1() invoking a single CGNode representing nested function f2())
61        * 
62        * Note that it is possible to not find a {@link CGNode} corresponding to some
63        * lexical parent; this occurs when a deeply nested function is returned
64        * before being invoked, so some lexical parent is no longer on the call stack
65        * when the function is allocated. See test case nested.js.
66        */
67       public class ScopeMappingInstanceKey implements InstanceKey {
68         /**
69          * the underlying instance key
70          */
71         private final InstanceKey base;
72     
73         /**
74          * the node in which base is allocated
75          */
76         private final CGNode creator;
77     
78         private ScopeMappingInstanceKey(CGNode creator, InstanceKey base) {
79           this.creator = creator;
80           this.base = base;
81         }
82     
83         @Override
84         public IClass getConcreteType() {
85           return base.getConcreteType();
86         }
87     
88         /**
89          * get the CGNode representing the lexical parent of {@link #creator} with
90          * name definer
91          * 
92          * @param definer
93          * @return
94          */
95         public Iterator<CGNode> getFunargNodes(Pair<String, String> name) {
96           Collection<CGNode> constructorCallers = getConstructorCallers(this, name);
97           assert constructorCallers != null && !constructorCallers.isEmpty() : "no callers for constructor";
98           Iterator<CGNode> result = EmptyIterator.instance();
99           for (CGNode callerOfConstructor : constructorCallers) {
100             if (callerOfConstructor.getMethod().getReference().getDeclaringClass().getName().toString().equals(name.snd)) {
101               result = new CompoundIterator<CGNode>(result, new NonNullSingletonIterator<CGNode>(callerOfConstructor));
102             } else {
103               PointerKey funcKey = builder.getPointerKeyForLocal(callerOfConstructor, 1);
104               for (InstanceKey funcPtr : builder.getPointerAnalysis().getPointsToSet(funcKey)) {
105                 if (funcPtr instanceof ScopeMappingInstanceKey) {
106                   result = new CompoundIterator<CGNode>(result, ((ScopeMappingInstanceKey) funcPtr).getFunargNodes(name));
107                 }
108               }
109             }
110           }
111           return result;
112         }
113     
114     
115         @Override
116         public int hashCode() {
117           return base.hashCode() * creator.hashCode();
118         }
119     
120         @Override
121         public boolean equals(Object o) {
122           return (o instanceof ScopeMappingInstanceKey) && ((ScopeMappingInstanceKey) o).base.equals(base)
123               && ((ScopeMappingInstanceKey) o).creator.equals(creator);
124         }
125     
126         @Override
127         public String toString() {
128           return "SMIK:" + base + "@creator:" + creator;
129         }
130     
131         public InstanceKey getBase() {
132           return base;
133         }
134         
135         public CGNode getCreator() {
136           return creator;
137         }
138     
139         @Override
140         public Iterator<Pair<CGNode, NewSiteReference>> getCreationSites(CallGraph CG) {
141           return new FilterIterator<Pair<CGNode, NewSiteReference>>(
142               base.getCreationSites(CG),
143               new Predicate<Pair<CGNode, NewSiteReference>>() {
144                 @Override public boolean test(Pair<CGNode, NewSiteReference> o) {
145                   return o.fst.equals(creator);
146                 }
147               });
148         }
149       }
150     
151       @Override
152       public InstanceKey getInstanceKeyForAllocation(CGNode creatorNode, NewSiteReference allocationSite) {
153         InstanceKey base = basic.getInstanceKeyForAllocation(creatorNode, allocationSite);
154         if (base != null && needsScopeMappingKey(base)) {
155           return new ScopeMappingInstanceKey(creatorNode, base);
156         } else {
157           return base;
158         }
159       }
160     
161       /**
162        * get the CGNodes corresponding to the method that invoked the constructor for smik
163        */
164       protected abstract Collection<CGNode> getConstructorCallers(ScopeMappingInstanceKey smik, Pair<String, String> name);
165       
166       @Override
167       public InstanceKey getInstanceKeyForMultiNewArray(CGNode node, NewSiteReference allocation, int dim) {
168         return basic.getInstanceKeyForMultiNewArray(node, allocation, dim);
169       }
170     
171       @Override
172       public InstanceKey getInstanceKeyForConstant(TypeReference type, Object S) {
173         return basic.getInstanceKeyForConstant(type, S);
174       }
175     
176       @Override
177       public InstanceKey getInstanceKeyForPEI(CGNode node, ProgramCounter instr, TypeReference type) {
178         return basic.getInstanceKeyForPEI(node, instr, type);
179       }
180     
181       @Override
182       public InstanceKey getInstanceKeyForMetadataObject(Object obj, TypeReference objType) {
183         return basic.getInstanceKeyForMetadataObject(obj, objType);
184       }
185     
186       public ScopeMappingInstanceKeys(PropagationCallGraphBuilder builder, InstanceKeyFactory basic) {
187         this.basic = basic;
188         this.builder = builder;
189       }
190     }
