11     package com.ibm.wala.cast.js.translator;
12     
13     import com.ibm.wala.cast.ir.translator.TranslatorToCAst;
14     import com.ibm.wala.cast.tree.CAst;
15     import com.ibm.wala.cast.tree.rewrite.AstLoopUnwinder;
16     import com.ibm.wala.cast.tree.rewrite.AstLoopUnwinder.UnwindKey;
17     import com.ibm.wala.cast.tree.rewrite.CAstRewriter;
18     import com.ibm.wala.cast.tree.rewrite.CAstRewriter.RewriteContext;
19     import com.ibm.wala.cast.tree.rewrite.CAstRewriterFactory;
20     import com.ibm.wala.classLoader.SourceFileModule;
21     import com.ibm.wala.classLoader.SourceModule;
22     
23     public abstract class JavaScriptLoopUnwindingTranslatorFactory 
24       implements JavaScriptTranslatorFactory 
25     {
26       private final int unwindFactor;
27     
28       protected JavaScriptLoopUnwindingTranslatorFactory(int unwindFactor) {
29         this.unwindFactor = unwindFactor;
30       }
31     
32       JavaScriptLoopUnwindingTranslatorFactory() {
33         this(3);
34       }
35     
36       protected abstract TranslatorToCAst translateInternal(CAst Ast, SourceModule M, String N);
37     
38       @Override
39       public TranslatorToCAst make(CAst ast, final SourceModule M) {
40     	  String N;
41     	  if (M instanceof SourceFileModule) {
42     		  N = ((SourceFileModule) M).getClassName();
43     	  } else {
44     	      N = M.getName();
45     	  }
46     
47     	  TranslatorToCAst xlator = translateInternal(ast, M, N);
48     	  xlator.addRewriter(new CAstRewriterFactory<CAstRewriter.RewriteContext<AstLoopUnwinder.UnwindKey>,AstLoopUnwinder.UnwindKey>() {
49     		@Override
50         public CAstRewriter<RewriteContext<UnwindKey>, UnwindKey> createCAstRewriter(CAst ast) {
51     			return new AstLoopUnwinder(ast, true, unwindFactor);
52     		}		  
53     	  }, false);
54     	  
55     	  return xlator;
56       }
57     }
