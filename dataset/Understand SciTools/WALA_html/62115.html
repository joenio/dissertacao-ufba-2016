36     public abstract class DynamicDalvikComparisonTest extends DalvikCallGraphTestBase {
37     
38     	protected void test(URI[] androidLibs, String mainClass, String javaScopeFile, String... args) throws ClassHierarchyException, IllegalArgumentException, IOException, CancelException, InterruptedException, ClassNotFoundException, SecurityException, NoSuchMethodException, IllegalAccessException, InvocationTargetException, InvalidClassFileException, FailureException {
39     		AnalysisScope javaScope = CallGraphTestUtil.makeJ2SEAnalysisScope(javaScopeFile, CallGraphTestUtil.REGRESSION_EXCLUSIONS);
40     		String javaJarPath = getJavaJar(javaScope);
41     		File androidDex = convertJarToDex(javaJarPath);
42     		Pair<CallGraph,PointerAnalysis<InstanceKey>> android = makeDalvikCallGraph(androidLibs, null, mainClass, androidDex.getAbsolutePath());
43     
44     		dynamicCG(new File(javaJarPath), mainClass, args);
45     		
46     		checkEdges(android.fst, new Predicate<MethodReference>() {
47     		  @Override
48     		  public boolean test(MethodReference t) {
49     		    return t.getDeclaringClass().getClassLoader().equals(ClassLoaderReference.Application);
50     		  }
51     		});
52     	}
53     	
54       protected File testFile(String file) throws IOException {
55         File inputFile = TemporaryFile.urlToFile(file, getClass().getClassLoader().getResource(file));
56         inputFile.deleteOnExit();
57         return inputFile;
58       }
