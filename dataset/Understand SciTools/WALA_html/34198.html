12     package com.ibm.wala.cast.tree.pattern;
13     
14     import com.ibm.wala.cast.tree.CAstNode;
15     
16     /**
17      * A node pattern that matches an AST node of a certain kind; additionally, the node's children
18      * have to match the pattern's child patterns.
19      * 
20      * @author mschaefer
21      *
22      */
23     public class NodeOfKind implements NodePattern {
24     	protected int kind;
25     	protected NodePattern[] children;
26     	
27     	public NodeOfKind(int kind, NodePattern... children) {
28     		this.kind = kind;
29     		this.children = new NodePattern[children.length];
30     		for(int i=0;i<children.length;++i)
31     			this.children[i] = children[i];
32     	}
33     	
34     	/* (non-Javadoc)
35     	 * @see pattern.NodePattern#matches(com.ibm.wala.cast.tree.CAstNode)
36     	 */
37     	@Override
38       public boolean matches(CAstNode node) {
39     		if(node == null || node.getKind() != kind || node.getChildCount() != children.length)
40     			return false;
41     		for(int i=0;i<children.length;++i)
42     			if(!children[i].matches(node.getChild(i)))
43     				return false;
44     		return true;
45     	}
46     }
