38     public class MultiNewArrayTest extends WalaTestCase {
39     
40       private static final ClassLoader MY_CLASSLOADER = MultiNewArrayTest.class.getClassLoader();
41       
42       @Test public void testMultiNewArray1() throws IOException, ClassHierarchyException {
43         AnalysisScope scope = null;
44         scope = AnalysisScopeReader.readJavaScope(TestConstants.WALA_TESTDATA, (new FileProvider()).getFile("J2SEClassHierarchyExclusions.txt"), MY_CLASSLOADER);
45         ClassHierarchy cha = ClassHierarchy.make(scope);
46         IClass klass = cha.lookupClass(TypeReference.findOrCreate(ClassLoaderReference.Application, TestConstants.MULTI_DIM_MAIN));
47         Assert.assertTrue(klass != null);
48         IMethod m = klass.getMethod(Selector.make(Language.JAVA, "testNewMultiArray()V"));
49         Assert.assertTrue(m != null);
50         AnalysisCache cache = new AnalysisCache();
51         IR ir = cache.getIRFactory().makeIR(m, Everywhere.EVERYWHERE, new SSAOptions());
52         Assert.assertTrue(ir != null);
53         SSAInstruction[] instructions = ir.getInstructions();
54         for (SSAInstruction instr : instructions) {
55           if (instr instanceof SSANewInstruction) {
56             System.err.println(instr.toString(ir.getSymbolTable()));
57             Assert.assertTrue(instr.getNumberOfUses() == 2);
58             Assert.assertTrue(ir.getSymbolTable().getIntValue(instr.getUse(0)) == 3);
59             Assert.assertTrue(ir.getSymbolTable().getIntValue(instr.getUse(1)) == 4);
60           }
61         }
62       }
63     }
