11     package com.ibm.wala.ssa;
12     
13     import com.ibm.wala.classLoader.CallSiteReference;
14     import com.ibm.wala.shrikeBT.IInvokeInstruction;
15     import com.ibm.wala.types.TypeReference;
16     
17     /**
18      */
19     public abstract class SSAInvokeInstruction extends SSAAbstractInvokeInstruction {
20     
21       protected final int result;
22     
23       /**
24        * The value numbers of the arguments passed to the call. For non-static methods, params[0] == this. If params == null, this
25        * should be a static method with no parameters.
26        */
27       protected final int[] params;
28     
29       protected SSAInvokeInstruction(int iindex, int result, int[] params, int exception, CallSiteReference site) {
30         super(iindex, exception, site);
31         this.result = result;
32         this.params = params;
33         assertParamsKosher(result, params, site);
34       }
35     
36       /**
37        * Constructor InvokeInstruction. This case for void return values
38        */
39       protected SSAInvokeInstruction(int iindex, int[] params, int exception, CallSiteReference site) {
40         this(iindex, -1, params, exception, site);
41       }
42     
43       @Override
44       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
45         // result == -1 for void-returning methods, which are the only calls
46         // that have a single value def.
47         return insts.InvokeInstruction(iindex, defs == null || result == -1 ? result : defs[0], uses == null ? params : uses,
48             defs == null ? exception : defs[result == -1 ? 0 : 1], site, null);
49       }
50     
51       public static void assertParamsKosher(int result, int[] params, CallSiteReference site) throws IllegalArgumentException {
52         if (site == null) {
53           throw new IllegalArgumentException("site cannot be null");
54         }
55         if (site.getDeclaredTarget().getReturnType().equals(TypeReference.Void)) {
56           if (result != -1) {
57             assert result == -1 : "bogus call to " + site;
58           }
59         }
60     
61         int nExpected = 0;
62         if (!site.isStatic()) {
63           nExpected++;
64         }
65     
66         nExpected += site.getDeclaredTarget().getNumberOfParameters();
67         if (nExpected > 0) {
68           if (params == null) {
69             assert params != null : "null params for " + site;
70           }
71           if (params.length != nExpected) {
72             assert params.length == nExpected : "wrong number of params for " + site + " Expected " + nExpected + " got "
73                 + params.length;
74           }
75         }
76       }
77     
78       /**
79        * @see com.ibm.wala.ssa.SSAInstruction#visit(IVisitor)
80        * @throws IllegalArgumentException if v is null
81        */
82       @Override
83       public void visit(IVisitor v) {
84         if (v == null) {
85           throw new IllegalArgumentException("v is null");
86         }
87         v.visitInvoke(this);
88       }
89     
90       /**
91        * @see com.ibm.wala.ssa.SSAInstruction#getNumberOfUses()
92        */
93       @Override
94       public int getNumberOfUses() {
95         if (params == null) {
96           assert site.getInvocationCode() == IInvokeInstruction.Dispatch.STATIC
97               || site.getInvocationCode() == IInvokeInstruction.Dispatch.SPECIAL;
98           assert site.getDeclaredTarget().getNumberOfParameters() == 0;
99           return 0;
100         } else {
101           return params.length;
102         }
103       }
104     
105       @Override
106       public int getNumberOfParameters() {
107         return getNumberOfUses();
108       }
109     
110       @Override
111       public int getNumberOfReturnValues() {
112         return (result == -1) ? 0 : 1;
113       }
114     
115       @Override
116       public int getReturnValue(int i) {
117         assert i == 0 : "i != 0";
118         assert (result != -1) : "SSA-Result is -1";
119         return result;
120       }
121     
122       /**
123        * @see com.ibm.wala.ssa.SSAInstruction#getUse(int)
124        */
125       @Override
126       public int getUse(int j) {
127         if (params == null) {
128           assert false : "Invalid getUse: " + j + " , null params " + this;
129         }
130         if (j >= params.length) {
131             throw new ArrayIndexOutOfBoundsException("Invalid getUse: " + this + ", index " + j + ", params.length " + params.length);
132         }
133         if (j < 0) {
134             throw new ArrayIndexOutOfBoundsException("j may not be negative! In getUse "  + this + ", index " + j + ", params.length " + params.length);
135         }
136         return params[j];
137       }
138     
139       @Override
140       public int hashCode() {
141         return (site.hashCode() * 7529) + (exception * 9823);
142       }
143     
144     }
