24     /**
25      * This class provides instance keys where for a given type T in a CGNode N, there is one "abstract allocation site" instance for
26      * all T allocations in node N.
27      */
28     public class SmushedAllocationSiteInstanceKeys implements InstanceKeyFactory {
29     
30       /**
31        * Governing call graph construction options
32        */
33       private final AnalysisOptions options;
34     
35       /**
36        * Governing class hierarchy
37        */
38       private final IClassHierarchy cha;
39     
40       private final ClassBasedInstanceKeys classBased;
41     
42       /**
43        * @param options Governing call graph construction options
44        */
45       public SmushedAllocationSiteInstanceKeys(AnalysisOptions options, IClassHierarchy cha) {
46         this.options = options;
47         this.cha = cha;
48         this.classBased = new ClassBasedInstanceKeys(options, cha);
49       }
50     
51       @Override
52       public InstanceKey getInstanceKeyForAllocation(CGNode node, NewSiteReference allocation) {
53         IClass type = options.getClassTargetSelector().getAllocatedTarget(node, allocation);
54         if (type == null) {
55           return null;
56         }
57     
58         // disallow recursion in contexts.
59         if (node.getContext() instanceof ReceiverInstanceContext) {
60           IMethod m = node.getMethod();
61           CGNode n = ContainerContextSelector.findNodeRecursiveMatchingContext(m, node.getContext());
62           if (n != null) {
63             return new SmushedAllocationSiteInNode(n, type);
64           }
65         }
66     
67         InstanceKey key = new SmushedAllocationSiteInNode(node, type);
68     
69         return key;
70       }
71     
72       @Override
73       public InstanceKey getInstanceKeyForMultiNewArray(CGNode node, NewSiteReference allocation, int dim) {
74         ArrayClass type = (ArrayClass) options.getClassTargetSelector().getAllocatedTarget(node, allocation);
75         if (type == null) {
76           return null;
77         }
78         InstanceKey key = new MultiNewArrayInNode(node, allocation, type, dim);
79     
80         return key;
81       }
82     
83       @Override
84       public <T> InstanceKey getInstanceKeyForConstant(TypeReference type, T S) {
85         if (options.getUseConstantSpecificKeys())
86           return new ConstantKey<T>(S, cha.lookupClass(type));
87         else
88           return new ConcreteTypeKey(cha.lookupClass(type));
89       }
90     
91       @Override
92       public InstanceKey getInstanceKeyForPEI(CGNode node, ProgramCounter pei, TypeReference type) {
93         return classBased.getInstanceKeyForPEI(node, pei, type);
94       }
95     
96       @Override
97       public InstanceKey getInstanceKeyForMetadataObject(Object obj, TypeReference objType) {
98         return classBased.getInstanceKeyForMetadataObject(obj, objType);
99       }
100     
101     }
