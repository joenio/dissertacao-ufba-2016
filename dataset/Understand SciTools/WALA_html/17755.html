33     /**
34      * Control Dependence Graph
35      */
36     public class ControlDependenceGraph<I, T extends IBasicBlock<I>> extends AbstractNumberedGraph<T> {
37     
38       /**
39        * Governing control flow-graph. The control dependence graph is computed from this cfg.
40        */
41       private final ControlFlowGraph<I, T> cfg;
42     
43       /**
44        * the EdgeManager for the CDG. It implements the edge part of the standard Graph abstraction, using the control-dependence edges
45        * of the cdg.
46        */
47       private final NumberedEdgeManager<T> edgeManager;
48     
49       /**
50        * If requested, this is a map from parentXchild Pairs representing edges in the CDG to the labels of the control flow edges that
51        * edge corresponds to. The labels are Boolean.True or Boolean.False for conditionals and an Integer for a switch label.
52        */
53       private Map<Pair, Set<Object>> edgeLabels;
54     
55       /**
56        * This is the heart of the CDG computation. Based on Cytron et al., this is the reverse dominance frontier based algorithm for
57        * computing control dependence edges.
58        * 
59        * @return Map: node n -> {x : n is control-dependent on x}
60        */
61       private Map<T, Set<T>> buildControlDependence(boolean wantEdgeLabels) {
62         Map<T, Set<T>> controlDependence = HashMapFactory.make(cfg.getNumberOfNodes());
63     
64         DominanceFrontiers<T> RDF = new DominanceFrontiers<T>(GraphInverter.invert(cfg), cfg.exit());
65     
66         if (wantEdgeLabels) {
67           edgeLabels = HashMapFactory.make();
68         }
69     
70         for (Iterator<? extends T> ns = cfg.iterator(); ns.hasNext();) {
71           HashSet<T> s = HashSetFactory.make(2);
72           controlDependence.put(ns.next(), s);
73         }
74     
75         for (Iterator<? extends T> ns = cfg.iterator(); ns.hasNext();) {
76           T y = ns.next();
77           for (Iterator<T> ns2 = RDF.getDominanceFrontier(y); ns2.hasNext();) {
78             T x = ns2.next();
79             controlDependence.get(x).add(y);
80             if (wantEdgeLabels) {
81               Set<Object> labels = HashSetFactory.make();
82               edgeLabels.put(Pair.make(x, y), labels);
83               for (Iterator<? extends T> ss = cfg.getSuccNodes(x); ss.hasNext();) {
84                 T s = ss.next();
85                 if (RDF.isDominatedBy(s, y)) {
86                   labels.add(s);
87                 }
88               }
89             }
90           }
91         }
92     
93         return controlDependence;
94       }
95     
96       /**
97        * Given the control-dependence edges in a forward direction (i.e. edges from control parents to control children), this method
98        * creates an EdgeManager that provides the edge half of the Graph abstraction.
99        */
100       private NumberedEdgeManager<T> constructGraphEdges(final Map<T, Set<T>> forwardEdges) {
101         return new NumberedEdgeManager<T>() {
102           Map<T, Set<T>> backwardEdges = HashMapFactory.make(forwardEdges.size());
103           {
104             for (Iterator<? extends T> x = cfg.iterator(); x.hasNext();) {
105               Set<T> s = HashSetFactory.make();
106               backwardEdges.put(x.next(), s);
107             }
108             for (Iterator<T> ps = forwardEdges.keySet().iterator(); ps.hasNext();) {
109               T p = ps.next();
110               for (Iterator ns = ((Set) forwardEdges.get(p)).iterator(); ns.hasNext();) {
111                 Object n = ns.next();
112                 backwardEdges.get(n).add(p);
113               }
114             }
115           }
116     
117           @Override
118           public Iterator<T> getPredNodes(T N) {
119             if (backwardEdges.containsKey(N))
120               return backwardEdges.get(N).iterator();
121             else
122               return EmptyIterator.instance();
123           }
124     
125           @Override
126           public IntSet getPredNodeNumbers(T node) {
127             MutableIntSet x = IntSetUtil.make();
128             if (backwardEdges.containsKey(node)) {
129               for(T pred : backwardEdges.get(node)) {
130                 x.add(pred.getNumber());
131               }
132             }
133             return x;
134           }
135     
136           @Override
137           public int getPredNodeCount(T N) {
138             if (backwardEdges.containsKey(N))
139               return ((Set) backwardEdges.get(N)).size();
140             else
141               return 0;
142           }
143     
144           @Override
145           public Iterator<T> getSuccNodes(T N) {
146             if (forwardEdges.containsKey(N))
147               return forwardEdges.get(N).iterator();
148             else
149               return EmptyIterator.instance();
150           }
151     
152           @Override
153           public IntSet getSuccNodeNumbers(T node) {
154             MutableIntSet x = IntSetUtil.make();
155             if (forwardEdges.containsKey(node)) {
156               for(T succ : forwardEdges.get(node)) {
157                 x.add(succ.getNumber());
158               }
159             }
160             return x;
161           }
162     
163           @Override
164           public int getSuccNodeCount(T N) {
165             if (forwardEdges.containsKey(N))
166               return ((Set) forwardEdges.get(N)).size();
167             else
168               return 0;
169           }
170     
171           @Override
172           public boolean hasEdge(T src, T dst) {
173             return forwardEdges.containsKey(src) && ((Set) forwardEdges.get(src)).contains(dst);
174           }
175     
176           @Override
177           public void addEdge(T src, T dst) {
178             throw new UnsupportedOperationException();
179           }
180     
181           @Override
182           public void removeEdge(T src, T dst) {
183             throw new UnsupportedOperationException();
184           }
185     
186           @Override
187           public void removeAllIncidentEdges(T node) {
188             throw new UnsupportedOperationException();
189           }
190     
191           @Override
192           public void removeIncomingEdges(T node) {
193             throw new UnsupportedOperationException();
194           }
195     
196           @Override
197           public void removeOutgoingEdges(T node) {
198             throw new UnsupportedOperationException();
199           }
200         };
201       }
202     
203       @Override
204       public String toString() {
205         StringBuffer sb = new StringBuffer();
206         for (Iterator<? extends T> ns = iterator(); ns.hasNext();) {
207           T n = ns.next();
208           sb.append(n.toString()).append("\n");
209           for (Iterator ss = getSuccNodes(n); ss.hasNext();) {
210             Object s = ss.next();
211             sb.append("  --> ").append(s);
212             if (edgeLabels != null)
213               for (Iterator labels = ((Set) edgeLabels.get(Pair.make(n, s))).iterator(); labels.hasNext();)
214                 sb.append("\n   label: ").append(labels.next());
215             sb.append("\n");
216           }
217         }
218     
219         return sb.toString();
220       }

222       /**
223        * @param cfg governing control flow graph
224        * @param wantEdgeLabels whether to compute edge labels for CDG edges
225        */
226       public ControlDependenceGraph(ControlFlowGraph<I, T> cfg, boolean wantEdgeLabels) {
227         if (cfg == null) {
228           throw new IllegalArgumentException("null cfg");
229         }
230         this.cfg = cfg;
231         this.edgeManager = constructGraphEdges(buildControlDependence(wantEdgeLabels));
232       }
233     
234       /**
235        * @param cfg governing control flow graph
236        */
237       public ControlDependenceGraph(ControlFlowGraph<I, T> cfg) {
238         this(cfg, false);
239       }
240     
241       public ControlFlowGraph getControlFlowGraph() {
242         return cfg;
243       }
244     
245       /**
246        * Return the set of edge labels for the control flow edges that cause the given edge in the CDG. Requires that the CDG be
247        * constructed with wantEdgeLabels being true.
248        */
249       public Set<Object> getEdgeLabels(Object from, Object to) {
250         return edgeLabels.get(Pair.make(from, to));
251       }
252     
253       @Override
254       public NumberedNodeManager<T> getNodeManager() {
255         return cfg;
256       }
257     
258       @Override
259       public NumberedEdgeManager<T> getEdgeManager() {
260         return edgeManager;
261       }
262     
263       public boolean controlEquivalent(T bb1, T bb2) {
264         if (getPredNodeCount(bb1) != getPredNodeCount(bb2)) {
265           return false;
266         }
267     
268         for (Iterator<? extends T> pbs1 = getPredNodes(bb1); pbs1.hasNext();) {
269           if (!hasEdge(pbs1.next(), bb2)) {
270             return false;
271           }
272         }
273     
274         return true;
275       }
