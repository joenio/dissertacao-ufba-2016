11     package com.ibm.wala.util.intset;
12     
13     
14     /**
15      * an int vector implementation designed for low occupancy. Note that get() from
16      * this vector is a binary search.
17      * 
18      * This should only be used for small sets ... insertion and deletion are linear
19      * in size of set.
20      * 
21      * In this optimizations, the indices are longs.
22      */
23     public class SparseLongIntVector {
24     
25       private final static int INITIAL_SIZE = 5;
26     
27       private final double EXPANSION = 1.5;
28     
29       /**
30        * if indices[i] = x, then data[i] == get(x)
31        */
32       final private MutableSparseLongSet indices = new MutableSparseLongSet();
33     
34       private int[] data = new int[INITIAL_SIZE];
35     
36       private final int defaultValue;
37     
38       public SparseLongIntVector(int defaultValue) {
39         this.defaultValue = defaultValue;
40       }
41     
42       /*
43        * @see com.ibm.wala.util.intset.IntVector#get(int)
44        */
45       public int get(long x) {
46         int index = indices.getIndex(x);
47         if (index == -1) {
48           return defaultValue;
49         } else {
50           return data[index];
51         }
52       }
53     
54       /*
55        * TODO: this can be optimized
56        * 
57        * @see com.ibm.wala.util.intset.IntVector#set(int, int)
58        */
59       public void set(long x, int value) {
60         int index = indices.getIndex(x);
61         if (index == -1) {
62           indices.add(x);
63           index = indices.getIndex(x);
64           ensureCapacity(indices.size() + 1);
65           if (index < (data.length - 1)) {
66             System.arraycopy(data, index, data, index + 1, indices.size() - index);
67           }
68         }
69         data[index] = value;
70       }
71     
72       private void ensureCapacity(int capacity) {
73         if (data.length < capacity + 1) {
74           int[] old = data;
75           data = new int[1 + (int) (capacity * EXPANSION)];
76           System.arraycopy(old, 0, data, 0, old.length);
77         }
78       }
79     
80       /*
81        * @see com.ibm.wala.util.debug.VerboseAction#performVerboseAction()
82        */
83       public void performVerboseAction() {
84         System.err.println((getClass() + " stats: "));
85         System.err.println(("data.length " + data.length));
86         System.err.println(("indices.size() " + indices.size()));
87     
88       }
89     }
