11     package com.ibm.wala.util.graph.traverse;
12     
13     import java.util.Iterator;
14     import java.util.List;
15     
16     import com.ibm.wala.util.Predicate;
17     import com.ibm.wala.util.collections.FilterIterator;
18     import com.ibm.wala.util.collections.Pair;
19     import com.ibm.wala.util.graph.Graph;
20     
21     /**
22      * Extends {@link DFSPathFinder} to discover all paths from a set of root nodes
23      * to nodes passing some {@link Predicate}.
24      * 
25      * Note that this code performs work that is potentially exponential in the size
26      * of the underlying graph, using exponential space. It most likely won't work
27      * even for graphs of moderate size.
28      */
29     public class DFSAllPathsFinder<T> extends DFSPathFinder<T> {
30     
31       public DFSAllPathsFinder(Graph<T> G, Iterator<T> nodes, Predicate<T> f) {
32         super(G, nodes, f);
33       }
34     
35       public DFSAllPathsFinder(Graph<T> G, T N, Predicate<T> f) throws IllegalArgumentException {
36         super(G, N, f);
37       }
38     
39       @Override
40       protected Iterator<? extends T> getConnected(T n) {
41         final List<T> cp = currentPath();
42         return new FilterIterator<T>(G.getSuccNodes(n), new Predicate<T>() {
43           @Override public boolean test(T o) {
44             return ! cp.contains(o);
45           }
46         });
47       }
48     
49       @Override
50       protected Iterator<? extends T> getPendingChildren(T n) {
51         Pair<List<T>,T> key = Pair.make(currentPath(), n);
52         return pendingChildren.get(key);
53       }
54     
55       @Override
56       protected void setPendingChildren(T v, Iterator<? extends T> iterator) {
57         Pair<List<T>,T> key = Pair.make(currentPath(), v);
58         pendingChildren.put(key, iterator);
59       }
60     
61       
62     }
