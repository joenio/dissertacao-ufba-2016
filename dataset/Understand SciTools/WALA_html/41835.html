11     package com.ibm.wala.escape;
12     
13     import com.ibm.wala.ipa.callgraph.CGNode;
14     import com.ibm.wala.ipa.callgraph.propagation.InstanceKey;
15     import com.ibm.wala.util.WalaException;
16     import com.ibm.wala.util.intset.IntSet;
17     
18     /**
19      * Basic interface for liveness analysis of heap-allocated objects.
20      */
21     public interface ILiveObjectAnalysis {
22     
23       /**
24        * @param allocMethod a method which holds an allocation site
25        * @param allocPC bytecode index of allocation site
26        * @param m method in question
27        * @param instructionIndex index of an instruction in SSA IR. in m. if -1, it is interpreted as a wildcard meaning "any statement"
28        * @throws WalaException
29        * @returns true if an object allocated at the allocation site &lt;allocMethod,allocPC> may be live immediately after the
30        *          statement <m,instructionIndex>
31        */
32       public boolean mayBeLive(CGNode allocMethod, int allocPC, CGNode m, int instructionIndex) throws WalaException;
33     
34       /**
35        * @param ik an instance key
36        * @param m method in question
37        * @param instructionIndex index of an instruction in SSA IR. in m. if -1, it is interpreted as a wildcard meaning "any statement"
38        * @throws WalaException
39        * @returns true if an object allocated at the allocation site &lt;allocMethod,allocPC> may be live immediately after the
40        *          statement <m,instructionIndex>
41        */
42       public boolean mayBeLive(InstanceKey ik, CGNode m, int instructionIndex) throws WalaException;
43     
44       /**
45        * @param ik an instance key
46        * @param m method in question
47        * @param instructionIndices indices of instructions in SSA IR.
48        * @returns true if an object allocated at the allocation site &lt;allocMethod,allocPC> may be live immediately after the
49        *          statement <m,instructionIndex> for any instructionIndex in the set
50        */
51       public boolean mayBeLive(InstanceKey ik, CGNode m, IntSet instructionIndices);
52     
53     }
