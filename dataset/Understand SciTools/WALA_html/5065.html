26     /**
27      * This class provides Instance Key call backs where each instance is in the same equivalence class as all other instances of the
28      * same concrete type.
29      */
30     public class ClassBasedInstanceKeys implements InstanceKeyFactory {
31     
32       private final static boolean DEBUG = false;
33     
34       private final AnalysisOptions options;
35     
36       private final IClassHierarchy cha;
37     
38       public ClassBasedInstanceKeys(AnalysisOptions options, IClassHierarchy cha) {
39         if (cha == null) {
40           throw new IllegalArgumentException("null cha");
41         }
42         this.cha = cha;
43         this.options = options;
44       }
45     
46       @Override
47       public InstanceKey getInstanceKeyForAllocation(CGNode node, NewSiteReference allocation) {
48         if (allocation == null) {
49           throw new IllegalArgumentException("allocation is null");
50         }
51     
52         if (options.getClassTargetSelector() == null) {
53           throw new IllegalStateException("options did not specify class target selector");
54         }
55         IClass type = options.getClassTargetSelector().getAllocatedTarget(node, allocation);
56         if (type == null) {
57           return null;
58         }
59     
60         ConcreteTypeKey key = new ConcreteTypeKey(type);
61     
62         return key;
63       }
64     
65       /**
66        * @see com.ibm.wala.ipa.callgraph.propagation.InstanceKeyFactory#getInstanceKeyForMultiNewArray(com.ibm.wala.ipa.callgraph.CGNode,
67        *      com.ibm.wala.classLoader.NewSiteReference, int) dim == 0 represents the first dimension, e.g., the [Object; instances in
68        *      [[Object; e.g., the [[Object; instances in [[[Object; dim == 1 represents the second dimension, e.g., the [Object
69        *      instances in [[[Object;
70        */
71       @Override
72       public InstanceKey getInstanceKeyForMultiNewArray(CGNode node, NewSiteReference allocation, int dim) {
73         if (DEBUG) {
74           System.err.println(("getInstanceKeyForMultiNewArray " + allocation + " " + dim));
75         }
76         ArrayClass type = (ArrayClass) options.getClassTargetSelector().getAllocatedTarget(node, allocation);
77         assert (type != null);
78         if (DEBUG) {
79           System.err.println(("type: " + type));
80         }
81         if (type == null) {
82           assert type != null : "null type for " + allocation;
83         }
84         int i = 0;
85         while (i <= dim) {
86           i++;
87           if (type == null) {
88             Assertions.UNREACHABLE();
89           }
90           type = (ArrayClass) type.getElementClass();
91           if (DEBUG) {
92             System.err.println(("intermediate: " + i + " " + type));
93           }
94         }
95         if (DEBUG) {
96           System.err.println(("final type: " + type));
97         }
98         if (type == null) {
99           return null;
100         }
101         ConcreteTypeKey key = new ConcreteTypeKey(type);
102     
103         return key;
104       }
105     
106       @Override
107       public <T> InstanceKey getInstanceKeyForConstant(TypeReference type, T S) {
108         if (type == null || cha.lookupClass(type) == null) {
109           return null;
110         } else {
111           if (options.getUseConstantSpecificKeys()) {
112             return new ConstantKey<T>(S, cha.lookupClass(type));
113           } else {
114             return new ConcreteTypeKey(cha.lookupClass(type));
115           }
116         }
117       }
118     
119       /**
120        * @return a set of ConcreteTypeKeys that represent the exceptions the PEI may throw.
121        */
122       @Override
123       public InstanceKey getInstanceKeyForPEI(CGNode node, ProgramCounter peiLoc, TypeReference type) {
124         IClass klass = cha.lookupClass(type);
125         if (klass == null) {
126           return null;
127         }
128         return new ConcreteTypeKey(cha.lookupClass(type));
129       }
130     
131       @Override
132       public InstanceKey getInstanceKeyForMetadataObject(Object obj, TypeReference objType) {
133         IClass cls = cha.lookupClass(objType);
134         assert cls != null : objType;
135         if (obj instanceof TypeReference) {
136           IClass klass = cha.lookupClass((TypeReference)obj);
137           if (klass == null) {
138             return new ConcreteTypeKey(cls);
139           } else {
140             // return the IClass itself, wrapped as a constant!
141             return new ConstantKey<IClass>(klass, cls);
142           }
143         } else if (obj instanceof MethodReference) {
144           IMethod m = cha.resolveMethod((MethodReference)obj);
145           if (m == null) {
146             return new ConcreteTypeKey(cls);
147           } else {
148             return new ConstantKey<IMethod>(m, cls);
149           }
150         } else if (obj instanceof Descriptor) {
151           return new ConstantKey<Descriptor>((Descriptor)obj, cls);
152         } else {
153           // other cases
154           throw new Error();
155         }
156       }
157     
158       /**
159        * @return Returns the class hierarchy.
160        */
161       public IClassHierarchy getClassHierarchy() {
162         return cha;
163       }
164     
165     }
