11     package com.ibm.wala.cast.tree.pattern;
12     
13     import com.ibm.wala.cast.tree.CAstNode;
14     
15     /**
16      * Pattern to match one of two alternatives.
17      * 
18      * @author mschaefer
19      *
20      */
21     public class Alt implements NodePattern {
22       private final NodePattern left, right;
23       
24       public Alt(NodePattern left, NodePattern right) {
25         this.left = left;
26         this.right = right;
27       }
28     
29       @Override
30       public boolean matches(CAstNode node) {
31         return left.matches(node) || right.matches(node);
32       }
33     
34     }
