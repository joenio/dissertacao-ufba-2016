86     /**
87      * A graph representation of statements flowing pointer values, but <em>not</em> primitive values. Nodes are variables, and edges
88      * are <em>against</em> value flow; assignment x = y yields edge from x to y with label {@link AssignLabel#noFilter()}
89      */
90     public class DemandPointerFlowGraph extends AbstractDemandFlowGraph implements IFlowGraph {
91     
92       public DemandPointerFlowGraph(CallGraph cg, HeapModel heapModel, MemoryAccessMap mam, IClassHierarchy cha) {
93         super(cg, heapModel, mam, cha);
94       }

96       /**
97        * add nodes for parameters and return values
98        */
99       @Override
100       protected void addNodesForParameters(CGNode node, IR ir) {
101         for (Iterator<Integer> iter = new PointerParamValueNumIterator(node); iter.hasNext();) {
102           int parameter = iter.next();
103           PointerKey paramPk = heapModel.getPointerKeyForLocal(node, parameter);
104           addNode(paramPk);
105           params.put(paramPk, node);
106         }
107         PointerKey returnKey = heapModel.getPointerKeyForReturnValue(node);
108         addNode(returnKey);
109         returns.put(returnKey, node);
110         PointerKey exceptionReturnKey = heapModel.getPointerKeyForExceptionalReturnValue(node);
111         addNode(exceptionReturnKey);
112         returns.put(exceptionReturnKey, node);
113       }

115       @Override
116       protected FlowStatementVisitor makeVisitor(CGNode node) {
117         return new StatementVisitor(heapModel, this, cha, cg, node);
118       }

176          * @see com.ibm.domo.ssa.SSAInstruction.Visitor#visitArrayLoad(com.ibm.domo.ssa.SSAArrayLoadInstruction)
177          */
178         @Override
179         public void visitArrayLoad(SSAArrayLoadInstruction instruction) {
180           // skip arrays of primitive type
181           if (instruction.typeIsPrimitive()) {
182             return;
183           }
184           PointerKey result = heapModel.getPointerKeyForLocal(node, instruction.getDef());
185           PointerKey arrayRef = heapModel.getPointerKeyForLocal(node, instruction.getArrayRef());
186           // TODO optimizations for purely local stuff
187           g.addNode(result);
188           g.addNode(arrayRef);
189           g.addEdge(result, arrayRef, GetFieldLabel.make(ArrayContents.v()));
190         }

193          * @see com.ibm.domo.ssa.SSAInstruction.Visitor#visitArrayStore(com.ibm.domo.ssa.SSAArrayStoreInstruction)
194          */
195         @Override
196         public void visitArrayStore(SSAArrayStoreInstruction instruction) {
197           // Assertions.UNREACHABLE();
198           // skip arrays of primitive type
199           if (instruction.typeIsPrimitive()) {
200             return;
201           }
202           // make node for used value
203           PointerKey value = heapModel.getPointerKeyForLocal(node, instruction.getValue());
204           PointerKey arrayRef = heapModel.getPointerKeyForLocal(node, instruction.getArrayRef());
205           // TODO purely local optimizations
206           g.addNode(value);
207           g.addNode(arrayRef);
208           g.addEdge(arrayRef, value, PutFieldLabel.make(ArrayContents.v()));
209         }

211         /*
212          * @see com.ibm.domo.ssa.SSAInstruction.Visitor#visitCheckCast(com.ibm.domo.ssa.SSACheckCastInstruction)
213          */
214         @Override
215         public void visitCheckCast(SSACheckCastInstruction instruction) {
216           Set<IClass> types = HashSetFactory.make();
217           
218           for(TypeReference t : instruction.getDeclaredResultTypes()) {
219             IClass cls = cha.lookupClass(t);
220             if (cls == null) {
221               return;
222             } else {
223               types.add(cls);
224             }
225           }
226           
227           FilteredPointerKey.MultipleClassesFilter filter = new FilteredPointerKey.MultipleClassesFilter(types.toArray(new IClass[ types.size() ]));
228           PointerKey result = heapModel.getPointerKeyForLocal(node, instruction.getResult());
229           PointerKey value = heapModel.getPointerKeyForLocal(node, instruction.getVal());
230           g.addNode(result);
231           g.addNode(value);
232           g.addEdge(result, value, AssignLabel.make(filter));
233         }

235         /*
236          * @see com.ibm.domo.ssa.SSAInstruction.Visitor#visitReturn(com.ibm.domo.ssa.SSAReturnInstruction)
237          */
238         @Override
239         public void visitReturn(SSAReturnInstruction instruction) {
240           // skip returns of primitive type
241           if (instruction.returnsPrimitiveType() || instruction.returnsVoid()) {
242             return;
243           } else {
244             // just make a node for the def'd value
245             PointerKey def = heapModel.getPointerKeyForLocal(node, instruction.getResult());
246             g.addNode(def);
247             PointerKey returnValue = heapModel.getPointerKeyForReturnValue(node);
248             g.addNode(returnValue);
249             g.addEdge(returnValue, def, AssignLabel.noFilter());
250           }
251         }

261         protected void visitGetInternal(int lval, int ref, boolean isStatic, FieldReference field) {
262     
263           // skip getfields of primitive type (optimisation)
264           if (field.getFieldType().isPrimitiveType()) {
265             return;
266           }
267           IField f = cg.getClassHierarchy().resolveField(field);
268           if (f == null) {
269             return;
270           }
271           PointerKey def = heapModel.getPointerKeyForLocal(node, lval);
272           assert def != null;
273     
274           if (isStatic) {
275             PointerKey fKey = heapModel.getPointerKeyForStaticField(f);
276             g.addNode(def);
277             g.addNode(fKey);
278             g.addEdge(def, fKey, AssignGlobalLabel.v());
279           } else {
280             PointerKey refKey = heapModel.getPointerKeyForLocal(node, ref);
281             g.addNode(def);
282             g.addNode(refKey);
283             // TODO purely local optimizations
284             g.addEdge(def, refKey, GetFieldLabel.make(f));
285           }
286         }

350         /*
351          * @see com.ibm.domo.ssa.Instruction.Visitor#visitNew(com.ibm.domo.ssa.NewInstruction)
352          */
353         @Override
354         public void visitNew(SSANewInstruction instruction) {
355     
356           InstanceKey iKey = heapModel.getInstanceKeyForAllocation(node, instruction.getNewSite());
357           if (iKey == null) {
358             // something went wrong. I hope someone raised a warning.
359             return;
360           }
361           PointerKey def = heapModel.getPointerKeyForLocal(node, instruction.getDef());
362           g.addNode(iKey);
363           g.addNode(def);
364           g.addEdge(def, iKey, NewLabel.v());
365     
366           NewMultiDimInfo multiDimInfo = getInfoForNewMultiDim(instruction, heapModel, node);
367           if (multiDimInfo != null) {
368             for (Pair<PointerKey, InstanceKey> newInstr : multiDimInfo.newInstrs) {
369               g.addNode(newInstr.fst);
370               g.addNode(newInstr.snd);
371               g.addEdge(newInstr.fst, newInstr.snd, NewLabel.v());
372             }
373             for (Pair<PointerKey, PointerKey> arrStoreInstr : multiDimInfo.arrStoreInstrs) {
374               g.addNode(arrStoreInstr.fst);
375               g.addNode(arrStoreInstr.snd);
376               g.addEdge(arrStoreInstr.fst, arrStoreInstr.snd, PutFieldLabel.make(ArrayContents.v()));
377             }
378           }
379         }

391          * @see com.ibm.domo.ssa.Instruction.Visitor#visitGetCaughtException(com.ibm.domo.ssa.GetCaughtExceptionInstruction)
392          */
393         @Override
394         public void visitGetCaughtException(SSAGetCaughtExceptionInstruction instruction) {
395           List<ProgramCounter> peis = SSAPropagationCallGraphBuilder.getIncomingPEIs(ir, getBasicBlock());
396           PointerKey def = heapModel.getPointerKeyForLocal(node, instruction.getDef());
397     
398           Set<IClass> types = SSAPropagationCallGraphBuilder.getCaughtExceptionTypes(instruction, ir);
399           addExceptionDefConstraints(ir, node, peis, def, types);
400         }

402         /**
403          * Generate constraints which assign exception values into an exception pointer
404          * 
405          * @param node governing node
406          * @param peis list of PEI instructions
407          * @param exceptionVar PointerKey representing a pointer to an exception value
408          * @param catchClasses the types "caught" by the exceptionVar
409          */
410         protected void addExceptionDefConstraints(IR ir, CGNode node, List<ProgramCounter> peis, PointerKey exceptionVar,
411             Set<IClass> catchClasses) {
412           for (Iterator<ProgramCounter> it = peis.iterator(); it.hasNext();) {
413             ProgramCounter peiLoc = it.next();
414             SSAInstruction pei = ir.getPEI(peiLoc);
415     
416             if (pei instanceof SSAAbstractInvokeInstruction) {
417               SSAAbstractInvokeInstruction s = (SSAAbstractInvokeInstruction) pei;
418               PointerKey e = heapModel.getPointerKeyForLocal(node, s.getException());
419               g.addNode(exceptionVar);
420               g.addNode(e);
421               g.addEdge(exceptionVar, e, AssignLabel.noFilter());
422     
423             } else if (pei instanceof SSAAbstractThrowInstruction) {
424               SSAAbstractThrowInstruction s = (SSAAbstractThrowInstruction) pei;
425               PointerKey e = heapModel.getPointerKeyForLocal(node, s.getException());
426               g.addNode(exceptionVar);
427               g.addNode(e);
428               g.addEdge(exceptionVar, e, AssignLabel.noFilter());
429             }
430     
431             // Account for those exceptions for which we do not actually have a
432             // points-to set for
433             // the pei, but just instance keys
434             Collection<TypeReference> types = pei.getExceptionTypes();
435             if (types != null) {
436               for (Iterator<TypeReference> it2 = types.iterator(); it2.hasNext();) {
437                 TypeReference type = it2.next();
438                 if (type != null) {
439                   InstanceKey ik = heapModel.getInstanceKeyForPEI(node, peiLoc, type);
440                   if (ik == null) {
441                     // probably due to exclusions
442                     continue;
443                   }
444                   assert ik instanceof ConcreteTypeKey : "uh oh: need to implement getCaughtException constraints for instance " + ik;
445                   ConcreteTypeKey ck = (ConcreteTypeKey) ik;
446                   IClass klass = ck.getType();
447                   if (PropagationCallGraphBuilder.catches(catchClasses, klass, cha)) {
448                     g.addNode(exceptionVar);
449                     g.addNode(ik);
450                     g.addEdge(exceptionVar, ik, NewLabel.v());
451                   }
452                 }
453               }
454             }
455           }
456         }

512       /**
513        * collect information about the new instructions and putfield instructions used to model an allocation of a multi-dimensional
514        * array. excludes the new instruction itself (i.e., the allocation of the top-level multi-dim array).
515        */
516       public static NewMultiDimInfo getInfoForNewMultiDim(SSANewInstruction instruction, HeapModel heapModel, CGNode node) {
517         if (heapModel == null) {
518           throw new IllegalArgumentException("null heapModel");
519         }
520         Collection<Pair<PointerKey, InstanceKey>> newInstrs = HashSetFactory.make();
521         Collection<Pair<PointerKey, PointerKey>> arrStoreInstrs = HashSetFactory.make();
522         InstanceKey iKey = heapModel.getInstanceKeyForAllocation(node, instruction.getNewSite());
523         if (iKey == null) {
524           // something went wrong. I hope someone raised a warning.
525           return null;
526         }
527         IClass klass = iKey.getConcreteType();
528         // if not a multi-dim array allocation, return null
529         if (!klass.isArrayClass() || ((ArrayClass) klass).getElementClass() == null
530             || !((ArrayClass) klass).getElementClass().isArrayClass()) {
531           return null;
532         }
533         PointerKey def = heapModel.getPointerKeyForLocal(node, instruction.getDef());
534     
535         int dim = 0;
536         InstanceKey lastInstance = iKey;
537         PointerKey lastVar = def;
538         while (klass != null && klass.isArrayClass()) {
539           klass = ((ArrayClass) klass).getElementClass();
540           // klass == null means it's a primitive
541           if (klass != null && klass.isArrayClass()) {
542             InstanceKey ik = heapModel.getInstanceKeyForMultiNewArray(node, instruction.getNewSite(), dim);
543             PointerKey pk = heapModel.getPointerKeyForArrayContents(lastInstance);
544             // if (DEBUG_MULTINEWARRAY) {
545             // Trace.println("multinewarray constraint: ");
546             // Trace.println(" pk: " + pk);
547             // Trace.println(" ik: " + system.findOrCreateIndexForInstanceKey(ik)
548             // + " concrete type " + ik.getConcreteType()
549             // + " is " + ik);
550             // Trace.println(" klass:" + klass);
551             // }
552             // g.addEdge(pk, ik, NewLabel.v());
553             newInstrs.add(Pair.make(pk, ik));
554             arrStoreInstrs.add(Pair.make(lastVar, pk));
555             lastInstance = ik;
556             lastVar = pk;
557             dim++;
558           }
559         }
560     
561         return new NewMultiDimInfo(newInstrs, arrStoreInstrs);
562       }
