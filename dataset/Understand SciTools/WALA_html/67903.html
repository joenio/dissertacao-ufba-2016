11     package com.ibm.wala.viz.viewer;
12     
13     import java.util.Collection;
14     
15     import javax.swing.DefaultListModel;
16     import javax.swing.JList;
17     import javax.swing.JScrollPane;
18     import javax.swing.JSplitPane;
19     import javax.swing.JTree;
20     import javax.swing.event.TreeExpansionEvent;
21     import javax.swing.event.TreeExpansionListener;
22     import javax.swing.event.TreeSelectionEvent;
23     import javax.swing.event.TreeSelectionListener;
24     import javax.swing.tree.DefaultMutableTreeNode;
25     import javax.swing.tree.TreePath;
26     
27     import com.ibm.wala.classLoader.IClass;
28     import com.ibm.wala.classLoader.IMethod;
29     import com.ibm.wala.ipa.cha.IClassHierarchy;
30     
31     public class ChaPanel extends JSplitPane {
32     
33       private final IClassHierarchy cha;
34     
35       public ChaPanel(IClassHierarchy cha) {
36         this.cha = cha;
37       
38         this.setDividerLocation(250);
39         JTree tree = buildTree();
40         this.setLeftComponent(new JScrollPane(tree));
41         
42         final DefaultListModel methodListModel = new DefaultListModel();
43         JList methodList = new JList(methodListModel);
44         this.setRightComponent(methodList);
45         
46         tree.addTreeSelectionListener(new TreeSelectionListener(){
47           @Override
48           public void valueChanged(TreeSelectionEvent e) {
49             TreePath newLeadSelectionPath = e.getNewLeadSelectionPath();
50             if (null == newLeadSelectionPath){
51               return;
52             }
53             DefaultMutableTreeNode treeNode = (DefaultMutableTreeNode) newLeadSelectionPath.getLastPathComponent();
54             IClass klass = (IClass) treeNode.getUserObject();
55             methodListModel.clear();
56             for (IMethod m : klass.getDeclaredMethods()){
57               methodListModel.addElement(m.toString());
58             }
59           }
60         });
61         
62       }
63     
64       private JTree buildTree() {
65         IClass rootClass = cha.getRootClass();
66         DefaultMutableTreeNode root = new DefaultMutableTreeNode(rootClass);
67     
68         expandNode(root);
69         JTree tree = new JTree(root);
70     
71         tree.addTreeExpansionListener(new TreeExpansionListener() {
72     
73           @Override
74           public void treeExpanded(TreeExpansionEvent event) {
75             TreePath path = event.getPath();
76             if (path == null) {
77               return;
78             }
79             DefaultMutableTreeNode lastNode = (DefaultMutableTreeNode) path.getLastPathComponent();
80             expandNode(lastNode);
81           }
82     
83           @Override
84           public void treeCollapsed(TreeExpansionEvent event) {
85     
86           }
87         });
88     
89         return tree;
90       }
91     
92       private void expandNode(DefaultMutableTreeNode treeNode) {
93         expandNode(treeNode, 3);
94       }
95     
96       private void expandNode(DefaultMutableTreeNode treeNode, int rec) {
97         if (rec == 0) {
98           return;
99         }
100     
101         if (treeNode.getChildCount() == 0) {
102           IClass klass = (IClass) treeNode.getUserObject();
103           Collection<IClass> immediateSubclasses = cha.getImmediateSubclasses(klass);
104           for (IClass immediateSubclass : immediateSubclasses){
105             treeNode.add(new DefaultMutableTreeNode(immediateSubclass));
106           }
107         }
108     
109         for (int i = 0; i < treeNode.getChildCount(); i++) {
110           DefaultMutableTreeNode child = (DefaultMutableTreeNode) treeNode.getChildAt(i);
111           expandNode(child, rec - 1);
112         }
113       }
114     
115       
116       
117     }
