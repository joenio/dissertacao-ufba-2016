11     package com.ibm.wala.util.graph.dominators;
12     
13     import java.util.Iterator;
14     import java.util.Map;
15     import java.util.Set;
16     
17     import com.ibm.wala.util.collections.HashMapFactory;
18     import com.ibm.wala.util.collections.HashSetFactory;
19     import com.ibm.wala.util.collections.NonNullSingletonIterator;
20     import com.ibm.wala.util.graph.Graph;
21     import com.ibm.wala.util.graph.traverse.DFS;
22     
23     /**
24      * An object that computes the dominance frontiers of a graph
25      */
26     public class DominanceFrontiers<T> {
27     
28       final private Map<T, Set<T>> DF = HashMapFactory.make();
29     
30       final private Dominators<T> dom;
31     
32       final private Graph<T> G;
33     
34       final private T root;
35     
36       /**
37        * @param G
38        *          The graph
39        * @param root
40        *          The root from which to compute dominators
41        */
42       public DominanceFrontiers(Graph<T> G, T root) {
43         this.root = root;
44         this.G = G;
45         this.dom = Dominators.make(G, root);
46         analyze();
47       }
48     
49       public Iterator<T> getDominanceFrontier(T n) {
50         Set<T> frontier = DF.get(n);
51         if (frontier == null) {
52           throw new IllegalArgumentException("no dominance frontier for node " + n);
53         }
54         return frontier.iterator();
55       }
56     
57       public boolean isDominatedBy(T node, T master) {
58         return dom.isDominatedBy(node, master);
59       }
60     
61       public Iterator<T> dominators(T node) {
62         return dom.dominators(node);
63       }
64     
65       public Graph<T> dominatorTree() {
66         return dom.dominatorTree();
67       }
68     
69       private void analyze() {
70         Graph<T> DT = dom.dominatorTree();
71     
72         Iterator<T> XS = DFS.iterateFinishTime(DT, new NonNullSingletonIterator<T>(root));
73         while (XS.hasNext()) {
74           T X = XS.next();
75           Set<T> DF_X = HashSetFactory.make();
76           DF.put(X, DF_X);
77     
78           // DF_local
79           for (Iterator<? extends T> YS = G.getSuccNodes(X); YS.hasNext();) {
80             T Y = YS.next();
81             if (dom.getIdom(Y) != X) {
82               DF_X.add(Y);
83             }
84           }
85     
86           // DF_up
87           for (Iterator<? extends T> ZS = DT.getSuccNodes(X); ZS.hasNext();) {
88             T Z = ZS.next();
89             for (Iterator<T> YS2 = getDominanceFrontier(Z); YS2.hasNext();) {
90               T Y2 = YS2.next();
91               if (dom.getIdom(Y2) != X)
92                 DF_X.add(Y2);
93             }
94           }
95         }
96       }
97     }
