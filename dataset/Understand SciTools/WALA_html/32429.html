11     package com.ibm.wala.shrikeCT;
12     
13     /**
14      * This class reads LineNumberTable attributes.
15      * 
16      * Instead of constructing a LineNumberTableReader directly, consider just calling LineNumberTableReader.makeBytecodeToSourceMap for
17      * convenient access to aggregate line number data from all the LineNumberTable attributes for a given Code.
18      */
19     public final class LineNumberTableReader extends AttributeReader {
20       /**
21        * Build a reader for a LineNumberTable attribute.
22        */
23       public LineNumberTableReader(ClassReader.AttrIterator iter) throws InvalidClassFileException {
24         super(iter, "LineNumberTable");
25     
26         int offset = attr + 6;
27         checkSize(offset, 2);
28         int count = cr.getUShort(offset);
29         offset += 2;
30         checkSize(offset, count * 4);
31       }
32     
33       /**
34        * @return the raw line number table data, a flattened sequence of (startPC, lineNumber) pairs
35        */
36       public int[] getRawTable() {
37         int count = cr.getUShort(attr + 6);
38         int[] r = new int[count * 2];
39         int offset = attr + 8;
40         for (int i = 0; i < r.length; i++) {
41           r[i] = cr.getUShort(offset);
42           offset += 2;
43         }
44         return r;
45       }
46     
47       /**
48        * Construct a "bytecode to source" map for the given code. This method aggregates all the LineNumberTable attributes for the code
49        * into one handy data structure.
50        * 
51        * @return an array mapping each byte of the bytecode bytes to the line number that that byte belongs to, or null if there is no
52        *         line number data in the Code
53        */
54       public static int[] makeBytecodeToSourceMap(CodeReader code) throws InvalidClassFileException, IllegalArgumentException {
55     
56         if (code == null) {
57           throw new IllegalArgumentException();
58         }
59         int[] r = null;
60         ClassReader cr = code.getClassReader();
61     
62         ClassReader.AttrIterator iter = new ClassReader.AttrIterator();
63         code.initAttributeIterator(iter);
64         for (; iter.isValid(); iter.advance()) {
65           if (iter.getName().equals("LineNumberTable")) {
66             if (r == null) {
67               r = new int[code.getBytecodeLength()];
68             }
69     
70             // check length
71             new LineNumberTableReader(iter);
72             int attr = iter.getRawOffset();
73             int count = cr.getUShort(attr + 6);
74             int offset = attr + 8;
75             for (int j = 0; j < count; j++) {
76               int startPC = cr.getUShort(offset);
77               int lineNum = cr.getUShort(offset + 2);
78               offset += 4;
79     
80               if (startPC < 0 || startPC >= r.length) {
81                 throw new InvalidClassFileException(offset, "Invalid bytecode offset " + startPC + " in LineNumberTable");
82               }
83               r[startPC] = lineNum;
84             }
85           }
86         }
87     
88         if (r != null) {
89           // fill in gaps in the old line number map
90           int last = 0;
91           for (int i = 0; i < r.length; i++) {
92             int cur = r[i];
93             if (cur == 0) {
94               r[i] = last;
95             } else {
96               last = cur;
97             }
98           }
99         }
100     
101         return r;
102       }
103     }
