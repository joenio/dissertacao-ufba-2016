93       @Ignore("support for this combination of context sensitivity and on-the-fly call graph refinement is not yet implemented")
94       @Test
95       public void testOnTheFlyCS() throws ClassHierarchyException, IllegalArgumentException, CancelException, IOException {
96         String mainClass = TestInfo.TEST_ONTHEFLY_CS;
97         final IDemandPointerAnalysis dmp = makeDemandPointerAnalysis(mainClass);
98         CGNode testMethod = AbstractPtrTest.findInstanceMethod(
99             dmp.getBaseCallGraph(),
100             dmp.getClassHierarchy().lookupClass(
101                 TypeReference.findOrCreate(ClassLoaderReference.Application, "Ldemandpa/TestOnTheFlyCS$C2")),
102             Atom.findOrCreateUnicodeAtom("doSomething"), Descriptor.findOrCreateUTF8("(Ljava/lang/Object;)V"));
103         PointerKey keyToQuery = AbstractPtrTest.getParam(testMethod, "testThisVar", dmp.getHeapModel());
104         Collection<InstanceKey> pointsTo = dmp.getPointsTo(keyToQuery);
105         if (debug) {
106           System.err.println("points-to for " + mainClass + ": " + pointsTo);
107         }
108         Assert.assertEquals(1, pointsTo.size());
109       }

177       @Override
178       protected DemandRefinementPointsTo makeDemandPointerAnalysis(String mainClass) throws ClassHierarchyException,
179           IllegalArgumentException, CancelException, IOException {
180         DemandRefinementPointsTo dmp = super.makeDemandPointerAnalysis(mainClass);
181         dmp.setRefinementPolicyFactory(new SinglePassRefinementPolicy.Factory(new AlwaysRefineFieldsPolicy(),
182             new AlwaysRefineCGPolicy()));
183         return dmp;
184       }
