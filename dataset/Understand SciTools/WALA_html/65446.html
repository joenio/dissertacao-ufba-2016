19     /**
20      * Utilities for iterating over graphs in topological order.
21      */
22     public class Topological {
23     
24       /**
25        * Build an Iterator over all the nodes in the graph, in an order such that SCCs are visited in topological order.
26        * 
27        * @throws IllegalArgumentException if graph == null
28        */
29       public static <T> Iterator<T> makeTopologicalIter(Graph<T> graph) throws IllegalArgumentException {
30         // the following code ensures a topological order over SCCs.
31         // note that the first two lines of the following give a topological
32         // order for dags, but that can get screwed up by cycles. so
33         // instead, we use Tarjan's SCC algorithm, which happens to
34         // visit nodes in an order consistent with a top. order over SCCs.
35     
36         if (graph == null) {
37           throw new IllegalArgumentException("graph == null");
38         }
39         // finish time is post-order
40         // note that if you pay attention only to the first representative
41         // of each SCC discovered, we have a top. order of these SCC
42         // representatives
43         Iterator<T> finishTime = DFS.iterateFinishTime(graph);
44         // reverse postorder is usual topological sort.
45         Iterator<T> rev = ReverseIterator.reverse(finishTime);
46         // the following statement helps out the GC; note that finishTime holds
47         // on to a large array
48         finishTime = null;
49         Graph<T> G_T = GraphInverter.invert(graph);
50         Iterator<T> order = DFS.iterateFinishTime(G_T, rev);
51         return order;
52       }
