11     package com.ibm.wala.cast.ir.ssa;
12     
13     import com.ibm.wala.classLoader.CallSiteReference;
14     import com.ibm.wala.ssa.SSAAbstractInvokeInstruction;
15     
16     public abstract class MultiReturnValueInvokeInstruction
17         extends SSAAbstractInvokeInstruction
18     {
19       protected final int results[];
20     
21       protected MultiReturnValueInvokeInstruction(int iindex, int results[], int exception, CallSiteReference site) {
22         super(iindex, exception, site);
23         this.results = results;
24       }
25     
26       @Override
27       public int getNumberOfReturnValues() {
28         return results==null? 0: results.length;
29       }
30     
31       @Override
32       public int getReturnValue(int i) {
33         return results[i];
34       }
35     
36     }
