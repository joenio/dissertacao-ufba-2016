36       /**
37        * List all the files in a directory that match a regular expression
38        * 
39        * @param recurse
40        *          recurse to subdirectories?
41        * @throws IllegalArgumentException
42        *           if dir is null
43        */
44       public static Collection<File> listFiles(String dir, String regex, boolean recurse) {
45         if (dir == null) {
46           throw new IllegalArgumentException("dir is null");
47         }
48         File d = new File(dir);
49         Pattern p = null;
50         if (regex != null) {
51           p = Pattern.compile(regex);
52         }
53         return listFiles(d, recurse, p);
54       }
55     
56       private static Collection<File> listFiles(File directory, boolean recurse, Pattern p) {
57         File[] files = directory.listFiles();
58         if (files == null) {
59           return Collections.emptyList();
60         }
61         HashSet<File> result = HashSetFactory.make();
62         for (int i = 0; i < files.length; i++) {
63           if (p == null || p.matcher(files[i].getAbsolutePath()).matches()) {
64             result.add(files[i]);
65           }
66           if (recurse && files[i].isDirectory()) {
67             result.addAll(listFiles(files[i], recurse, p));
68           }
69         }
70         return result;
71       }

80       public static void copy(String srcFileName, String destFileName) throws IOException {
81         if (srcFileName == null) {
82           throw new IllegalArgumentException("srcFileName is null");
83         }
84         if (destFileName == null) {
85           throw new IllegalArgumentException("destFileName is null");
86         }
87         FileChannel src = null;
88         FileChannel dest = null;
89         try {
90           src = new FileInputStream(srcFileName).getChannel();
91           dest = new FileOutputStream(destFileName).getChannel();
92           long n = src.size();
93           MappedByteBuffer buf = src.map(FileChannel.MapMode.READ_ONLY, 0, n);
94           dest.write(buf);
95         } finally {
96           if (dest != null) {
97             try {
98               dest.close();
99             } catch (IOException e1) {
100             }
101           }
102           if (src != null) {
103             try {
104               src.close();
105             } catch (IOException e1) {
106             }
107           }
108         }
109       }

111       /**
112        * delete all files (recursively) in a directory. This is dangerous. Use with
113        * care.
114        * 
115        * @throws IOException
116        *           if there's a problem deleting some file
117        */
118       public static void deleteContents(String directory) throws IOException {
119         File f = new File(directory);
120         if (!f.exists()) {
121           return;
122         }
123         if (!f.isDirectory()) {
124           throw new IOException(directory + " is not a vaid directory");
125         }
126         for (String s : f.list()) {
127           deleteRecursively(new File(f, s));
128         }
129       }
130     
131       private static void deleteRecursively(File f) throws IOException{
132         if (f.isDirectory()) {
133           for (String s : f.list()) {
134             deleteRecursively(new File(f, s));
135           }
136         }
137     
138         boolean b = f.delete();
139         if (!b) {
140           throw new IOException("failed to delete " + f);
141         }
142       }

144       /**
145        * Create a {@link FileOutputStream} corresponding to a particular file name.
146        * Delete the existing file if one exists.
147        */
148       public static final FileOutputStream createFile(String fileName) throws IOException {
149         if (fileName == null) {
150           throw new IllegalArgumentException("null file");
151         }
152         File f = new File(fileName);
153         if (f.getParentFile() != null && !f.getParentFile().exists()) {
154           boolean result = f.getParentFile().mkdirs();
155           if (!result) {
156             throw new IOException("failed to create " + f.getParentFile());
157           }
158         }
159         if (f.exists()) {
160           f.delete();
161         }
162         boolean result = f.createNewFile();
163         if (!result) {
164           throw new IOException("failed to create " + f);
165         }
166         return new FileOutputStream(f);
167       }

169       /**
170        * read fully the contents of s and return a byte array holding the result
171        * 
172        * @throws IOException
173        */
174       public static byte[] readBytes(InputStream s) throws IOException {
175         if (s == null) {
176           throw new IllegalArgumentException("null s");
177         }
178         ByteArrayOutputStream out = new ByteArrayOutputStream();
179         byte[] b = new byte[1024];
180         int n = s.read(b);
181         while (n != -1) {
182           out.write(b, 0, n);
183           n = s.read(b);
184         }
185         byte[] bb = out.toByteArray();
186         out.close();
187         return bb;
188       }

203       public static void recurseFiles(VoidFunction<File> action, final Predicate<File> filter, File top) {
204       	if (top.isDirectory()) {
205       		for(File f : top.listFiles(new FileFilter() {
206       			@Override
207       			public boolean accept(File file) {
208       				return filter.test(file) || file.isDirectory();
209       			}	
210       		})) {
211       			recurseFiles(action, filter, f);
212       		}
213       	} else {
214       		action.apply(top);
215       	}
216       }
