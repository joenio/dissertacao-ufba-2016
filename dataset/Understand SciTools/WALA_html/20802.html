35     /**
36      * Check handling of class constants (test for part of 1.5 support)
37      */
38     public class ClassConstantTest extends WalaTestCase {
39     
40       @Test public void testClassConstants() throws ClassHierarchyException, IllegalArgumentException, CancelException, IOException {
41     
42         AnalysisScope scope = CallGraphTestUtil.makeJ2SEAnalysisScope(TestConstants.WALA_TESTDATA,
43             CallGraphTestUtil.REGRESSION_EXCLUSIONS);
44         ClassHierarchy cha = ClassHierarchy.make(scope);
45     
46         // make sure we have the test class
47         TypeReference mainClassRef = TypeReference.findOrCreate(ClassLoaderReference.Application, TestConstants.CLASSCONSTANT_MAIN);
48         Assert.assertTrue(cha.lookupClass(mainClassRef) != null);
49     
50         // make call graph
51         Iterable<Entrypoint> entrypoints = Util.makeMainEntrypoints(scope, cha, TestConstants.CLASSCONSTANT_MAIN);
52         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
53         CallGraph cg = CallGraphTestUtil.buildZeroCFA(options, new AnalysisCache(), cha, scope, false);
54         // System.out.println("\nCall graph:");
55         // Trace.println(cg);
56     
57         // make sure the main method is reached
58         MethodReference mainMethodRef = MethodReference.findOrCreate(mainClassRef, "main", "([Ljava/lang/String;)V");
59         Set<CGNode> mainMethodNodes = cg.getNodes(mainMethodRef);
60         Assert.assertFalse(mainMethodNodes.isEmpty());
61         CGNode mainMethodNode = mainMethodNodes.iterator().next();
62         // Trace.println("main IR:");
63         // Trace.println(mainMethodNode.getIR());
64     
65         // Make sure call to hashCode is there (it uses the class constant)
66         TypeReference classRef = TypeReference.findOrCreate(ClassLoaderReference.Primordial, "Ljava/lang/Class");
67         MethodReference hashCodeRef = MethodReference.findOrCreate(classRef, "hashCode", "()I");
68         Set<CGNode> hashCodeNodes = cg.getNodes(hashCodeRef);
69         Assert.assertFalse(hashCodeNodes.isEmpty());
70     
71         // make sure call to hashCode from main
72         Assert.assertTrue(cg.hasEdge(mainMethodNode, hashCodeNodes.iterator().next()));
73       }

75     }
