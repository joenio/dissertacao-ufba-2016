11     package com.ibm.wala.ssa;
12     
13     
14     /**
15      * A value generated by a phi instruction.
16      * 
17      * Clients shouldn't use this ... it's only used internally during SSA construction.
18      */
19     public class PhiValue implements Value {
20       /**
21        * The phi instruction that defines this value
22        */
23       final private SSAPhiInstruction phi;
24     
25       /**
26        * @param phi The phi instruction that defines this value
27        */
28       PhiValue(SSAPhiInstruction phi) {
29         this.phi = phi;
30       }
31     
32       @Override
33       public String toString() {
34         return "v" + phi.getDef();
35       }
36     
37       /**
38        * @return The phi instruction that defines this value
39        */
40       public SSAPhiInstruction getPhiInstruction() {
41         return phi;
42       }
43     
44       /*
45        * @see com.ibm.wala.ssa.Value#isStringConstant()
46        */
47       @Override
48       public boolean isStringConstant() {
49         return false;
50       }
51     
52       /*
53        * @see com.ibm.wala.ssa.Value#isNullConstant()
54        */
55       @Override
56       public boolean isNullConstant() {
57         return false;
58       }
59     
60     }
