58     /**
59      * A CAst rewriter for extracting bits of code into one-shot closures. What to extract is determined 
60      * by an {@link ExtractionPolicy}.
61      * 
62      * <p>For instance, a {@link ForInBodyExtractionPolicy} extracts the body of every for-in loop in
63      * the program, whereas a {@link CorrelatedPairExtractionPolicy} extracts pieces of code containing
64      * correlated property reads and writes of the same property.</p>
65      * 
66      * <p>As an example, consider the following function:</p>
67      * <pre>
68      *   function extend(dest, src) {
69      *     for(var p in src)
70      *       dest[p] = src[p];
71      *   }
72      * </pre>
73      * <p>Under both {@link ForInBodyExtractionPolicy} and {@link CorrelatedPairExtractionPolicy}, this 
74      * should be transformed into</p>
75      * <pre>
76      *   function extend(dest, src) {
77      *     for(var p in src)
78      *       (function _forin_body_0(p) {
79      *         dest[p] = src[p];
80      *       })(p);
81      *   }
82      * </pre>
83      * 
84      * <p>There are four issues to be considered here.</p>
85      * <ul>
86      * <li><b>References to <code>this</code></b>: 
87      *   <p>If the code to extract contains references to <code>this</code>, these references have to be 
88      *   rewritten; otherwise they would refer to the global object in the transformed code.</p>
89      *   <p>We do this by giving the extracted function an extra parameter <code>thi$</code>, and rewriting
90      *   <code>this</code> to <code>thi$</code> within the extracted code.</p>
91      *   <p>For instance,</p>
92      *   <pre>
93      *   Object.prototype.extend = function(src) {
94      *     for(var p in src)
95      *       this[p] = src[p];
96      *   }
97      *   </pre>
98      *   <p>becomes</p>
99      *   <pre>
100      *   Object.prototype.extend = function(src) {
101      *     for(var p in src)
102      *       (function _forin_body_0(p, thi$) {
103      *         thi$[p] = src[p];
104      *       })(p, this);
105      *   }
106      *   </pre>
107      * </li>
108      * <li><b>Local variable declarations</b>:
109      *   <p>Local variable declarations inside the extracted code have to be hoisted to the enclosing function;
110      *   otherwise they would become local variables of the extracted function instead.</p>
111      *   <p>This is already taken care of by the translation from Rhino's AST to CAst.</p>
112      *   <p>Optionally, the policy can request that one local variable of the surrounding function be turned into 
113      *   a local variable of the extracted closure. The rewriter checks that this is possible: the code to extract
114      *   must not contain function calls or <code>new</code> expressions, and it must not contain <code>break</code>,
115      *   <code>continue</code>, or <code>return</code> statements. The former requirement prevents a called function
116      *   from observing a different value of the local variable than before. The latter requirement is necessary
117      *   because the final value of the localised variable needs to be returned and assigned to its counterpart in
118      *   the surrounding function; since non-local jumps are encoded by special return values (see next item),
119      *   this would no longer be possible.</p>
120      * </li>
121      * <li><b><code>break</code>, <code>continue</code>, <code>return</code></b>:
122      *   <p>A <code>break</code> or <code>continue</code> statement within the extracted loop body that refers
123      *   to the loop itself or an enclosing loop would become invalid in the transformed code. A <code>return</code>
124      *   statement would no longer return from the enclosing function, but instead from the extracted function.</p>
125      *   <p>We transform all three statements into <code>return</code> statements returning an object literal with a
126      *   property <code>type</code> indicating whether this is a 'goto' (i.e., <code>break</code> or <code>return</code>)
127      *   or a 'return'. In the former case, the 'target' property contains an integer identifying the jump target; in
128      *   the latter case, the 'value' property contains the value to return.</p>
129      *   <p>The return value of the extracted function is then examined to determine whether it completed normally
130      *   (i.e., returned <code>undefined</code>), or whether it returned an object indicating special control flow.</p>
131      *   <p>For example, consider this code from MooTools:</p>
132      *   <pre>
133      *   for(var style in Element.ShortStyles) {
134      *     if(property != style)
135      *       continue;
136      *     for(var s in Element.ShortStyles[style])
137      *       result.push(this.getStyle(s));
138      *     return result.join(' ');
139      *   }
140      *   </pre>
141      *   <p>Under {@link ForInBodyExtractionPolicy}, this is transformed into</p>
142      *   <pre>
143      *   for(var style in Element.ShortStyles) {
144      *     var s;
145      *     re$ = (function _forin_body_0(style, thi$) {
146      *       if(property != style)
147      *         return { type: 'goto', target: 1 };
148      *       for(s in Element.ShortStyles[style]) {
149      *         (function _forin_body_2(s) {
150      *           result.push(thi$.getStyle(s));
151      *         })(s);
152      *       }
153      *       return { type: 'return', value: result.join(' ') };
154      *     })(style, this);
155      *     if(re$) {
156      *       if(re$.type == 'return')
157      *         return re$.value;
158      *       if(re$.type == 'goto') {
159      *         if(re$.target == 1)
160      *           continue;
161      *       }
162      *     }
163      *   }
164      *   </pre>
165      *   <p>Note that at the CAst level, <code>break</code> and <code>continue</code> are represented as <code>goto</code>
166      *   statements, which simplifies the translation somewhat. The numerical encoding of jump targets does not matter
167      *   as long as the extracted function and the fixup code agree on which number represents which label.</p>
168      * </li>
169      * <li><b>Assignment to loop variable</b>:
170      *   <p>The loop body may assign to the loop variable. If the variable is referenced after the loop, this assignment
171      *   needs to be propagated back to the enclosing function in the extracted code.</p>
172      *   <p><b>TODO:</b> This is not handled at the moment.</p> 
173      * </li>
174      * </ul>
175      * 
176      * <p>Finally, note that exceptions do not need to be handled specially.</p>
177      * 
178      * @author mschaefer
179      *
180      */

197       @Override
198       protected void enterEntity(CAstEntity entity) {
199         policies.push(policyFactory.createPolicy(entity));

204         policies.pop();

207       @Override
208       protected CAstNode copyNodes(CAstNode root, CAstControlFlowMap cfg, NodePos context, Map<Pair<CAstNode, NoKey>, CAstNode> nodeMap) {
209         switch(root.getKind()) {
210         case OPERATOR:
211           return root;
212         case CONSTANT:
213           return copyConstant(root, cfg, context, nodeMap);
214         case BLOCK_STMT:
215           return copyBlock(root, cfg, context, nodeMap);
216         case RETURN: 
217           return copyReturn(root, cfg, context, nodeMap);
218         case VAR: 
219           return copyVar(root, cfg, context, nodeMap);
220         case GOTO:
221           return copyGoto(root, cfg, context, nodeMap);
222         default:
223           return copyNode(root, cfg, context, nodeMap);
224         }
225       }

227       /* Constants are not affected by the rewriting, they are just copied. */
228       private CAstNode copyConstant(CAstNode root, CAstControlFlowMap cfg, NodePos context, Map<Pair<CAstNode, NoKey>, CAstNode> nodeMap) {
229         CAstNode newNode = Ast.makeConstant(root.getValue());
230         nodeMap.put(Pair.make(root, context.key()), newNode);
231         return newNode;
232       }

234       /* Ask the policy whether it wants anything extracted from this block; otherwise the node is simply copied. */
235       private CAstNode copyBlock(CAstNode root, CAstControlFlowMap cfg, NodePos context, Map<Pair<CAstNode, NoKey>, CAstNode> nodeMap) {
236         List<ExtractionRegion> regions = policies.getFirst().extract(root);
237         if(regions == null || usesArguments(root)) {
238           return copyNode(root, cfg, context, nodeMap);
239         } else {
240           ArrayList<CAstNode> copied_children = new ArrayList<CAstNode>();
241           int next_child = 0;
242           // code in between regions is handled by invoking copyNodes, the regions themselves by extractRegion
243           for(ExtractionRegion region : regions) {
244             for(;next_child<region.getStart();++next_child)
245               copied_children.add(copyNodes(root.getChild(next_child), cfg, new ChildPos(root, next_child, context), nodeMap));
246             for(CAstNode stmt : extractRegion(root, cfg, new ExtractionPos(root, region, context), nodeMap))
247               copied_children.add(stmt);
248             next_child = region.getEnd();
249           }
250           for(;next_child<root.getChildCount();++next_child)
251             copied_children.add(copyNodes(root.getChild(next_child), cfg, new ChildPos(root, next_child, context), nodeMap));
252           CAstNode newNode = Ast.makeNode(root.getKind(), copied_children.toArray(new CAstNode[0]));
253           nodeMap.put(Pair.make(root, context.key()), newNode);
254           return newNode;
255         }
256       }

258       /*
259        * Normal variables are just copied, but 'this' references need to be rewritten if we are inside an extracted
260        * function body.
261        */
262       private CAstNode copyVar(CAstNode root, CAstControlFlowMap cfg, NodePos context, Map<Pair<CAstNode, NoKey>, CAstNode> nodeMap) {
263         /*
264          * If this node is a "this" reference, the outermost enclosing extracted function needs to pass in
265          * the value of "this" as a parameter.
266          * 
267          * NB: This has to be done by the _outermost_ function. If it were done by an inner function instead,
268          *     the outermost one may not pass in the value of "this" at all, so the inner one would
269          *     get the wrong value.
270          */
271         if(root.getChild(0).getValue().equals("this")) {
272           ExtractionPos epos = ExtractionPos.getOutermostEnclosingExtractionPos(context);
273           if(epos != null) {
274             epos.addThis();
275             CAstNode newNode = makeVarRef(epos.getThisParmName());
276             addExnFlow(newNode, JavaScriptTypes.ReferenceError, getCurrentEntity(), context);
277             nodeMap.put(Pair.make(root, context.key()), newNode);
278             return newNode;
279           } else {
280             return copyNode(root, cfg, context, nodeMap);
281           }
282         } else {
283           return copyNode(root, cfg, context, nodeMap);
284         }
285       }

287       /*
288        * 'break' and 'continue' statements are both encoded as GOTO. If they refer to a target outside the innermost
289        * enclosing extracted function body, they are rewritten into a 'return' statement.
290        */
291       private CAstNode copyGoto(CAstNode root, CAstControlFlowMap cfg, NodePos context, Map<Pair<CAstNode, NoKey>, CAstNode> nodeMap) {
292         CAstNode target = getCurrentEntity().getControlFlow().getTarget(root, null);
293         ExtractionPos epos = ExtractionPos.getEnclosingExtractionPos(context);
294         if(epos != null && !NodePos.inSubtree(target, epos.getParent())) {
295           epos.addGotoTarget(root.getChildCount() > 0 ? (String)root.getChild(0).getValue(): null, target);
296           int label = labeller.addNode(target);
297           // return { type: 'goto', target: <label> }
298           CAstNode returnLit = 
299               addNode(
300                   Ast.makeNode(OBJECT_LITERAL,
301                   addExnFlow(
302                       Ast.makeNode(CALL,
303                           addExnFlow(makeVarRef("Object"), JavaScriptTypes.ReferenceError, getCurrentEntity(), context),
304                           Ast.makeConstant("ctor")), 
305                       null, getCurrentEntity(), context),
306               Ast.makeConstant("type"),
307               Ast.makeConstant("goto"),
308               Ast.makeConstant("target"),
309               Ast.makeConstant(((double)label)+"")),
310               getCurrentEntity().getControlFlow());
311     
312           addNode(returnLit, getCurrentEntity().getControlFlow());
313           CAstNode newNode = Ast.makeNode(RETURN, returnLit);
314           // remove outgoing cfg edges of the old node
315           deleteFlow(root, getCurrentEntity());
316           nodeMap.put(Pair.make(root, context.key()), newNode);
317           return newNode;
318         } else {
319           return copyNode(root, cfg, context, nodeMap);
320         }
321       }

323       /* 'return' statements inside an extracted function body need to be encoded in a similar fashion. */
324       private CAstNode copyReturn(CAstNode root, CAstControlFlowMap cfg, NodePos context, Map<Pair<CAstNode, NoKey>, CAstNode> nodeMap) {
325         ExtractionPos epos = ExtractionPos.getEnclosingExtractionPos(context);
326     
327         if(epos == null || isSynthetic(root))
328           return copyNode(root, cfg, context, nodeMap);
329     
330         // add a return to every enclosing extracted function body
331         do {
332           epos.addReturn();
333           epos = ExtractionPos.getEnclosingExtractionPos(epos.getParentPos());
334         } while(epos != null);
335     
336         // emit appropriate 'return' statement
337         if(root.getChildCount() > 0) {
338           // return { type: 'return', value: <retval> }
339           CAstNode retval = copyNodes(root.getChild(0), cfg, new ChildPos(root, 0, context), nodeMap);
340           CAstNode newNode = 
341               Ast.makeNode(RETURN,
342                 addNode(
343                   Ast.makeNode(OBJECT_LITERAL,
344                       addExnFlow(Ast.makeNode(CALL,
345                           addExnFlow(makeVarRef("Object"), JavaScriptTypes.ReferenceError, getCurrentEntity(), context),
346                           Ast.makeConstant("ctor")), null, getCurrentEntity(), context),
347                           Ast.makeConstant("type"),
348                           Ast.makeConstant("return"),
349                           Ast.makeConstant("value"),
350                           retval),
351                   getCurrentEntity().getControlFlow()));
352           nodeMap.put(Pair.make(root, context.key()), newNode);
353           return newNode;
354         } else {
355           // return { type: 'return' }
356           CAstNode newNode = 
357               Ast.makeNode(RETURN,
358                 addNode(
359                   Ast.makeNode(OBJECT_LITERAL,
360                       addExnFlow(Ast.makeNode(CALL,
361                           addExnFlow(makeVarRef("Object"), JavaScriptTypes.ReferenceError, getCurrentEntity(), context),
362                           Ast.makeConstant("ctor")), null, getCurrentEntity(), context),
363                           Ast.makeConstant("type"),
364                           Ast.makeConstant("return")),
365                       getCurrentEntity().getControlFlow()));
366                    
367           nodeMap.put(Pair.make(root, context.key()), newNode);
368           return newNode;
369         }
370       }

372       /* Recursively copy child nodes. */
373       private CAstNode copyNode(CAstNode node, CAstControlFlowMap cfg, NodePos context, Map<Pair<CAstNode, NoKey>, CAstNode> nodeMap) {
374         CAstNode children[] = new CAstNode[node.getChildCount()];
375         
376         // copy children
377         for (int i = 0; i < children.length; i++) {
378           children[i] = copyNodes(node.getChild(i), cfg, new ChildPos(node, i, context), nodeMap);
379         }
380         
381         // for non-constant case labels, the case expressions appear as labels on CFG edges; rewrite those as well
382         for(Object label: cfg.getTargetLabels(node)) {
383           if (label instanceof CAstNode) {
384             copyNodes((CAstNode)label, cfg, new LabelPos(node, context), nodeMap);
385           }
386         }
387         
388         CAstNode newNode = Ast.makeNode(node.getKind(), children);
389         nodeMap.put(Pair.make(node, context.key()), newNode);
390     
391         // if this node has a control flow successor beyond the innermost enclosing extracted function loop, we need to reroute
392         ExtractionPos epos = ExtractionPos.getEnclosingExtractionPos(context);
393         if(!isFlowDeleted(newNode, getCurrentEntity()) && epos != null) {
394           // CAstControlFlowMap cfg = getCurrentEntity().getControlFlow();
395           Collection<Object> labels = cfg.getTargetLabels(node);
396           boolean invalidateCFlow = false;
397           for(Object label : labels) {
398             CAstNode target = cfg.getTarget(node, label);
399             if(target != CAstControlFlowMap.EXCEPTION_TO_EXIT && !epos.contains(target)) {
400               invalidateCFlow = true;
401               break;
402             }
403           }
404           if(invalidateCFlow) {
405             deleteFlow(node, getCurrentEntity());
406             for(Object label : labels) {
407               CAstNode target = cfg.getTarget(node, label);
408               if(epos.contains(target))
409                 addFlow(node, label, target, cfg);
410               else
411                 addFlow(node, label, CAstControlFlowMap.EXCEPTION_TO_EXIT, cfg);
412             }
413           }
414         }
415         return newNode;
416       }

418       private int anonymous_counter = 0;
419       
420       private List<CAstNode> extractRegion(CAstNode root, CAstControlFlowMap cfg, ExtractionPos context, Map<Pair<CAstNode, NoKey>, CAstNode> nodeMap) {
421         CAstEntity entity = getCurrentEntity();
422     
423         // whether we are extracting a single statement that is itself a block
424         boolean extractingBlock = context.getStart() + 1 == context.getEnd() && root.getChild(context.getStart()).getKind() == BLOCK_STMT;
425         
426         // whether we are extracting the body of a local scope
427         boolean extractingLocalScope = false;
428         
429         // whether we are extracting an empty loop body
430         boolean extractingEmpty = false;
431     
432         String name = EXTRACTED_FUN_BASENAME + (anonymous_counter++);
433         
434         // Create a new entity for the extracted function.
435         ExtractedFunction new_entity = new ExtractedFunction(name, context);
436         context.setExtractedEntity(new_entity);
437     
438         // rewrite the code to be extracted
439     
440         /*
441          * First, we need to massage the code a little bit, prepending an assignment of the form '<name> = <name>'
442          * and appending a RETURN statement if it may complete normally (i.e., if execution may 'fall off'
443          * the end). Additionally, if the extraction starts inside a nested BLOCK_EXPR, we flatten it out into a list
444          * of statements.
445          * 
446          * The whole thing is then wrapped into a block.
447          */
448         ArrayList<CAstNode> prologue = new ArrayList<CAstNode>();
449         ArrayList<CAstNode> fun_body_stmts = new ArrayList<CAstNode>();
450     
451         // if we are extracting a block, unwrap it
452         if(extractingBlock) {
453           CAstNode block = root.getChild(context.getStart());
454           for(int i=0;i<block.getChildCount();++i)
455             fun_body_stmts.add(block.getChild(i));
456         } else {
457           if(context.getRegion() instanceof TwoLevelExtractionRegion) {
458             CAstNode start = root.getChild(context.getStart());
459             TwoLevelExtractionRegion tler = (TwoLevelExtractionRegion)context.getRegion();
460             if(tler.getEndInner() != -1)
461               throw new UnimplementedError("Two-level extraction not fully implemented.");
462             int i;
463             if(start.getKind() == CAstNode.BLOCK_STMT) {
464               CAstNode[] before = new CAstNode[tler.getStartInner()];
465               for(i=0;i<tler.getStartInner();++i)
466                 before[i] = copyNodes(start.getChild(i), cfg, context, nodeMap);
467               for(int x = 0; x < before.length; x++) {
468                 prologue.add(before[x]);
469               }
470               if(i+1 == start.getChildCount()) {
471                 fun_body_stmts.add(addSpuriousExnFlow(start.getChild(i), cfg));            
472               } else {
473                 CAstNode[] after = new CAstNode[start.getChildCount()-i];
474                 for(int j=0;j+i<start.getChildCount();++j)
475                   after[j] = addSpuriousExnFlow(start.getChild(j+i), cfg);
476                 for(int x = 0; x < after.length; x++) {
477                   fun_body_stmts.add(after[x]);
478                 }
479               }
480               for(i=context.getStart()+1;i<context.getEnd();++i)
481                 fun_body_stmts.add(root.getChild(i));
482             } else if(start.getKind() == CAstNode.LOCAL_SCOPE) {
483               if(tler.getStartInner() != 0 || tler.getEnd() != tler.getStart() + 1)
484                 throw new UnimplementedError("Unsupported two-level extraction");
485               fun_body_stmts.add(start.getChild(0));
486               extractingLocalScope = true;
487             } else {
488               throw new UnimplementedError("Unsupported two-level.");
489             }
490           } else {
491             if(context.getEnd() > context.getStart()+1) {
492               CAstNode[] stmts = new CAstNode[context.getEnd()-context.getStart()];
493               for(int i=context.getStart();i<context.getEnd();++i)
494                 stmts[i-context.getStart()] = root.getChild(i);
495               fun_body_stmts.add(Ast.makeNode(root.getKind(), stmts));
496             } else {
497               CAstNode node_to_extract = root.getChild(context.getStart());
498               if(node_to_extract.getKind() == CAstNode.EMPTY)
499                 extractingEmpty = true;
500               fun_body_stmts.add(wrapIn(BLOCK_STMT, node_to_extract));
501             }
502           }
503         }
504         
505         List<String> locals = context.getRegion().getLocals();
506         String theLocal = null;
507         if(LOCALISE && locals.size() == 1 && noJumpsAndNoCalls(fun_body_stmts)) {
508           // the variable can be localised, remember its name
509           theLocal = locals.get(0);
510           
511           // append "return <theLocal>;" to the end of the function body
512           CAstNode retLocal = Ast.makeNode(RETURN, addExnFlow(makeVarRef(theLocal), JavaScriptTypes.ReferenceError, entity, context));
513           markSynthetic(retLocal);
514           // insert as last stmt if fun_body_stmts is a single block, otherwise append
515           if(fun_body_stmts.size() == 1 && fun_body_stmts.get(0).getKind() == BLOCK_STMT) {
516             CAstNode[] stmts = new CAstNode[fun_body_stmts.get(0).getChildCount()+1];
517             for(int i=0;i<stmts.length-1;++i)
518               stmts[i] = fun_body_stmts.get(0).getChild(i);
519             stmts[stmts.length-1] = retLocal;
520             fun_body_stmts.set(0, Ast.makeNode(BLOCK_STMT, stmts));
521           } else {
522             fun_body_stmts.add(retLocal);
523           }
524           
525           // prepend declaration "var <theLocal>;"
526           CAstNode theLocalDecl = Ast.makeNode(DECL_STMT, Ast.makeConstant(new CAstSymbolImpl(theLocal, JSAstTranslator.Any)),
527                                                addExnFlow(makeVarRef("$$undefined"), JavaScriptTypes.ReferenceError, entity, context));
528           
529           if(fun_body_stmts.size() > 1) {
530             CAstNode newBlock = Ast.makeNode(BLOCK_STMT, fun_body_stmts.toArray(new CAstNode[0]));
531             fun_body_stmts.clear();
532             fun_body_stmts.add(newBlock);
533           }
534           
535           
536           // fun_body_stmts.add(0, Ast.makeNode(BLOCK_STMT, theLocalDecl));
537           fun_body_stmts.add(0, theLocalDecl);
538         }
539     
540         CAstNode fun_body = Ast.makeNode(BLOCK_STMT, fun_body_stmts.toArray(new CAstNode[0]));
541     
542         /*
543          * Now we rewrite the body and construct a Rewrite object.
544          */
545         final Map<Pair<CAstNode, NoKey>, CAstNode> nodes = HashMapFactory.make();
546         final CAstNode newRoot = copyNodes(fun_body, cfg, context, nodes);
547         final CAstSourcePositionMap theSource = copySource(nodes, entity.getSourceMap());
548         final CAstControlFlowMap theCfg = copyFlow(nodes, entity.getControlFlow(), theSource);
549         final CAstNodeTypeMap theTypes = copyTypes(nodes, entity.getNodeTypeMap());
550         final Map<CAstNode, Collection<CAstEntity>> theChildren = HashMapFactory.make();
551         for(int i=context.getStart();i<context.getEnd();++i)
552           theChildren.putAll(copyChildren(root.getChild(i), nodes, entity.getAllScopedEntities()));
553     
554         Rewrite rw = new Rewrite() {
555           @Override
556           public CAstNode newRoot() { return newRoot; }
557           @Override
558           public CAstControlFlowMap newCfg() { return theCfg; }
559           @Override
560           public CAstSourcePositionMap newPos() { return theSource; }
561           @Override
562           public CAstNodeTypeMap newTypes() { return theTypes; }
563           @Override
564           public Map<CAstNode, Collection<CAstEntity>> newChildren() { return theChildren; }
565         };
566         new_entity.setRewrite(rw);
567     
568         /* Now we construct a call to the extracted function.
569          * 
570          * If the body never referenced 'this', the function call will be of the form
571          * 
572          *   <extracted_fun_expr>("do", <global_object>, <parm1>, ... , <parmn>)
573          *   
574          * The "do" argument is a dummy argument required by the CAst encoding for JavaScript.
575          * 
576          * If, on the other hand, there are references to 'this', these will have been rewritten into references
577          * to an additional parameter 'thi$', so the call will be of the form
578          * 
579          *   <extracted_fun_expr>("do", <global_object>, <parm1>, ..., <parmn>, this)
580          *   
581          * if we are in a function, and
582          * 
583          *   <extracted_fun_expr>("do", <global_object>, <parm1>, ..., <parmn>, <global_object>)
584          *   
585          * at the top-level.
586          * 
587          * In any case, we also add a CFG edge for a ReferenceError exception on the <parmi> arguments, and for
588          * a null pointer exception on the call; these are infeasible, but we add them anyway to get the same AST 
589          * as with a manual extraction.
590          */
591         List<CAstNode> args = new ArrayList<CAstNode>();
592         CAstNode funExpr = Ast.makeNode(FUNCTION_EXPR, Ast.makeConstant(new_entity));
593         args.add(funExpr);
594         context.setCallSite(funExpr);
595         ExtractionPos outer = ExtractionPos.getEnclosingExtractionPos(context.getParentPos());
596         if(outer == null) {
597           addEntity(funExpr, new_entity);
598         } else {
599           outer.addNestedPos(context);
600         }
601     
602         args.add(Ast.makeConstant("do"));
603         args.add(addNode(makeVarRef("__WALA__int3rnal__global"), entity.getControlFlow()));
604         for(String parmName : context.getParameters())
605           args.add(addExnFlow(makeVarRef(parmName), JavaScriptTypes.ReferenceError, entity, context));
606         if(context.containsThis())
607           args.add(inFunction() ? Ast.makeNode(VAR, Ast.makeConstant("this")) : Ast.makeConstant(null));
608         CAstNode call = Ast.makeNode(CALL, args.toArray(new CAstNode[0])); 
609         addExnFlow(call, null, entity, context);
610     
611         // if the extracted code contains jumps, we need to insert some fix-up code
612         List<CAstNode> stmts = new ArrayList<CAstNode>(prologue);
613         if(context.containsJump()) {
614           CAstNode decl = Ast.makeNode(ASSIGN,
615               addExnFlow(makeVarRef("re$"), JavaScriptTypes.ReferenceError, entity, context),
616               call);
617     
618           CAstNode fixup = null;
619           if(context.containsGoto())
620             fixup = createGotoFixup(context, entity);
621           if(context.containsReturn()) {
622             if(context.isOutermost()) {
623               CAstNode return_fixup = createReturnFixup(context, entity);
624               if(fixup != null)
625                 fixup = Ast.makeNode(BLOCK_STMT, return_fixup, fixup);
626               else
627                 fixup = return_fixup;
628             } else {
629               fixup = Ast.makeNode(RETURN, addExnFlow(makeVarRef("re$"), JavaScriptTypes.ReferenceError, entity, context));
630             }
631           }
632     
633           // if this is a nested for-in loop, we need to pass on unhandled jumps
634           if(!context.isOutermost() && (context.containsReturn() || context.containsOuterGoto()))
635             fixup = Ast.makeNode(RETURN, addExnFlow(makeVarRef("re$"), JavaScriptTypes.ReferenceError, entity, context));
636     
637           // if(re$) <check>;
638           fixup = Ast.makeNode(IF_STMT, 
639               addExnFlow(makeVarRef("re$"), JavaScriptTypes.ReferenceError, entity, context),
640               Ast.makeNode(LOCAL_SCOPE, wrapIn(BLOCK_STMT, fixup == null ? Ast.makeNode(EMPTY) : fixup)));
641     
642           stmts.add(decl);
643           stmts.add(fixup);
644         } else if(theLocal != null) {
645           // assign final value of the localised variable back
646           stmts.add(Ast.makeNode(CAstNode.ASSIGN, 
647               addExnFlow(makeVarRef(theLocal), JavaScriptTypes.ReferenceError, entity, context), 
648               call));
649         } else {
650           stmts.add(call);
651         }
652     
653         if(extractingBlock) {
654           // put the call and the fixup code together
655           CAstNode newNode = Ast.makeNode(BLOCK_STMT, stmts.toArray(new CAstNode[0]));
656           nodeMap.put(Pair.make(root, context.key()), newNode);
657           deleteFlow(root, getCurrentEntity());
658           stmts = Collections.singletonList(newNode);
659         }
660         
661         if(extractingLocalScope || extractingEmpty) {
662           CAstNode newNode = Ast.makeNode(LOCAL_SCOPE, wrapIn(BLOCK_STMT, stmts.toArray(new CAstNode[0])));
663           stmts = Collections.singletonList(newNode);
664         }
665         
666         return stmts;
667       }

669       private CAstNode addSpuriousExnFlow(CAstNode node, CAstControlFlowMap cfg) {
670         CAstControlFlowRecorder flow = (CAstControlFlowRecorder)cfg;
671         if(node.getKind() == ASSIGN) {
672           if(node.getChild(0).getKind() == VAR) {
673             CAstNode var = node.getChild(0);
674             if(!flow.isMapped(var))
675               flow.map(var, var);
676             flow.add(var, CAstControlFlowMap.EXCEPTION_TO_EXIT, JavaScriptTypes.ReferenceError);
677           }
678         }
679         return node;
680       }

682       private CAstNode createReturnFixup(ExtractionPos context, CAstEntity entity) {
683         return Ast.makeNode(IF_STMT,
684             Ast.makeNode(BINARY_EXPR,
685                 CAstOperator.OP_EQ,
686                 addExnFlow(Ast.makeNode(OBJECT_REF,
687                     addExnFlow(makeVarRef("re$"), JavaScriptTypes.ReferenceError, entity, context),
688                     Ast.makeConstant("type")), JavaScriptTypes.TypeError, entity, context),
689                 Ast.makeConstant("return")),
690             Ast.makeNode(RETURN,
691                 addExnFlow(Ast.makeNode(OBJECT_REF,
692                     addExnFlow(makeVarRef("re$"), JavaScriptTypes.ReferenceError, entity, context),
693                     Ast.makeConstant("value")), JavaScriptTypes.TypeError, entity, context)));
694       }
695     
696       private CAstNode createGotoFixup(ExtractionPos context, CAstEntity entity) {
697         CAstNode fixup = null;
698     
699         // add fixup code for every goto in the extracted code
700         for(Pair<String, CAstNode> goto_target : context.getGotoTargets()) {
701           // if(re$.target == <goto_target>) goto <goto_target>; else <fixup>
702           CAstNode cond = Ast.makeNode(BINARY_EXPR,
703               CAstOperator.OP_EQ,
704               addExnFlow(Ast.makeNode(OBJECT_REF,
705                   addExnFlow(makeVarRef("re$"), JavaScriptTypes.ReferenceError, entity, context),
706                   Ast.makeConstant("target")), JavaScriptTypes.TypeError, entity, context),
707               Ast.makeConstant((double)labeller.getLabel(goto_target.snd)+""));
708           CAstNode then_branch;
709           if(goto_target.fst != null)
710             then_branch = Ast.makeNode(GOTO, Ast.makeConstant(goto_target.fst));
711           else
712             then_branch = Ast.makeNode(GOTO);
713           addFlow(then_branch, null, goto_target.snd, entity.getControlFlow());
714           if(fixup != null)
715             fixup = Ast.makeNode(IF_STMT, cond, then_branch, fixup);
716           else
717             fixup = Ast.makeNode(IF_STMT, cond, then_branch);
718         }
719     
720         // add check whether re$ is actually a 'goto'
721         return Ast.makeNode(IF_STMT,
722             Ast.makeNode(BINARY_EXPR,
723                 CAstOperator.OP_EQ,
724                 addExnFlow(Ast.makeNode(OBJECT_REF,
725                     addExnFlow(makeVarRef("re$"), JavaScriptTypes.ReferenceError, entity, context),
726                     Ast.makeConstant("type")), JavaScriptTypes.TypeError, entity, context),
727                 Ast.makeConstant("goto")),
728             Ast.makeNode(LOCAL_SCOPE, wrapIn(BLOCK_STMT, fixup)));
729       }

731       // wrap given nodes into a node of the given kind, unless there is only a single node which is itself of the same kind
732       private CAstNode wrapIn(int kind, CAstNode... nodes) {
733         return nodes.length == 1 && nodes[0].getKind() == kind ? nodes[0] : Ast.makeNode(kind, nodes);
734       }

737       private CAstNode addExnFlow(CAstNode node, Object label, CAstEntity entity, NodePos pos) {
738         return addExnFlow(node, label, entity.getControlFlow(), pos);
739       }

741       private CAstNode addExnFlow(CAstNode node, Object label, CAstControlFlowMap flow, NodePos pos) {
742         CAstNode target = getThrowTarget(pos);
743         return addFlow(node, label, target, flow);
744       }

746       // determine the innermost enclosing throw target at position pos
747       private CAstNode getThrowTarget(NodePos pos) {
748         return pos.accept(new PosSwitch<CAstNode>() {
749           @Override 
750           public CAstNode caseRootPos(RootPos pos) { 
751             return CAstControlFlowMap.EXCEPTION_TO_EXIT; 
752           }
753     
754           @Override
755           public CAstNode caseChildPos(ChildPos pos) {
756             int kind = pos.getParent().getKind();
757             if(kind == TRY && pos.getIndex() == 0)
758               return pos.getParent().getChild(1);
759             if(kind == FUNCTION_EXPR || kind == FUNCTION_STMT)
760               return CAstControlFlowMap.EXCEPTION_TO_EXIT;
761             return getThrowTarget(pos.getParentPos());
762           }
763     
764           @Override
765           public CAstNode caseForInLoopBodyPos(ExtractionPos pos) {
766             return getThrowTarget(pos.getParentPos());
767           }
768     
769           @Override
770           public CAstNode caseLabelPos(LabelPos pos) {
771             return getThrowTarget(pos.getParentPos());
772           }
773         });
774       }

781       // determine whether we are inside a function
782       private boolean inFunction() {
783         for(CAstEntity e : getEnclosingEntities())
784           if(e.getKind() == CAstEntity.FUNCTION_ENTITY)
785             return true;
786         return false;
787       }

789       /*
790        * Due to the way CAst rewriting works, we sometimes have to insert nodes before rewriting
791        * a subtree. These nodes should not usually be rewritten again, however, so we mark them
792        * as "synthetic". Currently, this only applies to "return" statements.
793        */
794       private final Set<CAstNode> synthetic = HashSetFactory.make();
795       private void markSynthetic(CAstNode node) {
796         this.synthetic.add(node);
797       }
798       private boolean isSynthetic(CAstNode node) {
799         return synthetic.contains(node);
800       }

802       private boolean noJumpsAndNoCalls(Collection<CAstNode> nodes) {
803         for(CAstNode node : nodes)
804           if(!noJumpsAndNoCalls(node))
805             return false;
806         return true;
807       }
808       
809       // determine whether the given subtree contains no unstructured control flow and calls
810       private boolean noJumpsAndNoCalls(CAstNode node) {
811         switch(node.getKind()) {
812         case CAstNode.BREAK:
813         case CAstNode.CONTINUE:
814         case CAstNode.GOTO:
815         case CAstNode.RETURN:
816         case CAstNode.CALL:
817         case CAstNode.NEW:
818           return false;
819         }
820         for(int i=0;i<node.getChildCount();++i)
821           if(!noJumpsAndNoCalls(node.getChild(i)))
822             return false;
823         return true;
824       }
825       
826       // determines whether the given subtree refers to the variable "arguments"
827       private boolean usesArguments(CAstNode node) {
828         if(node.getKind() == CAstNode.VAR) {
829             return node.getChild(0).getValue().equals("arguments");
830         } else {
831           for(int i=0;i<node.getChildCount();++i)
832             if(usesArguments(node.getChild(i)))
833               return true;
834           return false;      
835         }
836       }
