11     package com.ibm.wala.ssa;
12     
13     import com.ibm.wala.types.TypeReference;
14     
15     /**
16      * SSA instruction representing an array load.
17      */
18     public abstract class SSAArrayLoadInstruction extends SSAArrayReferenceInstruction {
19       private final int result;
20     
21       protected SSAArrayLoadInstruction(int iindex, int result, int arrayref, int index, TypeReference elementType) {
22         super(iindex, arrayref, index, elementType);
23         this.result = result;
24       }
25     
26       @Override
27       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) throws IllegalArgumentException {
28         if (defs != null && defs.length == 0) {
29           throw new IllegalArgumentException("defs.length == 0");
30         }
31         if (uses != null && uses.length < 2) {
32           throw new IllegalArgumentException("uses.length < 2");
33         }
34         return insts.ArrayLoadInstruction(iindex, defs == null ? result : defs[0], uses == null ? getArrayRef() : uses[0],
35             uses == null ? getIndex() : uses[1], getElementType());
36       }
37     
38       @Override
39       public String toString(SymbolTable symbolTable) {
40         return getValueString(symbolTable, result) + " = arrayload " + getValueString(symbolTable, getArrayRef()) + "["
41             + getValueString(symbolTable, getIndex()) + "]";
42       }
43     
44       /**
45        * @see com.ibm.wala.ssa.SSAInstruction#visit(IVisitor)
46        * @throws IllegalArgumentException
47        *             if v is null
48        */
49       @Override
50       public void visit(IVisitor v) {
51         if (v == null) {
52           throw new IllegalArgumentException("v is null");
53         }
54         v.visitArrayLoad(this);
55       }
56     
57       /**
58        * @see com.ibm.wala.ssa.SSAInstruction#getDef()
59        */
60       @Override
61       public boolean hasDef() {
62         return true;
63       }
64     
65       @Override
66       public int getDef() {
67         return result;
68       }
69     
70       @Override
71       public int getDef(int i) {
72         if (i != 0) {
73           throw new IllegalArgumentException("illegal i: " + i);
74         }
75         return result;
76       }
77     
78       @Override
79       public int getNumberOfDefs() {
80         return 1;
81       }
82     
83       @Override
84       public int hashCode() {
85         return 6311 * result ^ 2371 * getArrayRef() + getIndex();
86       }
87     
88     
89     }
