11     package com.ibm.wala.types.annotations;
12     
13     import java.util.Collection;
14     
15     import com.ibm.wala.classLoader.FieldImpl;
16     import com.ibm.wala.classLoader.IClass;
17     import com.ibm.wala.classLoader.IField;
18     import com.ibm.wala.classLoader.IMethod;
19     import com.ibm.wala.classLoader.ShrikeCTMethod;
20     import com.ibm.wala.classLoader.ShrikeClass;
21     import com.ibm.wala.shrikeCT.InvalidClassFileException;
22     import com.ibm.wala.types.TypeName;
23     import com.ibm.wala.util.debug.Assertions;
24     
25     public class Annotations {
26       public static final TypeName INTERNAL = TypeName.findOrCreateClassName("com/ibm/wala/annotations", "Internal");
27     
28       public static final TypeName NONNULL = TypeName.findOrCreateClassName("com/ibm/wala/annotations", "NonNull");
29     
30       /**
31        * Does a particular method have a particular annotation?
32        */
33       public static boolean hasAnnotation(IMethod m, TypeName type) {
34         if (m instanceof ShrikeCTMethod) {
35           Collection<Annotation> annotations = null;
36           try {
37             annotations = ((ShrikeCTMethod) m).getRuntimeInvisibleAnnotations();
38           } catch (InvalidClassFileException e) {
39             e.printStackTrace();
40             Assertions.UNREACHABLE();
41           } 
42           for (Annotation a : annotations) {
43             if (a.getType().getName().equals(type)) {
44               return true;
45             }
46           }
47         }
48         return false;
49       }
50     
51       /**
52        * Does a particular class have a particular annotation?
53        */
54       public static boolean hasAnnotation(IClass c, TypeName type) {
55         if (c instanceof ShrikeClass) {
56           Collection<Annotation> annotations = null;
57           try {
58             annotations = ((ShrikeClass) c).getRuntimeInvisibleAnnotations();
59           } catch (InvalidClassFileException e) {
60             e.printStackTrace();
61             Assertions.UNREACHABLE();
62           }
63           for (Annotation a : annotations) {
64             if (a.getType().getName().equals(type)) {
65               return true;
66             }
67           }
68         }
69         return false;
70       }
71     
72       public static boolean hasAnnotation(IField field, TypeName type) {
73         if (field instanceof FieldImpl) {
74           FieldImpl f = (FieldImpl) field;
75           Collection<Annotation> annotations = f.getAnnotations();
76           if (annotations != null) {
77             for (Annotation a : annotations) {
78               if (a.getType().getName().equals(type)) {
79                 return true;
80               }
81             }
82           }
83         }
84         return false;
85       }
86     }
