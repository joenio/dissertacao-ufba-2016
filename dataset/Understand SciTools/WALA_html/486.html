29       public static <T> void assertColoring(Graph<T> G, Map<T,Integer> colors, boolean fullColor) {
30         for(T n : G) {
31           for(Iterator<T> ss = G.getSuccNodes(n); ss.hasNext(); ) {
32             T succ = ss.next();
33             if (!fullColor &&  (!colors.containsKey(n) || !colors.containsKey(succ)) ) {
34               continue;
35             }
36             Assert.assertTrue(n + " and succ: " + succ + " have same color: " + colors.get(n).intValue(), colors.get(n).intValue() != colors.get(succ).intValue()); 
37           }
38           for(Iterator<T> ps = G.getPredNodes(n); ps.hasNext(); ) {
39             T pred = ps.next();
40             if (!fullColor && (!colors.containsKey(n) || !colors.containsKey(pred)) ) {
41               continue;
42             }
43             Assert.assertTrue(n + " and pred: " + pred + " have same color:" + colors.get(n).intValue(), colors.get(n).intValue() != colors.get(pred).intValue()); 
44           }
45         }
46       }

48       private class TypedNode<T> extends NodeWithNumberedEdges {
49         private final T data;
50         
51         private TypedNode(T data) {
52           this.data = data;
53         }
54         
55         @Override
56         public String toString() {
57           return data.toString();
58         }
59       }
60         
61       private <T> NumberedGraph<TypedNode<T>> buildGraph(T[][] data) {
62         DelegatingNumberedGraph<TypedNode<T>> G = new DelegatingNumberedGraph<TypedNode<T>>();
63         Map<T,TypedNode<T>> nodes = HashMapFactory.make();
64         for(int i = 0; i < data.length; i++) {
65           TypedNode<T> n = new TypedNode<T>(data[i][0]);
66           nodes.put(data[i][0], n);
67           G.addNode(n);
68         }
69         for(int i = 0; i < data.length; i++) {
70           for(int j = 1; j < data[i].length; j++) {
71             G.addEdge(nodes.get(data[i][0]), nodes.get(data[i][j]));
72           }
73         }
74         
75         return G;
76       }

78         @Test
79         public void testOne() {
80           NumberedGraph<TypedNode<Integer>> G = 
81             buildGraph(new Integer[][]{
82                 new Integer[]{1, 6, 7, 8},
83                 new Integer[]{2, 5, 7, 8},
84                 new Integer[]{3, 5, 6, 8},
85                 new Integer[]{4, 7, 6, 5},
86                 new Integer[]{5, 2, 4, 3},
87                 new Integer[]{6, 3, 1, 4},
88                 new Integer[]{7, 1, 2, 4},
89                 new Integer[]{8, 1, 2, 3}});
90           ColoredVertices<TypedNode<Integer>> colors = new WelshPowell<TypedNode<Integer>>().color(G);
91           System.err.println(colors.getColors());
92           assertColoring(G, colors.getColors(), true);
93           Assert.assertTrue(colors.getNumColors() <= 4);
94         }
