11     package com.ibm.wala.util.collections;
12     
13     import java.util.Collection;
14     import java.util.Collections;
15     import java.util.Iterator;
16     import java.util.Map;
17     import java.util.Set;
18     
19     import com.ibm.wala.util.debug.UnimplementedError;
20     
21     /**
22      * This implementation of {@link Map} chooses between one of two implementations, depending on the size of the map.
23      */
24     public class BimodalMap<K, V> implements Map<K, V> {
25     
26       // what's the cutoff between small and big maps?
27       // this may be a time-space tradeoff; the caller must determine if
28       // it's willing to put up with slower random access in exchange for
29       // smaller footprint.
30       private final int cutOff;
31     
32       /**
33        * The implementation we delegate to
34        */
35       private Map<K, V> backingStore;
36     
37       /**
38        * @param cutoff the map size at which to switch from the small map implementation to the large map implementation
39        */
40       public BimodalMap(int cutoff) {
41         this.cutOff = cutoff;
42       }
43     
44       /*
45        * @see java.util.Map#size()
46        */
47       @Override
48       public int size() {
49         return (backingStore == null) ? 0 : backingStore.size();
50       }
51     
52       /*
53        * @see java.util.Map#isEmpty()
54        */
55       @Override
56       public boolean isEmpty() {
57         return (backingStore == null) ? true : backingStore.isEmpty();
58       }
59     
60       /*
61        * @see java.util.Map#containsKey(java.lang.Object)
62        */
63       @Override
64       public boolean containsKey(Object key) {
65         return (backingStore == null) ? false : backingStore.containsKey(key);
66       }
67     
68       /*
69        * @see java.util.Map#containsValue(java.lang.Object)
70        */
71       @Override
72       public boolean containsValue(Object value) {
73         return (backingStore == null) ? false : backingStore.containsValue(value);
74       }
75     
76       /*
77        * @see java.util.Map#get(java.lang.Object)
78        */
79       @Override
80       public V get(Object key) {
81         return (backingStore == null) ? null : backingStore.get(key);
82       }
83     
84       /*
85        * @see java.util.Map#put(java.lang.Object, java.lang.Object)
86        */
87       @Override
88       public V put(K key, V value) {
89         if (backingStore == null) {
90           backingStore = new SmallMap<K, V>();
91           backingStore.put(key, value);
92           return null;
93         } else {
94           if (backingStore instanceof SmallMap) {
95             V result = backingStore.put(key, value);
96             if (backingStore.size() > cutOff) {
97               transferBackingStore();
98             }
99             return result;
100           } else {
101             return backingStore.put(key, value);
102           }
103         }
104       }
105     
106       /**
107        * Switch backing implementation from a SmallMap to a HashMap
108        */
109       private void transferBackingStore() {
110         assert backingStore instanceof SmallMap;
111         SmallMap<K, V> S = (SmallMap<K, V>) backingStore;
112         backingStore = HashMapFactory.make(2 * S.size());
113         for (Iterator<K> it = S.keySet().iterator(); it.hasNext();) {
114           K key = it.next();
115           V value = S.get(key);
116           backingStore.put(key, value);
117         }
118       }
119     
120       /**
121        * @throws UnsupportedOperationException if the backingStore doesn't support remove
122        */
123       @Override
124       public V remove(Object key) {
125         return (backingStore == null) ? null : backingStore.remove(key);
126       }
127     
128       /*
129        * @see java.util.Map#putAll(java.util.Map)
130        */
131       @Override
132       @SuppressWarnings("unchecked")
133       public void putAll(Map<? extends K, ? extends V> t) throws UnsupportedOperationException {
134         if (t == null) {
135           throw new IllegalArgumentException("null t");
136         }
137         if (backingStore == null) {
138           int size = t.size();
139           if (size > cutOff) {
140             backingStore = HashMapFactory.make();
141           } else {
142             backingStore = new SmallMap<K, V>();
143           }
144           backingStore.putAll(t);
145           return;
146         } else {
147           if (backingStore instanceof SmallMap) {
148             if (t.size() > cutOff) {
149               Map<K, V> old = backingStore;
150               backingStore = (Map<K, V>) HashMapFactory.make(t);
151               backingStore.putAll(old);
152             } else {
153               backingStore.putAll(t);
154               if (backingStore.size() > cutOff) {
155                 transferBackingStore();
156               }
157               return;
158             }
159           } else {
160             backingStore.putAll(t);
161           }
162         }
163       }
164     
165       /*
166        * @see java.util.Map#clear()
167        */
168       @Override
169       public void clear() {
170         backingStore = null;
171       }
172     
173       /*
174        * @see java.util.Map#keySet()
175        */
176       @Override
177       @SuppressWarnings("unchecked")
178       public Set<K> keySet() {
179         return (Set<K>) ((backingStore == null) ? Collections.emptySet() : backingStore.keySet());
180       }
181     
182       /*
183        * @see java.util.Map#values()
184        */
185       @Override
186       @SuppressWarnings("unchecked")
187       public Collection<V> values() {
188         return (Collection<V>) ((backingStore == null) ? Collections.emptySet() : backingStore.values());
189       }
190     
191       /**
192        * @throws UnimplementedError if the backingStore implementation does
193        */
194       @Override
195       @SuppressWarnings("unchecked")
196       public Set<Map.Entry<K, V>> entrySet() {
197         return (Set<Entry<K, V>>) ((backingStore == null) ? Collections.emptySet() : backingStore.entrySet());
198       }
199     }
