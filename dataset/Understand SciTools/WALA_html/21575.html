37       private byte[] serialize(ClassWriter writer, List<StackMapFrame> frames) throws IOException {
38         ByteArrayOutputStream data = new ByteArrayOutputStream();
39         
40         for(StackMapFrame frame : frames) {
41           frame.write(data, writer);
42         }
43         
44         ByteArrayOutputStream bs = new ByteArrayOutputStream();
45         writeUShort(bs, writer.addCPUtf8("StackMapTable"));
46         writeInt(bs, data.size() + 2);
47         writeUShort(bs, frames.size());
48         data.writeTo(bs);
49         
50         return bs.toByteArray();
51     
52       }

62       private static List<StackMapFrame> remapStackFrames(List<StackMapFrame> sm, int[] newBytecodesToOldBytecodes) {
63         // mapping to new bytecode
64         Map<Integer,Integer> oldToNew = HashMapFactory.make();
65         for(int i = newBytecodesToOldBytecodes.length - 1; i >= 0; i--) {
66           oldToNew.put(newBytecodesToOldBytecodes[i], i);
67         }
68         
69         // positions of frames
70         int i = 1;
71         int positions[] = new int[ sm.size()];
72         Iterator<StackMapFrame> sms = sm.iterator();
73         int position = sms.next().getOffset();
74         positions[0] = oldToNew.get(position);
75         while (sms.hasNext()) {
76           position = position + sms.next().getOffset() + 1;
77           positions[i++] = oldToNew.get(position);
78         }
79         
80         // positions turned into offsets
81         for(i = positions.length-1; i > 0; i--) {
82           positions[i] = positions[i] - positions[i-1] - 1;
83         }
84         
85         // frames with new offsets
86         List<StackMapFrame> newFrames = new ArrayList<StackMapFrame>(sm.size());
87         for(i = 0; i < sm.size(); i++) {
88           newFrames.add(new StackMapFrame(sm.get(i), positions[i]));
89         }
90         
91         return newFrames;
92       }

98       static StackMapType item(String type) {
99         if (type == null) {
100           return Item.ITEM_Top;
101         } else if (type.equals(TYPE_null)) {
102           return Item.ITEM_Null;
103         } else if (type.equals(Analyzer.topType)) {
104           return Item.ITEM_Top;
105         } else if (type.equals(Analyzer.thisType)) {
106           return Item.ITEM_UninitializedThis;
107         } else if (type.equals(TYPE_int)) {
108           return Item.ITEM_Integer;
109         } else if (type.equals(TYPE_float)) {
110           return Item.ITEM_Float;
111         } else if (type.equals(TYPE_double)) {
112           return Item.ITEM_Double;
113         } else if (type.equals(TYPE_long)) {
114           return Item.ITEM_Long;
115         } else {
116           if (type.startsWith("#")) {
117             return new UninitializedType(type);        
118           } else {
119             return new ObjectType(type);
120           }
121         }
122       }

142       static StackMapType[] trim(StackMapType[] types) {
143         int i = types.length-1;
144         while (i >= 0 && (types[i] == null || types[i] == Item.ITEM_Null || types[i] == Item.ITEM_Top)) {
145           i--;
146         }
147         
148         if (i < 0) {
149           return new StackMapType[0];
150         } else if (i < types.length-1) {
151           StackMapType[] trimmed = new StackMapType[ i+1 ];
152           System.arraycopy(types, 0, trimmed, 0, i+1);
153           return trimmed;
154         } else {
155           return types;
156         }
157       }

159       private static String hackUnknown(String type) {
160         if (type == null) {
161           return type;
162         } else if (type.startsWith("[")) {
163           return "[" + hackUnknown(type.substring(1));
164         } else if ("L?;".equals(type)) {
165           return "Ljava/lang/Object;";
166         } else {
167           return type;
168         }
169       }

171       static StackMapType[] types(String[] types, boolean locals) {
172         StackMapType[] stackTypes = new StackMapType[ types.length ];
173         
174         int x = 0;
175         for(int j = 0; j < types.length; j++) {
176           StackMapType stackType = item(hackUnknown(types[j]));
177           stackTypes[x++] = stackType;
178           if (locals && stackType.size() == 2) {
179             j++;
180           }
181         }
182      
183         return trim(stackTypes);
184       }

186       private static boolean isUselessGoto(IInstruction inst, int index) {
187         if (inst instanceof GotoInstruction) {
188           if (((GotoInstruction)inst).getBranchTargets()[0] == index+1) {
189             return true;
190           }
191         }
192         
193         return false;
194       }

196       public static List<StackMapFrame> stackMapTable(ClassWriter writer, MethodData method, Output output, ClassHierarchyProvider cha, String[][] vars, List<StackMapFrame> reuseFrames) throws FailureException, IOException {
197         int idx = 0;
198         
199         List<StackMapFrame> frames = new ArrayList<StackMapFrame>();
200         
201         int[] instructionToBytecode = output.getInstructionOffsets();
202         IInstruction[] insts = method.getInstructions();
203         
204         Verifier typeChecker = new Verifier(method, instructionToBytecode, vars);
205         if (cha != null) {
206           typeChecker.setClassHierarchy(cha);
207         }
208         typeChecker.computeTypes();
209         BitSet bbs = typeChecker.getBasicBlockStarts();
210         
211         int offset = 0;
212         for(int i = 1; i < insts.length; i++) {
213           if (bbs.get(i)) {
214     
215             // Shrike does not generate goto i+1
216             if (isUselessGoto(insts[i], i)) {
217               continue;
218             }
219             
220             // offset delta
221             int position = instructionToBytecode[i];
222             assert position - offset > 0 || offset == 0;        
223             int frameOffset =  offset==0? position: position - offset - 1;
224             offset = position;
225             
226             if (reuseFrames != null) {
227               if (reuseFrames.get(idx).getOffset() == frameOffset) {
228                 frames.add(reuseFrames.get(idx++));
229                 continue;
230               } else {
231                 reuseFrames = null;
232               } 
233             } 
234             
235             // full frame
236             byte frameType = (byte)255;
237     
238             // locals
239             String[] localTypes = typeChecker.getLocalTypes()[i];
240             StackMapType[] localWriteTypes;
241             if (localTypes != null) {
242               localWriteTypes = types(localTypes, true);
243             } else {
244               localWriteTypes = new StackMapType[0]; 
245             }
246             
247             // stack
248             String[] stackTypes = typeChecker.getStackTypes()[i];
249             StackMapType[] stackWriteTypes;
250             if (stackTypes != null) {
251               stackWriteTypes = types(stackTypes, false);
252             } else {
253               stackWriteTypes = new StackMapType[0];         
254             }
255     
256             frames.add(new StackMapFrame(frameType, frameOffset, localWriteTypes, stackWriteTypes));
257           }
258         }
259         
260         return frames;
261       }
