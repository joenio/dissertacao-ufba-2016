11     package com.ibm.wala.classLoader;
12     
13     import java.io.File;
14     import java.util.Iterator;
15     import java.util.Set;
16     
17     import com.ibm.wala.util.collections.HashSetFactory;
18     
19     /**
20      * A module containing files under some directory.
21      */
22     public abstract class DirectoryTreeModule implements Module {
23     
24       protected final File root;
25     
26       /**
27        * @param root a directory
28        */
29       DirectoryTreeModule(File root) throws IllegalArgumentException {
30         this.root = root;
31         if (root == null) {
32           throw new IllegalArgumentException("null root");
33         }
34         if (!root.exists()) {
35           throw new IllegalArgumentException("root does not exist " + root);
36         }
37         if (!root.isDirectory()) {
38           throw new IllegalArgumentException("root is not a directory " + root);
39         }
40       }
41     
42       /**
43        * returns null if unsuccessful in creating FileModule
44        */
45       protected abstract FileModule makeFile(File file);
46     
47       protected abstract boolean includeFile(File file);
48     
49       private Set<FileModule> getEntriesRecursive(File dir) {
50         Set<FileModule> result = HashSetFactory.make();
51         File[] files = dir.listFiles();
52         if (files != null) {
53           for (int i = 0; i < files.length; i++) {
54             if (files[i].isDirectory()) {
55               result.addAll(getEntriesRecursive(files[i]));
56             } else if (includeFile(files[i])) {
57               FileModule fileModule = makeFile(files[i]);
58               if (fileModule != null) {
59                 result.add(fileModule);
60               }
61             }
62           }
63         } else {
64           // TODO: replace this with a real warning when the WarningSets are
65           // revamped
66           System.err.println(("Warning: failed to retrieve files in " + dir));
67         }
68     
69         return result;
70       }
71     
72       @Override
73       public Iterator<FileModule> getEntries() {
74         return getEntriesRecursive(root).iterator();
75       }
76     
77       public String getPath() {
78         return root.getAbsolutePath();
79       }
80     
81       @Override
82       public String toString() {
83         return getClass().getName() + ":" + getPath();
84       }
85     
86       @Override
87       public int hashCode() {
88         final int prime = 31;
89         int result = 1;
90         result = prime * result + ((root == null) ? 0 : root.hashCode());
91         return result;
92       }
93     
94       @Override
95       public boolean equals(Object obj) {
96         if (this == obj)
97           return true;
98         if (obj == null)
99           return false;
100         if (getClass() != obj.getClass())
101           return false;
102         final DirectoryTreeModule other = (DirectoryTreeModule) obj;
103         if (root == null) {
104           if (other.root != null)
105             return false;
106         } else if (!root.equals(other.root))
107           return false;
108         return true;
109       }
110     }
