11     package com.ibm.wala.util.graph.impl;
12     
13     import com.ibm.wala.util.graph.AbstractNumberedGraph;
14     import com.ibm.wala.util.graph.INodeWithNumber;
15     import com.ibm.wala.util.graph.NumberedEdgeManager;
16     import com.ibm.wala.util.graph.NumberedNodeManager;
17     import com.ibm.wala.util.intset.BasicNaturalRelation;
18     
19     /**
20      * A graph of numbered nodes, expected to have a fairly sparse edge structure.
21      */
22     public class SparseNumberedGraph<T extends INodeWithNumber> extends AbstractNumberedGraph<T> {
23     
24       private final DelegatingNumberedNodeManager<T> nodeManager;
25     
26       private final SparseNumberedEdgeManager<T> edgeManager;
27     
28       public SparseNumberedGraph() {
29         nodeManager = new DelegatingNumberedNodeManager<T>();
30         edgeManager = new SparseNumberedEdgeManager<T>(nodeManager);
31       }
32     
33       /**
34        * If normalCase == n, the s edge manager will eagerly allocated n words to hold out edges for each node. (performance
35        * optimization for time)
36        * 
37        * @param normalCase what is the "normal" number of out edges for a node?
38        */
39       public SparseNumberedGraph(int normalCase) {
40         nodeManager = new DelegatingNumberedNodeManager<T>();
41         edgeManager = new SparseNumberedEdgeManager<T>(nodeManager, normalCase, BasicNaturalRelation.TWO_LEVEL);
42       }
43     
44       public SparseNumberedGraph(DelegatingNumberedNodeManager<T> nodeManager, SparseNumberedEdgeManager<T> edgeManager) {
45         this.nodeManager = nodeManager;
46         this.edgeManager = edgeManager;
47       }
48     
49       /*
50        * @see com.ibm.wala.util.graph.AbstractGraph#getNodeManager()
51        */
52       @Override
53       protected NumberedNodeManager<T> getNodeManager() {
54         return nodeManager;
55       }
56     
57       /*
58        * @see com.ibm.wala.util.graph.AbstractGraph#getEdgeManager()
59        */
60       @Override
61       protected NumberedEdgeManager<T> getEdgeManager() {
62         return edgeManager;
63       }
64     }
