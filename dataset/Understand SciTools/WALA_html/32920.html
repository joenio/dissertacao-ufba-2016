50       @Test

54         Assert.assertTrue("cannot find project", p != null);
55       }

57       @Test
58       public void testProjectScope() throws IOException, CoreException {
59         IJavaScriptProject p = JavaScriptHeadlessUtil.getJavaScriptProjectFromWorkspace(project.projectName);
60         JSCallGraphUtil.setTranslatorFactory(new CAstRhinoTranslatorFactory());
61         AnalysisScope s = makeProjectPath(p).toAnalysisScope(new CAstAnalysisScope(JSCallGraphUtil.makeLoaders(), Collections.singleton(JavaScriptLoader.JS)));
62         System.err.println(s);
63         Assert.assertTrue("cannot make scope", s != null);
64         Assert.assertFalse("cannot find files", s.getModules(JavaScriptTypes.jsLoader).isEmpty());
65       }
66     
67       protected JavaScriptEclipseProjectPath makeProjectPath(IJavaScriptProject p) throws IOException, CoreException {
68         return JavaScriptEclipseProjectPath.make(p, Collections.<Pair<String,Plugin>>emptySet());
69       }

77         System.err.println(info.calls.size());

84       public void testEngine() throws IOException, CoreException, IllegalArgumentException, CancelException {
