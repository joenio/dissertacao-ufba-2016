53     /**
54      *  Stores references to the WALA-Intent objects.
55      *
56      *  This class is only of use in conjunction with the IntentContextSelector
57      *
58      *  @author Tobias Blaschke <code@tobiasblaschke.de>
59      */
60     /*package*/ class IntentMap {
61         private final Map<InstanceKey, Intent> seen = new HashMap<InstanceKey, Intent>();
62         private final Map<Intent, Intent> immutables = new HashMap<Intent, Intent>();
63     
64         public Intent findOrCreateImmutable(final Intent intent) {
65             if (immutables.containsKey(intent)) {
66                 final Intent immutable = immutables.get(intent);
67                 assert (immutable.getAction().equals(intent.getAction()));
68                 return immutable;
69             } else {
70                 final Intent immutable = intent.clone();
71                 immutable.setImmutable();
72                 immutables.put(intent, immutable);
73                 
74                 return immutable;
75             }
76         }
77     
78         public Intent find(final InstanceKey key) throws IndexOutOfBoundsException {
79             if (key == null) {
80                 throw new IllegalArgumentException("InstanceKey may not be null");
81             }
82             if (! seen.containsKey(key)) {
83                 throw new IndexOutOfBoundsException("No Intent was seen for key " + key);
84             }
85             return seen.get(key);
86         }
87     
88         public Intent create(final InstanceKey key, final String action) {
89             if (key == null) {
90                 throw new IllegalArgumentException("InstanceKey may not be null");
91             }
92             if (seen.containsKey(key)) {
93                 throw new IndexOutOfBoundsException("There may only be one Intent for " + key);
94             }
95             final Intent intent = new Intent(action);
96             seen.put(key, intent);
97             return intent;
98         }
99     
100         public Intent create(final InstanceKey key, final Atom action) {
101             if (key == null) {
102                 throw new IllegalArgumentException("InstanceKey may not be null");
103             }
104             if (seen.containsKey(key)) {
105                 throw new IndexOutOfBoundsException("There may only be one Intent for " + key);
106             }
107             final Intent intent = new Intent(action);
108             seen.put(key, intent);
109             return intent;
110         }
111     
112         public Intent create(final InstanceKey key) {
113             if (key == null) {
114                 throw new IllegalArgumentException("InstanceKey may not be null");
115             }
116             if (seen.containsKey(key)) {
117                 throw new IndexOutOfBoundsException("There may only be one Intent for " + key);
118             }
119             final Intent intent = new Intent();
120             seen.put(key, intent);
121             return intent;
122         }
123     
124         public Intent findOrCreate(final InstanceKey key) {
125             if (seen.containsKey(key)) {
126                 return find(key);
127             } else {
128                 return create(key);
129             }
130         }
131     
132         public void put(final InstanceKey key, final Intent intent) {
133             seen.put(key, intent);
134         }
135     
136         public boolean contains(final InstanceKey key) {
137             return seen.containsKey(key);
138         }
139     
140         public Intent findOrCreate(final InstanceKey key, String action) {
141             final Intent intent = findOrCreate(key);
142             final Atom foundAction = intent.getAction();
143             if (! foundAction.equals(Atom.findOrCreateAsciiAtom(action))) {
144                 throw new IllegalArgumentException("Actions differ (" + action + ", " +
145                         foundAction.toString() + ") for Intent " + key);
146             }
147             return intent;
148         }
149     
150         public Intent setAction(final InstanceKey key, final String action, boolean isExplicit) {
151             return setAction(key, Atom.findOrCreateAsciiAtom(action), isExplicit);
152         }
153     
154         public Intent unbind(final InstanceKey key) {
155             final Intent intent = find(key);
156             intent.unbind();
157             return intent;
158         }
159     
160         public Intent setExplicit(final InstanceKey key) {
161             if (contains(key)) {
162                 final Intent intent = find(key);
163                 intent.setExplicit();
164                 return intent;
165             } else {
166                 throw new IllegalArgumentException("setAction: No Intent found for key " + key);
167                 //final Intent intent = create(key);
168                 //intent.setExplicit();
169                 //return intent;
170             }
171         }
172     
173         public Intent setAction(final InstanceKey key, final Atom action, boolean isExplicit) {
174             if (contains(key)) {
175                 final Intent intent = find(key);
176                 if (isExplicit) {
177                     intent.setActionExplicit(action);
178                 } else {
179                     intent.setAction(action);
180                 }
181                 return intent;
182             } else {
183                 
184                 final Intent intent = create(key, action);
185                 return intent;
186             }
187         }
188     
189         public Intent setAction(final Intent intent, final String action, boolean isExplicit) {
190             for (final InstanceKey candKey : seen.keySet()) {
191                 if (seen.get(candKey).equals(intent)) {
192                     return setAction(candKey, action, isExplicit);
193                 }
194             }
195     
196             throw new IllegalStateException("The Intent " + intent + " was not registered before!");
197         }
198     
199         public Intent setAction(final InstanceKey key, final InstanceKey actionKey, boolean isExplicit) {
200             if (actionKey == null) {
201                 
202                 return find(key);
203             }
204             final String action;
205             {
206                 if (actionKey instanceof ConstantKey) {
207                     final Object actionO = ((ConstantKey)actionKey).getValue();
208                     if (actionO instanceof String) {
209                         action = StringStuff.deployment2CanonicalTypeString((String) actionO);
210                     } else if (actionO instanceof IClass) {
211                         action = ((IClass) actionO).getName().toString();
212                     } else {
213                         throw new IllegalArgumentException("Wrong action type: " + actionO.getClass());
214                     }
215                 } else {
216                     
217                     unbind(key);
218                     return null;
219                 }
220             }
221     
222             return setAction(key, action, isExplicit);
223         }
