11     package com.ibm.wala.shrikeBT;
12     
13     import com.ibm.wala.shrikeCT.BootstrapMethodsReader.BootstrapMethod;
14     
15     /**
16      * This class provides read-only access to a constant pool. It gets subclassed for each class reader/editor toolkit you want to work
17      * with.
18      */
19     public abstract class ConstantPoolReader {
20       /**
21        * Retrieve the JVM constant pool item type (a Constants.CONSTANT_xxx value). This method should be overriden by a
22        * toolkit-specific subclass.
23        * 
24        * @param index the constant pool item to examine
25        */
26       public abstract int getConstantPoolItemType(int index);
27     
28       /**
29        * Retrieve the value of a CONSTANT_Integer constant pool item. This method should be overriden by a toolkit-specific subclass.
30        * 
31        * @param index the constant pool item to examine
32        */
33       public abstract int getConstantPoolInteger(int index);
34     
35       /**
36        * Retrieve the value of a CONSTANT_Float constant pool item. This method should be overriden by a toolkit-specific subclass.
37        * 
38        * @param index the constant pool item to examine
39        */
40       public abstract float getConstantPoolFloat(int index);
41     
42       /**
43        * Retrieve the value of a CONSTANT_Long constant pool item. This method should be overriden by a toolkit-specific subclass.
44        * 
45        * @param index the constant pool item to examine
46        */
47       public abstract long getConstantPoolLong(int index);
48     
49       /**
50        * Retrieve the value of a CONSTANT_Double constant pool item. This method should be overriden by a toolkit-specific subclass.
51        * 
52        * @param index the constant pool item to examine
53        */
54       public abstract double getConstantPoolDouble(int index);
55     
56       /**
57        * Retrieve the value of a CONSTANT_String constant pool item. This method should be overriden by a toolkit-specific subclass.
58        * 
59        * @param index the constant pool item to examine
60        */
61       public abstract String getConstantPoolString(int index);
62     
63       /**
64        * Retrieve the value of a CONSTANT_MethodType constant pool item. This method should be overriden by a toolkit-specific subclass.
65        * 
66        * @param index the constant pool item to examine
67        */
68       public abstract String getConstantPoolMethodType(int index);
69     
70       /**
71        * Retrieve the value of a CONSTANT_Class constant pool item in JVM internal class format (e.g., java/lang/Object). This method
72        * should be overriden by a toolkit-specific subclass.
73        * 
74        * @param index the constant pool item to examine
75        */
76       public abstract String getConstantPoolClassType(int index);
77     
78       /**
79        * Retrieve the class part of a CONSTANT_FieldRef, CONSTANT_MethodRef, or CONSTANT_InterfaceMethodRef constant pool item, in JVM
80        * internal class format (e.g., java/lang/Object). This method should be overriden by a toolkit-specific subclass.
81        * 
82        * @param index the constant pool item to examine
83        */
84       public abstract String getConstantPoolMemberClassType(int index);
85     
86       /**
87        * Retrieve the name part of a CONSTANT_FieldRef, CONSTANT_MethodRef, or CONSTANT_InterfaceMethodRef constant pool item, This
88        * method should be overriden by a toolkit-specific subclass.
89        * 
90        * @param index the constant pool item to examine
91        */
92       public abstract String getConstantPoolMemberName(int index);
93     
94       /**
95        * Retrieve the type part of a CONSTANT_FieldRef, CONSTANT_MethodRef, or CONSTANT_InterfaceMethodRef constant pool item, in JVM
96        * internal type format (e.g., Ljava/lang/Object;). This method should be overriden by a toolkit-specific subclass.
97        * 
98        * @param index the constant pool item to examine
99        */
100       public abstract String getConstantPoolMemberType(int index);
101       
102       /**
103        * Retrieve the class part of the CONSTANT_FieldRef, CONSTANT_MethodRef, or CONSTANT_InterfaceMethodRef constant pool item pointed to by
104        * a CONSTANT_MethodHandle entry. This method should be overriden by a toolkit-specific subclass.
105        * 
106        * @param index the constant pool item to examine
107        */
108       public abstract String getConstantPoolHandleClassType(int index);
109     
110       /**
111        * Retrieve the name part of the CONSTANT_FieldRef, CONSTANT_MethodRef, or CONSTANT_InterfaceMethodRef constant pool item pointed to by
112        * a CONSTANT_MethodHandle entry. This method should be overriden by a toolkit-specific subclass.
113        * 
114        * @param index the constant pool item to examine
115        */
116       public abstract String getConstantPoolHandleName(int index);
117     
118       /**
119        * Retrieve the type part of the CONSTANT_FieldRef, CONSTANT_MethodRef, or CONSTANT_InterfaceMethodRef constant pool item pointed to by
120        * a CONSTANT_MethodHandle entry. This method should be overriden by a toolkit-specific subclass.
121        * 
122        * @param index the constant pool item to examine
123        */
124       public abstract String getConstantPoolHandleType(int index);
125     
126       public abstract byte getConstantPoolHandleKind(int index);
127     
128       public abstract BootstrapMethod getConstantPoolDynamicBootstrap(int index);
129       
130       public abstract String getConstantPoolDynamicName(int index);
131     
132       public abstract String getConstantPoolDynamicType(int index);
133     
134     }
