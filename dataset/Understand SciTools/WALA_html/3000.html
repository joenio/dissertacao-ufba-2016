89        * @see com.ibm.wala.demandpa.flowgraph.IFlowGraph#addSubgraphForNode(com.ibm.wala.ipa.callgraph.CGNode)

91       @Override
92       public void addSubgraphForNode(CGNode node) throws IllegalArgumentException {
93         if (node == null) {
94           throw new IllegalArgumentException("node == null");
95         }
96         IR ir = node.getIR();
97         if (ir == null) {
98           throw new IllegalArgumentException("no ir for node " + node);
99         }
100         int n = cg.getNumber(node);
101         if (!cgNodesVisited.contains(n)) {
102           cgNodesVisited.add(n);
103           unconditionallyAddConstraintsFromNode(node, ir);
104           addNodesForInvocations(node, ir);
105           addNodesForParameters(node, ir);
106         }
107       }

114         return cgNodesVisited.contains(cg.getNumber(node));

118        * @see com.ibm.wala.demandpa.flowgraph.IFlowGraph#getParamSuccs(com.ibm.wala.ipa.callgraph.propagation.LocalPointerKey)
119        */
120       public Iterator<PointerKeyAndCallSite> getParamSuccs(LocalPointerKey pk) {
121         // TODO cache this result
122         // TODO take some cgnode as parameter if we have calling context?
123         CGNode cgNode = params.get(pk);
124         if (cgNode == null) {
125           return EmptyIterator.instance();
126         }
127         int paramPos = pk.getValueNumber() - 1;
128         ArrayList<PointerKeyAndCallSite> paramSuccs = new ArrayList<PointerKeyAndCallSite>();
129         // iterate over callers
130         for (CGNode caller : cg) {
131           // TODO optimization: we don't need to add the graph if null is passed
132           // as the argument
133           addSubgraphForNode(caller);
134           IR ir = caller.getIR();
135           for (Iterator<CallSiteReference> iterator = ir.iterateCallSites(); iterator.hasNext();) {
136             CallSiteReference call = iterator.next();
137             if (cg.getPossibleTargets(caller, call).contains(cgNode)) {
138               SSAAbstractInvokeInstruction[] callInstrs = ir.getCalls(call);
139               for (int i = 0; i < callInstrs.length; i++) {
140                 SSAAbstractInvokeInstruction callInstr = callInstrs[i];
141                 PointerKey actualPk = heapModel.getPointerKeyForLocal(caller, callInstr.getUse(paramPos));
142                 assert containsNode(actualPk);
143                 assert containsNode(pk);
144                 paramSuccs.add(new PointerKeyAndCallSite(actualPk, call));
145               }
146             }
147           }
148         }
149         return paramSuccs.iterator();
150       }

186        */
187       public Iterator<PointerKeyAndCallSite> getReturnSuccs(LocalPointerKey pk) {
188         SSAAbstractInvokeInstruction callInstr = callDefs.get(pk);
189         if (callInstr == null)
190           return EmptyIterator.instance();
191         ArrayList<PointerKeyAndCallSite> returnSuccs = new ArrayList<PointerKeyAndCallSite>();
192         boolean isExceptional = pk.getValueNumber() == callInstr.getException();
193     
194         CallSiteReference callSiteRef = callInstr.getCallSite();
195         // get call targets
196         Collection<CGNode> possibleCallees = cg.getPossibleTargets(pk.getNode(), callSiteRef);
197         // construct graph for each target
198         for (CGNode callee : possibleCallees) {
199           addSubgraphForNode(callee);
200           PointerKey retVal = isExceptional ? heapModel.getPointerKeyForExceptionalReturnValue(callee) : heapModel
201               .getPointerKeyForReturnValue(callee);
202           assert containsNode(retVal);
203           returnSuccs.add(new PointerKeyAndCallSite(retVal, callSiteRef));
204         }
205     
206         return returnSuccs.iterator();
207       }

212       public Iterator<PointerKeyAndCallSite> getReturnPreds(LocalPointerKey pk) {
213         CGNode cgNode = returns.get(pk);
214         if (cgNode == null) {
215           return EmptyIterator.instance();
216         }
217         boolean isExceptional = pk == heapModel.getPointerKeyForExceptionalReturnValue(cgNode);
218         ArrayList<PointerKeyAndCallSite> returnPreds = new ArrayList<PointerKeyAndCallSite>();
219         // iterate over callers
220         for (CGNode caller : cg) {
221           // TODO we don't need to add the graph if null is passed
222           // as the argument
223           addSubgraphForNode(caller);
224           IR ir = caller.getIR();
225           for (Iterator<CallSiteReference> iterator = ir.iterateCallSites(); iterator.hasNext();) {
226             CallSiteReference call = iterator.next();
227             if (cg.getPossibleTargets(caller, call).contains(cgNode)) {
228               SSAAbstractInvokeInstruction[] callInstrs = ir.getCalls(call);
229               for (int i = 0; i < callInstrs.length; i++) {
230                 SSAAbstractInvokeInstruction callInstr = callInstrs[i];
231                 PointerKey returnPk = heapModel.getPointerKeyForLocal(caller, isExceptional ? callInstr.getException() : callInstr
232                     .getDef());
233                 assert containsNode(returnPk);
234                 assert containsNode(pk);
235                 returnPreds.add(new PointerKeyAndCallSite(returnPk, call));
236               }
237             }
238           }
239         }
240         return returnPreds.iterator();
241       }

245       protected void unconditionallyAddConstraintsFromNode(CGNode node, IR ir) {
246     
247         if (DEBUG) {
248           System.err.println(("Adding constraints for CGNode " + node));
249         }
250     
251         if (SSAPropagationCallGraphBuilder.PERIODIC_WIPE_SOFT_CACHES) {
252           wipeCount++;
253           if (wipeCount >= SSAPropagationCallGraphBuilder.WIPE_SOFT_CACHE_INTERVAL) {
254             wipeCount = 0;
255             ReferenceCleanser.clearSoftCaches();
256           }
257         }
258     
259         debugPrintIR(ir);
260     
261         if (ir == null) {
262           return;
263         }
264     
265         DefUse du = node.getDU();
266         addNodeInstructionConstraints(node, ir, du);
267         addNodePassthruExceptionConstraints(node, ir);
268         addNodeConstantConstraints(node, ir);
269       }

271       /**
272        * Add pointer flow constraints based on instructions in a given node
273        */
274       protected void addNodeInstructionConstraints(CGNode node, IR ir, DefUse du) {
275         FlowStatementVisitor v = makeVisitor(node);
276         ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg = ir.getControlFlowGraph();
277         for (ISSABasicBlock b : cfg) {
278           addBlockInstructionConstraints(node, cfg, b, v);
279         }
280       }

282       /**
283        * Add constraints for a particular basic block.
284        */
285       protected void addBlockInstructionConstraints(CGNode node, ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg,
286           ISSABasicBlock b, FlowStatementVisitor v) {
287         v.setBasicBlock(b);
288     
289         // visit each instruction in the basic block.
290         for (Iterator<SSAInstruction> it = b.iterator(); it.hasNext();) {
291           SSAInstruction s = it.next();
292           if (s != null) {
293             s.visit(v);
294           }
295         }
296     
297         addPhiConstraints(node, cfg, b);
298       }

300       private void addPhiConstraints(CGNode node, ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg, ISSABasicBlock b) {
301     
302         // visit each phi instruction in each successor block
303         for (Iterator<? extends IBasicBlock> iter = cfg.getSuccNodes(b); iter.hasNext();) {
304           ISSABasicBlock sb = (ISSABasicBlock) iter.next();
305           if (sb.isExitBlock()) {
306             // an optimization based on invariant that exit blocks should
307             // have no
308             // phis.
309             continue;
310           }
311           int n = 0;
312           // set n to be whichPred(this, sb);
313           for (Iterator<? extends IBasicBlock> back = cfg.getPredNodes(sb); back.hasNext(); n++) {
314             if (back.next() == b) {
315               break;
316             }
317           }
318           assert n < cfg.getPredNodeCount(sb);
319           for (Iterator<SSAPhiInstruction> phis = sb.iteratePhis(); phis.hasNext();) {
320             // Assertions.UNREACHABLE();
321             SSAPhiInstruction phi = phis.next();
322             if (phi == null) {
323               continue;
324             }
325             PointerKey def = heapModel.getPointerKeyForLocal(node, phi.getDef());
326             if (phi.getUse(n) > 0) {
327               PointerKey use = heapModel.getPointerKeyForLocal(node, phi.getUse(n));
328               addNode(def);
329               addNode(use);
330               addEdge(def, use, AssignLabel.noFilter());
331             }
332             // }
333             // }
334           }
335         }
336       }

355       final Map<CGNode, Set<CallerSiteContext>> callerCache = HashMapFactory.make();
356     
357       @Override
358       public Set<CallerSiteContext> getPotentialCallers(PointerKey formalPk) {
359         CGNode callee = null;
360         if (formalPk instanceof LocalPointerKey) {
361           callee = ((LocalPointerKey) formalPk).getNode();
362         } else if (formalPk instanceof ReturnValueKey) {
363           callee = ((ReturnValueKey) formalPk).getNode();
364         } else {
365           throw new IllegalArgumentException("formalPk must represent a local");
366         }
367         Set<CallerSiteContext> ret = callerCache.get(callee);
368         if (ret == null) {
369           ret = HashSetFactory.make();
370           for (Iterator<? extends CGNode> predNodes = cg.getPredNodes(callee); predNodes.hasNext();) {
371             CGNode caller = predNodes.next();
372             for (Iterator<CallSiteReference> iterator = cg.getPossibleSites(caller, callee); iterator.hasNext();) {
373               CallSiteReference call = iterator.next();
374               ret.add(new CallerSiteContext(caller, call));
375             }
376           }
377           callerCache.put(callee, ret);
378         }
379         return ret;
380       }
