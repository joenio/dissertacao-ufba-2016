48     package org.scandroid.prefixtransfer.modeledAllocations;
49     
50     import org.scandroid.prefixtransfer.InstanceKeySite;
51     import org.scandroid.prefixtransfer.PrefixVariable;
52     
53     public class UriParseString extends InstanceKeySite {
54     
55         final int stringInstanceID;
56         final int instanceID;
57     
58         public UriParseString(int instanceID, int stringInstanceID)
59         {
60             this.stringInstanceID = stringInstanceID;
61             this.instanceID = instanceID;
62         }
63     
64         @Override
65         public PrefixVariable propagate(PrefixVariable input) {
66     //      System.out.println("Propagating at: " + instanceID + " (" + constantValue + ")");
67             PrefixVariable retVal = new PrefixVariable();
68             retVal.copyState(input);
69             String prefix = input.getPrefix(stringInstanceID);
70             retVal.update(instanceID, prefix);
71             if (input.fullPrefixKnown.contains(stringInstanceID))
72                 retVal.include(instanceID);
73             return retVal;
74         }
75     
76         public String toString() {
77             return ("UriParseString(instanceID = " + instanceID + "; stringInstanceID = " + stringInstanceID + ")");
78         }
79     
80         @Override
81         public int instanceID() {
82             return instanceID;
83         }
84     
85     }
