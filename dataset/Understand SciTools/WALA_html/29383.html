50       /**
51        * prints the IR of each function in the script
52        * 
53        * @throws IOException
54        * @throws ClassHierarchyException
55        */
56       public static void printIRsForJS(String filename) throws IOException, ClassHierarchyException {
57         // use Rhino to parse JavaScript
58         JSCallGraphUtil.setTranslatorFactory(new CAstRhinoTranslatorFactory());
59         // build a class hierarchy, for access to code info
60         IClassHierarchy cha = JSCallGraphUtil.makeHierarchyForScripts(filename);
61         printIRsForCHA(cha, new Predicate<String>() {
62     
63           @Override
64           public boolean test(String t) {
65             return t.startsWith("Lprologue.js");
66           }
67         });
68       }
69     
70       protected static void printIRsForCHA(IClassHierarchy cha, Predicate<String> exclude) {
71         // for constructing IRs
72         IRFactory<IMethod> factory = AstIRFactory.makeDefaultFactory();
73         for (IClass klass : cha) {
74           // ignore models of built-in JavaScript methods
75           String name = klass.getName().toString();
76           if (exclude.test(name)) continue;
77           // get the IMethod representing the code
78           IMethod m = klass.getMethod(AstMethodReference.fnSelector);
79           if (m != null) {
80             IR ir = factory.makeIR(m, Everywhere.EVERYWHERE, new SSAOptions());
81             System.out.println(ir);
82             if (m instanceof AstMethod) {
83               AstMethod astMethod = (AstMethod) m;
84               System.out.println(astMethod.getSourcePosition());
85             }
86             System.out.println("===================================================\n");
87           }
88         }
89       }
90     
91       private static void printIRsForHTML(String filename) throws IllegalArgumentException, MalformedURLException, IOException,
92           CancelException, WalaException, Error {
93         // use Rhino to parse JavaScript
94         JSCallGraphUtil.setTranslatorFactory(new CAstRhinoTranslatorFactory());
95         // add model for DOM APIs
96         JavaScriptLoader.addBootstrapFile(WebUtil.preamble);
97         URL url = (new File(filename)).toURI().toURL();
98         Pair<Set<MappedSourceModule>, File> p = WebUtil.extractScriptFromHTML(url, true);
99         SourceModule[] scripts = p.fst.toArray(new SourceModule[] {});
100         JavaScriptLoaderFactory loaders = new WebPageLoaderFactory(JSCallGraphUtil.getTranslatorFactory());
101         CAstAnalysisScope scope = new CAstAnalysisScope(scripts, loaders, Collections.singleton(JavaScriptLoader.JS));
102         IClassHierarchy cha = ClassHierarchy.make(scope, loaders, JavaScriptLoader.JS);
103         Util.checkForFrontEndErrors(cha);
104         printIRsForCHA(cha, new Predicate<String>() {
105     
106           @Override
107           public boolean test(String t) {
108             return t.startsWith("Lprologue.js") || t.startsWith("Lpreamble.js");
109           }
110         });
111       }
112     
113       /**
114        * 
115        * @param args
116        * @throws IOException
117        * @throws WalaException
118        * @throws CancelException
119        * @throws IllegalArgumentException
120        * @throws Error
121        */
122       public static void main(String[] args) throws IOException, IllegalArgumentException, CancelException, WalaException, Error {
123         String filename = args[0];
124         if (filename.endsWith(".js")) {
125           printIRsForJS(filename);
126         } else if (filename.endsWith(".html")) {
127           printIRsForHTML(filename);
128         }
129     
130       }
