19     /**
20      * Representation of a node's position in a CAst entity's syntax tree. The position is stored as a zipper
21      * data structure.
22      * 
23      * @author mschaefer
24      *
25      */
26     public abstract class NodePos implements CAstRewriter.RewriteContext<CAstBasicRewriter.NoKey> {
27     	public abstract <A> A accept(PosSwitch<A> ps);
28     	
29     	@Override
30       public NoKey key() {
31     		return null;
32     	}
33     
34     	/**
35     	 * Determines whether a node is inside the subtree rooted at some other node.
36     	 * 
37     	 * @param node the node
38     	 * @param tree the subtree
39     	 * @return {@literal true} if {@code node} is a descendant of {@code tree}, {@literal false} otherwise
40     	 */
41     	public static boolean inSubtree(CAstNode node, CAstNode tree) {
42     		if(node == tree)
43     			return true;
44     		if(tree == null)
45     			return false;
46     		for(int i=0;i<tree.getChildCount();++i)
47     			if(inSubtree(node, tree.getChild(i)))
48     				return true;
49     		return false;
50     	}
51     }
