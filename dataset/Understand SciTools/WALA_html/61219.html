60     /**
61      * utility methods for mapping various program entities from one call graph to the corresponding entity in another one
62      * 
63      * @author Manu Sridharan
64      * 
65      */
66     public class CallGraphMapUtil {
67     
68       /**
69        * map a call graph node from one call graph to the corresponding node in another. Note that the target call graph must be
70        * context-insensitive for the method, i.e., the only context for the method should be Everywhere.EVERYWHERE.
71        * 
72        * @return the corresponding node, or <code>null</code> if the method is not in the target call graph
73        * @throws IllegalArgumentException if fromCG == null
74        */
75       public static CGNode mapCGNode(CGNode orig, CallGraph fromCG, CallGraph toCG) throws IllegalArgumentException {
76         if (fromCG == null) {
77           throw new IllegalArgumentException("fromCG == null");
78         }
79         if (orig == fromCG.getFakeRootNode()) {
80           return toCG.getFakeRootNode();
81         } else {
82           MethodReference methodRef = orig.getMethod().getReference();
83           if (methodRef.toString().equals("< Primordial, Ljava/lang/Object, clone()Ljava/lang/Object; >")) {
84             // NOTE: clone() is cloned one level, even by RTA, so we need to handle it
85             CGNode ret = toCG.getNode(orig.getMethod(), orig.getContext());
86             if (ret == null) {
87               System.err.println(("WEIRD can't map node " + orig));
88             }
89             return ret;
90           } else {
91             Set<CGNode> nodes = toCG.getNodes(methodRef);
92             int size = nodes.size();
93             assert size <= 1;
94             return (size == 0) ? null : nodes.iterator().next();
95           }
96         }
97       }
98     
99       public static InstanceKey mapInstKey(InstanceKey ik, CallGraph fromCG, CallGraph toCG, HeapModel heapModel)
100           throws UnimplementedError, NullPointerException {
101         InstanceKey ret = null;
102         if (ik instanceof InstanceKeyWithNode) {
103           CGNode oldCGNode = ((InstanceKeyWithNode) ik).getNode();
104           CGNode newCGNode = mapCGNode(oldCGNode, fromCG, toCG);
105           if (newCGNode == null) {
106             return null;
107           }
108           if (ik instanceof AllocationSiteInNode) {
109             if (ik instanceof NormalAllocationInNode) {
110               ret = heapModel.getInstanceKeyForAllocation(newCGNode, ((AllocationSiteInNode) ik).getSite());
111             } else if (ik instanceof MultiNewArrayInNode) {
112               MultiNewArrayInNode mnik = (MultiNewArrayInNode) ik;
113     
114               ret = heapModel.getInstanceKeyForMultiNewArray(newCGNode, mnik.getSite(), mnik.getDim());
115             } else {
116               Assertions.UNREACHABLE();
117             }
118           } else {
119             Assertions.UNREACHABLE();
120           }
121         } else if (ik instanceof ConcreteTypeKey) {
122           return ik;
123         } else {
124           Assertions.UNREACHABLE();
125         }
126         assert ret != null;
127         assert ret.getClass() == ik.getClass();
128         return ret;
129       }
130     
131       public static PointerKey mapPointerKey(PointerKey pk, CallGraph fromCG, CallGraph toCG, HeapModel heapModel)
132           throws UnimplementedError {
133         PointerKey ret = null;
134         if (pk instanceof AbstractLocalPointerKey) {
135           CGNode oldCGNode = ((AbstractLocalPointerKey) pk).getNode();
136           CGNode newCGNode = mapCGNode(oldCGNode, fromCG, toCG);
137           if (newCGNode == null) {
138             return null;
139           }
140           if (pk instanceof LocalPointerKey) {
141             ret = heapModel.getPointerKeyForLocal(newCGNode, ((LocalPointerKey) pk).getValueNumber());
142           } else if (pk instanceof ReturnValueKey) {
143             // NOTE: must check for ExceptionReturnValueKey first,
144             // since its a subclass if ReturnValueKey
145             if (pk instanceof ExceptionReturnValueKey) {
146               ret = heapModel.getPointerKeyForExceptionalReturnValue(newCGNode);
147             } else {
148               ret = heapModel.getPointerKeyForReturnValue(newCGNode);
149             }
150           } else {
151             Assertions.UNREACHABLE();
152           }
153         } else {
154           Assertions.UNREACHABLE();
155         }
156         assert ret != null;
157         return ret;
158       }
