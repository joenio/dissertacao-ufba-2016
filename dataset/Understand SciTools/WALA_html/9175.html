38     package com.ibm.wala.demandpa.alg.refinepolicy;
39     
40     import com.ibm.wala.ipa.cha.IClassHierarchy;
41     
42     /**
43      * A refinement policy specified manually with annotations.
44      * @author Manu Sridharan
45      *
46      */
47     public class ManualRefinementPolicy extends AbstractRefinementPolicy {
48     
49       private ManualRefinementPolicy(IClassHierarchy cha) {
50         // since we have specified what to refine manually, no need to run multiple passes
51         super(new ContainersFieldPolicy(cha), new AlwaysRefineCGPolicy(), 1, new int[] { LONGER_PASS_BUDGET });
52       }
53     
54       public static class Factory implements RefinementPolicyFactory {
55         private final IClassHierarchy cha;
56     
57         public Factory(IClassHierarchy cha) {
58           this.cha = cha;
59         }
60     
61         @Override
62         public RefinementPolicy make() {
63           return new ManualRefinementPolicy(cha);
64         }
65       }
66     }
