11     package com.ibm.wala.cast.js.callgraph.fieldbased.flowgraph.vertices;
12     
13     /**
14      * A lexical access vertex represents a lexical variable, i.e., a local variable
15      * that is accessed from within a nested function. It is identified by the name
16      * of its defining function, and its own name.
17      * 
18      * @author mschaefer
19      *
20      */
21     public class LexicalVarVertex extends Vertex {
22     	// name of the function defining this lexical variable
23     	private final String definer;
24     	
25     	// name of the lexical variable itself
26     	private final String name;
27     	
28     	LexicalVarVertex(String definer, String name) {
29     		this.definer = definer;
30     		this.name = name;
31     	}
32     
33     	@Override
34     	public <T> T accept(VertexVisitor<T> visitor) {
35     		return visitor.visitLexicalAccessVertex(this);
36     	}
37     	
38     	@Override
39     	public String toString() {
40     		return "LexVar(" + definer.substring(definer.lastIndexOf('/')+1) + ", " + name + ")";
41     	}
42     }
