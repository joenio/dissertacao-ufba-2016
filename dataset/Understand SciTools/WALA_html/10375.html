28     public abstract class TestLexicalModRef {
29     
30       @Test
31       public void testSimpleLexical() throws IOException, IllegalArgumentException, CancelException, WalaException {
32         JSCFABuilder b = JSCallGraphBuilderUtil.makeScriptCGBuilder("tests", "simple-lexical.js");
33         CallGraph CG = b.makeCallGraph(b.getOptions());    
34         LexicalModRef lexAccesses = LexicalModRef.make(CG, b.getPointerAnalysis());
35         Map<CGNode, OrdinalSet<Pair<CGNode, String>>> readResult = lexAccesses.computeLexicalRef();
36         Map<CGNode, OrdinalSet<Pair<CGNode, String>>> writeResult = lexAccesses.computeLexicalMod();
37         for (CGNode n : readResult.keySet()) {
38           if (n.toString().contains("Node: <Code body of function Ltests/simple-lexical.js/outer/inner>")) {
39             // function "inner" reads exactly x and z 
40             OrdinalSet<Pair<CGNode, String>> readVars = readResult.get(n);
41             Assert.assertEquals(2, readVars.size());
42             Assert.assertEquals("[[Node: <Code body of function Ltests/simple-lexical.js/outer> Context: Everywhere,x], [Node: <Code body of function Ltests/simple-lexical.js/outer> Context: Everywhere,z]]", readVars.toString());
43             // writes x and z as well
44             OrdinalSet<Pair<CGNode, String>> writtenVars = writeResult.get(n);
45             Assert.assertEquals(2, writtenVars.size());
46             Assert.assertEquals("[[Node: <Code body of function Ltests/simple-lexical.js/outer> Context: Everywhere,x], [Node: <Code body of function Ltests/simple-lexical.js/outer> Context: Everywhere,z]]", writtenVars.toString());
47             
48           }
49           if (n.toString().contains("Node: <Code body of function Ltests/simple-lexical.js/outer/inner2>")) {
50             // function "inner3" reads exactly innerName, inner3, and x and z via callees
51             OrdinalSet<Pair<CGNode, String>> readVars = readResult.get(n);
52             Assert.assertEquals(4, readVars.size());
53             for(Pair<CGNode, String> rv : readVars) {
54               Assert.assertTrue(rv.toString(), 
55                                 "[Node: <Code body of function Ltests/simple-lexical.js/outer> Context: Everywhere,x]".equals(rv.toString()) ||
56                                 "[Node: <Code body of function Ltests/simple-lexical.js/outer> Context: Everywhere,inner3]".equals(rv.toString()) ||
57                                 "[Node: <Code body of function Ltests/simple-lexical.js/outer> Context: Everywhere,innerName]".equals(rv.toString()) ||
58                                 "[Node: <Code body of function Ltests/simple-lexical.js/outer> Context: Everywhere,z]".equals(rv.toString()));
59               
60             }
61           }
62         }
63       }
