129         public final int pc;
130         final protected Opcode opcode;
131         final protected DexIMethod method;
132     
133     
134         public final static int[] noInstructions = new int[0];
135     
136         protected Instruction(int pc, Opcode op, DexIMethod method)
137         {
138             this.pc = pc;
139             this.opcode = op;
140             this.method = method;
141         }
142     
143     
144         /**
145          * True if the instruction can continue.
146          * @see com.ibm.wala.shrikeBT.IInstruction#isFallThrough()
147          */
148         public boolean isFallThrough() {
149             return opcode.canContinue();
150         }
151     
152         /**
153          * True if the instruction can throw an exception
154          * @see com.ibm.wala.shrikeBT.IInstruction#isPEI()
155          */
156         public boolean isPEI() {
157             return opcode.canThrow();
158         }
159     
160         /**
161          * @return The DexIMethod which contains this instruction.
162          */
163         public DexIMethod getParentMethod(){
164             return method;
165         }
166     
167         /**
168          * @return The opcode associated with this instruction.
169          */
170         public Opcode getOpcode(){
171             return opcode;
172         }
173     
174         public int[] getBranchTargets() {
175             return noInstructions;
176         }
177     
178     
179     
180         public abstract void visit(Visitor visitor);
181     
182     }
