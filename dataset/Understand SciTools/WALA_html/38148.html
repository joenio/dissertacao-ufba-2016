34     public class AstJavaTypeInference extends AstTypeInference {
35     
36       protected IClass stringClass;
37     
38       protected class AstJavaTypeOperatorFactory extends AstTypeOperatorFactory implements AstJavaInstructionVisitor {
39         @Override
40         public void visitBinaryOp(SSABinaryOpInstruction instruction) {
41           if (doPrimitives) {
42             IBinaryOpInstruction.IOperator op = instruction.getOperator();
43             if (op == CAstBinaryOp.EQ || op == CAstBinaryOp.NE || op == CAstBinaryOp.LT
44                 || op == CAstBinaryOp.GE || op == CAstBinaryOp.GT || op == CAstBinaryOp.LE) {
45               result = new DeclaredTypeOperator(language.getPrimitive(language.getConstantType(Boolean.TRUE)));
46             } else {
47               result = new PrimAndStringOp();
48             }
49           }
50         }
51     
52         @Override
53         public void visitEnclosingObjectReference(EnclosingObjectReference inst) {
54           TypeReference type = inst.getEnclosingType();
55           IClass klass = cha.lookupClass(type);
56           if (klass == null) {
57             Assertions.UNREACHABLE();
58           } else {
59             result = new DeclaredTypeOperator(new ConeType(klass));
60           }
61         }
62     
63         @Override
64         public void visitJavaInvoke(AstJavaInvokeInstruction instruction) {
65           TypeReference type = instruction.getDeclaredResultType();
66           if (type.isReferenceType()) {
67             IClass klass = cha.lookupClass(type);
68             if (klass == null) {
69               // a type that cannot be loaded.
70               // be pessimistic
71               result = new DeclaredTypeOperator(BOTTOM);
72             } else {
73               result = new DeclaredTypeOperator(new ConeType(klass));
74             }
75           } else {
76             if (doPrimitives && type.isPrimitiveType()) {
77               result = new DeclaredTypeOperator(PrimitiveType.getPrimitive(type));
78             } else {
79               result = null;
80             }
81           }
82         }
83       }

85       public class AstJavaTypeVarFactory extends TypeVarFactory {
86     
87         @Override
88         public IVariable makeVariable(int valueNumber) {
89           SymbolTable st = ir.getSymbolTable();
90           if (st.isStringConstant(valueNumber)) {
91             IClass klass = cha.lookupClass(TypeReference.JavaLangString);
92             TypeAbstraction stringTypeAbs = new PointType(klass);
93             return new TypeVariable(stringTypeAbs);
94           } else {
95             return super.makeVariable(valueNumber);
96           }
97         }
98     
99       }
100     
101       public AstJavaTypeInference(IR ir, IClassHierarchy cha, boolean doPrimitives) {
102         super(ir, cha, JavaPrimitiveType.BOOLEAN, doPrimitives);
103       }
104     
105       IClass getStringClass() {
106         if (stringClass == null) {
107           this.stringClass = cha.lookupClass(TypeReference.JavaLangString);
108         }
109         return stringClass;
110       }
111     
112       @Override
113       protected void initialize() {
114         init(ir, new AstJavaTypeVarFactory(), new AstJavaTypeOperatorFactory());
115       }
116     
117       @Override
118       public TypeAbstraction getConstantPrimitiveType(int valueNumber) {
119         SymbolTable st = ir.getSymbolTable();
120         if (st.isBooleanConstant(valueNumber)) {
121           return language.getPrimitive(language.getConstantType(Boolean.TRUE));
122         } else {
123           return super.getConstantPrimitiveType(valueNumber);
124         }
125       }
126     
127       protected class PrimAndStringOp extends PrimitivePropagateOperator {
128     
129         private PrimAndStringOp() {
130         }
131     
132         @Override
133         public byte evaluate(TypeVariable lhs, TypeVariable[] rhs) {
134           TypeAbstraction meet = null;
135     
136           for (int i = 0; i < rhs.length; i++) {
137             if (rhs[i] != null) {
138               TypeVariable r = rhs[i];
139               TypeAbstraction ta = r.getType();
140               if (ta instanceof PointType) {
141                 if (ta.getType().equals(getStringClass())) {
142                   meet = new PointType(ta.getType());
143                   break;
144                 }
145               } else if (ta instanceof ConeType) {
146                 if (ta.getType().equals(getStringClass())) {
147                   meet = new PointType(ta.getType());
148                   break;
149                 }
150               }
151             }
152           }
153     
154           if (meet == null) {
155             return super.evaluate(lhs, rhs);
156           } else {
157             TypeVariable L = lhs;
158             TypeAbstraction lhsType = L.getType();
159     
160             if (lhsType.equals(meet)) {
161               return NOT_CHANGED;
162             } else {
163               L.setType(meet);
164               return CHANGED;
165             }
166           }
167         }
