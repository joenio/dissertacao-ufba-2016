132     	public static <E extends ISSABasicBlock>
133           TabulationResult<BasicBlockInContext<E>, CGNode, DomainElement> 
134           analyze(final ISupergraph<BasicBlockInContext<E>, 
135         		  CGNode> graph,
136                   CallGraph cg,
137                   PointerAnalysis<InstanceKey> pa,
138                   Map<BasicBlockInContext<E>, Map<FlowType<E>,Set<CodeElement>>> initialTaints,
139                   IFDSTaintDomain<E> d,
140                   IProgressMonitor progressMonitor, 
141                   final IFlowFunctionMap<BasicBlockInContext<E>> flowFunctionMap
142                 ) {
143     
144             final IFDSTaintDomain<E> domain = d;
145     
146             final List<PathEdge<BasicBlockInContext<E>>>
147                initialEdges = new ArrayList<PathEdge<BasicBlockInContext<E>>>();
148     
149             //Add PathEdges to the taints
150             //Places that initial taints occur, and where they initially flow into
151             for(BasicBlockInContext<E> taintBB:initialTaints.keySet())
152             {
153             	Map<FlowType<E>, Set<CodeElement>> bbTaints = initialTaints.get(taintBB);
154             	for(FlowType<E> taintType:bbTaints.keySet())
155                 {
156                     for(CodeElement taintElement:bbTaints.get(taintType))
157                     {
158                     	BasicBlockInContext<E>[] entryBlocks = graph.getEntriesForProcedure(taintBB.getNode());
159                     	for (int i = 0; i < entryBlocks.length; i++) {
160                     		//Add PathEdge <s_p,0> -> <n,d1>
161                     		initialEdges.add(PathEdge.createPathEdge(entryBlocks[i], 0, taintBB, domain.getMappedIndex(new DomainElement(taintElement,taintType))));
162                     	}
163                         //initialEdges.add(PathEdge.createPathEdge(e.getKey(), 0, e.getKey(), domain.getMappedIndex(new DomainElement(o,e2.getKey()))));
164                     }
165                 }
166             }
167             //Add PathEdges to the entry points of the supergraph <s_main,0> -> <s_main,0>
168             for (CGNode entry : cg.getEntrypointNodes()) {
169             	BasicBlockInContext<E>[] bbic = graph.getEntriesForProcedure(entry);
170             	for (int i = 0; i < bbic.length; i++)
171             		initialEdges.add(PathEdge.createPathEdge(bbic[i], 0, bbic[i], 0));
172             }
173             
174             final TabulationProblem<BasicBlockInContext<E>, CGNode, DomainElement>
175               problem =
176                 new TabulationProblem<BasicBlockInContext<E>, CGNode, DomainElement>() {
177     
178                 public TabulationDomain<DomainElement, BasicBlockInContext<E>> getDomain() {
179                     return domain;
180                 }
181     
182                 public IFlowFunctionMap<BasicBlockInContext<E>> getFunctionMap() {
183                     return flowFunctionMap;
184                 }
185     
186                 public IMergeFunction getMergeFunction() {
187                     return null;
188                 }
189     
190                 public ISupergraph<BasicBlockInContext<E>, CGNode> getSupergraph() {
191                     return graph;
192                 }
193     
194                 public Collection<PathEdge<BasicBlockInContext<E>>> initialSeeds() {
195                     return initialEdges;
196     //              CGNode entryProc = cfg.getCallGraph().getEntrypointNodes()
197     //                      .iterator().next();
198     //              BasicBlockInContext<ISSABasicBlock> entryBlock = cfg
199     //                      .getEntry(entryProc);
200     //              for (int i = 0; i < entryProc.getIR().getNumberOfParameters(); i++) {
201     //                  list.add(PathEdge.createPathEdge(entryBlock, 0, entryBlock,
202     //                          domain.getMappedIndex(new LocalElement(i + 1))));
203     //              }
204     //              return list;
205                 }
206     
207             };
208             TabulationSolver<BasicBlockInContext<E>, CGNode, DomainElement> solver =
209                 TabulationSolver.make(problem/**, progressMonitor*/);
210     
211             try {
212             	TabulationResult<BasicBlockInContext<E>,CGNode, DomainElement> flowResult = solver.solve();
213     //        	if (options.ifdsExplorer()) {
214     //        		for (int i = 1; i < domain.getSize(); i++) {        			
215     //                            			
216     //        		}
217     //        		GraphUtil.exploreIFDS(flowResult);
218     //        	}
219                 return flowResult;
220             } catch (CancelException e) {
221                 throw new CancelRuntimeException(e);
222             }
223         }
