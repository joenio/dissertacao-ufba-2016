11     package com.ibm.wala.shrikeBT;
12     
13     /**
14      * An ExceptionHandler represents a single handler covering a single instruction. It simply tells us what kind of exception to catch
15      * and where to dispatch the exception to.
16      * 
17      * ExceptionHandlers are immutable. It is quite legal to save a reference to an exception handler and use it in any other context.
18      * We also treat arrays of ExceptionHandlers as immutable. Therefore the following code can be used to build an exception handler
19      * table that specifies two handlers covering an entire block of code:
20      * 
21      * <pre>
22      * 
23      *   ExceptionHandler[] hs = {
24      *     new ExceptionHandler(110, &quot;Ljava.lang.NullPointerException;&quot;),
25      *     new ExceptionHandler(220, &quot;Ljava.io.IOException;&quot;);
26      *   };
27      *   for (int i = 0; i &lt; 100; i++) {
28      *     handlers[i] = hs;
29      *   }
30      * </pre>
31      */
32     final public class ExceptionHandler {
33     
34       int handler;
35     
36       final String catchClass;
37     
38       /**
39        * @param handler the label for the handler code
40        * @param catchClass the type of exception that should be caught (in JVM format), or null if all exceptions should be caught (as
41        *          with 'finally')
42        */
43       public ExceptionHandler(int handler, String catchClass) {
44         this.handler = handler;
45         this.catchClass = catchClass;
46       }
47     
48       /**
49        * @return the label of the handler code
50        */
51       public int getHandler() {
52         return handler;
53       }
54     
55       /**
56        * @return the type of exceptions to be caught, or null if all exceptions should be caught
57        */
58       public String getCatchClass() {
59         return catchClass;
60       }
61     
62       public boolean equals(ExceptionHandler h) {
63         if (h == null) {
64           throw new IllegalArgumentException("h is null");
65         }
66         return h.handler == handler && (catchClass == null ? h.catchClass == null : catchClass.equals(h.catchClass));
67       }
68     
69       @Override
70       public int hashCode() {
71         return 1069 * handler + ((catchClass == null) ? 0 : catchClass.hashCode());
72       }
73     
74       @Override
75       public boolean equals(Object o) {
76         if (o instanceof ExceptionHandler) {
77           return equals((ExceptionHandler) o);
78         } else {
79           return false;
80         }
81       }
82     }
