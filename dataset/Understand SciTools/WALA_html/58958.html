15     package com.ibm.wala.cast.ir.translator;
16     
17     import com.ibm.wala.cast.tree.CAstControlFlowMap;
18     import com.ibm.wala.cast.tree.CAstNode;
19     import com.ibm.wala.cast.tree.CAstNodeTypeMap;
20     import com.ibm.wala.cast.tree.CAstSourcePositionMap;
21     import com.ibm.wala.cast.tree.CAstSourcePositionMap.Position;
22     import com.ibm.wala.cast.tree.CAstType;
23     import com.ibm.wala.cast.tree.impl.CAstControlFlowRecorder;
24     import com.ibm.wala.cast.tree.impl.CAstNodeTypeMapRecorder;
25     import com.ibm.wala.cast.tree.impl.CAstSourcePositionRecorder;
26     
27     public abstract class AbstractCodeEntity extends AbstractEntity {
28       protected final CAstSourcePositionRecorder src = new CAstSourcePositionRecorder();
29     
30       protected final CAstControlFlowRecorder cfg = new CAstControlFlowRecorder(src);
31     
32       protected final CAstNodeTypeMapRecorder types = new CAstNodeTypeMapRecorder();
33     
34       protected final CAstType type;
35     
36       protected CAstNode Ast;
37     
38       protected AbstractCodeEntity(CAstType type) {
39         this.type = type;
40       }
41     
42       @Override
43       public CAstNode getAST() {
44         return Ast;
45       }
46     
47       @Override
48       public CAstType getType() {
49         return type;
50       }
51     
52       @Override
53       public CAstControlFlowMap getControlFlow() {
54         return cfg;
55       }
56     
57       @Override
58       public CAstSourcePositionMap getSourceMap() {
59         return src;
60       }
61     
62       @Override
63       public CAstNodeTypeMap getNodeTypeMap() {
64         return types;
65       }
66     
67       public void setGotoTarget(CAstNode from, CAstNode to) {
68         setLabelledGotoTarget(from, to, null);
69       }
70     
71       public void setLabelledGotoTarget(CAstNode from, CAstNode to, Object label) {
72         if (!cfg.isMapped(from)) {
73           cfg.map(from, from);
74         }
75         if (!cfg.isMapped(to)) {
76           cfg.map(to, to);
77         }
78         cfg.add(from, to, label);
79       }
80     
81       public void setNodePosition(CAstNode n, Position pos) {
82         src.setPosition(n, pos);
83       }
84     
85       public void setNodeType(CAstNode n, CAstType type) {
86         types.add(n, type);
87       }
88       
89       public void setAst(CAstNode Ast) {
90         this.Ast = Ast;
91       }
92     }
