49     package com.ibm.wala.dalvik.dex.instructions;
50     
51     import org.jf.dexlib.Code.Opcode;
52     
53     import com.ibm.wala.dalvik.classLoader.DexIMethod;
54     import com.ibm.wala.dalvik.dex.instructions.ArrayGet.Type;
55     import com.ibm.wala.types.TypeReference;
56     
57     public class ArrayPut extends Instruction {
58     
59         public final int array;
60         public final int source;
61         public final int offset;
62         public final Type type;
63     
64         public ArrayPut(int pc, int source, int array, int offset, Type type, Opcode op, DexIMethod method) {
65             super(pc, op, method);
66             this.source = source;
67             this.array = array;
68             this.offset = offset;
69             this.type = type;
70         }
71     
72         @Override
73         public void visit(Visitor visitor) {
74             visitor.visitArrayPut(this);
75         }
76     
77         public TypeReference getType() {
78             return ArrayGet.getType(type);
79         }
80     
81     }
