31        * @return the Set corresponding to key in M; create one if needed

36       public static <K, T> Set<T> findOrCreateSet(Map<K, Set<T>> M, K key) {
37         if (M == null) {
38           throw new IllegalArgumentException("M is null");
39         }
40         Set<T> result = M.get(key);
41         if (result == null) {
42           result = HashSetFactory.make(2);
43           M.put(key, result);
44         }
45         return result;
46       }

155       /**
156        * @param m a map from key -> Set <value>
157        * @return inverted map, value -> Set <key>
158        * @throws IllegalArgumentException if m is null
159        */
160       public static <K, V> Map<V, Set<K>> inverseMap(Map<K, Set<V>> m) {
161         if (m == null) {
162           throw new IllegalArgumentException("m is null");
163         }
164         Map<V, Set<K>> result = HashMapFactory.make(m.size());
165         for (Iterator<Map.Entry<K, Set<V>>> it = m.entrySet().iterator(); it.hasNext();) {
166           Map.Entry<K, Set<V>> E = it.next();
167           K key = E.getKey();
168           Set<V> values = E.getValue();
169           for (Iterator<V> it2 = values.iterator(); it2.hasNext();) {
170             V v = it2.next();
171             Set<K> s = findOrCreateSet(result, v);
172             s.add(key);
173           }
174         }
175         return result;
176       }

178       /**
179        * invert an input map that is one-to-one (i.e., it does not map two different keys to the same value)
180        * 
181        * @throws IllegalArgumentException if m is null
182        * @throws IllegalArgumentException if m is not one-to-one
183        */
184       public static <K, V> Map<V, K> invertOneToOneMap(Map<K, V> m) {
185         if (m == null) {
186           throw new IllegalArgumentException("m is null");
187         }
188         Map<V, K> result = HashMapFactory.make(m.size());
189         for (Map.Entry<K, V> entry : m.entrySet()) {
190           K key = entry.getKey();
191           V val = entry.getValue();
192           if (result.containsKey(val)) {
193             throw new IllegalArgumentException("input map not one-to-one");
194           }
195           result.put(val, key);
196         }
197         return result;
198       }

200       public static <K, V> Map<Set<K>, V> groupKeysByValue(Map<K, V> m) {
201         if (m == null) {
202           throw new IllegalArgumentException("m is null");
203         }
204         Map<Set<K>, V> result = HashMapFactory.make();
205         Map<V, Set<K>> valueToKeys = HashMapFactory.make();
206         for (Iterator<Map.Entry<K, V>> it = m.entrySet().iterator(); it.hasNext();) {
207           Map.Entry<K, V> E = it.next();
208           K key = E.getKey();
209           V value = E.getValue();
210           findOrCreateSet(valueToKeys, value).add(key);
211         }
212         for (Iterator<Map.Entry<V, Set<K>>> it = valueToKeys.entrySet().iterator(); it.hasNext();) {
213           Map.Entry<V, Set<K>> E = it.next();
214           V value = E.getKey();
215           Set<K> keys = E.getValue();
216           result.put(keys, value);
217         }
218         return result;
219       }
