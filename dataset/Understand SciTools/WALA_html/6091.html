93         public WDexClassLoaderImpl(ClassLoaderReference loader,IClassLoader parent,
94                 SetOfClasses exclusions, IClassHierarchy cha) {
95             super(loader, cha.getScope().getArrayClassLoader(), parent, exclusions, cha);
96             lParent = parent;
97             this.exclusions = exclusions;
98             //DEBUG_LEVEL = 0;
99         }

101         @Override
102         public void init(List<Module> modules) throws IOException {
103         	super.init(modules);
104             // module are loaded according to the given order (same as in Java VM)
105             Set<ModuleEntry> classModuleEntries = HashSetFactory.make();
106             
107             for (Iterator<Module> it = modules.iterator(); it.hasNext();) {
108                 Module archive = it.next();
109                 Set<ModuleEntry> classFiles = getDexFiles(archive);
110                 
111                 removeClassFiles(classFiles, classModuleEntries);
112                 loadAllDexClasses(classFiles);
113                 
114                 for (Iterator<ModuleEntry> it2 = classFiles.iterator(); it2.hasNext();) {
115                 	ModuleEntry file = it2.next();
116                 	classModuleEntries.add(file);
117                 }
118             }                       
119         }

122         /**
123          * Remove from s any class file module entries which already are in t
124          */
125         private void removeClassFiles(Set<ModuleEntry> s, Set<ModuleEntry> t) {
126         	Set<String> old = HashSetFactory.make();
127         	for (Iterator<ModuleEntry> it = t.iterator(); it.hasNext();) {
128         		ModuleEntry m = it.next();
129         		old.add(m.getClassName());
130         	}
131         	HashSet<ModuleEntry> toRemove = HashSetFactory.make();
132         	for (Iterator<ModuleEntry> it = s.iterator(); it.hasNext();) {
133         		ModuleEntry m = it.next();
134         		if (old.contains(m.getClassName())) {
135         			toRemove.add(m);
136         		}
137         	}
138         	s.removeAll(toRemove);
139         }
140         
141         private Set<ModuleEntry> getDexFiles(Module M) throws IOException {
142         	TreeSet<ModuleEntry> sortedEntries = new TreeSet<ModuleEntry>(HashCodeComparator.instance());
143         	sortedEntries.addAll(Iterator2Collection.toSet(M.getEntries()));
144         	HashSet<ModuleEntry> result = HashSetFactory.make();
145         	for (Iterator<ModuleEntry> it = sortedEntries.iterator(); it.hasNext();) {
146         		ModuleEntry entry = (ModuleEntry) it.next();
147         		if (entry instanceof DexModuleEntry) {    		
148         			result.add(entry);
149         		} 
150         	}
151         	return result;
152         }

154         
155         private void loadAllDexClasses(Collection<ModuleEntry> moduleEntries) {
156         	
157         	for (Iterator<ModuleEntry> it = moduleEntries.iterator(); it.hasNext();) {
158         		ModuleEntry entry = it.next();
159     
160         		// Dalvik class
161         		if (entry instanceof DexModuleEntry) {
162         			DexModuleEntry dexEntry = ((DexModuleEntry) entry);
163         			String className = dexEntry.getClassName();
164     				TypeName tName = TypeName.string2TypeName(className);
165     
166         			//if (DEBUG_LEVEL > 0) {
167         			//  System.err.println("Consider dex class: " + tName);
168         			//}
169     
170         			//System.out.println("Typename: " + tName.toString());
171         			//System.out.println(tName.getClassName());
172         			if (loadedClasses.get(tName) != null) {
173         				Warnings.add(MultipleDexImplementationsWarning
174         						.create(className));
175         			} else if (lParent != null && lParent.lookupClass(tName) != null) {
176         				Warnings.add(MultipleDexImplementationsWarning
177         						.create(className));
178         			}
179         			//if the class is empty, ie an interface
180         			//                  else if (dexEntry.getClassDefItem().getClassData() == null) {
181         			//                      System.out.println("Jumping over (classdata null): "+dexEntry.getClassName());
182         			//                      Warnings.add(MultipleDexImplementationsWarning
183         			//                              .create(dexEntry.getClassName()));
184         			//                  }
185         			else {
186         				IClass iClass = new DexIClass(this, cha, dexEntry);
187         				if (iClass.getReference().getName().equals(tName)) {
188         					
189         					// className is a descriptor, so strip the 'L'
190         					if (exclusions != null && exclusions.contains(className.substring(1))) {
191         						if (DEBUG_LEVEL > 0) {
192         							System.err.println("Excluding " + className);
193         						}
194         						continue;
195         					}
196      
197         					loadedClasses.put(tName, iClass);
198         				} else {
199         					Warnings.add(InvalidDexFile.create(className));
200         				}
201         			}
202         		}
203         	}
204         }

214       /**
215        * A warning when we find more than one implementation of a given class name
216        */
217       private static class MultipleDexImplementationsWarning extends Warning {
218     
219         final String className;
220     
221         MultipleDexImplementationsWarning(String className) {
222           super(Warning.SEVERE);
223           this.className = className;
224         }
225     
226         @Override
227         public String getMsg() {
228           return getClass().toString() + " : " + className;
229         }
230     
231         public static MultipleDexImplementationsWarning create(String className) {
232           return new MultipleDexImplementationsWarning(className);
233         }
234       }
