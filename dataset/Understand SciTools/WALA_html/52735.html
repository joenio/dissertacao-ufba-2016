29     /**
30      * A test of support for source file mapping
31      */
32     public class SourceMapTest extends WalaTestCase {
33       private static final ClassLoader MY_CLASSLOADER = SourceMapTest.class.getClassLoader();
34     
35       private final static String CLASS_IN_PRIMORDIAL_JAR = "Lcom/ibm/wala/model/SyntheticFactory";
36     
37       @Test public void testHello() throws ClassHierarchyException, IOException {
38         if (analyzingJar()) return;
39         AnalysisScope scope = null;
40         scope = AnalysisScopeReader.readJavaScope(TestConstants.HELLO, null, MY_CLASSLOADER);
41         // TODO: it's annoying to have to build a class hierarchy here.
42         // see feature 38676
43         ClassHierarchy cha = ClassHierarchy.make(scope);
44         TypeReference t = TypeReference.findOrCreate(scope.getApplicationLoader(), TestConstants.HELLO_MAIN);
45         IClass klass = cha.lookupClass(t);
46         Assert.assertTrue("failed to load " + t, klass != null);
47         String sourceFile = klass.getSourceFileName();
48         System.err.println("Source file: " + sourceFile);
49         Assert.assertTrue(sourceFile != null);
50       }
51     
52       @Test public void testFromJar() throws ClassHierarchyException, IOException {
53         if (analyzingJar()) return;
54         AnalysisScope scope = null;
55         scope = AnalysisScopeReader.readJavaScope(TestConstants.HELLO, null, MY_CLASSLOADER);
56         // TODO: it's annoying to have to build a class hierarchy here.
57         // open a feature to fix this
58         ClassHierarchy cha = ClassHierarchy.make(scope);
59         TypeReference t = TypeReference.findOrCreate(scope.getPrimordialLoader(), CLASS_IN_PRIMORDIAL_JAR);
60         IClass klass = cha.lookupClass(t);
61         Assert.assertTrue(klass != null);
62         String sourceFile = klass.getSourceFileName();
63         Assert.assertTrue(sourceFile != null);
64         System.err.println("Source file: " + sourceFile);
65         Module container = ((BytecodeClass)klass).getContainer();
66         Assert.assertTrue(container != null);
67         System.err.println("container: " + container);
68       }
69     }
