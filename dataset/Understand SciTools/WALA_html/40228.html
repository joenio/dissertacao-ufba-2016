11     package com.ibm.wala.cast.ir.ssa;
12     
13     import com.ibm.wala.ssa.SSAInstruction;
14     import com.ibm.wala.ssa.SSAInstructionFactory;
15     import com.ibm.wala.ssa.SSAUnaryOpInstruction;
16     import com.ibm.wala.ssa.SymbolTable;
17     
18     /**
19      * A simple assignment statement. Only appears in the IR before SSA conversion, and temporarily when needed to undo copy propagation
20      * during processing of new lexical definitions and uses.
21      * 
22      * @author Julian Dolby (dolby@us.ibm.com)
23      * 
24      */
25     public class AssignInstruction extends SSAUnaryOpInstruction {
26     
27       /**
28        * create the assignment v_result := v_val
29        * 
30        * @param result
31        * @param val
32        */
33       public AssignInstruction(int iindex, int result, int val) {
34         super(iindex, null, result, val);
35         assert result != val;
36         assert result != -1;
37         assert val != -1;
38       }
39     
40       /*
41        * @see com.ibm.wala.ssa.SSAInstruction#copyForSSA(int[], int[])
42        */
43       @Override
44       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
45         return ((AstInstructionFactory) insts)
46             .AssignInstruction(iindex, defs == null ? getDef(0) : defs[0], uses == null ? getUse(0) : uses[0]);
47       }
48     
49       /*
50        * @see com.ibm.wala.ssa.SSAInstruction#toString(com.ibm.wala.ssa.SymbolTable, com.ibm.wala.ssa.ValueDecorator)
51        */
52       @Override
53       public String toString(SymbolTable symbolTable) {
54         return getValueString(symbolTable, result) + " := " + getValueString(symbolTable, val);
55       }
56     
57       /*
58        * @see com.ibm.wala.ssa.SSAInstruction#visit(com.ibm.wala.ssa.SSAInstruction.Visitor)
59        */
60       @Override
61       public void visit(IVisitor v) {
62         ((AstPreInstructionVisitor) v).visitAssign(this);
63       }
64     
65       /**
66        * @return
67        */
68       public int getVal() {
69         return getUse(0);
70       }
71     }
