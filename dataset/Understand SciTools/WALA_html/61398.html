63     /**
64      *  Access parameters without confusion on their numbers.
65      *
66      *  Depending on the representation of a method (IMethod, MethodReference) parameters are placed at a
67      *  different position. Functions furthermore may have an implicit this-pointer which alters the 
68      *  positions again.
69      *
70      *  Accessing parameters of these functions by their numbers only is error prone and leads to confusion.
71      *  This class tries to leverage parameter-access.
72      *
73      *  You can use this class using now numbers at all. However if you choose to use numbers this class has yet 
74      *  another numbering convention (jupeee): 1 is the first parameter appearing in the Selector, no matter if 
75      *  the Method has an implicit this. It is not zero as Java initializes new integer-arrays with zero.
76      *
77      *  If you want to alter the values of the incoming parameters you may also want to use the ParameterManager
78      *  which tracks the changes.
79      *
80      *  @author Tobias Blaschke <code@tobiasblaschke.de>
81      *  @since  2013-10-19
82      */

156             /**
157              *  Create Parameters using ParameterAccessor.
158              *
159              *  @param  number  SSA-Value to access this parameter
160              *  @param  name    Optional variable-name - may be null
161              *  @param  type    Variable Type to this parameter
162              *  @param  disp    Implicit this, regular parameter or return value?
163              *  @param  basedOn Is Accessor constructed with IMethod or MethodReference
164              *  @param  mRef    Method this parameter belongs to
165              *  @param  descriptorOffset add to number to get position in descriptor
166              */
167             protected Parameter(final int number, final String name, final TypeReference type, final ParamerterDisposition disp, 
168                     final BasedOn basedOn, final MethodReference mRef, final int descriptorOffset) {
169                 super(number, type, mRef, new ParameterKey(type.getName(), number + descriptorOffset, name));
170     
171                 if (mRef == null) {
172                     throw new IllegalArgumentException("MethodReference (mRef) of a Parameter may not be null");
173                 }
174                 if (basedOn == null) {
175                     throw new IllegalArgumentException("Argument basedOn of a Parameter may not be null");
176                 }
177                 if (disp == null) {
178                     throw new IllegalArgumentException("ParamerterDisposition (disp) of a Parameter may not be null");
179                 }
180                 // If type was null the call to super failed
181                 if ((number < 1) && (basedOn == BasedOn.METHOD_REFERENCE)) {
182                     throw new IllegalArgumentException("The first accessible SSA-Value of a MethodReference is 1 but the " +
183                             "Value-Number given is " + number);
184                 }
185                 if ((number < 0) && (basedOn == BasedOn.IMETHOD)) {
186                     throw new IllegalArgumentException("The first accessible SSA-Value of an IMethod is 0 but the " +
187                             "Value-Number given is " + number);
188                 }
189                 if ((disp == ParamerterDisposition.PARAM) && (number + descriptorOffset > mRef.getNumberOfParameters())) {
190                     throw new IllegalArgumentException("The SSA-Value " + number + " (with added offset " + descriptorOffset + 
191                             ") is beyond the number of Arguments (" + mRef.getNumberOfParameters() + ") of the Method " +
192                             mRef.getName() + "\n" + mRef.getSignature());
193                 }
194                 if ((disp == ParamerterDisposition.THIS) && (basedOn == BasedOn.METHOD_REFERENCE) && (number != 1)) {
195                     throw new IllegalArgumentException("The implicit this-pointer of a MethodReference is located at SSA-Value 1. " +
196                             "The SSA-Value given is " + number);
197                 }
198                 if ((disp == ParamerterDisposition.THIS) && (basedOn == BasedOn.IMETHOD) && (number != 0)) {
199                     throw new IllegalArgumentException("The implicit this-pointer of an IMethod is located at SSA-Value 0. " +
200                             "The SSA-Value given is " + number);
201                 }
202                 if ((descriptorOffset < -2) || (descriptorOffset > 1)) {
203                     throw new IllegalArgumentException("The descriptor-offset given is not within its expected bounds: " +
204                             "-1 (for a method without implicit this-pointer) to 1. The given offset is " + descriptorOffset);
205                 }
206     
207                 this.disp = disp;
208                 this.basedOn = basedOn;
209                 this.descriptorOffset = descriptorOffset;
210                 super.isAssigned();
211             }

290         /**
291          *  Reads the parameters of a MethodReference CAUTION:.
292          *
293          *  Do _not_ use ParameterAceesor(IMethod.getReference()), but ParameterAceesor(IMehod)!
294          *
295          *  Using this Constructor influences the SSA-Values returned later. The cha is needed to
296          *  determine whether mRef is static. If this is already known one should prefer the faster
297          *  {@link #ParameterAccessor(ParameterAccessor, boolean)}.
298          *
299          *  @param  mRef    The method to read the parameters from.
300          */
301         public ParameterAccessor(final MethodReference mRef, final IClassHierarchy cha) {
302             if (mRef == null) {
303                 throw new IllegalArgumentException("Can't read the arguments from null.");
304             }
305     
306             this.mRef = mRef;
307             this.method = null;
308             this.base = BasedOn.METHOD_REFERENCE;
309             this.numberOfParameters = mRef.getNumberOfParameters();
310     
311             final boolean hasImplicitThis;
312             Set<IMethod> targets = cha.getPossibleTargets(mRef);
313             if (targets.size() < 1) {
314                 warn("Unable to look up the method {} starting extensive search...", mRef);
315     
316                 targets = new HashSet<IMethod>();
317                 final TypeReference mClass = mRef.getDeclaringClass();
318                 final Selector mSel = mRef.getSelector();
319                 final Set<IClass> testClasses = new HashSet<IClass>();
320     
321                 // Look up all classes matching exactly
322                 for (IClassLoader loader : cha.getLoaders()) {
323                     final IClass cand = loader.lookupClass(mClass.getName());
324                     
325                     if (cand != null) {
326                         testClasses.add(cand);
327                     }
328                 }
329     
330                 // Try lookupClass..
331                 final IClass lookedUp;
332                 lookedUp = cha.lookupClass(mClass);
333                 if (lookedUp != null) {
334                     debug("Found using cha.lookupClass()");
335                     testClasses.add(lookedUp);
336                 }
337     
338                 info("Searching the classes {} for the method", testClasses);
339     
340                 for (IClass testClass : testClasses) {
341                     final IMethod cand = testClass.getMethod(mSel);
342     
343                     if (cand != null) {
344                         targets.add(cand);
345                     }
346                 }
347     
348                 if (targets.size() < 1) {
349                     warn("Still no candidates for the method - continuing with super-classes (TODO)");
350     
351                     // TODO
352                    
353                     { // DEBUG
354                         for (IClass testClass : testClasses) {
355                             info("Known Methods in " + testClass);
356                             for (IMethod contained : testClass.getAllMethods()) {
357                                 System.out.println(contained);
358                                 info("\t" + contained);
359                             }
360                         }
361                     } // */
362                     
363                     throw new IllegalStateException("Unable to look up the method " + mRef);
364                 }
365             }
366     
367             { // Iterate all candidates
368                 final Iterator<IMethod> it = targets.iterator();
369                 final boolean testStatic = it.next().isStatic();
370                 while (it.hasNext()) {
371                     final boolean tmpStatic = it.next().isStatic();
372                     if (testStatic != tmpStatic) {
373                         throw new IllegalStateException("The ClassHierarchy knows multiple (" + targets.size() + ") targets for " + mRef +
374                                 ". The targets contradict themselves if they have an implicit this!");
375                     }
376                 }
377                 hasImplicitThis = (! testStatic);
378             }
379     
380             if (hasImplicitThis) {
381                 info("The method {} has an implicit this pointer", mRef);
382                 this.implicitThis = 1;
383                 this.descriptorOffset = -1;  
384             } else {
385                 info("The method {} has no implicit this pointer", mRef);
386                 this.implicitThis = -1;
387                 this.descriptorOffset = 0;
388             }
389         }

446                 this.descriptorOffset = -1;

573          *
574          *  This obviously only works on non-static methods. You probably want to check if the method has
575          *  such an implicit this using hasImplicitThis() as this method will throw if there is none.

583             final int self = getThisNo();

585             switch (this.base) {

588                     break;
589                 case METHOD_REFERENCE:
590                     selfType = this.mRef.getDeclaringClass();
591                     break;
592                 default:
593                     throw new UnsupportedOperationException("No implementation of getThis() for base " + this.base);
594             }
595             return getThisAs(selfType);

1445         public static boolean isAssignable(final TypeReference from, final TypeReference to, final IClassHierarchy cha) {
1446             if (cha == null) {
1447                 throw new IllegalArgumentException("ClassHierarchy may not be null");
1448             }
1449     
1450             if (from.getName().equals(to.getName())) return true;
