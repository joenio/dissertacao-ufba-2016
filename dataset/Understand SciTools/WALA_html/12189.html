14     import java.io.FileWriter;
15     import java.io.Writer;
16     
17     import com.ibm.wala.shrikeBT.Constants;
18     import com.ibm.wala.shrikeBT.IInstruction;
19     import com.ibm.wala.shrikeBT.InvokeInstruction;
20     import com.ibm.wala.shrikeBT.MethodData;
21     import com.ibm.wala.shrikeBT.Util;
22     import com.ibm.wala.shrikeBT.shrikeCT.ClassInstrumenter;
23     import com.ibm.wala.shrikeBT.shrikeCT.OfflineInstrumenter;
24     import com.ibm.wala.shrikeCT.ClassReader;
25     
26     /**
27      * This is a demo class.
28      * 
29      * Class files are taken as input arguments (or if there are none, from standard input). The methods in those files are
30      * instrumented: we insert a System.err.println() at ever method call, and a System.err.println() at every method entry.
31      * 
32      * In Unix, I run it like this: java -cp ~/dev/shrike/shrike com.ibm.wala.shrikeBT.shrikeCT.tools.Bench test.jar -o output.jar
33      * 
34      * The instrumented classes are placed in the directory "output" under the current directory. Disassembled code is written to the
35      * file "report" under the current directory.
36      */
37     public class Statistics {
38       private static OfflineInstrumenter instrumenter;
39     
40       public static void main(String[] args) throws Exception {
41         for (int i = 0; i < 1; i++) {
42           instrumenter = new OfflineInstrumenter(true);
43     
44           Writer w = new BufferedWriter(new FileWriter("report", false));
45     
46           args = instrumenter.parseStandardArgs(args);
47     
48           instrumenter.beginTraversal();
49           ClassInstrumenter ci;
50           while ((ci = instrumenter.nextClass()) != null) {
51             doClass(ci, w);
52           }
53           instrumenter.close();
54           w.close();
55         }
56       }
57     
58       private static void doClass(final ClassInstrumenter ci, Writer w) throws Exception {
59         ClassReader cr = ci.getReader();
60         final String className = cr.getName();
61         w.write("Class: " + className + "\n");
62     
63         boolean allPrivateConstructors = true;
64         boolean methodCallsConstructor = false;
65         boolean classInitCallsConstructor = false;
66     
67         for (int m = 0; m < cr.getMethodCount(); m++) {
68           MethodData d = ci.visitMethod(m);
69     
70           // d could be null, e.g., if the method is abstract or native
71           if (d != null) {
72             if (d.getName().equals("<init>")) {
73               int f = cr.getMethodAccessFlags(m);
74               if ((f & Constants.ACC_PRIVATE) == 0
75                   && ((f & Constants.ACC_PROTECTED) == 0 || (cr.getAccessFlags() & Constants.ACC_FINAL) == 0)) {
76                 allPrivateConstructors = false;
77               }
78             }
79     
80             int constructorCalls = 0;
81             IInstruction[] instrs = d.getInstructions();
82             for (int i = 0; i < instrs.length; i++) {
83               if (instrs[i] instanceof InvokeInstruction) {
84                 InvokeInstruction invoke = (InvokeInstruction) instrs[i];
85                 if (invoke.getMethodName().equals("<init>") && invoke.getClassType().equals(Util.makeType(className))) {
86                   constructorCalls++;
87                 }
88               }
89             }
90             if (!d.getName().equals("<init>") && !d.getName().equals("<clinit>")) {
91               if (constructorCalls > 0) {
92                 methodCallsConstructor = true;
93               }
94             } else if (d.getName().equals("<clinit>")) {
95               classInitCallsConstructor = true;
96             }
97           }
98         }
99     
100         if (allPrivateConstructors && !methodCallsConstructor && classInitCallsConstructor) {
101           w.write("Restricted Creation\n");
102         }
103       }
104     }
