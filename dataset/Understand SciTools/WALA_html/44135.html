19     import com.ibm.wala.ipa.cha.IClassHierarchy;
20     import com.ibm.wala.ssa.IR;
21     import com.ibm.wala.ssa.SSAInstruction;
22     import com.ibm.wala.types.MethodReference;
23     import com.ibm.wala.util.collections.Iterator2Iterable;
24     
25     final class TypeInferenceAssertion implements IRAssertion {
26       private String typeName;
27     
28       public TypeInferenceAssertion(String packageName) {
29         this.typeName = packageName;
30       }
31     
32       // For now just check things in the main method
33       @Override
34       public void check(CallGraph cg) {
35         IR ir = getIR(cg, typeName, "main", "[Ljava/lang/String;", "V");
36         AstJavaTypeInference inference = new AstJavaTypeInference(ir, cg.getClassHierarchy(), true);
37     
38         for (SSAInstruction instr : Iterator2Iterable.make(ir.iterateAllInstructions())) {
39           // Check defs
40           for (int def = 0; def < instr.getNumberOfDefs(); def++) {
41             int ssaVariable = instr.getDef(def);
42             inference.getType(ssaVariable);
43           }
44     
45           // Check uses
46           for (int def = 0; def < instr.getNumberOfUses(); def++) {
47             int ssaVariable = instr.getUse(def);
48             inference.getType(ssaVariable);
49           }
50         }
51     
52       }
53     
54       private IR getIR(CallGraph cg, String fullyQualifiedTypeName, String methodName, String methodParameter, String methodReturnType) {
55         IClassHierarchy classHierarchy = cg.getClassHierarchy();
56         MethodReference methodRef = IRTests
57             .descriptorToMethodRef(
58                 String.format("Source#%s#%s#(%s)%s", fullyQualifiedTypeName, methodName, methodParameter, methodReturnType),
59                 classHierarchy);
60         IMethod method = classHierarchy.resolveMethod(methodRef);
61         CGNode node = cg.getNode(method, Everywhere.EVERYWHERE);
62         return node.getIR();
63       }
64     
65     }
