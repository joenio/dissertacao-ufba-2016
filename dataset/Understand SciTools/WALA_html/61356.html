11     package com.ibm.wala.ssa;
12     
13     import com.ibm.wala.types.TypeReference;
14     
15     /**
16      * A checkcast (dynamic type test) instruction. This instruction produces a new value number (like an assignment) if the check
17      * succeeds.
18      * 
19      * Note that this instruction generalizes the meaning of checkcast in Java since it supports
20      * multiple types for which to check.  The meaning is that the case succeeds if the object
21      * is of any of the desired types.
22      *  
23      */
24     public abstract class SSACheckCastInstruction extends SSAInstruction {
25     
26       /**
27        * A new value number def'fed by this instruction when the type check succeeds.
28        */
29       private final int result;
30     
31       /**
32        * The value being checked by this instruction
33        */
34       private final int val;
35     
36       /**
37        * The types for which this instruction checks; the assignment succeeds if the val is a subtype of one of these types
38        */
39       private final TypeReference[] declaredResultTypes;
40     
41       /**
42        * whether the type test throws an exception
43        */
44       private final boolean isPEI;
45       
46       /**
47        * @param result A new value number def'fed by this instruction when the type check succeeds.
48        * @param val The value being checked by this instruction
49        * @param type The type which this instruction checks
50        */
51       protected SSACheckCastInstruction(int iindex, int result, int val, TypeReference[] types, boolean isPEI) {
52         super(iindex);
53         this.result = result;
54         this.val = val;
55         this.declaredResultTypes = types;
56         this.isPEI = isPEI;
57       }
58     
59       @Override
60       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
61         if (defs != null && defs.length == 0) {
62           throw new IllegalArgumentException("(defs != null) and (defs.length == 0)");
63         }
64         if (uses != null && uses.length == 0) {
65           throw new IllegalArgumentException("(uses != null) and (uses.length == 0)");
66         }
67         return insts.CheckCastInstruction(iindex, defs == null ? result : defs[0], uses == null ? val : uses[0], declaredResultTypes, isPEI);
68       }
69     
70       @Override
71       public String toString(SymbolTable symbolTable) {
72         String v = getValueString(symbolTable, result) + " = checkcast";
73         for (TypeReference t : declaredResultTypes) {
74             v = v + " " + t;
75         }
76         return v + getValueString(symbolTable, val);
77       }
78     
79       /*
80        * @see com.ibm.wala.ssa.SSAInstruction#visit(IVisitor)
81        * 
82        * @throws IllegalArgumentException if v is null
83        */
84       @Override
85       public void visit(IVisitor v) {
86         if (v == null) {
87           throw new IllegalArgumentException("v is null");
88         }
89         v.visitCheckCast(this);
90       }
91     
92       /*
93        * @see com.ibm.wala.ssa.SSAInstruction#hasDef()
94        */
95       @Override
96       public boolean hasDef() {
97         return true;
98       }
99     
100       /**
101        * @return A new value number def'fed by this instruction when the type check succeeds.
102        */
103       @Override
104       public int getDef() {
105         return result;
106       }
107     
108       /*
109        * @see com.ibm.wala.ssa.SSAInstruction#getDef(int)
110        */
111       @Override
112       public int getDef(int i) {
113         assert i == 0;
114         return result;
115       }
116     
117       /*
118        * @see com.ibm.wala.ssa.SSAInstruction#getNumberOfUses()
119        */
120       @Override
121       public int getNumberOfDefs() {
122         return 1;
123       }
124     
125       @Override
126       public int getNumberOfUses() {
127         return 1;
128       }
129     
130       /**
131        * @see com.ibm.wala.ssa.SSAInstruction#getUse(int)
132        */
133       @Override
134       public int getUse(int j) {
135         assert j == 0;
136         return val;
137       }
138     
139       /**
140        * @deprecated the system now supports multiple types, so this
141        * accessor will not work for all languages.
142        */
143       @Deprecated
144       public TypeReference getDeclaredResultType() {
145         assert declaredResultTypes.length == 1;
146         return declaredResultTypes[0];
147       }
148     
149       public TypeReference[] getDeclaredResultTypes() {
150         return declaredResultTypes;
151       }
152     
153       public int getResult() {
154         return result;
155       }
156     
157       public int getVal() {
158         return val;
159       }
160     
161       @Override
162       public int hashCode() {
163         return result * 7529 + val;
164       }
165     
166       /*
167        * @see com.ibm.wala.ssa.Instruction#isPEI()
168        */
169       @Override
170       public boolean isPEI() {
171         return isPEI;
172       }
173     
174       /*
175        * @see com.ibm.wala.ssa.Instruction#isFallThrough()
176        */
177       @Override
178       public boolean isFallThrough() {
179         return true;
180       }
181     
182       @Override
183       public String toString() {
184         String s = super.toString();
185         for (TypeReference t : declaredResultTypes) {
186           s = s + " " + t;
187         }
188         return s;
189       }
190     
191     }
