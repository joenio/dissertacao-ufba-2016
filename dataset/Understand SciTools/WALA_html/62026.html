25       static {
26         MutableLongSetFactory defaultFactory = new MutableSparseLongSetFactory();

28           try {
29             Class<?> intSetFactoryClass = Class.forName(System.getProperty(INT_SET_FACTORY_CONFIG_PROPERTY_NAME));
30             MutableLongSetFactory intSetFactory = (MutableLongSetFactory) intSetFactoryClass.newInstance();
31             setDefaultLongSetFactory(intSetFactory);
32           } catch (Exception e) {
33             System.err.println(("Cannot use int set factory " + System.getProperty(INT_SET_FACTORY_CONFIG_PROPERTY_NAME)));
34             setDefaultLongSetFactory(defaultFactory);
35           }
36         } else {
37           setDefaultLongSetFactory(defaultFactory);
38         }
39         assert defaultLongSetFactory != null;
40       }

48       /**
49        * This method constructs an appropriate mutable copy of set.
50        * 
51        * @param set
52        * @return a new MutableLongSet object with the same value as set
53        * @throws UnimplementedError if (not ( set instanceof com.ibm.wala.util.intset.SparseLongSet ) ) and (not ( set instanceof
54        *           com.ibm.wala.util.intset.BitVectorLongSet ) ) and (not ( set instanceof
55        *           com.ibm.wala.util.intset.BimodalMutableLongSet ) ) and (not ( set instanceof
56        *           com.ibm.wala.util.intset.DebuggingMutableLongSet ) ) and (not ( set instanceof
57        *           com.ibm.wala.util.intset.SemiSparseMutableLongSet ) ) and (not ( set instanceof
58        *           com.ibm.wala.util.intset.MutableSharedBitVectorLongSet ) )
59        * @throws IllegalArgumentException if set == null
60        */
61       public static MutableLongSet makeMutableCopy(LongSet set) throws IllegalArgumentException, UnimplementedError {
62         if (set == null) {
63           throw new IllegalArgumentException("set == null");
64         }
65         if (set instanceof SparseLongSet) {
66           return MutableSparseLongSet.make(set);
67         } else {
68           Assertions.UNREACHABLE(set.getClass().toString());
69           return null;
70         }
71       }

73       /**
74        * Compute the asymmetric difference of two sets, a \ b.
75        */
76       public static LongSet diff(LongSet A, LongSet B) {
77         if (A == null) {
78           throw new IllegalArgumentException("null A");
79         }
80         if (B == null) {
81           throw new IllegalArgumentException("null B");
82         }
83         return diff(A, B, LongSetUtil.getDefaultLongSetFactory());
84       }

108       public static LongSet diff(LongSet A, LongSet B, MutableLongSetFactory factory) {
109         if (A == null) {
110           throw new IllegalArgumentException("null A");
111         }
112         if (B == null) {
113           throw new IllegalArgumentException("null B");
114         }
115         if (A instanceof SparseLongSet && B instanceof SparseLongSet) {
116           return SparseLongSet.diff((SparseLongSet) A, (SparseLongSet) B);
117         } else {
118           return defaultSlowDiff(A, B, factory);
119         }
120       }

147       /**
148        * @return index \in [low,high] s.t. data[index] = key, or -1 if not found
149        */
150       public static int binarySearch(long[] data, long key, int low, int high) throws IllegalArgumentException {
151         if (data == null) {
152           throw new IllegalArgumentException("null array");
153         }
154         if (data.length == 0) {
155           return -1;
156         }
157         if (low <= high && (low < 0 || high < 0)) {
158           throw new IllegalArgumentException("can't search negative indices");
159         }
160         if (high > data.length - 1) {
161           high = data.length - 1;
162         }
163         if (low <= high) {
164           int mid = (low + high) / 2;
165           long midValue = data[mid];
166           if (midValue == key) {
167             return mid;
168           } else if (midValue > key) {
169             return binarySearch(data, key, low, mid - 1);
170           } else {
171             return binarySearch(data, key, mid + 1, high);
172           }
173         } else {
174           return -1;
175         }
176       }

189       /**
190        * @return a new sparse int set which adds j to s
191        * @throws IllegalArgumentException if s == null
192        */
193       public static LongSet add(LongSet s, int j) throws IllegalArgumentException {
194         if (s == null) {
195           throw new IllegalArgumentException("s == null");
196         }
197         if (s instanceof SparseLongSet) {
198           SparseLongSet sis = (SparseLongSet) s;
199           return SparseLongSet.add(sis, j);
200         } else {
201           // really slow. optimize as needed.
202           MutableSparseLongSet result = MutableSparseLongSet.make(s);
203           result.add(j);
204           return result;
205         }
206       }
