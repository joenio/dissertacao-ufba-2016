11     package com.ibm.wala.cast.java.translator.polyglot;
12     
13     import polyglot.ast.ArrayInit;
14     import polyglot.ast.Expr;
15     import polyglot.frontend.Job;
16     import polyglot.frontend.Scheduler;
17     import polyglot.frontend.VisitorGoal;
18     import polyglot.types.SemanticException;
19     import polyglot.types.Type;
20     import polyglot.visit.AscriptionVisitor;
21     
22     /**
23      * Runs an AscriptionVisitor to make sure that empty array literals actually get a type.
24      * @author rfuhrer
25      */
26     public class AscriptionGoal extends VisitorGoal {
27       /**
28        * 
29        */
30       private static final long serialVersionUID = 7416951196743862079L;
31     
32       public AscriptionGoal(Job job) {
33         super(job,
34           new AscriptionVisitor(job, job.extensionInfo().typeSystem(), job.extensionInfo().nodeFactory()) {
35             @Override
36             public Expr ascribe(Expr e, Type toType) throws SemanticException {
37               if (e instanceof ArrayInit && e.type().isNull()) {
38                 return e.type(toType);
39               }
40               return super.ascribe(e, toType);
41             }
42         }
43         );
44         Scheduler scheduler= job.extensionInfo().scheduler();
45     
46     	addPrereq(scheduler.TypeChecked(job));
47       }
48     }
