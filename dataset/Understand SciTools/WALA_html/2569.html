158         /**
159          * The section is used to build classes of EntryPoints on how they are to be called.
160          *
161          * The section is represented by the last label passed before the entity is reached.
162          *
163          * @return the section of this entrypoint
164          */
165         public ExecutionOrder getSection() {
166             if (this.order.compareTo(ExecutionOrder.BEFORE_LOOP) == -1) return ExecutionOrder.AT_FIRST;
167             if (this.order.compareTo(ExecutionOrder.START_OF_LOOP) == -1) return ExecutionOrder.BEFORE_LOOP;
168             if (this.order.compareTo(ExecutionOrder.MIDDLE_OF_LOOP) == -1) return ExecutionOrder.START_OF_LOOP;
169             if (this.order.compareTo(ExecutionOrder.MULTIPLE_TIMES_IN_LOOP) == -1) return ExecutionOrder.MIDDLE_OF_LOOP;
170             if (this.order.compareTo(ExecutionOrder.END_OF_LOOP) == -1) return ExecutionOrder.MULTIPLE_TIMES_IN_LOOP;
171             if (this.order.compareTo(ExecutionOrder.AFTER_LOOP) == -1) return ExecutionOrder.END_OF_LOOP;
172             if (this.order.compareTo(ExecutionOrder.AT_LAST) == -1) return ExecutionOrder.AFTER_LOOP;
173             return ExecutionOrder.AT_LAST;
174         }

183         /**
184          * The ExecutionOrder is used to partially order EntryPoints. 
185          *
186          * The order has to be understood inclusive! E.g. "after(END_OF_LOOP)" means that the position is __BEFORE__ the
187          * loop is actually closed!
188          *
189          * Before building the model a list of AdroidEntryPoints is to be sorted by that criterion. 
190          * You can use AndroidEntryPoint.ExecutionOrderComperator for that task.
191          */
192         public static class ExecutionOrder implements Comparable<IExecutionOrder>, IExecutionOrder {
193             // This is an Enum-Style class
194             /** Visit the EntryPoint once at the beginning of the model use that for initialization stuff  */
195             public final static ExecutionOrder AT_FIRST = new ExecutionOrder(0);
196             /** Basicly the same as AT_FIRST but visited after AT_FIRST */
197             public final static ExecutionOrder BEFORE_LOOP = new ExecutionOrder (Integer.MAX_VALUE / 8);
198             /** Visit multiple times (endless) in the loop */
199             public final static ExecutionOrder START_OF_LOOP = new ExecutionOrder (Integer.MAX_VALUE / 8 * 2);
200             /** Basicly the same as START_OF_LOOP */
201             public final static ExecutionOrder MIDDLE_OF_LOOP = new ExecutionOrder (Integer.MAX_VALUE / 8 * 3);
202             /** Do multiple calls in the loop. Visited after MIDDLE_OF_LOOP, before EEN_OF_LOOP */
203             public final static ExecutionOrder MULTIPLE_TIMES_IN_LOOP = new ExecutionOrder (Integer.MAX_VALUE / 8 * 4);
204             /** Things in END_OF_LOOP are acutually part of the loop. Use AFTER_LOOP if you want them executed only once */
205             public final static ExecutionOrder END_OF_LOOP = new ExecutionOrder (Integer.MAX_VALUE / 8 * 5);
206             /** Basicly the same as AT_LAST but visited before */
207             public final static ExecutionOrder AFTER_LOOP = new ExecutionOrder (Integer.MAX_VALUE / 8 * 6);
208             /** Last calls in the model */
209             public final static ExecutionOrder AT_LAST = new ExecutionOrder (Integer.MAX_VALUE / 8 * 7);
210             /** This value getts used by the detection heuristic - It is not recommended for manual use. */
211     		public final static ExecutionOrder DEFAULT = MIDDLE_OF_LOOP;

213             private int value;
214             /**
215              *  Unrecommended way to generate the Order based on an Integer.
216              *
217              *  This method is handy when reading back files. In your code you should prefer the methods
218              *  {@link #after(IExecutionOrder)} and {@link #between(IExecutionOrder, IExecutionOrder)}.
219              */
220             public ExecutionOrder(int val) { 
221                 this.value=val; 
222             }
223             /**
224              *  Unrecommended way to generate the Order based on a Label-String.
225              *
226              *  This method is handy when reading back files. If you want to refer to a label you
227              *  should prefer the static members.
228              */
229             public ExecutionOrder(String label) {
230                 if (label.equals("AT_FIRST")) { this.value = ExecutionOrder.AT_FIRST.getOrderValue(); return; }
231                 if (label.equals("BEFORE_LOOP")) { this.value =  ExecutionOrder.BEFORE_LOOP.getOrderValue(); return; }
232                 if (label.equals("START_OF_LOOP")) { this.value = ExecutionOrder.START_OF_LOOP.getOrderValue(); return; }
233                 if (label.equals("MIDDLE_OF_LOOP")) { this.value = ExecutionOrder.MIDDLE_OF_LOOP.getOrderValue(); return; }
234                 if (label.equals("MULTIPLE_TIMES_IN_LOOP")) { this.value = ExecutionOrder.MULTIPLE_TIMES_IN_LOOP.getOrderValue(); return; }
235                 if (label.equals("END_OF_LOOP")) { this.value = ExecutionOrder.END_OF_LOOP.getOrderValue(); return; }
236                 if (label.equals("AFTER_LOOP")) { this.value = ExecutionOrder.AFTER_LOOP.getOrderValue(); return; }
237                 if (label.equals("AT_LAST")) { this.value = ExecutionOrder.AT_LAST.getOrderValue(); return; }
238                 throw new IllegalArgumentException("ExecutionOrder was constructed from an illegal label: " + label);
239             }
240             /** {@inheritDoc} */
241             public int getOrderValue() { return this.value; }

243             /**
244              *  Use {@link #between(IExecutionOrder, IExecutionOrder)} instead.
245              *
246              *  Does the internal calculations for the placement.
247              */
248             private static ExecutionOrder between(int after, int before) {
249                 if ((before - after) == 1) {
250                     // It could be ok to warn here instead of throwing: Functions would be located at the same
251                     // 'slot' in this case.
252                     // An alternative solution could be to choose other values for before or after.
253                     throw new ArithmeticException("Precision to low when cascading the ExecutionOrder. You can prevent this error by "
254                             + "assigning the ExecutionOrders (e.g. using before() and after()) in a different order.");
255                 }
256                 if (after > before) {
257                     throw new IllegalArgumentException("The requested ordering could not be established due to the after-parameter "
258                             + "being greater than the before parameter! after=" + after + " before=" + before);
259                 }
260                 return new ExecutionOrder(after + (before - after) / 2);
261             }

263             /** 
264              *  Use this to place a call to an EntryPoint between two other EntryPoint calls or ExecutionOrder "labels".
265              *  between() does not care about section-boundaries by itself! 
266              *
267              *  Use {@link between(IExecutionOrder[], IExecutionOrder[])} and use labels as additional placement-information 
268              *  to prevent unexpected misplacement.
269              *
270              *  @param  after   the call or "label" to be executed before this one
271              *  @param  before  the call or "label" to be executed after this one (inclusive)
272              *  @return A sortable object to represent the execution order
273              *  @throws ArithmeticException when the precision is no more suitable for further cascading
274              *  @throws IllegalArgumentException if parameter after is larger than before.
275              *  @throws NullPointerException if either parameter is null
276              */
277             public static ExecutionOrder between(IExecutionOrder after, IExecutionOrder before) { 
278                 if (after == null) {
279                     throw new NullPointerException("after may not be null");
280                 }
281                 if (before == null) {
282                     throw new NullPointerException("after may not be null");
283                 }
284                 return between(after.getOrderValue(), before.getOrderValue()); 
285             }

288              *  Helper for internal use.
289              */
290             private static ExecutionOrder between(IExecutionOrder after, int before) { 
291                 return between(after.getOrderValue(), before);

294             /**
295              *  Use this variant to refer to multiple locations.
296              *
297              *  The minimum / maximum is computed before the placement of the ExecutionOrder.
298              *
299              *  This method is intended to be more robust when changing the position-information of referred-to
300              *  ExecutionOrders.
301              *
302              *  In any other means it behaves exactly like {@link #between(IExecutionOrder, IExecutionOrder)}.
303              *
304              *  @param  after   the calls or "labels"  to be executed before this one
305              *  @param  before  the calls or "labels" to be executed after this one (inclusive)
306              *  @return A sortable object to represent the execution order
307              *  @throws ArithmeticException when the precision is no more suitable for further cascading
308              *  @throws IllegalArgumentException if parameter after is larger than before.
309              *  @throws NullPointerException if either parameter is null
310              */
311             public static ExecutionOrder between(IExecutionOrder[] after, IExecutionOrder[] before) {
312                 if ((after == null) || (after.length == 0)) {
313                     throw new NullPointerException("after may not be null or empty array");
314                 }
315                 if ((before == null) || (before.length == 0)) {
316                     throw new NullPointerException("before may not be null or empty array");
317                 }
318                 int max = Integer.MIN_VALUE;
319                 int min = Integer.MAX_VALUE;
320                 for (IExecutionOrder i : after) if (max < i.getOrderValue()) max=i.getOrderValue();
321                 for (IExecutionOrder i : before) if (min > i.getOrderValue()) min=i.getOrderValue();
322                 return between(max, min);
323             }

326                 if ((before == null) || (before.length == 0)) {
327                     throw new NullPointerException("after may not be null or empty array");
328                 }
329                 int min = Integer.MAX_VALUE;
330                 for (IExecutionOrder i : before) if (min > i.getOrderValue()) min=i.getOrderValue();
331                 return between(after, min);
332             }

334             public static ExecutionOrder between(IExecutionOrder[] after, IExecutionOrder before) {
335                 if ((after == null) || (after.length == 0)) {
336                     throw new NullPointerException("after may not be null or empty array");
337                 }
338                 if (before == null) {
339                     throw new NullPointerException("before may not be null");
340                 }
341                 int max = Integer.MIN_VALUE;
342                 for (IExecutionOrder i : after) if (max < i.getOrderValue()) max=i.getOrderValue();
343                 return between(max, before.getOrderValue());
344             }

400                 if (this.compareTo(ExecutionOrder.BEFORE_LOOP) == -1) return ExecutionOrder.AT_FIRST;
401                 if (this.compareTo(ExecutionOrder.START_OF_LOOP) == -1) return ExecutionOrder.BEFORE_LOOP;

405                 if (this.compareTo(ExecutionOrder.AFTER_LOOP) == -1) return ExecutionOrder.END_OF_LOOP;
406                 if (this.compareTo(ExecutionOrder.AT_LAST) == -1) return ExecutionOrder.AFTER_LOOP;
407                 return ExecutionOrder.AT_LAST;
