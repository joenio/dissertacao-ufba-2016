11     package com.ibm.wala.util;
12     
13     import java.io.File;
14     import java.util.ArrayList;
15     
16     /**
17      * Platform-specific utility functions.
18      */
19     public class PlatformUtil {
20     
21       /**
22        * are we running on Mac OS X?
23        */
24       public static boolean onMacOSX() {
25         String osname = System.getProperty("os.name");
26         return osname.toLowerCase().contains("mac");
27         // return System.getProperty("mrj.version") != null;
28       }
29     
30       /**
31        * are we running on Linux?
32        */
33       public static boolean onLinux() {
34         String osname = System.getProperty("os.name");
35         return osname.equalsIgnoreCase("linux");
36       }
37     
38       /**
39        * are we running on Windows?
40        */
41       public static boolean onWindows() {
42         String osname = System.getProperty("os.name");
43         return osname.toLowerCase().contains("windows");
44       }
45     
46       /**
47        * are we running on IKVM? see http://www.ikvm.net
48        */
49       public static boolean onIKVM() {
50         return "IKVM.NET".equals(System.getProperty("java.runtime.name"));
51       }
52     
53       /**
54        * get the jars in the boot classpath. TODO test on more JVMs
55        * 
56        * @throws IllegalStateException
57        *           if boot classpath cannot be found
58        */
59       public static String[] getBootClassPathJars() {
60         String classpath = System.getProperty("sun.boot.class.path");
61         if (classpath == null) {
62           throw new IllegalStateException("could not find boot classpath");
63         }
64         String[] jars = classpath.split(File.pathSeparator);
65         ArrayList<String> result = new ArrayList<String>();
66         for (String jar : jars) {
67           if (jar.endsWith(".jar") && (new File(jar)).exists()) {
68             result.add(jar);
69           }
70         }
71         return result.toArray(new String[result.size()]);
72       }
73     }
