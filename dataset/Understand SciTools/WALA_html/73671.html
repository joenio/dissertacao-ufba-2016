11     package com.ibm.wala.cast.tree;
12     
13     import java.util.Collection;
14     
15     import com.ibm.wala.util.debug.Assertions;
16     
17     /**
18      * The control flow information for the CAPA AST of a particular entity. An ast
19      * may contain various nodes that pertain to control flow---such as gotos,
20      * branches, exceptions and so on---and this map denotes the target ast nodes of
21      * ast nodes that are control flow instructions. The label is fairly
22      * arbitrary---it will depend on the language, producers and consumers of the
23      * tree---but is generally expected to be things like case labels, exception
24      * types, conditional outcomes and so on.
25      * 
26      * @author Julian Dolby (dolby@us.ibm.com)
27      * 
28      */
29     public interface CAstControlFlowMap {
30     
31       /**
32        * A distinguished label that means this control flow is the default target of
33        * a switch (or case) statement as found in many procedural languages.
34        */
35       public static final Object SWITCH_DEFAULT = new Object();
36     
37       /**
38        * A distinguished target that means this control flow is the target of an
39        * uncaught exception.
40        */
41       public static final CAstNode EXCEPTION_TO_EXIT = new CAstNode() {
42         @Override
43         public int getKind() {
44           return CAstNode.CONSTANT;
45         }
46     
47         @Override
48         public Object getValue() {
49           return this;
50         }
51     
52         @Override
53         public CAstNode getChild(int n) {
54           Assertions.UNREACHABLE();
55           return null;
56         }
57     
58         @Override
59         public int getChildCount() {
60           return 0;
61         }
62     
63         @Override
64         public String toString() {
65           return "EXCEPTION_TO_EXIT";
66         }
67     
68         @Override
69         public int hashCode() {
70           return getKind() * toString().hashCode();
71         }
72     
73         @Override
74         public boolean equals(Object o) {
75           return o == this;
76         }
77       };
78     
79       /**
80        * Return the target ast node of the control-flow instruction denoted by from
81        * with respect to the given label.
82        */
83       CAstNode getTarget(CAstNode from, Object label);
84     
85       /**
86        * Return a collection of all labels for which the control-flow ast node
87        * <code>from</code> has a target.
88        */
89       Collection<Object> getTargetLabels(CAstNode from);
90     
91       /**
92        * Return a collection of control-flow ast nodes that have this one as a
93        * possible target.
94        */
95       Collection getSourceNodes(CAstNode to);
96     
97       /**
98        * Returns an iterator of all CAstNodes for which this map contains control
99        * flow mapping information.
100        */
101       Collection<CAstNode> getMappedNodes();
102     }
