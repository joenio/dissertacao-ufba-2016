55       public ShrikeClass(ShrikeClassReaderHandle reader, IClassLoader loader, IClassHierarchy cha) throws InvalidClassFileException {
56         super(loader, cha);
57         if (reader == null) {
58           throw new IllegalArgumentException("reader is null");
59         }
60         this.reader = reader;
61         computeTypeReference();
62         this.hashCode = 2161 * getReference().hashCode();
63         // as long as the reader is around, pull more data out
64         // of it before the soft reference to it disappears
65         computeSuperName();
66         computeModifiers();
67         computeInterfaceNames();
68         computeFields();
69       }

71       /**
72        * Compute the fields declared by this class
73        * 
74        * @throws InvalidClassFileException
75        *           iff Shrike fails to read the class file correctly
76        */
77       private void computeFields() throws InvalidClassFileException {
78         ClassReader cr = reader.get();
79         int fieldCount = cr.getFieldCount();
80         List<FieldImpl> instanceList = new ArrayList<FieldImpl>(fieldCount);
81         List<FieldImpl> staticList = new ArrayList<FieldImpl>(fieldCount);
82         try {
83           for (int i = 0; i < fieldCount; i++) {
84             int accessFlags = cr.getFieldAccessFlags(i);
85             Atom name = Atom.findOrCreateUnicodeAtom(cr.getFieldName(i));
86             ImmutableByteArray b = ImmutableByteArray.make(cr.getFieldType(i));
87             Collection<Annotation> annotations = HashSetFactory.make();
88             annotations.addAll(getRuntimeInvisibleAnnotations(i));
89             annotations.addAll(getRuntimeVisibleAnnotations(i));
90             annotations = annotations.isEmpty() ? null : annotations;
91             
92             TypeSignature sig = null;
93             SignatureReader signatureReader = getSignatureReader(i);
94             if (signatureReader != null) {
95               String signature = signatureReader.getSignature();
96               if (signature != null) {
97                 sig = TypeSignature.make(signature);
98               }
99             }
100             
101             if ((accessFlags & ClassConstants.ACC_STATIC) == 0) {
102               addFieldToList(instanceList, name, b, accessFlags, annotations, sig);
103             } else {
104               addFieldToList(staticList, name, b, accessFlags, annotations, sig);
105             }
106           }
107           instanceFields = new IField[instanceList.size()];
108           populateFieldArrayFromList(instanceList, instanceFields);
109           staticFields = new IField[staticList.size()];
110           populateFieldArrayFromList(staticList, staticFields);
111     
112         } catch (InvalidClassFileException e) {
113           e.printStackTrace();
114           Assertions.UNREACHABLE();
115         }
116       }

125       /**
126        * Note that this is called from the constructor, at which point this class is
127        * not yet ready to actually load the superclass. Instead, we pull out the
128        * name of the superclass and cache it here, to avoid hitting the reader
129        * later.
130        */
131       private void computeSuperName() {
132         try {
133           String s = reader.get().getSuperName();
134           if (s != null) {
135             superName = ImmutableByteArray.make("L" + s);
136           }
137         } catch (InvalidClassFileException e) {
138           Assertions.UNREACHABLE();
139         }
140       }

142       /**
143        * Note that this is called from the constructor, at which point this class is
144        * not yet ready to actually load the interfaces. Instead, we pull out the
145        * name of the interfaces and cache it here, to avoid hitting the reader
146        * later.
147        */
148       private void computeInterfaceNames() {
149         try {
150           String[] s = reader.get().getInterfaceNames();
151           interfaceNames = new ImmutableByteArray[s.length];
152           for (int i = 0; i < interfaceNames.length; i++) {
153             interfaceNames[i] = ImmutableByteArray.make("L" + s[i]);
154           }
155         } catch (InvalidClassFileException e) {
156           Assertions.UNREACHABLE();
157         }
158       }

160       /**
161        * initialize the declared methods array
162        * 
163        * @throws InvalidClassFileException
164        */
165       @Override
166       protected ShrikeCTMethod[] computeDeclaredMethods() throws InvalidClassFileException {
167         int methodCount = reader.get().getMethodCount();
168         ShrikeCTMethod[] result = new ShrikeCTMethod[methodCount];
169         for (int i = 0; i < methodCount; i++) {
170           ShrikeCTMethod m = new ShrikeCTMethod(this, i);
171           if (DEBUG) {
172             System.err.println(("Register method " + m + " for class " + this));
173           }
174           result[i] = m;
175         }
176         return result;
177       }

185       private void computeTypeReference() throws InvalidClassFileException {
186         String className = "L" + reader.get().getName();
187         ImmutableByteArray name = ImmutableByteArray.make(className);
188     
189         typeReference = TypeReference.findOrCreate(getClassLoader().getReference(), TypeName.findOrCreate(name));

216       /**
217        * Clear all optional cached data associated with this class
218        */
219       public void clearSoftCaches() {
220         // toss optional information from each method.
221         if (methodMap != null) {
222           for (Iterator it = getDeclaredMethods().iterator(); it.hasNext();) {
223             ShrikeCTMethod m = (ShrikeCTMethod) it.next();
224             m.clearCaches();
225           }
226         }
227         // clear the methodMap cache
228         // SJF: don't do this!!! makes it hard to clear caches on methods.
229         // methodMap = null;
230         inheritCache = null;
231         // clear the cached interfaces
232         allInterfaces = null;
233         // toss away the Shrike reader
234         reader.clear();
235       }

263       private AnnotationsReader getAnnotationsReader(boolean runtimeInvisable) throws InvalidClassFileException {
264         ClassReader r = reader.get();
265         ClassReader.AttrIterator attrs = new ClassReader.AttrIterator();
266         r.initClassAttributeIterator(attrs);
267     
268         return AnnotationsReader.getReaderForAnnotation(runtimeInvisable ? AnnotationType.RuntimeInvisibleAnnotations
269             : AnnotationType.RuntimeVisibleAnnotations, attrs);
270       }

272       private InnerClassesReader getInnerClassesReader() throws InvalidClassFileException {
273         ClassReader r = reader.get();
274         ClassReader.AttrIterator attrs = new ClassReader.AttrIterator();
275         r.initClassAttributeIterator(attrs);
276     
277         // search for the desired attribute
278         InnerClassesReader result = null;
279         try {
280           for (; attrs.isValid(); attrs.advance()) {
281             if (attrs.getName().equals("InnerClasses")) {
282               result = new InnerClassesReader(attrs);
283               break;
284             }
285           }
286         } catch (InvalidClassFileException e) {
287           Assertions.UNREACHABLE();
288         }
289         return result;
290       }

296         return AnnotationsReader.getReaderForAnnotation(runtimeInvisible ? AnnotationType.RuntimeInvisibleAnnotations
297             : AnnotationType.RuntimeVisibleAnnotations, iter);

319       private SignatureReader getSignatureReader(int index) throws InvalidClassFileException {
320         ClassReader r = reader.get();
321         ClassReader.AttrIterator attrs = new ClassReader.AttrIterator();
322         if (index == -1) {
323           r.initClassAttributeIterator(attrs);
324         } else {
325           r.initFieldAttributeIterator(index, attrs);
326         }
327         // search for the desired attribute
328         SignatureReader result = null;
329         try {
330           for (; attrs.isValid(); attrs.advance()) {
331             if (attrs.getName().toString().equals("Signature")) {
332               result = new SignatureReader(attrs);
333               break;
334             }
335           }
336         } catch (InvalidClassFileException e) {
337           Assertions.UNREACHABLE();
338         }
339         return result;
340       }

356       /**
357        * Does the class file indicate that this class is a member of some other
358        * class?
359        * 
360        * @throws InvalidClassFileException
361        */
362       public boolean isInnerClass() throws InvalidClassFileException {
363         InnerClassesReader r = getInnerClassesReader();
364         if (r != null) {
365           for (String s : r.getInnerClasses()) {
366             if (s.equals(getName().toString().substring(1))) {
367               String outer = r.getOuterClass(s);
368               return outer != null;
369             }
370           }
371         }
372         return false;
373       }
374     
375       /**
376        * Does the class file indicate that this class is a static inner class?
377        * 
378        * @throws InvalidClassFileException
379        */
380       public boolean isStaticInnerClass() throws InvalidClassFileException {
381         InnerClassesReader r = getInnerClassesReader();
382         if (r != null) {
383           for (String s : r.getInnerClasses()) {
384             if (s.equals(getName().toString().substring(1))) {
385               String outer = r.getOuterClass(s);
386               if (outer != null) {
387                 int modifiers = r.getAccessFlags(s);
388                 boolean result = ((modifiers & Constants.ACC_STATIC) != 0);
389                 return result;
390               }
391             }
392           }
393         }
394         return false;
395       }

397       /**
398        * If this is an inner class, return the outer class. Else return null.
399        * 
400        * @throws InvalidClassFileException
401        */
402       public TypeReference getOuterClass() throws InvalidClassFileException {
403         if (!isInnerClass()) {
404           return null;
405         }
406         InnerClassesReader r = getInnerClassesReader();
407         for (String s : r.getInnerClasses()) {
408           if (s.equals(getName().toString().substring(1))) {
409             String outer = r.getOuterClass(s);
410             if (outer != null) {
411               return TypeReference.findOrCreate(getClassLoader().getReference(), "L" + outer);
412             }
413           }
414         }
415         return null;
416       }
