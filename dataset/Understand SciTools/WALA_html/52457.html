119         /**
120          *  Generate clinit for AndroidModelClass.
121          *
122          *  clinit initializes AndroidComponents
123          */
124         private SummarizedMethod clinit() {
125             final MethodReference clinitRef = MethodReference.findOrCreate(this.getReference(), MethodReference.clinitSelector);
126             final VolatileMethodSummary clinit = new VolatileMethodSummary(new MethodSummary(clinitRef));
127             clinit.setStatic(true);
128             final TypeSafeInstructionFactory instructionFactory = new TypeSafeInstructionFactory(cha);
129             
130             final Set<TypeReference> components = AndroidEntryPointManager.MANAGER.getComponents();
131             int ssaNo = 1;
132     
133             if (AndroidEntryPointManager.MANAGER.doFlatComponents()) {
134                 for (TypeReference component : components) {
135                     final SSAValue instance = new SSAValue(ssaNo++, component, clinitRef);
136                     { // New
137                         final int pc = clinit.getNextProgramCounter();
138                         final NewSiteReference nRef = NewSiteReference.make(pc, component);
139                         final SSAInstruction instr = instructionFactory.NewInstruction(pc, instance, nRef);
140                         clinit.addStatement(instr);
141                     }
142                     { // Call cTor
143                         final int pc = clinit.getNextProgramCounter();
144                         final MethodReference ctor = MethodReference.findOrCreate(component, MethodReference.initSelector);
145                         final CallSiteReference site = CallSiteReference.make(pc, ctor, IInvokeInstruction.Dispatch.SPECIAL);
146                         final SSAValue exception = new SSAValue(ssaNo++, TypeReference.JavaLangException, clinitRef);
147                         final List<SSAValue> params = new ArrayList<SSAValue>();
148                         params.add(instance);
149                         final SSAInstruction ctorCall = instructionFactory.InvokeInstruction(pc, params, exception, site);
150                         clinit.addStatement(ctorCall);
151                     }
152                     { // Put into AndroidModelClass
153                         final Atom fdName = component.getName().getClassName();
154                         putField(fdName, component);
155                         final int pc = clinit.getNextProgramCounter();
156                         final FieldReference fdRef = FieldReference.findOrCreate(this.getReference(), fdName, component);
157                         final SSAInstruction putInst = instructionFactory.PutInstruction(pc, instance, fdRef);
158                         clinit.addStatement(putInst);
159                     }
160                 }
161             }
162     
163             return new SummarizedMethodWithNames(clinitRef, clinit, this);
164         }
