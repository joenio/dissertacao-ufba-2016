11     package com.ibm.wala.ipa.callgraph;
12     
13     import com.ibm.wala.classLoader.IMethod;
14     import com.ibm.wala.ipa.callgraph.impl.Everywhere;
15     import com.ibm.wala.ssa.DefUse;
16     import com.ibm.wala.ssa.DefaultIRFactory;
17     import com.ibm.wala.ssa.IR;
18     import com.ibm.wala.ssa.IRFactory;
19     import com.ibm.wala.ssa.SSACache;
20     import com.ibm.wala.ssa.SSAOptions;
21     import com.ibm.wala.util.ref.ReferenceCleanser;
22     
23     /**
24      * A place to hold onto caches of various analysis artifacts.
25      * 
26      * Someday this should maybe go away?
27      */
28     public class AnalysisCache {
29       private final IRFactory<IMethod> irFactory;
30     
31       private final SSACache ssaCache;
32     
33       public AnalysisCache(IRFactory<IMethod> irFactory) {
34         super();
35         this.irFactory = irFactory;
36         this.ssaCache = new SSACache(irFactory);
37         ReferenceCleanser.registerCache(this);
38       }
39     
40       public AnalysisCache() {
41         this(new DefaultIRFactory());
42       }
43     
44       public void invalidate(IMethod method, Context C) {
45         ssaCache.invalidate(method, C);
46       }
47     
48       public SSACache getSSACache() {
49         return ssaCache;
50       }
51     
52       public IRFactory<IMethod> getIRFactory() {
53         return irFactory;
54       }
55     
56       /**
57        * Find or create an IR for the method using the {@link Everywhere} context and default {@link SSAOptions}
58        */
59       public IR getIR(IMethod method) {
60         if (method == null) {
61           throw new IllegalArgumentException("method is null");
62         }
63         return ssaCache.findOrCreateIR(method, Everywhere.EVERYWHERE, new AnalysisOptions().getSSAOptions());
64       }
65     
66     
67       /**
68        * Find or create a DefUse for the IR using the {@link Everywhere} context 
69        */
70       public DefUse getDefUse(IR ir) {
71         if (ir == null) {
72           throw new IllegalArgumentException("ir is null");
73         }
74         return ssaCache.findOrCreateDU(ir, Everywhere.EVERYWHERE);
75       }
76     }
