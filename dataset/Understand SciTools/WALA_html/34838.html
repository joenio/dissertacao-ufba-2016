11     package com.ibm.wala.classLoader;
12     
13     import java.io.IOException;
14     import java.io.InputStream;
15     import java.util.Enumeration;
16     import java.util.HashMap;
17     import java.util.HashSet;
18     import java.util.Iterator;
19     import java.util.jar.JarFile;
20     import java.util.zip.ZipEntry;
21     
22     import com.ibm.wala.util.collections.HashMapFactory;
23     import com.ibm.wala.util.collections.HashSetFactory;
24     import com.ibm.wala.util.debug.Assertions;
25     import com.ibm.wala.util.io.FileUtil;
26     import com.ibm.wala.util.ref.CacheReference;
27     
28     /**
29      * A module which is a wrapper around a Jar file
30      */
31     public class JarFileModule implements Module {
32     
33       private final JarFile file;
34     
35       /**
36        * For efficiency, try to cache the byte[] holding each ZipEntries contents; this will help avoid multiple unzipping
37        */
38       private final HashMap<ZipEntry, Object> cache = HashMapFactory.make();
39     
40       public JarFileModule(JarFile f) {
41         if (f == null) {
42           throw new IllegalArgumentException("null f");
43         }
44         this.file = f;
45       }
46     
47       public String getAbsolutePath() {
48         return file.getName();
49       }
50     
51       @Override
52       public String toString() {
53         return "JarFileModule:" + file.getName();
54       }
55     
56       protected ModuleEntry createEntry(ZipEntry z) {
57         return new JarFileEntry(z.getName(), this);
58       }
59       
60       /*
61        * @see com.ibm.wala.classLoader.Module#getEntries()
62        */
63       @Override
64       public Iterator<ModuleEntry> getEntries() {
65         HashSet<ModuleEntry> result = HashSetFactory.make();
66         for (Enumeration e = file.entries(); e.hasMoreElements();) {
67           ZipEntry Z = (ZipEntry) e.nextElement();
68           result.add(createEntry(Z));
69         }
70         return result.iterator();
71       }
72     
73       // need to do equals() and hashCode() based on file name, since JarFile
74       // does not implement equals() / hashCode()
75     
76       @Override
77       public int hashCode() {
78         return file.getName().hashCode();
79       }
80     
81       @Override
82       public boolean equals(Object obj) {
83         if (this == obj)
84           return true;
85         if (obj == null)
86           return false;
87         if (getClass() != obj.getClass())
88           return false;
89         final JarFileModule other = (JarFileModule) obj;
90         if (!file.getName().equals(other.file.getName()))
91           return false;
92         return true;
93       }
94     
95       public byte[] getContents(ZipEntry entry) {
96         byte[] b = (byte[]) CacheReference.get(cache.get(entry));
97     
98         if (b != null) {
99           return b;
100         }
101     
102         try {
103           InputStream s = file.getInputStream(entry);
104           byte[] bb = FileUtil.readBytes(s);
105           cache.put(entry, CacheReference.make(bb));
106           s.close();
107           return bb;
108         } catch (IOException e) {
109           e.printStackTrace();
110           Assertions.UNREACHABLE();
111           return null;
112         }
113       }
114     
115       public JarFile getJarFile() {
116         return file;
117       }
118     
119     }
