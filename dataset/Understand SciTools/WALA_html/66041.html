157     	  private void checkFunctionBodies(String fileName, String[][] assertions) throws IOException, ClassHierarchyException {
158     		  Map<String, String> sources = HashMapFactory.make();
159     		  for(String[] assertion : assertions) {
160     			  sources.put(assertion[0], assertion[1]);
161     		  }
162     		  
163     		  JavaScriptLoaderFactory loaders = makeLoaders(null);
164     		  AnalysisScope scope = makeScriptScope("tests", fileName, loaders);
165     		  IClassHierarchy cha = makeHierarchy(scope, loaders);
166     		  for(IClass cls : cha) {
167     			  if (cls.getName().toString().contains(fileName)) {
168     				  AstMethod fun = (AstMethod)cls.getMethod(AstMethodReference.fnSelector);
169     				 //System.err.println(fun.getDeclaringClass().getName() + " " + fun.getSourcePosition());
170     				  SourceBuffer sb = new SourceBuffer(fun.getSourcePosition());
171     				  //System.err.println(sb);
172     				  if (sources.containsKey(fun.getDeclaringClass().getName().toString())) {
173     					  System.err.println("checking source of " + fun.getDeclaringClass().getName() + " at " + fun.getSourcePosition());
174     					  Assert.assertEquals(sources.get(fun.getDeclaringClass().getName().toString()), sb.toString());
175     				  }
176     			  }			  
177     		  }
178     	}
