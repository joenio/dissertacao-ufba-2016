18       private static final String prefix = System.getProperty("java.io.tmpdir") + File.separator + "loggedClasses" + File.separator + System.currentTimeMillis();
19       
20       static {
21         System.err.println("scraping to " + prefix);

24       
25       @Override
26       public byte[] transform(ClassLoader loader, String className, Class<?> classBeingRedefined, ProtectionDomain protectionDomain,
27           byte[] classfileBuffer) throws IllegalClassFormatException {
28         try {
29           String sourceFile = null;
30           ClassReader reader = new ClassReader(classfileBuffer);
31           AttrIterator attrs = new ClassReader.AttrIterator();
32           reader.initClassAttributeIterator(attrs);
33           for (; attrs.isValid(); attrs.advance()) {
34             if (attrs.getName().equals("SourceFile")) {
35               SourceFileReader file = new SourceFileReader(attrs);
36               int index = file.getSourceFileCPIndex();
37               sourceFile = reader.getCP().getCPUtf8(index);
38             }
39           }
40           if (className == null || sourceFile == null || !sourceFile.endsWith("java") || true) try {
41             String log = prefix + File.separator + reader.getName() + ".class";
42             (new File(log)).getParentFile().mkdirs();
43             FileOutputStream f = new FileOutputStream(log);
44             f.write(classfileBuffer);
45             f.close();
46           } catch (IOException e) {
47             assert false : e;
48           }
49     
50           return classfileBuffer;
51         } catch (InvalidClassFileException e1) {
52           e1.printStackTrace();
53           throw new IllegalClassFormatException(e1.getLocalizedMessage());
54         }
55       }
56     
57       public static void premain(String agentArgs, Instrumentation inst) {
58         inst.addTransformer(new CodeScraper());
59       }
