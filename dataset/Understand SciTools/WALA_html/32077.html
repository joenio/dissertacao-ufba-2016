91       /**
92        * Create a class instrumenter from a preinitialized class reader.
93        * 
94        * @throws IllegalArgumentException if cr is null
95        */
96       public ClassInstrumenter(String inputName, ClassReader cr, ClassHierarchyProvider cha, boolean reuseStackMaps) throws InvalidClassFileException {
97         if (cr == null) {
98           throw new IllegalArgumentException("cr is null");
99         }
100         this.cr = cr;
101         this.cha = cha;
102         this.reuseStackMaps = reuseStackMaps;
103         methods = new MethodData[cr.getMethodCount()];
104         oldCode = new CodeReader[methods.length];
105         cpr = CTDecoder.makeConstantPoolReader(cr);
106         deletedMethods = new boolean[methods.length];
107         this.inputName = inputName;
108       }

117       /**
118        * Implement this interface to instrument every method of a class using visitMethods() below.
119        */
120       public static interface MethodExaminer {
121         /**
122          * Do something to the method.
123          */
124         public void examineCode(MethodData data);
125       }

127       private void prepareMethod(int i) throws InvalidClassFileException {
128         if (deletedMethods[i]) {
129           methods[i] = null;
130         } else if (methods[i] == null) {
131           ClassReader.AttrIterator iter = new ClassReader.AttrIterator();
132           cr.initMethodAttributeIterator(i, iter);
133           for (; iter.isValid(); iter.advance()) {
134             if (iter.getName().equals("Code")) {
135               CodeReader code = new CodeReader(iter);
136               CTDecoder d = new CTDecoder(code, cpr);
137               try {
138                 d.decode();
139               } catch (InvalidBytecodeException e) {
140                 throw new InvalidClassFileException(code.getRawOffset(), e.getMessage());
141               }
142               MethodData md = new MethodData(d, cr.getMethodAccessFlags(i), CTDecoder.convertClassToType(cr.getName()), cr
143                   .getMethodName(i), cr.getMethodType(i));
144               methods[i] = md;
145               oldCode[i] = code;
146               return;
147             }
148           }
149         }
150       }

154        * 
155        * @param i the index of the method to delete

157       public void deleteMethod(int i) {

164       // create a empty method body and then user can apply patches later on
165       public MethodData createEmptyMethodData(String name, String sig, int access) {
166         // Instruction[] instructions=new Instruction[0];
167         Instruction[] instructions = new Instruction[1];
168         instructions[0] = ReturnInstruction.make(Constants.TYPE_void);
169         ExceptionHandler[][] handlers = new ExceptionHandler[instructions.length][];
170         Arrays.fill(handlers, noHandlers);
171         int[] i2b = new int[instructions.length];
172         for (int i = 0; i < i2b.length; i++) {
173           i2b[i] = i;
174         }
175         MethodData md = null;
176         try {
177           md = new MethodData(access, Util.makeType(cr.getName()), name, sig, instructions, handlers, i2b);
178     
179         } catch (InvalidClassFileException ex) {
180           ex.printStackTrace();
181         }
182         return md;
183     
184       }

186       /**
187        * Do something to every method in the class. This will visit all methods, including those already marked for deletion.
188        * 
189        * @param me the visitor to apply to each method
190        */
191       public void visitMethods(MethodExaminer me) throws InvalidClassFileException {
192         for (int i = 0; i < methods.length; i++) {
193           prepareMethod(i);
194           if (methods[i] != null) {
195             me.examineCode(methods[i]);
196           }
197         }
198       }

225       public void resetMethod(int i) {

230       /**
231        * Replace the code for method i with new code. This also "undeletes" the method if it was marked for deletion.
232        * 
233        * @param i the index of the method to replace
234        * @throws IllegalArgumentException if md is null
235        */
236       public void replaceMethod(int i, MethodData md) {
237         if (md == null) {
238           throw new IllegalArgumentException("md is null");
239         }
240         deletedMethods[i] = false;
241         methods[i] = md;
242         oldCode[i] = null;
243         md.setHasChanged();
244       }

246       /**
247        * Check whether any methods in the class have actually been changed.
248        */
249       public boolean isChanged() {
250         for (int i = 0; i < methods.length; i++) {
251           if (deletedMethods[i] || (methods[i] != null && methods[i].getHasChanged())) {
252             return true;
253           }
254         }
255         return false;
256       }

273       /**
274        * Copy the contents of the old class, plus any method modifications, into a new ClassWriter. The ClassWriter must be empty!
275        * 
276        * @param w the classwriter to copy into.
277        */
278       private void emitClassInto(ClassWriter w) throws InvalidClassFileException {
279         w.setMajorVersion(cr.getMajorVersion());
280         w.setMinorVersion(cr.getMinorVersion());
281         w.setRawCP(cr.getCP(), false);
282         w.setAccessFlags(cr.getAccessFlags());
283         w.setNameIndex(cr.getNameIndex());
284         w.setSuperNameIndex(cr.getSuperNameIndex());
285         w.setInterfaceNameIndices(cr.getInterfaceNameIndices());
286     
287         int fieldCount = cr.getFieldCount();
288         for (int i = 0; i < fieldCount; i++) {
289           w.addRawField(new ClassWriter.RawElement(cr.getBytes(), cr.getFieldRawOffset(i), cr.getFieldRawSize(i)));
290         }
291     
292         for (int i = 0; i < methods.length; i++) {
293           MethodData md = methods[i];
294           if (!deletedMethods[i]) {
295             if (md == null || !md.getHasChanged()) {
296               w.addRawMethod(new ClassWriter.RawElement(cr.getBytes(), cr.getMethodRawOffset(i), cr.getMethodRawSize(i)));
297             } else {
298               CTCompiler comp = CTCompiler.make(w, md);
299               comp.setPresetConstants(cpr);
300     
301               try {
302                 comp.compile();
303               } catch (Error ex) {
304                 ex.printStackTrace();
305                 throw new Error("Error compiling method " + md + ": " + ex.getMessage());
306               } catch (Exception ex) {
307                 ex.printStackTrace();
308                 throw new Error("Error compiling method " + md + ": " + ex.getMessage());
309               }
310     
311               CodeReader oc = oldCode[i];
312               int flags = cr.getMethodAccessFlags(i);
313               // we're not installing a native method here
314               flags &= ~ClassConstants.ACC_NATIVE;
315               w.addMethod(flags, cr.getMethodNameIndex(i), cr.getMethodTypeIndex(i), makeMethodAttributes(i, w, oc, comp.getOutput(), md));
316               Compiler.Output[] aux = comp.getAuxiliaryMethods();
317               if (aux != null) {
318                 for (int j = 0; j < aux.length; j++) {
319                   Compiler.Output a = aux[j];
320                   w.addMethod(a.getAccessFlags(), a.getMethodName(), a.getMethodSignature(), makeMethodAttributes(i, w, oc, a, md));
321                 }
322               }
323             }
324           }
325         }
326     
327         ClassReader.AttrIterator iter = new ClassReader.AttrIterator();
328         cr.initClassAttributeIterator(iter);
329         for (; iter.isValid(); iter.advance()) {
330           w.addClassAttribute(new ClassWriter.RawElement(cr.getBytes(), iter.getRawOffset(), iter.getRawSize()));
331         }
332       }

334       private static CodeWriter makeNewCode(ClassWriter w, Compiler.Output output) {
335         CodeWriter code = new CodeWriter(w);
336         code.setMaxStack(output.getMaxStack());
337         code.setMaxLocals(output.getMaxLocals());
338         code.setCode(output.getCode());
339         code.setRawHandlers(output.getRawHandlers());
340         return code;
341       }

343       private LineNumberTableWriter makeNewLines(ClassWriter w, CodeReader oldCode, Compiler.Output output)
344           throws InvalidClassFileException {
345         int[] newLineMap = null;
346         int[] oldLineMap = LineNumberTableReader.makeBytecodeToSourceMap(oldCode);
347         if (oldLineMap != null) {
348           // Map the old line number map onto the new bytecodes
349           int[] newToOldMap = output.getNewBytecodesToOldBytecodes();
350           newLineMap = new int[newToOldMap.length];
351           for (int i = 0; i < newToOldMap.length; i++) {
352             int old = newToOldMap[i];
353             if (old >= 0) {
354               newLineMap[i] = oldLineMap[old];
355             }
356           }
357         } else if (createFakeLineNumbers) {
358           newLineMap = new int[output.getCode().length];
359           for (int i = 0; i < newLineMap.length; i++) {
360             newLineMap[i] = i + fakeLineOffset;
361           }
362         } else {
363           return null;
364         }
365     
366         // Now compress it into the JVM form
367         int[] rawTable = LineNumberTableWriter.makeRawTable(newLineMap);
368         if (rawTable == null || rawTable.length == 0) {
369           return null;
370         } else {
371           LineNumberTableWriter lines = new LineNumberTableWriter(w);
372           lines.setRawTable(rawTable);
373           return lines;
374         }
375       }

377       private static LocalVariableTableWriter makeNewLocals(ClassWriter w, CodeReader oldCode, Compiler.Output output)
378           throws InvalidClassFileException {
379         int[][] oldMap = LocalVariableTableReader.makeVarMap(oldCode);
380         if (oldMap != null) {
381           // Map the old map onto the new bytecodes
382           int[] newToOldMap = output.getNewBytecodesToOldBytecodes();
383           int[][] newMap = new int[newToOldMap.length][];
384           int[] lastLocals = null;
385           for (int i = 0; i < newToOldMap.length; i++) {
386             int old = newToOldMap[i];
387             if (old >= 0) {
388               newMap[i] = oldMap[old];
389               lastLocals = newMap[i];
390             } else {
391               newMap[i] = lastLocals;
392             }
393           }
394     
395           int[] rawTable = LocalVariableTableWriter.makeRawTable(newMap, output);
396           if (rawTable == null || rawTable.length == 0) {
397             return null;
398           } else {
399             LocalVariableTableWriter locals = new LocalVariableTableWriter(w);
400             locals.setRawTable(rawTable);
401             return locals;
402           }
403         } else {
404           return null;
405         }
406       }

408       private ClassWriter.Element[] makeMethodAttributes(int m, ClassWriter w, CodeReader oldCode, Compiler.Output output, MethodData md)
409           throws InvalidClassFileException {
410         CodeWriter code = makeNewCode(w, output);
411     
412         int codeAttrCount = 0;
413         LineNumberTableWriter lines = null;
414         LocalVariableTableWriter locals = null;
415         StackMapTableWriter stacks = null;
416         if (oldCode != null) {
417           lines = makeNewLines(w, oldCode, output);
418           if (lines != null) {
419             codeAttrCount++;
420           }
421           locals = makeNewLocals(w, oldCode, output);
422           if (locals != null) {
423             codeAttrCount++;
424           }
425           if (oldCode.getClassReader().getMajorVersion() > 50) {
426             try { 
427               List<StackMapFrame> sm = StackMapTableReader.readStackMap(oldCode);
428     
429               String[][] varTypes = null;
430               int[] newToOld = output.getNewBytecodesToOldBytecodes();
431               int[][] vars = LocalVariableTableReader.makeVarMap(oldCode);
432               if (vars != null) {
433                 varTypes = new String[newToOld.length][];
434                 for(int i = 0; i < newToOld.length; i++) {
435                   int idx = newToOld[i];
436                   if (idx != -1 && vars[idx] != null) {
437                     varTypes[i] = new String[vars[idx].length / 2];
438                     for(int j = 1; j < vars[idx].length; j += 2) {
439                       int type = vars[idx][j];
440                       varTypes[i][j/2] = type==0? null: oldCode.getClassReader().getCP().getCPUtf8(type);
441                     }
442                   }
443                 }
444               }
445               
446               stacks = new StackMapTableWriter(w, md, output, cha, varTypes , sm);
447               codeAttrCount++;
448             } catch (IOException | FailureException e) {
449               // TODO Auto-generated catch block
450               e.printStackTrace();
451             }
452           }
453         }
454         ClassWriter.Element[] codeAttributes = new ClassWriter.Element[codeAttrCount];
455         int codeAttrIndex = 0;
456         if (lines != null) {
457           codeAttributes[codeAttrIndex++] = lines;
458         }
459         if (locals != null) {
460           codeAttributes[codeAttrIndex++] = locals;
461         }
462         if (stacks != null) {
463           codeAttributes[codeAttrIndex++] = stacks;      
464         }
465         code.setAttributes(codeAttributes);
466     
467         ClassReader.AttrIterator iter = new ClassReader.AttrIterator();
468         cr.initMethodAttributeIterator(m, iter);
469         int methodAttrCount = iter.getRemainingAttributesCount();
470         if (oldCode == null) {
471           methodAttrCount++;
472         }
473         ClassWriter.Element[] methodAttributes = new ClassWriter.Element[methodAttrCount];
474         for (int i = 0; iter.isValid(); iter.advance()) {
475           if (iter.getName().equals("Code")) {
476             methodAttributes[i] = code;
477             code = null;
478             if (oldCode == null) {
479               throw new Error("No old code provided, but Code attribute found");
480             }
481           } else {
482             methodAttributes[i] = new ClassWriter.RawElement(cr.getBytes(), iter.getRawOffset(), iter.getRawSize());
483           }
484           i++;
485         }
486         if (oldCode == null) {
487           if (code == null) {
488             throw new Error("Old code not provided but existing code was found and replaced");
489           }
490           methodAttributes[methodAttrCount - 1] = code;
491         }
492     
493         return methodAttributes;
494       }
