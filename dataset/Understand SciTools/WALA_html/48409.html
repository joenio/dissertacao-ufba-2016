11     package com.ibm.wala.util.collections;
12     
13     import java.lang.reflect.Method;
14     import java.util.LinkedHashMap;
15     import java.util.Map;
16     
17     import com.ibm.wala.util.debug.UnimplementedError;
18     
19     /**
20      * a debugging aid. This implementation complains if you stick an object in here which appears to use System.identityHashCode()
21      */
22     public class ParanoidHashMap<K, V> extends LinkedHashMap<K, V> {
23       public static final long serialVersionUID = 909018793791787198L;
24     
25       /**
26        * @param t
27        * @throws NullPointerException if t is null
28        */
29       public ParanoidHashMap(Map<K, V> t) throws NullPointerException {
30         super(t.size());
31         putAll(t);
32       }
33     
34       public ParanoidHashMap(int size) {
35         super(size);
36       }
37     
38       public ParanoidHashMap() {
39       }
40     
41       /*
42        * @see java.util.Map#put(java.lang.Object, java.lang.Object)
43        */
44       @Override
45       public V put(K arg0, V arg1) {
46         assertOverridesHashCode(arg0);
47         return super.put(arg0, arg1);
48       }
49     
50       public static void assertOverridesHashCode(Object o) throws UnimplementedError {
51         if (o != null && o.hashCode() == System.identityHashCode(o)) {
52           try {
53             Method method = o.getClass().getMethod("hashCode");
54             if (method.getDeclaringClass() == Object.class) {
55               assert false : o.getClass().toString();
56             }
57           } catch (Exception e) {
58             assert false : "Could not find hashCode method";
59           }
60         }
61       }
62     
63       /*
64        * @see java.util.Map#putAll(java.util.Map)
65        */
66       @Override
67       public void putAll(Map<? extends K, ? extends V> arg0) {
68         if (arg0 == null) {
69           throw new IllegalArgumentException("arg0 is null");
70         }
71         for (Map.Entry<? extends K, ? extends V> E : arg0.entrySet()) {
72           put(E.getKey(), E.getValue());
73         }
74       }
75     
76     }
