11     package com.ibm.wala.util.shrike;
12     
13     import com.ibm.wala.types.MemberReference;
14     import com.ibm.wala.util.warnings.Warning;
15     
16     /**
17      * Utility class to help deal with analysis of exceptions.
18      */
19     public class Exceptions {
20     
21       /**
22        * A warning for when we fail to resolve the type for a checkcast
23        */
24       public static class MethodResolutionFailure extends Warning {
25     
26         final MemberReference method;
27     
28         MethodResolutionFailure(byte code, MemberReference method) {
29           super(code);
30           this.method = method;
31         }
32     
33         @Override
34         public String getMsg() {
35           return getClass().toString() + " : " + method;
36         }
37     
38         public static MethodResolutionFailure moderate(MemberReference method) {
39           return new MethodResolutionFailure(Warning.MODERATE, method);
40         }
41     
42         public static MethodResolutionFailure severe(MemberReference method) {
43           return new MethodResolutionFailure(Warning.SEVERE, method);
44         }
45       }
46       
47     }
