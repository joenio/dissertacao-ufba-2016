11     package com.ibm.wala.ipa.callgraph.propagation;
12     
13     import com.ibm.wala.classLoader.IClass;
14     import com.ibm.wala.classLoader.NewSiteReference;
15     import com.ibm.wala.ipa.callgraph.CGNode;
16     
17     /**
18      * Represents an array with length zero. Useful for precision since such arrays
19      * cannot have contents.
20      */
21     public final class ZeroLengthArrayInNode extends AllocationSiteInNode {
22     
23       public ZeroLengthArrayInNode(CGNode node, NewSiteReference allocation, IClass type) {
24         super(node, allocation, type);
25       }
26     
27       @Override
28       public boolean equals(Object obj) {
29         // instanceof is OK because this class is final
30         if (obj instanceof ZeroLengthArrayInNode) {
31           AllocationSiteInNode other = (AllocationSiteInNode) obj;
32           return getNode().equals(other.getNode()) && getSite().equals(other.getSite());
33         } else {
34           return false;
35         }
36       }
37     
38       @Override
39       public int hashCode() {
40         return getNode().hashCode() * 8647 + getSite().hashCode();
41       }
42     
43     }
