26     /**
27      * transforms each synchronized block to execute under a conditional test
28      * calling some method m(), where the block is duplicated in both the if and
29      * else branches. The transformation enables a static analysis to separately
30      * analyze the synchronized block for true and false return values from m().
31      * 
32      * See "Finding Concurrency-Related Bugs using Random Isolation," Kidd et al.,
33      * VMCAI'09, Section 3
34      */
35     public class SynchronizedBlockDuplicator extends
36         CAstRewriter<CAstRewriter.RewriteContext<SynchronizedBlockDuplicator.UnwindKey>, SynchronizedBlockDuplicator.UnwindKey> {
37     
38       /**
39        * key type used for cloning the synchronized blocks and the true and false
40        * branches of the introduced conditional
41        */
42       class UnwindKey implements CAstRewriter.CopyKey<UnwindKey> {
43         /**
44          * are we on the true or false branch?
45          */
46         private boolean testDirection;
47     
48         /**
49          * the AST node representing the synchronized block
50          */
51         private CAstNode syncNode;
52     
53         /**
54          * key associated with the {@link RewriteContext context} of the parent AST
55          * node of the synchronized block
56          */
57         private UnwindKey rest;
58     
59         private UnwindKey(boolean testDirection, CAstNode syncNode, UnwindKey rest) {
60           this.rest = rest;
61           this.syncNode = syncNode;
62           this.testDirection = testDirection;
63         }
64     
65         @Override
66         public int hashCode() {
67           return (testDirection ? 1 : -1) * System.identityHashCode(syncNode) * (rest == null ? 1 : rest.hashCode());
68         }
69     
70         @Override
71         public UnwindKey parent() {
72           return rest;
73         }
74     
75         @Override
76         public boolean equals(Object o) {
77           return (o instanceof UnwindKey) && ((UnwindKey) o).testDirection == testDirection && ((UnwindKey) o).syncNode == syncNode
78               && (rest == null ? ((UnwindKey) o).rest == null : rest.equals(((UnwindKey) o).rest));
79         }
80     
81         @Override
82         public String toString() {
83           return "#" + testDirection + ((rest == null) ? "" : rest.toString());
84         }
85       }
86     
87       // private static final boolean DEBUG = false;
88     
89       /**
90        * method to be invoked in the conditional test (program counter is ignored?
91        * --MS)
92        */
93       private final CallSiteReference f;
94     
95       public SynchronizedBlockDuplicator(CAst Ast, boolean recursive, CallSiteReference f) {
96         super(Ast, recursive, new RootContext());
97         this.f = f;
98       }
99     
100       public CAstEntity translate(CAstEntity original) {
101         return rewrite(original);
102       }
103     
104       /**
105        * context used for nodes not contained in a synchronized block
106        */
107       private static class RootContext implements RewriteContext<UnwindKey> {
108     
109         @Override
110         public UnwindKey key() {
111           return null;
112         }
113       }
114     
115       /**
116        * context used within synchronized blocks
117        */
118       class SyncContext implements RewriteContext<UnwindKey> {
119         /**
120          * context used for the parent AST node of the synchronized block
121          */
122         private final CAstRewriter.RewriteContext<UnwindKey> parent;
123     
124         /**
125          * are we on the true or false branch of the introduced conditional?
126          */
127         private final boolean testDirection;
128     
129         /**
130          * the AST node representing the synchronized block
131          */
132         private final CAstNode syncNode;
133     
134         private SyncContext(boolean testDirection, CAstNode syncNode, RewriteContext<UnwindKey> parent) {
135           this.testDirection = testDirection;
136           this.syncNode = syncNode;
137           this.parent = parent;
138         }
139     
140         @Override
141         public UnwindKey key() {
142           return new UnwindKey(testDirection, syncNode, parent.key());
143         }
144     
145         /**
146          * is n our synchronized block node or the synchronized block node of a
147          * parent?
148          */
149         private boolean containsNode(CAstNode n) {
150           if (n == syncNode) {
151             return true;
152           } else if (parent != null) {
153             return contains(parent, n);
154           } else {
155             return false;
156           }
157         }
158       }
159     
160       @Override
161       protected CAstNode flowOutTo(Map nodeMap, CAstNode oldSource, Object label, CAstNode oldTarget, CAstControlFlowMap orig,
162           CAstSourcePositionMap src) {
163         assert oldTarget == CAstControlFlowMap.EXCEPTION_TO_EXIT;
164         return oldTarget;
165       }
166     
167       private boolean contains(RewriteContext<UnwindKey> c, CAstNode n) {
168         if (c instanceof SyncContext) {
169           return ((SyncContext) c).containsNode(n);
170         } else {
171           return false;
172         }
173       }
174     
175       /**
176        * does root represent a synchronized block? if so, return the variable whose
177        * lock is acquired. otherwise, return <code>null</code>
178        */
179       private String isSynchronizedOnVar(CAstNode root) {
180         if (root.getKind() == CAstNode.UNWIND) {
181           CAstNode unwindBody = root.getChild(0);
182           if (unwindBody.getKind() == CAstNode.BLOCK_STMT) {
183             CAstNode firstStmt = unwindBody.getChild(0);
184             if (firstStmt.getKind() == CAstNode.MONITOR_ENTER) {
185               CAstNode expr = firstStmt.getChild(0);
186               if (expr.getKind() == CAstNode.VAR) {
187                 String varName = (String) expr.getChild(0).getValue();
188     
189                 CAstNode protectBody = root.getChild(1);
190                 if (protectBody.getKind() == CAstNode.MONITOR_EXIT) {
191                   CAstNode expr2 = protectBody.getChild(0);
192                   if (expr2.getKind() == CAstNode.VAR) {
193                     String varName2 = (String) expr2.getChild(0).getValue();
194                     if (varName.equals(varName2)) {
195                       return varName;
196                     }
197                   }
198                 }
199               }
200             }
201           }
202         }
203     
204         return null;
205       }
206     
207       @Override
208       protected CAstNode copyNodes(CAstNode n, final CAstControlFlowMap cfg, RewriteContext<UnwindKey> c, Map<Pair<CAstNode, UnwindKey>, CAstNode> nodeMap) {
209         String varName;
210         // don't copy operators or constants (presumably since they are immutable?)
211         if (n instanceof CAstOperator) {
212           return n;
213     
214         } else if (n.getValue() != null) {
215           return Ast.makeConstant(n.getValue());
216         } else if (!contains(c, n) && (varName = isSynchronizedOnVar(n)) != null) {
217           // we call contains() above since we pass n to copyNodes() below for the
218           // true and false branches of the conditional, and in those recursive
219           // calls we want n to be copied normally
220     
221           // the conditional test
222           CAstNode test = Ast.makeNode(CAstNode.CALL, Ast.makeNode(CAstNode.VOID), Ast.makeConstant(f),
223               Ast.makeNode(CAstNode.VAR, Ast.makeConstant(varName), Ast.makeConstant(JavaPrimitiveTypeMap.lookupType("boolean"))));
224     
225           // the new if conditional
226           return Ast.makeNode(CAstNode.IF_STMT, test, copyNodes(n, cfg, new SyncContext(true, n, c), nodeMap),
227               copyNodes(n, cfg, new SyncContext(false, n, c), nodeMap));
228     
229         } else {
230           // invoke copyNodes() on the children with context c, ensuring, e.g., that
231           // the body of a synchronized block gets cloned
232           CAstNode[] newChildren = new CAstNode[n.getChildCount()];
233           for (int i = 0; i < newChildren.length; i++)
234             newChildren[i] = copyNodes(n.getChild(i), cfg, c, nodeMap);
235     
236           CAstNode newN = Ast.makeNode(n.getKind(), newChildren);
237     
238           nodeMap.put(Pair.make(n, c.key()), newN);
239     
240           return newN;
241         }
242       }
