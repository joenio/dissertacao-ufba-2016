197         private SSAInstruction undo(SSAInstruction inst, int use, int val) {
198           int c = getNumberOfUses(inst);
199           int[] newUses = new int[c];
200           for (int i = 0; i < c; i++) {
201             if (i == use)
202               newUses[i] = val;
203             else
204               newUses[i] = getUse(inst, i);
205           }
206     
207           return inst.copyForSSA(CFG.getMethod().getDeclaringClass().getClassLoader().getInstructionFactory(), null, newUses);
208         }
209     
210         private void undo(int rhs) {
211           int lhs = symtab.newSymbol();
212     
213           instructions[instructionIndex] = new AssignInstruction(instructionIndex, lhs, rhs);
214     
215           if (DEBUG_UNDO)
216             System.err.println(("recreating assignment at " + instructionIndex + " as " + lhs + " = " + rhs));
217     
218           for (Iterator<Object> uses = renamedUses.iterator(); uses.hasNext();) {
219             Object x = uses.next();
220             if (x instanceof UseRecord) {
221               UseRecord use = (UseRecord) x;
222               int idx = use.instructionIndex;
223               SSAInstruction inst = instructions[idx];
224     
225               if (DEBUG_UNDO)
226                 System.err.println(("Changing use #" + use.useNumber + " of inst #" + idx + " to val " + lhs));
227     
228               if (use.useNumber >= 0) {
229                 instructions[idx] = undo(inst, use.useNumber, lhs);
230               } else {
231                 lexicalInfo.getExposedUses(idx)[-use.useNumber - 1] = lhs;
232               }
233               copyPropagationMap.remove(use);
234             } else {
235               PhiUseRecord use = (PhiUseRecord) x;
236               int bb = use.BBnumber;
237               int phi = use.phiNumber;
238               SSACFG.BasicBlock BB = CFG.getNode(bb);
239               BB.addPhiForLocal(phi, (SSAPhiInstruction) undo(BB.getPhiForLocal(phi), use.useNumber, lhs));
240               copyPropagationMap.remove(use);
241             }
242           }
243     
244           for (Iterator<CopyPropagationRecord> cs = childRecords.iterator(); cs.hasNext();) {
245             cs.next().undo(lhs);
246           }
247         }
248     
249         public void undo() {
250           undo(this.rhs);
251           copyPropagationMap.remove(new UseRecord(instructionIndex, rhs));
252         }
253       }

271         @Override
272         public String[] getLocalNames(int pc, int vn) {
273           
274           if (computedNames[vn] != null) {
275             return computedNames[vn];
276           }
277           
278           int v = skip(vn) || vn >= valueMap.length ? vn : valueMap[vn];
279           String[][] namesData = debugInfo.getSourceNamesForValues();
280           String[] vNames = namesData[v];
281           Set<String> x = HashSetFactory.make();
282           x.addAll(Arrays.asList(vNames));
283      
284           MutableIntSet vals = IntSetUtil.make();
285           while (assignments.containsKey(v) && !vals.contains(v)) {
286             vals.add(v);
287             v = assignments.get(v);
288             vNames = namesData[v];
289             x.addAll(Arrays.asList(vNames));        
290           }
291     
292           return computedNames[vn] = x.toArray(new String[x.size()]);
293         }

329       private CopyPropagationRecord topR(int v) {
330         if (R[v] != null && !R[v].isEmpty()) {
331           CopyPropagationRecord rec = R[v].peek();
332           if (top(v) == rec.rhs) {
333             return rec;
334           }
335         }
336     
337         return null;
338       }

417       @Override
418       protected void repairPhiUse(SSACFG.BasicBlock BB, int phiIndex, int rvalIndex, int newRval) {
419         SSAPhiInstruction phi = getPhi(BB, phiIndex);
420     
421         int[] newUses = new int[getNumberOfUses(phi)];
422         for (int v = 0; v < newUses.length; v++) {
423           int oldUse = getUse(phi, v);
424           int newUse = (v == rvalIndex) ? newRval : oldUse;
425           newUses[v] = newUse;
426     
427           if (v == rvalIndex && topR(oldUse) != null) {
428             topR(oldUse).addUse(BB.getGraphNodeId(), phiIndex, v);
429           }
430         }
431     
432         phi.setValues(newUses);
433       }

435       @Override
436       protected void pushAssignment(SSAInstruction inst, int index, int newRhs) {
437         int lhs = getDef(inst, 0);
438         int rhs = getUse(inst, 0);
439     
440         assignments.put(rhs, lhs);
441         
442         CopyPropagationRecord rec = new CopyPropagationRecord(index, lhs, newRhs);
443         R[lhs].push(rec);
444         if (topR(rhs) != null) {
445           topR(rhs).addChild(rec);
446         }
447       }

449       @Override
450       protected void repairInstructionUses(SSAInstruction inst, int index, int[] newUses) {
451         for (int j = 0; j < getNumberOfUses(inst); j++) {
452           if (topR(getUse(inst, j)) != null) {
453             topR(getUse(inst, j)).addUse(index, j);
454           }
455         }
456     
457         int[] lexicalUses = lexicalInfo.getExposedUses(index);
458         if (lexicalUses != null) {
459           for (int j = 0; j < lexicalUses.length; j++) {
460             int lexicalUse = lexicalUses[j];
461             if (lexicalUse != -1 && !skip(lexicalUse)) {
462               if (S.length <= lexicalUse ||  S[lexicalUse].isEmpty()) {
463                 lexicalUses[j] = -1;
464               } else {
465                 int newUse = top(lexicalUse);
466     
467                 lexicalUses[j] = newUse;
468     
469                 if (topR(lexicalUse) != null) {
470                   topR(lexicalUse).addUse(index, -j - 1);
471                 }
472               }
473             }
474           }
475         }
476       }

518       @SuppressWarnings("unchecked")
519       private SSAConversion(AstMethod M, AstIR ir, SSAOptions options) {
520         super(ir, options);
521         HashMap<Object, CopyPropagationRecord> m = HashMapFactory.make();
522         this.copyPropagationMap = (ir.getLocalMap() instanceof SSAInformation) ? ((SSAInformation) ir.getLocalMap()).getCopyHistory()
523             : m;
524     
525         this.ir = ir;
526         this.debugInfo = M.debugInfo();
527         this.lexicalInfo = ir.lexicalInfo();
528         this.symtab = ir.getSymbolTable();
529         this.R = new Stack[ir.getSymbolTable().getMaxValueNumber() + 1];
530     
531         for (int i = 0; i < CFG.getNumberOfNodes(); i++) {
532           SSACFG.BasicBlock bb = CFG.getNode(i);
533           if (bb.hasPhi()) {
534             int n = 0;
535             for (Iterator X = bb.iteratePhis(); X.hasNext(); n++)
536               X.next();
537             phiCounts[i] = n;
538           }
539         }
540     
541         this.nextSSAValue = ir.getNumberOfParameters() + 1;
542     
543         int[] exitLive = lexicalInfo.getExitExposedUses();
544         BitVector v = new BitVector();
545         if (exitLive != null) {
546           for (int i = 0; i < exitLive.length; i++) {
547             if (exitLive[i] > -1) {
548               v.set(exitLive[i]);
549             }
550           }
551         }
552         this.liveness = LiveAnalysis.perform(CFG, symtab, v);
553     
554         if (DEBUG) {
555           System.err.println(liveness);
556         }
557       }

561         while (symtab.isConstant(nextSSAValue) || skip(nextSSAValue))
562           ++nextSSAValue;
563         symtab.ensureSymbol(nextSSAValue);
564         int v = nextSSAValue++;
565         return v;
566       }

579             S[params[i]].push(params[i]);

637       private static IntSet valuesToConvert(AstIR ir) {
638         SSAInstruction[] insts = ir.getInstructions();
639         MutableIntSet foundOne = new BitVectorIntSet();
640         MutableIntSet foundTwo = new BitVectorIntSet();
641         for (int i = 0; i < insts.length; i++) {
642           SSAInstruction inst = insts[i];
643           if (inst != null) {
644             for (int j = 0; j < inst.getNumberOfDefs(); j++) {
645               int def = inst.getDef(j);
646               if (def != -1) {
647                 if (foundOne.contains(def) || ir.getSymbolTable().isConstant(def) || def <= ir.getNumberOfParameters()
648                     || inst instanceof AssignInstruction) {
649                   foundTwo.add(def);
650                 } else {
651                   foundOne.add(def);
652                 }
653               }
654             }
655           }
656         }
657     
658         return foundTwo;
659       }

665       public static SSA2LocalMap convert(AstMethod M, final AstIR ir, SSAOptions options, final IntSet values) {
666         try {
667           if (DEBUG) {
668             System.err.println(("starting conversion for " + values));
669             System.err.println(ir);
670           }
671           if (DEBUG_UNDO)
672             System.err.println((">>> starting " + ir.getMethod()));
673           SSAConversion ssa = new SSAConversion(M, ir, options) {
674             final int limit = ir.getSymbolTable().getMaxValueNumber();
675     
676             @Override
677             protected boolean skip(int i) {
678               return (i >= 0) && (i <= limit) && (!values.contains(i));
679             }
680           };
681           ssa.perform();
682           if (DEBUG_UNDO)
683             System.err.println(("<<< done " + ir.getMethod()));
684           return ssa.getComputedLocalMap();
685         } catch (RuntimeException e) {
686     //      System.err.println(("exception " + e + " while converting:"));
687     //      System.err.println(ir);
