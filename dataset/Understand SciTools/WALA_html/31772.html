304     	public void addPossibleListeners(ClassHierarchy cha) {
305     		Set<String> ignoreMethods = new HashSet<String>();
306     		ignoreMethods.add("<init>");
307     		ignoreMethods.add("<clinit>");
308     		ignoreMethods.add("registerNatives");
309     		ignoreMethods.add("getClass");
310     		ignoreMethods.add("hashCode");
311     		ignoreMethods.add("equals");
312     		ignoreMethods.add("clone");
313     		ignoreMethods.add("toString");
314     		ignoreMethods.add("notify");
315     		ignoreMethods.add("notifyAll");
316     		ignoreMethods.add("finalize");
317     		ignoreMethods.add("wait");		
318     
319     		List<MethodNamePattern> moreEntryPointSpecs = new ArrayList<MethodNamePattern> ();
320     		
321     		//add default entrypoints from AndroidSpecs.entrypointSpecs
322     		//Currently adds methods even if they exist in the ignnoreMethods
323     		//set.
324     		for (MethodNamePattern mnp: defaultCallbacks) {
325     			moreEntryPointSpecs.add(mnp);
326     		}
327     
328     		for (IClass ic:cha) {
329     			if (!LoaderUtils.fromLoader(ic, ClassLoaderReference.Application)) {
330     				continue;
331     			}
332     
333     			//finds all *Listener classes and fetches all methods for the listener
334     			if (ic.getName().getClassName().toString().endsWith("Listener")) {
335     				for (IMethod im: ic.getAllMethods()) {
336     					//TODO: add isAbstract()?
337     					if (!ignoreMethods.contains(im.getName().toString()) && !im.isPrivate()) {
338     						moreEntryPointSpecs.add(
339     								new MethodNamePattern(ic.getName().toString(), 
340     										im.getName().toString()));
341     					}
342     				}
343     			}
344     			//not a listener, just find all the methods that start with "on____"
345     			else {
346     				for (IMethod im:ic.getAllMethods()) {
347     					//TODO: add isAbstract()?
348     					if (!ignoreMethods.contains(im.getName().toString()) &&
349     							im.getName().toString().startsWith("on") && !im.isPrivate()) {
350     						moreEntryPointSpecs.add(new MethodNamePattern(ic.getName().toString(),
351     								im.getName().toString()));
352     					}
353     				}
354     			}
355     		}
356     
357     //		entrypointSpecs =
358     		callBacks = 
359     				moreEntryPointSpecs.toArray(new MethodNamePattern[moreEntryPointSpecs.size()]);
360     
361     	}
362     	
363     	public MethodNamePattern[] getCallBacks() {
364     		return callBacks;
365     	}
