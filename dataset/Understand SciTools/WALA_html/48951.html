143       /**
144        * If n is relevant and its cfg has not already been added, add nodes and edges for n
145        * 
146        * @param n
147        */
148       private void addIntraproceduralNodesAndEdgesForCGNodeIfNeeded(CGNode n) {
149         if (!cgNodesVisited.contains(cg.getNumber(n)) && relevant.test(n)) {
150           if (DEBUG_LEVEL > 0) {
151             System.err.println("Adding nodes and edges for cg node: " + n);
152           }
153           cgNodesVisited.add(cg.getNumber(n));
154           // retrieve a cfg for node n.
155           ControlFlowGraph<SSAInstruction, T> cfg = getCFG(n);
156           if (cfg != null) {
157             // create a node for each basic block.
158             addNodeForEachBasicBlock(cfg, n);
159             SSAInstruction[] instrs = cfg.getInstructions();
160             // create edges for node n.
161             for (Iterator<T> bbs = cfg.iterator(); bbs.hasNext();) {
162               T bb = bbs.next();
163               if (bb != cfg.entry())
164                 addEdgesToNonEntryBlock(n, cfg, instrs, bb);
165             }
166           }
167         }
168       }

170       /**
171        * Add edges to the IPCFG for the incoming edges incident on a basic block bb.
172        * 
173        * @param n a call graph node
174        * @param cfg the CFG for n
175        * @param instrs the instructions for node n
176        * @param bb a basic block in the CFG
177        */
178       protected void addEdgesToNonEntryBlock(CGNode n, ControlFlowGraph<?, T> cfg, SSAInstruction[] instrs, T bb) {
179         if (DEBUG_LEVEL > 1) {
180           System.err.println("addEdgesToNonEntryBlock: " + bb);
181           System.err.println("nPred: " + cfg.getPredNodeCount(bb));
182         }
183     
184         for (Iterator<? extends T> ps = cfg.getPredNodes(bb); ps.hasNext();) {
185           T pb = ps.next();
186           if (DEBUG_LEVEL > 1) {
187             System.err.println("Consider previous block: " + pb);
188           }
189     
190           if (pb.equals(cfg.entry())) {
191             // entry block has no instructions
192             BasicBlockInContext<T> p = new BasicBlockInContext<T>(n, pb);
193             BasicBlockInContext<T> b = new BasicBlockInContext<T>(n, bb);
194             g.addEdge(p, b);
195             continue;
196           }
197           
198           SSAInstruction inst = getLastInstructionForBlock(pb, instrs);
199     
200           if (DEBUG_LEVEL > 1) {
201             System.err.println("Last instruction is : " + inst);
202           }
203           if (inst instanceof SSAAbstractInvokeInstruction) {
204             if (CALL_TO_RETURN_EDGES) {
205               // Add a "normal" edge from the predecessor block to this block.
206               BasicBlockInContext<T> p = new BasicBlockInContext<T>(n, pb);
207               BasicBlockInContext<T> b = new BasicBlockInContext<T>(n, bb);
208               g.addEdge(p, b);
209             }
210           } else {
211             // previous instruction is not a call instruction.
212             BasicBlockInContext<T> p = new BasicBlockInContext<T>(n, pb);
213             BasicBlockInContext<T> b = new BasicBlockInContext<T>(n, bb);
214             if (!g.containsNode(p) || !g.containsNode(b)) {
215               assert g.containsNode(p) : "IPCFG does not contain " + p;
216               assert g.containsNode(b) : "IPCFG does not contain " + b;
217             }
218             g.addEdge(p, b);
219           }
220         }
221       }

223       protected SSAInstruction getLastInstructionForBlock(T pb, SSAInstruction[] instrs) {
224         int index = pb.getLastInstructionIndex();
225         SSAInstruction inst = instrs[index];
226         return inst;
227       }

230       /**
231        * Add an edge from the exit() block of a callee to a return site in the caller
232        * 
233        * @param returnBlock the return site for a call
234        * @param targetCFG the called method
235        */
236       private void addEdgesFromExitToReturn(CGNode caller, T returnBlock, CGNode target,
237           ControlFlowGraph<SSAInstruction, ? extends T> targetCFG) {
238         T texit = targetCFG.exit();
239         BasicBlockInContext<T> exit = new BasicBlockInContext<T>(target, texit);
240         addNodeForBasicBlockIfNeeded(exit);
241         BasicBlockInContext<T> ret = new BasicBlockInContext<T>(caller, returnBlock);
242         if (!g.containsNode(exit) || !g.containsNode(ret)) {
243           assert g.containsNode(exit) : "IPCFG does not contain " + exit;
244           assert g.containsNode(ret) : "IPCFG does not contain " + ret;
245         }
246         if (DEBUG_LEVEL > 1) {
247           System.err.println("addEdgeFromExitToReturn " + exit + ret);
248         }
249         g.addEdge(exit, ret);
250       }

252       /**
253        * Add an edge from the exit() block of a callee to a return site in the caller
254        * 
255        * @param callBlock the return site for a call
256        * @param targetCFG the called method
257        */
258       private void addEdgesFromCallToEntry(CGNode caller, T callBlock, CGNode target,
259           ControlFlowGraph<SSAInstruction, ? extends T> targetCFG) {
260         T tentry = targetCFG.entry();
261         BasicBlockInContext<T> entry = new BasicBlockInContext<T>(target, tentry);
262         addNodeForBasicBlockIfNeeded(entry);
263         BasicBlockInContext<T> call = new BasicBlockInContext<T>(caller, callBlock);
264         if (!g.containsNode(entry) || !g.containsNode(call)) {
265           assert g.containsNode(entry) : "IPCFG does not contain " + entry;
266           assert g.containsNode(call) : "IPCFG does not contain " + call;
267         }
268         if (DEBUG_LEVEL > 1) {
269           System.err.println("addEdgeFromCallToEntry " + call + " " + entry);
270         }
271         g.addEdge(call, entry);
272       }

274       /**
275        * Add the incoming edges to the entry() block and the outgoing edges from the exit() block for a call graph node.
276        * 
277        * @param n a node in the call graph
278        * @param bb the entry() block for n
279        */
280       private void addInterproceduralEdgesForEntryAndExitBlocks(CGNode n, ControlFlowGraph<SSAInstruction, ? extends T> cfg) {
281     
282         T entryBlock = cfg.entry();
283         T exitBlock = cfg.exit();
284         if (DEBUG_LEVEL > 0) {
285           System.err.println("addInterproceduralEdgesForEntryAndExitBlocks " + n);
286         }
287     
288         for (Iterator callers = cg.getPredNodes(n); callers.hasNext();) {
289           CGNode caller = (CGNode) callers.next();
290           if (DEBUG_LEVEL > 1) {
291             System.err.println("got caller " + caller);
292           }
293           if (relevant.test(caller)) {
294             addEntryAndExitEdgesToCaller(n, entryBlock, exitBlock, caller);
295           }
296         }
297       }
298     
299       private void addEntryAndExitEdgesToCaller(CGNode n, T entryBlock, T exitBlock, CGNode caller) {
300         if (DEBUG_LEVEL > 0) {
301           System.err.println("caller " + caller + "is relevant");
302         }
303         ControlFlowGraph<SSAInstruction, T> ccfg = getCFG(caller);
304         if (ccfg != null) {
305           SSAInstruction[] cinsts = ccfg.getInstructions();
306     
307           if (DEBUG_LEVEL > 1) {
308             System.err.println("Visiting " + cinsts.length + " instructions");
309           }
310           for (int i = 0; i < cinsts.length; i++) {
311             if (cinsts[i] instanceof SSAAbstractInvokeInstruction) {
312               if (DEBUG_LEVEL > 1) {
313                 System.err.println("Checking invokeinstruction: " + cinsts[i]);
314               }
315               SSAAbstractInvokeInstruction call = (SSAAbstractInvokeInstruction) cinsts[i];
316               CallSiteReference site = call.getCallSite();
317               assert site.getProgramCounter() == ccfg.getProgramCounter(i);
318               if (cg.getPossibleTargets(caller, site).contains(n)) {
319                 if (DEBUG_LEVEL > 1) {
320                   System.err.println("Adding edge " + ccfg.getBlockForInstruction(i) + " to " + entryBlock);
321                 }
322                 T callerBB = ccfg.getBlockForInstruction(i);
323                 BasicBlockInContext<T> b1 = new BasicBlockInContext<T>(caller, callerBB);
324                 // need to add a node for caller basic block, in case we haven't processed caller yet
325                 addNodeForBasicBlockIfNeeded(b1);
326                 BasicBlockInContext<T> b2 = new BasicBlockInContext<T>(n, entryBlock);
327                 g.addEdge(b1, b2);
328                 // also add edges from exit node to all return nodes (successor of call bb)
329                 for (Iterator<? extends T> succIter = ccfg.getSuccNodes(callerBB); succIter.hasNext();) {
330                   T returnBB = succIter.next();
331                   BasicBlockInContext<T> b3 = new BasicBlockInContext<T>(n, exitBlock);
332                   BasicBlockInContext<T> b4 = new BasicBlockInContext<T>(caller, returnBB);
333                   addNodeForBasicBlockIfNeeded(b4);
334                   g.addEdge(b3, b4);
335                 }
336               }
337             }
338           }
339         }
340       }

342       /**
343        * Add a node to the IPCFG for each node in a CFG. side effect: populates the hasCallVector
344        * 
345        * @param cfg a control-flow graph
346        */
347       private void addNodeForEachBasicBlock(ControlFlowGraph<? extends SSAInstruction, ? extends T> cfg, CGNode N) {
348         for (Iterator<? extends T> bbs = cfg.iterator(); bbs.hasNext();) {
349           T bb = bbs.next();
350           if (DEBUG_LEVEL > 1) {
351             System.err.println("IPCFG Add basic block " + bb);
352           }
353           BasicBlockInContext<T> b = new BasicBlockInContext<T>(N, bb);
354           addNodeForBasicBlockIfNeeded(b);
355         }
356       }

358       private void addNodeForBasicBlockIfNeeded(BasicBlockInContext<T> b) {
359         if (!g.containsNode(b)) {
360           g.addNode(b);
361           ControlFlowGraph<SSAInstruction, T> cfg = getCFG(b);
362           if (hasCall(b, cfg)) {
363             hasCallVector.set(g.getNumber(b));
364           }
365         }
366       }

430       private void constructFullGraph() {
431         if (!constructedFullGraph) {
432           for (CGNode n : cg) {
433             addIntraproceduralNodesAndEdgesForCGNodeIfNeeded(n);
434             addEdgesToCallees(n);
435           }
436           for (int i = 0; i < g.getMaxNumber(); i++) {
437             addedSuccs.add(i);
438             addedPreds.add(i);
439           }
440           constructedFullGraph = true;
441         }
442       }

444       /**
445        * add interprocedural edges to nodes in callees of n
446        * 
447        * @param n
448        */
449       private void addEdgesToCallees(CGNode n) {
450         ControlFlowGraph<SSAInstruction, T> cfg = getCFG(n);
451         if (cfg != null) {
452           for (Iterator<T> bbs = cfg.iterator(); bbs.hasNext();) {
453             T bb = bbs.next();
454             BasicBlockInContext<T> block = new BasicBlockInContext<T>(n, bb);
455             if (hasCall(block)) {
456               addCalleeEdgesForCall(n, block);
457             }
458           }
459         }
460       }

462       /**
463        * add edges to callees for return block and corresponding call block(s)
464        */
465       private void addCalleeEdgesForReturn(CGNode node, BasicBlockInContext<T> returnBlock) {
466         final int num = g.getNumber(returnBlock);
467         if (!handledReturns.contains(num)) {
468           handledReturns.add(num);
469           // compute calls for return
470           ControlFlowGraph<SSAInstruction, T> cfg = getCFG(returnBlock);
471           for (Iterator<? extends T> it = cfg.getPredNodes(returnBlock.getDelegate()); it.hasNext();) {
472             T b = it.next();
473             final BasicBlockInContext<T> block = new BasicBlockInContext<T>(node, b);
474             if (hasCall(block)) {
475               addCalleeEdgesForCall(node, block);
476             }
477           }
478         }
479       }

481       /**
482        * add edges to callee entry for call block, and edges from callee exit to corresponding return blocks
483        */
484       private void addCalleeEdgesForCall(CGNode n, BasicBlockInContext<T> callBlock) {
485         int num = g.getNumber(callBlock);
486         if (!handledCalls.contains(num)) {
487           handledCalls.add(num);
488           ControlFlowGraph<SSAInstruction, T> cfg = getCFG(n);
489           CallSiteReference site = getCallSiteForCallBlock(callBlock, cfg);
490           if (DEBUG_LEVEL > 1) {
491             System.err.println("got Site: " + site);
492           }
493           boolean irrelevantTargets = false;
494           for (Iterator ts = cg.getPossibleTargets(n, site).iterator(); ts.hasNext();) {
495             CGNode tn = (CGNode) ts.next();
496             if (!relevant.test(tn)) {
497               if (DEBUG_LEVEL > 1) {
498                 System.err.println("Irrelevant target: " + tn);
499               }
500               irrelevantTargets = true;
501               continue;
502             }
503     
504             if (DEBUG_LEVEL > 1) {
505               System.err.println("Relevant target: " + tn);
506             }
507             // add an edge from tn exit to this node
508             ControlFlowGraph<SSAInstruction, ? extends T> tcfg = getCFG(tn);
509             // tcfg might be null if tn is an unmodelled native method
510             if (tcfg != null) {
511               final T cbDelegate = callBlock.getDelegate();
512               addEdgesFromCallToEntry(n, cbDelegate, tn, tcfg);
513               for (Iterator<? extends T> returnBlocks = cfg.getSuccNodes(cbDelegate); returnBlocks.hasNext();) {
514                 T retBlock = returnBlocks.next();
515                 addEdgesFromExitToReturn(n, retBlock, tn, tcfg);
516                 if (irrelevantTargets) {
517                   // Add a "normal" edge from the call block to the return block.
518                   g.addEdge(callBlock, new BasicBlockInContext<T>(n, retBlock));
519                 }
520               }
521             }
522           }
523         }
524       }

526       /**
527        * add edges to nodes in callers of n
528        * 
529        * @param n
530        */
531       private void addCallerEdges(CGNode n) {
532         final int num = cg.getNumber(n);
533         if (!cgNodesWithCallerEdges.contains(num)) {
534           cgNodesWithCallerEdges.add(num);
535           ControlFlowGraph<SSAInstruction, T> cfg = getCFG(n);
536           addInterproceduralEdgesForEntryAndExitBlocks(n, cfg);
537         }
538       }

565       /**
566        * add enough nodes and edges to the graph to allow for computing predecessors of N
567        */
568       private void initForPred(BasicBlockInContext<T> N) {
569         CGNode node = getCGNode(N);
570         addIntraproceduralNodesAndEdgesForCGNodeIfNeeded(node);
571         int num = g.getNumber(N);
572         if (!addedPreds.contains(num)) {
573           addedPreds.add(num);
574           if (N.getDelegate().isEntryBlock()) {
575             addCallerEdges(node);
576           }
577           if (isReturn(N)) {
578             addCalleeEdgesForReturn(node, N);
579           }
580         }
581       }

583       /**
584        * add enough nodes and edges to the graph to allow for computing successors of N
585        */
586       private void initForSucc(BasicBlockInContext<T> N) {
587         CGNode node = getCGNode(N);
588         addIntraproceduralNodesAndEdgesForCGNodeIfNeeded(node);
589         int num = g.getNumber(N);
590         if (!addedSuccs.contains(num)) {
591           addedSuccs.add(num);
592           if (N.getDelegate().isExitBlock()) {
593             addCallerEdges(node);
594           }
595           if (hasCall(N)) {
596             addCalleeEdgesForCall(node, N);
597           }
598         }
599       }

672        * @return true iff basic block B ends in a call instuction
673        */
674       protected boolean hasCall(BasicBlockInContext<T> B, ControlFlowGraph<SSAInstruction, T> cfg) {
675         SSAInstruction[] statements = cfg.getInstructions();
676     
677         int lastIndex = B.getLastInstructionIndex();
678         if (lastIndex >= 0) {
679     
680           if (statements.length <= lastIndex) {
681             System.err.println(statements.length);
682             System.err.println(cfg);
683             assert lastIndex < statements.length : "bad BB " + B + " and CFG for " + getCGNode(B);
684           }
685           SSAInstruction last = statements[lastIndex];
686           return (last instanceof SSAAbstractInvokeInstruction);
687         } else {

713       /**
714        * get the {@link CallSiteReference} corresponding to the last instruction in B (assumed to be a call)
715        */
716       protected CallSiteReference getCallSiteForCallBlock(IBasicBlock<SSAInstruction> B, ControlFlowGraph<SSAInstruction, T> cfg) {
717         SSAInstruction[] statements = cfg.getInstructions();
718         SSAAbstractInvokeInstruction call = (SSAAbstractInvokeInstruction) statements[B.getLastInstructionIndex()];
719         int pc = cfg.getProgramCounter(B.getLastInstructionIndex());
720         CallSiteReference site = call.getCallSite();
721         assert site.getProgramCounter() == pc;
722         return site;
723       }

735       @Override
736       public boolean hasEdge(BasicBlockInContext<T> src, BasicBlockInContext<T> dst) {
737         if (!addedSuccs.contains(getNumber(src))) {
738           if (!src.getNode().equals(dst.getNode())) {
739             if (src.getDelegate().isExitBlock()) {
740               // checking for an exit to return edge
741               CGNode callee = src.getNode();
742               if (!cgNodesWithCallerEdges.contains(cg.getNumber(callee))) {
743                 CGNode caller = dst.getNode();
744                 T exitBlock = src.getDelegate();
745                 T entryBlock = getCFG(callee).entry();
746                 addEntryAndExitEdgesToCaller(callee, entryBlock, exitBlock, caller);
747               }
748             } else if (hasCall(src) && dst.getDelegate().isEntryBlock()) {
749               // checking for a call to entry edge
750               CGNode callee = dst.getNode();
751               if (!cgNodesWithCallerEdges.contains(cg.getNumber(callee))) {
752                 CGNode caller = src.getNode();
753                 T entryBlock = dst.getDelegate();
754                 T exitBlock = getCFG(callee).exit();
755                 addEntryAndExitEdgesToCaller(callee, entryBlock, exitBlock, caller);
756               }
757             }
758           } else {
759             // if it exists, edge must be intraprocedural
760             addIntraproceduralNodesAndEdgesForCGNodeIfNeeded(src.getNode());
761           }
762         }
763         addedSuccs.add(getNumber(src));
764         return g.hasEdge(src, dst);
765       }

824       /**
825        * @param callBlock node in the IPCFG that ends in a call
826        * @return the nodes that are return sites for this call.
827        * @throws IllegalArgumentException if bb is null
828        */
829       public Iterator<BasicBlockInContext<T>> getReturnSites(BasicBlockInContext<T> callBlock) {
830         if (callBlock == null) {
831           throw new IllegalArgumentException("bb is null");
832         }
833         final CGNode node = callBlock.getNode();
834     
835         // a successor node is a return site if it is in the same
836         // procedure, and is not the entry() node.
837         Predicate isReturn = new Predicate() {
838           @Override public boolean test(Object o) {
839             BasicBlockInContext other = (BasicBlockInContext) o;
840             return !other.isEntryBlock() && node.equals(other.getNode());
841           }
842         };
843         return new FilterIterator<BasicBlockInContext<T>>(getSuccNodes(callBlock), isReturn);
844       }

846       /**
847        * get the basic blocks which are call sites that may call callee and return to returnBlock if callee is null, answer return sites
848        * for which no callee was found.
849        */
850       public Iterator<BasicBlockInContext<T>> getCallSites(BasicBlockInContext<T> returnBlock, final CGNode callee) {
851         if (returnBlock == null) {
852           throw new IllegalArgumentException("bb is null");
853         }
854         final ControlFlowGraph<SSAInstruction, T> cfg = getCFG(returnBlock);
855         Iterator<? extends T> it = cfg.getPredNodes(returnBlock.getDelegate());
856         final CGNode node = returnBlock.getNode();
857     
858         Predicate<? extends T> dispatchFilter = new Predicate<T>() {
859           @Override public boolean test(T callBlock) {
860             BasicBlockInContext<T> bb = new BasicBlockInContext<T>(node, callBlock);
861             if (!hasCall(bb, cfg)) {
862               return false;
863             }
864             if (callee != null) {
865               return getCallTargets(bb).contains(callee);
866             } else {
867               return getCallTargets(bb).isEmpty();
868             }
869           }
870         };
871         it = new FilterIterator<T>(it, dispatchFilter);
872     
873         Function<T, BasicBlockInContext<T>> toContext = new Function<T, BasicBlockInContext<T>>() {
874           @Override
875           public BasicBlockInContext<T> apply(T object) {
876             T b = object;
877             return new BasicBlockInContext<T>(node, b);
878           }
879         };
880         MapIterator<T, BasicBlockInContext<T>> m = new MapIterator<T, BasicBlockInContext<T>>(it, toContext);
881         return new FilterIterator<BasicBlockInContext<T>>(m, isCall);
882       }
883     
884       private final Predicate<BasicBlockInContext<T>> isCall = new Predicate<BasicBlockInContext<T>>() {
885         @Override public boolean test(BasicBlockInContext<T> o) {
886           return hasCall(o);
887         }
888       };

890       public boolean isReturn(BasicBlockInContext<T> bb) throws IllegalArgumentException {
891         if (bb == null) {
892           throw new IllegalArgumentException("bb == null");
893         }
894         ControlFlowGraph<SSAInstruction, T> cfg = getCFG(bb);
895         for (Iterator<? extends T> it = cfg.getPredNodes(bb.getDelegate()); it.hasNext();) {
896           T b = it.next();
897           if (hasCall(new BasicBlockInContext<T>(bb.getNode(), b))) {
898             return true;
899           }
900         }
901         return false;
902       }
