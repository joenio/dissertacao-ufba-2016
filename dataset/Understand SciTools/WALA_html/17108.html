70     /**
71      *  Instantiates certain android-types differently.
72      *
73      *  For example instantiating an android.content.Context would pull in all Android-components in
74      *  scope resulting in a massivly overapproximated model.
75      *
76      *  @author Tobias Blaschke <code@tobiasblaschke.de>
77      */

82         public SpecializedInstantiator(final VolatileMethodSummary body, final TypeSafeInstructionFactory instructionFactory,
83                 final SSAValueManager pm, final IClassHierarchy cha, final MethodReference scope, final AnalysisScope analysisScope,
84                 final IInstantiator parent) {
85             super(body, instructionFactory, pm, cha, scope, analysisScope, 100);
86             this.parent = parent;
87         }

89         /**
90          *  Creates a new instance of type calling all that's necessary.
91          *
92          *  If T is a class-type all its constructors are searched for the one found best suited (takes the least arguments, ...).
93          *  New instances are created for all parameters, then the constructor is called.
94          *
95          *  If T represents multiple types (is an interface, abstract class, ...) _all_ implementors of that type are instantiated
96          *  After that they get Phi-ed together.
97          *
98          *  If T is an array-type a new array of length 1 is generated.
99          *
100          *  @todo   Do we want to mix in REUSE-Parameters?
101          */
102         public SSAValue createInstance(final TypeReference T, final boolean asManaged, VariableKey key, Set<? extends SSAValue> seen) {
103             return createInstance(T, asManaged, key, seen, 0);
104         }

106         /* package private */ SSAValue createInstance(final TypeReference T, final boolean asManaged, VariableKey key, Set<? extends SSAValue> seen, int currentDepth) {
107             if (seen == null) {
108                 
109                 seen = new HashSet<SSAValue>();
110             }
111            
112             if (currentDepth > this.maxDepth) {
113                 final SSAValue instance = this.pm.getUnmanaged(T, key);
114                 instance.setAssigned();
115                 return instance;
116             }
117     
118     //        final IClass klass = this.cha.lookupClass(T);
119             { // fetch new value
120                 if (asManaged) {
121                     if (key == null) {
122                         throw new IllegalArgumentException("A managed variable needs a key - null given.");
123                     }
124                 } else {
125                     if (key == null) {
126                         key = new UniqueKey();
127                     }
128                 }
129             }
130     
131             assert(understands(T));
132     
133             if (T.equals(AndroidTypes.Context)) {
134                 return createContext(T, asManaged, key, seen);
135             }
136     
137             if (T.equals(AndroidTypes.ContextWrapper)) {
138                 return createContextWrapper(T, asManaged, key, seen);
139             }
140     
141             return null;
142         }

159         /**
160          *  Creates a new instance of android/content/Context.
161          */
162         public SSAValue createContext(final TypeReference T, final boolean asManaged, VariableKey key, Set<? extends SSAValue> seen) {
163             final List<SSAValue> appComponents = new ArrayList<SSAValue>();
164             {
165                 // TODO: Can we create a tighter conterxt?
166                 // TODO: Force an Application-Context?
167     
168                 if (AndroidEntryPointManager.MANAGER.doFlatComponents()) {
169                     final AndroidModelClass mClass = AndroidModelClass.getInstance(cha);
170     
171                     // At a given time context is expected to be only of one component already seen.
172                     // If it's seen there is a field in AndroidModelClass.
173                     for (final IField f : mClass.getAllFields()) {
174                         assert(f.isStatic()) : "All fields of AndroidModelClass are expected to be static! " + f + " is not.";
175                         
176                         final TypeReference fdType = f.getReference().getFieldType();
177                         { // Test assignable
178                             if (! ParameterAccessor.isAssignable(fdType, T, cha) ) {
179                                 assert(false) : "Unexpected but not fatal - remove assertion if this happens";
180                                 continue;
181                             }
182                         }
183                                                                                                                
184                         final VariableKey iKey = new SSAValue.TypeKey(fdType.getName());
185                         final SSAValue instance;
186                         if (this.pm.isSeen(iKey)) {
187                             instance = this.pm.getCurrent(iKey);
188                         } else {
189                             final int pc = this.body.getNextProgramCounter();
190                             final VariableKey subKey = new SSAValue.WeaklyNamedKey(fdType.getName(), "ctx" + fdType.getName().getClassName().toString());
191                             instance = this.pm.getUnallocated(fdType, subKey);
192                             final SSAInstruction getInst = instructionFactory.GetInstruction(pc, instance, f.getReference());
193                             this.body.addStatement(getInst);
194                             this.pm.setAllocation(instance, getInst);
195                         }
196     
197                         appComponents.add(instance);
198                     }
199                 } else {
200                     for (TypeReference component : AndroidEntryPointManager.MANAGER.getComponents()) {
201                         final VariableKey iKey = new SSAValue.TypeKey(component.getName());
202     
203                         if (this.pm.isSeen(iKey)) {
204                             final SSAValue instance;
205                             instance = this.pm.getCurrent(iKey);
206                             assert (instance.getNumber() > 0);
207                             appComponents.add(instance);
208                         }
209                     }
210                 }
211             }
212     
213             final SSAValue instance;
214             if ( appComponents.size() == 1) {
215                 instance = appComponents.get(0);
216             } else if ( appComponents.size() > 0) {
217                 { // Phi them together
218                     final int pc = this.body.getNextProgramCounter();
219                     instance = this.pm.getFree(T, key);
220                     assert (pc > 0);
221                     assert (instance.getNumber() > 0);
222                     final SSAInstruction phi = instructionFactory.PhiInstruction(pc, instance, appComponents);
223                     this.body.addStatement(phi);
224                     this.pm.setPhi(instance, phi);
225                 }
226             } else {
227                 instance = this.pm.getUnmanaged(T, key);
228                 this.body.addConstant(instance.getNumber(), new ConstantValue(null));
229                 instance.setAssigned();
230             }
231     
232             return instance;
233         }

236         public SSAValue createContextWrapper(final TypeReference T, final boolean asManaged, VariableKey key, Set<? extends SSAValue> seen) {
237             final VariableKey contextKey = new SSAValue.TypeKey(AndroidTypes.ContextName);
238             final SSAValue context;
239             {
240                 if (this.pm.isSeen(contextKey)) {
241                     context = this.pm.getCurrent(contextKey);
242                 } else {
243                     context = createContext(AndroidTypes.Context, true, contextKey, seen);
244                 }
245             }
246     
247             final SSAValue instance = this.pm.getUnallocated(T, key);
248             {
249                 // call: ContextWrapper(Context base)
250                 final MethodReference ctor = MethodReference.findOrCreate(T, MethodReference.initAtom, 
251                         Descriptor.findOrCreate(new TypeName[] { AndroidTypes.ContextName }, TypeReference.VoidName));
252                 final List<SSAValue> params = new ArrayList<SSAValue>();
253                 params.add(context);
254                 addCallCtor(instance, ctor, params);
255             }
256     
257             return instance;
258         }

261         /**
262          *  Satisfy the interface.
263          */
264         public int createInstance(TypeReference type, Object... instantiatorArgs) {
265             // public SSAValue createInstance(final TypeReference T, final boolean asManaged, VariableKey key, Set<SSAValue> seen) {
266             if (! (instantiatorArgs[0] instanceof Boolean)) {
267                 throw new IllegalArgumentException("Argument 0 to createInstance has to be boolean.");
268             }
269             if (! ((instantiatorArgs[1] == null) || (instantiatorArgs[1] instanceof VariableKey))) {
270                 throw new IllegalArgumentException("Argument 1 to createInstance has to be null or an instance of VariableKey"); 
271             }
272             if (! ((instantiatorArgs[2] == null) || (instantiatorArgs[2] instanceof Set))) {
273                 throw new IllegalArgumentException("Argument 2 to createInstance has to be null or an instance of Set<? extends SSAValue>, " +
274                         "got: " + instantiatorArgs[2].getClass()); 
275             }
276             final int currentDepth;
277             {
278                 if (instantiatorArgs.length == 4) {
279                     currentDepth = (Integer) instantiatorArgs[3];
280                 } else {
281                     currentDepth = 0;
282                 }
283             }
284             if (instantiatorArgs[2] != null) {
285                 final Set seen = (Set) instantiatorArgs[2];
286                 if (! seen.isEmpty()) {
287                     final Object o = seen.iterator().next();
288                     if (! (o instanceof SSAValue)) {
289                         throw new IllegalArgumentException("Argument 2 to createInstance has to be null or an instance of Set<? extends SSAValue>, " +
290                                 "got Set<" + o.getClass() + ">");
291                     }
292                 }
293             }
294     
295             return createInstance(type, (Boolean) instantiatorArgs[0], (VariableKey) instantiatorArgs[1], (Set<? extends SSAValue>) 
296                     instantiatorArgs[2], currentDepth).getNumber();
297         }
