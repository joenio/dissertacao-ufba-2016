73       public static class EnclosingObjectReferenceKey extends AbstractFieldPointerKey {
74         private final IClass outer;
75     
76         public EnclosingObjectReferenceKey(InstanceKey inner, IClass outer) {
77           super(inner);
78           this.outer = outer;
79         }
80     
81         @Override
82         public int hashCode() {
83           return getInstanceKey().hashCode() * outer.hashCode();
84         }
85     
86         @Override
87         public boolean equals(Object o) {
88           return (o instanceof EnclosingObjectReferenceKey) && ((EnclosingObjectReferenceKey) o).outer.equals(outer)
89               && ((EnclosingObjectReferenceKey) o).getInstanceKey().equals(getInstanceKey());
90         }
91       }

145       protected static class AstJavaConstraintVisitor extends AstConstraintVisitor implements AstJavaInstructionVisitor {
146     
147         public AstJavaConstraintVisitor(
148     			     AstSSAPropagationCallGraphBuilder builder,
149     			     CGNode node) 
150         {
151           super(builder, node);
152         }
153     
154         /**
155          * For each of objKey's instance keys ik, adds the constraint lvalKey = EORK(ik,cls),
156          * where EORK(ik,cls) will be made equivalent to the actual enclosing class by
157          * the handleNew() function below.
158          * @param lvalKey
159          * @param cls
160          * @param objKey
161          */
162         private void handleEnclosingObject(final PointerKey lvalKey, final IClass cls, final PointerKey objKey) {
163           SymbolTable symtab = ir.getSymbolTable();
164           int objVal;
165           if (objKey instanceof LocalPointerKey) {
166             objVal = ((LocalPointerKey) objKey).getValueNumber();
167           } else {
168             objVal = 0;
169           }
170     
171           if (objVal > 0 && contentsAreInvariant(symtab, du, objVal)) {
172             system.recordImplicitPointsToSet(objKey);
173     
174             InstanceKey[] objs = getInvariantContents(objVal);
175     
176             for (int i = 0; i < objs.length; i++) {
177               PointerKey enclosing = new EnclosingObjectReferenceKey(objs[i], cls);
178               system.newConstraint(lvalKey, assignOperator, enclosing);
179             }
180     
181           } else {
182             system.newSideEffect(new UnaryOperator<PointsToSetVariable>() {
183               @Override
184               public byte evaluate(PointsToSetVariable lhs, PointsToSetVariable rhs) {
185                 IntSetVariable tv = rhs;
186                 if (tv.getValue() != null) {
187                   tv.getValue().foreach(new IntSetAction() {
188                     @Override
189                     public void act(int ptr) {
190                       InstanceKey iKey = system.getInstanceKey(ptr);
191                       PointerKey enclosing = new EnclosingObjectReferenceKey(iKey, cls);
192                       system.newConstraint(lvalKey, assignOperator, enclosing);
193                     }
194                   });
195                 }
196                 return NOT_CHANGED;
197               }
198     
199               @Override
200               public int hashCode() {
201                 return System.identityHashCode(this);
202               }
203     
204               @Override
205               public boolean equals(Object o) {
206                 return o == this;
207               }
208     
209               @Override
210               public String toString() {
211                 return "enclosing objects of " + objKey;
212               }
213             }, objKey);
214           }
215         }
216     
217         @Override
218         public void visitEnclosingObjectReference(EnclosingObjectReference inst) {
219           PointerKey lvalKey = getPointerKeyForLocal(inst.getDef());
220           PointerKey objKey = getPointerKeyForLocal(1);
221           IClass cls = getClassHierarchy().lookupClass(inst.getEnclosingType());
222           handleEnclosingObject(lvalKey, cls, objKey);
223         }
224     
225         @Override
226         public void visitNew(SSANewInstruction instruction) {
227           super.visitNew(instruction);
228           InstanceKey iKey = getInstanceKeyForAllocation(instruction.getNewSite());
229     
230           if (iKey != null) {
231             IClass klass = iKey.getConcreteType();
232     
233             // in the case of a AstJavaNewEnclosingInstruction (a new instruction like outer.new Bla()),
234             // we may need to record the instance keys if the pointer key outer is invariant (and thus implicit) 
235             InstanceKey enclosingInvariantKeys[] = null;
236            
237             if (klass instanceof JavaClass) {
238               IClass enclosingClass = ((JavaClass) klass).getEnclosingClass(); // the immediate enclosing class.
239               if (enclosingClass != null) {
240                 IClass currentCls = node.getMethod().getDeclaringClass();
241                 PointerKey objKey;
242                 
243                 if ( instruction instanceof AstJavaNewEnclosingInstruction ) {
244                   
245     
246                   int enclosingVal = ((AstJavaNewEnclosingInstruction) instruction).getEnclosing();
247                   SymbolTable symtab = ir.getSymbolTable();
248     
249                   // pk 'outer' is invariant, which means it's implicit, so can't add a constraint with the pointer key.
250                   // we should just add constraints directly to the instance keys (below) 
251                   if ( contentsAreInvariant(symtab, du, enclosingVal) )
252                     enclosingInvariantKeys = getInvariantContents(enclosingVal);
253                   
254                   // what happens if objKey is implicit but the contents aren't invariant?! (it this possible?) big trouble!
255                   
256                   objKey = getPointerKeyForLocal(enclosingVal);
257                 }
258                 else
259                   objKey = getPointerKeyForLocal(1);
260     
261                 System.err.println(("class is " + klass + ", enclosing is " + enclosingClass + ", method is " + node.getMethod()));
262     
263                 if (node.getMethod().isSynthetic()) {
264                   return;
265                 }
266     
267                 currentCls = enclosingClass;
268                 
269                 PointerKey x = new EnclosingObjectReferenceKey(iKey, currentCls);
270                 if ( enclosingInvariantKeys != null )
271                   for ( InstanceKey obj: enclosingInvariantKeys )
272                     system.newConstraint(x, obj);
273                 else
274                   system.newConstraint(x, assignOperator, objKey);
275     
276                 // If the immediate inclosing class is not a top-level class, we must make EORKs for all enclosing classes up to the top level.
277                 // for instance, if we have "D d = c.new D()", and c is of type A$B$C, methods in D may reference variables and functions from
278                 // A, B, and C. Therefore we must also make the links from EORK(allocsite of d,enc class B) and EORK(allocsite of d,en class A).
279                 // We do this by getting the enclosing class of C and making a link from EORK(d,B) -> EORK(c,B), etc.
280                 currentCls = ((JavaClass) currentCls).getEnclosingClass();
281                 while (currentCls != null) { 
282                   x = new EnclosingObjectReferenceKey(iKey, currentCls); // make EORK(d,B), EORK(d,A), etc.
283                   handleEnclosingObject(x, currentCls, objKey);
284                   // objKey is the pointer key representing the immediate inner class.
285                   // handleEnclosingObject finds x's instance keys and for each one "ik" links x to EORK(ik,currentCls)
286                   // thus, for currentCls=B, it will find the allocation site of c and make a link from EORK(d,B) to EORK(c,B)
287                   
288                   currentCls = ((JavaClass) currentCls).getEnclosingClass();
289                 }
290                 
291               }
292             }
293           }
294         }

307       @Override
308       protected boolean sameMethod(CGNode opNode, String definingMethod) {
309         MethodReference reference = opNode.getMethod().getReference();
310         String selector = reference.getSelector().toString();
311         String containingClass = reference.getDeclaringClass().getName().toString();
312         return definingMethod.equals(containingClass + "/" + selector);
313       }
