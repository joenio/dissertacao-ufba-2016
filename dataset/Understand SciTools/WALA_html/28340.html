11     package com.ibm.wala.cast.js.callgraph.fieldbased;
12     
13     import com.ibm.wala.cast.ir.ssa.AstGlobalRead;
14     import com.ibm.wala.cast.js.ssa.JSAbstractInstructionVisitor;
15     import com.ibm.wala.cast.js.ssa.JavaScriptInvoke;
16     import com.ibm.wala.cast.js.types.JavaScriptMethods;
17     import com.ibm.wala.classLoader.IMethod;
18     import com.ibm.wala.ssa.DefUse;
19     import com.ibm.wala.ssa.SSAInstruction;
20     import com.ibm.wala.ssa.SymbolTable;
21     
22     /**
23      * A {@link JSAbstractInstructionVisitor} that is used to only visit instructions of a single method.
24      * 
25      * @author mschaefer
26      *
27      */
28     public class JSMethodInstructionVisitor extends JSAbstractInstructionVisitor {
29     
30     	protected final IMethod method;
31     	protected final SymbolTable symtab;
32     	protected final DefUse du;
33     
34     	public JSMethodInstructionVisitor(IMethod method, SymbolTable symtab, DefUse du) {
35     		this.method = method;
36     		this.symtab = symtab;
37     		this.du = du;
38     	}
39     
40     	/**
41     	 * Determine whether <code>invk</code> corresponds to a function declaration or function expression.
42     	 * 
43     	 * TODO: A bit hackish. Is there a more principled way to do this?
44     	 */
45     	protected boolean isFunctionConstructorInvoke(JavaScriptInvoke invk) {
46     	    /*
47     	     * Function objects are allocated by explicit constructor invocations like this:
48     	     * 
49     	     *   v8 = global:global Function
50     	     *   v4 = construct v8@2 v6:#L<fullFunctionName> exception:<nd>
51     	     */
52     		if(invk.getDeclaredTarget().equals(JavaScriptMethods.ctorReference)) {
53     			int fn = invk.getFunction();
54     			SSAInstruction fndef = du.getDef(fn);
55     			if(fndef instanceof AstGlobalRead) {
56     				AstGlobalRead agr = (AstGlobalRead)fndef;
57     				if(agr.getGlobalName().equals("global Function")) {
58     					if(invk.getNumberOfParameters() != 2)
59     						return false;
60     					// this may be a genuine use of "new Function()", not a declaration/expression
61     					if(!symtab.isStringConstant(invk.getUse(1)))
62     						return false;
63     					return true;
64     				}
65     			}
66     		}
67     		return false;
68     	}
69     }
