85       /**
86        * create an Atom from utf8[off] of length len
87        * 
88        * @throws IllegalArgumentException if utf8.length <= off
89        */
90       public static Atom findOrCreate(byte utf8[], int off, int len) throws IllegalArgumentException, IllegalArgumentException,
91           IllegalArgumentException {
92     
93         if (utf8 == null) {
94           throw new IllegalArgumentException("utf8 == null");
95         }
96         if (len < 0) {
97           throw new IllegalArgumentException("len must be >= 0, " + len);
98         }
99         if (off < 0) {
100           throw new IllegalArgumentException("off must be >= 0, " + off);
101         }
102         if (utf8.length < off + len) {
103           throw new IllegalArgumentException("utf8.length < off + len");
104         }
105         if (off + len < 0) {
106           throw new IllegalArgumentException("off + len is too big: " + off + " + " + len);
107         }
108         byte val[] = new byte[len];
109         for (int i = 0; i < len; ++i) {
110           val[i] = utf8[off++];
111         }
112         return findOrCreate(val);
113     
114       }

116       public static synchronized Atom findOrCreate(byte[] bytes) {
117         if (bytes == null) {
118           throw new IllegalArgumentException("bytes is null");
119         }
120         AtomKey key = new AtomKey(bytes);
121         Atom val = dictionary.get(key);
122         if (val != null) {
123           return val;
124         }
125         val = new Atom(key);
126         dictionary.put(key, val);
127         return val;
128       }

130       public static synchronized Atom findOrCreate(ImmutableByteArray b) {
131         if (b == null) {
132           throw new IllegalArgumentException("b is null");
133         }
134         return findOrCreate(b.b);
135       }

138         if (b == null) {
139           throw new IllegalArgumentException("b is null");
140         }
141         return findOrCreate(b.b, start, length);

173       public final boolean startsWith(Atom start) {
174           assert (start != null);
175     
176           for (int i = 0; i < start.val.length; ++i) {
177               if (val[i] != start.val[i])
178                   return false;
179           }

181           return true;
182       }

184       /**
185        * Return array descriptor corresponding to "this" array-element descriptor. this: array-element descriptor - something like "I"
186        * or "Ljava/lang/Object;"
187        * 
188        * @return array descriptor - something like "[I" or "[Ljava/lang/Object;"
189        */
190       public final Atom arrayDescriptorFromElementDescriptor() {
191         byte sig[] = new byte[1 + val.length];
192         sig[0] = (byte) '[';
193         for (int i = 0, n = val.length; i < n; ++i)
194           sig[i + 1] = val[i];
195         return findOrCreate(sig);
196       }

251       /**
252        * Parse "this" array descriptor to obtain descriptor for array's element type. this: array descriptor - something like "[I"
253        * 
254        * @return array element descriptor - something like "I"
255        */
256       public final Atom parseForArrayElementDescriptor() throws IllegalArgumentException {
257         if (val.length == 0) {
258           throw new IllegalArgumentException("empty atom is not an array");
259         }
260         return findOrCreate(val, 1, val.length - 1);
261       }
262     
263       /**
264        * Parse "this" array descriptor to obtain number of dimensions in corresponding array type. this: descriptor - something like
265        * "[Ljava/lang/String;" or "[[I"
266        * 
267        * @return dimensionality - something like "1" or "2"
268        * @throws IllegalStateException if this Atom does not represent an array
269        */
270       public final int parseForArrayDimensionality() throws IllegalArgumentException {
271         if (val.length == 0) {
272           throw new IllegalArgumentException("empty atom is not an array");
273         }
274         try {
275           for (int i = 0;; ++i) {
276             if (val[i] != '[') {
277               return i;
278             }
279           }
280         } catch (ArrayIndexOutOfBoundsException e) {
281           throw new IllegalStateException("not an array: " + this);
282         }
283     
284       }
285     
286       /**
287        * Return the innermost element type reference for an array
288        * 
289        * @throws IllegalStateException if this Atom does not represent an array descriptor
290        */
291       public final Atom parseForInnermostArrayElementDescriptor() throws IllegalArgumentException {
292         if (val.length == 0) {
293           throw new IllegalArgumentException("empty atom is not an array");
294         }
295         try {
296           int i = 0;
297           while (val[i] == '[') {
298             i++;
299           }
300           return findOrCreate(val, i, val.length - i);
301         } catch (ArrayIndexOutOfBoundsException e) {
302           throw new IllegalStateException("not an array: " + this);
303         }
304       }

306       /**
307        * key for the dictionary.
308        */
309       private final static class AtomKey {
310         /**
311          * The utf8 value this atom key represents
312          */
313         private final byte val[];
314     
315         /**
316          * Cached hash code for this atom key.
317          */
318         private final int hash;
319     
320         /**
321          * Create atom from given utf8 sequence.
322          */
323         private AtomKey(byte utf8[]) {
324           int tmp = 99989;
325           for (int i = utf8.length; --i >= 0;) {
326             tmp = 99991 * tmp + utf8[i];
327           }
328           this.val = utf8;
329           this.hash = tmp;
330         }

332         /**
333          * @see java.lang.Object#equals(Object)
334          */
335         @Override
336         public final boolean equals(Object other) {
337     
338           assert (other != null && this.getClass().equals(other.getClass()));
339           if (this == other) {
340             return true;
341           }
342     
343           AtomKey that = (AtomKey) other;
344           if (hash != that.hash)
345             return false;
346           if (val.length != that.val.length)
347             return false;
348           for (int i = 0; i < val.length; i++) {
349             if (val[i] != that.val[i])
350               return false;
351           }
352     
353           return true;
354     
355         }

431       private static Atom concat(byte c, byte[] bs) {
432         byte[] val = new byte[bs.length + 1];
433         val[0] = c;
434         System.arraycopy(bs, 0, val, 1, bs.length);
435         return findOrCreate(val);
436       }

439         if (b == null) {
440           throw new IllegalArgumentException("b is null");
441         }
442         return concat(c, b.b);
443       }

445       public static Atom concat(Atom ma, Atom mb) {
446          if ((ma == null ) || (mb == null)) {
447           throw new IllegalArgumentException("argument may not be null!");
448         }
449     
450         byte[] val = new byte[ma.val.length + mb.val.length];
451      
452         System.arraycopy(ma.val, 0, val, 0, ma.val.length);
453         System.arraycopy(mb.val, 0, val, ma.val.length, mb.val.length);
454     
455         return findOrCreate(val);
456       }

459         if (b == null) {

463           return false;
464         }
