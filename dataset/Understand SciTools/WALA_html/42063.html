11     package com.ibm.wala.shrikeCT;
12     
13     /**
14      * This class reads LocalVariableTable attributes.
15      * 
16      * Instead of constructing a LocalVariableTable directly, consider just calling LocalVariableTable.makeVarMap for convenient access
17      * to aggregate local variable data from all the LocalVariableTable attributes for a given Code.
18      */
19     public final class LocalVariableTableReader extends AttributeReader {
20       public LocalVariableTableReader(ClassReader.AttrIterator iter) throws InvalidClassFileException {
21         super(iter, "LocalVariableTable");
22     
23         int offset = attr + 6;
24         checkSize(offset, 2);
25         int count = cr.getUShort(offset);
26         offset += 2;
27         checkSize(offset, count * 10);
28       }
29     
30       /**
31        * @return the raw line number table data, a flattened sequence of (startPC, PClength, nameIndex, typeIndex, var) tuples
32        */
33       public int[] getRawTable() {
34         int count = cr.getUShort(attr + 6);
35         int[] r = new int[count * 5];
36         int offset = attr + 8;
37         for (int i = 0; i < r.length; i++) {
38           r[i] = cr.getUShort(offset);
39           offset += 2;
40         }
41         return r;
42       }
43     
44       private static int[] makeVarVector(int[] curVector, int varIndex, int nameIndex, int typeIndex) {
45         int[] newVector;
46         if (curVector == null) {
47           newVector = new int[(varIndex + 1) * 2];
48         } else {
49           newVector = new int[Math.max(curVector.length, (varIndex + 1) * 2)];
50           System.arraycopy(curVector, 0, newVector, 0, curVector.length);
51         }
52         newVector[varIndex * 2] = nameIndex;
53         newVector[varIndex * 2 + 1] = typeIndex;
54         return newVector;
55       }
56     
57       /**
58        * @return an array mapping bytecode offsets to arrays representing the local variable maps for each offset; a local variable map
59        *         is represented as an array of localVars*2 elements, containing a pair (nameIndex, typeIndex) for each local variable; a
60        *         pair (0,0) indicates there is no information for that local variable at that offset
61        */
62       public static int[][] makeVarMap(CodeReader code) throws InvalidClassFileException, IllegalArgumentException {
63         if (code == null) {
64           throw new IllegalArgumentException();
65         }
66     
67         int[][] r = null;
68         ClassReader cr = code.getClassReader();
69     
70         ClassReader.AttrIterator iter = new ClassReader.AttrIterator();
71         code.initAttributeIterator(iter);
72         for (; iter.isValid(); iter.advance()) {
73           if (iter.getName().equals("LocalVariableTable")) {
74             if (r == null) {
75               r = new int[code.getBytecodeLength()][];
76             }
77     
78             // check length
79             new LocalVariableTableReader(iter);
80             int attr = iter.getRawOffset();
81             int count = cr.getUShort(attr + 6);
82             int offset = attr + 8;
83             for (int j = 0; j < count; j++) {
84               int startPC = cr.getUShort(offset);
85               int length = cr.getUShort(offset + 2);
86               int nameIndex = cr.getUShort(offset + 4);
87               int typeIndex = cr.getUShort(offset + 6);
88               int varIndex = cr.getUShort(offset + 8);
89               offset += 10;
90     
91               if (varIndex < 0) {
92                 throw new InvalidClassFileException(offset, "Invalid variable index " + varIndex + " in LocalVariableTable");
93               } else if (startPC < 0) {
94                 throw new InvalidClassFileException(offset, "Invalid startPC " + startPC + " in LocalVariableTable");
95               } else if (startPC + length > r.length) {
96                 throw new InvalidClassFileException(offset, "Invalid startPC+length " + (startPC + length) + " > " + r.length
97                     + " in LocalVariableTable");
98               }
99     
100               for (int k = startPC; k < startPC + length; k++) {
101                 int[] newVector = makeVarVector(r[k], varIndex, nameIndex, typeIndex);
102                 r[k] = newVector;
103               }
104             }
105           }
106         }
107     
108         return r;
109       }
110     }
