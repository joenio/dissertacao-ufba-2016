11     package com.ibm.wala.ssa;
12     
13     /**
14      * An instruction which unconditionally throws an exception
15      */
16     public abstract class SSAThrowInstruction extends SSAAbstractThrowInstruction {
17     
18       protected SSAThrowInstruction(int iindex, int exception) {
19         super(iindex, exception);
20       }
21     
22       /* 
23        * @see com.ibm.wala.ssa.SSAInstruction#copyForSSA(com.ibm.wala.ssa.SSAInstructionFactory, int[], int[])
24        */
25       @Override
26       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) throws IllegalArgumentException {
27         if (uses != null && uses.length != 1) {
28           throw new IllegalArgumentException("if non-null, uses.length must be 1");
29         }
30         return insts.ThrowInstruction(iindex, uses == null ? getException() : uses[0]);
31       }
32     
33       /**
34        * @see com.ibm.wala.ssa.SSAInstruction#visit(IVisitor)
35        */
36       @Override
37       public void visit(IVisitor v) throws NullPointerException {
38         v.visitThrow(this);
39       }
40     }
