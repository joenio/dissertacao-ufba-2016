54       public static String findJarFiles(String[] directories) throws WalaException {
55         Collection<String> result = HashSetFactory.make();
56         for (int i = 0; i < directories.length; i++) {
57           for (Iterator<File> it = FileUtil.listFiles(directories[i], ".*\\.jar", true).iterator(); it.hasNext();) {
58             File f = (File) it.next();
59             result.add(f.getAbsolutePath());
60           }
61         }
62         return composeString(result);
63       }
64       
65       private static String composeString(Collection<String> s) {
66         StringBuffer result = new StringBuffer();
67         Iterator<String> it = s.iterator();
68         for (int i = 0; i < s.size() - 1; i++) {
69           result.append(it.next());
70           result.append(File.pathSeparator);
71         }
72         if (it.hasNext()) {
73           result.append(it.next());
74         }
75         return result.toString();
76       }

104       /**
105        * @param appJar something like "c:/temp/testdata/java_cup.jar"
106        * @throws CancelException
107        * @throws IllegalArgumentException
108        */
109       public static Process run(String appJar, String exclusionFile) throws IllegalArgumentException, CancelException {
110         try {
111           Graph<CGNode> g = buildPrunedCallGraph(appJar, (new FileProvider()).getFile(exclusionFile));
112     
113           Properties p = null;
114           try {
115             p = WalaExamplesProperties.loadProperties();
116             p.putAll(WalaProperties.loadProperties());
117           } catch (WalaException e) {
118             e.printStackTrace();
119             Assertions.UNREACHABLE();
120           }
121           String pdfFile = p.getProperty(WalaProperties.OUTPUT_DIR) + File.separatorChar + PDF_FILE;
122     
123           String dotExe = p.getProperty(WalaExamplesProperties.DOT_EXE);
124           DotUtil.dotify(g, null, PDFTypeHierarchy.DOT_FILE, pdfFile, dotExe);
125     
126           String gvExe = p.getProperty(WalaExamplesProperties.PDFVIEW_EXE);
127           return PDFViewUtil.launchPDFView(pdfFile, gvExe);
128     
129         } catch (WalaException e) {
130           e.printStackTrace();
131           return null;
132         } catch (IOException e) {
133           e.printStackTrace();
134           return null;
135         }
136       }

138       /**
139        * @param appJar something like "c:/temp/testdata/java_cup.jar"
140        * @return a call graph
141        * @throws CancelException
142        * @throws IllegalArgumentException
143        * @throws IOException 
144        */
145       public static Graph<CGNode> buildPrunedCallGraph(String appJar, File exclusionFile) throws WalaException,
146           IllegalArgumentException, CancelException, IOException {
147         AnalysisScope scope = AnalysisScopeReader.makeJavaBinaryAnalysisScope(appJar, exclusionFile != null ? exclusionFile : new File(
148             CallGraphTestUtil.REGRESSION_EXCLUSIONS));
149     
150         ClassHierarchy cha = ClassHierarchy.make(scope);
151     
152         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha);
153         AnalysisOptions options = new AnalysisOptions(scope, entrypoints);
154     
155         // //
156         // build the call graph
157         // //
158         com.ibm.wala.ipa.callgraph.CallGraphBuilder builder = Util.makeZeroCFABuilder(options, new AnalysisCache(), cha, scope);
159         CallGraph cg = builder.makeCallGraph(options, null);
160     
161         System.err.println(CallGraphStats.getStats(cg));
162     
163         Graph<CGNode> g = pruneForAppLoader(cg);
164     
165         return g;
166       }

168       public static Graph<CGNode> pruneForAppLoader(CallGraph g) throws WalaException {
169         return PDFTypeHierarchy.pruneGraph(g, new ApplicationLoaderFilter());

188       /**
189        * A filter that accepts WALA objects that "belong" to the application loader.
190        * 
191        * Currently supported WALA types include
192        * <ul>
193        * <li> {@link CGNode}
194        * <li> {@link LocalPointerKey}
195        * </ul>
196        */
197       private static class ApplicationLoaderFilter extends Predicate<CGNode> {
198     
199         @Override public boolean test(CGNode o) {
200           if (o instanceof CGNode) {
201             CGNode n = (CGNode) o;
202             return n.getMethod().getDeclaringClass().getClassLoader().getReference().equals(ClassLoaderReference.Application);
203           } else if (o instanceof LocalPointerKey) {
204             LocalPointerKey l = (LocalPointerKey) o;
205             return test(l.getNode());
206           } else {
207             return false;
208           }
209         }
210       }
