11     package com.ibm.wala.cast.ir.translator;
12     
13     import java.util.Map;
14     
15     import com.ibm.wala.cast.tree.CAst;
16     import com.ibm.wala.cast.tree.CAstControlFlowMap;
17     import com.ibm.wala.cast.tree.CAstNode;
18     import com.ibm.wala.cast.tree.impl.CAstOperator;
19     import com.ibm.wala.cast.tree.rewrite.CAstBasicRewriter;
20     import com.ibm.wala.util.collections.Pair;
21     
22     public abstract class ConstantFoldingRewriter extends CAstBasicRewriter {
23     
24       protected ConstantFoldingRewriter(CAst Ast) {
25         super(Ast, true);
26       }
27     
28       protected abstract Object eval(CAstOperator op, Object lhs, Object rhs);
29       
30       @Override
31       protected CAstNode copyNodes(CAstNode root, 
32           CAstControlFlowMap cfg, 
33           NonCopyingContext context,
34           Map<Pair<CAstNode, NoKey>, CAstNode> nodeMap) 
35       {
36         CAstNode result;
37         if (root.getKind() == CAstNode.BINARY_EXPR) {
38           CAstNode left = copyNodes(root.getChild(1), cfg, context, nodeMap);
39           CAstNode right = copyNodes(root.getChild(2), cfg, context, nodeMap);
40           Object v;
41           if (left.getKind() == CAstNode.CONSTANT && right.getKind() == CAstNode.CONSTANT && (v = eval((CAstOperator) root.getChild(0), left.getValue(), right.getValue())) != null) {
42             result = Ast.makeConstant(v);
43           } else {
44             result = Ast.makeNode(CAstNode.BINARY_EXPR, root.getChild(0), left, right);
45           }
46     
47         } else if (root.getKind() == CAstNode.IF_EXPR || root.getKind() == CAstNode.IF_STMT) {
48           CAstNode expr = copyNodes(root.getChild(0), cfg, context, nodeMap);
49           if (expr.getKind() == CAstNode.CONSTANT && expr.getValue() == Boolean.TRUE) {
50             result = copyNodes(root.getChild(1), cfg, context, nodeMap);
51           } else if (expr.getKind() == CAstNode.CONSTANT && root.getChildCount() > 2 && expr.getValue() == Boolean.FALSE) {
52             result = copyNodes(root.getChild(2), cfg, context, nodeMap);
53           } else {
54             CAstNode then = copyNodes(root.getChild(1), cfg, context, nodeMap);
55             if (root.getChildCount() == 3) {
56               result = Ast.makeNode(root.getKind(), expr, then, copyNodes(root.getChild(2), cfg, context, nodeMap));
57             } else {
58               result = Ast.makeNode(root.getKind(), expr, then);
59             }
60           }
61           
62         } else if (root.getKind() == CAstNode.CONSTANT) {
63           result = Ast.makeConstant(root.getValue());
64     
65         } else if (root.getKind() == CAstNode.OPERATOR) {
66           result = root;
67           
68         } else {
69           CAstNode children[] = new CAstNode[root.getChildCount()];
70           for (int i = 0; i < children.length; i++) {
71             children[i] = copyNodes(root.getChild(i), cfg, context, nodeMap);
72           }
73           for(Object label: cfg.getTargetLabels(root)) {
74             if (label instanceof CAstNode) {
75               copyNodes((CAstNode)label, cfg, context, nodeMap);
76             }
77           }
78           CAstNode copy = Ast.makeNode(root.getKind(), children);
79           result = copy;
80         }
81     
82         nodeMap.put(Pair.make(root, context.key()), result);
83         return result;
84       }
85     
86     }
