11     package com.ibm.wala.util.processes;
12     
13     import java.io.BufferedOutputStream;
14     import java.io.IOException;
15     import java.util.logging.Logger;
16     
17     /**
18      * A generic process launcher
19      */
20     public class BasicLauncher extends Launcher {
21     
22       protected String cmd;
23     
24       public BasicLauncher(boolean captureOutput, boolean captureErr, Logger logger) {
25         super(captureOutput, captureErr, logger);
26       }
27     
28       public String getCmd() {
29         return cmd;
30       }
31     
32       public void setCmd(String newCmd) {
33         cmd = newCmd;
34       }
35     
36       @Override
37       public String toString() {
38         StringBuffer result = new StringBuffer(super.toString());
39         result.append(" (cmd: ");
40         result.append(cmd);
41         return result.toString();
42       }
43     
44       /**
45        * Launch the process and wait until it is finished.  Returns the exit value of the process.
46        */
47       public int launch() throws  IllegalArgumentException, IOException {
48         Process p = spawnProcess(getCmd());
49         Thread d1 = isCaptureErr() ? captureStdErr(p) : drainStdErr(p);
50         Thread d2 = isCaptureOutput() ? captureStdOut(p) : drainStdOut(p);
51         if (getInput() != null) {
52           final BufferedOutputStream input = new BufferedOutputStream(p.getOutputStream());
53           try {
54             input.write(getInput(), 0, getInput().length);
55             input.flush();
56             input.close();
57           } catch (IOException e) {
58             e.printStackTrace();
59             throw new IOException("error priming stdin", e);
60           }
61         }
62         try {
63           d1.join();
64           d2.join();
65         } catch (InterruptedException e) {
66           throw new Error("Internal error", e);
67         }
68         if (isCaptureErr()) {
69           Drainer d = (Drainer) d1;
70           setStdErr(d.getCapture().toByteArray());
71         }
72         if (isCaptureOutput()) {
73           Drainer d = (Drainer) d2;
74           setStdOut(d.getCapture().toByteArray());
75         }
76         return p.exitValue();
77       }
78     }
