205       /**
206        * Copy a constant pool from some ClassReader into this class. This must be done before any entries are allocated in this
207        * ClassWriter's constant pool, and it can only be done once. If and only if this is done, it is safe to copy "raw" fields,
208        * methods and attributes from the ClassReader into this class, because the constant pool references in those fields, methods and
209        * attributes are guaranteed to point to the same constant pool items in this new class.
210        * 
211        * @param cacheEntries records whether to parse the raw constant pool completely so that if new entries are required which are the
212        *          same as entries already in the raw pool, the existing entries in the raw pool are used instead. Setting this to 'true'
213        *          produces smaller constant pools but may slow down performance because the raw pool must be completely parsed
214        */
215       public void setRawCP(ConstantPoolParser cp, boolean cacheEntries) throws InvalidClassFileException, IllegalArgumentException {
216         if (cp == null) {
217           throw new IllegalArgumentException();
218         }
219         if (rawCP != null) {
220           throw new IllegalArgumentException("Cannot set raw constant pool twice");
221         }
222         if (nextCPIndex != 1) {
223           throw new IllegalArgumentException("Cannot set raw constant pool after allocating new entries");
224         }
225         rawCP = cp;
226         nextCPIndex = cp.getItemCount();
227     
228         if (cacheEntries) {
229           for (int i = 1; i < nextCPIndex; i++) {
230             byte t = cp.getItemType(i);
231             switch (t) {
232             case CONSTANT_String:
233               cachedCPEntries.put(new CWStringItem(cp.getCPString(i), CONSTANT_String), new Integer(i));
234               break;
235             case CONSTANT_Class:
236               cachedCPEntries.put(new CWStringItem(cp.getCPClass(i), CONSTANT_Class), new Integer(i));
237               break;
238             case CONSTANT_MethodType:
239               cachedCPEntries.put(new CWStringItem(cp.getCPMethodType(i), CONSTANT_MethodType), new Integer(i));
240               break;
241             case CONSTANT_MethodHandle:
242             case CONSTANT_FieldRef:
243             case CONSTANT_InterfaceMethodRef:
244             case CONSTANT_MethodRef:
245               cachedCPEntries.put(new CWRef(t, cp.getCPRefClass(i), cp.getCPRefName(i), cp.getCPRefType(i)), new Integer(i));
246               break;
247             case CONSTANT_NameAndType:
248               cachedCPEntries.put(new CWNAT(cp.getCPNATName(i), cp.getCPNATType(i)), new Integer(i));
249               break;
250             case CONSTANT_Integer:
251               cachedCPEntries.put(new Integer(cp.getCPInt(i)), new Integer(i));
252               break;
253             case CONSTANT_Float:
254               cachedCPEntries.put(new Float(cp.getCPFloat(i)), new Integer(i));
255               break;
256             case CONSTANT_Long:
257               cachedCPEntries.put(new Long(cp.getCPLong(i)), new Integer(i));
258               break;
259             case CONSTANT_Double:
260               cachedCPEntries.put(new Double(cp.getCPDouble(i)), new Integer(i));
261               break;
262             case CONSTANT_Utf8:
263               cachedCPEntries.put(cp.getCPUtf8(i), new Integer(i));
264               break;
265             }
266           }
267         }
268       }

278       protected int addCPEntry(Object o, int size) {
279         if (cachedCPEntries == null) {
280           throw new IllegalArgumentException("Cannot add a new constant pool entry during makeBytes() processing!");
281         }
282     
283         Integer i = forceAddCPEntries ? null : cachedCPEntries.get(o);
284         if (i != null) {
285           return i.intValue();
286         } else {
287           int index = nextCPIndex;
288           nextCPIndex += size;
289           i = new Integer(index);
290           cachedCPEntries.put(o, i);
291           newCPEntries.add(o);
292           if (nextCPIndex > 0xFFFF) {
293             throw new IllegalArgumentException("Constant pool item count exceeded");
294           }
295           return index;
296         }
297       }

511       /**
512        * An Element is an object that can be serialized into a byte buffer. Serialization via 'copyInto' is performed when the user
513        * calls makeBytes() on the ClassWriter. At this time no new constant pool items can be allocated, so any item indices that need
514        * to be emitted must be allocated earlier.
515        */
516       public static abstract class Element {
517         public Element() {
518         }
519     
520         /**
521          * @return the number of bytes that will be generated.
522          */
523         public abstract int getSize();
524     
525         /**
526          * Copy the bytes into 'buf' at offset 'offset'.
527          * 
528          * @return the number of bytes copies, which must be equal to getSize()
529          */
530         public abstract int copyInto(byte[] buf, int offset);
531       }

533       /**
534        * A RawElement is an Element that is already available as some chunk of a byte buffer.
535        */
536       public static final class RawElement extends Element {
537         final private byte[] buf;
538     
539         final private int offset;
540     
541         final private int len;
542     
543         /**
544          * Create an Element for the 'len' bytes in 'buf' at offset 'offset'.
545          */
546         public RawElement(byte[] buf, int offset, int len) {
547           this.buf = buf;
548           this.offset = offset;
549           this.len = len;
550         }
551     
552         @Override
553         public int getSize() {
554           return len;
555         }
556     
557         @Override
558         public int copyInto(byte[] dest, int destOffset) {
559           System.arraycopy(buf, offset, dest, destOffset, len);
560           return destOffset + len;
561         }
562       }

602       static final class MemberElement extends Element {
603         final private int access;
604     
605         final private int name;
606     
607         final private int type;
608     
609         final private Element[] attributes;
610     
611         public MemberElement(int access, int name, int type, Element[] attributes) {
612           if (access < 0 || access > 0xFFFF) {
613             throw new IllegalArgumentException("Access flags out of range: " + access);
614           }
615           if (name < 1 || name > 0xFFFF) {
616             throw new IllegalArgumentException("Name constant pool index out of range: " + name);
617           }
618           if (type < 1 || type > 0xFFFF) {
619             throw new IllegalArgumentException("Type constant pool index out of range: " + name);
620           }
621           if (attributes == null) {
622             throw new IllegalArgumentException("Atrtributes are null");
623           }
624           if (attributes.length > 0xFFFF) {
625             throw new IllegalArgumentException("Too many attributes: " + attributes.length);
626           }
627     
628           this.access = access;
629           this.name = name;
630           this.type = type;
631           this.attributes = attributes;
632         }
633     
634         @Override
635         public int getSize() {
636           int size = 8;
637           if (attributes != null) {
638             for (int i = 0; i < attributes.length; i++) {
639               size += attributes[i].getSize();
640             }
641           }
642           return size;
643         }
644     
645         @Override
646         public int copyInto(byte[] buf, int offset) {
647           setUShort(buf, offset, access);
648           setUShort(buf, offset + 2, name);
649           setUShort(buf, offset + 4, type);
650           if (attributes != null) {
651             setUShort(buf, offset + 6, attributes.length);
652             offset += 8;
653             for (int i = 0; i < attributes.length; i++) {
654               offset = attributes[i].copyInto(buf, offset);
655             }
656           } else {
657             setUShort(buf, offset + 6, 0);
658             offset += 8;
659           }
660           return offset;
661         }
662       }

709       private int reserveBuf(int size) {
710         if (buf == null) {
711           buf = new byte[size];
712         } else if (bufLen + size > buf.length) {
713           byte[] newBuf = new byte[Math.max(buf.length * 2, bufLen + size)];
714           System.arraycopy(buf, 0, newBuf, 0, bufLen);
715           buf = newBuf;
716         }
717         int offset = bufLen;
718         bufLen += size;
719         return offset;
720       }
721     
722       private void emitElement(Element e) {
723         int size = e.getSize();
724         int offset = reserveBuf(size);
725         int finalOffset = e.copyInto(buf, offset);
726         if (finalOffset - offset != size) {
727           throw new Error("Element failed to output the promised bytes: promised " + size + ", got " + (finalOffset - offset));
728         }
729       }

731       private static final char[] noChars = new char[0];
732     
733       private void emitConstantPool() {
734         if (rawCP != null) {
735           int len = rawCP.getRawSize();
736           int offset = reserveBuf(len);
737           System.arraycopy(rawCP.getRawBytes(), rawCP.getRawOffset(), buf, offset, len);
738         }
739     
740         char[] chars = noChars;
741     
742         // BE CAREFUL: the newCPEntries array grows during this loop.
743         for (int i = 0; i < newCPEntries.size(); i++) {
744           Object o = newCPEntries.get(i);
745           if (o instanceof CWItem) {
746             CWItem item = (CWItem) o;
747             byte t = item.getType();
748             int offset;
749             switch (t) {
750             case CONSTANT_Class:
751             case CONSTANT_String:
752             case CONSTANT_MethodType:
753               offset = reserveBuf(3);
754               setUShort(buf, offset + 1, addCPUtf8(((CWStringItem) item).s));
755               break;
756             case CONSTANT_NameAndType: {
757               offset = reserveBuf(5);
758               CWNAT nat = (CWNAT) item;
759               setUShort(buf, offset + 1, addCPUtf8(nat.n));
760               setUShort(buf, offset + 3, addCPUtf8(nat.t));
761               break;
762             }
763             case CONSTANT_MethodHandle: {
764               offset = reserveBuf(4);
765               CWHandle handle = (CWHandle) item;
766               setUByte(buf, offset + 1, handle.getKind());
767               switch (handle.getKind()) {
768               case REF_getStatic:
769               case REF_getField:
770               case REF_putField:
771               case REF_putStatic: {
772                 int x = addCPFieldRef(handle.c, handle.n, handle.t);
773                 setUShort(buf, offset + 2, x);
774                 break;
775               }
776               case REF_invokeVirtual:
777               case REF_newInvokeSpecial: {
778                 int x = addCPMethodRef(handle.c, handle.n, handle.t);
779                 setUShort(buf, offset + 2, x);
780                 break;
781               }
782               case REF_invokeSpecial:
783               case REF_invokeStatic: {
784                 int x = addCPMethodRef(handle.c, handle.n, handle.t);
785                 setUShort(buf, offset + 2, x);
786                 break;
787               }
788               case REF_invokeInterface: {
789                 int x = addCPInterfaceMethodRef(handle.c, handle.n, handle.t);
790                 setUShort(buf, offset + 2, x);
791                 break;
792               }
793               }
794              break; 
795             }
796             case CONSTANT_MethodRef:
797             case CONSTANT_FieldRef:
798             case CONSTANT_InterfaceMethodRef: {
799               offset = reserveBuf(5);
800               CWRef ref = (CWRef) item;
801               setUShort(buf, offset + 1, addCPClass(ref.c));
802               setUShort(buf, offset + 3, addCPNAT(ref.n, ref.t));
803               break;
804             }
805             default:
806               throw new Error("Invalid type: " + t);
807             }
808             buf[offset] = t;
809           } else {
810             if (o instanceof String) {
811               String s = (String) o;
812               int slen = s.length();
813     
814               if (chars.length < slen) {
815                 chars = new char[slen];
816               }
817               s.getChars(0, slen, chars, 0);
818     
819               int offset = reserveBuf(3);
820               buf[offset] = CONSTANT_Utf8;
821     
822               int maxBytes = slen * 3;
823               int p = reserveBuf(maxBytes); // worst case reservation
824     
825               for (int j = 0; j < slen; j++) {
826                 char ch = chars[j];
827                 if (ch == 0) {
828                   setUShort(buf, p, 0xC080);
829                   p += 2;
830                 } else if (ch < 0x80) {
831                   buf[p] = (byte) ch;
832                   p += 1;
833                 } else if (ch < 0x800) {
834                   buf[p] = (byte) ((ch >> 6) | 0xC0);
835                   buf[p + 1] = (byte) ((ch & 0x3F) | 0x80);
836                   p += 2;
837                 } else {
838                   buf[p] = (byte) ((ch >> 12) | 0xE0);
839                   buf[p + 1] = (byte) (((ch >> 6) & 0x3F) | 0x80);
840                   buf[p + 2] = (byte) ((ch & 0x3F) | 0x80);
841                   p += 3;
842                 }
843               }
844               int bytes = p - (offset + 3);
845               reserveBuf(bytes - maxBytes); // negative reservation to push back buf
846               // size
847               if (bytes > 0xFFFF) {
848                 throw new IllegalArgumentException("String too long: " + bytes + " bytes");
849               }
850               setUShort(buf, offset + 1, bytes);
851             } else if (o instanceof Integer) {
852               int offset = reserveBuf(5);
853               buf[offset] = CONSTANT_Integer;
854               setInt(buf, offset + 1, ((Integer) o).intValue());
855             } else if (o instanceof Long) {
856               int offset = reserveBuf(9);
857               buf[offset] = CONSTANT_Long;
858               setLong(buf, offset + 1, ((Long) o).longValue());
859             } else if (o instanceof Float) {
860               int offset = reserveBuf(5);
861               buf[offset] = CONSTANT_Float;
862               setFloat(buf, offset + 1, ((Float) o).intValue());
863             } else if (o instanceof Double) {
864               int offset = reserveBuf(9);
865               buf[offset] = CONSTANT_Double;
866               setDouble(buf, offset + 1, ((Double) o).intValue());
867             }
868           }
869         }
870       }

872       /**
873        * After you've added everything you need to the class, call this method to generate the actual class file data. This can only be
874        * called once.
875        */
876       public byte[] makeBytes() throws IllegalArgumentException {
877         if (buf != null) {
878           throw new IllegalArgumentException("Can't call makeBytes() twice");
879         }
880     
881         if (thisClass == 0) {
882           throw new IllegalArgumentException("No class name set");
883         }
884     
885         reserveBuf(10);
886         setInt(buf, 0, MAGIC);
887         setUShort(buf, 4, minorVersion);
888         setUShort(buf, 6, majorVersion);
889     
890         emitConstantPool();
891         // The constant pool can grow during emmission, so store the size last
892         setUShort(buf, 8, nextCPIndex);
893         // No new constant pool entries can be allocated; make sure we
894         // catch any such error by client code
895         cachedCPEntries = null;
896     
897         int offset = reserveBuf(8);
898         setUShort(buf, offset, accessFlags);
899         setUShort(buf, offset + 2, thisClass);
900         setUShort(buf, offset + 4, superClass);
901         if (superInterfaces != null) {
902           setUShort(buf, offset + 6, superInterfaces.length);
903           reserveBuf(superInterfaces.length * 2);
904           for (int i = 0; i < superInterfaces.length; i++) {
905             setUShort(buf, offset + 8 + i * 2, superInterfaces[i]);
906           }
907         } else {
908           setUShort(buf, offset + 6, 0);
909         }
910     
911         offset = reserveBuf(2);
912         int numFields = fields.size();
913         setUShort(buf, offset, numFields);
914         for (int i = 0; i < numFields; i++) {
915           emitElement(fields.get(i));
916         }
917     
918         offset = reserveBuf(2);
919         int numMethods = methods.size();
920         // Xiangyu, debug
921         // System.out.println("numMethods="+numMethods);
922         setUShort(buf, offset, numMethods);
923         for (int i = 0; i < numMethods; i++) {
924           emitElement(methods.get(i));
925         }
926     
927         offset = reserveBuf(2);
928         int numAttrs = classAttributes.size();
929         setUShort(buf, offset, numAttrs);
930         for (int i = 0; i < numAttrs; i++) {
931           emitElement(classAttributes.get(i));
932         }
933     
934         if (buf.length == bufLen) {
935           return buf;
936         } else {
937           byte[] b = new byte[bufLen];
938           System.arraycopy(buf, 0, b, 0, bufLen);
939           return b;
940         }
941       }

943       /**
944        * Set the byte at offset 'offset' in 'buf' to the unsigned 8-bit value in v.
945        * 
946        * @throws IllegalArgumentException if buf is null
947        */
948       public static void setUByte(byte[] buf, int offset, int v) throws IllegalArgumentException {
949         if (buf == null) {
950           throw new IllegalArgumentException("buf is null");
951         }
952         try {
953           buf[offset] = (byte) v;
954         } catch (ArrayIndexOutOfBoundsException e) {
955           throw new IllegalArgumentException("invalid offset: " + offset);
956         }
957       }

959       /**
960        * Set the 4 bytes at offset 'offset' in 'buf' to the signed 32-bit value in v.
961        * 
962        * @throws IllegalArgumentException if buf is null
963        */
964       public static void setInt(byte[] buf, int offset, int v) throws IllegalArgumentException {
965         if (buf == null) {
966           throw new IllegalArgumentException("buf is null");
967         }
968         try {
969           buf[offset] = (byte) (v >> 24);
970           buf[offset + 1] = (byte) (v >> 16);
971           buf[offset + 2] = (byte) (v >> 8);
972           buf[offset + 3] = (byte) v;
973         } catch (ArrayIndexOutOfBoundsException e) {
974           throw new IllegalArgumentException("illegal offset " + offset);
975         }
976       }

979        * Set the 8 bytes at offset 'offset' in 'buf' to the signed 64-bit value in v.
980        */
981       public static void setLong(byte[] buf, int offset, long v) throws IllegalArgumentException {
982         setInt(buf, offset, (int) (v >> 32));
983         setInt(buf, offset + 4, (int) v);

1000       /**
1001        * Set the 2 bytes at offset 'offset' in 'buf' to the unsigned 16-bit value in v.
1002        * 
1003        * @throws IllegalArgumentException if buf is null
1004        */
1005       public static void setUShort(byte[] buf, int offset, int v) throws IllegalArgumentException {
1006         if (buf == null) {
1007           throw new IllegalArgumentException("buf is null");
1008         }
1009         if (offset < 0 || offset + 1 >= buf.length) {
1010           throw new IllegalArgumentException("buf is too short " + buf.length + " " + offset);
1011         }
1012         try {
1013           buf[offset] = (byte) (v >> 8);
1014           buf[offset + 1] = (byte) v;
1015         } catch (ArrayIndexOutOfBoundsException e) {
1016           throw new IllegalArgumentException("invalid offset: " + offset);
1017         }
1018       }
