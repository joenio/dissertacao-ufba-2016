38     /**
39      * A simple liveness analysis based on flow-insensitive pointer analysis.
40      */
41     public class FILiveObjectAnalysis implements ILiveObjectAnalysis {
42     
43       /**
44        * governing call graph
45        */
46       private final CallGraph callGraph;
47     
48       /**
49        * Graph view of pointer analysis results
50        */
51       private final HeapGraph heapGraph;
52     
53       /**
54        * Cached map from InstanceKey -> Set<CGNode>
55        */
56       final private Map<InstanceKey, Set<CGNode>> liveNodes = HashMapFactory.make();
57     
58       /**
59        * Set of instanceKeys which are live everywhere
60        */
61       final private Set<InstanceKey> liveEverywhere = HashSetFactory.make();
62     
63       /**
64        * A hack for now .. since right now the intraprocedural analysis is expensive
65        */
66       private final boolean expensiveIntraproceduralAnalysis;
67     
68       /**
69        * 
70        */
71       public FILiveObjectAnalysis(CallGraph callGraph, HeapGraph heapGraph, boolean expensiveIntraproceduralAnalysis) {
72         super();
73         this.callGraph = callGraph;
74         this.heapGraph = heapGraph;
75         this.expensiveIntraproceduralAnalysis = expensiveIntraproceduralAnalysis;
76       }
77     
78       @Override
79       public boolean mayBeLive(CGNode allocMethod, int allocPC, CGNode m, int instructionIndex) throws IllegalArgumentException,
80           WalaException {
81         if (allocMethod == null) {
82           throw new IllegalArgumentException("allocMethod == null");
83         }
84         NewSiteReference site = TrivialMethodEscape.findAlloc(allocMethod, allocPC);
85         InstanceKey ik = heapGraph.getHeapModel().getInstanceKeyForAllocation(allocMethod, site);
86         return mayBeLive(ik, m, instructionIndex);
87       }
88     
89       /**
90        * @param instructionIndex index of an SSA instruction
91        */
92       @Override
93       public boolean mayBeLive(InstanceKey ik, CGNode m, int instructionIndex) {
94         if (liveEverywhere.contains(ik)) {
95           return true;
96         } else {
97           Set<CGNode> live = liveNodes.get(ik);
98           if (live != null) {
99             if (live.contains(m)) {
100               if (instructionIndex == -1) {
101                 return true;
102               } else {
103                 if (mayBeLiveInSomeCaller(ik, m)) {
104                   // a hack. if it's live in some caller of m, assume it's live
105                   // throughout m. this may be imprecise.
106                   return true;
107                 } else {
108                   if (expensiveIntraproceduralAnalysis) {
109                     return mayBeLiveIntraprocedural(ik, m, instructionIndex);
110                   } else {
111                     // be conservative
112                     return true;
113                   }
114                 }
115               }
116             } else {
117               return false;
118             }
119           } else {
120             live = computeLiveNodes(ik);
121             liveNodes.put(ik, live);
122             return mayBeLive(ik, m, instructionIndex);
123           }
124         }
125       }
126     
127       private boolean mayBeLiveInSomeCaller(InstanceKey ik, CGNode m) {
128         for (Iterator it = callGraph.getPredNodes(m); it.hasNext();) {
129           CGNode n = (CGNode) it.next();
130           if (mayBeLive(ik, n, -1)) {
131             return true;
132           }
133         }
134         return false;
135       }
136     
137       /**
138        * precondition: !mayBeLiveInSomeCaller(ik, m)
139        * 
140        * @param instructionIndex index of an SSA instruction
141        */
142       private boolean mayBeLiveIntraprocedural(InstanceKey ik, CGNode m, int instructionIndex) {
143     
144         IR ir = m.getIR();
145         DefUse du = m.getDU();
146     
147         for (Iterator it = DFS.iterateDiscoverTime(GraphInverter.invert(heapGraph), ik); it.hasNext();) {
148           Object p = it.next();
149           if (p instanceof LocalPointerKey) {
150             LocalPointerKey lpk = (LocalPointerKey) p;
151             if (lpk.getNode().equals(m)) {
152               if (LocalLiveRangeAnalysis.isLive(lpk.getValueNumber(), instructionIndex, ir, du)) {
153                 return true;
154               }
155             }
156           }
157         }
158         return false;
159       }
160     
161       /**
162        * Compute the set of nodes in which ik may be live. If it's live everywhere, return an EMPTY_SET, but also record this fact in
163        * the "liveEverywhere" set as a side effect
164        */
165       private Set<CGNode> computeLiveNodes(InstanceKey ik) {
166         Set<CGNode> localRootNodes = HashSetFactory.make();
167         for (Iterator it = DFS.iterateDiscoverTime(GraphInverter.invert(heapGraph), ik); it.hasNext();) {
168           Object node = it.next();
169           if (node instanceof StaticFieldKey) {
170             liveEverywhere.add(ik);
171             return Collections.emptySet();
172           } else {
173             if (node instanceof AbstractLocalPointerKey) {
174               AbstractLocalPointerKey local = (AbstractLocalPointerKey) node;
175               localRootNodes.add(local.getNode());
176             } else if (node instanceof TypedPointerKey) {
177               TypedPointerKey t = (TypedPointerKey) node;
178               node = t.getBase();
179               if (node instanceof AbstractLocalPointerKey) {
180                 AbstractLocalPointerKey local = (AbstractLocalPointerKey) node;
181                 localRootNodes.add(local.getNode());
182               } else {
183                 Assertions.UNREACHABLE("unexpected base of TypedPointerKey: " + node.getClass() + " " + node);
184               }
185             }
186           }
187         }
188         return DFS.getReachableNodes(callGraph, localRootNodes);
189       }
190     
191       @Override
192       public boolean mayBeLive(InstanceKey ik, CGNode m, IntSet instructionIndices) {
193         if (instructionIndices == null) {
194           throw new IllegalArgumentException("instructionIndices is null");
195         }
196         // TODO this sucks
197         for (IntIterator it = instructionIndices.intIterator(); it.hasNext();) {
198           int i = it.next();
199           if (mayBeLive(ik, m, i)) {
200             return true;
201           }
202         }
203         return false;
204       }
