11     package com.ibm.wala.ssa;
12     
13     import com.ibm.wala.shrikeBT.IConditionalBranchInstruction;
14     import com.ibm.wala.shrikeBT.IConditionalBranchInstruction.IOperator;
15     import com.ibm.wala.types.TypeReference;
16     
17     /**
18      * A conditional branch instruction, which tests two values according to some {@link IOperator}.
19      */
20     public class SSAConditionalBranchInstruction extends SSAInstruction {
21       private final IConditionalBranchInstruction.IOperator operator;
22     
23       private final int val1;
24     
25       private final int val2;
26     
27       private final TypeReference type;
28     
29       final private int target;
30     
31       public SSAConditionalBranchInstruction(int iindex, IConditionalBranchInstruction.IOperator operator, TypeReference type, int val1, int val2, int target)
32           throws IllegalArgumentException {
33         super(iindex);
34         this.target = target;
35         this.operator = operator;
36         this.val1 = val1;
37         this.val2 = val2;
38         this.type = type;
39         if (val1 <= 0) {
40           throw new IllegalArgumentException("Invalid val1: " + val1);
41         }
42         if (val2 <= 0) {
43           throw new IllegalArgumentException("Invalid val2: " + val2);
44         }
45       }
46     
47       public int getTarget() {
48         return target;
49       }
50     
51       @Override
52       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) throws IllegalArgumentException {
53         if (uses != null && uses.length < 2) {
54           throw new IllegalArgumentException("(uses != null) and (uses.length < 2)");
55         }
56         return insts.ConditionalBranchInstruction(iindex, operator, type, uses == null ? val1 : uses[0], uses == null ? val2 : uses[1], target);
57       }
58     
59       public IConditionalBranchInstruction.IOperator getOperator() {
60         return operator;
61       }
62     
63     
64       @Override
65       public String toString(SymbolTable symbolTable) {
66         return "conditional branch(" + operator + ", to iindex=" + target + ") " + getValueString(symbolTable, val1) + "," + getValueString(symbolTable, val2);
67       }
68     
69       /**
70        * @see com.ibm.wala.ssa.SSAInstruction#visit(IVisitor)
71        * @throws IllegalArgumentException if v is null
72        */
73       @Override
74       public void visit(IVisitor v) {
75         if (v == null) {
76           throw new IllegalArgumentException("v is null");
77         }
78         v.visitConditionalBranch(this);
79       }
80     
81       /**
82        * @see com.ibm.wala.ssa.SSAInstruction#getNumberOfUses()
83        */
84       @Override
85       public int getNumberOfUses() {
86         return 2;
87       }
88     
89       /**
90        * @see com.ibm.wala.ssa.SSAInstruction#getUse(int)
91        */
92       @Override
93       public int getUse(int j) {
94         assert j <= 1;
95         return (j == 0) ? val1 : val2;
96       }
97     
98       public TypeReference getType() {
99         return type;
100       }
101     
102       public boolean isObjectComparison() {
103         return type.isReferenceType();
104       }
105     
106       public boolean isIntegerComparison() {
107         return type == TypeReference.Int;
108       }
109     
110       @Override
111       public int hashCode() {
112         return 7151 * val1 + val2;
113       }
114     
115       /*
116        * @see com.ibm.wala.ssa.Instruction#isFallThrough()
117        */
118       @Override
119       public boolean isFallThrough() {
120         return true;
121       }
122     
123     }
