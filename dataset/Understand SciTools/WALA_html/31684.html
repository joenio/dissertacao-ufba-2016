11     package com.ibm.wala.fixpoint;
12     
13     import com.ibm.wala.util.graph.impl.NodeWithNumber;
14     
15     /**
16      * Represents a single variable in a fixed-point system.
17      */
18     @SuppressWarnings("rawtypes")
19     public abstract class AbstractVariable<T extends AbstractVariable> extends NodeWithNumber implements IVariable<T> {
20     
21       private static int nextHashCode = 0;
22       
23       private int orderNumber;
24       
25       private final int hashCode;
26       
27       protected AbstractVariable() {
28         this.hashCode = nextHash();
29       }
30     
31       @Override
32       public boolean equals(Object obj) {
33         // we assume the solver manages these canonically
34         return this == obj;
35       }
36       
37       /**
38        * I know this is theoretically bad.   However,
39        * <ul>
40        * <li> we need this to be extremely fast .. it's in the inner loop of lots of stuff.
41        * <li> these objects will probably only be hashed with each other {@link AbstractVariable}s, 
42        * in which case incrementing hash codes is OK.
43        * <li> we want determinism, so we don't want to rely on System.identityHashCode
44        * </ul>
45        */
46       public static synchronized int nextHash() {
47         return nextHashCode++;
48       }
49     
50       @Override
51       public final int hashCode() {
52         return hashCode;
53       }
54     
55       @Override
56       public int getOrderNumber() {
57         return orderNumber;
58       }
59     
60       @Override
61       public void setOrderNumber(int orderNumber) {
62         this.orderNumber = orderNumber;
63       }
64     }
