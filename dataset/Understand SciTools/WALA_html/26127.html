41     package com.ibm.wala.dalvik.util.androidEntryPoints;
42     
43     import java.util.List;
44     
45     import com.ibm.wala.dalvik.ipa.callgraph.impl.AndroidEntryPoint;
46     import com.ibm.wala.dalvik.ipa.callgraph.impl.AndroidEntryPoint.ExecutionOrder;
47     import com.ibm.wala.dalvik.util.AndroidComponent;
48     import com.ibm.wala.dalvik.util.AndroidEntryPointLocator.AndroidPossibleEntryPoint;
49     
50     /**
51      *  Hardcoded EntryPoint-specifications for an Android-Application.
52      *
53      *  The specifications are read and handled by AndroidEntryPointLocator.
54      *  see: http://developer.android.com/reference/android/app/Application.html
55      *
56      *  @see    com.ibm.wala.dalvik.util.AndroidEntryPointLocator
57      *  @author Tobias Blaschke <code@tobiasblaschke.de>
58      */
59     public final class ApplicationEP {
60     
61         /**
62          * Called when the application is starting, before any activity, service, or receiver objects (excluding content providers) have been created.
63          */
64     	public static final AndroidPossibleEntryPoint onCreate = new AndroidPossibleEntryPoint(AndroidComponent.APPLICATION, 
65                 "onCreate",
66     			ExecutionOrder.between(
67                     new AndroidEntryPoint.IExecutionOrder[] {
68                         ExecutionOrder.AT_FIRST,
69                         ProviderEP.onCreate     /* Yes, ContentProviders come before App! */
70                     },
71                     new AndroidEntryPoint.IExecutionOrder[] {
72                         ActivityEP.onCreate,
73                         ServiceEP.onCreate
74                     }
75                 ));
76     
77         /**
78          *  Called by the system when the device configuration changes while your component is running.
79          *
80          *  Note that, unlike activities, other components are never restarted when a configuration changes: they must always deal with the 
81          *  results of the change, such as by re-retrieving resources. 
82          */
83     	public static final AndroidPossibleEntryPoint onConfigurationChanged = new AndroidPossibleEntryPoint(AndroidComponent.APPLICATION, 
84                 "onConfigurationChanged",   // TODO: Position
85     			ExecutionOrder.between(
86                     new AndroidEntryPoint.IExecutionOrder[] {
87                         ActivityEP.onConfigurationChanged,
88                         ExecutionOrder.END_OF_LOOP
89                     },
90                     new AndroidEntryPoint.IExecutionOrder[] {
91                         ExecutionOrder.AT_LAST
92                    }
93                 ));
94         /**
95          *  This is called when the overall system is running low on memory, and actively running processes should trim their memory usage.
96          *
97          *  While the exact point at which this will be called is not defined, generally it will happen when all background process have been killed.
98          *  That is, before reaching the point of killing processes hosting service and foreground UI that we would like to avoid killing. 
99          */
100     	public static final AndroidPossibleEntryPoint onLowMemory = new AndroidPossibleEntryPoint(AndroidComponent.APPLICATION, 
101                 "onLowMemory",  // TODO: Position
102     			ExecutionOrder.between(
103                     new AndroidEntryPoint.IExecutionOrder[] {
104                         ExecutionOrder.END_OF_LOOP,
105                         ActivityEP.onLowMemory
106                     },
107                     new AndroidEntryPoint.IExecutionOrder[] {
108                         ExecutionOrder.AFTER_LOOP,
109                         onConfigurationChanged  // XXX ??!
110                     }
111                 ));
112     
113         /**
114          *  This method is for use in emulated process environments.
115          */
116         /*
117     	public static final AndroidPossibleEntryPoint onTerminate = new AndroidPossibleEntryPoint(AndroidComponent.APPLICATION, 
118                 "onTerminate",
119     			ExecutionOrder.AT_LAST
120                 );
121         */
122     
123         /**
124          *  Called when the operating system has determined that it is a good time for a process to trim unneeded memory from its process.
125          */
126     	public static final AndroidPossibleEntryPoint onTrimMemory = new AndroidPossibleEntryPoint(AndroidComponent.APPLICATION, 
127                 "onTrimMemory",
128                 ExecutionOrder.directlyBefore(onLowMemory)     // may potentially come before onLowMemory 
129                 );
130     
131         /**
132          *  Add the EntryPoint specifications defined in this file to the given list.
133          *
134          *  @param  possibleEntryPoints the list to extend.
135          */
136     	public static void populate(List<? super AndroidPossibleEntryPoint> possibleEntryPoints) {
137     		possibleEntryPoints.add(onCreate);
138     		possibleEntryPoints.add(onConfigurationChanged);
139             possibleEntryPoints.add(onLowMemory);
140             //possibleEntryPoints.add(onTerminate);
141     		possibleEntryPoints.add(onTrimMemory);
142     	}
143     }
