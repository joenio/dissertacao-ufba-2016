39       public static class ZippedProjectData {
40         public final Plugin sourcePlugin;
41         public final String projectName;
42         public final String zipFileName;
43      
44         public ZippedProjectData(Plugin sourcePlugin, String projectName, String zipFileName) {
45           this.sourcePlugin = sourcePlugin;
46           this.projectName = projectName;
47           this.zipFileName = zipFileName;
48           open();
49         }
50         
51         private void open() {
52           importZippedProject(sourcePlugin, projectName, zipFileName, new NullProgressMonitor());
53         }
54         
55         public void close() {
56           destroyProject(projectName);
57         }
58       }
59       
60       public static void importZippedProject(Plugin plugin, String projectName, String zipFileName, IProgressMonitor monitor) {
61         ZipFile zipFile = getZipFile(plugin, zipFileName);
62         createOpenProject(projectName);
63         importZipfile(projectName, zipFile, monitor);
64       }
65     
66       public static void createOpenProject(String projectName) {
67         IWorkspaceRoot root = getWorkspace();
68         IProject project = root.getProject(projectName);
69         try {
70           project.create(null);
71           project.open(null);
72         } catch (CoreException e) {
73           e.printStackTrace();
74         }
75       }
76     
77       public static void destroyProject(String projectName) {
78         IWorkspaceRoot root = getWorkspace();
79         IProject project = root.getProject(projectName);
80         try {
81           project.delete(true, null);
82         } catch (CoreException e) {
83           e.printStackTrace();
84         }
85       }
86     
87       public static void importProjectFromFilesystem(String projectName, File root, IProgressMonitor monitor) {
88         FileSystemStructureProvider fs = FileSystemStructureProvider.INSTANCE;
89         importProject(fs, monitor, projectName, root);
90       }
91       
92       public static void importZipfile(String projectName, ZipFile zipFile, IProgressMonitor monitor) {
93         ZipFileStructureProvider provider = new ZipFileStructureProvider(zipFile);
94     
95         importProject(provider, monitor, projectName, provider.getRoot());
96     
97         try {
98           zipFile.close();
99         } catch (IOException e) {
100           e.printStackTrace();
101         }
102       }
103     
104       protected static <T> void importProject(IImportStructureProvider provider, IProgressMonitor monitor, String projectName, T root) {
105         IPath containerPath = getWorkspacePath().append(projectName).addTrailingSeparator();
106     
107         ImportOperation importOp = new ImportOperation(containerPath, root, provider, new IOverwriteQuery() {
108           @Override
109           public String queryOverwrite(String pathString) {
110             return IOverwriteQuery.ALL;
111           }
112         });
113     
114         importOp.setCreateContainerStructure(false);
115         importOp.setOverwriteResources(true);
116         try {
117           importOp.run(monitor);
118         } catch (InvocationTargetException e) {
119           e.printStackTrace();
120         } catch (InterruptedException e) {
121           e.printStackTrace();
122         }
123       }
124     
125       public static File getTestDataFile(Plugin plugin, String filename) {
126         Bundle bundle = plugin.getBundle();
127         IPath path = new Path("testdata").append(filename);
128     
129         URL url = FileLocator.find(bundle, path, null);
130         assert url != null : bundle.toString() + " path " + path.toString();
131         try {
132           URL fileURL = FileLocator.toFileURL(url);
133           File file = new File(fileURL.getPath());
134           return file;
135         } catch (IOException e) {
136           reportException(e);
137         }
138     
139         return null;
140       }
141     
142       public static ZipFile getZipFile(Plugin plugin, String testArchive) {
143         File file = getTestDataFile(plugin, testArchive);
144         if (file != null) {
145           try {
146             return new ZipFile(file);
147           } catch (ZipException e) {
148             reportException(e);
149           } catch (IOException e) {
150             reportException(e);
151           }
152         }
153     
154         return null;
155       }
156     
157       public static IWorkspaceRoot getWorkspace() {

159       }

162         return ResourcesPlugin.getWorkspace().getRoot().getFullPath();
163       }
