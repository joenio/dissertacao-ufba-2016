11     package com.ibm.wala.ide.util;
12     
13     import static com.ibm.wala.ide.util.JdtUtil.getAST;
14     import static com.ibm.wala.ide.util.JdtUtil.getOriginalNode;
15     
16     import java.lang.ref.SoftReference;
17     import java.util.Map;
18     
19     import org.eclipse.core.resources.IFile;
20     import org.eclipse.jdt.core.dom.ASTNode;
21     
22     import com.ibm.wala.util.collections.HashMapFactory;
23     
24     public class ASTNodeFinder {
25     
26     	private final Map<IFile, SoftReference<ASTNode>> fileASTs = HashMapFactory.make();
27     	
28     	public ASTNode getASTNode(JdtPosition pos) {
29     		IFile sourceFile = pos.getEclipseFile();
30     		if (!fileASTs.containsKey(sourceFile) || fileASTs.get(sourceFile).get() == null) {
31     			fileASTs.put(sourceFile, new SoftReference<ASTNode>(getAST(sourceFile)));
32     		}
33     		
34     		return getOriginalNode(fileASTs.get(sourceFile).get(), pos);
35     	}
36     	
37     }
