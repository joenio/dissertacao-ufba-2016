11     package com.ibm.wala.shrikeBT.shrikeCT;

13     import java.io.BufferedInputStream;
14     import java.io.IOException;
15     import java.io.OutputStream;
16     
17     import com.ibm.wala.shrikeBT.Util;
18     import com.ibm.wala.shrikeBT.analysis.ClassHierarchyProvider;
19     import com.ibm.wala.shrikeBT.tools.OfflineInstrumenterBase;
20     import com.ibm.wala.shrikeCT.ClassWriter;
21     import com.ibm.wala.shrikeCT.InvalidClassFileException;
22     
23     /**
24      * This class provides a convenient way to iterate through a collection of Java classes and instrument their code. This is just a
25      * specialization of OfflineInstrumenterBase to use the shrikeCT functionality.
26      */
27     final public class OfflineInstrumenter extends OfflineInstrumenterBase {
28       private final boolean reuseStackMaps;
29       /**
30        * Create an empty collection of classes to instrument.
31        */
32       public OfflineInstrumenter(boolean reuseStackMaps) {
33         this.reuseStackMaps = reuseStackMaps;
34       }
35     
36       @Override
37       protected Object makeClassFromStream(String inputName, BufferedInputStream s) throws IOException {
38         byte[] bytes = new byte[s.available()];
39         Util.readFully(s, bytes);
40         try {
41           return new ClassInstrumenter(inputName, bytes, cha, reuseStackMaps);
42         } catch (InvalidClassFileException e) {
43           throw new IOException("Class is invalid: " + e.getMessage());
44         }
45       }
46     
47       @Override
48       protected String getClassName(Object cl) {
49         try {
50           return ((ClassInstrumenter) cl).getReader().getName().replace('/', '.');
51         } catch (InvalidClassFileException e) {
52           return null;
53         }
54       }
55     
56       @Override
57       protected void writeClassTo(Object cl, Object mods, OutputStream s) throws IOException {
58         ClassInstrumenter ci = (ClassInstrumenter) cl;
59         ClassWriter cw = (ClassWriter) mods;
60         if (cw == null) {
61           s.write(ci.getReader().getBytes());
62         } else {
63           s.write(cw.makeBytes());
64         }
65       }
66     
67       /**
68        * Get the next class to be instrumented.
69        */
70       public ClassInstrumenter nextClass() throws IOException {
71         return (ClassInstrumenter) internalNextClass();
72       }
73     
74       /**
75        * Update the original class with some method changes. 'code' should be the result of out.emitClass(). You can add new fields and
76        * methods to 'code' (or make other changes) before calling this method.
77        */
78       public void outputModifiedClass(ClassInstrumenter out, ClassWriter code) throws IllegalStateException, IOException {
79         internalOutputModifiedClass(out, out.getInputName(), code);
80       }
81     
82       /**
83        * Update the original class with some method changes. This method calls out.emitClass() for you.
84        */
85       public void outputModifiedClass(ClassInstrumenter out) throws IllegalArgumentException, IOException {
86         if (out == null) {
87           throw new IllegalArgumentException();
88         }
89         try {
90           internalOutputModifiedClass(out, out.getInputName(), out.emitClass());
91         } catch (InvalidClassFileException e) {
92           e.printStackTrace();
93           throw new IOException("Invalid class file");
94         }
95       }
96     }
