12     package com.ibm.wala.classLoader;
13     
14     import java.io.InputStream;
15     import java.util.jar.JarFile;
16     
17     import com.ibm.wala.util.debug.Assertions;
18     import com.ibm.wala.util.io.FileSuffixes;
19     
20     /**
21      * An entry in a Jar file.
22      */
23     public class JarFileEntry implements ModuleEntry {
24     
25       private final String entryName;
26     
27       private final JarFileModule jarFileModule;
28     
29       protected JarFileEntry(String entryName, JarFileModule jarFile) {
30         this.entryName = entryName;
31         this.jarFileModule = jarFile;
32       }
33     
34       /*
35        * @see com.ibm.wala.classLoader.ModuleEntry#getName()
36        */
37       @Override
38       public String getName() {
39         return entryName;
40       }
41     
42       /*
43        * @see com.ibm.wala.classLoader.ModuleEntry#isClassFile()
44        */
45       @Override
46       public boolean isClassFile() {
47         return FileSuffixes.isClassFile(getName());
48       }
49     
50       /*
51        * @see com.ibm.wala.classLoader.ModuleEntry#getInputStream()
52        */
53       @Override
54       public InputStream getInputStream() {
55         try {
56           JarFile jarFile = jarFileModule.getJarFile();
57           return jarFile.getInputStream(jarFile.getEntry(entryName));
58         } catch (Exception e) {
59           // TODO Auto-generated catch block
60           e.printStackTrace();
61           Assertions.UNREACHABLE();
62           return null;
63         }
64       }
65     
66       /*
67        * @see com.ibm.wala.classLoader.ModuleEntry#getSize()
68        */
69       public long getSize() {
70         // TODO: cache this?
71         return jarFileModule.getJarFile().getEntry(entryName).getSize();
72       }
73     
74       @Override
75       public String toString() {
76         return jarFileModule.getJarFile().getName() + ":" + getName();
77       }
78     
79       /*
80        * @see com.ibm.wala.classLoader.ModuleEntry#isModuleFile()
81        */
82       @Override
83       public boolean isModuleFile() {
84         return FileSuffixes.isJarFile(getName()) || FileSuffixes.isWarFile(getName());
85       }
86     
87       /*
88        * @see com.ibm.wala.classLoader.ModuleEntry#asModule()
89        */
90       @Override
91       public Module asModule() {
92         return new NestedJarFileModule(jarFileModule, jarFileModule.getJarFile().getEntry(entryName));
93       }
94     
95       public JarFile getJarFile() {
96         return jarFileModule.getJarFile();
97       }
98       
99       @Override
100       public JarFileModule getContainer() {
101         return jarFileModule;
102       }
103     
104       @Override
105       public int hashCode() {
106         return entryName.hashCode() * 5059 + jarFileModule.getJarFile().hashCode();
107       }
108     
109       /*
110        * @see com.ibm.wala.classLoader.ModuleEntry#getClassName()
111        */
112       @Override
113       public String getClassName() {
114         return FileSuffixes.stripSuffix(getName());
115       }
116     
117       /*
118        * @see com.ibm.wala.classLoader.ModuleEntry#isSourceFile()
119        */
120       @Override
121       public boolean isSourceFile() {
122         return FileSuffixes.isSourceFile(getName());
123       }
124     
125       @Override
126       public boolean equals(Object obj) {
127         return this == obj;
128       }
129     }
