41     /**
42      * abstract class loader that performs CAst and IR generation for relevant
43      * entities in a list of {@link Module}s. Subclasses provide the CAst / IR
44      * translators appropriate for the language.
45      * 
46      */
47     public abstract class CAstAbstractModuleLoader extends CAstAbstractLoader {
48     
49       private static final boolean DEBUG = false;
50     
51       public CAstAbstractModuleLoader(IClassHierarchy cha, IClassLoader parent) {
52         super(cha, parent);
53       }
54     
55       public CAstAbstractModuleLoader(IClassHierarchy cha) {
56         this(cha, null);
57       }
58     
59       /**
60        * create the appropriate CAst translator for the language and source module
61        */
62       protected abstract TranslatorToCAst getTranslatorToCAst(CAst ast, SourceModule M) throws IOException;
63     
64       /**
65        * should IR be generated for entity?
66        */
67       protected abstract boolean shouldTranslate(CAstEntity entity);
68     
69       /**
70        * create the appropriate IR translator for the language
71        */
72       protected abstract TranslatorToIR initTranslator();
73     
74       protected File getLocalFile(SourceModule M) throws IOException {
75         if (M instanceof SourceFileModule) {
76           return ((SourceFileModule) M).getFile();
77         } else {
78           File f = File.createTempFile("module", ".txt");
79           f.deleteOnExit();
80           TemporaryFile.streamToFile(f, M.getInputStream());
81           return f;
82         }
83       }
84     
85       /**
86        * subclasses should override to perform actions after CAst and IR have been
87        * generated. by default, do nothing
88        */
89       protected void finishTranslation() {
90     
91       }
92     
93       @Override
94       public void init(final List<Module> modules) {
95     
96         final CAst ast = new CAstImpl();
97     
98         // convert everything to CAst
99         final Set<Pair<CAstEntity, ModuleEntry>> topLevelEntities = new LinkedHashSet<Pair<CAstEntity, ModuleEntry>>();
100         for (Iterator<Module> mes = modules.iterator(); mes.hasNext();) {
101           translateModuleToCAst(mes.next(), ast, topLevelEntities);
102         }
103     
104         // generate IR as needed
105         final TranslatorToIR xlatorToIR = initTranslator();
106     
107         for (Iterator<Pair<CAstEntity, ModuleEntry>> tles = topLevelEntities.iterator(); tles.hasNext();) {
108           Pair<CAstEntity, ModuleEntry> p = tles.next();
109           if (shouldTranslate(p.fst)) {
110             xlatorToIR.translate(p.fst, p.snd);
111           }
112         }
113     
114         if (DEBUG) {
115           for (Iterator ts = types.keySet().iterator(); ts.hasNext();) {
116             TypeName tn = (TypeName) ts.next();
117             try {
118               System.err.println(("found type " + tn + " : " + types.get(tn) + " < " + types.get(tn).getSuperclass()));
119             } catch (Exception e) {
120               System.err.println(e);
121             }
122           }
123         }
124     
125         finishTranslation();
126       }
127     
128       /**
129        * translate moduleEntry to CAst and store result in topLevelEntities
130        * 
131        * @param ast
132        * @param topLevelEntities
133        */
134       private void translateModuleEntryToCAst(ModuleEntry moduleEntry, CAst ast, Set<Pair<CAstEntity, ModuleEntry>> topLevelEntities) {
135         try {
136           if (moduleEntry.isModuleFile()) {
137             // nested module
138             translateModuleToCAst(moduleEntry.asModule(), ast, topLevelEntities);
139           } else if (moduleEntry instanceof SourceModule) {
140             TranslatorToCAst xlatorToCAst = getTranslatorToCAst(ast, (SourceModule) moduleEntry);
141     
142             CAstEntity fileEntity = null;
143             try {
144               fileEntity = xlatorToCAst.translateToCAst();
145             
146               if (DEBUG) {
147                 CAstPrinter.printTo(fileEntity, new PrintWriter(System.err));
148               }
149               topLevelEntities.add(Pair.make(fileEntity, moduleEntry));
150     
151             } catch (TranslatorToCAst.Error e) {
152               addMessage(moduleEntry, e.warning);
153             }
154           }
155         } catch (final MalformedURLException e) {
156           addMessage(moduleEntry, new Warning(Warning.SEVERE) {
157             @Override
158             public String getMsg() {
159               return "Malformed URL issue: " + e.getMessage();
160             }
161           });
162         } catch (final IOException e) {
163           addMessage(moduleEntry, new Warning(Warning.SEVERE) {
164             @Override
165             public String getMsg() {
166               return "I/O issue: " + e.getMessage();
167             }
168           });
169         } catch (final RuntimeException e) {
170           final ByteArrayOutputStream s = new ByteArrayOutputStream();
171           PrintStream ps = new PrintStream(s);
172           e.printStackTrace(ps);
173           addMessage(moduleEntry, new Warning(Warning.SEVERE) {
174             @Override
175             public String getMsg() {
176               return "Parsing issue: " + new String(s.toByteArray());
177             }
178           });
179         }
180       }
181     
182       /**
183        * translate all relevant entities in the module to CAst, storing the results
184        * in topLevelEntities
185        */
186       private void translateModuleToCAst(Module module, CAst ast, Set<Pair<CAstEntity, ModuleEntry>> topLevelEntities) {
187         for (Iterator<? extends ModuleEntry> mes = module.getEntries(); mes.hasNext();) {
188           translateModuleEntryToCAst(mes.next(), ast, topLevelEntities);
189         }
190       }
191     
192     }
