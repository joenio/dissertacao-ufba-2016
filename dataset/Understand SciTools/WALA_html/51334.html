37       /**
38        * @return number of annotations in this attribute
39        * @throws InvalidClassFileException
40        */
41       public int getAnnotationCount() throws InvalidClassFileException {
42         int offset = beginOffset + 6;
43         checkSize(offset, 2);
44         return cr.getUShort(offset);

47       /**
48        * @return total length of this attribute in bytes, <bf>including</bf> the
49        *         first 6 bytes
50        * @throws InvalidClassFileException
51        */
52       public int getAttributeSize() throws InvalidClassFileException {
53         int offset = beginOffset + 2;
54         checkSize(offset, 4);

58       /**
59        * get the Utf8 constant pool value, where the constant pool offset is given
60        * in the class
61        * 
62        * @param offset
63        *          offset in the class file at which the constant pool offset is
64        *          given
65        */
66       private String getUtf8ConstantPoolValue(int offset) throws InvalidClassFileException {
67         checkSize(offset, 2);
68         int cpOffset = cr.getUShort(offset);
69         return cr.getCP().getCPUtf8(cpOffset);

155       /**
156        * get all the annotations declared in this attribute.
157        * 
158        * @throws InvalidClassFileException
159        */
160       public AnnotationAttribute[] getAllAnnotations() throws InvalidClassFileException {
161         AnnotationAttribute[] result = new AnnotationAttribute[getAnnotationCount()];
162         int offset = beginOffset + 8; // skip attribute_name_index,
163                                       // attribute_length, and num_annotations
164         for (int i = 0; i < result.length; i++) {
165           Pair<AnnotationAttribute, Integer> attributeAndSize = getAttributeAndSize(offset);
166           result[i] = attributeAndSize.fst;
167           offset += attributeAndSize.snd;
168         }
169         return result;
170       }
171     
172       /**
173        * <pre>
174        * param_annotations {
175        *   u2 attribute_name_index;
176        *   u4 attribute_length;
177        *   u1 num_parameters;
178        *   {
179        *     u2 num_annotations;
180        *     annotation annotations[num_annotations];
181        *   } parameter_annotations[num_parameters];
182        * </pre>
183        */
184       public AnnotationAttribute[][] getAllParameterAnnotations() throws InvalidClassFileException {
185         int numParamOffset = beginOffset + 6;
186         checkSize(numParamOffset, 1);
187         int paramCount = cr.getByte(numParamOffset);
188         AnnotationAttribute[][] result = new AnnotationAttribute[paramCount][];
189         // skip attribute_name_index, attribute_length, and num_parameters
190         int offset = beginOffset + 7;
191         for (int i = 0; i < result.length; i++) {
192           checkSize(offset, 2);
193           result[i] = new AnnotationAttribute[cr.getUShort(offset)];
194           offset += 2;
195           for (int j = 0; j < result[i].length; j++) {
196             Pair<AnnotationAttribute, Integer> attributeAndSize = getAttributeAndSize(offset);
197             result[i][j] = attributeAndSize.fst;
198             offset += attributeAndSize.snd;        
199           }
200         }
201         return result;
202       }
203       /**
204        * <pre>
205        * annotation { 
206        *   u2 type_index;
207        *   u2 num_element_value_pairs; 
208        *   { u2 element_name_index; 
209        *     element_value value; 
210        *   } element_value_pairs[num_element_value_pairs]
211        * </pre>
212        * 
213        * @throws InvalidClassFileException
214        */
215       private Pair<AnnotationAttribute, Integer> getAttributeAndSize(int begin) throws InvalidClassFileException {
216         String type = getUtf8ConstantPoolValue(begin);
217         int numElementValuePairs = cr.getUShort(begin + 2);
218         int size = 4;
219         int offset = begin + 4;
220         Map<String, ElementValue> elementName2Val = HashMapFactory.make();
221         for (int i = 0; i < numElementValuePairs; i++) {
222           String elementName = getUtf8ConstantPoolValue(offset);
223           offset += 2;
224           Pair<ElementValue, Integer> elementValAndSize = readElementValueAndSize(offset);
225           offset += elementValAndSize.snd;
226           size += elementValAndSize.snd + 2;
227           elementName2Val.put(elementName, elementValAndSize.fst);
228         }
229         return Pair.make(new AnnotationAttribute(type, elementName2Val), size);
230       }

232       /**
233        * Representation of an annotation attribute. An annotation has the following
234        * format in the bytecode:
235        * 
236        * <pre>
237        * annotation {
238        *   u2 type_index;
239        *   u2 num_element_value_pairs;
240        *   {  u2 element_name_index;
241        *      element_value value;
242        * } element_value_pairs[num_element_value_pairs];
243        * </pre>
244        * 
245        * See the JVM specification section 4.7.16 for details.
246        * 
247        * This class implements {@link ElementValue} to handle nested annotations.
248        */
249       public static class AnnotationAttribute implements ElementValue {
250     
251         /**
252          * the type of the annotation
253          */
254         public final String type;
255     
256         /**
257          * the arguments to the annotation
258          */
259         public final Map<String, ElementValue> elementValues;
260     
261         public AnnotationAttribute(String type, Map<String, ElementValue> elementValues) {
262           super();
263           this.type = type;
264           this.elementValues = elementValues;
265         }
266     
267         @Override
268         public String toString() {
269           return "AnnotationElementValue [type=" + type + ", elementValues=" + elementValues + "]";
270         }
271     
272       }
273     
274       /**
275        * <pre>
276        * element_value { 
277        *   u1 tag; 
278        *   union {
279        *     u2 const_value_index; 
280        *     {  u2 type_name_index;
281        *        u2 const_name_index; 
282        *     } enum_const_value; 
283        *     u2 class_info_index; 
284        *     annotation annotation_value; 
285        *     {  u2 num_values;
286        *        element_value values[num_values]; 
287        *     } array_value;
288        *   } value;
289        * </pre>
290        * 
291        * A constant value (including class info) is represented by a
292        * {@link ConstantElementValue}. An enum constant value is represented by an
293        * {@link EnumElementValue}. An array value is represented by an
294        * {@link ArrayElementValue}. Finally, a nested annotation is represented by
295        * an {@link AnnotationAttribute}.
296        * 
297        * @throws InvalidClassFileException
298        * @throws IllegalArgumentException
299        */
300       private Pair<ElementValue, Integer> readElementValueAndSize(int offset) throws IllegalArgumentException,
301           InvalidClassFileException {
302         char tag = (char) cr.getByte(offset);
303         // meaning of this short depends on the tag
304         int nextShort = cr.getUShort(offset + 1);
305         switch (tag) {
306         case 'B':
307         case 'C':
308         case 'I':
309         case 'S':
310         case 'Z':
311           return Pair.<ElementValue, Integer> make(new ConstantElementValue(cr.getCP().getCPInt(nextShort)), 3);
312         case 'J':
313           return Pair.<ElementValue, Integer> make(new ConstantElementValue(cr.getCP().getCPLong(nextShort)), 3);
314         case 'D':
315           return Pair.<ElementValue, Integer> make(new ConstantElementValue(cr.getCP().getCPDouble(nextShort)), 3);
316         case 'F':
317           return Pair.<ElementValue, Integer> make(new ConstantElementValue(cr.getCP().getCPFloat(nextShort)), 3);
318         case 's': // string
319         case 'c': // class; just represent as a constant element with the type name
320           return Pair.<ElementValue, Integer> make(new ConstantElementValue(cr.getCP().getCPUtf8(nextShort)), 3);
321         case 'e': // enum
322           return Pair.<ElementValue, Integer> make(
323               new EnumElementValue(cr.getCP().getCPUtf8(nextShort), cr.getCP().getCPUtf8(cr.getUShort(offset + 3))), 5);
324         case '[': // array
325           int numValues = nextShort;
326           int numArrayBytes = 3; // start with 3 for the tag and num_values bytes
327           ElementValue[] vals = new ElementValue[numValues];
328           // start curOffset at beginning of array values
329           int curArrayOffset = offset + 3;
330           for (int i = 0; i < numValues; i++) {
331             Pair<ElementValue, Integer> arrayElemValueAndSize = readElementValueAndSize(curArrayOffset);
332             vals[i] = arrayElemValueAndSize.fst;
333             curArrayOffset += arrayElemValueAndSize.snd;
334             numArrayBytes += arrayElemValueAndSize.snd;
335           }
336           return Pair.<ElementValue, Integer> make(new ArrayElementValue(vals), numArrayBytes);
337         case '@': // annotation
338           Pair<AnnotationAttribute, Integer> attributeAndSize = getAttributeAndSize(offset + 1);
339           // add 1 to size for the tag
340           return Pair.<ElementValue, Integer> make(attributeAndSize.fst, attributeAndSize.snd + 1);
341         default:
342           assert false;
343           return null;
344         }
345       }

351       public static enum AnnotationType {
352         RuntimeVisibleAnnotations, RuntimeInvisibleAnnotations,RuntimeVisibleParameterAnnotations, RuntimeInvisibleParameterAnnotations
353       }
354       
355       public static boolean isKnownAnnotation(String name) {
356         for (AnnotationType t : AnnotationType.values()) {
357           if (t.name().equals(name)) {
358             return true;
359           }
360         }
361         return false;
362       }
363       
364       public static AnnotationsReader getReaderForAnnotation(AnnotationType type, ClassReader.AttrIterator iter) {
365         // search for the desired attribute
366         final String attrName = type.toString();
367         try {
368           for (; iter.isValid(); iter.advance()) {
369             if (iter.getName().equals(attrName)) {
370               return new AnnotationsReader(iter, attrName);
371             }
372           }
373         } catch (InvalidClassFileException e) {
374           Assertions.UNREACHABLE();
375         }
376         return null;
377       }
