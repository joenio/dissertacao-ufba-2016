160       /**
161        * Open a JAR/ZIP file. This routine caches the last JAR file opened to save effort when the same file is accessed again and
162        * again. DO NOT close the file returned by this routine until you've finished with this OfflineInstrumente completely. Also, this
163        * JarFile will be closed the next time someone calls openCachedJar.
164        */
165       private JarFile openCachedJar(File file) throws IOException {
166         if (cachedJarFile != null && cachedJarFile.equals(file)) {
167           return cachedJar;
168         } else {
169           if (cachedJar != null) {
170             cachedJar.close();
171           }
172           cachedJarFile = file;
173           cachedJar = new JarFile(file);
174           return cachedJar;
175         }
176       }

256       /**
257        * Add a directory containing class files to instrument. All subdirectories are also scanned.
258        * 
259        * @throws IllegalArgumentException if d is null
260        */
261       final public void addInputDirectory(File baseDirectory, File d) throws IOException, IllegalArgumentException {
262         if (d == null) {
263           throw new IllegalArgumentException("d is null");
264         }
265         File[] fs = d.listFiles(new FileFilter() {
266           @Override
267           public boolean accept(File f) {
268             return f.isDirectory() || f.getName().endsWith(".class");
269           }
270         });
271         if (fs == null) {
272           throw new IllegalArgumentException("bad directory " + d.getAbsolutePath());
273         }
274         for (int i = 0; i < fs.length; i++) {
275           File f = fs[i];
276           if (f.isDirectory()) {
277             addInputDirectory(baseDirectory, f);
278           } else {
279             addInputClass(baseDirectory, f);
280           }
281         }
282       }

284       /**
285        * Add something to instrument --- the name of a JAR file, a class file, a directory or an entry within a jar file (as
286        * filename#entryname). If we can't identify it, nothing is added and we return false.
287        * 
288        * @throws IllegalArgumentException if a is null
289        */
290       final public boolean addInputElement(File baseDirectory, String a) throws IOException {
291         if (a == null) {
292           throw new IllegalArgumentException("a is null");
293         }
294         try {
295           int poundIndex = a.indexOf('#');
296           if (poundIndex > 0) {
297             addInputJarEntry(new File(a.substring(0, poundIndex)), a.substring(poundIndex + 1));
298             return true;
299           }
300           File f = new File(a);
301           if (f.isDirectory()) {
302             addInputDirectory(baseDirectory, f);
303             return true;
304           } else if (f.exists()) {
305             if (a.endsWith(".class")) {
306               addInputClass(baseDirectory, f);
307               return true;
308             } else if (a.endsWith(".jar") || a.endsWith(".zip")) {
309               addInputJar(new File(a));
310               return true;
311             }
312           }
313         } catch (IOException ex) {
314           throw new IOException("Error reading input element '" + a + "': " + ex.getMessage());
315         }
316         return false;
317       }

319       /**
320        * Parse an argument list to find elements to instrument and the name of the output file. The "-o filename" option selects the
321        * output JAR file name. Any other argument not starting with "-" is added to the list of elements to instrument, if it appears to
322        * be the name of a class file, JAR file, or directory. If any argument starting with "--" is encountered, the rest of the
323        * command-line is considered leftover
324        * 
325        * @return the arguments that were not understood
326        * @throws IllegalArgumentException if args == null
327        */
328       final public String[] parseStandardArgs(String[] args) throws IllegalArgumentException, IOException {
329         if (args == null) {
330           throw new IllegalArgumentException("args == null");
331         }
332         ArrayList<String> leftover = new ArrayList<String>();
333     
334         for (int i = 0; i < args.length; i++) {
335           String a = args[i];
336           if (a == null) {
337             throw new IllegalArgumentException("args[" + i + "] is null");
338           }
339           if (a.equals("-o") && i + 1 < args.length) {
340             setOutputJar(new File(args[i + 1]));
341             i++;
342             continue;
343           } else if (!a.startsWith("-")) {
344             if (addInputElement(new File(a), a)) {
345               continue;
346             }
347           } else if (a.startsWith("--")) {
348             for (int j = i; j < args.length; j++) {
349               leftover.add(args[j]);
350             }
351             break;
352           }
353           leftover.add(a);
354         }
355     
356         String[] r = new String[leftover.size()];
357         leftover.toArray(r);
358         return r;
359       }

381       final protected Object internalNextClass() throws IOException {
382         while (true) {
383           if (inputIndex >= inputs.size()) {
384             return null;
385           } else {
386             Input in = inputs.get(inputIndex);
387             inputIndex++;
388             if (ignoringInputs.get(inputIndex - 1) || !in.isClass()) {
389               continue;
390             }
391             BufferedInputStream s = new BufferedInputStream(in.open());
392             try {
393               Object r = makeClassFromStream(in.getInputName(), s);
394               String name = getClassName(r);
395               in.setClassName(name);
396               return r;
397             } finally {
398               s.close();
399             }
400           }
401         }
402       }

430       final protected boolean internalOutputModifiedClass(Object cf, String name, Object mods) throws IOException {
431         makeOutputJar();
432         if (entryNames.contains(name)) {
433           return false;
434         } else {
435           putNextEntry(new ZipEntry(name));
436           BufferedOutputStream s = new BufferedOutputStream(outputJar);
437           writeClassTo(cf, mods, s);
438           s.flush();
439           outputJar.closeEntry();
440           return true;
441         }
442       }

473       private static byte[] cachedBuf;
474     
475       private static synchronized byte[] makeBuf() {
476         if (cachedBuf != null) {
477           byte[] r = cachedBuf;
478           cachedBuf = null;
479           return r;
480         } else {
481           return new byte[60000];
482         }
483       }
484     
485       private static synchronized void releaseBuf(byte[] buf) {
486         cachedBuf = buf;
487       }
488     
489       public static void copyStream(InputStream in, OutputStream out) throws IllegalArgumentException, IOException {
490         if (in == null) {
491           throw new IllegalArgumentException("in == null");
492         }
493         byte[] buf = makeBuf();
494         try {
495           while (true) {
496             int read = in.read(buf);
497             if (read < 0) {
498               return;
499             }
500             out.write(buf, 0, read);
501           }
502         } finally {
503           releaseBuf(buf);
504         }
505       }

509        *

530       /**
531        * Call this to copy any unmodified classes to the output. This is called automatically by close(); you should only call this if
532        * you want to write an entry to the JAR file *after* the unmodified classes. This will only ever be called once per output JAR.
533        */
534       final public void writeUnmodifiedClasses() throws IOException, IllegalStateException {
535         passUnmodifiedClasses = false;
536         makeOutputJar();
537         for (int i = 0; i < inputs.size(); i++) {
538           Input in = inputs.get(i);
539           if (!in.isClass()) {
540             if (in instanceof JarInput) {
541               JarInput jin = (JarInput) in;
542               ZipEntry entry = jin.getEntry();
543               InputStream s = jin.open();
544               try {
545                 ZipEntry newEntry = new ZipEntry(entry.getName());
546                 newEntry.setComment(entry.getComment());
547                 newEntry.setExtra(entry.getExtra());
548                 newEntry.setTime(entry.getTime());
549                 putNextEntry(newEntry);
550                 copyStream(s, outputJar);
551                 outputJar.closeEntry();
552               } finally {
553                 s.close();
554               }
555             } else {
556               throw new Error("Unknown non-class input: " + in);
557             }
558           } else {
559             String name = in.getClassName();
560             if (name == null) {
561               BufferedInputStream s = new BufferedInputStream(in.open(), 65536);
562               try {
563                 Object cl = makeClassFromStream(in.getInputName(), s);
564                 String entryName = toEntryName(getClassName(cl));
565                 if (!entryNames.contains(entryName)) {
566                   putNextEntry(new ZipEntry(entryName));
567                   BufferedOutputStream clOut = new BufferedOutputStream(outputJar);
568                   writeClassTo(cl, null, clOut);
569                   clOut.flush();
570                   outputJar.closeEntry();
571                 }
572               } finally {
573                 s.close();
574               }
575             } else {
576               String entryName = toEntryName(name);
577               if (!entryNames.contains(entryName)) {
578                 BufferedInputStream s = new BufferedInputStream(in.open());
579                 try {
580                   putNextEntry(new ZipEntry(entryName));
581                   BufferedOutputStream clOut = new BufferedOutputStream(outputJar);
582                   copyStream(s, clOut);
583                   clOut.flush();
584                   outputJar.closeEntry();
585                 } finally {
586                   s.close();
587                 }
588               }
589             }
590           }
591         }
592       }

594       /**
595        * Call this when you're done modifying classes.
596        */
597       final public void close() throws IOException, IllegalStateException {
598         if (passUnmodifiedClasses) {
599           writeUnmodifiedClasses();
600         }
601     
602         if (outputJar != null) {
603           outputJar.close();
604         }
605     
606         if (cachedJar != null) {
607           cachedJar.close();
608         }
609       }

611       private void putNextEntry(ZipEntry newEntry) throws IOException, IllegalStateException {
612         if (outputJar == null) {
613           throw new IllegalStateException();
614         }
615         outputJar.putNextEntry(newEntry);
616         entryNames.add(newEntry.getName());
617         if (manifestBuilder != null) {
618           manifestBuilder.addEntry(newEntry);
619         }
620       }
