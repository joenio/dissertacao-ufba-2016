38     /**
39      * tests for weird corner cases, such as when the input program doesn't verify
40      * 
41      * @author sfink
42      */
43     public class CornerCasesTest extends WalaTestCase {
44     
45       private static final ClassLoader MY_CLASSLOADER = CornerCasesTest.class.getClassLoader();
46     
47       /**
48        * test that getMethod() works even if a declared ancestor interface doesn't
49        * exist
50        * 
51        * @throws ClassHierarchyException
52        * @throws IOException 
53        */
54       @Test public void testBug38484() throws ClassHierarchyException, IOException {
55         AnalysisScope scope = null;
56         scope = AnalysisScopeReader.readJavaScope(TestConstants.WALA_TESTDATA, (new FileProvider()).getFile("J2SEClassHierarchyExclusions.txt"), MY_CLASSLOADER);
57         ClassHierarchy cha = ClassHierarchy.make(scope);
58         TypeReference t = TypeReference.findOrCreateClass(scope.getApplicationLoader(), "cornerCases", "YuckyInterface");
59         IClass klass = cha.lookupClass(t);
60         Assert.assertTrue(klass != null);
61         IMethod m = klass.getMethod(new Selector(Atom.findOrCreateAsciiAtom("x"), Descriptor.findOrCreateUTF8("()V")));
62         Assert.assertTrue(m == null);
63       }
64     
65       /**
66        * test that type inference works in the presence of a getfield where the
67        * field's declared type cannot be loaded
68        * 
69        * @throws ClassHierarchyException
70        * @throws IOException 
71        */
72       @Test public void testBug38540() throws ClassHierarchyException, IOException {
73         AnalysisScope scope = null;
74         scope = AnalysisScopeReader.readJavaScope(TestConstants.WALA_TESTDATA, (new FileProvider()).getFile("J2SEClassHierarchyExclusions.txt"), MY_CLASSLOADER);
75         AnalysisOptions options = new AnalysisOptions();
76         ClassHierarchy cha = ClassHierarchy.make(scope);
77         TypeReference t = TypeReference.findOrCreateClass(scope.getApplicationLoader(), "cornerCases", "Main");
78         IClass klass = cha.lookupClass(t);
79         Assert.assertTrue(klass != null);
80         ShrikeCTMethod m = (ShrikeCTMethod) klass.getMethod(new Selector(Atom.findOrCreateAsciiAtom("foo"), Descriptor
81             .findOrCreateUTF8("()Ljava/lang/Object;")));
82         Assert.assertTrue(m != null);
83         IR ir = new AnalysisCache().getSSACache().findOrCreateIR(m, Everywhere.EVERYWHERE, options.getSSAOptions());
84         TypeInference.make(ir, false);
85       }
