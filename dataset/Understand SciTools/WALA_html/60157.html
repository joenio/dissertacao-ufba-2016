22     /**
23      * Convenience methods for navigating a {@link ControlFlowGraph}.
24      */
25     public class Util {

27       /**
28        * @return the last instruction in basic block b, as stored in the instruction array for cfg
29        */
30       public static SSAInstruction getLastInstruction(ControlFlowGraph cfg, IBasicBlock b) {
31         if (b == null) {
32           throw new IllegalArgumentException("b is null");
33         }
34         if (cfg == null) {
35           throw new IllegalArgumentException("G is null");
36         }
37         return (SSAInstruction) cfg.getInstructions()[b.getLastInstructionIndex()];
38       }
39     
40       /**
41        * Does basic block b end with a conditional branch instruction?
42        */
43       public static boolean endsWithConditionalBranch(ControlFlowGraph G, IBasicBlock b) {
44         return getLastInstruction(G, b) instanceof SSAConditionalBranchInstruction;
45       }
46     
47       /**
48        * Does basic block b end with a switch instruction?
49        */
50       public static boolean endsWithSwitch(ControlFlowGraph G, IBasicBlock b) {
51         return getLastInstruction(G, b) instanceof SSASwitchInstruction;
52       }
53     
54       /**
55        * Given that b falls through to the next basic block, what basic block does it fall through to?
56        */
57       public static <I, T extends IBasicBlock<I>> T getFallThruBlock(ControlFlowGraph<I, T> G, T b) {
58         if (b == null) {
59           throw new IllegalArgumentException("b is null");
60         }
61         if (G == null) {
62           throw new IllegalArgumentException("G is null");
63         }
64         return G.getBlockForInstruction(b.getLastInstructionIndex() + 1);
65       }
66     
67       /**
68        * Given that b ends with a conditional branch, return the basic block to
69        * which control transfers if the branch is not taken.
70        */
71       public static <I, T extends IBasicBlock<I>> T getNotTakenSuccessor(ControlFlowGraph<I, T> G, T b) {
72         if (G == null) {
73           throw new IllegalArgumentException("G is null");
74         }
75         if (!endsWithConditionalBranch(G, b)) {
76           throw new IllegalArgumentException(b.toString() + " does not end with a conditional branch");
77         }
78         return getFallThruBlock(G, b);
79       }
80     
81       /**
82        * Given that b ends with a conditional branch, return the basic block to
83        * which control transfers if the branch is taken.
84        */
85       public static <I, T extends IBasicBlock<I>> T getTakenSuccessor(ControlFlowGraph<I, T> G, T b) {
86         if (G == null) {
87           throw new IllegalArgumentException("G is null");
88         }
89         if (!endsWithConditionalBranch(G, b)) {
90           throw new IllegalArgumentException(b.toString() + " does not end with a conditional branch");
91         }
92         T fs = getNotTakenSuccessor(G, b);
93         for (Iterator<? extends T> ss = G.getSuccNodes(b); ss.hasNext();) {
94           T s = ss.next();
95           if (s != fs)
96             return s;
97         }
98     
99         // under pathological conditions, b may have exactly one successor (in other
100         // words, the
101         // branch is irrelevant
102         return fs;
103       }

105       /**
106        * When the tested value of the switch statement in b has value c, which basic
107        * block does control transfer to.
108        */
109       public static <I, T extends IBasicBlock<I>> T resolveSwitch(ControlFlowGraph<I, T> G, T b, int c) {
110         assert endsWithSwitch(G, b);
111         SSASwitchInstruction s = (SSASwitchInstruction) getLastInstruction(G, b);
112         int[] casesAndLabels = s.getCasesAndLabels();
113         for (int i = 0; i < casesAndLabels.length; i += 2)
114           if (casesAndLabels[i] == c)
115             return G.getBlockForInstruction(casesAndLabels[i + 1]);
116     
117         return G.getBlockForInstruction(s.getDefault());
118       }
119     
120       /**
121        * Is block s the default case for the switch instruction which is the last instruction of block b?
122        */
123       public static <I, T extends IBasicBlock<I>> boolean isSwitchDefault(ControlFlowGraph<I, T> G, T b, T s) {
124         if (G == null) {
125           throw new IllegalArgumentException("G is null");
126         }
127         assert endsWithSwitch(G, b);
128         SSASwitchInstruction sw = (SSASwitchInstruction) getLastInstruction(G, b);
129         assert G.getBlockForInstruction(sw.getDefault()) != null;
130         return G.getBlockForInstruction(sw.getDefault()).equals(s);
131       }

133       /**
134        * When a switch statement at the end of block b transfers control to block s,
135        * which case was taken? TODO: Is this correct? Can't we have multiple cases
136        * that apply? Check on this.
137        */
138       public static <I, T extends IBasicBlock<I>> int getSwitchLabel(ControlFlowGraph<I, T> G, T b, T s) {
139         assert endsWithSwitch(G, b);
140         SSASwitchInstruction sw = (SSASwitchInstruction) getLastInstruction(G, b);
141         int[] casesAndLabels = sw.getCasesAndLabels();
142         for (int i = 0; i < casesAndLabels.length; i += 2) {
143           if (G.getBlockForInstruction(casesAndLabels[i + 1]).equals(s)) {
144             return casesAndLabels[i];
145           }
146         }
147     
148         Assertions.UNREACHABLE();
149         return -1;
150       }

152       /**
153        * To which {@link IBasicBlock} does control flow from basic block bb, which ends in a
154        * conditional branch, when the conditional branch operands evaluate to the
155        * constants c1 and c2, respectively.
156        * 
157        * Callers must resolve the constant values from the {@link SymbolTable}
158        * before calling this method. These integers are <bf>not</bf> value numbers;
159        */
160       public static <I, T extends IBasicBlock<I>> T resolveBranch(ControlFlowGraph<I, T> G, T bb, int c1, int c2) {
161         SSAConditionalBranchInstruction c = (SSAConditionalBranchInstruction) getLastInstruction(G, bb);
162         switch ((ConditionalBranchInstruction.Operator) c.getOperator()) {
163         case EQ:
164           if (c1 == c2)
165             return getTakenSuccessor(G, bb);
166           else
167             return getNotTakenSuccessor(G, bb);
168         case NE:
169           if (c1 != c2)
170             return getTakenSuccessor(G, bb);
171           else
172             return getNotTakenSuccessor(G, bb);
173         case LT:
174           if (c1 < c2)
175             return getTakenSuccessor(G, bb);
176           else
177             return getNotTakenSuccessor(G, bb);
178         case GE:
179           if (c1 >= c2)
180             return getTakenSuccessor(G, bb);
181           else
182             return getNotTakenSuccessor(G, bb);
183         case GT:
184           if (c1 > c2)
185             return getTakenSuccessor(G, bb);
186           else
187             return getNotTakenSuccessor(G, bb);
188         case LE:
189           if (c1 <= c2)
190             return getTakenSuccessor(G, bb);
191           else
192             return getNotTakenSuccessor(G, bb);
193         }
194     
195         Assertions.UNREACHABLE();
196         return null;
197       }

199       /**
200        * Given that a is a predecessor of b in the cfg ..
201        * 
202        * When we enumerate the predecessors of b in order, which is the first index
203        * in this order in which a appears? Note that this order corresponds to the
204        * order of operands in a phi instruction.
205        */
206       public static <I, T extends IBasicBlock<I>> int whichPred(ControlFlowGraph<I, T> cfg, T a, T b) {
207         if (cfg == null) {
208           throw new IllegalArgumentException("cfg is null");
209         }
210         if (a == null) {
211           throw new IllegalArgumentException("a is null");
212         }
213         if (b == null) {
214           throw new IllegalArgumentException("b is null");
215         }
216         int i = 0;
217         for (Iterator it = cfg.getPredNodes(b); it.hasNext();) {
218           if (it.next().equals(a)) {
219             return i;
220           }
221           i++;
222         }
223         Assertions.UNREACHABLE("Invalid: a must be a predecessor of b! " + a + " " + b);
224         return -1;
225       }
