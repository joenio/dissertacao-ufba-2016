21     public final class SourcePositionTableReader extends AttributeReader {

23       protected SourcePositionTableReader(AttrIterator attr) throws InvalidClassFileException {
24         super(attr, CRTable.ATTRIBUTE_NAME);
25       }

27       public static final class Position implements Comparable<Object> {
28         
29         public final int firstLine;
30         public final int lastLine;
31         public final int firstCol;
32         public final int lastCol;
33         
34         private Position(int firstLine, int lastLine, int firstCol, int lastCol) {
35           this.firstLine = firstLine;
36           this.lastLine = lastLine;
37           this.firstCol = firstCol;
38           this.lastCol = lastCol;
39         }
40         
41         public int compareTo(Object o) {
42           if (o instanceof Position) {
43             Position p = (Position) o;
44             if (firstLine != p.firstLine) {
45               return firstLine - p.firstLine;
46             } else if (firstCol != p.firstCol) {
47               return firstCol - p.firstCol;
48             } else if (lastLine != p.lastLine) {
49               return lastLine - p.lastLine;
50             } else if (lastCol != p.lastCol) {
51               return lastCol - p.lastCol;
52             } else {
53               return 0;
54             }
55           } else {
56             return -1;
57           }
58         } 
59       }

61       public static Position findParameterPosition(int methodNr, CodeReader code) throws InvalidClassFileException, IOException {
62         if (code == null) {
63           throw new IllegalArgumentException();
64         }
65     
66         Position params = null;
67     
68         // read parameter positions
69         {
70           ClassReader.AttrIterator cIter = new ClassReader.AttrIterator();
71           ClassReader cr = code.getClassReader();
72           cr.initMethodAttributeIterator(methodNr, cIter);
73           
74           for (;cIter.isValid(); cIter.advance()) {
75             if (MethodPositions.ATTRIBUTE_NAME.equals(cIter.getName())) {
76               byte data[] = getData(cr, cIter.getRawOffset(), cIter.getRawSize());
77               MethodPositions mPos = new MethodPositions(data);
78               Range r = mPos.getMethodInfo();
79               params = convert(r);
80             }
81           }
82         }
83       
84         return params;
85       }
86       
87       public static Position[] makeBytecodeToPositionMap(CodeReader code) throws InvalidClassFileException, IOException {
88         if (code == null) {
89           throw new IllegalArgumentException();
90         }
91         
92         Position pos[] = null;
93         ClassReader.AttrIterator iter = new ClassReader.AttrIterator();
94         code.initAttributeIterator(iter);
95         
96         for (; iter.isValid(); iter.advance()) {
97           if (CRTable.ATTRIBUTE_NAME.equals(iter.getName())) {
98             if (pos == null) {
99               pos = new Position[code.getBytecodeLength()];
100             }
101             
102             SourcePositionTableReader spRead = new SourcePositionTableReader(iter);
103             spRead.fillBytecodeToPositionMap(pos);
104           }
105         }
106     
107         if (pos != null) {
108           // fill in gaps
109           Position last = new Position(0, 0, 0, 0);
110           for (int i = 0; i < pos.length; i++) {
111             Position cur = pos[i];
112             if (cur == null) {
113               pos[i] = last;
114             } else {
115               last = cur;
116             }
117           }
118         }
119         
120         return pos;
121       }
122     
123       private static final int ATTRIBUTE_HEADER_SIZE = 6;
124       
125       private static final byte[] getData(ClassReader cr, int rawOffset, int rawSize) {
126         // prepare raw data of attribute to pass to sourceinfo
127         byte klass[] = cr.getBytes();
128         int size = rawSize - ATTRIBUTE_HEADER_SIZE;
129         byte data[] = new byte[size];
130         System.arraycopy(klass, rawOffset + ATTRIBUTE_HEADER_SIZE, data, 0, size);
131         
132         return data;
133       }
134       
135       private void fillBytecodeToPositionMap(Position[] pos) throws IOException {
136         byte tableData[] = getData(getClassReader(), getRawOffset(), getRawSize());
137         
138         CRTable crTable = new CRTable(tableData);
139         
140         for (int pc = 0; pc < pos.length; pc++) {
141           Range r = crTable.getSourceInfo(pc);
142           Position p = convert(r);
143           pos[pc] = p;
144         }
145       }
146       
147       private final static Position convert(Range r) {
148         Position pos = null;
149         
150         if (r != null) {
151           com.ibm.wala.sourcepos.Position start = r.getStartPosition();
152           com.ibm.wala.sourcepos.Position end = r.getEndPosition();
153           
154           if (start != null && !start.isUndefined()) {
155             if (end != null && !end.isUndefined()) {
156               pos = new Position(start.getLine(), end.getLine(), start.getColumn(), end.getColumn());
157             } else {
158               pos = new Position(start.getLine(), start.getLine(), -1, -1);
159             }
160           }
161         }    
162     
163         return pos;
164       }
165     
166       public static Position[] makeLineNumberToPositionMap(int[] lineNumberMap) {
167         Position pos[] = new Position[lineNumberMap.length];
168         
169         for (int i = 0; i < pos.length; i++) {
170           int line = lineNumberMap[i];
171           pos[i] = new Position(line, line, -1, -1);
172         }
173         
174         return pos;
175       }
