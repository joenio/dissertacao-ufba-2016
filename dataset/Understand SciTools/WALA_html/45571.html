11     package com.ibm.wala.ssa;
12     
13     import com.ibm.wala.shrikeBT.IUnaryOpInstruction;
14     
15     /**
16      * An SSA instruction for some unary operator.
17      * 
18      * @see IUnaryOpInstruction for a list of operators
19      */
20     public class SSAUnaryOpInstruction extends SSAAbstractUnaryInstruction {
21     
22       private final IUnaryOpInstruction.IOperator operator;
23     
24       public SSAUnaryOpInstruction(int iindex, IUnaryOpInstruction.IOperator operator, int result, int val) {
25         super(iindex, result, val);
26         this.operator = operator;
27       }
28     
29       @Override
30       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) throws IllegalArgumentException {
31         if (uses != null && uses.length == 0) {
32           throw new IllegalArgumentException("(uses != null) and (uses.length == 0)");
33         }
34         return insts.UnaryOpInstruction(iindex, operator, defs == null || defs.length == 0 ? result : defs[0], uses == null ? val : uses[0]);
35       }
36     
37       @Override
38       public String toString(SymbolTable symbolTable) {
39         return getValueString(symbolTable, result) + " = " + operator + " " + getValueString(symbolTable, val);
40       }
41     
42       /**
43        * @see com.ibm.wala.ssa.SSAInstruction#visit(IVisitor)
44        */
45       @Override
46       public void visit(IVisitor v) throws NullPointerException {
47         v.visitUnaryOp(this);
48       }
49     
50       public IUnaryOpInstruction.IOperator getOpcode() {
51         return operator;
52       }
53     
54      }
