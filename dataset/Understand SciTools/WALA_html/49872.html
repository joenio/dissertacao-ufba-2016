11     package com.ibm.wala.util.collections;
12     
13     import java.util.HashMap;
14     import java.util.Iterator;
15     import java.util.NoSuchElementException;
16     
17     import com.ibm.wala.util.debug.Assertions;
18     import com.ibm.wala.util.debug.UnimplementedError;
19     import com.ibm.wala.util.intset.OrdinalSetMapping;
20     
21     /**
22      * A bit set mapping based on an immutable object array. This is not terribly efficient, but is useful for prototyping.
23      */
24     public class ObjectArrayMapping<T> implements OrdinalSetMapping<T> {
25     
26       final private T[] array;
27     
28       /**
29        * A mapping from object to Integer
30        */
31       final private HashMap<T, Integer> map = HashMapFactory.make();
32     
33       public ObjectArrayMapping(final T[] array) {
34         if (array == null) {
35           throw new IllegalArgumentException("null array");
36         }
37         this.array = array;
38         for (int i = 0; i < array.length; i++) {
39           map.put(array[i], Integer.valueOf(i));
40         }
41       }
42     
43       @Override
44       public T getMappedObject(int n) throws NoSuchElementException {
45         try {
46           return array[n];
47         } catch (ArrayIndexOutOfBoundsException e) {
48           throw new IllegalArgumentException("invalid n: " + n);
49         }
50       }
51     
52       @Override
53       public int getMappedIndex(Object o) {
54         if (map.get(o) == null) {
55           return -1;
56         }
57         return map.get(o).intValue();
58       }
59     
60       @Override
61       public boolean hasMappedIndex(Object o) {
62         return map.get(o) != null;
63       }
64     
65       @Override
66       public Iterator<T> iterator() {
67         return map.keySet().iterator();
68       }
69     
70       @Override
71       public int add(Object o) throws UnimplementedError {
72         Assertions.UNREACHABLE();
73         return 0;
74       }
75     
76       @Override
77       public int getMaximumIndex() {
78         return array.length - 1;
79       }
80     
81       @Override
82       public int getSize() {
83         return map.size();
84       }
85     }
