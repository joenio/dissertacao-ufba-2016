84       /**
85        * For each statement s, return the set of statements that may def the heap value read by s.
86        * 
87        * @param node the node we are computing heap reaching defs for
88        * @param ir IR for the node
89        * @param pa governing pointer analysis
90        * @param mod the set of heap locations which may be written (transitively) by this node. These are logically return values in the
91        *          SDG.
92        * @param statements the statements whose def-use are considered interesting
93        * @param exclusions heap locations that should be excluded from data dependence tracking
94        * 
95        * @throws IllegalArgumentException if pa is null
96        * @throws IllegalArgumentException if statements is null
97        */
98       @SuppressWarnings("unused")
99       public Map<Statement, OrdinalSet<Statement>> computeReachingDefs(CGNode node, IR ir, PointerAnalysis<InstanceKey> pa,
100           Map<CGNode, OrdinalSet<PointerKey>> mod, Collection<Statement> statements, HeapExclusions exclusions, CallGraph cg) {
101     
102         if (statements == null) {
103           throw new IllegalArgumentException("statements is null");
104         }
105         if (pa == null) {
106           throw new IllegalArgumentException("pa is null");
107         }
108         if (VERBOSE | DEBUG) {
109           System.err.println("Reaching Defs " + node);
110           System.err.println(statements.size());
111         }
112     
113         if (DEBUG) {
114           System.err.println(ir);
115         }
116     
117         // create a control flow graph with one instruction per basic block.
118         ExplodedControlFlowGraph cfg = ExplodedControlFlowGraph.make(ir);
119     
120         // create a mapping between statements and integers, used in bit vectors
121         // shortly
122         OrdinalSetMapping<Statement> domain = createStatementDomain(statements);
123     
124         // map SSAInstruction indices to statements
125         Map<Integer, NormalStatement> ssaInstructionIndex2Statement = mapInstructionsToStatements(domain);
126     
127         // solve reaching definitions as a dataflow problem
128         BitVectorFramework<IExplodedBasicBlock, Statement> rd = new BitVectorFramework<IExplodedBasicBlock, Statement>(cfg, new RD(node,
129             cfg, pa, domain, ssaInstructionIndex2Statement, exclusions), domain);
130         if (VERBOSE) {
131           System.err.println("Solve ");
132         }
133         BitVectorSolver<? extends ISSABasicBlock> solver = new BitVectorSolver<IExplodedBasicBlock>(rd);
134         try {
135           solver.solve(null);
136         } catch (CancelException e) {
137           throw new CancelRuntimeException(e);
138         }
139         if (VERBOSE) {
140           System.err.println("Solved. ");
141         }
142         return makeResult(solver, domain, node, heapModel, pa, mod, cfg,
143             ssaInstructionIndex2Statement, exclusions, cg);
144       }

153         RDMap(BitVectorSolver<? extends ISSABasicBlock> solver, OrdinalSetMapping<Statement> domain, CGNode node, ExtendedHeapModel h,
154             PointerAnalysis<InstanceKey> pa, Map<CGNode, OrdinalSet<PointerKey>> mod, ExplodedControlFlowGraph cfg,

171         private void eagerPopulate(Map<PointerKey, MutableIntSet> pointerKeyMod, BitVectorSolver<? extends ISSABasicBlock> solver,
172             OrdinalSetMapping<Statement> domain, CGNode node, ExtendedHeapModel h, PointerAnalysis<InstanceKey> pa,
173             Map<CGNode, OrdinalSet<PointerKey>> mod, ExplodedControlFlowGraph cfg,
174             Map<Integer, NormalStatement> ssaInstruction2Statement) {
175           for (Statement s : domain) {
176             delegate.put(s, computeResult(s, pointerKeyMod, solver, domain, node, h, pa, mod, cfg, ssaInstruction2Statement));
177           }
178         }

180         /**
181          * For each pointerKey, which statements may def it
182          */
183         private Map<PointerKey, MutableIntSet> initPointerKeyMod(OrdinalSetMapping<Statement> domain, CGNode node, ExtendedHeapModel h,
184             PointerAnalysis<InstanceKey> pa) {
185           Map<PointerKey, MutableIntSet> pointerKeyMod = HashMapFactory.make();
186           for (Statement s : domain) {
187             switch (s.getKind()) {
188             case HEAP_PARAM_CALLEE:
189             case HEAP_RET_CALLER: {
190               HeapStatement hs = (HeapStatement) s;
191               MutableIntSet set = findOrCreateIntSet(pointerKeyMod, hs.getLocation());
192               set.add(domain.getMappedIndex(s));
193               break;
194             }
195             default: {
196               Collection<PointerKey> m = getMod(s, node, h, pa, exclusions);
197               for (PointerKey p : m) {
198                 MutableIntSet set = findOrCreateIntSet(pointerKeyMod, p);
199                 set.add(domain.getMappedIndex(s));
200               }
201               break;
202             }
203             }
204           }
205           return pointerKeyMod;
206         }

208         private MutableIntSet findOrCreateIntSet(Map<PointerKey, MutableIntSet> map, PointerKey key) {
209           MutableIntSet result = map.get(key);
210           if (result == null) {
211             result = MutableSparseIntSet.makeEmpty();
212             map.put(key, result);
213           }
214           return result;
215         }

304         /**
305          * For a statement s, compute the set of statements that may def the heap value read by s.
306          */
307         OrdinalSet<Statement> computeResult(Statement s, Map<PointerKey, MutableIntSet> pointerKeyMod,
308             BitVectorSolver<? extends ISSABasicBlock> solver, OrdinalSetMapping<Statement> domain, CGNode node, ExtendedHeapModel h,
309             PointerAnalysis<InstanceKey> pa, Map<CGNode, OrdinalSet<PointerKey>> mod, ExplodedControlFlowGraph cfg,
310             Map<Integer, NormalStatement> ssaInstructionIndex2Statement) {
311           switch (s.getKind()) {
312           case NORMAL:
313             NormalStatement n = (NormalStatement) s;
314             Collection<PointerKey> ref = modRef.getRef(node, h, pa, n.getInstruction(), exclusions);
315             if (!ref.isEmpty()) {
316               ISSABasicBlock bb = cfg.getBlockForInstruction(n.getInstructionIndex());
317               BitVectorVariable v = solver.getIn(bb);
318               MutableSparseIntSet defs = MutableSparseIntSet.makeEmpty();
319               if (v.getValue() != null) {
320                 for (PointerKey p : ref) {
321                   if (pointerKeyMod.get(p) != null) {
322                     defs.addAll(pointerKeyMod.get(p).intersection(v.getValue()));
323                   }
324                 }
325               }
326               return new OrdinalSet<Statement>(defs, domain);
327             } else {
328               return OrdinalSet.empty();
329             }
330           case HEAP_RET_CALLEE: {
331             HeapStatement.HeapReturnCallee r = (HeapStatement.HeapReturnCallee) s;
332             PointerKey p = r.getLocation();
333             BitVectorVariable v = solver.getIn(cfg.exit());
334             if (DEBUG) {
335               System.err.println("computeResult " + cfg.exit() + " " + s + " " + pointerKeyMod.get(p) + " " + v);
336             }
337             if (pointerKeyMod.get(p) == null) {
338               return OrdinalSet.empty();
339             }
340             return new OrdinalSet<Statement>(pointerKeyMod.get(p).intersection(v.getValue()), domain);
341           }
342           case HEAP_RET_CALLER: {
343             HeapStatement.HeapReturnCaller r = (HeapStatement.HeapReturnCaller) s;
344             ISSABasicBlock bb = cfg.getBlockForInstruction(r.getCallIndex());
345             BitVectorVariable v = solver.getIn(bb);
346             if (allCalleesMod(cg, r, mod) || pointerKeyMod.get(r.getLocation()) == null || v.getValue() == null) {
347               // do nothing ... force flow into and out of the callees
348               return OrdinalSet.empty();
349             } else {
350               // the defs that flow to the call may flow to this return, since
351               // the callees may have no relevant effect.
352               return new OrdinalSet<Statement>(pointerKeyMod.get(r.getLocation()).intersection(v.getValue()), domain);
353             }
354           }
355           case HEAP_PARAM_CALLER: {
356             HeapStatement.HeapParamCaller r = (HeapStatement.HeapParamCaller) s;
357             NormalStatement call = ssaInstructionIndex2Statement.get(r.getCallIndex());
358             ISSABasicBlock callBlock = cfg.getBlockForInstruction(call.getInstructionIndex());
359             if (callBlock.isEntryBlock()) {
360               int x = domain.getMappedIndex(new HeapStatement.HeapParamCallee(node, r.getLocation()));
361               assert x >= 0;
362               IntSet xset = SparseIntSet.singleton(x);
363               return new OrdinalSet<Statement>(xset, domain);
364             }
365             BitVectorVariable v = solver.getIn(callBlock);
366             if (pointerKeyMod.get(r.getLocation()) == null || v.getValue() == null) {
367               // do nothing ... force flow into and out of the callees
368               return OrdinalSet.empty();
369             } else {
370               return new OrdinalSet<Statement>(pointerKeyMod.get(r.getLocation()).intersection(v.getValue()), domain);
371             }
372           }
373           case NORMAL_RET_CALLEE:
374           case NORMAL_RET_CALLER:
375           case PARAM_CALLEE:
376           case PARAM_CALLER:
377           case EXC_RET_CALLEE:
378           case EXC_RET_CALLER:
379           case PHI:
380           case PI:
381           case CATCH:
382           case METHOD_ENTRY:
383           case METHOD_EXIT:
384             return OrdinalSet.empty();
385           case HEAP_PARAM_CALLEE:
386             // no statements in this method will def the heap being passed in
387             return OrdinalSet.empty();
388           default:
389             Assertions.UNREACHABLE(s.getKind().toString());
390             return null;
391           }
392         }

407       /**
408        * Do all callees corresponding to the given call site def the pointer key being tracked by r?
409        */
410       private static boolean allCalleesMod(CallGraph cg, HeapReturnCaller r, Map<CGNode, OrdinalSet<PointerKey>> mod) {
411         Collection<CGNode> targets = cg.getPossibleTargets(r.getNode(), r.getCall().getCallSite());
412         if (targets.isEmpty()) {
413           return false;
414         }
415         for (CGNode t : targets) {
416           if (!mod.get(t).contains(r.getLocation())) {
417             return false;
418           }
419         }
420         return true;
421       }

423       private Collection<PointerKey> getMod(Statement s, CGNode n, ExtendedHeapModel h, PointerAnalysis<InstanceKey> pa, HeapExclusions exclusions) {
424         switch (s.getKind()) {
425         case NORMAL:
426           NormalStatement ns = (NormalStatement) s;
427           return modRef.getMod(n, h, pa, ns.getInstruction(), exclusions);
428         case HEAP_PARAM_CALLEE:
429         case HEAP_RET_CALLER:
430           HeapStatement hs = (HeapStatement) s;
431           return Collections.singleton(hs.getLocation());
432         case HEAP_RET_CALLEE:
433         case HEAP_PARAM_CALLER:
434         case EXC_RET_CALLEE:
435         case EXC_RET_CALLER:
436         case NORMAL_RET_CALLEE:
437         case NORMAL_RET_CALLER:
438         case PARAM_CALLEE:
439         case PARAM_CALLER:
440         case PHI:
441         case PI:
442         case METHOD_ENTRY:
443         case METHOD_EXIT:
444         case CATCH:
445           // doesn't mod anything in the heap.
446           return Collections.emptySet();
447         default:
448           Assertions.UNREACHABLE(s.getKind() + " " + s.toString());
449           return null;
450         }
451       }

476       private class RD implements ITransferFunctionProvider<IExplodedBasicBlock, BitVectorVariable> {
477     
478         private final CGNode node;
479     
480         private final ExplodedControlFlowGraph cfg;
481     
482         private final OrdinalSetMapping<Statement> domain;
483

485

487

490         /**
491          * if (i,j) \in heapReturnCaller, then statement j is a HeapStatement.ReturnCaller for statement i, a NormalStatement
492          * representing an invoke
493          */
494         private final IBinaryNaturalRelation heapReturnCaller = new BasicNaturalRelation();
495     
496         public RD(CGNode node, ExplodedControlFlowGraph cfg, PointerAnalysis<InstanceKey> pa, OrdinalSetMapping<Statement> domain,
497             Map<Integer, NormalStatement> ssaInstructionIndex2Statement, HeapExclusions exclusions) {
498           this.node = node;
499           this.cfg = cfg;
500           this.domain = domain;
501           this.pa = pa;
502           this.ssaInstructionIndex2Statement = ssaInstructionIndex2Statement;
503           this.exclusions = exclusions;
504           initHeapReturnCaller();
505         }
506     
507         private void initHeapReturnCaller() {
508           for (Statement s : domain) {
509             if (s.getKind().equals(Kind.HEAP_RET_CALLER)) {
510               if (DEBUG) {
511                 System.err.println("initHeapReturnCaller " + s);
512               }
513               HeapStatement.HeapReturnCaller r = (HeapReturnCaller) s;
514               NormalStatement call = ssaInstructionIndex2Statement.get(r.getCallIndex());
515               int i = domain.getMappedIndex(call);
516               int j = domain.getMappedIndex(r);
517               heapReturnCaller.add(i, j);
518             }
519           }
520         }

522         @Override
523         public UnaryOperator<BitVectorVariable> getEdgeTransferFunction(IExplodedBasicBlock src, IExplodedBasicBlock dst) {
524           if (DEBUG) {
525             System.err.println("getEdgeXfer: " + src + " " + dst + " " + src.isEntryBlock());
526           }
527           if (src.isEntryBlock()) {
528             if (DEBUG) {
529               System.err.println("heapEntry " + heapEntryStatements());
530             }
531             return new BitVectorUnionVector(new BitVectorIntSet(heapEntryStatements()).getBitVector());
532           }
533           if (src.getInstruction() != null && !(src.getInstruction() instanceof SSAAbstractInvokeInstruction)
534               && !cfg.getNormalSuccessors(src).contains(dst)) {
535             // if the edge only happens due to exceptional control flow, then no
536             // heap locations
537             // are def'ed or used
538             if (DEBUG) {
539               System.err.println("Identity");
540             }
541             return BitVectorIdentity.instance();
542           } else {
543             BitVector kill = kill(src);
544             IntSet gen = gen(src);
545             if (DEBUG) {
546               System.err.println("gen: " + gen + " kill: " + kill);
547             }
548             if (kill == null) {
549               if (gen == null) {
550                 return BitVectorIdentity.instance();
551               } else {
552                 return new BitVectorUnionVector(new BitVectorIntSet(gen).getBitVector());
553               }
554             } else {
555               if (gen == null) {
556                 return new BitVectorMinusVector(kill);
557               } else {
558                 return new BitVectorKillGen(kill, new BitVectorIntSet(gen).getBitVector());
559               }
560             }
561           }
562         }

584         /**
585          * @return int set representing the heap def statements that are gen'ed by the basic block. null if none.
586          */
587         IntSet gen(IExplodedBasicBlock b) {
588     
589           SSAInstruction s = b.getInstruction();
590           if (DEBUG) {
591             System.err.println("gen " + b + " " + s);
592           }
593           if (s == null) {
594             return null;
595           } else {
596             if (s instanceof SSAAbstractInvokeInstruction) {
597               // it's a normal statement ... we better be able to find it in the
598               // domain.
599               Statement st = ssaInstructionIndex2Statement.get(b.getLastInstructionIndex());
600               if (st == null) {
601                 System.err.println(ssaInstructionIndex2Statement);
602                 Assertions.UNREACHABLE("bang " + b + " " + b.getLastInstructionIndex() + " " + s);
603               }
604               int domainIndex = domain.getMappedIndex(st);
605               assert (domainIndex != -1);
606               if (DEBUG) {
607                 System.err.println("GEN FOR " + s + " " + heapReturnCaller.getRelated(domainIndex));
608               }
609               return heapReturnCaller.getRelated(domainIndex);
610             } else {
611               Collection<PointerKey> gen = modRef.getMod(node, heapModel, pa, s, exclusions);
612               if (gen.isEmpty()) {
613                 return null;
614               } else {
615                 NormalStatement n = ssaInstructionIndex2Statement.get(b.getLastInstructionIndex());
616                 return SparseIntSet.singleton(domain.getMappedIndex(n));
617               }
618             }
619           }
620         }

622         /**
623          * @return int set representing all HEAP_PARAM_CALLEE statements in the domain.
624          */
625         private IntSet heapEntryStatements() {
626           BitVectorIntSet result = new BitVectorIntSet();
627           for (Statement s : domain) {
628             if (s.getKind().equals(Kind.HEAP_PARAM_CALLEE)) {
629               result.add(domain.getMappedIndex(s));
630             }
631           }
632           return result;
633         }

635         /**
636          * @return int set representing the heap def statements that are killed by the basic block. null if none.
637          */
638         BitVector kill(IExplodedBasicBlock b) {
639           SSAInstruction s = b.getInstruction();
640           if (s == null) {
641             return null;
642           } else {
643             Collection<PointerKey> mod = modRef.getMod(node, heapModel, pa, s, exclusions);
644             if (mod.isEmpty()) {
645               return null;
646             } else {
647               // only static fields are actually killed
648               Predicate staticFilter = new Predicate() {
649                 @Override public boolean test(Object o) {
650                   return o instanceof StaticFieldKey;
651                 }
652               };
653               final Collection<PointerKey> kill = Iterator2Collection
654                   .toSet(new FilterIterator<PointerKey>(mod.iterator(), staticFilter));
655               if (kill.isEmpty()) {
656                 return null;
657               } else {
658                 Predicate f = new Predicate() {
659                   // accept any statement which writes a killed location.
660                   @Override public boolean test(Object o) {
661                     Statement s = (Statement) o;
662                     Collection m = getMod(s, node, heapModel, pa, exclusions);
663                     for (PointerKey k : kill) {
664                       if (m.contains(k)) {
665                         return true;
666                       }
667                     }
668                     return false;
669                   }
670                 };
671                 BitVector result = new BitVector();
672                 for (Statement k : Iterator2Iterable.make(new FilterIterator<Statement>(domain.iterator(), f))) {
673                   result.set(domain.getMappedIndex(k));
674                 }
675                 return result;
676               }
677             }
678           }
679         }
