11     package com.ibm.wala.ipa.cha;
12     
13     import com.ibm.wala.classLoader.IClass;
14     import com.ibm.wala.classLoader.IMethod;
15     import com.ibm.wala.types.MethodReference;
16     
17     /**
18      * Utilities for querying a class hierarchy
19      */
20     public class ClassHierarchyUtil {
21     
22       /**
23        * find the root of the inheritance tree for method m.
24        */
25       public static IMethod getRootOfInheritanceTree(IMethod m) {
26         IClass c = m.getDeclaringClass();
27         IClass parent = c.getSuperclass();
28         if (parent == null) {
29           return m;
30         } else {
31           MethodReference ref = MethodReference.findOrCreate(parent.getReference(), m.getSelector());
32           IMethod m2 = m.getClassHierarchy().resolveMethod(ref);
33           if (m2 != null && !m2.equals(m)) {
34             return getRootOfInheritanceTree(m2);
35           }
36           return m;
37         }
38       }
39     
40       /**
41        * Return the method that m overrides, or null if none
42        */
43       public static IMethod getOverriden(IMethod m) {
44         IClass c = m.getDeclaringClass();
45         IClass parent = c.getSuperclass();
46         if (parent == null) {
47           return null;
48         } else {
49           MethodReference ref = MethodReference.findOrCreate(parent.getReference(), m.getSelector());
50           IMethod m2 = m.getClassHierarchy().resolveMethod(ref);
51           if (m2 != null && !m2.equals(m)) {
52             return m2;
53           }
54           return null;
55         }
56       }
57     
58     }
