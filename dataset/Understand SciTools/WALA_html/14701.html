85        */
86       public static <T> void dotify(Graph<T> g, NodeDecorator<T> labels, String title, String dotFile, String outputFile, String dotExe)
87           throws WalaException {
88         if (g == null) {
89           throw new IllegalArgumentException("g is null");
90         }
91         File f = DotUtil.writeDotFile(g, labels, title, dotFile);
92         if (dotExe != null) {
93           spawnDot(dotExe, outputFile, f);
94         }
95       }

97       public static void spawnDot(String dotExe, String outputFile, File dotFile) throws WalaException {
98         if (dotFile == null) {
99           throw new IllegalArgumentException("dotFile is null");
100         }
101         String[] cmdarray = { dotExe, outputTypeCmdLineParam(), "-o", outputFile, "-v", dotFile.getAbsolutePath() };
102         System.out.println("spawning process " + Arrays.toString(cmdarray));
103         BufferedInputStream output = null;
104         BufferedInputStream error = null;
105         try {
106           Process p = Runtime.getRuntime().exec(cmdarray);
107           output = new BufferedInputStream(p.getInputStream());
108           error = new BufferedInputStream(p.getErrorStream());
109           boolean repeat = true;
110           while (repeat) {
111             try {
112               Thread.sleep(500);
113             } catch (InterruptedException e1) {
114               e1.printStackTrace();
115               // just ignore and continue
116             }
117             if (output.available() > 0) {
118               byte[] data = new byte[output.available()];
119               int nRead = output.read(data);
120               System.err.println("read " + nRead + " bytes from output stream");
121             }
122             if (error.available() > 0) {
123               byte[] data = new byte[error.available()];
124               int nRead = error.read(data);
125               System.err.println("read " + nRead + " bytes from error stream");
126             }
127             try {
128               p.exitValue();
129               // if we get here, the process has terminated
130               repeat = false;
131               System.out.println("process terminated with exit code " + p.exitValue());
132             } catch (IllegalThreadStateException e) {
133               // this means the process has not yet terminated.
134               repeat = true;
135             }
136           }
137         } catch (IOException e) {
138           e.printStackTrace();
139           throw new WalaException("IOException in " + DotUtil.class);
140         } finally {
141           if (output != null) {
142             try {
143               output.close();
144             } catch (IOException e) {
145               e.printStackTrace();
146             }
147           }
148           if (error != null) {
149             try {
150               error.close();
151             } catch (IOException e) {
152               e.printStackTrace();
153             }
154           }
155         }
156       }

158       public static <T> File writeDotFile(Graph<T> g, NodeDecorator<T> labels, String title, String dotfile) throws WalaException {
159     
160         if (g == null) {
161           throw new IllegalArgumentException("g is null");
162         }
163         StringBuffer dotStringBuffer = dotOutput(g, labels, title);
164     
165         // retrieve the filename parameter to this component, a String
166         if (dotfile == null) {
167           throw new WalaException("internal error: null filename parameter");
168         }
169         try {
170           File f = new File(dotfile);
171           FileWriter fw = new FileWriter(f);
172           fw.write(dotStringBuffer.toString());
173           fw.close();
174           return f;
175     
176         } catch (Exception e) {
177           throw new WalaException("Error writing dot file " + dotfile);
178         }
179       }

181       /**
182        * @return StringBuffer holding dot output representing G
183        * @throws WalaException
184        */
185       private static <T> StringBuffer dotOutput(Graph<T> g, NodeDecorator<T> labels, String title) throws WalaException {
186         StringBuffer result = new StringBuffer("digraph \"DirectedGraph\" {\n");
187     
188         if (title != null) {
189           result.append("graph [label = \""+title+"\", labelloc=t, concentrate = true];");
190         } else {
191           result.append("graph [concentrate = true];");
192         }
193         
194         String rankdir = getRankDir();
195         if (rankdir != null) {
196           result.append("rankdir=" + rankdir + ";");
197         }
198         String fontsizeStr = "fontsize=" + fontSize;
199         String fontcolorStr = (fontColor != null) ? ",fontcolor="+fontColor : "";
200         String fontnameStr = (fontName != null) ? ",fontname="+fontName : "";
201              
202         result.append("center=true;");
203         result.append(fontsizeStr);
204         result.append(";node [ color=blue,shape=\"box\"");
205         result.append(fontsizeStr);
206         result.append(fontcolorStr);
207         result.append(fontnameStr);
208         result.append("];edge [ color=black,");
209         result.append(fontsizeStr);
210         result.append(fontcolorStr);
211         result.append(fontnameStr);
212         result.append("]; \n");
213     
214         Collection<T> dotNodes = computeDotNodes(g);
215     
216         outputNodes(labels, result, dotNodes);
217     
218         for (Iterator<? extends T> it = g.iterator(); it.hasNext();) {
219           T n = it.next();
220           for (Iterator<? extends T> it2 = g.getSuccNodes(n); it2.hasNext();) {
221             T s = it2.next();
222             result.append(" ");
223             result.append(getPort(n, labels));
224             result.append(" -> ");
225             result.append(getPort(s, labels));
226             result.append(" \n");
227           }
228         }
229     
230         result.append("\n}");
231         return result;
232       }

234       private static <T> void outputNodes(NodeDecorator<T> labels, StringBuffer result, Collection<T> dotNodes) throws WalaException {
235         for (Iterator<T> it = dotNodes.iterator(); it.hasNext();) {
236           outputNode(labels, result, it.next());
237         }
238       }
239     
240       private static <T> void outputNode(NodeDecorator<T> labels, StringBuffer result, T n) throws WalaException {
241         result.append("   ");
242         result.append("\"");
243         result.append(getLabel(n, labels));
244         result.append("\"");
245         result.append(decorateNode(n, labels));
246       }

260        * @param n node to decorate
261        * @param d decorating master
262        */
263       private static <T> String decorateNode(T n, NodeDecorator<T> d) throws WalaException {
264         StringBuffer result = new StringBuffer();
265         result.append(" [ ]\n");
266         return result.toString();
267       }

269       private static <T> String getLabel(T n, NodeDecorator<T> d) throws WalaException {
270         String result = null;
271         if (d == null) {
272           result = n.toString();
273         } else {
274           result = d.getLabel(n);
275           result = result == null ? n.toString() : result;
276         }
277         if (result.length() >= MAX_LABEL_LENGTH) {
278           result = result.substring(0, MAX_LABEL_LENGTH - 3) + "...";
279         }
280         return result;
281       }
282     
283       private static <T> String getPort(T n, NodeDecorator<T> d) throws WalaException {
284         return "\"" + getLabel(n, d) + "\"";
