11     package com.ibm.wala.ssa;
12     
13     import java.util.Collection;
14     
15     /**
16      * A mapping that tells, for a given instruction s, what "names" does s def and use <em>indirectly</em>.
17      * 
18      * For example, an {@link SSALoadIndirectInstruction} takes as an argument a pointer value. This map tells which locals that pointer
19      * may alias.
20      * 
21      * So if we have
22      * 
23      * Example A:
24      * <tt>
25      * v2 = SSAAddressOf v1;
26      * v7 = #1;
27      * v3 = SSALoadIndirect v2; (1)
28      * <\tt>
29      * 
30      * Then this map will tell us that instruction (1) indirectly uses whichever source-level entity (in the source or bytecode) v1 represents.
31      * 
32      * (Don't be confused by AddressOf v1 .. we're not actually taking the address of v1 ... we're taking the address of some source level entity (like a local variable in source code or bytecode) for which v1 is just an SSA name)
33      * 
34      * As a more complex example, when we have lexical scoping, we can have the following IR generated, which passes a local by reference:
35      * 
36      * Example B:
37      * <tt>
38      * foo:
39      *   v3 = AddressOf v2;
40      *   bar(v3)  (1)
41      * bar(v1):
42      *   StoreIndirect v1, #7.
43      * </tt>
44      * 
45      * In this case, the instruction (1) potentially defs the locals aliased by v2. The lexical scoping support could/should use this
46      * information to rebuild SSA accounting for the fact that (1) defs v2.
47      * 
48      */
49     public interface SSAIndirectionData<T extends SSAIndirectionData.Name> {
50     
51       /**
52        * A Name is a mock interface introduced just for strong typing. A Name represents some semantic entity in the program
53        * representation (e.g. a local variable in the source code or a local number in the bytecode)
54        */
55       public interface Name {
56       }
57     
58       /**
59        * Returns the set of "source" level names (e.g. local variables in bytecode or source code) for which this map holds information.
60        */
61       Collection<T> getNames();
62     
63       /**
64        * For the instruction at the given index, and a source-level name, return the SSA value number which represents this
65        * instruction's def of that name.
66        * 
67        * For example, in Example B in header comment above, suppose v2 referred to a "source"-entity called "Local1".
68        * Since instruction (1) (call to bar) defs "Local1", we introduce in this table a new SSA value number, say v7, which represents
69        * the value of "Local1" immediately after this instruction.   
70        */
71       int getDef(int instructionIndex, T name);
72     
73       /**
74        * Record the fact that a particular instruction defs a particular SSA value number (newDef), representing the value of a "source" entity
75        * "name".
76        * 
77        * @see #getDef
78        */
79       void setDef(int instructionIndex, T name, int newDef);
80     
81       /**
82        * For the instruction at the given index, and a source-level name, return the SSA value number which represents this
83        * instruction's use of that name.
84        * 
85        * For example, in Example A in header comment above, suppose v1 referred to a "source"-entity called "Local1".
86        * Since instruction (1) (LoadIndirect) uses "Local1", we record in this table the SSA value number that represents "Local1"
87        * immediately before instruction (1).   So if v1 and v7 both refer to "Local1", then (1) uses v7.  If v7 does NOT refer to "Local1",
88        * then (1) uses v1.
89        */
90       int getUse(int instructionIndex, T name);
91     
92       /**
93        *  @see #getUse
94        */
95       void setUse(int instructionIndex, T name, int newUse);
96     
97     }
