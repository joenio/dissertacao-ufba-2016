11     package com.ibm.wala.ssa;
12     
13     import com.ibm.wala.classLoader.IMethod;
14     import com.ibm.wala.ipa.callgraph.Context;
15     import com.ibm.wala.ipa.callgraph.impl.Everywhere;
16     
17     /**
18      * A mapping from IMethod -> SSAOptions -> SoftReference -> Something
19      * 
20      * This doesn't work very well ... GCs don't do such a great job with SoftReferences ... revamp it.
21      */
22     public class SSACache {
23     
24       /**
25        * used for debugging
26        */
27       private static final boolean DISABLE = false;
28     
29       /**
30        * The factory that actually creates new IR objects
31        */
32       private final IRFactory<IMethod> factory;
33     
34       /**
35        * A cache of SSA IRs
36        */
37       final private AuxiliaryCache irCache = new AuxiliaryCache();
38     
39       /**
40        * A cache of DefUse information
41        */
42       final private AuxiliaryCache duCache = new AuxiliaryCache();
43     
44       /**
45        * @param factory a factory for creating IRs
46        */
47       public SSACache(IRFactory<IMethod> factory) {
48         this.factory = factory;
49       }
50     
51       /**
52        * @param m a "normal" (bytecode-based) method
53        * @param options options governing ssa construction
54        * @return an IR for m, built according to the specified options. null if m is abstract or native.
55        * @throws IllegalArgumentException if m is null
56        */
57       public synchronized IR findOrCreateIR(final IMethod m, Context c, final SSAOptions options) {
58     
59         if (m == null) {
60           throw new IllegalArgumentException("m is null");
61         }
62         if (m.isAbstract() || m.isNative()) {
63           return null;
64         }
65     
66         if (factory.contextIsIrrelevant(m)) {
67           c = Everywhere.EVERYWHERE;
68         }
69     
70         if (DISABLE) {
71           return factory.makeIR(m, c, options);
72         }
73     
74         IR ir = (IR) irCache.find(m, c, options);
75         if (ir == null) {
76           ir = factory.makeIR(m, c, options);
77           irCache.cache(m, c, options, ir);
78         }
79         return ir;
80       }
81     
82       /**
83        * @param m a method
84        * @param options options governing ssa construction
85        * @return DefUse information for m, built according to the specified options. null if unavailable
86        * @throws IllegalArgumentException if m is null
87        */
88       public synchronized DefUse findOrCreateDU(IMethod m, Context c, SSAOptions options) {
89         if (m == null) {
90           throw new IllegalArgumentException("m is null");
91         }
92         if (m.isAbstract() || m.isNative()) {
93           return null;
94         }
95         if (factory.contextIsIrrelevant(m)) {
96           c = Everywhere.EVERYWHERE;
97         }
98     
99         DefUse du = (DefUse) duCache.find(m, c, options);
100         if (du == null) {
101           IR ir = findOrCreateIR(m, c, options);
102           du = new DefUse(ir);
103           duCache.cache(m, c, options, du);
104         }
105         return du;
106       }
107     
108       /**
109        * @return {@link DefUse} information for m, built according to the specified options. null if unavailable
110        * @throws IllegalArgumentException if ir is null
111        */
112       public synchronized DefUse findOrCreateDU(IR ir, Context C) {
113         if (ir == null) {
114           throw new IllegalArgumentException("ir is null");
115         }
116         DefUse du = (DefUse) duCache.find(ir.getMethod(), C, ir.getOptions());
117         if (du == null) {
118           du = new DefUse(ir);
119           duCache.cache(ir.getMethod(), C, ir.getOptions(), du);
120         }
121         return du;
122       }
123     
124       /**
125        * The existence of this is unfortunate.
126        */
127       public void wipe() {
128         irCache.wipe();
129         duCache.wipe();
130       }
131     
132       /**
133        * Invalidate the cached IR for a <method,context> pair
134        */
135       public void invalidateIR(IMethod method, Context c) {
136         irCache.invalidate(method, c);
137       }
138     
139       /**
140        * Invalidate the cached {@link DefUse} for a <method,context> pair
141        */
142       public void invalidateDU(IMethod method, Context c) {
143         duCache.invalidate(method, c);
144       }
145     
146       /**
147        * Invalidate all cached information for a <method,context> pair
148        */
149       public void invalidate(IMethod method, Context c) {
150         invalidateIR(method, c);
151         invalidateDU(method, c);
152       }
153     }
