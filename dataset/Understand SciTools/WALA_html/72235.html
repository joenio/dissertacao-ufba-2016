14     import java.util.Set;
15     
16     import com.ibm.wala.classLoader.ClassLoaderFactory;

18     import com.ibm.wala.classLoader.IClassLoader;

20     import com.ibm.wala.classLoader.IMethod;
21     import com.ibm.wala.ipa.callgraph.AnalysisScope;
22     import com.ibm.wala.types.ClassLoaderReference;
23     import com.ibm.wala.types.FieldReference;
24     import com.ibm.wala.types.MethodReference;
25     import com.ibm.wala.types.Selector;
26     import com.ibm.wala.types.TypeReference;

28     /**
29      * General interface for a type hierarchy
30      */
31     public interface IClassHierarchy extends Iterable<IClass> {
32     
33       public ClassLoaderFactory getFactory();
34     
35       public AnalysisScope getScope();
36     
37       public IClassLoader[] getLoaders();
38     
39       public IClassLoader getLoader(ClassLoaderReference loaderRef);
40     
41       /**
42        * @return true if the add succeeded; false if it failed for some reason
43        * @throws IllegalArgumentException if klass is null
44        */
45       public boolean addClass(IClass klass);
46     
47       /**
48        * @return The number of classes present in the class hierarchy.
49        */
50       public int getNumberOfClasses();
51     
52       public boolean isRootClass(IClass c);
53     
54       public IClass getRootClass();
55     
56       public int getNumber(IClass c);
57     
58     /** BEGIN Custom change: remember unresolved classes */
59       public Set<TypeReference> getUnresolvedClasses();
60       
61     /** END Custom change: remember unresolved classes */
62       /**
63        * Find the possible targets of a call to a method reference
64        * 
65        * @param ref method reference
66        * @return the set of IMethods that this call can resolve to.
67        * @throws IllegalArgumentException if ref is null
68        */
69       public Set<IMethod> getPossibleTargets(MethodReference ref);
70     
71       /**
72        * Find the possible targets of a call to a method reference where the receiver is of a certain type
73        * 
74        * @param receiverClass the class of the receiver
75        * @param ref method reference
76        * @return the set of IMethods that this call can resolve to.
77        */
78       public Set<IMethod> getPossibleTargets(IClass receiverClass, MethodReference ref);
79     
80       /**
81        * Return the unique receiver of an invocation of method on an object of type m.getDeclaredClass
82        * 
83        * @return IMethod, or null if no appropriate receiver is found.
84        * @throws IllegalArgumentException if m is null
85        */
86       public IMethod resolveMethod(MethodReference m);
87     
88       /**
89        * @return the canonical IField that represents a given field , or null if none found
90        * @throws IllegalArgumentException if f is null
91        */
92       public IField resolveField(FieldReference f);
93     
94       /**
95        * @return the canonical IField that represents a given field , or null if none found
96        * @throws IllegalArgumentException if f is null
97        * @throws IllegalArgumentException if klass is null
98        */
99       public IField resolveField(IClass klass, FieldReference f);
100     
101       /**
102        * Return the unique receiver of an invocation of method on an object of type declaringClass
103        * 
104        * @param receiverClass type of receiver
105        * @param selector method signature
106        * @return Method resolved method abstraction
107        * @throws IllegalArgumentException if receiverClass is null
108        */
109       public IMethod resolveMethod(IClass receiverClass, Selector selector);
110     
111       /**
112        * Load a class using one of the loaders specified for this class hierarchy
113        * 
114        * @return null if can't find the class.
115        * @throws IllegalArgumentException if A is null
116        */
117       public IClass lookupClass(TypeReference A);
118     
119       // public boolean isSyntheticClass(IClass c);
120     
121       public boolean isInterface(TypeReference type);
122     
123       public IClass getLeastCommonSuperclass(IClass A, IClass B);
124     
125       public TypeReference getLeastCommonSuperclass(TypeReference A, TypeReference B);
126     
127       /**
128        * Is c a subclass of T?
129        * 
130        * @throws IllegalArgumentException if c is null
131        */
132       public boolean isSubclassOf(IClass c, IClass T);
133     
134       /**
135        * Does c implement i?
136        * 
137        * @return true iff i is an interface and c is a class that implements i, or c is an interface that extends i.
138        */
139       public boolean implementsInterface(IClass c, IClass i);
140     
141       /**
142        * Return set of all subclasses of type in the Class Hierarchy
143        */
144       public Collection<IClass> computeSubClasses(TypeReference type);
145     
146       /**
147        * Solely for optimization; return a Collection<TypeReference> representing the subclasses of Error
148        * 
149        * kind of ugly. a better scheme?
150        */
151       public Collection<TypeReference> getJavaLangErrorTypes();
152     
153       /**
154        * Solely for optimization; return a Collection<TypeReference> representing the subclasses of {@link RuntimeException}
155        * 
156        * kind of ugly. a better scheme?
157        */
158       public Collection<TypeReference> getJavaLangRuntimeExceptionTypes();
159     
160       /**
161        * @param type an interface
162        * @return Set of IClass that represent implementors of the interface
163        */
164       public Set<IClass> getImplementors(TypeReference type);
165     
166       /**
167        * @return the number of classes that immediately extend klass.
168        */
169       public int getNumberOfImmediateSubclasses(IClass klass);
170     
171       /**
172        * @return the classes that immediately extend klass.
173        */
174       public Collection<IClass> getImmediateSubclasses(IClass klass);
175     
176       /**
177        * Does an expression c1 x := c2 y typecheck?
178        * 
179        * i.e. is c2 a subtype of c1?
180        * 
181        * @throws IllegalArgumentException if c1 is null
182        * @throws IllegalArgumentException if c2 is null
183        */
184       public boolean isAssignableFrom(IClass c1, IClass c2);
185     
186     }
