371         @Override
372         public void visitAstLexicalRead(AstLexicalRead instruction) {
373           visitLexical(instruction, new LexicalOperator((AstCGNode) node, instruction.getAccesses(), true) {
374             @Override
375             protected void action(PointerKey lexicalKey, int vn) {
376               PointerKey lval = getPointerKeyForLocal(vn);
377               if (lexicalKey instanceof LocalPointerKey) {
378                 CGNode lnode = ((LocalPointerKey) lexicalKey).getNode();
379                 int lvn = ((LocalPointerKey) lexicalKey).getValueNumber();
380                 IR lir = getBuilder().getCFAContextInterpreter().getIR(lnode);
381                 SymbolTable lsymtab = lir.getSymbolTable();
382                 DefUse ldu = getAnalysisCache().getSSACache().findOrCreateDU(lir, lnode.getContext());
383                 if (contentsAreInvariant(lsymtab, ldu, lvn)) {
384                   InstanceKey[] ik = getInvariantContents(lsymtab, ldu, lnode, lvn);
385                   system.recordImplicitPointsToSet(lexicalKey);
386                   for (int i = 0; i < ik.length; i++) {
387                     system.findOrCreateIndexForInstanceKey(ik[i]);
388                     system.newConstraint(lval, ik[i]);
389                   }
390     
391                   return;
392                 }
393               }
394     
395               system.newConstraint(lval, assignOperator, lexicalKey);
396             }
397           });
398         }

400         @Override
401         public void visitAstLexicalWrite(AstLexicalWrite instruction) {
402           visitLexical(instruction, new LexicalOperator((AstCGNode) node, instruction.getAccesses(), false) {
403             @Override
404             protected void action(PointerKey lexicalKey, int vn) {
405               PointerKey rval = getPointerKeyForLocal(vn);
406               if (contentsAreInvariant(symbolTable, du, vn)) {
407                 InstanceKey[] ik = getInvariantContents(vn);

413               } else {
414                 system.newConstraint(lexicalKey, assignOperator, rval);

416             }

430         @Override
431         public void visitPut(SSAPutInstruction inst) {
432           super.visitPut(inst);
433     
434           if (inst.isStatic() || !getBuilder().useObjectCatalog())
435             return;
436     
437           // update the object catalog corresponding to the base pointer, adding the
438           // name of the field as a property
439     
440           SymbolTable symtab = ir.getSymbolTable();
441     
442           int objVn = inst.getRef();
443           String fieldName = null;
444           try {
445             fieldName = inst.getDeclaredField().getName().toUnicodeString();
446           } catch (UTFDataFormatException e) {
447             Assertions.UNREACHABLE();
448           }
449     
450           final PointerKey objKey = getPointerKeyForLocal(objVn);
451     
452           final InstanceKey[] fieldNameKeys = new InstanceKey[] { getInstanceKeyForConstant(fieldName) };
453           assert fieldNameKeys.length == 1;
454     
455           if (contentsAreInvariant(symtab, du, objVn)) {
456             system.recordImplicitPointsToSet(objKey);
457             final InstanceKey[] objKeys = getInvariantContents(objVn);
458     
459             for (int i = 0; i < objKeys.length; i++) {
460               if (!getBuilder().isUncataloguedField(objKeys[i].getConcreteType(), fieldName)) {
461                 PointerKey objCatalog = getPointerKeyForObjectCatalog(objKeys[i]);
462                 if (objCatalog != null) {
463                   system.newConstraint(objCatalog, fieldNameKeys[0]);
464                 }
465               }
466             }
467     
468           } else {
469             final String hack = fieldName;
470             system.newSideEffect(new UnaryOperator<PointsToSetVariable>() {
471               @Override
472               public byte evaluate(PointsToSetVariable lhs, PointsToSetVariable rhs) {
473                 final IntSetVariable objects = rhs;
474                 if (objects.getValue() != null) {
475                   objects.getValue().foreach(new IntSetAction() {
476                     @Override
477                     public void act(int optr) {
478                       InstanceKey object = system.getInstanceKey(optr);
479                       if (!getBuilder().isUncataloguedField(object.getConcreteType(), hack)) {
480                         PointerKey cat = getPointerKeyForObjectCatalog(object);
481                         if (cat != null) {
482                           system.newConstraint(cat, fieldNameKeys[0]);
483                         }
484                       }
485                     }
486                   });
487                 }
488                 return NOT_CHANGED;
489               }
490     
491               @Override
492               public int hashCode() {
493                 return System.identityHashCode(this);
494               }
495     
496               @Override
497               public boolean equals(Object o) {
498                 return o == this;
499               }
500     
501               @Override
502               public String toString() {
503                 return "field name record: " + objKey;
504               }
505             }, objKey);
506           }
507         }

589         private abstract class LexicalOperator extends UnaryOperator<PointsToSetVariable> {
590           /**
591            * node in which lexical accesses are performed
592            */
593           private final AstCGNode node;
594     
595           /**
596            * the lexical accesses to be handled
597            */
598           private final Access[] accesses;
599     
600           /**
601            * are all the lexical accesses loads? if false, they are all stores
602            */
603           private final boolean isLoad;
604     
605           private LexicalOperator(AstCGNode node, Access[] accesses, boolean isLoad) {
606             this.node = node;
607             this.isLoad = isLoad;
608             this.accesses = accesses;
609           }
610     
611           /**
612            * perform the necessary {@link #action(PointerKey, int)}s for the
613            * accesses. For each access, we determine the possible {@link CGNode}s
614            * corresponding to its definer (see
615            * {@link AstConstraintVisitor#getLexicalDefiners(CGNode, String)). Handle
616            * using
617            * {@link AstConstraintVisitor#handleRootLexicalReference(String, String, CGNode)}
618            * .
619            */
620           private void doLexicalPointerKeys(boolean funargsOnly) {
621             for (int i = 0; i < accesses.length; i++) {
622               final String name = accesses[i].variableName;
623               final String definer = accesses[i].variableDefiner;
624               final int vn = accesses[i].valueNumber;
625     
626               if (AstTranslator.DEBUG_LEXICAL)
627                 System.err.println(("looking up lexical parent " + definer));
628     
629               Set<CGNode> creators = getLexicalDefiners(node, Pair.make(name, definer));
630               
631               System.err.println("definers " + creators.size());
632               
633               for (CGNode n : creators) {
634                 PointerKey funargKey = handleRootLexicalReference(name, definer, n);
635                 action(funargKey, vn);
636               }
637             }
638           }
639     
640           @Override
641           public byte evaluate(PointsToSetVariable lhs, PointsToSetVariable rhs) {
642             doLexicalPointerKeys(true);
643             return NOT_CHANGED;
644           }
645     
646           abstract protected void action(PointerKey lexicalKey, int vn);
647     
648           @Override
649           public String toString() {
650             return "lexical op";
651           }
652     
653           @Override
654           public boolean equals(Object o) {
655             if (!(o instanceof LexicalOperator)) {
656               return false;
657             } else {
658               LexicalOperator other = (LexicalOperator) o;
659     
660               if (isLoad != other.isLoad) {
661                 return false;
662     
663               } else if (!node.equals(other.node)) {
664                 return false;
665     
666               } else if (accesses.length != other.accesses.length) {
667                 return false;
668     
669               } else {
670                 for (int i = 0; i < accesses.length; i++) {
671                   if (!accesses[i].equals(other.accesses[i])) {
672                     return false;
673                   }
674                 }
675     
676                 for (int i = 0; i < accesses.length; i++) {
677                   if (!accesses[i].equals(other.accesses[i])) {
678                     return false;
679                   }
680                 }
681     
682                 return true;
683               }
684             }
685           }

687           @Override
688           public int hashCode() {
689             return node.hashCode() * accesses[0].hashCode() * accesses.length;
690           }

693         private Set<CGNode> getLexicalDefiners(final CGNode opNode, final Pair<String, String> definer) {
694           if (definer == null) {
695             return Collections.singleton(getBuilder().getCallGraph().getFakeRootNode());
696           } else if (getBuilder().sameMethod(opNode, definer.snd)) {
697             // lexical access to a variable declared in opNode itself
698             return Collections.singleton(opNode);
699           } else {
700             final Set<CGNode> result = HashSetFactory.make();
701             PointerKey F = getBuilder().getPointerKeyForLocal(opNode, 1);
702     
703             IR ir = getBuilder().getCFAContextInterpreter().getIR(opNode);
704             SymbolTable symtab = ir.getSymbolTable();
705             DefUse du = getBuilder().getCFAContextInterpreter().getDU(opNode);
706             if (contentsAreInvariant(symtab, du, 1)) {
707               system.recordImplicitPointsToSet(F);
708               final InstanceKey[] functionKeys = getInvariantContents(symtab, du, opNode, 1);
709               for (int f = 0; f < functionKeys.length; f++) {
710                 system.findOrCreateIndexForInstanceKey(functionKeys[f]);
711                 ScopeMappingInstanceKey K = (ScopeMappingInstanceKey) functionKeys[f];
712                 Iterator<CGNode> x = K.getFunargNodes(definer);
713                 while (x.hasNext()) {
714                   result.add(x.next());
715                 }
716               }
717             } else {
718               PointsToSetVariable FV = system.findOrCreatePointsToSet(F);
719               if (FV.getValue() != null) {
720                 FV.getValue().foreach(new IntSetAction() {
721                   @Override
722                   public void act(int ptr) {
723                     InstanceKey iKey = system.getInstanceKey(ptr);
724                     if (iKey instanceof ScopeMappingInstanceKey) {
725                       ScopeMappingInstanceKey K = (ScopeMappingInstanceKey) iKey;
726                       Iterator<CGNode> x = K.getFunargNodes(definer);
727                       while (x.hasNext()) {
728                         result.add(x.next());
729                       }
730                     } else {
731                       Assertions.UNREACHABLE("unexpected instance key " + iKey);
732                     }
733                   }
734                 });
735               }
736             }
737     
738             return result;
739           }
740         }

749         private Set<PointerKey> discoveredUpwardFunargs = HashSetFactory.make();

751         /**
752          * add constraints that assign the final value of name in definingNode to
753          * the upward funarg (lhs), modeling adding of the state to the closure
754          */
755         private void addUpwardFunargConstraints(PointerKey lhs, String name, String definer, CGNode definingNode) {
756           discoveredUpwardFunargs.add(lhs);
757     
758           LexicalInformation LI = ((AstIR) definingNode.getIR()).lexicalInfo();
759           Pair[] names = LI.getExposedNames();
760           for (int i = 0; i < names.length; i++) {
761             if (name.equals(names[i].fst) && definer.equals(names[i].snd)) {
762               int vn = LI.getExitExposedUses()[i];
763               if (vn > 0) {
764                 IR ir = getBuilder().getCFAContextInterpreter().getIR(definingNode);
765                 DefUse du = getBuilder().getCFAContextInterpreter().getDU(definingNode);
766                 SymbolTable st = ir.getSymbolTable();
767     
768                 PointerKey rhs = getBuilder().getPointerKeyForLocal(definingNode, vn);
769     
770                 if (contentsAreInvariant(st, du, vn)) {
771                   system.recordImplicitPointsToSet(rhs);
772                   final InstanceKey[] objs = getInvariantContents(st, du, definingNode, vn);
773                   for (int f = 0; f < objs.length; f++) {
774                     system.findOrCreateIndexForInstanceKey(objs[f]);
775                     system.newConstraint(lhs, objs[f]);
776                   }
777                 } else {
778                   system.newConstraint(lhs, assignOperator, rhs);
779                 }
780               }
781     
782               return;
783             }
784           }
785     
786           Assertions.UNREACHABLE();
787         }

789         /**
790          * handle a lexical reference where we found no parent call graph node
791          * defining the name; it's either a global or an upward funarg
792          */
793         private PointerKey handleRootLexicalReference(String name, String definer, final CGNode definingNode) {
794           // global variable
795           if (definer == null) {
796             return new AstGlobalPointerKey(name);
797     
798             // upward funarg
799           } else {
800             class UpwardFunargPointerKey extends AstGlobalPointerKey {
801               UpwardFunargPointerKey(String name) {
802                 super(name);
803               }
804     
805               public CGNode getDefiningNode() {
806                 return definingNode;
807               }
808     
809               @Override
810               public boolean equals(Object x) {
811                 return (x instanceof UpwardFunargPointerKey)
812                     && super.equals(x)
813                     && (definingNode == null ? definingNode == ((UpwardFunargPointerKey) x).getDefiningNode() : definingNode
814                         .equals(((UpwardFunargPointerKey) x).getDefiningNode()));
815               }
816     
817               @Override
818               public int hashCode() {
819                 return super.hashCode() * ((definingNode == null) ? 17 : definingNode.hashCode());
820               }
821     
822               @Override
823               public String toString() {
824                 return "[upward:" + getName() + ":" + definingNode + "]";
825               }
826             }
827     
828             PointerKey result = new UpwardFunargPointerKey(name);
829     
830             if (!discoveredUpwardFunargs.contains(result) && definingNode != null) {
831               addUpwardFunargConstraints(result, name, definer, definingNode);
832             }
833     
834             return result;
835           }
836         }

852         private void newFieldOperation(CGNode opNode, final int objVn, final int fieldsVn, final boolean isLoadOperation,
853             final ReflectedFieldAction action) {
854           IR ir = getBuilder().getCFAContextInterpreter().getIR(opNode);
855           SymbolTable symtab = ir.getSymbolTable();
856           DefUse du = getBuilder().getCFAContextInterpreter().getDU(opNode);
857           PointerKey objKey = getBuilder().getPointerKeyForLocal(opNode, objVn);
858           final PointerKey fieldKey = getBuilder().getPointerKeyForLocal(opNode, fieldsVn);
859     
860           // log field access
861           if (DEBUG_PROPERTIES) {
862             if (isLoadOperation)
863               System.err.print(("adding read of " + objKey + "." + fieldKey + ":"));
864             else
865               System.err.print(("adding write of " + objKey + "." + fieldKey + ":"));
866     
867             if (contentsAreInvariant(symtab, du, objVn)) {
868               System.err.print(" constant obj:");
869               InstanceKey[] x = getInvariantContents(symtab, du, opNode, objVn);
870               for (int i = 0; i < x.length; i++) {
871                 System.err.print((x[i].toString() + " "));
872               }
873             } else {
874               System.err.print((" obj:" + system.findOrCreatePointsToSet(objKey)));
875             }
876     
877             if (contentsAreInvariant(symtab, du, fieldsVn)) {
878               System.err.print(" constant prop:");
879               InstanceKey[] x = getInvariantContents(symtab, du, opNode, fieldsVn);
880               for (int i = 0; i < x.length; i++) {
881                 System.err.print((x[i].toString() + " "));
882               }
883             } else {
884               System.err.print((" props:" + system.findOrCreatePointsToSet(fieldKey)));
885             }
886     
887             System.err.print("\n");
888           }
889     
890           // make sure instance keys get mapped for PointerAnalysisImpl
891           if (contentsAreInvariant(symtab, du, objVn)) {
892             InstanceKey[] x = getInvariantContents(symtab, du, opNode, objVn);
893             for (int i = 0; i < x.length; i++) {
894               system.findOrCreateIndexForInstanceKey(x[i]);
895             }
896           }
897           if (contentsAreInvariant(symtab, du, fieldsVn)) {
898             InstanceKey[] x = getInvariantContents(symtab, du, opNode, fieldsVn);
899             for (int i = 0; i < x.length; i++) {
900               system.findOrCreateIndexForInstanceKey(x[i]);
901             }
902           }
903     
904           // process field access
905           if (contentsAreInvariant(symtab, du, objVn)) {
906             system.recordImplicitPointsToSet(objKey);
907             final InstanceKey[] objKeys = getInvariantContents(symtab, du, opNode, objVn);
908     
909             if (contentsAreInvariant(symtab, du, fieldsVn)) {
910               system.recordImplicitPointsToSet(fieldKey);
911               InstanceKey[] fieldsKeys = getInvariantContents(symtab, du, opNode, fieldsVn);
912     
913               newFieldOperationObjectAndFieldConstant(isLoadOperation, action, objKeys, fieldsKeys);
914     
915             } else {
916               newFieldOperationOnlyObjectConstant(isLoadOperation, action, fieldKey, objKeys);
917             }
918     
919           } else {
920             if (contentsAreInvariant(symtab, du, fieldsVn)) {
921               system.recordImplicitPointsToSet(fieldKey);
922               final InstanceKey[] fieldsKeys = getInvariantContents(symtab, du, opNode, fieldsVn);
923     
924               newFieldOperationOnlyFieldConstant(isLoadOperation, action, objKey, fieldsKeys);
925     
926             } else {
927               newFieldFullOperation(isLoadOperation, action, objKey, fieldKey);
928             }
929           }
930     
931           if (DEBUG_PROPERTIES) {
932             System.err.println("finished\n");
933           }
934         }

955         protected void newFieldFullOperation(final boolean isLoadOperation, final ReflectedFieldAction action, PointerKey objKey,
956             final PointerKey fieldKey) {
957           system.newSideEffect(new AbstractOperator<PointsToSetVariable>() {
958             private final MutableIntSet doneReceiver = IntSetUtil.make();
959             private final MutableIntSet doneField = IntSetUtil.make();
960     
961             @Override
962             public byte evaluate(PointsToSetVariable lhs, final PointsToSetVariable[] rhs) {
963               final IntSetVariable receivers = rhs[0];
964               final IntSetVariable fields = rhs[1];
965               if (receivers.getValue() != null && fields.getValue() != null) {
966                 receivers.getValue().foreach(new IntSetAction() {
967                   @Override
968                   public void act(final int rptr) {
969                     final InstanceKey receiver = system.getInstanceKey(rptr);
970     
971                     if (!isLoadOperation) {
972                       PointerKey cat = getPointerKeyForObjectCatalog(receiver);
973                       if (cat != null) {
974                         system.newConstraint(cat, assignOperator, fieldKey);
975                       }
976                     }
977     
978                     fields.getValue().foreach(new IntSetAction() {
979                       @Override
980                       public void act(int fptr) {
981                         if (!doneField.contains(fptr) || !doneReceiver.contains(rptr)) {
982                           InstanceKey field = system.getInstanceKey(fptr);
983                           for (Iterator keys = isLoadOperation ? getPointerKeysForReflectedFieldRead(receiver, field)
984                               : getPointerKeysForReflectedFieldWrite(receiver, field); keys.hasNext();) {
985                             AbstractFieldPointerKey key = (AbstractFieldPointerKey) keys.next();
986                             if (DEBUG_PROPERTIES)
987                               action.dump(key, false, false);
988                             action.action(key);
989                           }
990                         }
991                       }
992                     });
993                   }
994                 });
995                 doneReceiver.addAll(receivers.getValue());
996                 doneField.addAll(fields.getValue());
997               }
998     
999               return NOT_CHANGED;
1000             }
1001     
1002             @Override
1003             public String toString() {
1004               return "field op";
1005             }
1006     
1007             @Override
1008             public boolean equals(Object o) {
1009               return o == this;
1010             }
1011     
1012             @Override
1013             public int hashCode() {
1014               return System.identityHashCode(this);
1015             }
1016           }, objKey, fieldKey);
1017         }
