83       public class AstCGNode extends ExplicitNode {
84         private Set<Function<Object, Object>> callbacks;
85     
86         private IR cachedIR;
87         
88         private DefUse cachedDU;
89         
90         private AstCGNode(IMethod method, Context context) {
91           super(method, context);
92         }
93     
94         private void fireCallbacks() {
95           if (callbacks != null) {
96             boolean done = false;
97             while (!done) {
98               try {
99                 for (Iterator<Function<Object, Object>> x = callbacks.iterator(); x.hasNext();) {
100                   x.next().apply(null);
101                 }
102               } catch (ConcurrentModificationException e) {
103                 done = false;
104                 continue;
105               }
106               done = true;
107             }
108           }
109         }
110     
111         private boolean hasCallback(Function<Object, Object> callback) {
112           return callbacks != null && callbacks.contains(callback);
113         }
114     
115         private boolean hasAllCallbacks(Set<Function<Object, Object>> callbacks) {
116           return callbacks != null && callbacks.containsAll(callbacks);
117         }
118     
119         public void addCallback(Function<Object, Object> callback) {
120           if (!hasCallback(callback)) {
121             if (callbacks == null) {
122               callbacks = HashSetFactory.make(1);
123             }
124     
125             callbacks.add(callback);
126     
127             for (Iterator ps = getCallGraph().getPredNodes(this); ps.hasNext();) {
128               ((AstCGNode) ps.next()).addCallback(callback);
129             }
130           }
131         }
132     
133         public void addAllCallbacks(Set<Function<Object, Object>> callback) {
134           if (!hasAllCallbacks(callbacks)) {
135             if (callbacks == null) {
136               callbacks = HashSetFactory.make(1);
137             }
138     
139             callbacks.addAll(callback);
140     
141             for (Iterator ps = getCallGraph().getPredNodes(this); ps.hasNext();) {
142               ((AstCGNode) ps.next()).addAllCallbacks(callback);
143             }
144           }
145         }
146     
147         public void clearMutatedCache(CallSiteReference cs) {
148           targets.remove(cs.getProgramCounter());
149         }
150         
151         @Override
152         public boolean addTarget(CallSiteReference site, CGNode node) {
153           if (super.addTarget(site, node)) {
154             if (((AstCGNode) node).callbacks != null) {
155               ((AstCGNode) node).fireCallbacks();
156               addAllCallbacks(((AstCGNode) node).callbacks);
157             }
158             return true;
159           } else {
160             return false;
161           }
162         }
163       }
164

166       protected ExplicitNode makeNode(IMethod method, Context context) {
167         return new AstCGNode(method, context);
168       }
