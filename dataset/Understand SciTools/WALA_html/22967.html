11     package com.ibm.wala.ssa;
12     
13     import java.util.ArrayList;
14     import java.util.Iterator;
15     
16     import com.ibm.wala.util.collections.EmptyIterator;
17     import com.ibm.wala.util.debug.Assertions;
18     import com.ibm.wala.util.intset.IntIterator;
19     import com.ibm.wala.util.intset.IntSet;
20     import com.ibm.wala.util.intset.IntSetUtil;
21     import com.ibm.wala.util.intset.MutableIntSet;
22     
23     /**
24      * An object which represent Def-Use information for an SSA {@link IR}
25      */
26     public class DefUse {
27       static final boolean DEBUG = false;
28     
29       /**
30        * A mapping from integer (value number) -> {@link SSAInstruction} that defines the value
31        */
32       final private SSAInstruction[] defs;
33     
34       /**
35        * A mapping from integer (value number) -> bit vector holding integers representing instructions that use the value number
36        */
37       final private MutableIntSet[] uses;
38     
39       /**
40        * A Mapping from integer -> Instruction
41        */
42       final protected ArrayList<SSAInstruction> allInstructions = new ArrayList<SSAInstruction>();
43     
44       /**
45        * prevent the IR from being collected while this is live.
46        */
47       private final IR ir;
48     
49       /**
50        * @param ir an IR in SSA form.
51        * @throws IllegalArgumentException if ir is null
52        */
53       public DefUse(final IR ir) {
54         if (ir == null) {
55           throw new IllegalArgumentException("ir is null");
56         }
57         this.ir = ir;
58     
59         // set up mapping from integer -> instruction
60         initAllInstructions();
61         defs = new SSAInstruction[getMaxValueNumber() + 1];
62         uses = new MutableIntSet[getMaxValueNumber() + 1];
63         if (DEBUG) {
64           System.err.println(("DefUse: defs.length " + defs.length));
65         }
66         Iterator it = allInstructions.iterator();
67         for (int i = 0; i < allInstructions.size(); i++) {
68           SSAInstruction s = (SSAInstruction) it.next();
69           if (s == null) {
70             continue;
71           }
72           for (int j = 0; j < getNumberOfDefs(s); j++) {
73             defs[getDef(s, j)] = s;
74           }
75           for (int j = 0; j < getNumberOfUses(s); j++) {
76             int use = getUse(s, j);
77             try {
78               if (use != -1) {
79                 if (uses[use] == null) {
80                   uses[use] = IntSetUtil.make();
81                 }
82                 uses[use].add(i);
83               }
84             } catch (ArrayIndexOutOfBoundsException e) {
85               throw new InternalError("unexpected value number " + use);
86             }
87           }
88         }
89       }
90     
91       /**
92        * @return the maximum value number in a particular IR
93        */
94       protected int getMaxValueNumber() {
95         return ir.getSymbolTable().getMaxValueNumber();
96       }
97     
98       /**
99        * Initialize the allInstructions field with every {@link SSAInstruction} in the ir.
100        */
101       protected void initAllInstructions() {
102         for (Iterator<SSAInstruction> it = ir.iterateAllInstructions(); it.hasNext();) {
103           allInstructions.add(it.next());
104         }
105       }
106     
107       /**
108        * What is the ith value number defined by instruction s?
109        */
110       protected int getDef(SSAInstruction s, int i) {
111         return s.getDef(i);
112       }
113     
114       /**
115        * What is the ith value number used by instruction s?
116        */
117       protected int getUse(SSAInstruction s, int i) {
118         return s.getUse(i);
119       }
120     
121       /**
122        * How many value numbers does instruction s def?
123        */
124       protected int getNumberOfDefs(SSAInstruction s) {
125         return s.getNumberOfDefs();
126       }
127     
128       /**
129        * How many value numbers does instruction s use?
130        */
131       protected int getNumberOfUses(SSAInstruction s) {
132         return s.getNumberOfUses();
133       }
134     
135       /**
136        * @return the {@link SSAInstruction} that defines the variable with value number v.
137        */
138       public SSAInstruction getDef(int v) {
139         return (v < defs.length) ? defs[v] : null;
140       }
141     
142       /**
143        * Return all uses of the variable with the given value number
144        */
145       public Iterator<SSAInstruction> getUses(int v) {
146         if (uses[v] == null) {
147           return EmptyIterator.instance();
148         } else {
149           return new UseIterator(uses[v]);
150         }
151       }
152     
153       /**
154        * return an {@link Iterator} of all instructions that use any of a set of variables
155        */
156       private class UseIterator implements Iterator<SSAInstruction> {
157         final IntIterator it;
158     
159         /**
160          * @param uses the set of value numbers whose uses this object iterates over
161          */
162         UseIterator(IntSet uses) {
163           it = uses.intIterator();
164         }
165     
166         @Override
167         public boolean hasNext() {
168           return it.hasNext();
169         }
170     
171         @Override
172         public SSAInstruction next() {
173           return allInstructions.get(it.next());
174         }
175     
176         @Override
177         public void remove() {
178           Assertions.UNREACHABLE();
179         }
180       }
181     
182       /**
183        * @param v a value number
184        * @return the number of uses of the variable with the given value number
185        */
186       public int getNumberOfUses(int v) {
187         return uses[v] == null ? 0 : uses[v].size();
188       }
189     }
