11     package com.ibm.wala.types;
12     
13     import java.util.HashMap;
14     
15     import com.ibm.wala.util.collections.HashMapFactory;
16     import com.ibm.wala.util.debug.Assertions;
17     import com.ibm.wala.util.debug.UnimplementedError;
18     import com.ibm.wala.util.shrike.ShrikeUtil;
19     import com.ibm.wala.util.strings.Atom;
20     
21     /**
22      * A class to represent the reference in a class file to a field.
23      */
24     public final class FieldReference extends MemberReference {
25       private final static boolean DEBUG = false;
26     
27       /**
28        * Used to canonicalize MemberReferences a mapping from Key -> MemberReference
29        */
30       final private static HashMap<Key, FieldReference> dictionary = HashMapFactory.make();
31     
32       private final TypeReference fieldType;
33     
34       @Override
35       public String getSignature() {
36         return getDeclaringClass().getName() + "." + getName() + " " + getFieldType().getName();
37       }
38     
39       /**
40        * Find or create the canonical MemberReference instance for the given tuple.
41        * 
42        * @param mn the name of the member
43        */
44       public static synchronized FieldReference findOrCreate(TypeReference tref, Atom mn, TypeReference fieldType) {
45         if (tref == null) {
46           throw new IllegalArgumentException("null tref");
47         }
48         Key key = new Key(tref, mn, fieldType);
49     
50         FieldReference val = dictionary.get(key);
51         if (val != null) {
52           return val;
53         }
54     
55         val = new FieldReference(key, fieldType);
56     
57         dictionary.put(key, val);
58         return val;
59       }
60     
61       /**
62        * Find or create the canonical MemberReference instance for the given tuple.
63        */
64       public static FieldReference findOrCreate(ClassLoaderReference loader, String classType, String fieldName, String fieldType)
65           throws IllegalArgumentException {
66         TypeReference c = ShrikeUtil.makeTypeReference(loader, classType);
67         TypeReference ft = ShrikeUtil.makeTypeReference(loader, fieldType);
68         Atom name = Atom.findOrCreateUnicodeAtom(fieldName);
69         return findOrCreate(c, name, ft);
70       }
71     
72       private FieldReference(Key key, TypeReference fieldType) {
73         super(key.type, key.name, key.hashCode());
74         this.fieldType = fieldType;
75         if (DEBUG) {
76           if (getName().toString().indexOf('.') > -1)
77             throw new UnimplementedError();
78           if (fieldType.toString().indexOf('.') > -1)
79             Assertions.UNREACHABLE("Field name: " + fieldType.toString());
80           if (getName().toString().length() == 0)
81             throw new UnimplementedError();
82           if (fieldType.toString().length() == 0)
83             throw new UnimplementedError();
84         }
85       }
86     
87       /**
88        * @return the descriptor component of this member reference
89        */
90       public final TypeReference getFieldType() {
91         return fieldType;
92       }
93     
94       @Override
95       public final String toString() {
96         return "< " + getDeclaringClass().getClassLoader().getName() + ", " + getDeclaringClass().getName() + ", " + getName() + ", "
97             + fieldType + " >";
98       }
99     
100       /**
101        * An identifier/selector for fields.
102        */
103       protected static class Key {
104         final TypeReference type;
105     
106         final Atom name;
107     
108         private final TypeReference fieldType;
109     
110         Key(TypeReference type, Atom name, TypeReference fieldType) {
111           this.type = type;
112           this.name = name;
113           this.fieldType = fieldType;
114         }
115     
116         @Override
117         public final int hashCode() {
118           return 7487 * type.hashCode() + name.hashCode();
119         }
120     
121         @Override
122         public final boolean equals(Object other) {
123           assert other != null && this.getClass().equals(other.getClass());
124     
125           Key that = (Key) other;
126           return type.equals(that.type) && name.equals(that.name) && fieldType.equals(that.fieldType);
127         }
128       }
129     
130     }
