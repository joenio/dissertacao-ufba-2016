38     public class JavaEclipseProjectPath extends EclipseProjectPath<IClasspathEntry, IJavaProject> {
39     
40     	public enum JavaSourceLoader implements ILoader {
41     		SOURCE(JavaSourceAnalysisScope.SOURCE);
42     		
43     	    private ClassLoaderReference ref;
44     
45     	    JavaSourceLoader(ClassLoaderReference ref) {
46     	      this.ref = ref;
47     	    }
48     
49     		@Override
50     		public ClassLoaderReference ref() {
51     			return ref;
52     		}
53     	}
54     	
55       protected JavaEclipseProjectPath(com.ibm.wala.ide.util.EclipseProjectPath.AnalysisScopeType scopeType)
56           throws IOException, CoreException {
57         super(scopeType);
58       }
59     
60       public static JavaEclipseProjectPath make(IJavaProject p, AnalysisScopeType scopeType) throws IOException, CoreException {
61         JavaEclipseProjectPath path = new JavaEclipseProjectPath(scopeType);
62         path.create(p.getProject());
63         return path;
64       }
65     
66       @Override
67       protected IJavaProject makeProject(IProject p) {
68         try {
69           if (p.hasNature(JavaCore.NATURE_ID)) {
70             return JavaCore.create(p);
71           }
72         } catch (CoreException e) {
73           // not a Java project
74         } 
75         return null;
76       }
77     
78       @Override
79       protected IClasspathEntry resolve(IClasspathEntry entry) {
80         return JavaCore.getResolvedClasspathEntry(entry);
81       }
82     
83       @Override
84       protected void resolveClasspathEntry(IJavaProject project, IClasspathEntry entry, ILoader loader, boolean includeSource, boolean cpeFromMainProject) {
85     	  entry = JavaCore.getResolvedClasspathEntry(entry);
86     	  switch (entry.getEntryKind()) {
87     	  case IClasspathEntry.CPE_SOURCE: {
88     		  resolveSourcePathEntry(includeSource? JavaSourceLoader.SOURCE: Loader.APPLICATION, includeSource, cpeFromMainProject, entry.getPath(), entry.getOutputLocation(), entry.getExclusionPatterns()
89     , "java");
90     		  break;
91     	  }
92     	  case IClasspathEntry.CPE_LIBRARY: {
93     		  resolveLibraryPathEntry(loader, entry.getPath());
94     		  break;
95     	  }
96     	  case IClasspathEntry.CPE_PROJECT: {
97     		  resolveProjectPathEntry(loader, includeSource, entry.getPath());
98     		  break;
99     	  }
100     	  case IClasspathEntry.CPE_CONTAINER: {
101     		  try {
102     		    IClasspathContainer cont = JavaCore.getClasspathContainer(entry.getPath(), project);
103     		    IClasspathEntry[] entries = cont.getClasspathEntries();
104     		    resolveClasspathEntries(project, Arrays.asList(entries), cont.getKind() == IClasspathContainer.K_APPLICATION ? loader : Loader.PRIMORDIAL,
105     		        includeSource, false);
106     		  } catch (CoreException e) {
107     			  System.err.println(e);
108     			  Assertions.UNREACHABLE();
109     		  }
110     	  }
111     	  }
112       }
113     
114       @Override
115       protected void resolveProjectClasspathEntries(IJavaProject project, boolean includeSource) {
116     	try {
117     		resolveClasspathEntries(project, Arrays.asList(project.getRawClasspath()), Loader.EXTENSION, includeSource, true);
118     		File output = makeAbsolute(project.getOutputLocation()).toFile();
119     		if (!includeSource) {
120     			if (output.exists()) {
121     				List<Module> s = MapUtil.findOrCreateList(modules, Loader.APPLICATION);
122     				s.add(new BinaryDirectoryTreeModule(output));
123     			}
124     		}
125     	} catch (JavaModelException e) {
126     		e.printStackTrace();
127     		Assertions.UNREACHABLE();
128     	}
129       }
130     }
