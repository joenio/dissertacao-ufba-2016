38     /**
39      * Abstract superclass for types performing a rewrite operation on a CAst. The
40      * CAst is not mutated; instead, a new CAst is created which delegates to the
41      * original CAst where no transformation was performed.
42      * 
43      * @param <C>
44      *          type of the RewriteContext used when traversing the original CAst
45      *          during the rewrite operation
46      * @param <K>
47      *          a key used to ease cloning of partial ASTs. When rewriting an AST,
48      *          sub-classes maintain a mapping from (original node, key) pairs
49      *          (where key is of type K) to new nodes; see
50      *          {@link #copyNodes(CAstNode, RewriteContext, Map)}
51      */
52     public abstract class CAstRewriter<C extends CAstRewriter.RewriteContext<K>, K extends CAstRewriter.CopyKey<K>> {
53     
54       protected static final boolean DEBUG = false;
55     
56       /**
57        * interface to be implemented by keys used for cloning sub-trees during the
58        * rewrite
59        */
60       public interface CopyKey<Self extends CopyKey> {
61     
62         @Override
63         int hashCode();
64     
65         @Override
66         boolean equals(Object o);
67     
68         /**
69          * keys have parent pointers, useful for when nesting cloning must occur
70          * (e.g., unrolling of nested loops)
71          */
72         Self parent();
73     
74       };
75     
76       /**
77        * interface to be implemented by contexts used while traversing the AST
78        */
79       public interface RewriteContext<K extends CopyKey> {
80     
81         /**
82          * get the cloning key for this context
83          */
84         K key();
85     
86       };

121       /**
122        * rewrite the CAst rooted at root under some context, returning the node at
123        * the root of the rewritten tree. mutate nodeMap in the process, indicating
124        * how (original node, copy key) pairs are mapped to nodes in the rewritten
125        * tree.
126        */
127       protected abstract CAstNode copyNodes(CAstNode root, final CAstControlFlowMap cfg, C context, Map<Pair<CAstNode, K>, CAstNode> nodeMap);
128     
129       /**
130        * in {@link #copyFlow(Map, CAstControlFlowMap, CAstSourcePositionMap)}, if
131        * the source of some original CFG edge is replicated, but we find no replica
132        * for the target, what node should be the target of the CFG edge in the
133        * rewritten AST? By default, just uses the original target.
134        * 
135        */
136       protected CAstNode flowOutTo(Map<Pair<CAstNode, K>, CAstNode> nodeMap, CAstNode oldSource, Object label, CAstNode oldTarget,
137           CAstControlFlowMap orig, CAstSourcePositionMap src) {
138         return oldTarget;
139       }

141       /**
142        * create a control-flow map for the rewritten tree, given the mapping from
143        * (original node, copy key) pairs ot new nodes and the original control-flow
144        * map.
145        */
146       protected CAstControlFlowMap copyFlow(Map<Pair<CAstNode, K>, CAstNode> nodeMap, CAstControlFlowMap orig,
147           CAstSourcePositionMap newSrc) {
148     
149         // the new control-flow map
150         final CAstControlFlowRecorder newMap = new CAstControlFlowRecorder(newSrc);
151         // tracks which CAstNodes not present in nodeMap's key set (under any copy
152         // key) are added as targets of CFG edges
153         // via a call to flowOutTo() (see below); used to ensure these nodes are
154         // only mapped to themselves once in newMap
155         final Set<CAstNode> mappedOutsideNodes = HashSetFactory.make(1);
156         // all edge targets in new control-flow map; must all be mapped to
157         // themselves
158         Set<CAstNode> allNewTargetNodes = HashSetFactory.make(1);
159         Collection<CAstNode> oldSources = orig.getMappedNodes();
160     
161         for (Iterator<Entry<Pair<CAstNode, K>, CAstNode>> NS = nodeMap.entrySet().iterator(); NS.hasNext();) {
162           Entry<Pair<CAstNode, K>, CAstNode> entry = NS.next();
163           Pair<CAstNode, K> N = entry.getKey();
164           CAstNode oldSource = N.fst;
165           K key = N.snd;
166     
167           CAstNode newSource = entry.getValue();
168           assert newSource != null;
169     
170           newMap.map(newSource, newSource);
171     
172           if (DEBUG) {
173             System.err.println(("\n\nlooking at " + key + ":" + CAstPrinter.print(oldSource)));
174           }
175     
176           if (oldSources.contains(oldSource)) {
177             Iterator<Object> LS = orig.getTargetLabels(oldSource).iterator();
178             //if (orig.getTarget(oldSource, null) != null) {
179             //  LS = IteratorPlusOne.make(LS, null);
180             //}
181     
182             while (LS.hasNext()) {
183               Object origLabel = LS.next();
184               CAstNode oldTarget = orig.getTarget(oldSource, origLabel);
185               assert oldTarget != null;
186     
187               if (DEBUG) {
188                 System.err.println(("old: " + origLabel + " --> " + CAstPrinter.print(oldTarget)));
189               }
190     
191               // try to find a k in key's parent chain such that (oldTarget, k) is
192               // in nodeMap's key set
193               Pair targetKey;
194               CopyKey k = key;
195               do {
196                 targetKey = Pair.make(oldTarget, k);
197                 if (k != null) {
198                   k = k.parent();
199                 } else {
200                   break;
201                 }
202               } while (!nodeMap.containsKey(targetKey));
203     
204               Object newLabel;
205               if (nodeMap.containsKey(Pair.make(origLabel, targetKey.snd))) { // label
206                                                                               // is
207                                                                               // mapped
208                                                                               // too
209                 newLabel = nodeMap.get(Pair.make(origLabel, targetKey.snd));
210               } else {
211                 newLabel = origLabel;
212               }
213     
214               CAstNode newTarget;
215               if (nodeMap.containsKey(targetKey)) {
216                 newTarget = nodeMap.get(targetKey);
217                 newMap.add(newSource, newTarget, newLabel);
218                 allNewTargetNodes.add(newTarget);
219     
220               } else {
221                 // could not discover target of CFG edge in nodeMap under any key related to the current source key.
222                 // the edge might have been deleted, or it may end at a node above the root where we were
223                 // rewriting
224                 // ask flowOutTo() to just choose a target
225                 newTarget = flowOutTo(nodeMap, oldSource, origLabel, oldTarget, orig, newSrc);
226                 allNewTargetNodes.add(newTarget);
227                 newMap.add(newSource, newTarget, newLabel);
228                 if (newTarget != CAstControlFlowMap.EXCEPTION_TO_EXIT && !mappedOutsideNodes.contains(newTarget)) {
229                   mappedOutsideNodes.add(newTarget);
230                   newMap.map(newTarget, newTarget);
231                 }
232               }
233     
234               if (DEBUG) {
235                 System.err.println(("mapping:old: " + CAstPrinter.print(oldSource) + "-- " + origLabel + " --> " + CAstPrinter
236                     .print(oldTarget)));
237                 System.err.println(("mapping:new: " + CAstPrinter.print(newSource) + "-- " + newLabel + " --> " + CAstPrinter
238                     .print(newTarget)));
239               }
240             }
241           }
242         }
243     
244         allNewTargetNodes.removeAll(newMap.getMappedNodes());
245         for (CAstNode newTarget : allNewTargetNodes) {
246           if (newTarget != CAstControlFlowMap.EXCEPTION_TO_EXIT) {
247             newMap.map(newTarget, newTarget); 
248           }
249         }
250         
251         
252         assert !oldNodesInNewMap(nodeMap, newMap);
253     
254         return newMap;
255       }

257       // check whether newMap contains any CFG edges involving nodes in the domain of nodeMap
258       private boolean oldNodesInNewMap(Map<Pair<CAstNode, K>, CAstNode> nodeMap, final CAstControlFlowRecorder newMap) {
259         HashSet<CAstNode> oldNodes = HashSetFactory.make();
260         for(Entry<Pair<CAstNode, K>, CAstNode> e : nodeMap.entrySet())
261           oldNodes.add(e.getKey().fst);
262         for(CAstNode mappedNode : newMap.getMappedNodes()) {
263           if(oldNodes.contains(mappedNode))
264             return true;
265           for(Object lbl : newMap.getTargetLabels(mappedNode))
266             if(oldNodes.contains(newMap.getTarget(mappedNode, lbl)))
267               return true;
268         }
269         return false;
270       }

272       protected CAstSourcePositionMap copySource(Map<Pair<CAstNode, K>, CAstNode> nodeMap, CAstSourcePositionMap orig) {
273         CAstSourcePositionRecorder newMap = new CAstSourcePositionRecorder();
274         for (Iterator<Map.Entry<Pair<CAstNode, K>, CAstNode>> NS = nodeMap.entrySet().iterator(); NS.hasNext();) {
275           Map.Entry<Pair<CAstNode, K>, CAstNode> entry = NS.next();
276           Pair<CAstNode, K> N = entry.getKey();
277           CAstNode oldNode = N.fst;
278     
279           CAstNode newNode = entry.getValue();
280     
281           if (orig.getPosition(oldNode) != null) {
282             newMap.setPosition(newNode, orig.getPosition(oldNode));
283           }
284         }
285     
286         return newMap;
287       }

289       protected CAstNodeTypeMap copyTypes(Map<Pair<CAstNode, K>, CAstNode> nodeMap, CAstNodeTypeMap orig) {
290         if (orig != null) {
291           CAstNodeTypeMapRecorder newMap = new CAstNodeTypeMapRecorder();
292           for (Iterator<Entry<Pair<CAstNode, K>, CAstNode>> NS = nodeMap.entrySet().iterator(); NS.hasNext();) {
293             Entry<Pair<CAstNode, K>, CAstNode> entry = NS.next();
294             Pair<CAstNode, K> N = entry.getKey();
295             CAstNode oldNode = N.fst;
296     
297             CAstNode newNode = entry.getValue();
298     
299             if (orig.getNodeType(oldNode) != null) {
300               newMap.add(newNode, orig.getNodeType(oldNode));
301             }
302           }
303     
304           return newMap;
305         } else {
306           return null;
307         }
308       }

310       protected Map<CAstNode, Collection<CAstEntity>> copyChildren(CAstNode root, Map<Pair<CAstNode, K>, CAstNode> nodeMap,
311           Map<CAstNode, Collection<CAstEntity>> children) {
312         final Map<CAstNode, Collection<CAstEntity>> newChildren = new LinkedHashMap<CAstNode, Collection<CAstEntity>>();
313     
314         for (Iterator<Entry<Pair<CAstNode, K>, CAstNode>> NS = nodeMap.entrySet().iterator(); NS.hasNext();) {
315           Entry<Pair<CAstNode, K>, CAstNode> entry = NS.next();
316           Pair<CAstNode, K> N = entry.getKey();
317           CAstNode oldNode = N.fst;
318     
319           CAstNode newNode = entry.getValue();
320     
321           if (children.containsKey(oldNode)) {
322             Set<CAstEntity> newEntities = new LinkedHashSet<CAstEntity>();
323             newChildren.put(newNode, newEntities);
324             for (Iterator oldEntities = ((Collection) children.get(oldNode)).iterator(); oldEntities.hasNext();) {
325               newEntities.add(rewrite((CAstEntity) oldEntities.next()));
326             }
327           }
328         }
329     
330         for (Iterator<Map.Entry<CAstNode, Collection<CAstEntity>>> keys = children.entrySet().iterator(); keys.hasNext();) {
331           Map.Entry<CAstNode, Collection<CAstEntity>> entry = keys.next();
332           CAstNode key = entry.getKey();
333           if (key == null) {
334             Set<CAstEntity> newEntities = new LinkedHashSet<CAstEntity>();
335             newChildren.put(key, newEntities);
336             for (Iterator oldEntities = ((Collection) entry.getValue()).iterator(); oldEntities.hasNext();) {
337               newEntities.add(rewrite((CAstEntity) oldEntities.next()));
338             }
339           }
340         }
341     
342         return newChildren;
343       }

345       /**
346        * rewrite the CAst sub-tree rooted at root
347        */
348       public Rewrite rewrite(final CAstNode root, final CAstControlFlowMap cfg, final CAstSourcePositionMap pos, final CAstNodeTypeMap types,
349           final Map<CAstNode, Collection<CAstEntity>> children) {
350         final Map<Pair<CAstNode, K>, CAstNode> nodes = HashMapFactory.make();
351         final CAstNode newRoot = copyNodes(root, cfg, rootContext, nodes);
352         return new Rewrite() {
353           private CAstControlFlowMap theCfg = null;
354     
355           private CAstSourcePositionMap theSource = null;
356     
357           private CAstNodeTypeMap theTypes = null;
358     
359           private Map<CAstNode, Collection<CAstEntity>> theChildren = null;
360     
361           @Override
362           public CAstNode newRoot() {
363             return newRoot;
364           }
365     
366           @Override
367           public CAstControlFlowMap newCfg() {
368             if (theCfg == null)
369               theCfg = copyFlow(nodes, cfg, newPos());
370             return theCfg;
371           }
372     
373           @Override
374           public CAstSourcePositionMap newPos() {
375             if (theSource == null)
376               theSource = copySource(nodes, pos);
377             return theSource;
378           }
379     
380           @Override
381           public CAstNodeTypeMap newTypes() {
382             if (theTypes == null)
383               theTypes = copyTypes(nodes, types);
384             return theTypes;
385           }
386     
387           @Override
388           public Map<CAstNode, Collection<CAstEntity>> newChildren() {
389             if (theChildren == null)
390               theChildren = copyChildren(root, nodes, children);
391             return theChildren;
392           }
393         };
394       }
395     
396       /**
397        * perform the rewrite on a {@link CAstEntity}, returning the new
398        * {@link CAstEntity} as the result
399        */
400       public CAstEntity rewrite(final CAstEntity root) {
401     
402         if (root.getAST() != null) {
403           final Rewrite rewrite = rewrite(root.getAST(), root.getControlFlow(), root.getSourceMap(), root.getNodeTypeMap(),
404               root.getAllScopedEntities());
405     
406           return new DelegatingEntity(root) {
407             @Override
408             public String toString() {
409               return root.toString() + " (clone)";
410             }
411     
412             @Override
413             public Iterator<CAstEntity> getScopedEntities(CAstNode construct) {
414               Map<CAstNode, Collection<CAstEntity>> newChildren = getAllScopedEntities();
415               if (newChildren.containsKey(construct)) {
416                 return newChildren.get(construct).iterator();
417               } else {
418                 return EmptyIterator.instance();
419               }
420             }
421     
422             @Override
423             public Map<CAstNode, Collection<CAstEntity>> getAllScopedEntities() {
424               return rewrite.newChildren();
425             }
426     
427             @Override
428             public CAstNode getAST() {
429               return rewrite.newRoot();
430             }
431     
432             @Override
433             public CAstNodeTypeMap getNodeTypeMap() {
434               return rewrite.newTypes();
435             }
436     
437             @Override
438             public CAstSourcePositionMap getSourceMap() {
439               return rewrite.newPos();
440             }
441     
442             @Override
443             public CAstControlFlowMap getControlFlow() {
444               return rewrite.newCfg();
445             }
446           };
447     
448         } else if (recursive) {
449     
450           Map<CAstNode, Collection<CAstEntity>> children = root.getAllScopedEntities();
451           final Map<CAstNode, Collection<CAstEntity>> newChildren = new LinkedHashMap<CAstNode, Collection<CAstEntity>>();
452           for (Iterator<Map.Entry<CAstNode, Collection<CAstEntity>>> keys = children.entrySet().iterator(); keys.hasNext();) {
453             Map.Entry<CAstNode, Collection<CAstEntity>> entry = keys.next();
454             CAstNode key = entry.getKey();
455             Set<CAstEntity> newValues = new LinkedHashSet<CAstEntity>();
456             newChildren.put(key, newValues);
457             for (Iterator es = entry.getValue().iterator(); es.hasNext();) {
458               newValues.add(rewrite((CAstEntity) es.next()));
459             }
460           }
461     
462           return new DelegatingEntity(root) {
463             @Override
464             public String toString() {
465               return root.toString() + " (clone)";
466             }
467     
468             @Override
469             public Iterator<CAstEntity> getScopedEntities(CAstNode construct) {
470               if (newChildren.containsKey(construct)) {
471                 return newChildren.get(construct).iterator();
472               } else {
473                 return EmptyIterator.instance();
474               }
475             }
476     
477             @Override
478             public Map<CAstNode, Collection<CAstEntity>> getAllScopedEntities() {
479               return newChildren;
480             }
481           };
482     
483         } else {
484           return root;
485         }
486       }
