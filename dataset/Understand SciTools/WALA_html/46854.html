14     package com.ibm.wala.cast.java.translator.polyglot;
15     
16     import java.io.File;
17     import java.io.IOException;
18     import java.io.InputStream;
19     
20     import polyglot.frontend.FileSource;
21     import polyglot.frontend.Resource;
22     
23     import com.ibm.wala.classLoader.SourceModule;
24     
25     /**
26      * A Polyglot Source whose input comes from an InputStream.<br>
27      * Currently extends FileSource since that's all that the Polyglot Compiler class
28      * will accept.
29      * @author rfuhrer
30      */
31     public class ModuleSource extends FileSource {
32       private final SourceModule module;
33       
34       SourceModule getModule() {
35         return module;
36       }
37       
38       public ModuleSource(final SourceModule module) throws IOException {
39         super(new Resource() {
40           @Override
41           public File file() {
42             return new File(module.getURL().getFile());
43           }
44     
45           @Override
46           public InputStream getInputStream() throws IOException {
47             return module.getInputStream();
48           }
49     
50           @Override
51           public String name() {
52             String fullPath = module.getURL().getFile();
53             int idx= fullPath.lastIndexOf(File.separatorChar);
54             return (idx > 0) ? fullPath.substring(idx+1) : fullPath;
55           }
56           @Override
57           public String toString() {
58             return module.getName();
59           }
60         }, true);
61         this.module = module;
62       }
63     }
