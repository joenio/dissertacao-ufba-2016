31     public class EclipseWebAnalysisEngine extends EclipseJavaScriptAnalysisEngine {
32     
33       private final Set<Pair<String, Plugin>> models = HashSetFactory.make();
34       
35       public EclipseWebAnalysisEngine(IJavaScriptProject project, Collection<Pair<String, Plugin>> models, BuilderType builderType) throws IOException, CoreException {
36         super(project, builderType);
37         // core DOM model
38         this.models.add(Pair.make("preamble.js", (Plugin)Activator.getDefault()));
39         this.models.addAll(models);
40       }
41     
42       @Override
43       protected ClassLoaderFactory makeClassLoaderFactory(SetOfClasses exclusions) {
44         return new WebPageLoaderFactory(new CAstRhinoTranslatorFactory());
45       }
46     
47       @Override
48       protected AnalysisScope makeAnalysisScope() {
49         return new CAstAnalysisScope(new WebPageLoaderFactory(new CAstRhinoTranslatorFactory()), Collections.singleton(JavaScriptLoader.JS));
50       }
51     
52       @Override
53       protected JavaScriptEclipseProjectPath createProjectPath(IJavaScriptProject project) throws IOException, CoreException {
54         return EclipseWebProjectPath.make(project, models);
55       }
56     
57       @Override
58       public Pair<JSCallGraph, PointerAnalysis<ObjectVertex>> getFieldBasedCallGraph(String scriptName) throws CancelException {
59         Set<Entrypoint> eps= HashSetFactory.make();
60         eps.add(JSCallGraphUtil.makeScriptRoots(getClassHierarchy()).make(scriptName));
61         eps.add(JSCallGraphUtil.makeScriptRoots(getClassHierarchy()).make("Lprologue.js"));
62         
63         for(Pair<String,Plugin> model : models) {
64           eps.add(JSCallGraphUtil.makeScriptRoots(getClassHierarchy()).make("L" + model.fst));
65         }
66     
67         return getFieldBasedCallGraph(eps);
68       }
