22     import com.ibm.wala.util.collections.HashMapFactory;

24     import com.ibm.wala.util.collections.NonNullSingletonIterator;

27     /**
28      * This class searches breadth-first for node that matches some criteria. If found, it reports a path to the first node found.
29      * 
30      * This class follows the outNodes of the graph nodes to define the graph, but this behavior can be changed by overriding the
31      * getConnected method.
32      * 
33      * TODO: if finding many paths, use a dynamic programming algorithm instead of calling this repeatedly.
34      */
35     public class BFSPathFinder<T> {
36     
37       private final boolean DEBUG = false;
38     
39       /**
40        * The graph to search
41        */
42       final private Graph<T> G;
43     
44       /**
45        * The Filter which defines the target set of nodes to find
46        */
47       final private Predicate<T> filter;
48     
49       /**
50        * an enumeration of all nodes to search from
51        */
52       final private Iterator<T> roots;
53     
54       /**
55        * Construct a breadth-first enumerator starting with a particular node in a directed graph.
56        * 
57        * @param G the graph whose nodes to enumerate
58        */
59       public BFSPathFinder(Graph<T> G, T N, Predicate<T> f) {
60         if (G == null) {
61           throw new IllegalArgumentException("G is null");
62         }
63         if (f == null) {
64           throw new IllegalArgumentException("null f");
65         }
66         this.G = G;
67         this.roots = new NonNullSingletonIterator<T>(N);
68         this.filter = f;
69       }
70     
71       /**
72        * Construct a breadth-first enumerator starting with a particular node in a directed graph.
73        * 
74        * @param G the graph whose nodes to enumerate
75        * @throws IllegalArgumentException if G is null
76        */
77       public BFSPathFinder(Graph<T> G, T src, final T target) throws IllegalArgumentException {
78         if (G == null) {
79           throw new IllegalArgumentException("G is null");
80         }
81         this.G = G;
82         this.roots = new NonNullSingletonIterator<T>(src);
83         if (!G.containsNode(src)) {
84           throw new IllegalArgumentException("src is not in graph " + src);
85         }
86         this.filter = new Predicate<T>() {
87           @Override public boolean test(T o) {
88             return target.equals(o);
89           }
90         };
91       }
92     
93       /**
94        * Construct a breadth-first enumerator starting with a particular node in a directed graph.
95        * 
96        * @param G the graph whose nodes to enumerate
97        */
98       public BFSPathFinder(Graph<T> G, T src, Iterator<T> targets) {
99         if (targets == null) {
100           throw new IllegalArgumentException("targets is null");
101         }
102         final Set<T> ts = HashSetFactory.make();
103         while (targets.hasNext()) {
104           ts.add(targets.next());
105         }
106     
107         this.G = G;
108         this.roots = new NonNullSingletonIterator<T>(src);
109     
110         this.filter = new Predicate<T>() {
111           @Override public boolean test(T o) {
112             return ts.contains(o);
113           }
114         };
115       }
116     
117       /**
118        * Construct a breadth-first enumerator starting with any of a set of nodes in a directed graph.
119        * 
120        * @param G the graph whose nodes to enumerate
121        */
122       public BFSPathFinder(Graph<T> G, Iterator<T> sources, final T target) {
123         if (G == null) {
124           throw new IllegalArgumentException("G is null");
125         }
126         if (sources == null) {
127           throw new IllegalArgumentException("sources is null");
128         }
129         this.G = G;
130         this.roots = sources;
131         this.filter = new Predicate<T>() {
132           @Override public boolean test(T o) {
133             return target.equals(o);
134           }
135         };
136       }
137     
138       /**
139        * Construct a breadth-first enumerator across the (possibly improper) subset of nodes reachable from the nodes in the given
140        * enumeration.
141        * 
142        * @param nodes the set of nodes from which to start searching
143        */
144       public BFSPathFinder(Graph<T> G, Iterator<T> nodes, Predicate<T> f) {
145         this.G = G;
146         this.roots = nodes;
147         this.filter = f;
148         if (G == null) {
149           throw new IllegalArgumentException("G is null");
150         }
151         if (roots == null) {
152           throw new IllegalArgumentException("roots is null");
153         }
154       }
155     
156       /**
157        * @return a List of nodes that specifies the first path found from a root to a node accepted by the filter. Returns null if no
158        *         path found.
159        */
160       public List<T> find() {
161     
162         LinkedList<T> Q = new LinkedList<T>();
163         HashMap<Object, T> history = HashMapFactory.make();
164         while (roots.hasNext()) {
165           T next = roots.next();
166           Q.addLast(next);
167           history.put(next, null);
168         }
169         while (!Q.isEmpty()) {
170           T N = Q.removeFirst();
171           if (DEBUG) {
172             System.err.println(("visit " + N));
173           }
174           if (filter.test(N)) {
175             return makePath(N, history);
176           }
177           Iterator<? extends T> children = getConnected(N);
178           while (children.hasNext()) {
179             T c = children.next();
180             if (!history.containsKey(c)) {
181               Q.addLast(c);
182               history.put(c, N);
183             }
184           }
185         }
186     
187         return null;
188       }
189     
190       /**
191        * @return a List which represents a path in the breadth-first search to Q[i]. Q holds the nodes visited during the BFS, in order.
192        */
193       private List<T> makePath(T node, Map<Object, T> history) {
194         ArrayList<T> result = new ArrayList<T>();
195         T n = node;
196         result.add(n);
197         while (true) {
198           T parent = history.get(n);
199           if (parent == null)
200             return result;
201           else {
202             result.add(parent);
203             n = parent;
204           }
205         }
206       }
207     
208       /**
209        * get the out edges of a given node
210        * 
211        * @param n the node of which to get the out edges
212        * @return the out edges
213        * 
214        */
215       protected Iterator<? extends T> getConnected(T n) {
216         return G.getSuccNodes(n);
217       }
218     }
