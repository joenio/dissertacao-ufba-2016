43     /**
44      * An {@link SSAContextInterpreter} specialized to interpret Class.newInstance in a {@link JavaTypeContext} which represents the
45      * point-type of the class object created by the call.
46      */
47     public class ClassNewInstanceContextInterpreter extends AbstractReflectionInterpreter {
48     
49       public final static Atom newInstanceAtom = Atom.findOrCreateUnicodeAtom("newInstance");
50     
51       private final static Descriptor classNewInstanceDescriptor = Descriptor.findOrCreateUTF8("()Ljava/lang/Object;");
52     
53       public final static MethodReference CLASS_NEW_INSTANCE_REF = MethodReference.findOrCreate(TypeReference.JavaLangClass,
54           newInstanceAtom, classNewInstanceDescriptor);
55     
56       private final static Atom defCtorAtom = Atom.findOrCreateUnicodeAtom("<init>");
57     
58       private final static Descriptor defCtorDescriptor = Descriptor.findOrCreateUTF8("()V");
59     
60       private final static Selector defCtorSelector = new Selector(defCtorAtom, defCtorDescriptor);
61     
62       private final IClassHierarchy cha;
63     
64     /** BEGIN Custom change: caching */
65       private final Map<String, IR> cache = HashMapFactory.make();
66       
67     /** END Custom change: caching */
68       public ClassNewInstanceContextInterpreter(IClassHierarchy cha) {
69         this.cha = cha;
70       }
71     
72       @Override
73       public IR getIR(CGNode node) {
74         if (node == null) {
75           throw new IllegalArgumentException("node is null");
76         }
77         assert understands(node);
78         if (DEBUG) {
79           System.err.println("generating IR for " + node);
80         }
81     /** BEGIN Custom change: caching */
82         
83         final JavaTypeContext context = (JavaTypeContext) node.getContext();
84         final IMethod method = node.getMethod();
85         final String hashKey = method.toString() + "@" + context.toString();
86         
87         IR result = cache.get(hashKey);
88         
89         if (result == null) {
90           result = makeIR(method, context);
91           cache.put(hashKey, result);
92         }
93         
94     /** END Custom change: caching */
95         return result;
96       }
97     
98       @Override
99       public int getNumberOfStatements(CGNode node) {
100         assert understands(node);
101         return getIR(node).getInstructions().length;
102       }
103     
104       @Override
105       public boolean understands(CGNode node) {
106         if (node == null) {
107           throw new IllegalArgumentException("node is null");
108         }
109         if (!(node.getContext() instanceof JavaTypeContext)) {
110           return false;
111         }
112         return node.getMethod().getReference().equals(CLASS_NEW_INSTANCE_REF);
113       }
114     
115       @Override
116       public Iterator<NewSiteReference> iterateNewSites(CGNode node) {
117         if (node == null) {
118           throw new IllegalArgumentException("node is null");
119         }
120         assert understands(node);
121         JavaTypeContext context = (JavaTypeContext) node.getContext();
122         TypeReference tr = context.getType().getTypeReference();
123         if (tr != null) {
124           return new NonNullSingletonIterator<NewSiteReference>(NewSiteReference.make(0, tr));
125         }
126         return EmptyIterator.instance();
127       }
128     
129       @Override
130       public Iterator<CallSiteReference> iterateCallSites(CGNode node) {
131         assert understands(node);
132         return EmptyIterator.instance();
133       }
134     
135       private IR makeIR(IMethod method, JavaTypeContext context) {
136         SSAInstructionFactory insts = context.getType().getType().getClassLoader().getInstructionFactory();
137         TypeReference tr = context.getType().getTypeReference();
138         if (tr != null) {
139           SpecializedMethod m = new SpecializedMethod(method, method.getDeclaringClass(), method.isStatic(), false);
140           IClass klass = cha.lookupClass(tr);
141           IMethod publicDefaultCtor = getPublicDefaultCtor(klass);
142           if (publicDefaultCtor != null) {
143             m.addStatementsForConcreteSimpleType(tr);
144           } else if (klass.getMethod(defCtorSelector) == null) {
145             TypeReference instantiationExceptionRef = TypeReference.findOrCreateClass(ClassLoaderReference.Primordial, "java/lang",
146                 "InstantiationException");
147             int xobj = method.getNumberOfParameters() + 1;
148             SSAInstruction newStatement = insts.NewInstruction(m.allInstructions.size(), xobj, NewSiteReference.make(2, instantiationExceptionRef));
149             m.addInstruction(tr, newStatement, true);
150             SSAInstruction throwStatement = insts.ThrowInstruction(m.allInstructions.size(), xobj);
151             m.addInstruction(tr, throwStatement, false);
152           } else {
153             TypeReference illegalAccessExceptionRef = TypeReference.findOrCreateClass(ClassLoaderReference.Primordial, "java/lang",
154                 "IllegalAccessException");
155             int xobj = method.getNumberOfParameters() + 1;
156             SSAInstruction newStatement = insts.NewInstruction(m.allInstructions.size(), xobj, NewSiteReference.make(2, illegalAccessExceptionRef));
157             m.addInstruction(tr, newStatement, true);
158             SSAInstruction throwStatement = insts.ThrowInstruction(m.allInstructions.size(), xobj);
159             m.addInstruction(tr, throwStatement, false);
160           }
161     
162           SSAInstruction[] instrs = new SSAInstruction[m.allInstructions.size()];
163           m.allInstructions.<SSAInstruction> toArray(instrs);
164           return new SyntheticIR(method, context, new InducedCFG(instrs, method, context), instrs, SSAOptions.defaultOptions(), null);
165         }
166     
167         return null;
168       }
169     
170       private IMethod getPublicDefaultCtor(IClass klass) {
171         IMethod ctorMethod = klass.getMethod(defCtorSelector);
172         if (ctorMethod != null && ctorMethod.isPublic() && ctorMethod.getDeclaringClass() == klass) {
173           return ctorMethod;
174         }
175         return null;
176       }

196       }

201       }
