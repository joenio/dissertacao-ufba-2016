11     package com.ibm.wala.util.graph.traverse;
12     
13     import java.util.Iterator;
14     import java.util.NoSuchElementException;
15     import java.util.Stack;
16     
17     import com.ibm.wala.util.collections.EmptyIterator;
18     import com.ibm.wala.util.debug.UnimplementedError;
19     import com.ibm.wala.util.graph.Graph;
20     
21     /**
22      * This class implements depth-first search over a {@link Graph}, return an enumeration of the nodes of the graph in order of increasing
23      * finishing time. This class follows the outNodes of the graph nodes to define the graph, but this behavior can be changed by
24      * overriding the getConnected method.
25      */
26     public abstract class DFSFinishTimeIterator<T> extends Stack<T> implements Iterator<T> {
27     
28       /**
29        * the current next element in finishing time order
30        */
31       private T theNextElement;
32     
33       /**
34        * an enumeration of all nodes to search from
35        */
36       private Iterator<? extends T> roots;
37     
38       /**
39        * The governing graph.
40        */
41       private Graph<T> G;
42     
43       /**
44        * Subclasses must call this in the constructor!
45        * 
46        * @param G
47        * @param nodes
48        */
49       protected void init(Graph<T> G, Iterator<? extends T> nodes) {
50         this.G = G;
51         roots = nodes;
52         if (roots.hasNext())
53           theNextElement = roots.next();
54       }
55     
56       /**
57        * Return whether there are any more nodes left to enumerate.
58        * 
59        * @return true if there nodes left to enumerate.
60        */
61       @Override
62       public boolean hasNext() {
63         return (!empty() || (theNextElement != null && getPendingChildren(theNextElement) == null));
64       }
65     
66       abstract Iterator<T> getPendingChildren(T n);
67     
68       abstract void setPendingChildren(T v, Iterator<T> iterator);
69     
70       /**
71        * Find the next graph node in finishing time order.
72        * 
73        * @return the next graph node in finishing time order.
74        */
75       @Override
76       @SuppressWarnings("unchecked")
77       public T next() throws NoSuchElementException {
78         if (!hasNext()) {
79           throw new NoSuchElementException();
80         }
81         if (empty()) {
82           T v = theNextElement;
83           setPendingChildren(v, getConnected(v));
84           push(v);
85         }
86         recurse: while (!empty()) {
87           T v = peek();
88           Iterator<? extends T> pc = getPendingChildren(v);
89           for (Iterator<? extends T> e = pc; e.hasNext();) {
90             T n = e.next();
91             assert n != null : "null node in pc";
92             Iterator<T> nChildren = getPendingChildren(n);
93             if (nChildren == null) {
94               // found a new child: recurse to it.
95               setPendingChildren(n, getConnected(n));
96               push(n);
97               continue recurse;
98             }
99           }
100           // the following saves space by allowing the original iterator to be GCed
101           setPendingChildren(v, (Iterator<T>) EmptyIterator.instance());
102     
103           // no more children to visit: finished this vertex
104           while (getPendingChildren(theNextElement) != null && roots.hasNext()) {
105             theNextElement = roots.next();
106           }
107     
108           return pop();
109         }
110         return null;
111       }
112     
113       /**
114        * get the out edges of a given node
115        * 
116        * @param n the node of which to get the out edges
117        * @return the out edges
118        * 
119        */
120       protected Iterator<T> getConnected(T n) {
121         return G.getSuccNodes(n);
122       }
123     
124       /**
125        * @see java.util.Iterator#remove()
126        */
127       @Override
128       public void remove() throws UnimplementedError {
129         throw new UnimplementedError();
130       }
131     
132     }
