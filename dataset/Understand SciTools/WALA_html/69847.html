109       /**
110        * @return true if the instruction can "fall through" to the following instruction
111        */
112       public boolean isFallThrough();
113     
114       /**
115        * @return an array containing the labels this instruction can branch to (not including the following instruction if this
116        *         instruction 'falls through')
117        */
118       public int[] getBranchTargets();
119     
120       /**
121        * @return an Instruction equivalent to this one but with any branch labels updated by looking them up in the targetMap array
122        */
123       public IInstruction redirectTargets(int[] targetMap);
124     
125       /**
126        * @return the number of values this instruction pops off the working stack
127        */
128       public int getPoppedCount();
129     
130       /**
131        * Computes the type of data pushed onto the stack, or null if none is pushed.
132        * 
133        * @param poppedTypesToCheck the types of the data popped off the stack by this instruction; if poppedTypes is null, then we don't
134        *          know the incoming stack types and the result of this method may be less accurate
135        */
136       public String getPushedType(String[] poppedTypesToCheck);
137     
138       /**
139        * @return the JVM word size of the value this instruction pushes onto the stack, or 0 if this instruction doesn't push anything

142       public byte getPushedWordSize();
143     
144       /**
145        * Apply a Visitor to this instruction. We invoke the appropriate Visitor method according to the type of this instruction.

147       public void visit(IInstruction.Visitor v);
