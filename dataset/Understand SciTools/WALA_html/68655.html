23     /**
24      * 
25      * 0-1-CFA Call graph builder which analyzes calls to "container methods" in a context which is defined by the receiver
26      * instance.
27      * 
28      * @author sfink
29      */
30     public class AstJavaZeroOneContainerCFABuilder extends AstJavaCFABuilder {
31     
32       /**
33        * @param cha governing class hierarchy
34        * @param options call graph construction options
35        * @param appContextSelector application-specific logic to choose contexts
36        * @param appContextInterpreter application-specific logic to interpret a method in context
37        */
38       public AstJavaZeroOneContainerCFABuilder(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache,
39           ContextSelector appContextSelector, SSAContextInterpreter appContextInterpreter) {
40         super(cha, options, cache);
41     
42         ContextSelector def = new DefaultContextSelector(options, cha);
43         ContextSelector contextSelector = appContextSelector == null ? def : new DelegatingContextSelector(appContextSelector, def);
44     
45         SSAContextInterpreter contextInterpreter = makeDefaultContextInterpreters(appContextInterpreter, options, cha);
46         setContextInterpreter(contextInterpreter);
47     
48         ZeroXInstanceKeys zik = makeInstanceKeys(cha, options, contextInterpreter);
49         setInstanceKeys(new JavaScopeMappingInstanceKeys(cha, this, zik));
50     
51         ContextSelector CCS = makeContainerContextSelector(cha, zik);
52         DelegatingContextSelector DCS = new DelegatingContextSelector(CCS, contextSelector);
53         setContextSelector(DCS);
54       }
55     
56       protected ZeroXInstanceKeys makeInstanceKeys(IClassHierarchy cha, AnalysisOptions options,
57           SSAContextInterpreter contextInterpreter) {
58         ZeroXInstanceKeys zik = new ZeroXInstanceKeys(options, cha, contextInterpreter, ZeroXInstanceKeys.ALLOCATIONS
59             | ZeroXInstanceKeys.SMUSH_PRIMITIVE_HOLDERS | ZeroXInstanceKeys.SMUSH_STRINGS | ZeroXInstanceKeys.SMUSH_MANY
60             | ZeroXInstanceKeys.SMUSH_THROWABLES);
61         return zik;
62       }
63     
64       /**
65        * @param cha
66        * @param keys
67        * @return an object which creates contexts for call graph nodes based on the container disambiguation policy
68        */
69       protected ContextSelector makeContainerContextSelector(IClassHierarchy cha, ZeroXInstanceKeys keys) {
70         return new ContainerContextSelector(cha, keys);
71       }
72     
73     }
