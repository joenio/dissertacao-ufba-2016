48     package org.scandroid.spec;
49     
50     import java.util.Collection;
51     import java.util.LinkedList;
52     import com.ibm.wala.classLoader.IField;
53     import com.ibm.wala.classLoader.IClass;
54     import com.ibm.wala.classLoader.IClassLoader;
55     import com.ibm.wala.types.TypeName;
56     import com.ibm.wala.util.strings.Atom;
57     
58     public class FieldNamePattern {
59         final String className;   // null = match any class
60     
61         final String memberName;  // null = match any method
62                                   // * used to match arbitrary substrings
63         public FieldNamePattern(String c, String m) {
64             className = c;
65             memberName = m;
66         }
67     
68         Collection<IField> lookupFields(IClassLoader cl) {
69             Collection<IField> matching = new LinkedList<IField>();
70             IClass c = cl.lookupClass(TypeName.findOrCreate(className));
71             if(c == null) return matching;
72             Atom atom = Atom.findOrCreateUnicodeAtom(memberName);
73             Collection<IField> allFields = c.getAllFields();
74             for(IField f: allFields) {
75                 if(f.getName().equals(atom)) {
76                     matching.add(f);
77                 }
78             }
79             return matching;
80         }
81     }
