18     /**
19      * Provides minimal abstraction layer to a stream of bytecodes from the code attribute of a method.
20      */
21     public class BytecodeStream implements BytecodeConstants {
22       private final IMethod method;
23     
24       private final IClass declaringClass;

26       private final int bcLength;

36       /**
37        * @param m the method containing the bytecodes
38        * @param bc the array of bytecodes
39        * @throws IllegalArgumentException if bc is null
40        * @throws IllegalArgumentException if m is null
41        */
42       public BytecodeStream(IMethod m, byte[] bc) {
43         if (m == null) {
44           throw new IllegalArgumentException("m is null");
45         }
46         if (bc == null) {
47           throw new IllegalArgumentException("bc is null");
48         }
49         method = m;
50         declaringClass = m.getDeclaringClass();
51         bcodes = bc;
52         bcLength = bc.length;
53         bcIndex = 0;
54       }

134        * @see #nextInstruction()

139       }

142        * Are we currently processing a wide instruction?
143        * 
144        * @return true if current instruction is wide
145        * @see #nextInstruction()
146        * @see #getOpcode()
147        */
148       public final boolean isWide() {

150       }
151     
152       /**
153        * Skips the current instruction
154        * 
155        * @see #skipInstruction(int,boolean)
156        */
157       public final void skipInstruction() {
158         int len = JBC_length[opcode] - 1;
159         if (wide)
160           len += len;
161         if (len >= 0) {
162           bcIndex += len;
163         } else {
164           skipSpecialInstruction(opcode);
165         }
166       }
167     
168       /**
169        * Skips the current instruction (without using the opcode field) A slightly optimized version of skipInstruction()
170        * 
171        * @param opc current opcode
172        * @param w whether current instruction follows wide
173        * @see #skipInstruction()
174        */
175       public final void skipInstruction(int opc, boolean w) {
176         int len = JBC_length[opc] - 1;
177         if (w)
178           len += len;
179         if (len >= 0)
180           bcIndex += len;
181         else
182           skipSpecialInstruction(opc);
183       }

316        * 
317        * @param num the number of the offset to retrieve
318        * @return switch offset
319        */
320       public final int getTableSwitchOffset(int num) {
321         return getSignedInt(bcIndex + (num << 2));
322       }
323     
324       /**
325        * Returns the offset for a given value of the tableswitch (as a signed int) or 0 if the value is out of range. Used for
326        * tableswitch The "cursor" has to be positioned at the start of the offset table NOTE: Will NOT advance cursor
327        * 
328        * @param value the value to retrieve offset for
329        * @param low the lowest value of the tableswitch
330        * @param high the highest value of the tableswitch
331        * @return switch offset
332        */
333       public final int computeTableSwitchOffset(int value, int low, int high) {
334         if (value < low || value > high)
335           return 0;
336         return getSignedInt(bcIndex + ((value - low) << 2));
337       }

360        * Returns the numbered offset of the lookupswitch (as a signed int) Used for lookupswitch The "cursor" has to be positioned at
361        * the start of the pair table NOTE: Will NOT advance cursor

364        * @return switch offset
365        * @see #getLookupSwitchValue(int)
366        */
367       public final int getLookupSwitchOffset(int num) {
368         return getSignedInt(bcIndex + (num << 3) + 4);
369       }

371       /**
372        * Returns the numbered value of the lookupswitch (as a signed int) Used for lookupswitch The "cursor" has to be positioned at the
373        * start of the pair table NOTE: Will NOT advance cursor
374        * 
375        * @param num the number of the value to retrieve
376        * @return switch value
377        * @see #getLookupSwitchOffset(int)
378        */
379       public final int getLookupSwitchValue(int num) {
380         return getSignedInt(bcIndex + (num << 3));
381       }
382     
383       /**
384        * Returns the offset for a given value of the lookupswitch (as a signed int) or 0 if the value is not in the table. Used for
385        * lookupswitch The "cursor" has to be positioned at the start of the offset table NOTE: Will NOT advance cursor WARNING: Uses
386        * LINEAR search. Whoever has time on their hands can re-implement this as a binary search.
387        * 
388        * @param value the value to retrieve offset for
389        * @param num the number of match-offset pairs in the lookupswitch
390        * @return switch offset
391        */
392       public final int computeLookupSwitchOffset(int value, int num) {
393         for (int i = 0; i < num; i++)
394           if (getSignedInt(bcIndex + (i << 3)) == value)
395             return getSignedInt(bcIndex + (i << 3) + 4);
396         return 0;
397       }

457       // Skip a tableswitch or a lookupswitch instruction
458       private void skipSpecialInstruction(int opcode) {
459         switch (opcode) {
460         case JBC_tableswitch: {
461           alignSwitch();
462           getDefaultSwitchOffset();
463           int l = getLowSwitchValue();
464           int h = getHighSwitchValue();
465           skipTableSwitchOffsets(h - l + 1); // jump offsets
466         }
467           break;
468         case JBC_lookupswitch: {
469           alignSwitch();
470           getDefaultSwitchOffset();
471           int n = getSwitchLength();
472           skipLookupSwitchPairs(n); // match-offset pairs
473         }
474           break;
475         case JBC_wide: {
476           int oc = getWideOpcode();
477           int len = JBC_length[oc] - 1;
478           bcIndex += len + len;
479         }
480           break;
481         default:
482           throw new NullPointerException();
483         }
484       }

486       // // READ BYTECODES
487       private final byte readSignedByte() {
488         return bcodes[bcIndex++];
489       }
490     
491       private final int readUnsignedByte() {
492         return (bcodes[bcIndex++] & 0xFF);
493       }
494     
495       private final int getUnsignedByte(int index) {
496         return (bcodes[index] & 0xFF);
497       }
498     
499       private final int readSignedShort() {
500         int i = bcodes[bcIndex++] << 8;
501         i |= (bcodes[bcIndex++] & 0xFF);
502         return i;
503       }
504     
505       private final int readUnsignedShort() {
506         int i = (bcodes[bcIndex++] & 0xFF) << 8;
507         i |= (bcodes[bcIndex++] & 0xFF);
508         return i;
509       }
510     
511       private final int readSignedInt() {
512         int i = bcodes[bcIndex++] << 24;
513         i |= (bcodes[bcIndex++] & 0xFF) << 16;
514         i |= (bcodes[bcIndex++] & 0xFF) << 8;
515         i |= (bcodes[bcIndex++] & 0xFF);
516         return i;
517       }
518     
519       private final int getSignedInt(int index) {
520         int i = bcodes[index++] << 24;
521         i |= (bcodes[index++] & 0xFF) << 16;
522         i |= (bcodes[index++] & 0xFF) << 8;
523         i |= (bcodes[index] & 0xFF);
524         return i;
525       }
