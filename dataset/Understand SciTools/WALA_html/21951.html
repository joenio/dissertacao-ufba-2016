39     /**
40      * An example of using {@link IFDSExplorer}. We visualize the result of running {@link ContextSensitiveReachingDefs} on a simple
41      * test example.
42      */
43     public class IFDSExplorerExample {
44     
45       /**
46        * Usage: IFDSExplorerExample -dotExe <path_to_dot_exe> -viewerExe <path_to_viewer_exe>
47        * 
48        * @param args
49        * @throws IOException
50        * @throws CallGraphBuilderCancelException
51        * @throws IllegalArgumentException
52        * @throws WalaException
53        */
54       public static void main(String[] args) throws IOException, IllegalArgumentException, CallGraphBuilderCancelException,
55           WalaException {
56         Properties p = CommandLine.parse(args);
57         AnalysisScope scope = AnalysisScopeReader.readJavaScope(TestConstants.WALA_TESTDATA, null, IFDSExplorer.class.getClassLoader());
58         IClassHierarchy cha = ClassHierarchy.make(scope);
59         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha,
60             "Ldataflow/StaticDataflow");
61         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
62     
63         CallGraphBuilder builder = Util.makeZeroOneCFABuilder(options, new AnalysisCache(), cha, scope);
64         CallGraph cg = builder.makeCallGraph(options, null);
65         AnalysisCache cache = new AnalysisCache();
66         ContextSensitiveReachingDefs reachingDefs = new ContextSensitiveReachingDefs(cg, cache);
67         TabulationResult<BasicBlockInContext<IExplodedBasicBlock>, CGNode, Pair<CGNode, Integer>> result = reachingDefs.analyze();
68         IFDSExplorer.setDotExe(p.getProperty("dotExe"));
69         IFDSExplorer.setGvExe(p.getProperty("viewerExe"));
70         IFDSExplorer.viewIFDS(result);
71       }
