23     /**
24      * For some reason (either a bug in our code that defeats soft references, or a
25      * bad policy in the GC), leaving soft reference caches to clear themselves out
26      * doesn't work. Help it out.
27      * 
28      * It's unfortunate that this class exists.
29      */
30     public class ReferenceCleanser {
31       
32       private final static float OCCUPANCY_TRIGGER = 0.5f;
33     
34       private static WeakReference<IClassHierarchy> cha;
35     
36       private static WeakReference<AnalysisCache> cache;
37     
38       public static void registerClassHierarchy(IClassHierarchy cha) {
39         ReferenceCleanser.cha = new WeakReference<IClassHierarchy>(cha);
40       }
41     
42       private static IClassHierarchy getClassHierarchy() {
43         IClassHierarchy result = null;
44         if (cha != null) {
45           result = cha.get();
46         }
47         return result;
48       }
49     
50       public static void registerCache(AnalysisCache cache) {
51         ReferenceCleanser.cache = new WeakReference<AnalysisCache>(cache);
52       }
53     
54       private static AnalysisCache getAnalysisCache() {
55         AnalysisCache result = null;
56         if (cache != null) {
57           result = cache.get();
58         }
59         return result;
60       }
61     
62       /**
63        * A debugging aid. TODO: move this elsewhere
64        */
65       public static void clearSoftCaches() {
66         float occupancy = 1f - ((float)Runtime.getRuntime().freeMemory() / (float)Runtime.getRuntime().totalMemory());
67         if (occupancy < OCCUPANCY_TRIGGER) {
68           return;
69         }
70         AnalysisCache cache = getAnalysisCache();
71         if (cache != null) {
72           cache.getSSACache().wipe();
73         }
74         IClassHierarchy cha = getClassHierarchy();
75         if (cha != null) {
76           for (IClass klass : cha) {
77             if (klass instanceof ShrikeClass) {
78               ShrikeClass c = (ShrikeClass) klass;
79               c.clearSoftCaches();
80             } else {
81               for (Iterator it2 = klass.getDeclaredMethods().iterator(); it2.hasNext(); ) {
82                 IMethod m = (IMethod)it2.next();
83                 if (m instanceof ShrikeCTMethod) {
84                   ((ShrikeCTMethod)m).clearCaches();
85                 }
86               }
87             }
88           }
89         }
90       }
91     
92     }
