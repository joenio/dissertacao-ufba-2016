39       public static <T> boolean isAcyclic(NumberedGraph<T> G, T root) {
40         IBinaryNaturalRelation r = computeBackEdges(G, root);
41         Iterator<IntPair> it = r.iterator();
42         return !it.hasNext();
43       }

45       public static final int THRESHOLD_FOR_NONRECURSIVE_DFS = 1000;
46       
47       /**
48        * Compute a relation R s.t. (i,j) \in R iff (i,j) is a backedge according to a DFS of a numbered graph starting from some root.
49        * 
50        * Not efficient. Recursive and uses hash sets.
51        */
52       public static <T> IBinaryNaturalRelation computeBackEdges(NumberedGraph<T> G, T root) {
53         if (G == null) {
54           throw new IllegalArgumentException("G is null");
55         }
56         
57         final BasicNaturalRelation result = new BasicNaturalRelation();
58     
59         // for large methods (e.g. obfuscated library code as found in android libraries 'com.google.ads.ad.a([B[B)V')
60         // the recursive dfs can lead to a stack overflow error.
61         // for smaller methods the recursive solution seems to be faster, so we keep it.
62         if (G.getNumberOfNodes() <= THRESHOLD_FOR_NONRECURSIVE_DFS) {
63           final Set<T> visited = HashSetFactory.make();
64           final Set<T> onstack = HashSetFactory.make();
65           dfs(result, root, G, visited, onstack);
66         } else { 
67           dfsNonRecursive(result, root, G);
68         }
69         
70         return result;
71       }
72       
73       private static <T> void dfs(BasicNaturalRelation result, T root, NumberedGraph<T> G, Set<T> visited, Set<T> onstack) {
74         visited.add(root);
75         onstack.add(root);
76         for (Iterator<? extends T> it = G.getSuccNodes(root); it.hasNext();) {
77           T dstNode = it.next();
78           if (onstack.contains(dstNode)) {
79             int src = G.getNumber(root);
80             int dst = G.getNumber(dstNode);
81             result.add(src, dst);
82           }
83           if (!visited.contains(dstNode)) {
84             dfs(result, dstNode, G, visited, onstack);
85           }
86         }
87         onstack.remove(root);
88       }

90       private static <T> void dfsNonRecursive(final BasicNaturalRelation result, final T root, final NumberedGraph<T> G) {
91         final Stack<T> stack = new Stack<T>();
92         final Set<T> stackSet = new HashSet<T>();
93         final Stack<Iterator<? extends T>> stackIt = new Stack<Iterator<? extends T>>();
94         final Set<T> finished = new HashSet<T>();
95         stack.push(root);
96         stackSet.add(root);
97         stackIt.push(G.getSuccNodes(root));
98         
99         while (!stack.isEmpty()) {
100           final T current = stack.pop();
101           stackSet.remove(current);
102           final Iterator<? extends T> currentIt = stackIt.pop();
103           if (finished.contains(current)) { continue; }
104           
105           boolean isFinished = true;
106           while (isFinished && currentIt.hasNext() ) {
107             final T succ = currentIt.next();
108             if (!finished.contains(succ)) {
109               if (succ == current || stackSet.contains(succ)) {
110                 // found a backedge
111                 final int src = G.getNumber(current);
112                 final int dst = G.getNumber(succ);
113                 result.add(src, dst);
114               } else {
115                 stack.push(current);
116                 stackSet.add(current);
117                 stackIt.push(currentIt);
118                 stack.push(succ);
119                 stackSet.add(succ);
120                 stackIt.push(G.getSuccNodes(succ));
121                 isFinished = false;
122               }
123             }
124           }
125           
126           if (isFinished) {
127             finished.add(current);
128           }
129         }
130       }

132       public static <T> boolean hasIncomingBackEdges(Path p, NumberedGraph<T> G, T root) {
133         /*
134          * TODO: pull out computeBackEdges, and pass in the backedge relation as a parameter to this call
135          */
136         IBinaryNaturalRelation backedges = computeBackEdges(G, root);
137         for (int index = 0; index < p.size(); index++) {
138           int gn = p.get(index);
139           Iterator<? extends T> predIter = G.getPredNodes(G.getNode(gn));
140           while (predIter.hasNext()) {
141             if (backedges.contains(G.getNumber(predIter.next()), gn))
142               return true;
143           }
144         }
145         return false;
146       }

148       /**
149        * Compute a set of acyclic paths through a graph G from a node src to a node sink.
150        * 
151        * This is not terribly efficient.
152        * 
153        * @param max the max number of paths to return.
154        */
155       public static <T> Collection<Path> computeAcyclicPaths(NumberedGraph<T> G, T root, T src, T sink, int max) {
156         Collection<Path> result = HashSetFactory.make();
157         EdgeFilteredNumberedGraph<T> acyclic = new EdgeFilteredNumberedGraph<T>(G, computeBackEdges(G, root));
158     
159         Collection<Path> worklist = HashSetFactory.make();
160         Path sinkPath = Path.make(G.getNumber(sink));
161         worklist.add(sinkPath);
162         while (!worklist.isEmpty() && result.size() <= max) {
163           Path p = worklist.iterator().next();
164           worklist.remove(p);
165           int first = p.get(0);
166           if (first == G.getNumber(src)) {
167             result.add(p);
168           } else {
169             for (IntIterator it = acyclic.getPredNodeNumbers(acyclic.getNode(first)).intIterator(); it.hasNext();) {
170               worklist.add(Path.prepend(it.next(), p));
171             }
172           }
173         }
174     
175         return result;
176       }
