11     package com.ibm.wala.shrikeCT;
12     
13     /**
14      * This class builds serializable SourceFile attributes.
15      * 
16      * After constructing a SourceFileWriter, you must call setSourceFileCPIndex.
17      */
18     public final class SourceFileWriter extends ClassWriter.Element {
19       final private int attrID;
20     
21       private int index = -1;
22     
23       /**
24        * Build an empty writer.
25        * 
26        * @throws IllegalArgumentException if w is null
27        */
28       public SourceFileWriter(ClassWriter w) {
29         if (w == null) {
30           throw new IllegalArgumentException("w is null");
31         }
32         attrID = w.addCPUtf8("SourceFile");
33       }
34     
35       private void verify() {
36         if (index < 0) {
37           throw new IllegalArgumentException("The value's constant pool index is not set");
38         }
39       }
40     
41       @Override
42       public int getSize() throws IllegalArgumentException {
43         verify();
44         return 8;
45       }
46     
47       @Override
48       public int copyInto(byte[] buf, int offset) throws IllegalArgumentException {
49         verify();
50         ClassWriter.setUShort(buf, offset, attrID);
51         ClassWriter.setInt(buf, offset + 2, 2);
52         ClassWriter.setUShort(buf, offset + 6, index);
53         return offset + 8;
54       }
55     
56       /**
57        * Set the index of the constant pool item holding the source file name.
58        */
59       public void setSourceFileCPIndex(int index) throws IllegalArgumentException {
60         if (index < 1 || index > 0xFFFF) {
61           throw new IllegalArgumentException("Invalid CP index: " + index);
62         }
63     
64         this.index = index;
65       }
66     }
