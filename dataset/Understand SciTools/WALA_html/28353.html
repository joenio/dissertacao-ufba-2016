26     /**
27      * A {@link ContextSelector} to intercept calls to Object.getClass()
28      */
29     public class GetClassContextSelector implements ContextSelector {
30     
31       public final static MethodReference GET_CLASS = MethodReference.findOrCreate(TypeReference.JavaLangObject, "getClass",
32           "()Ljava/lang/Class;");
33     
34       public GetClassContextSelector() {
35       }
36     
37       /*
38        * @see com.ibm.wala.ipa.callgraph.ContextSelector#getCalleeTarget(com.ibm.wala.ipa.callgraph.CGNode,
39        *      com.ibm.wala.classLoader.CallSiteReference, com.ibm.wala.classLoader.IMethod,
40        *      com.ibm.wala.ipa.callgraph.propagation.InstanceKey)
41        */
42       @Override
43       public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] receiver) {
44         if (callee.getReference().equals(GET_CLASS)) {
45           return new JavaTypeContext(new PointType(receiver[0].getConcreteType()));
46         }
47         return null;
48       }
49       
50       private static final IntSet thisParameter = IntSetUtil.make(new int[]{0});
51     
52       @Override
53       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
54         if (site.isDispatch() || site.getDeclaredTarget().getNumberOfParameters() > 0) {
55           return thisParameter;
56         } else {
57           return EmptyIntSet.instance;
58         }
59       }
60     
61     }
