11     package com.ibm.wala.cast.tree.impl;
12     
13     import com.ibm.wala.cast.tree.CAstNode;
14     
15     /**
16      * An implementation of CAst, i.e. a simple factory for creating capa
17      * ast nodes.  This class simply creates generic nodes with a kind
18      * field, and either an array of children.  Note that there is no easy
19      * way to mutate these trees; do not changes this (see CAstNode for
20      * the rationale for this rule).
21      *
22      * @author Julian Dolby (dolby@us.ibm.com)
23      *
24      */
25     public class CAstValueImpl extends CAstImpl {
26     
27       protected static class CAstNodeValueImpl extends CAstNodeImpl {
28           
29         protected CAstNodeValueImpl(int kind, CAstNode cs[]) {
30           super(kind, cs);
31         }
32     
33         @Override
34         public int hashCode() {
35           int value = 1237 * kind;
36           for(int i = 0; i < cs.length; i++) 
37     	value *= cs[i].hashCode();
38     
39           return value;
40         }
41     
42         @Override
43         public boolean equals(Object o) {
44           if (! (o instanceof CAstNode)) return false;
45           if (kind != ((CAstNode)o).getKind()) return false;
46           if (((CAstNode)o).getChildCount() != cs.length) return false;
47           for(int i = 0; i < cs.length; i++)
48     	if (! cs[i].equals(((CAstNode)o).getChild(i)))
49     	  return false;
50     
51           return true;
52         }
53       }
54       
55       @Override
56       public CAstNode makeNode(final int kind, final CAstNode[] cs) {
57         return new CAstNodeValueImpl(kind, cs);
58       }
59     
60       protected static class CAstValueValueImpl extends CAstValueImpl {
61     
62         protected CAstValueValueImpl(Object value) {
63           super(value);
64         }
65     
66         @Override
67         public int hashCode() { 
68           return value.hashCode(); 
69         }
70     
71         @Override
72         public boolean equals(Object o) {
73           if (o instanceof CAstNode) {
74     	return value==null? 
75     	    ((CAstNode)o).getValue()==null:
76     	    value.equals(((CAstNode)o).getValue());
77           } else {
78     	return false;
79           }
80         }
81       }
82     
83       @Override
84       public CAstNode makeConstant(final Object value) {
85         return new CAstValueValueImpl(value);
86       }
87     
88     }
