14     package com.ibm.wala.cast.java.ipa.callgraph;
15     
16     import java.util.Collection;
17     import java.util.Collections;
18     
19     import com.ibm.wala.classLoader.Language;
20     import com.ibm.wala.classLoader.Module;
21     import com.ibm.wala.classLoader.SourceDirectoryTreeModule;
22     import com.ibm.wala.ipa.callgraph.AnalysisScope;
23     import com.ibm.wala.types.ClassLoaderReference;
24     import com.ibm.wala.util.strings.Atom;
25     
26     public class JavaSourceAnalysisScope extends AnalysisScope {
27     
28       public final static ClassLoaderReference SOURCE = new ClassLoaderReference(Atom.findOrCreateAsciiAtom("Source"), Atom
29           .findOrCreateAsciiAtom("Java"), ClassLoaderReference.Application);
30      
31       public JavaSourceAnalysisScope() {
32         this(Collections.singleton(Language.JAVA));
33       }
34     
35       protected void initCoreForJavaSource() {
36         initCoreForJava();
37         loadersByName.put(SOURCE.getName(),SOURCE);
38     
39         setLoaderImpl(SOURCE, "com.ibm.wala.cast.java.translator.polyglot.PolyglotSourceLoaderImpl");   
40       }
41       
42       protected JavaSourceAnalysisScope(Collection<? extends Language> languages) {
43         super(languages);
44         initCoreForJavaSource();
45         initSynthetic(SOURCE);
46        }
47     
48       public ClassLoaderReference getSourceLoader() {
49         return SOURCE;
50       }
51       
52       @Override
53       public void addToScope(ClassLoaderReference loader, Module m) {
54         if (m instanceof SourceDirectoryTreeModule && loader.equals(ClassLoaderReference.Application)) {
55           super.addToScope(SOURCE, m);
56         } else {
57           super.addToScope(loader, m);
58         }
59       }
60     }
