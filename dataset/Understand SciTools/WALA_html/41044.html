38     public class JSCallGraph extends AstCallGraph {
39     
40       public JSCallGraph(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache) {
41         super(cha, options, cache);
42       }
43     
44       public final static MethodReference fakeRoot = MethodReference.findOrCreate(JavaScriptTypes.FakeRoot, FakeRootMethod.name,
45           FakeRootMethod.descr);
46     
47       public static class JSFakeRoot extends ScriptFakeRoot {
48     
49         public JSFakeRoot(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache) {
50           super(fakeRoot, cha.lookupClass(JavaScriptTypes.FakeRoot), cha, options, cache);
51         }
52     
53         @Override
54         public InducedCFG makeControlFlowGraph(SSAInstruction[] instructions) {
55           return new JSInducedCFG(instructions, this, Everywhere.EVERYWHERE);
56         }
57     
58         @Override
59         public SSANewInstruction addAllocation(TypeReference T) {
60           if (cha.isSubclassOf(cha.lookupClass(T), cha.lookupClass(JavaScriptTypes.Root))) {
61             int instance = nextLocal++;
62             NewSiteReference ref = NewSiteReference.make(statements.size(), T);
63             SSANewInstruction result = getDeclaringClass().getClassLoader().getInstructionFactory().NewInstruction(statements.size(), instance, ref);
64             statements.add(result);
65             return result;
66           } else {
67             return super.addAllocation(T);
68           }
69         }
70     
71         @Override
72         public SSAAbstractInvokeInstruction addDirectCall(int function, int[] params, CallSiteReference site) {
73           CallSiteReference newSite = new JSCallSiteReference(statements.size());
74     
75           JavaScriptInvoke s = new JavaScriptInvoke(statements.size(), function, nextLocal++, params, nextLocal++, newSite);
76           statements.add(s);
77     
78           return s;
79         }
80       }
81     
82       @Override
83       protected CGNode makeFakeWorldClinitNode() {
84         return null;
85       }
86     
87       @Override
88       protected CGNode makeFakeRootNode() throws com.ibm.wala.util.CancelException  {
89         return findOrCreateNode(new JSFakeRoot(cha, options, getAnalysisCache()), Everywhere.EVERYWHERE);
90       }
91     
92       @Override
93       public Set<CGNode> getNodes(MethodReference m) {
94         if (m.getName().equals(JavaScriptMethods.ctorAtom)) {
95           // TODO cache this?
96           Set<CGNode> result = HashSetFactory.make(1);
97           for (CGNode n : this) {
98             IMethod method = n.getMethod();
99             if (method.getName().equals(JavaScriptMethods.ctorAtom) && method.getDeclaringClass().getReference().equals(m.getDeclaringClass())) {
100               result.add(n);
101             }
102           }
103           return result;      
104         } else {
105           return super.getNodes(m);
106         }
107       }
108       
109     }
