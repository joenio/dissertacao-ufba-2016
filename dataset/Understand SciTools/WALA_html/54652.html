33     public class JavaScopeMappingInstanceKeys extends ScopeMappingInstanceKeys {
34     
35       public JavaScopeMappingInstanceKeys(IClassHierarchy cha, PropagationCallGraphBuilder builder, InstanceKeyFactory basic) {
36         super(builder, basic);
37     
38       }
39     
40       protected LexicalParent[] getParents(InstanceKey base) {
41         IClass cls = base.getConcreteType();
42         if (isPossiblyLexicalClass(cls)) {
43           Set<LexicalParent> result = HashSetFactory.make();
44     
45           for (Iterator MS = cls.getAllMethods().iterator(); MS.hasNext();) {
46             IMethod m = (IMethod) MS.next();
47             if ((m instanceof AstMethod) && !m.isStatic()) {
48               AstMethod M = (AstMethod) m;
49               LexicalParent[] parents = M.getParents();
50               for (int i = 0; i < parents.length; i++) {
51                 result.add(parents[i]);
52               }
53             }
54           }
55     
56           if (!result.isEmpty()) {
57             if (AstTranslator.DEBUG_LEXICAL)
58               System.err.println((base + " has parents: " + result));
59     
60             return result.toArray(new LexicalParent[result.size()]);
61           }
62     
63         }
64     
65         if (AstTranslator.DEBUG_LEXICAL)
66           System.err.println((base + " has no parents"));
67     
68         return new LexicalParent[0];
69       }
70     
71       protected boolean isPossiblyLexicalClass(IClass cls) {
72         return cls instanceof JavaClass;
73       }
74     
75       @Override
76       protected boolean needsScopeMappingKey(InstanceKey base) {
77         boolean result = getParents(base).length > 0;
78         if (AstTranslator.DEBUG_LEXICAL)
79           System.err.println(("does " + base + " need scope mapping? " + result));
80     
81         return result;
82       }
83       
84       @Override
85       protected Collection<CGNode> getConstructorCallers(ScopeMappingInstanceKey smik, Pair<String, String> name) {
86         // for Java, the creator node is exactly what we want
87         return Collections.singleton(smik.getCreator());
88       }
