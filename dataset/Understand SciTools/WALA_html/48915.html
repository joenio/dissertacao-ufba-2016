11     package com.ibm.wala.ipa.callgraph;

13     import com.ibm.wala.classLoader.CallSiteReference;
14     import com.ibm.wala.classLoader.IClass;
15     import com.ibm.wala.classLoader.IMethod;

17     /**
18      * This interface represents policies for selecting a method to call at a given invocation site. The most obvious such policy would
19      * be to look at the relevant class hierarchy and lookup the appropriate method based on the type reference and name and descriptor
20      * at the call site. However, other policies are possible for purposes such as providing an abstraction of unanalyzed libraries or
21      * specialized J2EE functionality.
22      * 
23      * Such policies are consulted by the different analysis mechanisms, both the flow-based and non-flow algorithms. The current
24      * mechanism is that the policy object are registered with the AnalysisOptions object, and all analyses that need to analyze
25      * invocations ask that object for the method selector to use.
26      * 
27      * In general, for specialized selectors, it is good practice to build selectors that handle the special case of interest, and
28      * otherwise delegate to a child selector. When registering with the AnalysisOptions object, make the child selector be whatever the
29      * options object had before.
30      */
31     public interface MethodTargetSelector {
32     
33       /**
34        * Given a calling node, a call site and (optionally) a dispatch type, return the target method to be called.
35        * 
36        * @param caller the GCNode in the call graph containing the call
37        * @param site the call site reference of the call site
38        * @param receiver the type of the target object or null
39        * @return the method to be called.
40        */
41       IMethod getCalleeTarget(CGNode caller, CallSiteReference site, IClass receiver);
42     }
