126       public static void runTestCase(String mainClass, String scopeFile, String benchName) throws IllegalArgumentException,
127           CancelException, IOException {
128         System.err.println("=====BENCHMARK " + benchName + "=====");
129         System.err.println("analyzing " + benchName);
130         DemandRefinementPointsTo dmp = null;
131         try {
132           dmp = makeDemandPointerAnalysis(scopeFile, mainClass, benchName);
133           findFailingCasts(dmp.getBaseCallGraph(), dmp);
134         } catch (ClassHierarchyException e) {
135           e.printStackTrace();
136         }
137         System.err.println("=*=*=*=*=*=*=*=*=*=*=*=*=*=*");
138         System.err.println("");
139         System.err.println("");
140       }

142       private static DemandRefinementPointsTo makeDemandPointerAnalysis(String scopeFile, String mainClass, String benchName)
143           throws ClassHierarchyException, IllegalArgumentException, CancelException, IOException {
144         AnalysisScope scope = CallGraphTestUtil.makeJ2SEAnalysisScope(scopeFile, getExclusions(benchName));
145         // build a type hierarchy
146         ClassHierarchy cha = ClassHierarchy.make(scope);
147     
148         // set up call graph construction options; mainly what should be considered
149         // entrypoints?
150         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha, mainClass);
151         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
152     
153         System.err.print("constructing call graph...");
154         final Pair<CallGraph, PointerAnalysis> cgAndPA = buildCallGraph(scope, cha, options);
155         CallGraph cg = cgAndPA.fst;    
156         System.err.println("done");
157         System.err.println(CallGraphStats.getStats(cg));
158         MemoryAccessMap fam = new SimpleMemoryAccessMap(cg, cgAndPA.snd.getHeapModel(), false);
159         DemandRefinementPointsTo fullDemandPointsTo = DemandRefinementPointsTo.makeWithDefaultFlowGraph(cg, heapModel, fam, cha, options, makeStateMachineFactory());
160         fullDemandPointsTo.setRefinementPolicyFactory(chooseRefinePolicyFactory(cha));
161         return fullDemandPointsTo;
162       }

168       // if true, construct up-front call graph cheaply (0-CFA)
169       private static final boolean CHEAP_CG = true;

173       /**
174        * builds a call graph, and sets the corresponding heap model for analysis
175        * 
176        * @param scope
177        * @param cha
178        * @param options
179        * @return
180        * @throws CancelException
181        * @throws IllegalArgumentException
182        */
183       private static Pair<CallGraph, PointerAnalysis> buildCallGraph(AnalysisScope scope, ClassHierarchy cha, AnalysisOptions options)
184           throws IllegalArgumentException, CancelException {
185         CallGraph retCG = null;
186         PointerAnalysis retPA = null;
187         final AnalysisCache cache = new AnalysisCache();
188         CallGraphBuilder builder;
189         if (CHEAP_CG) {
190           builder = Util.makeZeroCFABuilder(options, cache, cha, scope);
191           // we want vanilla 0-1 CFA, which has one abstract loc per allocation
192           heapModel = Util.makeVanillaZeroOneCFABuilder(options, cache, cha, scope);
193         } else {
194           builder = Util.makeZeroOneContainerCFABuilder(options, cache, cha, scope);
195           heapModel = (HeapModel) builder;
196         }
197         ProgressMaster master = ProgressMaster.make(new NullProgressMonitor(), 360000, false);
198         master.beginTask("runSolver", 1);
199         try {
200           retCG = builder.makeCallGraph(options, master);
201           retPA = builder.getPointerAnalysis();
202         } catch (CallGraphBuilderCancelException e) {
203           System.err.println("TIMED OUT!!");
204           retCG = e.getPartialCallGraph();
205           retPA = e.getPartialPointerAnalysis();
206         }
207         return Pair.make(retCG, retPA);
208       }

223       private static List<Pair<CGNode, SSACheckCastInstruction>> findFailingCasts(CallGraph cg, DemandRefinementPointsTo dmp) {
224         final IClassHierarchy cha = dmp.getClassHierarchy();
225         List<Pair<CGNode, SSACheckCastInstruction>> failing = new ArrayList<Pair<CGNode, SSACheckCastInstruction>>();
226     
227         int numSafe = 0, numMightFail = 0;
228         outer: for (Iterator<? extends CGNode> nodeIter = cg.iterator(); nodeIter.hasNext();) {
229           CGNode node = nodeIter.next();
230           TypeReference declaringClass = node.getMethod().getReference().getDeclaringClass();
231           // skip library classes
232           if (declaringClass.getClassLoader().equals(ClassLoaderReference.Primordial)) {
233             continue;
234           }
235           IR ir = node.getIR();
236           if (ir == null)
237             continue;
238           SSAInstruction[] instrs = ir.getInstructions();
239           for (int i = 0; i < instrs.length; i++) {
240             if (numSafe + numMightFail > MAX_CASTS)
241               break outer;
242             SSAInstruction instruction = instrs[i];
243             if (instruction instanceof SSACheckCastInstruction) {
244               SSACheckCastInstruction castInstr = (SSACheckCastInstruction) instruction;
245               final TypeReference[] declaredResultTypes = castInstr.getDeclaredResultTypes();
246          
247               boolean primOnly = true;
248               for (TypeReference t : declaredResultTypes) {
249                 if (! t.isPrimitiveType()) {
250                   primOnly = false;
251                 }
252               }
253               if (primOnly) {
254                 continue;
255               }
256               
257               System.err.println("CHECKING " + castInstr + " in " + node.getMethod());
258               PointerKey castedPk = heapModel.getPointerKeyForLocal(node, castInstr.getUse(0));
259               Predicate<InstanceKey> castPred = new Predicate<InstanceKey>() {
260     
261                 @Override
262                 public boolean test(InstanceKey ik) {
263                   TypeReference ikTypeRef = ik.getConcreteType().getReference();
264                   for (TypeReference t : declaredResultTypes) {
265                     if (cha.isAssignableFrom(cha.lookupClass(t), cha.lookupClass(ikTypeRef))) {
266                       return true;
267                     }
268                   }
269                   return false;
270                 }
271     
272               };
273               long startTime = System.currentTimeMillis();
274               Pair<PointsToResult, Collection<InstanceKey>> queryResult = dmp.getPointsTo(castedPk, castPred);
275               long runningTime = System.currentTimeMillis() - startTime;
276               System.err.println("running time: " + runningTime + "ms");
277               final FieldRefinePolicy fieldRefinePolicy = dmp.getRefinementPolicy().getFieldRefinePolicy();
278               switch (queryResult.fst) {
279               case SUCCESS:
280                 System.err.println("SAFE: " + castInstr + " in " + node.getMethod());
281                 if (fieldRefinePolicy instanceof ManualFieldPolicy) {
282                   ManualFieldPolicy hackedFieldPolicy = (ManualFieldPolicy) fieldRefinePolicy;
283                   System.err.println(hackedFieldPolicy.getHistory());
284                 }
285                 System.err.println("TRAVERSED " + dmp.getNumNodesTraversed() + " nodes");
286                 numSafe++;
287                 break;
288               case NOMOREREFINE:
289                 if (queryResult.snd != null) {
290                   System.err.println("MIGHT FAIL: no more refinement possible for " + castInstr + " in " + node.getMethod());
291                 } else {
292                   System.err.println("MIGHT FAIL: exceeded budget for " + castInstr + " in " + node.getMethod());
293                 }
294                 failing.add(Pair.make(node, castInstr));
295                 numMightFail++;
296                 break;
297               case BUDGETEXCEEDED:
298                 System.err.println("MIGHT FAIL: exceeded budget for " + castInstr + " in " + node.getMethod());
299                 failing.add(Pair.make(node, castInstr));
300                 numMightFail++;
301                 break;
302               default:
303                 Assertions.UNREACHABLE();
304               }
305             }
306           }
307           // break outer;
308         }
309         System.err.println("TOTAL SAFE: " + numSafe);
310         System.err.println("TOTAL MIGHT FAIL: " + numMightFail);
311         return failing;
312       }
