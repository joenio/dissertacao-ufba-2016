126     	class AB {
127     		public int getA_X_from_AB() {
128     			return a_x; // CHECK enclosing is an A
129     		}
130     
131     		public int getA_X_thru_AB() {
132     			return getA_X(); // CHECK enclosing is an A
133     		}

135     		class ABA {
136     			int aba_x;
137     			class ABAA {
138     				int getABA_X() {
139     					return aba_x; // CHECK enclosing is an ABA or ABSubA
140     				}
141     				int getA_X() {
142     					return a_x; // CHECK enclosing is an A
143     				}
144     			}
145     			class ABAB {
146     				int getABA_X() {
147     					return aba_x; // CHECK enclosing is an ABA
148     				}
149     			}
150     		}
151     		
152     		class ABSubA extends ABA {
153     			class ABSubAA {
154     				int getABA_X() {
155     					return aba_x; // CHECK enclosing is an ABSubA
156     				}
157     				int getA_X() {
158     					return a_x; // CHECK enclosing is an A
159     				}
160     				ABA.ABAA makeABAA() {
161     					return new ABAA(); // this new instruction requires us to know that ABSubA is a subclass of ABA.
162     									   // thus when we call getABA_X() on the result, it will need to find a EORK(this site,class ABA) -> THIS (of type ABSubAA) 
163     				}
164     			}
165     			int getA_X() {
166     				return a_x; // CHECK enclosing is an A
167     			}
168     		}
