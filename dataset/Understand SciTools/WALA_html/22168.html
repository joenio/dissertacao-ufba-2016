11     package com.ibm.wala.ssa;
12     
13     import com.ibm.wala.classLoader.NewSiteReference;
14     import com.ibm.wala.types.ClassLoaderReference;
15     import com.ibm.wala.types.TypeReference;
16     
17     /**
18      * An allocation instruction ("new") for SSA form.
19      * 
20      * This includes allocations of both scalars and arrays.
21      */
22     public abstract class SSANewInstruction extends SSAInstruction {
23       private final int result;
24     
25       private final NewSiteReference site;
26     
27       /**
28        * The value numbers of the arguments passed to the call. If params == null, this should be a static this statement allocates a
29        * scalar. if params != null, then params[i] is the size of the ith dimension of the array.
30        */
31       private final int[] params;
32     
33       /**
34        * Create a new instruction to allocate a scalar.
35        */
36       protected SSANewInstruction(int iindex, int result, NewSiteReference site) throws IllegalArgumentException {
37         super(iindex);
38         if (site == null) {
39           throw new IllegalArgumentException("site cannot be null");
40         }
41         this.result = result;
42         this.site = site;
43         this.params = null;
44       }
45     
46       /**
47        * Create a new instruction to allocate an array.
48        * 
49        * @throws IllegalArgumentException if site is null
50        * @throws IllegalArgumentException if params is null
51        */
52       protected SSANewInstruction(int iindex, int result, NewSiteReference site, int[] params) {
53         super(iindex);
54         if (params == null) {
55           throw new IllegalArgumentException("params is null");
56         }
57         if (site == null) {
58           throw new IllegalArgumentException("site is null");
59         }
60         assert site.getDeclaredType().isArrayType()
61             || site.getDeclaredType().getClassLoader().getLanguage() != ClassLoaderReference.Java;
62         this.result = result;
63         this.site = site;
64         this.params = new int[params.length];
65         System.arraycopy(params, 0, this.params, 0, params.length);
66       }
67     
68       @Override
69       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
70         if (params == null) {
71           return insts.NewInstruction(iindex, defs == null ? result : defs[0], site);
72         } else {
73           return insts.NewInstruction(iindex, defs == null ? result : defs[0], site, uses == null ? params : uses);
74         }
75       }
76     
77       @Override
78       public String toString(SymbolTable symbolTable) {
79         return getValueString(symbolTable, result) + " = new " + site.getDeclaredType() + "@" + site.getProgramCounter()
80             + (params == null ? "" : array2String(params, symbolTable));
81       }
82     
83       private String array2String(int[] params, SymbolTable symbolTable) {
84         StringBuffer result = new StringBuffer();
85         for (int i = 0; i < params.length; i++) {
86           result.append(getValueString(symbolTable, params[i]));
87           result.append(" ");
88         }
89         return result.toString();
90       }
91     
92       /**
93        * @see com.ibm.wala.ssa.SSAInstruction#visit(IVisitor)
94        * @throws IllegalArgumentException if v is null
95        */
96       @Override
97       public void visit(IVisitor v) {
98         if (v == null) {
99           throw new IllegalArgumentException("v is null");
100         }
101         v.visitNew(this);
102       }
103     
104       /**
105        * @see com.ibm.wala.ssa.SSAInstruction#getDef()
106        */
107       @Override
108       public boolean hasDef() {
109         return true;
110       }
111     
112       @Override
113       public int getDef() {
114         return result;
115       }
116     
117       @Override
118       public int getDef(int i) {
119         assert i == 0;
120         return result;
121       }
122     
123       @Override
124       public int getNumberOfDefs() {
125         return 1;
126       }
127     
128       /**
129        * @return TypeReference
130        */
131       public TypeReference getConcreteType() {
132         return site.getDeclaredType();
133       }
134     
135       public NewSiteReference getNewSite() {
136         return site;
137       }
138     
139       @Override
140       public int hashCode() {
141         return result * 7529 + site.getDeclaredType().hashCode();
142       }
143     
144       @Override
145       public int getNumberOfUses() {
146         return params == null ? 0 : params.length;
147       }
148     
149       @Override
150       public int getUse(int j) {
151         assert params != null : "expected params but got null in " + this.toString();
152         assert params.length > j : "found too few parameters";
153         return params[j];
154       }
155     
156       /*
157        * @see com.ibm.wala.ssa.Instruction#isPEI()
158        */
159       @Override
160       public boolean isPEI() {
161         return true;
162       }
163     
164       /*
165        * @see com.ibm.wala.ssa.Instruction#isFallThrough()
166        */
167       @Override
168       public boolean isFallThrough() {
169         return true;
170       }
171     
172     }
