11     package com.ibm.wala.util.intset;
12     
13     import java.util.Iterator;
14     import java.util.TreeSet;
15     
16     /**
17      * An object that creates mutable sparse int sets.
18      */
19     public class MutableSparseLongSetFactory implements MutableLongSetFactory {
20     
21       /**
22        * @param set
23        * @throws IllegalArgumentException  if set is null
24        */
25       @Override
26       public MutableLongSet make(long[] set) {
27         if (set == null) {
28           throw new IllegalArgumentException("set is null");
29         }
30         if (set.length == 0) {
31           return new MutableSparseLongSet();
32         } else {
33           // XXX not very efficient.
34           TreeSet<Long> T = new TreeSet<Long>();
35           for (int i = 0; i < set.length; i++) {
36             T.add(Long.valueOf(set[i]));
37           }
38           long[] copy = new long[T.size()];
39           int i = 0;
40           for (Iterator<Long> it = T.iterator(); it.hasNext();) {
41             Long I = it.next();
42             copy[i++] = I.longValue();
43           }
44           MutableSparseLongSet result = new MutableSparseLongSet(copy);
45           return result;
46         }
47       }
48     
49       @Override
50       public MutableLongSet parse(String string) throws NumberFormatException {
51         int[] backingStore = SparseIntSet.parseIntArray(string);
52         long[] bs = new long[ backingStore.length ];
53         for(int i = 0; i < bs.length; i++) bs[i] = backingStore[i];
54         return new MutableSparseLongSet(bs);
55       }
56     
57       /*
58        * @see com.ibm.wala.util.intset.MutableLongSetFactory#make(com.ibm.wala.util.intset.LongSet)
59        */
60       @Override
61       public MutableLongSet makeCopy(LongSet x) throws IllegalArgumentException {
62         if (x == null) {
63           throw new IllegalArgumentException("x == null");
64         }
65         return MutableSparseLongSet.make(x);
66       }
67     
68       /*
69        * @see com.ibm.wala.util.intset.MutableLongSetFactory#make()
70        */
71       @Override
72       public MutableLongSet make() {
73         return new MutableSparseLongSet();
74       }
75     
76     }
