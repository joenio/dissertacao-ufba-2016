13     /**
14      * A Context is a mapping from a name (ContextKey) to a value (ContextItem)
15      * 
16      * For example, for CFA-1, there is only one name ("caller"); and the context maps "caller" to an IMethod
17      * 
18      * As another example, for CPA, there would be name for each parameter slot ("zero","one","two"), and the Context provides a mapping
19      * from this name to a set of types. eg. "one" -> {java.lang.String, java.lang.Date}
20      */
21     public interface Context {
22       /**
23        * @return the objects corresponding to a given name
24        */
25       ContextItem get(ContextKey name);
26     }
