46       public JavaJavaScriptHybridCallGraphBuilder(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache) {
47         super(cha, options, cache, new AstCFAPointerKeys());
48         globalObject = new GlobalObjectKey(cha.lookupClass(JavaScriptTypes.Root));
49         
50         SSAContextInterpreter contextInterpreter = makeDefaultContextInterpreters(null, options, cha);
51         setContextInterpreter( contextInterpreter );
52     
53         ContextSelector def = new DefaultContextSelector(options, cha);
54         Map<Atom,ContextSelector> languageSelectors = HashMapFactory.make();
55         languageSelectors.put(JavaScriptTypes.jsName, 
56           new JavaScriptFunctionApplyContextSelector(new JavaScriptConstructorContextSelector(def)));
57         languageSelectors.put(Language.JAVA.getName(), def);
58         setContextSelector(new CrossLanguageContextSelector(languageSelectors));
59         
60         Map<Atom,InstanceKeyFactory> instanceKeys = HashMapFactory.make();
61         instanceKeys.put(
62           JavaScriptTypes.jsName,
63           new JavaScriptScopeMappingInstanceKeys(cha, this, new JavaScriptConstructorInstanceKeys(new ZeroXInstanceKeys(
64               options, cha, contextInterpreter, ZeroXInstanceKeys.ALLOCATIONS))));
65         instanceKeys.put(
66             Language.JAVA.getName(),
67             new ZeroXInstanceKeys(options, cha, contextInterpreter, ZeroXInstanceKeys.NONE));
68         setInstanceKeys(new CrossLanguageInstanceKeys(instanceKeys));
69      }

81         return new TargetLanguageSelector<ConstraintVisitor, CGNode>() {
82           @Override

84             if (JavaScriptTypes.jsName.equals(language)) {
85               return new JSConstraintVisitor(JavaJavaScriptHybridCallGraphBuilder.this, construct);

150         if (JavaScriptLoader.JS.equals(opNode.getMethod().getDeclaringClass().getClassLoader().getLanguage())) {
151           return definingMethod.equals(opNode.getMethod().getReference().getDeclaringClass().getName().toString());
