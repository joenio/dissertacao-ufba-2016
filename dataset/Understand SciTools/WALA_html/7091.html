20     
21       private ClassHierarchyStore cha = new ClassHierarchyStore();
22     
23       private Writer out = new PrintWriter(System.err);
24       
25       public OnlineDynamicCallGraph() throws IllegalArgumentException, IOException, InvalidClassFileException {
26         OfflineInstrumenter libReader = new OfflineInstrumenter(true);
27         for (String cps : new String[]{ System.getProperty("java.class.path"), System.getProperty("sun.boot.class.path") }) {
28           for (String cp : cps.split(File.pathSeparator)) {
29             File x = new File(cp);
30             if (x.exists()) {
31               if (x.isDirectory()) {
32                 libReader.addInputDirectory(x, x);
33               } else {
34                 libReader.addInputJar(x);
35               }
36             }
37           } 
38         }
39         
40         ClassInstrumenter ci;
41         while ((ci = libReader.nextClass()) != null) {
42           CTUtils.addClassToHierarchy(cha, ci.getReader());
43         }
44       }
45       
46       @Override
47       public byte[] transform(ClassLoader loader, String className, Class<?> classBeingRedefined, ProtectionDomain protectionDomain,
48           byte[] classfileBuffer) throws IllegalClassFormatException {
49         try {
50           if (className.contains("com/ibm/wala") || className.contains("java/lang") || (className.contains("java/") && !className.matches("java/util/[A-Z]")) || className.contains("sun/")) {
51             return classfileBuffer;
52           } else {
53             ClassInstrumenter ci = new ClassInstrumenter(className, classfileBuffer, cha, false);
54             return OfflineDynamicCallGraph.doClass(ci, out).makeBytes();
55           }
56         } catch (InvalidClassFileException | IOException | FailureException e) {
57           e.printStackTrace();
58           System.err.println("got here with " + e.getMessage());
59           throw new IllegalClassFormatException(e.getMessage());
60         }
61       }
62       
63       public static void premain(String agentArgs, Instrumentation inst) throws IllegalArgumentException, IOException, InvalidClassFileException {
64         inst.addTransformer(new OnlineDynamicCallGraph());
65       }
