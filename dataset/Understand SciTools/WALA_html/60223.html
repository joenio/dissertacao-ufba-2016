66     	protected static <T> Set<T> processCG(CallGraph cg, Predicate<CGNode> filter, Function<CGNode,T> map) {
67     		Set<T> result = HashSetFactory.make();
68     		for(CGNode n : cg) {
69     			if (filter.test(n)) {
70     				result.add(map.apply(n));
71     			}
72     		}
73     		return result;
74     	}
75     	
76     	protected static Set<MethodReference> applicationMethods(CallGraph cg) {
77     		return processCG(cg,
78     			new Predicate<CGNode>() {
79     				@Override
80     				public boolean test(CGNode t) {
81     					return t.getMethod().getReference().getDeclaringClass().getClassLoader().equals(ClassLoaderReference.Application);
82     				}
83     			},
84     			new Function<CGNode,MethodReference>() {
85     				@Override
86     				public MethodReference apply(CGNode object) {
87     					return object.getMethod().getReference();
88     				}
89     			});
90     	}

101     	@SuppressWarnings("unused")
102       private static SSAContextInterpreter makeDefaultInterpreter(AnalysisOptions options, AnalysisCache cache) {
103     		return new DefaultSSAInterpreter(options, cache) {
104     			@Override
105     			public Iterator<NewSiteReference> iterateNewSites(CGNode node) {
106     				return 
107     					new MapIterator<SSAInstruction,NewSiteReference>(
108     						new FilterIterator<SSAInstruction>(
109     								node.getIR().iterateAllInstructions(), 
110     								new Predicate<SSAInstruction>() {
111     									@Override
112     									public boolean test(SSAInstruction t) {
113     										return t instanceof SSANewInstruction;
114     									} 
115     								}), 
116     						new Function<SSAInstruction,NewSiteReference>() {
117     							@Override
118     							public NewSiteReference apply(SSAInstruction object) {
119     								return ((SSANewInstruction)object).getNewSite();
120     							} 
121     						}
122     					);
123     			}
124     		};
125     	}

127     	public static Pair<CallGraph, PointerAnalysis<InstanceKey>> makeAPKCallGraph(URI[] androidLibs, File androidAPIJar, String apkFileName, IProgressMonitor monitor, ReflectionOptions policy) throws IOException, ClassHierarchyException, IllegalArgumentException, CancelException {
128     		AnalysisScope scope = makeDalvikScope(androidLibs, androidAPIJar, apkFileName);
129     
130     		final IClassHierarchy cha = ClassHierarchy.make(scope);
131     
132     		AnalysisCache cache = new AnalysisCache(new DexIRFactory());
133     
134     		Set<LocatorFlags> flags = HashSetFactory.make();
135     		flags.add(LocatorFlags.INCLUDE_CALLBACKS);
136     		flags.add(LocatorFlags.EP_HEURISTIC);
137     		flags.add(LocatorFlags.CB_HEURISTIC);
138     		AndroidEntryPointLocator eps = new AndroidEntryPointLocator(flags);
139     		List<? extends Entrypoint> es = eps.getEntryPoints(cha);
140     
141     		assert ! es.isEmpty();
142     		
143     		AnalysisOptions options = new AnalysisOptions(scope, es);
144     		options.setReflectionOptions(policy);
145     		
146     		// SSAPropagationCallGraphBuilder cgb = Util.makeZeroCFABuilder(options, cache, cha, scope, null, makeDefaultInterpreter(options, cache));
147     		SSAPropagationCallGraphBuilder cgb = Util.makeZeroCFABuilder(options, cache, cha, scope);
148       
149     		CallGraph callGraph = cgb.makeCallGraph(options, monitor);
150     		
151     		PointerAnalysis<InstanceKey> ptrAnalysis = cgb.getPointerAnalysis();
152     		
153     		return Pair.make(callGraph, ptrAnalysis);
154     	}

156     	public static Pair<CallGraph, PointerAnalysis<InstanceKey>> makeDalvikCallGraph(URI[] androidLibs, File androidAPIJar, String mainClassName, String dexFileName) throws IOException, ClassHierarchyException, IllegalArgumentException, CancelException {
157     		AnalysisScope scope = makeDalvikScope(androidLibs, androidAPIJar, dexFileName);
158     		
159     		final IClassHierarchy cha = ClassHierarchy.make(scope);
160     
161     		TypeReference mainClassRef = TypeReference.findOrCreate(ClassLoaderReference.Application, mainClassName);
162     		IClass mainClass = cha.lookupClass(mainClassRef);
163     		assert mainClass != null;
164     
165     		System.err.println("building call graph for " + mainClass + ":" + mainClass.getClass());
166     		
167     		Iterable<Entrypoint> entrypoints = Util.makeMainEntrypoints(scope, cha, mainClassName);
168     		
169     		AnalysisCache cache = new AnalysisCache(new DexIRFactory());
170     
171     		AnalysisOptions options = new AnalysisOptions(scope, entrypoints);
172     
173     		SSAPropagationCallGraphBuilder cgb = Util.makeZeroCFABuilder(options, cache, cha, scope);
174       
175     		CallGraph callGraph = cgb.makeCallGraph(options);
176     
177     		MethodReference mmr = MethodReference.findOrCreate(mainClassRef, "main", "([Ljava/lang/String;)V");
178     		assert !callGraph.getNodes(mmr).isEmpty();
179     		
180     		PointerAnalysis<InstanceKey> ptrAnalysis = cgb.getPointerAnalysis();
181     		
182     		return Pair.make(callGraph, ptrAnalysis);
183     	}
