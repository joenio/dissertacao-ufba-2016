189       public XMLMethodSummaryReader(InputStream xmlFile, AnalysisScope scope) {
190         super();
191         if (xmlFile == null) {
192           throw new IllegalArgumentException("null xmlFile");
193         }
194         if (scope == null) {
195           throw new IllegalArgumentException("null scope");
196         }
197         this.scope = scope;
198         try {
199           readXML(xmlFile);
200         } catch (Exception e) {
201           e.printStackTrace();
202           Assertions.UNREACHABLE();
203         }
204       }

207         SAXHandler handler = new SAXHandler();
208     
209         assert xml != null : "Null xml stream";
210         SAXParserFactory factory = SAXParserFactory.newInstance();
211         factory.newSAXParser().parse(new InputSource(xml), handler);

276         /*
277          * @see org.xml.sax.ContentHandler#startElement(java.lang.String, java.lang.String, java.lang.String, org.xml.sax.Attributes)
278          */
279         @Override
280         public void startElement(String uri, String name, String qName, Attributes atts) {
281           Integer element = elementMap.get(qName);
282           if (element == null) {
283             Assertions.UNREACHABLE("Invalid element: " + qName);
284           }
285           switch (element.intValue()) {
286           case E_CLASSLOADER: {
287             String clName = atts.getValue(A_NAME);
288             governingLoader = classLoaderName2Ref(clName);
289           }
290             break;
291           case E_METHOD:
292             String mname = atts.getValue(A_NAME);
293             if (mname.equals(A_WILDCARD)) {
294               Assertions.UNREACHABLE("Wildcards not currently implemented.");
295             } else {
296               startMethod(atts);
297             }
298             break;
299           case E_CLASS:
300             String cname = atts.getValue(A_NAME);
301             if (cname.equals(A_WILDCARD)) {
302               Assertions.UNREACHABLE("Wildcards not currently implemented");
303             } else {
304               startClass(cname, atts);
305             }
306             break;
307           case E_PACKAGE:
308             governingPackage = Atom.findOrCreateUnicodeAtom(atts.getValue(A_NAME));
309             String ignore = atts.getValue(A_IGNORE);
310             if (ignore != null && ignore.equals(V_TRUE)) {
311               ignoredPackages.add(governingPackage);
312             }
313             break;
314           case E_CALL:
315             processCallSite(atts);
316             break;
317           case E_NEW:
318             processAllocation(atts);
319             break;
320           case E_PUTSTATIC:
321             processPutStatic(atts);
322             break;
323           case E_PUTFIELD:
324             processPutField(atts);
325             break;
326           case E_GETFIELD:
327             processGetField(atts);
328             break;
329           case E_ATHROW:
330             processAthrow(atts);
331             break;
332           case E_AASTORE:
333             processAastore(atts);
334             break;
335           case E_AALOAD:
336             processAaload(atts);
337             break;
338           case E_RETURN:
339             processReturn(atts);
340             break;
341           case E_POISON:
342             processPoison(atts);
343             break;
344           case E_CONSTANT:
345             processConstant(atts);
346             break;
347           case E_SUMMARY_SPEC:
348             break;
349           default:
350             Assertions.UNREACHABLE("Unexpected element: " + name);
351             break;
352           }
353         }

355         private void startClass(String cname, Attributes atts) {
356           String clName = "L" + governingPackage + "/" + cname;
357           governingClass = className2Ref(clName);
358           String allocString = atts.getValue(A_ALLOCATABLE);
359           if (allocString != null) {
360             Assertions.productionAssertion(allocString.equals("true"));
361             allocatable.add(governingClass);
362           }
363         }

365         /*
366          * @see org.xml.sax.ContentHandler#endElement(java.lang.String, java.lang.String, java.lang.String)
367          */
368         @Override
369         public void endElement(String uri, String name, String qName) {
370           Integer element = elementMap.get(qName);
371           if (element == null) {
372             Assertions.UNREACHABLE("Invalid element: " + name);
373           }
374           switch (element.intValue()) {
375           case E_CLASSLOADER:
376             governingLoader = null;
377             break;
378           case E_METHOD:
379             if (governingMethod != null) {
380               checkReturnValue(governingMethod);
381             }
382             governingMethod = null;
383             symbolTable = null;
384             break;
385           case E_CLASS:
386             governingClass = null;
387             break;
388           case E_PACKAGE:
389             governingPackage = null;
390             break;
391           case E_CALL:
392           case E_GETFIELD:
393           case E_NEW:
394           case E_POISON:
395           case E_PUTSTATIC:
396           case E_PUTFIELD:
397           case E_AALOAD:
398           case E_AASTORE:
399           case E_ATHROW:
400           case E_SUMMARY_SPEC:
401           case E_RETURN:
402           case E_CONSTANT:
403             break;
404           default:
405             Assertions.UNREACHABLE("Unexpected element: " + name);
406             break;
407           }
408         }

410         /**
411          * If a method is declared to return a value, be sure the method summary includes a return statement. Throw an assertion if not.
412          * 
413          * @param governingMethod
414          */
415         private void checkReturnValue(MethodSummary governingMethod) {
416           Assertions.productionAssertion(governingMethod != null);
417           Assertions.productionAssertion(governingMethod.getReturnType() != null);
418           if (governingMethod.getReturnType().isReferenceType()) {
419             SSAInstruction[] statements = governingMethod.getStatements();
420             for (int i = 0; i < statements.length; i++) {
421               if (statements[i] instanceof SSAReturnInstruction) {
422                 return;
423               }
424             }
425             Assertions.UNREACHABLE("Method summary " + governingMethod + " must have a return statement.");
426           }
427     
428         }

430         /**
431          * Process an element indicating a call instruction
432          * 
433          * @param atts
434          */
435         private void processCallSite(Attributes atts) {
436           String typeString = atts.getValue(A_TYPE);
437           String nameString = atts.getValue(A_NAME);
438           String classString = atts.getValue(A_CLASS);
439           String descString = atts.getValue(A_DESCRIPTOR);
440           TypeReference type = TypeReference.findOrCreate(governingLoader, TypeName.string2TypeName(classString));
441           Atom nm = Atom.findOrCreateAsciiAtom(nameString);
442           Language lang = scope.getLanguage(governingLoader.getLanguage());
443           SSAInstructionFactory insts = lang.instructionFactory();
444           Descriptor D = Descriptor.findOrCreateUTF8(lang, descString);
445           MethodReference ref = MethodReference.findOrCreate(type, nm, D);
446           CallSiteReference site = null;
447           int nParams = ref.getNumberOfParameters();
448           if (typeString.equals("virtual")) {
449             site = CallSiteReference.make(governingMethod.getNextProgramCounter(), ref, IInvokeInstruction.Dispatch.VIRTUAL);
450             nParams++;
451           } else if (typeString.equals("special")) {
452             site = CallSiteReference.make(governingMethod.getNextProgramCounter(), ref, IInvokeInstruction.Dispatch.SPECIAL);
453             nParams++;
454           } else if (typeString.equals("interface")) {
455             site = CallSiteReference.make(governingMethod.getNextProgramCounter(), ref, IInvokeInstruction.Dispatch.INTERFACE);
456             nParams++;
457           } else if (typeString.equals("static")) {
458             site = CallSiteReference.make(governingMethod.getNextProgramCounter(), ref, IInvokeInstruction.Dispatch.STATIC);
459           } else {
460             Assertions.UNREACHABLE("Invalid call type " + typeString);
461           }
462           int[] params = new int[nParams];
463     
464           for (int i = 0; i < params.length; i++) {
465             String argString = atts.getValue(A_ARG + i);
466             Assertions.productionAssertion(argString != null, "unspecified arg in method " + governingMethod + " " + site);
467             Integer valueNumber = symbolTable.get(argString);
468             if (valueNumber == null) {
469               Assertions.UNREACHABLE("Cannot lookup value: " + argString);
470             }
471             params[i] = valueNumber.intValue();
472           }
473     
474           // allocate local for exceptions
475           int exceptionValue = nextLocal++;
476     
477           // register the local variable defined by this call, if appropriate
478           String defVar = atts.getValue(A_DEF);
479           if (defVar != null) {
480     
481             if (symbolTable.keySet().contains(defVar)) {
482               Assertions.UNREACHABLE("Cannot def variable twice: " + defVar + " in " + governingMethod);
483             }
484     
485             int defNum = nextLocal;
486             symbolTable.put(defVar, new Integer(nextLocal++));
487     
488             governingMethod.addStatement(insts.InvokeInstruction(governingMethod.getNumberOfStatements(), defNum, params, exceptionValue, site, null));
489           } else {
490             // ignore return value, if any
491             governingMethod.addStatement(insts.InvokeInstruction(governingMethod.getNumberOfStatements(), params, exceptionValue, site, null));
492           }
493         }

495         /**
496          * Process an element indicating a new allocation site.
497          * 
498          * @param atts
499          */
500         private void processAllocation(Attributes atts) {
501           Language lang = scope.getLanguage(governingLoader.getLanguage());
502           SSAInstructionFactory insts = lang.instructionFactory();
503     
504           // deduce the concrete type allocated
505           String classString = atts.getValue(A_CLASS);
506           final TypeReference type = TypeReference.findOrCreate(governingLoader, TypeName.string2TypeName(classString));
507     
508           // register the local variable defined by this allocation
509           String defVar = atts.getValue(A_DEF);
510           if (symbolTable.keySet().contains(defVar)) {
511             Assertions.UNREACHABLE("Cannot def variable twice: " + defVar + " in " + governingMethod);
512           }
513           if (defVar == null) {
514             // the method summary ignores the def'ed variable.
515             // just allocate a temporary
516             defVar = "L" + nextLocal;
517           }
518           int defNum = nextLocal;
519           symbolTable.put(defVar, new Integer(nextLocal++));
520     
521           // create the allocation statement and add it to the method summary
522           NewSiteReference ref = NewSiteReference.make(governingMethod.getNextProgramCounter(), type);
523     
524           SSANewInstruction a = null;
525           if (type.isArrayType()) {
526             String size = atts.getValue(A_SIZE);
527             Assertions.productionAssertion(size != null);
528             Integer sNumber = symbolTable.get(size);
529             Assertions.productionAssertion(sNumber != null);
530             Assertions.productionAssertion(
531                 // array of objects
532                 type.getDerivedMask()==ArrayMask || 
533                 // array of primitives
534                 type.getDerivedMask()==((ArrayMask<<ElementBits)|PrimitiveMask));  
535             a = insts.NewInstruction(governingMethod.getNumberOfStatements(), defNum, ref, new int[] { sNumber.intValue() });
536           } else {
537             a = insts.NewInstruction(governingMethod.getNumberOfStatements(), defNum, ref);
538           }
539           governingMethod.addStatement(a);
540         }

543          * Process an element indicating an Athrow
544          * 
545          * @param atts
546          */
547         private void processAthrow(Attributes atts) {
548           Language lang = scope.getLanguage(governingLoader.getLanguage());
549           SSAInstructionFactory insts = lang.instructionFactory();
550     
551           // get the value thrown
552           String V = atts.getValue(A_VALUE);
553           if (V == null) {
554             Assertions.UNREACHABLE("Must specify value for putfield " + governingMethod);
555           }
556           Integer valueNumber = symbolTable.get(V);
557           if (valueNumber == null) {
558             Assertions.UNREACHABLE("Cannot lookup value: " + V);
559           }
560     
561           SSAThrowInstruction T = insts.ThrowInstruction(governingMethod.getNumberOfStatements(), valueNumber.intValue());
562           governingMethod.addStatement(T);
563         }

565         /**
566          * Process an element indicating a putfield.
567          * 
568          * @param atts
569          */
570         private void processGetField(Attributes atts) {
571           Language lang = scope.getLanguage(governingLoader.getLanguage());
572           SSAInstructionFactory insts = lang.instructionFactory();
573     
574           // deduce the field written
575           String classString = atts.getValue(A_CLASS);
576           TypeReference type = TypeReference.findOrCreate(governingLoader, TypeName.string2TypeName(classString));
577     
578           String fieldString = atts.getValue(A_FIELD);
579           Atom fieldName = Atom.findOrCreateAsciiAtom(fieldString);
580     
581           String ftString = atts.getValue(A_FIELD_TYPE);
582           TypeReference fieldType = TypeReference.findOrCreate(governingLoader, TypeName.string2TypeName(ftString));
583     
584           FieldReference field = FieldReference.findOrCreate(type, fieldName, fieldType);
585     
586           // get the value def'fed
587           String defVar = atts.getValue(A_DEF);
588           if (symbolTable.keySet().contains(defVar)) {
589             Assertions.UNREACHABLE("Cannot def variable twice: " + defVar + " in " + governingMethod);
590           }
591           if (defVar == null) {
592             Assertions.UNREACHABLE("Must specify def for getfield " + governingMethod);
593           }
594           int defNum = nextLocal;
595           symbolTable.put(defVar, new Integer(nextLocal++));
596     
597           // get the ref read from
598           String R = atts.getValue(A_REF);
599           if (R == null) {
600             Assertions.UNREACHABLE("Must specify ref for getfield " + governingMethod);
601           }
602           Integer refNumber = symbolTable.get(R);
603           if (refNumber == null) {
604             Assertions.UNREACHABLE("Cannot lookup ref: " + R);
605           }
606     
607           SSAGetInstruction G = insts.GetInstruction(governingMethod.getNumberOfStatements(), defNum, refNumber.intValue(), field);
608           governingMethod.addStatement(G);
609         }

670           Atom fieldName = Atom.findOrCreateAsciiAtom(fieldString);
671     
672           String ftString = atts.getValue(A_FIELD_TYPE);
673           TypeReference fieldType = TypeReference.findOrCreate(governingLoader, TypeName.string2TypeName(ftString));
674     
675           FieldReference field = FieldReference.findOrCreate(type, fieldName, fieldType);
676     
677           // get the value stored
678           String V = atts.getValue(A_VALUE);
679           if (V == null) {
680             Assertions.UNREACHABLE("Must specify value for putstatic " + governingMethod);
681           }
682           Integer valueNumber = symbolTable.get(V);
683           if (valueNumber == null) {
684             Assertions.UNREACHABLE("Cannot lookup value: " + V);
685           }
686           SSAPutInstruction P = insts.PutInstruction(governingMethod.getNumberOfStatements(), valueNumber.intValue(), field);
687           governingMethod.addStatement(P);
688         }

692          * 
693          * @param atts
694          */
695         private void processAastore(Attributes atts) {
696           Language lang = scope.getLanguage(governingLoader.getLanguage());
697           SSAInstructionFactory insts = lang.instructionFactory();
698     
699           String R = atts.getValue(A_REF);
700           if (R == null) {
701             Assertions.UNREACHABLE("Must specify ref for aastore " + governingMethod);
702           }
703           Integer refNumber = symbolTable.get(R);
704           if (refNumber == null) {
705             Assertions.UNREACHABLE("Cannot lookup value: " + R);
706           }
707           // N.B: we currently ignore the index
708           String I = atts.getValue(A_INDEX);
709           if (I == null) {
710             Assertions.UNREACHABLE("Must specify index for aastore " + governingMethod);
711           }
712           String V = atts.getValue(A_VALUE);
713           if (V == null) {
714             Assertions.UNREACHABLE("Must specify value for aastore " + governingMethod);
715           }
716           /** BEGIN Custom change: expect type information in array-store instructions */
717           String strType = atts.getValue(A_TYPE);
718           TypeReference type;
719           if (strType == null) {
720             type = TypeReference.JavaLangObject;
721           } else {
722             type = TypeReference.findOrCreate(governingLoader, strType);
723           }
724           /** END Custom change: get type information in array-store instructions */
725           Integer valueNumber = symbolTable.get(V);
726           if (valueNumber == null) {
727             Assertions.UNREACHABLE("Cannot lookup value: " + V);
728           }
729           /** BEGIN Custom change: expect type information in array-store instructions */
730           SSAArrayStoreInstruction S = insts.ArrayStoreInstruction(governingMethod.getNumberOfStatements(), refNumber.intValue(), 0, valueNumber.intValue(), type);
731           /** END Custom change: get type information in array-store instructions */
732           governingMethod.addStatement(S);
733         }

735         /**
736          * Process an element indicating an Aaload
737          * 
738          * @param atts
739          */
740         private void processAaload(Attributes atts) {
741           //<aaload def="foo" ref="arg1" index="the-answer" />
742           Language lang = scope.getLanguage(governingLoader.getLanguage());
743           SSAInstructionFactory insts = lang.instructionFactory();
744     
745           String R = atts.getValue(A_REF);
746           if (R == null) {
747             Assertions.UNREACHABLE("Must specify ref for aaload " + governingMethod);
748           }
749           Integer refNumber = symbolTable.get(R);
750           if (refNumber == null) {
751             Assertions.UNREACHABLE("Cannot lookup value: " + R);
752           }
753           // N.B: we currently ignore the index
754           String I = atts.getValue(A_INDEX);
755           if (I == null) {
756             Assertions.UNREACHABLE("Must specify index for aaload " + governingMethod);
757           }
758           String strType = atts.getValue(A_TYPE);
759           TypeReference type;
760           if (strType == null) {
761             type = TypeReference.JavaLangObject;
762           } else {
763             type = TypeReference.findOrCreate(governingLoader, strType);
764           }
765        // get the value def'fed
766           String defVar = atts.getValue(A_DEF);
767           if (symbolTable.keySet().contains(defVar)) {
768             Assertions.UNREACHABLE("Cannot def variable twice: " + defVar + " in " + governingMethod);
769           }
770           if (defVar == null) {
771             Assertions.UNREACHABLE("Must specify def for getfield " + governingMethod);
772           }
773           int defNum = nextLocal;
774           symbolTable.put(defVar, new Integer(nextLocal++));
775           SSAArrayLoadInstruction S = insts.ArrayLoadInstruction(governingMethod.getNumberOfStatements(), defNum, refNumber.intValue(), 0,
776               type);
777           governingMethod.addStatement(S);
778         }

780         /**
781          * Process an element indicating a return statement.
782          * 
783          * @param atts
784          */
785         private void processReturn(Attributes atts) {
786           Language lang = scope.getLanguage(governingLoader.getLanguage());
787           SSAInstructionFactory insts = lang.instructionFactory();
788     
789           if (governingMethod.getReturnType() != null) {
790             String retV = atts.getValue(A_VALUE);
791             if (retV == null) {
792               SSAReturnInstruction R = insts.ReturnInstruction(governingMethod.getNumberOfStatements());
793               governingMethod.addStatement(R);
794             } else {
795               Integer valueNumber = symbolTable.get(retV);
796               if (valueNumber == null) {
797                 if (!retV.equals(V_NULL)) {
798                   Assertions.UNREACHABLE("Cannot return value with no def: " + retV);
799                 } else {
800                   valueNumber = symbolTable.get(V_NULL);
801                   if (valueNumber == null) {
802                     valueNumber = new Integer(nextLocal++);
803                     symbolTable.put(V_NULL, valueNumber);
804                   }
805                 }
806               }
807               boolean isPrimitive = governingMethod.getReturnType().isPrimitiveType();
808               SSAReturnInstruction R = insts.ReturnInstruction(governingMethod.getNumberOfStatements(), valueNumber.intValue(), isPrimitive);
809               governingMethod.addStatement(R);
810             }
811           }
812         }

814         /**
815          * @param atts
816          */
817         private void processConstant(Attributes atts) {
818           String var = atts.getValue(A_NAME);
819           if (var == null)
820             Assertions.UNREACHABLE("Must give name for constant");
821           Integer valueNumber = new Integer(nextLocal++);
822           symbolTable.put(var, valueNumber);
823     
824           String typeString = atts.getValue(A_TYPE);
825           String valueString = atts.getValue(A_VALUE);
826     
827           governingMethod.addConstant(valueNumber, (typeString.equals("int")) ? new ConstantValue(new Integer(valueString))
828               : (typeString.equals("long")) ? new ConstantValue(new Long(valueString))
829                   : (typeString.equals("short")) ? new ConstantValue(new Short(valueString))
830                       : (typeString.equals("float")) ? new ConstantValue(new Float(valueString))
831                           : (typeString.equals("double")) ? new ConstantValue(new Double(valueString)) : null);
832         }

835          * Process an element which indicates this method is "poison"
836          * 
837          * @param atts
838          */
839         private void processPoison(Attributes atts) {
840           String reason = atts.getValue(A_REASON);

849           } else {
850             Assertions.UNREACHABLE("Unexpected level: " + level);
851           }
852         }

854         /**
855          * Begin processing of a method. 1. Set the governing method. 2. Initialize the nextLocal variable
856          * 
857          * @param atts
858          */
859         private void startMethod(Attributes atts) {
860     
861           String methodName = atts.getValue(A_NAME);
862     
863           Atom mName = Atom.findOrCreateUnicodeAtom(methodName);
864           String descString = atts.getValue(A_DESCRIPTOR);
865           Language lang = scope.getLanguage(governingLoader.getLanguage());
866           Descriptor D = Descriptor.findOrCreateUTF8(lang, descString);
867     
868           MethodReference ref = MethodReference.findOrCreate(governingClass, mName, D);
869           governingMethod = new MethodSummary(ref);
870     
871           if (DEBUG) {
872             System.err.println(("Register method summary: " + ref));
873           }
874           summaries.put(ref, governingMethod);
875     
876           boolean isStatic = false;
877           String staticString = atts.getValue(A_STATIC);
878           if (staticString != null) {
879             if (staticString.equals("true")) {
880               isStatic = true;
881               governingMethod.setStatic(true);
882             } else if (staticString.equals("false")) {
883               isStatic = false;
884               governingMethod.setStatic(false);
885             } else {
886               Assertions.UNREACHABLE("Invalid attribute value " + A_STATIC + ": " + staticString);
887             }
888           }
889     
890           String factoryString = atts.getValue(A_FACTORY);
891           if (factoryString != null) {
892             if (factoryString.equals("true")) {
893               governingMethod.setFactory(true);
894             } else if (factoryString.equals("false")) {
895               governingMethod.setFactory(false);
896             } else {
897               Assertions.UNREACHABLE("Invalid attribute value " + A_FACTORY + ": " + factoryString);
898             }
899           }
900     
901           // This is somewhat gross, but it is to deal with the fact that
902           // some non-Java languages have notions of arguments that do not
903           // map nicely to descriptors.
904           int nParams;
905           String specifiedArgs = atts.getValue(A_NUM_ARGS);
906           if (specifiedArgs == null) {
907             nParams = ref.getNumberOfParameters();
908             if (!isStatic) {
909               nParams += 1;
910             }
911           } else {
912             nParams = Integer.parseInt(specifiedArgs);
913           }
914     
915           // note that symbol tables reserve v0 for "unknown", so v1 gets assigned
916           // to the first parameter "arg0", and so forth.
917           nextLocal = nParams + 1;
918           symbolTable = HashMapFactory.make(5);
919           // create symbols for the parameters
920           for (int i = 0; i < nParams; i++) {
921             symbolTable.put("arg" + i, new Integer(i + 1));
922           }
923         }
