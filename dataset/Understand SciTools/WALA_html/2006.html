25     /**
26      * For synthetic methods marked as "Factories", we analyze in a context defined by the caller.
27      */
28     class FactoryContextSelector implements ContextSelector {
29     
30       public FactoryContextSelector() {
31       }
32       
33       /*
34        * @see com.ibm.wala.ipa.callgraph.ContextSelector#getCalleeTarget(com.ibm.wala.ipa.callgraph.CGNode, com.ibm.wala.classLoader.CallSiteReference, com.ibm.wala.classLoader.IMethod)
35        */
36       @Override
37       public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] receiver) {
38         if (callee == null) {
39           throw new IllegalArgumentException("callee is null");
40         }
41         if (callee.isSynthetic()) {
42           SyntheticMethod s = (SyntheticMethod) callee;
43           if (s.isFactoryMethod()) {
44             return new CallStringContext(new CallString(site, caller.getMethod()));
45           }
46         }
47         return null;
48       }
49     
50       @Override
51       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
52         return EmptyIntSet.instance;
53       }
54     
55     }
