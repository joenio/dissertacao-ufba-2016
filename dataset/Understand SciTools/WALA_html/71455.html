17     /**
18      * A Decoder translates a method's Java bytecode into shrikeBT code, i.e. an array of Instruction objects and an array of lists of
19      * ExceptionHandlers.
20      * 
21      * This class implements basic decoding functionality. A toolkit for reading class files must specialize this class with particular
22      * constant pool reader implementation.
23      * 
24      * Normal usage of this class looks like this:
25      * 
26      * <pre>
27      * 
28      *    Decoder d = new MyToolkitDecoder(...);
29      *    try {
30      *      d.decode();
31      *    } catch (Decoder.InvalidBytecodeException ex) {
32      *      ...
33      *    }
34      *    Instruction[] myInstructions = d.getInstructions();
35      *    ExceptionHandler[][] exnHandlers = d.getHandlers();
36      * 
37      * </pre>
38      */

53       private static Instruction[] makeSimpleInstructions() {
54         Instruction[] table = new Instruction[256];
55     
56         table[OP_aconst_null] = ConstantInstruction.make(TYPE_null, null);
57         for (int i = OP_iconst_m1; i <= OP_iconst_5; i++) {
58           table[i] = ConstantInstruction.make(TYPE_int, new Integer(i - OP_iconst_m1 - 1));
59         }
60         for (int i = OP_lconst_0; i <= OP_lconst_1; i++) {
61           table[i] = ConstantInstruction.make(TYPE_long, new Long(i - OP_lconst_0));
62         }
63         for (int i = OP_fconst_0; i <= OP_fconst_2; i++) {
64           table[i] = ConstantInstruction.make(TYPE_float, new Float(i - OP_fconst_0));
65         }
66         for (int i = OP_dconst_0; i <= OP_dconst_1; i++) {
67           table[i] = ConstantInstruction.make(TYPE_double, new Double(i - OP_dconst_0));
68         }
69     
70         for (int i = OP_iload_0; i <= OP_aload_3; i++) {
71           table[i] = LoadInstruction.make(indexedTypes[(i - OP_iload_0) / 4], (i - OP_iload_0) % 4);
72         }
73         for (int i = OP_iaload; i <= OP_saload; i++) {
74           table[i] = ArrayLoadInstruction.make(indexedTypes[i - OP_iaload]);
75         }
76         for (int i = OP_istore_0; i <= OP_astore_3; i++) {
77           table[i] = StoreInstruction.make(indexedTypes[(i - OP_istore_0) / 4], (i - OP_istore_0) % 4);
78         }
79         for (int i = OP_iastore; i <= OP_sastore; i++) {
80           table[i] = ArrayStoreInstruction.make(indexedTypes[i - OP_iastore]);
81         }
82     
83         table[OP_pop] = PopInstruction.make(1);
84         table[OP_dup] = DupInstruction.make(1, 0);
85         table[OP_dup_x1] = DupInstruction.make(1, 1);
86         table[OP_swap] = SwapInstruction.make();
87     
88         for (int i = OP_iadd; i <= OP_drem; i++) {
89           table[i] = BinaryOpInstruction
90               .make(indexedTypes[(i - OP_iadd) % 4], BinaryOpInstruction.Operator.values()[(i - OP_iadd) / 4]);
91         }
92         for (int i = OP_ineg; i <= OP_dneg; i++) {
93           table[i] = UnaryOpInstruction.make(indexedTypes[i - OP_ineg], IUnaryOpInstruction.Operator.NEG);
94         }
95         for (int i = OP_ishl; i <= OP_lushr; i++) {
96           table[i] = ShiftInstruction.make(indexedTypes[(i - OP_ishl) % 2], ShiftInstruction.Operator.values()[(i - OP_ishl) / 2]);
97         }
98         for (int i = OP_iand; i <= OP_lxor; i++) {
99           table[i] = BinaryOpInstruction.make(indexedTypes[(i - OP_iand) % 2],
100               BinaryOpInstruction.Operator.values()[BinaryOpInstruction.Operator.AND.ordinal() + (i - OP_iand) / 2]);
101         }
102     
103         for (int i = OP_i2l; i <= OP_d2f; i++) {
104           table[i] = ConversionInstruction.make(indexedTypes[(i - OP_i2l) / 3], indexedTypes[skip((i - OP_i2l) % 3, (i - OP_i2l) / 3)]);
105         }
106         for (int i = OP_i2b; i <= OP_i2s; i++) {
107           table[i] = ConversionInstruction.make(TYPE_int, indexedTypes[5 + (i - OP_i2b)]);
108         }
109     
110         table[OP_lcmp] = ComparisonInstruction.make(TYPE_long, ComparisonInstruction.Operator.CMP);
111         for (int i = OP_fcmpl; i <= OP_dcmpg; i++) {
112           table[i] = ComparisonInstruction.make(indexedTypes[2 + (i - OP_fcmpl) / 2],
113               ComparisonInstruction.Operator.values()[ComparisonInstruction.Operator.CMPL.ordinal() + (i - OP_fcmpl) % 2]);
114         }
115     
116         for (int i = OP_ireturn; i <= OP_areturn; i++) {
117           table[i] = ReturnInstruction.make(indexedTypes[i - OP_ireturn]);
118         }
119         table[OP_return] = ReturnInstruction.make(TYPE_void);
120     
121         table[OP_athrow] = ThrowInstruction.make(false);
122     
123         table[OP_monitorenter] = MonitorInstruction.make(true);
124         table[OP_monitorexit] = MonitorInstruction.make(false);
125     
126         table[OP_arraylength] = ArrayLengthInstruction.make();
127     
128         return table;
129       }

176       private int decodeShort(int index) {
177         return (code[index] << 8) | (code[index + 1] & 0xFF);
178       }
179     
180       private int decodeUShort(int index) {
181         return ((code[index] & 0xFF) << 8) | (code[index + 1] & 0xFF);
182       }
183     
184       private int decodeInt(int index) {
185         return (code[index] << 24) | ((code[index + 1] & 0xFF) << 16) | ((code[index + 2] & 0xFF) << 8) | (code[index + 3] & 0xFF);
186       }

197       private static int elemCount(byte[] stack, int stackPtr) throws InvalidBytecodeException {
198         if (stackPtr < 0) {
199           throw new InvalidBytecodeException("Stack underflow");
200         }
201     
202         if (stack[stackPtr] == 2) {
203           return 1;
204         } else {
205           if (stackPtr < 1) {
206             throw new InvalidBytecodeException("Stack underflow");
207           }
208     
209           if (stack[stackPtr - 1] != 1) {
210             throw new InvalidBytecodeException("Trying to manipulate a pair of " + "one-word items but one of them is two words");
211           }
212     
213           return 2;
214         }
215       }

217       private String getPrimitiveType(int t) throws InvalidBytecodeException {
218         switch (t) {
219         case T_BOOLEAN:
220           return TYPE_boolean;
221         case T_CHAR:
222           return TYPE_char;
223         case T_FLOAT:
224           return TYPE_float;
225         case T_DOUBLE:
226           return TYPE_double;
227         case T_BYTE:
228           return TYPE_byte;
229         case T_SHORT:
230           return TYPE_short;
231         case T_INT:
232           return TYPE_int;
233         case T_LONG:
234           return TYPE_long;
235         default:
236           throw new InvalidBytecodeException("Unknown primitive type " + t);
237         }

257       private boolean doesSubroutineReturn(int sub) {
258         for (int j = 0; j < retInfo.length; j++) {
259           if (retInfo[j] != null && retInfo[j].sub == sub) {
260             return true;
261           }
262         }
263         return false;

266       private int findReturnToVar(int v, int addr, boolean[] visited) throws InvalidBytecodeException {
267         while (true) {
268           if (visited[addr]) {
269             return 0;
270           } else if (retInfo[addr] != null && retInfo[addr].retVar == v) {
271             return addr;
272           } else {
273             int offset = decodedOffset[addr];
274     
275             if (offset == UNSEEN) {
276               return 0;
277             }
278     
279             int size = decodedSize[addr];
280             Instruction instr = null;
281     
282             visited[addr] = true;
283     
284             for (int j = 0; j < rawHandlers.length; j += 4) {
285               if (rawHandlers[j] <= addr && addr < rawHandlers[j + 1]) {
286                 int handlerAddr = rawHandlers[j + 2];
287     
288                 if (decodedOffset[handlerAddr] < 0) {
289                   byte[] stackWords = new byte[code.length * 2];
290                   // the bottom element on the stack must be a return address.
291                   stackWords[0] = 1;
292                   decodeAt(handlerAddr, 1, stackWords);
293                 }
294     
295                 int r = findReturnToVar(v, handlerAddr, visited);
296                 if (r != 0) {
297                   return r;
298                 }
299               }
300             }
301     
302             // If there's a JSR here, see if it ever returns. If it does not
303             // then this instruction does not fall through and so we should
304             // stop searching.
305             if (JSRs[addr] != 0) {
306               if (!doesSubroutineReturn(JSRs[addr])) {
307                 return 0;
308               }
309             } else {
310               for (int j = 0; j < size; j++) {
311                 instr = decoded.get(offset + j);
312                 if (instr instanceof StoreInstruction && ((StoreInstruction) instr).getVarIndex() == v) {
313                   return 0;
314                 }
315     
316                 int[] targets = instr.getBranchTargets();
317                 for (int k = 0; k < targets.length; k++) {
318                   if (targets[k] >= 0) {
319                     int r = findReturnToVar(v, targets[k], visited);
320                     if (r != 0) {
321                       return r;
322                     }
323                   }
324                 }
325               }
326     
327               if (instr != null && !instr.isFallThrough()) {
328                 return 0;
329               }
330             }
331     
332             do {
333               addr++;
334             } while (decodedOffset[addr] == INSIDE_INSTRUCTION);
335           }
336         }
337       }

339       /**
340        * Locate an instruction that returns from this subroutine; return 0 if one cannot be found.
341        */
342       private int findReturn(int subAddr) throws InvalidBytecodeException {
343         if (decodedSize[subAddr] < 1) {
344           throw new InvalidBytecodeException("Subroutine at " + subAddr + " does not start with an astore or pop instruction");
345         }
346         Instruction instr = decoded.get(decodedOffset[subAddr]);
347     
348         if (instr instanceof PopInstruction) {
349           // this subroutine can't return
350           return 0;
351         }
352     
353         if (!(instr instanceof StoreInstruction)) {
354           throw new InvalidBytecodeException("Subroutine at " + subAddr + " does not start with an astore or pop instruction");
355         }
356     
357         int localIndex = ((StoreInstruction) instr).getVarIndex();
358     
359         do {
360           subAddr++;
361         } while (decodedOffset[subAddr] == INSIDE_INSTRUCTION);
362     
363         return findReturnToVar(localIndex, subAddr, new boolean[code.length]);
364       }

366       private void decodeSubroutine(int jsrAddr, int retToAddr, int subAddr, int stackLen, byte[] stackWords)
367           throws InvalidBytecodeException {
368         if (JSRs == null) {
369           JSRs = new int[code.length];
370           retInfo = new RetInfo[code.length];
371         }
372     
373         JSRs[jsrAddr] = subAddr;
374     
375         if (decodedOffset[subAddr] < 0) {
376           stackWords[stackLen] = 1;
377           stackLen++;
378     
379           decodeAt(subAddr, stackLen, stackWords);
380         }
381     
382         int retAddr = findReturn(subAddr);
383         if (retAddr > 0) {
384           RetInfo r = retInfo[retAddr];
385           r.sub = subAddr;
386           byte[] cloneStackWords = new byte[r.stackWords.length];
387           System.arraycopy(r.stackWords, 0, cloneStackWords, 0, cloneStackWords.length);
388           decodeAt(retToAddr, r.stackLen, cloneStackWords);
389         }
390       }

392       private void assignReachablesToSubroutine(int addr, int sub) {
393     
394         while (belongsToSub[addr] < 0) {
395           int size = decodedSize[addr];
396     
397           belongsToSub[addr] = sub;
398     
399           for (int j = 0; j < rawHandlers.length; j += 4) {
400             if (rawHandlers[j] <= addr && addr < rawHandlers[j + 1]) {
401               assignReachablesToSubroutine(rawHandlers[j + 2], sub);
402             }
403           }
404     
405           Instruction instr = null;
406           if (size > 0 && JSRs[addr] == 0) {
407             int offset = decodedOffset[addr];
408             instr = decoded.get(offset + size - 1);
409             int[] targets = instr.getBranchTargets();
410             for (int k = 0; k < targets.length; k++) {
411               if (targets[k] >= 0) {
412                 // only chase real gotos; ignore rets
413                 assignReachablesToSubroutine(targets[k], sub);
414               }
415             }
416           }
417     
418           if (instr != null && !instr.isFallThrough()) {
419             return;
420           }
421           if (JSRs[addr] != 0 && !doesSubroutineReturn(JSRs[addr])) {
422             return;
423           }
424     
425           do {
426             addr++;
427           } while (decodedOffset[addr] < 0);
428         }
429       }
430     
431       private void assignSubroutine(int sub) {
432         assignReachablesToSubroutine(sub, sub);
433     
434         for (int i = 0; i < belongsToSub.length; i++) {
435           if (JSRs[i] > 0 && belongsToSub[i] == sub && belongsToSub[JSRs[i]] < 0) {
436             assignSubroutine(JSRs[i]);
437           }
438         }
439       }

441       private void computeSubroutineMap() {
442         belongsToSub = new int[code.length];
443     
444         for (int i = 0; i < belongsToSub.length; i++) {
445           belongsToSub[i] = -1;
446         }
447     
448         assignSubroutine(0);
449       }

451       private int decodeBytecodeInstruction(int index, int stackLen, byte[] stackWords) throws InvalidBytecodeException {
452         int opcode = code[index] & 0xFF;
453     
454         Instruction i = simpleInstructions[opcode];
455         if (i != null) {
456           decoded.add(i);
457           return index + 1;
458         }
459     
460         boolean wide = false;
461     
462         while (true) {
463           index++;
464     
465           switch (opcode) {
466           case OP_nop:
467             break;
468           case OP_bipush:
469             i = ConstantInstruction.make(code[index]);
470             index++;
471             break;
472           case OP_sipush:
473             i = ConstantInstruction.make(decodeShort(index));
474             index += 2;
475             break;
476           case OP_ldc:
477             i = makeConstantPoolLoad(code[index] & 0xFF);
478             index++;
479             break;
480           case OP_ldc_w:
481             i = makeConstantPoolLoad(decodeShort(index));
482             index += 2;
483             break;
484           case OP_ldc2_w:
485             i = makeConstantPoolLoad(decodeShort(index));
486             index += 2;
487             break;
488           case OP_iload:
489           case OP_lload:
490           case OP_fload:
491           case OP_dload:
492           case OP_aload:
493             i = LoadInstruction.make(indexedTypes[opcode - OP_iload], wide ? decodeUShort(index) : (code[index] & 0xFF));
494             index += wide ? 2 : 1;
495             break;
496           case OP_istore:
497           case OP_lstore:
498           case OP_fstore:
499           case OP_dstore:
500           case OP_astore:
501             i = StoreInstruction.make(indexedTypes[opcode - OP_istore], wide ? decodeUShort(index) : (code[index] & 0xFF));
502             index += wide ? 2 : 1;
503             break;
504           case OP_pop2:
505             i = PopInstruction.make(elemCount(stackWords, stackLen - 1));
506             break;
507           case OP_dup_x2:
508             i = DupInstruction.make(1, elemCount(stackWords, stackLen - 2));
509             break;
510           case OP_dup2:
511             i = DupInstruction.make(elemCount(stackWords, stackLen - 1), 0);
512             break;
513           case OP_dup2_x1:
514             i = DupInstruction.make(elemCount(stackWords, stackLen - 1), 1);
515             break;
516           case OP_dup2_x2:
517             i = DupInstruction.make(elemCount(stackWords, stackLen - 1), elemCount(stackWords, stackLen - 2));
518             break;
519           case OP_iinc: {
520             int v = wide ? decodeUShort(index) : (code[index] & 0xFF);
521             int c = wide ? decodeShort(index + 2) : code[index + 1];
522     
523             decoded.add(LoadInstruction.make(TYPE_int, v));
524             decoded.add(ConstantInstruction.make(c));
525             decoded.add(BinaryOpInstruction.make(TYPE_int, Operator.ADD));
526             i = StoreInstruction.make(TYPE_int, v);
527             index += wide ? 4 : 2;
528             break;
529           }
530           case OP_ifeq:
531           case OP_ifne:
532           case OP_iflt:
533           case OP_ifle:
534           case OP_ifgt:
535           case OP_ifge:
536             decoded.add(makeZero);
537             i = ConditionalBranchInstruction.make(TYPE_int, ConditionalBranchInstruction.Operator.values()[opcode - OP_ifeq],
538                 (index - 1) + decodeShort(index));
539             index += 2;
540             break;
541           case OP_if_icmpeq:
542           case OP_if_icmpne:
543           case OP_if_icmplt:
544           case OP_if_icmple:
545           case OP_if_icmpgt:
546           case OP_if_icmpge:
547             i = ConditionalBranchInstruction.make((short) opcode, (index - 1) + decodeShort(index));
548             index += 2;
549             break;
550           case OP_if_acmpeq:
551           case OP_if_acmpne:
552             i = ConditionalBranchInstruction.make(TYPE_Object, ConditionalBranchInstruction.Operator.values()[opcode - OP_if_acmpeq],
553                 (index - 1) + decodeShort(index));
554             index += 2;
555             break;
556           case OP_goto:
557             i = GotoInstruction.make((index - 1) + decodeShort(index));
558             index += 2;
559             break;
560           case OP_jsr: {
561             index += 2;
562             break;
563           }
564           case OP_jsr_w: {
565             index += 4;
566             break;
567           }
568           case OP_ret:
569             int v = wide ? decodeUShort(index) : (code[index] & 0xFF);
570             i = GotoInstruction.make(-1 - v);
571     
572             if (retInfo == null) {
573               throw new InvalidBytecodeException("'ret' outside of subroutine");
574             }
575             retInfo[index - (wide ? 2 : 1)] = new RetInfo(-1, v, stackLen, stackWords);
576     
577             index += wide ? 2 : 1;
578             break;
579           case OP_tableswitch: {
580             int start = index - 1;
581             while ((index & 3) != 0) {
582               index++;
583             }
584             int def = start + decodeInt(index);
585             int low = decodeInt(index + 4);
586             int high = decodeInt(index + 8);
587             int[] t = new int[(high - low + 1) * 2];
588     
589             for (int j = 0; j < t.length; j += 2) {
590               t[j] = j / 2 + low;
591               t[j + 1] = start + decodeInt(index + 12 + j * 2);
592             }
593             i = SwitchInstruction.make(t, def);
594             index += 12 + (high - low + 1) * 4;
595             break;
596           }
597           case OP_lookupswitch: {
598             int start = index - 1;
599             while ((index & 3) != 0) {
600               index++;
601             }
602             int def = start + decodeInt(index);
603             int n = decodeInt(index + 4);
604             int[] t = new int[n * 2];
605     
606             for (int j = 0; j < t.length; j += 2) {
607               t[j] = decodeInt(index + 8 + j * 4);
608               t[j + 1] = start + decodeInt(index + 12 + j * 4);
609             }
610             i = SwitchInstruction.make(t, def);
611             index += 8 + n * 8;
612             break;
613           }
614           case OP_getstatic:
615           case OP_getfield: {
616             int f = decodeUShort(index);
617             i = GetInstruction.make(constantPool, f, opcode == OP_getstatic);
618             index += 2;
619             break;
620           }
621           case OP_putstatic:
622           case OP_putfield: {
623             int f = decodeUShort(index);
624             i = PutInstruction.make(constantPool, f, opcode == OP_putstatic);
625             index += 2;
626             break;
627           }
628           case OP_invokevirtual:
629           case OP_invokespecial:
630           case OP_invokestatic: {
631             int m = decodeUShort(index);
632             i = InvokeInstruction.make(constantPool, m, opcode);
633             index += 2;
634             break;
635           }
636           case OP_invokeinterface: {
637             int m = decodeUShort(index);
638             i = InvokeInstruction.make(constantPool, m, opcode);
639             index += 4;
640             break;
641           }
642           case OP_invokedynamic: {
643             int m = decodeUShort(index);
644             i = InvokeDynamicInstruction.make(constantPool, m, opcode);
645             index += 4;
646             break;
647           }
648           case OP_new:
649             i = NewInstruction.make(constantPool.getConstantPoolClassType(decodeUShort(index)), 0);
650             index += 2;
651             break;
652           case OP_newarray:
653             i = NewInstruction.make(Util.makeArray(getPrimitiveType(code[index])), 1);
654             index++;
655             break;
656           case OP_anewarray:
657             i = NewInstruction.make(Util.makeArray(constantPool.getConstantPoolClassType(decodeUShort(index))), 1);
658             index += 2;
659             break;
660           case OP_checkcast:
661             i = CheckCastInstruction.make(constantPool.getConstantPoolClassType(decodeUShort(index)));
662             index += 2;
663             break;
664           case OP_instanceof:
665             i = InstanceofInstruction.make(constantPool.getConstantPoolClassType(decodeUShort(index)));
666             index += 2;
667             break;
668           case OP_wide:
669             wide = true;
670             opcode = code[index] & 0xFF;
671             continue;
672           case OP_multianewarray:
673             i = NewInstruction.make(constantPool.getConstantPoolClassType(decodeUShort(index)), code[index + 2] & 0xFF);
674             index += 3;
675             break;
676           case OP_ifnull:
677           case OP_ifnonnull:
678             decoded.add(ConstantInstruction.make(TYPE_Object, null));
679             i = ConditionalBranchInstruction.make(TYPE_Object, ConditionalBranchInstruction.Operator.values()[opcode - OP_ifnull],
680                 (index - 1) + decodeShort(index));
681             index += 2;
682             break;
683           case OP_goto_w:
684             i = GotoInstruction.make((index - 1) + decodeInt(index));
685             index += 4;
686             break;
687           default:
688             throw new InvalidBytecodeException("Unknown opcode " + opcode);
689           }
690     
691           break;
692         }
693     
694         if (i != null) {
695           decoded.add(i);
696         }
697     
698         return index;
699       }

701       private int applyInstructionToStack(Instruction i, int stackLen, byte[] stackWords) throws InvalidBytecodeException {
702         stackLen -= i.getPoppedCount();
703     
704         if (stackLen < 0) {
705           throw new InvalidBytecodeException("Stack underflow");
706         }
707     
708         if (i instanceof DupInstruction) {
709           DupInstruction d = (DupInstruction) i;
710           int delta = d.getDelta();
711           int size = d.getSize();
712     
713           System.arraycopy(stackWords, stackLen + delta, stackWords, stackLen + size + delta, size);
714           System.arraycopy(stackWords, stackLen, stackWords, stackLen + size, delta);
715           System.arraycopy(stackWords, stackLen + size + delta, stackWords, stackLen, size);
716           stackLen += size * 2 + delta;
717         } else if (i instanceof SwapInstruction) {
718           if (stackWords[stackLen] != stackWords[stackLen + 1]) {
719             throw new Error("OP_swap must always be swapping the same size, 1");
720           }
721           stackLen += 2;
722         } else {
723           byte pushedWords = i.getPushedWordSize();
724           if (pushedWords > 0) {
725             stackWords[stackLen] = pushedWords;
726             stackLen++;
727           }
728         }
729     
730         return stackLen;
731       }

733       private void decodeAt(int index, int stackLen, byte[] stackWords) throws InvalidBytecodeException {
734         if (index < 0 || index >= decodedOffset.length) {
735           throw new InvalidBytecodeException(index, "Branch index " + index + " out of range");
736         }
737     
738         while (decodedOffset[index] < 0) {
739           int s = decoded.size();
740     
741           decodedOffset[index] = s;
742     
743           int newIndex;
744     
745           try {
746             newIndex = decodeBytecodeInstruction(index, stackLen, stackWords);
747             int instrCount = decoded.size() - s;
748     
749             decodedSize[index] = (byte) instrCount;
750     
751             // mark invalid offsets
752             for (int i = index + 1; i < newIndex; i++) {
753               decodedOffset[i] = INSIDE_INSTRUCTION;
754             }
755     
756             if (instrCount > 0) {
757               for (int i = s; i < s + instrCount; i++) {
758                 stackLen = applyInstructionToStack(decoded.get(i), stackLen, stackWords);
759               }
760     
761               Instruction instr = decoded.get(s + instrCount - 1);
762               int[] targets = instr.getBranchTargets();
763     
764               for (int i = 0; i < targets.length; i++) {
765                 int t = targets[i];
766     
767                 if (t >= 0) {
768                   decodeAt(t, stackLen, stackWords.clone());
769                 }
770               }
771     
772               if (!instr.isFallThrough()) {
773                 return;
774               }
775             } else { // possibly the jsr case
776               int jIndex = index;
777               int opcode = code[jIndex] & 0xFF;
778               if (opcode == OP_wide) {
779                 jIndex++;
780                 opcode = code[jIndex] & 0xFF;
781               }
782     
783               if (opcode == OP_jsr || opcode == OP_jsr_w) {
784                 jIndex++;
785                 int offset = opcode == OP_jsr_w ? decodeInt(jIndex) : decodeShort(jIndex);
786     
787                 decoded.add(GotoInstruction.make(0));
788                 decodedSize[index] = 1;
789     
790                 decodeSubroutine(index, newIndex, index + offset, stackLen, stackWords);
791                 return;
792               }
793             }
794           } catch (InvalidBytecodeException ex) {
795             ex.setIndex(index);
796             throw ex;
797           } catch (Error ex) {
798             System.err.println("Fatal error at index " + index);
799             throw ex;
800           } catch (RuntimeException ex) {
801             System.err.println("Fatal error at index " + index);
802             throw ex;
803           }
804     
805           index = newIndex;
806     
807           if (index >= decodedOffset.length) {
808             throw new InvalidBytecodeException(index, "Fell off end of bytecode array");
809           }
810         }
811       }

814        * This exception is thrown when the Decoder detects invalid incoming bytecode (code that would not pass the Java verifier). We
815        * don't guarantee to perform full verification in the Decoder, however.
816        */
817       public static class InvalidBytecodeException extends Exception {
818     
819         private static final long serialVersionUID = -8807125136613458111L;
820     
821         private int index;
822     
823         InvalidBytecodeException(String s) {
824           super(s);
825           index = -1;
826         }
827     
828         InvalidBytecodeException(int i, String s) {
829           super(s);
830           index = i;
831         }
832     
833         void setIndex(int i) {
834           if (index < 0) {
835             index = i;
836           }
837         }
838     
839         /**
840          * @return the offset of the bytecode instruction deemed to be invalid
841          */
842         public int getIndex() {
843           return index;
844         }
845       }

847       private ExceptionHandler[] makeHandlers(int i, int[] addrMap) {
848         int numHandlers = 0;
849         for (int j = 0; j < rawHandlers.length; j += 4) {
850           if (rawHandlers[j] <= i && i < rawHandlers[j + 1]) {
851             numHandlers++;
852           }
853         }
854     
855         return makeHandlers(i, numHandlers, addrMap);
856       }
857     
858       private ExceptionHandler[] makeHandlers(int i, int numHandlers, int[] addrMap) {
859         if (numHandlers == 0) {
860           return noHandlers;
861         } else {
862           ExceptionHandler[] hs = new ExceptionHandler[numHandlers];
863           numHandlers = 0;
864           for (int j = 0; j < rawHandlers.length; j += 4) {
865             if (rawHandlers[j] <= i && i < rawHandlers[j + 1]) {
866               int classIndex = rawHandlers[j + 3];
867               String catchClass = classIndex == 0 ? null : constantPool.getConstantPoolClassType(classIndex);
868     
869               hs[numHandlers] = new ExceptionHandler(addrMap[rawHandlers[j + 2]], catchClass);
870               numHandlers++;
871             }
872           }
873           return hs;
874         }
875       }

877       private int computeSubroutineLength(int sub) {
878         int len = 1; // extra instruction for "push null"
879     
880         for (int i = sub; i < belongsToSub.length; i++) {
881           if (belongsToSub[i] == sub) {
882             len += decodedSize[i];
883             if (JSRs[i] > 0) {
884               len += computeSubroutineLength(JSRs[i]);
885             }
886           }
887         }
888     
889         return len;
890       }

892       private int appendSubroutineCode(int callSite, int newCodeIndex, int[] callerMap) {
893         instructions[callerMap[callSite]] = GotoInstruction.make(newCodeIndex);
894     
895         instructions[newCodeIndex] = ConstantInstruction.make(TYPE_Object, null);
896         newCodeIndex++;
897     
898         int subStart = newCodeIndex;
899         int[] map = callerMap.clone();
900         int sub = JSRs[callSite];
901     
902         // emit the subroutine code
903         for (int i = sub; i < belongsToSub.length; i++) {
904           if (belongsToSub[i] == sub) {
905             int s = decodedSize[i];
906             int offset = decodedOffset[i];
907     
908             map[i] = newCodeIndex;
909     
910             for (int j = 0; j < s; j++) {
911               Instruction instr = decoded.get(offset + j);
912               instructions[newCodeIndex] = instr;
913               instructionsToBytecodes[newCodeIndex] = i;
914               newCodeIndex++;
915             }
916           }
917         }
918     
919         // fix up branch targets within emitted subroutine
920         for (int i = subStart; i < newCodeIndex; i++) {
921           IInstruction instr = instructions[i];
922           if (instr instanceof GotoInstruction && ((GotoInstruction) instr).getLabel() < 0) {
923             // fix up 'ret' instruction to branch back to return address
924             instructions[i] = GotoInstruction.make(callerMap[callSite] + 1);
925           } else {
926             instructions[i] = instr.redirectTargets(map);
927           }
928           handlers[i] = makeHandlers(instructionsToBytecodes[i], map);
929         }
930     
931         // extend handlers to cover the fake "push null"
932         handlers[subStart - 1] = handlers[subStart];
933     
934         // resolve callee subroutines
935         for (int i = sub; i < belongsToSub.length; i++) {
936           if (belongsToSub[i] == sub && JSRs[i] > 0) {
937             newCodeIndex = appendSubroutineCode(i, newCodeIndex, map);
938           }
939         }
940     
941         return newCodeIndex;
942       }

944       /**
945        * Perform the decoding.
946        * 
947        * @throws InvalidBytecodeException the incoming code is invalid and would fail Java bytecode verification
948        */
949       final public void decode() throws InvalidBytecodeException {
950         byte[] stackWords = new byte[code.length * 2];
951     
952         decoded = new ArrayList<Instruction>();
953         decodedOffset = new int[code.length];
954         for (int i = 0; i < decodedOffset.length; i++) {
955           decodedOffset[i] = UNSEEN;
956         }
957         decodedSize = new byte[code.length];
958     
959         decodeAt(0, 0, stackWords);
960         // Decode code that's only reachable through exception handlers
961         for (int i = 0; i < rawHandlers.length; i += 4) {
962           stackWords[0] = 1;
963           decodeAt(rawHandlers[i + 2], 1, stackWords);
964         }
965     
966         if (retInfo != null) {
967           computeSubroutineMap();
968           retInfo = null;
969         }
970     
971         int instructionsLen = decoded.size();
972     
973         if (belongsToSub != null) {
974           for (int i = 0; i < belongsToSub.length; i++) {
975             if (belongsToSub[i] == 0) {
976               if (JSRs[i] > 0) {
977                 instructionsLen += computeSubroutineLength(JSRs[i]);
978               }
979             } else if (belongsToSub[i] > 0) {
980               instructionsLen -= decodedSize[i];
981             }
982           }
983         }
984     
985         instructions = new Instruction[instructionsLen];
986         instructionsToBytecodes = new int[instructionsLen];
987         handlers = new ExceptionHandler[instructionsLen][];
988     
989         // shuffle decoded instructions into method order
990         int p = 0;
991         for (int i = 0; i < decodedOffset.length; i++) {
992           int offset = decodedOffset[i];
993     
994           if (offset >= 0 && (belongsToSub == null || belongsToSub[i] == 0)) {
995             decodedOffset[i] = p;
996     
997             int s = decodedSize[i];
998             for (int j = 0; j < s; j++) {
999               instructions[p] = decoded.get(offset + j);
1000               instructionsToBytecodes[p] = i;
1001               p++;
1002             }
1003           }
1004         }
1005     
1006         // fix up instructions to refer to the instruction vector instead of
1007         // bytecode offsets
1008         for (int i = 0; i < p; i++) {
1009           instructions[i] = instructions[i].redirectTargets(decodedOffset);
1010         }
1011     
1012         // emit subroutines
1013         if (JSRs != null) {
1014           for (int i = 0; i < JSRs.length; i++) {
1015             if (JSRs[i] > 0 && belongsToSub[i] == 0) {
1016               p = appendSubroutineCode(i, p, decodedOffset);
1017             }
1018           }
1019         }
1020     
1021         // generate exception handlers
1022         if (rawHandlers.length > 0) {
1023           ExceptionHandler[] hs = null;
1024           int handlersValidBefore = -1;
1025     
1026           p = 0;
1027           for (int i = 0; i < decodedOffset.length; i++) {
1028             if (decodedOffset[i] >= 0 && (belongsToSub == null || belongsToSub[i] == 0)) {
1029               if (i >= handlersValidBefore) {
1030                 // We just crossed a handler range boundary
1031                 // compute new exception handler array
1032                 int numHandlers = 0;
1033                 handlersValidBefore = Integer.MAX_VALUE;
1034                 for (int j = 0; j < rawHandlers.length; j += 4) {
1035                   if (rawHandlers[j] <= i) {
1036                     if (i < rawHandlers[j + 1]) {
1037                       numHandlers++;
1038                       handlersValidBefore = Math.min(handlersValidBefore, rawHandlers[j + 1]);
1039                     }
1040                   } else {
1041                     handlersValidBefore = Math.min(handlersValidBefore, rawHandlers[j]);
1042                   }
1043                 }
1044     
1045                 hs = makeHandlers(i, numHandlers, decodedOffset);
1046               }
1047     
1048               int s = decodedSize[i];
1049               for (int j = 0; j < s; j++) {
1050                 handlers[p] = hs;
1051                 p++;
1052               }
1053             }
1054           }
1055         } else {
1056           for (int i = 0; i < handlers.length; i++) {
1057             handlers[i] = noHandlers;
1058           }
1059         }
1060     
1061         decoded = null;
1062         decodedOffset = null;
1063         decodedSize = null;
1064         belongsToSub = null;
1065         JSRs = null;
1066       }
