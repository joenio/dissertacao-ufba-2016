96       /**
97        * Find or create the canonical MethodReference instance for the given tuple.
98        * 
99        * @param mn the name of the member
100        * @param md the descriptor of the member
101        */
102       public static synchronized MethodReference findOrCreate(TypeReference tref, Atom mn, Descriptor md) {
103         if (tref == null) {
104           throw new IllegalArgumentException("null tref");
105         }
106         Key key = new Key(tref, mn, md);
107     
108         MethodReference val = dictionary.get(key);
109         if (val != null)
110           return val;
111         val = new MethodReference(key);
112         dictionary.put(key, val);
113         return val;
114       }
115     
116       /**
117        * Find or create the canonical MethodReference instance for the given tuple.
118        * 
119        * @param tref the type reference
120        * @param selector the selector for the method
121        * @throws IllegalArgumentException if selector is null
122        */
123       public static synchronized MethodReference findOrCreate(TypeReference tref, Selector selector) {
124         if (selector == null) {
125           throw new IllegalArgumentException("selector is null");
126         }
127         return findOrCreate(tref, selector.getName(), selector.getDescriptor());
128       }
129     
130       public static MethodReference findOrCreate(TypeReference t, String methodName, String descriptor) throws IllegalArgumentException {
131         return findOrCreate(Language.JAVA, t, methodName, descriptor);
132       }
133       
134       public static MethodReference findOrCreate(Language l, TypeReference t, String methodName, String descriptor) throws IllegalArgumentException {
135         Descriptor d = Descriptor.findOrCreateUTF8(l, descriptor);
136         return findOrCreate(t, Atom.findOrCreateUnicodeAtom(methodName), d);
137       }
138       
139       public static MethodReference findOrCreate(ClassLoaderReference loader, String methodClass, String methodName,
140           String methodSignature) throws IllegalArgumentException {
141         return findOrCreate(Language.JAVA, loader, methodClass, methodName, methodSignature);
142       }
143     
144       public static MethodReference findOrCreate(Language l, ClassLoaderReference loader, String methodClass, String methodName,
145           String methodSignature) throws IllegalArgumentException {
146         TypeReference t = ShrikeUtil.makeTypeReference(loader, methodClass);
147         Atom name = Atom.findOrCreateUnicodeAtom(methodName);
148         Descriptor d = Descriptor.findOrCreateUTF8(l, methodSignature);
149         return findOrCreate(t, name, d);
150       }

152       /**
153        * @return the descriptor component of this member reference
154        */
155       public final Descriptor getDescriptor() {
156         return selector.getDescriptor();
157       }
158     
159       @Override
160       public final String toString() {
161         return "< " + getDeclaringClass().getClassLoader().getName() + ", " + getDeclaringClass().getName() + ", " + selector + " >";
162       }
163     
164       MethodReference(Key key) {
165         super(key.type, key.name, key.hashCode());
166         selector = new Selector(key.name, key.descriptor);
167         TypeName[] parameterNames = key.descriptor.getParameters();
168         if (parameterNames != null) {
169           parameterTypes = new TypeReference[parameterNames.length];
170         } else {
171           parameterTypes = null;
172         }
173         ClassLoaderReference loader = getDeclaringClass().getClassLoader();
174         if (parameterNames != null) {
175           for (int i = 0; i < parameterNames.length; i++) {
176             parameterTypes[i] = TypeReference.findOrCreate(loader, parameterNames[i]);
177           }
178         }
179     
180         TypeName r = key.descriptor.getReturnType();
181         returnType = TypeReference.findOrCreate(getDeclaringClass().getClassLoader(), r);
182       }
183     
184       /**
185        * @return return type of the method
186        */
187       public final TypeReference getReturnType() {
188         return returnType;
189       }
190     
191       /**
192        * @return ith parameter to the method. This does NOT include the implicit "this" pointer.
193        */
194       public final TypeReference getParameterType(int i) throws IllegalArgumentException {
195         if (parameterTypes == null || i >= parameterTypes.length) {
196           throw new IllegalArgumentException("illegal parameter number " + i + " for " + this);
197         }
198         return parameterTypes[i];
199       }
200     
201       public boolean isInit() {
202         return getName().equals(MethodReference.initAtom);
203       }
204     
205       /**
206        * @return something like: com.foo.bar.createLargeOrder(IILjava/lang/String;Ljava/sql/Date;)Ljava/lang/Integer;
207        */
208       @Override
209       public String getSignature() {
210         // TODO: check that we're not calling this often.
211         String s = getDeclaringClass().getName().toString().substring(1).replace('/', '.') + "." + getName() + getDescriptor();
212         return s;
213       }
214     
215       /**
216        * @return something like: createLargeOrder(IILjava.lang.String;SLjava.sql.Date;)Ljava.lang.Integer;
217        */
218       public Selector getSelector() {
219         return selector;
220       }
221     
222       /**
223        * This method does NOT include the implicit "this" parameter
224        */
225       public int getNumberOfParameters() {
226         return parameterTypes == null ? 0 : parameterTypes.length;
227       }

229       /**
230        * An identifier/selector for methods.
231        */
232       protected static class Key {
233         private final TypeReference type;
234     
235         private final Atom name;
236     
237         private final Descriptor descriptor;
238     
239         Key(TypeReference type, Atom name, Descriptor descriptor) {
240           this.type = type;
241           this.name = name;
242           this.descriptor = descriptor;
243         }
244     
245         @Override
246         public final int hashCode() {
247     
248           return 7001 * type.hashCode() + 7013 * name.hashCode() + descriptor.hashCode();
249         }
250     
251         @Override
252         public final boolean equals(Object other) {
253           assert other != null && this.getClass().equals(other.getClass());
254           Key that = (Key) other;
255           return type.equals(that.type) && name.equals(that.name) && descriptor.equals(that.descriptor);
256         }
257       }
