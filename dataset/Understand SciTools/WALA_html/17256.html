50     	private static Pair<CallGraph,PointerAnalysis<InstanceKey>> makeJavaBuilder(String scopeFile, String mainClass) throws IOException, ClassHierarchyException, IllegalArgumentException, CancelException {
51     		AnalysisScope scope = CallGraphTestUtil.makeJ2SEAnalysisScope(scopeFile, CallGraphTestUtil.REGRESSION_EXCLUSIONS);
52     		ClassHierarchy cha = ClassHierarchy.make(scope);
53     		Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha, mainClass);
54     		AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
55     		SSAPropagationCallGraphBuilder builder = Util.makeZeroCFABuilder(options, new AnalysisCache(), cha, scope);
56     		CallGraph CG = builder.makeCallGraph(options);
57     		return Pair.make(CG, builder.getPointerAnalysis());
58     	}

60     	private static Set<Pair<CGNode,CGNode>> edgeDiff(CallGraph from, CallGraph to) {
61     		Set<Pair<CGNode,CGNode>> result = HashSetFactory.make();
62     		for(CGNode f : from) {
63     			if (! f.getMethod().isSynthetic()) {
64     			outer: for(CGNode t : from) {
65     				if (!t.getMethod().isSynthetic() && from.hasEdge(f, t)) {
66     					Set<CGNode> fts = to.getNodes(f.getMethod().getReference());
67     					Set<CGNode> tts = to.getNodes(t.getMethod().getReference());
68     					for(CGNode x : fts) {
69     						for(CGNode y : tts) {
70     							if (to.hasEdge(x, y)) {
71     								continue outer;
72     							}
73     						}
74     					}
75     					result.add(Pair.make(f, t));
76     				}
77     			}
78     			}
79     		}
80     		return result;
81     	}

83     	private static void test(String mainClass, String javaScopeFile) throws ClassHierarchyException, IllegalArgumentException, IOException, CancelException, InterruptedException {
84     		Pair<CallGraph, PointerAnalysis<InstanceKey>> java = makeJavaBuilder(javaScopeFile, mainClass);
85     
86     		AnalysisScope javaScope = java.fst.getClassHierarchy().getScope();
87     		String javaJarPath = getJavaJar(javaScope);
88     		File androidDex = convertJarToDex(javaJarPath);
89     		Pair<CallGraph,PointerAnalysis<InstanceKey>> android = makeDalvikCallGraph(null, null, mainClass, androidDex.getAbsolutePath());
90     	
91     		Set<MethodReference> androidMethods = applicationMethods(android.fst);
92     		Set<MethodReference> javaMethods = applicationMethods(java.fst);
93     		
94     		Iterator<Pair<CGNode, CGNode>> javaExtraEdges = edgeDiff(java.fst, android.fst).iterator();
95     		boolean fail = false;
96     		if (javaExtraEdges.hasNext()) {
97     			fail = true;
98     			Set<MethodReference> javaExtraNodes = HashSetFactory.make(javaMethods);
99     			javaExtraNodes.removeAll(androidMethods);		
100     
101     			System.err.println(Iterator2Collection.toSet(javaExtraEdges));
102     			System.err.println(javaExtraNodes);
103     			
104     			System.err.println(android.fst);
105     			
106     			for(CGNode n : android.fst) {
107     				System.err.println("### " + n);
108     				if (n.getIR() != null) {
109     					System.err.println(n.getIR());
110     				
111     					System.err.println("return: " + android.snd.getPointsToSet(new ReturnValueKey(n)));
112     					System.err.println("exceptions: " + android.snd.getPointsToSet(new ExceptionReturnValueKey(n)));					
113     					for(int i = 1; i < n.getIR().getSymbolTable().getMaxValueNumber(); i++) {
114     						LocalPointerKey x = new LocalPointerKey(n, i);
115     						OrdinalSet<InstanceKey> s = android.snd.getPointsToSet(x);
116     						if (s != null && !s.isEmpty()) {
117     							System.err.println(i + ": " + s);
118     						}
119     					}
120     				}
121     			}
122     		}
123     		
124     		Assert.assertTrue(!fail);		
125     	}
