38     package com.ibm.wala.util;
39     
40     import java.util.ArrayList;
41     import java.util.Collection;
42     import java.util.Iterator;
43     import java.util.List;
44     
45     /**
46      * Interface for defining an arbitrary predicate on {@link Object}s.
47      */
48     public abstract class Predicate<T> {
49       @SuppressWarnings("rawtypes")
50       public static final Predicate FALSE = new Predicate() {
51         @Override
52         public boolean test(Object o) {
53           return false;
54         }
55       };
56     
57       @SuppressWarnings("rawtypes")
58       public static final Predicate TRUE = FALSE.not();
59     
60       @SuppressWarnings("unchecked")
61       public static <T> Predicate<T> truePred() {
62         return TRUE;
63       }
64     
65       @SuppressWarnings("unchecked")
66       public static <T> Predicate<T> falsePred() {
67         return FALSE;
68       }
69     
70       /** Test whether an {@link Object} satisfies this {@link Predicate} */
71       public abstract boolean test(T t);
72     
73       /** Return a predicate that is a negation of this predicate */
74       public Predicate<T> not() {
75         final Predicate<T> originalPredicate = this;
76         return new Predicate<T>() {
77           @Override
78           public boolean test(T t) {
79             return !originalPredicate.test(t);
80           }
81         };
82       }
83     
84       /**
85        * Return a predicate that is a conjunction of this predicate and another predicate
86        */
87       public Predicate<T> and(final Predicate<T> conjunct) {
88         final Predicate<T> originalPredicate = this;
89         return new Predicate<T>() {
90           @Override
91           public boolean test(T t) {
92             return originalPredicate.test(t) && conjunct.test(t);
93           }
94         };
95       }
96     
97       /**
98        * Return a predicate that is a conjunction of this predicate and another predicate
99        */
100       public Predicate<T> or(final Predicate<T> disjunct) {
101         final Predicate<T> originalPredicate = this;
102         return new Predicate<T>() {
103           @Override
104           public boolean test(T t) {
105             return originalPredicate.test(t) || disjunct.test(t);
106           }
107         };
108       }
109       
110       /**
111        * Create the predicate "is an element of c"
112        */
113       public static <T> Predicate<T> isElementOf(final Collection<T> c) {
114         return new Predicate<T>() {
115           @Override
116           public boolean test(T t) {
117             return c.contains(t);
118           }
119         };
120       }
121       
122       /**
123        * Filter a collection: generate a new list from an existing collection, consisting of the elements satisfying some predicate.
124        * 
125        * @throws IllegalArgumentException if src == null
126        */
127       public static <T> List<T> filter(Iterator<T> src, Predicate<T> pred) throws IllegalArgumentException {
128         if (src == null) {
129           throw new IllegalArgumentException("src == null");
130         }
131         ArrayList<T> result = new ArrayList<T>();
132         for (; src.hasNext();) {
133           T curElem = src.next();
134           if (pred.test(curElem))
135             result.add(curElem);
136         }
137         return result;
138       }
139     }
