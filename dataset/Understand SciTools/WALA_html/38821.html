33     /**
34      * Compute mod-ref information limited to accesses of lexical variables.
35      * 
36      */
37     public class LexicalModRef {
38     
39       public static LexicalModRef make(CallGraph cg, PointerAnalysis<InstanceKey> pa) {
40         return new LexicalModRef(cg, pa);
41       }
42     
43       private final CallGraph cg;
44     
45       private final PointerAnalysis<InstanceKey> pa;
46     
47       protected LexicalModRef(CallGraph cg, PointerAnalysis<InstanceKey> pa) {
48         this.cg = cg;
49         this.pa = pa;
50       }
51     
52       /**
53        * Compute the lexical variables possibly read by each {@link CGNode} and its
54        * transitive callees. A lexical variable is represented as a pair (C,N),
55        * where C is the defining {@link CGNode} and N is the {@link String} name.
56        */
57       public Map<CGNode, OrdinalSet<Pair<CGNode, String>>> computeLexicalRef() {
58         Map<CGNode, Collection<Pair<CGNode, String>>> scan = CallGraphTransitiveClosure.collectNodeResults(cg,
59             new Function<CGNode, Collection<Pair<CGNode, String>>>() {
60     
61               @Override
62               public Collection<Pair<CGNode, String>> apply(CGNode n) {
63                 return scanNodeForLexReads(n);
64               }
65             });
66         return CallGraphTransitiveClosure.transitiveClosure(cg, scan);
67       }
68     
69       /**
70        * Compute the lexical variables possibly modified by each {@link CGNode} and
71        * its transitive callees. A lexical variable is represented as a pair (C,N),
72        * where C is the defining {@link CGNode} and N is the {@link String} name.
73        */
74       public Map<CGNode, OrdinalSet<Pair<CGNode, String>>> computeLexicalMod() {
75         Map<CGNode, Collection<Pair<CGNode, String>>> scan = CallGraphTransitiveClosure.collectNodeResults(cg,
76             new Function<CGNode, Collection<Pair<CGNode, String>>>() {
77     
78               @Override
79               public Collection<Pair<CGNode, String>> apply(CGNode n) {
80                 return scanNodeForLexWrites(n);
81               }
82             });
83         return CallGraphTransitiveClosure.transitiveClosure(cg, scan);
84       }
85     
86       protected Collection<Pair<CGNode, String>> scanNodeForLexReads(CGNode n) {
87         Collection<Pair<CGNode, String>> result = HashSetFactory.make();
88         IR ir = n.getIR();
89         if (ir != null) {
90           for (SSAInstruction instr : Iterator2Iterable.make(ir.iterateNormalInstructions())) {
91             if (instr instanceof AstLexicalRead) {
92               AstLexicalRead read = (AstLexicalRead) instr;
93               for (Access a : read.getAccesses()) {
94                 Pair<String, String> nameAndDefiner = a.getName();
95                 result.addAll(getNodeNamePairsForAccess(n, nameAndDefiner));
96               }
97             }
98           }
99         }
100         return result;
101       }
102     
103       protected Collection<Pair<CGNode, String>> scanNodeForLexWrites(CGNode n) {
104         Collection<Pair<CGNode, String>> result = HashSetFactory.make();
105         IR ir = n.getIR();
106         if (ir != null) {
107           for (SSAInstruction instr : Iterator2Iterable.make(ir.iterateNormalInstructions())) {
108             if (instr instanceof AstLexicalWrite) {
109               AstLexicalWrite write = (AstLexicalWrite) instr;
110               for (Access a : write.getAccesses()) {
111                 Pair<String, String> nameAndDefiner = a.getName();
112                 result.addAll(getNodeNamePairsForAccess(n, nameAndDefiner));
113               }
114             }
115           }
116         }
117         return result;
118       }
119     
120       private Collection<Pair<CGNode, String>> getNodeNamePairsForAccess(CGNode n, Pair<String, String> nameAndDefiner) {
121         Collection<Pair<CGNode, String>> result = HashSetFactory.make();
122         // use scope-mapping instance keys in pointer analysis. may need a different
123         // scheme for CG construction not based on pointer analysis
124         OrdinalSet<InstanceKey> functionValues = pa.getPointsToSet(pa.getHeapModel().getPointerKeyForLocal(n, 1));
125         for (InstanceKey ik : functionValues) {
126           if (ik instanceof ScopeMappingInstanceKey) {
127             ScopeMappingInstanceKey smik = (ScopeMappingInstanceKey) ik;
128             for (CGNode definerNode : Iterator2Iterable.make(smik.getFunargNodes(nameAndDefiner))) {
129               result.add(Pair.make(definerNode, nameAndDefiner.fst));
130             }
131           }
132         }
133         return result;
134       }
135     }
