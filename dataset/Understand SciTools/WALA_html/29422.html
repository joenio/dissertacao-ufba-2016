11     package com.ibm.wala.shrikeBT;
12     
13     /**
14      * This class represents the ?aload instructions.
15      */
16     final public class ArrayLoadInstruction extends Instruction implements IArrayLoadInstruction {
17       protected ArrayLoadInstruction(short opcode) {
18         super(opcode);
19       }
20     
21       private final static ArrayLoadInstruction[] preallocated = preallocate();
22     
23       private static ArrayLoadInstruction[] preallocate() {
24         ArrayLoadInstruction[] r = new ArrayLoadInstruction[OP_saload - OP_iaload + 2];
25         for (short i = OP_iaload; i <= OP_saload; i++) {
26           r[i - OP_iaload] = new ArrayLoadInstruction(i);
27         }
28         r[OP_saload - OP_iaload + 1] = r[OP_baload - OP_iaload];
29         return r;
30       }
31     
32       public static ArrayLoadInstruction make(String type) throws IllegalArgumentException {
33         int i = Util.getTypeIndex(type);
34         if (i < 0 || i > TYPE_boolean_index) {
35           throw new IllegalArgumentException("Invalid type " + type + " for ArrayLoadInstruction");
36         }
37         return preallocated[i];
38       }
39     
40       @Override
41       public boolean equals(Object o) {
42         if (o instanceof ArrayLoadInstruction) {
43           ArrayLoadInstruction i = (ArrayLoadInstruction) o;
44           return i.opcode == opcode;
45         } else {
46           return false;
47         }
48       }
49     
50       @Override
51       public int hashCode() {
52         return opcode + 9109101;
53       }
54     
55       @Override
56       public int getPoppedCount() {
57         return 2;
58       }
59     
60       @Override
61       public String toString() {
62         return "ArrayLoad(" + getType() + ")";
63       }
64     
65       @Override
66       public String getPushedType(String[] types) throws IllegalArgumentException {
67     
68         if (types == null) {
69           return getType();
70         } else {
71           if (types.length <= 1) {
72             throw new IllegalArgumentException("types.length <= 1");
73           }
74           String t = types[1];
75           if (t == null) {
76             throw new IllegalArgumentException("types[1] cannot be null");
77           }
78           if (t.startsWith("[")) {
79             return t.substring(1);
80           } else if (t.equals(TYPE_null)) {
81             return TYPE_null;
82           } else {
83             return TYPE_unknown;
84           }
85         }
86       }
87     
88       @Override
89       public byte getPushedWordSize() {
90         return Util.getWordSize(getType());
91       }
92     
93       @Override
94       public String getType() {
95         return Constants.indexedTypes[opcode - OP_iaload];
96       }
97     
98       @Override
99       public void visit(IInstruction.Visitor v) throws NullPointerException {
100         v.visitArrayLoad(this);
101       }
102     
103       @Override
104       public boolean isPEI() {
105         return true;
106       }
107     
108       /**
109        * Java bytecode does not permit this. 
110        * @see com.ibm.wala.shrikeBT.IMemoryOperation#isAddressOf()
111        */
112       @Override
113       public boolean isAddressOf() {
114         return false;
115       }

117     }
