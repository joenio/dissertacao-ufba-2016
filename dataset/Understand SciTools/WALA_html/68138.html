147         public void attachActivities(Set<? extends SSAValue> activities, SSAValue application, SSAValue thread, SSAValue context,
148                 SSAValue iBinderToken, SSAValue intent) {
149             // call: final void Activity.attach(Context context, ActivityThread aThread, Instrumentation instr, IBinder token,
150             //    Application application, Intent intent, ActivityInfo info, CharSequence title, 
151             //    Activity parent, String id, Object lastNonConfigurationInstance,
152             //    Configuration config)
153             
154             final SSAValue nullInstrumentation;
155             {
156                 nullInstrumentation = pm.getUnmanaged(AndroidTypes.Instrumentation, "nullInstrumentation");
157                 this.redirect.addConstant(nullInstrumentation.getNumber(), new ConstantValue(null));
158                 nullInstrumentation.setAssigned();
159             }
160             final SSAValue nullInfo;
161             {
162                 nullInfo = pm.getUnmanaged(AndroidTypes.ActivityInfo, "nullInfo");
163                 this.redirect.addConstant(nullInfo.getNumber(), new ConstantValue(null));
164                 nullInfo.setAssigned();
165             }
166             final SSAValue title;
167             {
168                 title = pm.getUnmanaged(TypeReference.JavaLangString, "title");   // XXX CharSequence
169                 this.redirect.addConstant(title.getNumber(), new ConstantValue("title"));
170                 title.setAssigned();
171             }
172             final SSAValue nullParent;
173             {
174                 nullParent = pm.getUnmanaged(AndroidTypes.Activity, "nullParent");   
175                 this.redirect.addConstant(nullParent.getNumber(), new ConstantValue(null));
176                 nullParent.setAssigned();
177             }
178             final SSAValue nullConfigInstance;
179             {
180                 final TypeName cName = TypeName.string2TypeName("Landroid/app/Activity$NonConfigurationInstances");
181                 final TypeReference type = TypeReference.findOrCreate(com.ibm.wala.types.ClassLoaderReference.Primordial, cName);
182                 nullConfigInstance = pm.getUnmanaged(type, "noState");   
183                 this.redirect.addConstant(nullConfigInstance.getNumber(), new ConstantValue(null));
184                 nullConfigInstance.setAssigned();
185             }
186              final SSAValue nullConfiguration;
187             {
188                 nullConfiguration = pm.getUnmanaged(AndroidTypes.Configuration, "nullConfig");   
189                 this.redirect.addConstant(nullConfiguration.getNumber(), new ConstantValue(null));
190                 nullConfiguration.setAssigned();
191             }
192            
193             final Descriptor desc = Descriptor.findOrCreate(new TypeName[] {
194                     AndroidTypes.ContextName,
195                     AndroidTypes.ActivityThreadName,
196                     AndroidTypes.InstrumentationName,
197                     AndroidTypes.IBinderName,
198                     AndroidTypes.ApplicationName,
199                     AndroidTypes.IntentName,
200                     AndroidTypes.ActivityInfoName,
201                     TypeName.string2TypeName("Ljava/lang/CharSequence"),
202                     AndroidTypes.ActivityName,
203                     TypeReference.JavaLangString.getName(),
204                     TypeName.string2TypeName("Landroid/app/Activity$NonConfigurationInstances"),
205                     AndroidTypes.ConfigurationName }, TypeReference.VoidName);
206             final Selector mSel = new Selector(Atom.findOrCreateAsciiAtom("attach"), desc);
207             final MethodReference mRef = MethodReference.findOrCreate(AndroidTypes.Activity, mSel);
208             final List<SSAValue> params = new ArrayList<SSAValue>(13);
209             params.add(null);   // activity
210             params.add(context);
211             params.add(thread);
212             params.add(nullInstrumentation);
213             params.add(iBinderToken);
214             params.add(application);
215             params.add(intent);
216             params.add(nullInfo);
217             params.add(title);
218             params.add(nullParent);
219             params.add(title);
220             params.add(nullConfigInstance);
221             params.add(nullConfiguration);
222     
223             for (final SSAValue activity: activities) {
224                 final int callPC = redirect.getNextProgramCounter();
225                 final CallSiteReference site = CallSiteReference.make(callPC, mRef, IInvokeInstruction.Dispatch.VIRTUAL);
226                 final SSAValue exception = pm.getException();
227                 params.set(0, activity);
228                 final SSAInstruction invokation = instructionFactory.InvokeInstruction(callPC, params, exception, site);
229                 redirect.addStatement(invokation);    
230             }
231         }

243         public SSAValue fetchCallerContext() {
244             /*if (flags.contains(StarterFlags.CONTEXT_FREE)) {

248             if (caller == null) {
249                 return null;

252             final IClass iCaller = cha.lookupClass(caller);
253             final IClass iActivity = cha.lookupClass(AndroidTypes.Activity);
254             final IClass iApp = cha.lookupClass(AndroidTypes.Application);
255             final IClass iService = cha.lookupClass(AndroidTypes.Service);
256     
257             
258             final SSAValue androidContext;
259             if (caller.getName().equals(AndroidTypes.ContextWrapperName)) {
260                 this.callerContext = AndroidTypes.AndroidContextType.USELESS;
261                 return null;
262                 /*{ // Fetch ContextWrapperName.mBase => androidContext

269                     final SSAInstruction getInst = instructionFactory.GetInstruction(instPC, androidContext, self, mBaseRef);
270                     redirect.addStatement(getInst);

274                     logger.info("Caller has android-context type: ContextWrapper(ContextImpl)");

277             } else if (caller.getName().equals(AndroidTypes.ContextImplName)) {
278                 { // self is already the right context
279                     androidContext = self;
280                     this.callerContext = AndroidTypes.AndroidContextType.CONTEXT_IMPL;

284             } else if (cha.isAssignableFrom(iActivity, iCaller)) {

286                 androidContext = null;
287                 this.callerContext = AndroidTypes.AndroidContextType.ACTIVITY;
288                 
289                 return androidContext;
290             } else if (caller.equals(AndroidModelClass.ANDROID_MODEL_CLASS)) {
291                 // TODO: Return something useful
292                 this.callerContext = AndroidTypes.AndroidContextType.USELESS;

324         public SSAValue fetchIBinder(SSAValue androidContext) {
325             
326             final SSAValue iBinder = pm.getUnmanaged(AndroidTypes.IBinder, "foreignIBinder");
327     
328             if (flags.contains(StarterFlags.CONTEXT_FREE)) {
329                 // TODO: Can we do somethig?
330                 return null;
331             } else if (flags.contains(StarterFlags.QUENCH_PERMISSIONS)) {
332                 // If this flag is set the given asMethod has a IntentSender-Parameter
333     
334                 final Parameter intentSender = acc.firstOf(AndroidTypes.IntentSenderName);
335                 assert (intentSender != null) : "Unable to look up the IntentSender-Object";
336                 assert (intentSender.getNumber() == 2) : "The IntentSender-Object was not located at SSA-Number 2. This may be entirely " +
337                         "ok! I left this assertion to ashure the ParameterAccessor does its job right.";
338     
339                 // retreive the IBinder: IIntentSender.asBinder()
340                 final SSAValue iIntentSender = pm.getUnmanaged(AndroidTypes.IIntentSender, "iIntentSender");
341                 { // call IIntentSender IntentSender.getTarget()
342                     final int callPC = redirect.getNextProgramCounter();
343                     final Selector mSel = Selector.make("getTarget()Landroid/content/IIntentSender;");
344                     final MethodReference mRef = MethodReference.findOrCreate(AndroidTypes.IntentSender, mSel);
345                     final CallSiteReference site = CallSiteReference.make(callPC, mRef, IInvokeInstruction.Dispatch.VIRTUAL);
346                     final SSAValue exception = pm.getException();
347                     final List<SSAValue> params = new ArrayList<SSAValue>(1);
348                     params.add(intentSender);
349                     final SSAInstruction invokation = instructionFactory.InvokeInstruction(callPC, iIntentSender, params, exception, site);
350                     redirect.addStatement(invokation);
351                 }
352     
353                 { // call IBinder IIntentSender.asBinder()
354                     final int callPC = redirect.getNextProgramCounter();
355                     final Selector mSel = Selector.make("asBinder()Landroid/os/IBinder;");
356                     final MethodReference mRef = MethodReference.findOrCreate(AndroidTypes.IntentSender, mSel);
357                     final CallSiteReference site = CallSiteReference.make(callPC, mRef, IInvokeInstruction.Dispatch.VIRTUAL);

362                     redirect.addStatement(invokation);
363                 }

365                 
366                 return iBinder;
367             //} else if (caller.getName().equals(AndroidTypes.ActivityName)) {
368             } else if (this.callerContext == AndroidTypes.AndroidContextType.ACTIVITY) {
369                 
370                 // The IBinder is Activity.mMainThread.getApplicationThread()   // TODO: Verify
371                 final SSAValue mMainThread = pm.getUnmanaged(AndroidTypes.ActivityThread, "callersMainThred");
372                 { // Fetch mMainthred
373                     final int instPC = redirect.getNextProgramCounter();
374                     final FieldReference mMainThreadRef = FieldReference.findOrCreate(AndroidTypes.Activity, Atom.findOrCreateAsciiAtom("mMainThread"),
375                             AndroidTypes.ActivityThread);
376                     final SSAInstruction getInst = instructionFactory.GetInstruction(instPC, mMainThread, self, mMainThreadRef);
377                     redirect.addStatement(getInst);
378                 }
379     
380                 /*{ // DEBUG
381                     final com.ibm.wala.classLoader.IClass activityThread = cha.lookupClass(AndroidTypes.ActivityThread);
382                     assert (activityThread != null);
383                     for (com.ibm.wala.classLoader.IMethod m : activityThread.getDeclaredMethods()) {
384                         System.out.println(m);
385                     }
386                 } // */
387     
388                 { // Call getApplicationThread() on it
389                     final int callPC = redirect.getNextProgramCounter();
390                     final Selector mSel = Selector.make("getApplicationThread()Landroid/app/ActivityThread$ApplicationThread;");
391                     final MethodReference mRef = MethodReference.findOrCreate(AndroidTypes.ActivityThread, mSel);
392                     final CallSiteReference site = CallSiteReference.make(callPC, mRef, IInvokeInstruction.Dispatch.VIRTUAL);
393                     final SSAValue exception = pm.getException();
394                     final List<SSAValue> params = new ArrayList<SSAValue>(1);
395                     params.add(mMainThread);
396                     final SSAInstruction invokation = instructionFactory.InvokeInstruction(callPC, iBinder,  params, exception, site);
397                     redirect.addStatement(invokation);
398                 } // */
399             
400                 return iBinder;
401             } else if (this.callerContext == AndroidTypes.AndroidContextType.CONTEXT_IMPL) {
402                 // For bindService its mActivityToken - TODO: For the rest? 
403                 // startActivity uses mMainThread.getApplicationThread()
404     
405                 { // read mActivityToken -> iBinder
406                     
407     
408                     final FieldReference mActivityTokenRef = FieldReference.findOrCreate(AndroidTypes.ContextImpl,
409                             Atom.findOrCreateAsciiAtom("mActivityToken"), AndroidTypes.IBinder);
410                     final int instPC = redirect.getNextProgramCounter();
411                     final SSAInstruction getInst = instructionFactory.GetInstruction(instPC, iBinder, androidContext, mActivityTokenRef);
412                     redirect.addStatement(getInst);
413                 }
414             
415                 return iBinder;
416             } else if (this.callerContext == AndroidTypes.AndroidContextType.CONTEXT_BRIDGE) {
417                 // TODO: Return something useful
418                 
419                 return null;
420             } else if (caller.equals(AndroidModelClass.ANDROID_MODEL_CLASS)) {
421                 // TODO: Return something useful

517          *  Add Phi (if necessary) - not if only one from.
518          */
519         public SSAValue addPhi(List<? extends SSAValue> from) {
520             if (from.size() == 1) {
521                 return from.get(0);
522             } else {
523                 final SSAValue retVal = this.pm.getUnmanaged(from.get(0).getType(), "forPhi");
524                 final int phiPC = redirect.getNextProgramCounter();
525                 final SSAInstruction phi = instructionFactory.PhiInstruction(phiPC, retVal, from);
526                 this.redirect.addStatement(phi);
527                 return retVal;
528             }
