26     import com.ibm.wala.types.TypeReference;
27     import com.ibm.wala.util.strings.Atom;
28     
29     public class LambdaMethodTargetSelector implements MethodTargetSelector {
30     
31       private final WeakHashMap<BootstrapMethod, SummarizedMethod> summaries = new WeakHashMap<BootstrapMethod, SummarizedMethod>();
32     
33       private final MethodTargetSelector base;
34       
35       public LambdaMethodTargetSelector(MethodTargetSelector base) {
36         this.base = base;
37       }
38     
39       @Override
40       public IMethod getCalleeTarget(CGNode caller, CallSiteReference site, IClass receiver) {
41         if (! site.getDeclaredTarget().getName().equals(MethodReference.clinitName) && 
42             caller.getClassHierarchy().lookupClass(TypeReference.LambdaMetaFactory) != null &&
43             caller.getClassHierarchy().lookupClass(TypeReference.LambdaMetaFactory).equals(
44                 caller.getClassHierarchy().lookupClass(site.getDeclaredTarget().getDeclaringClass()))) 
45         {
46           SSAInvokeDynamicInstruction invoke = (SSAInvokeDynamicInstruction)caller.getIR().getCalls(site)[0];
47           
48           if (!summaries.containsKey(invoke.getBootstrap())) {
49             String cls = caller.getMethod().getDeclaringClass().getName().toString().replace("/", "$").substring(1);
50             int bootstrapIndex = invoke.getBootstrap().getIndexInClassFile();
51             MethodReference ref = 
52                 MethodReference.findOrCreate(
53                     site.getDeclaredTarget().getDeclaringClass(), 
54                     Atom.findOrCreateUnicodeAtom(site.getDeclaredTarget().getName().toString() +"$" + cls + "$" + bootstrapIndex),
55                     site.getDeclaredTarget().getDescriptor());
56             
57             MethodSummary summary = new MethodSummary(ref);
58             
59             if (site.isStatic()) {
60               summary.setStatic(true);
61             }
62             
63             int index = 0;
64             int v = site.getDeclaredTarget().getNumberOfParameters() + 2;
65             IClass lambda = LambdaSummaryClass.findOrCreate(caller, invoke);
66             SSAInstructionFactory insts = Language.JAVA.instructionFactory();
67             summary.addStatement(insts.NewInstruction(index, v, NewSiteReference.make(index, lambda.getReference())));
68             index++;
69             for(int i = 0; i < site.getDeclaredTarget().getNumberOfParameters(); i++) {
70               summary.addStatement(
71                   insts.PutInstruction(index++, v, i+1, lambda.getField(Atom.findOrCreateUnicodeAtom("c" + i)).getReference()));
72             }
73             summary.addStatement(insts.ReturnInstruction(index++, v, false));
74             
75             summaries.put(invoke.getBootstrap(), new SummarizedMethod(ref, summary, caller.getClassHierarchy().lookupClass(site.getDeclaredTarget().getDeclaringClass())));
76           }
77           
78           return summaries.get(invoke.getBootstrap());
79           
80         } else {
81           return base.getCalleeTarget(caller, site, receiver);
82         }
83       }
84     
85     }
