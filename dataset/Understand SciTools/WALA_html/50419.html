35     /**
36      * An SWT action that spawns spawns a ghostview to see the local supergraph for a procedure node which is the current selection in a
37      * tree viewer.
38      * 
39      * @author sfink
40      */
41     public class ViewIFDSLocalAction<T, P, F> extends Action {
42       /**
43        * Governing tree viewer
44        */
45       private final SWTTreeViewer viewer;
46     
47       /**
48        * Governing supergraph
49        */
50       private final ISupergraph<T, P> supergraph;
51     
52       /**
53        * name of PDF file to generate
54        */
55       private final String pdfFile;
56     
57       /**
58        * name of dot file to generate
59        */
60       private final String dotFile;
61     
62       /**
63        * path to dot.exe
64        */
65       private final String dotExe;
66     
67       /**
68        * path to pdf view executable
69        */
70       private final String pdfViewExe;

74       public ViewIFDSLocalAction(SWTTreeViewer viewer, TabulationResult<T, P, F> result, String pdfFile, String dotFile, String dotExe,
75           String pdfViewExe, NodeDecorator labels) {
76         if (result == null) {
77           throw new IllegalArgumentException("null result");
78         }
79         this.viewer = viewer;
80         this.supergraph = result.getProblem().getSupergraph();
81         this.pdfFile = pdfFile;
82         this.dotFile = dotFile;
83         this.dotExe = dotExe;
84         this.pdfViewExe = pdfViewExe;
85         this.labels = labels;
86         setText("View Local Supergraph");
87       }

89       public ViewIFDSLocalAction(SWTTreeViewer viewer, TabulationResult<T, P, F> result, String psFile, String dotFile, String dotExe,
90           String gvExe) {
91         if (result == null) {
92           throw new IllegalArgumentException("null result");
93         }
94         this.viewer = viewer;
95         this.supergraph = result.getProblem().getSupergraph();
96         this.pdfFile = psFile;
97         this.dotFile = dotFile;
98         this.dotExe = dotExe;
99         this.pdfViewExe = gvExe;
100         this.labels = new Labels<T, P, F>(result);
101         setText("View Local Supergraph");
102       }
103     
104       private static class Labels<T, P, F> implements NodeDecorator {
105         private TabulationResult<T, P, F> result;
106     
107         Labels(TabulationResult<T, P, F> result) {
108           this.result = result;
109         }
110     
111         @Override
112         @SuppressWarnings("unchecked")
113         public String getLabel(Object o) throws WalaException {
114           T t = (T) o;
115           if (t instanceof BasicBlockInContext) {
116             BasicBlockInContext bb = (BasicBlockInContext) t;
117             if (bb.getDelegate() instanceof IExplodedBasicBlock) {
118               IExplodedBasicBlock delegate = (IExplodedBasicBlock) bb.getDelegate();
119               String s = delegate.getNumber() + " " + result.getResult(t) + "\\n" + stringify(delegate.getInstruction());
120               for (Iterator<SSAPhiInstruction> phis = delegate.iteratePhis(); phis.hasNext();) {
121                 SSAPhiInstruction phi = phis.next();
122                 s += " " + phi;
123               }
124               if (delegate.isCatchBlock()) {
125                 s += " " + delegate.getCatchInstruction();
126               }
127               return s;
128             }
129           }
130           return t + " " + result.getResult(t);
131         }
132       }
133     
134       /**
135        * Print a short-ish representation of s as a String
136        */
137       public static String stringify(SSAInstruction s) {
138         if (s == null) {
139           return null;
140         }
141         if (s instanceof SSAAbstractInvokeInstruction) {
142           SSAAbstractInvokeInstruction call = (SSAAbstractInvokeInstruction) s;
143           String def = call.hasDef() ? Integer.valueOf(call.getDef()) + "=" : "";
144           String result = def + "call " + call.getDeclaredTarget().getDeclaringClass().getName().getClassName() + "."
145               + call.getDeclaredTarget().getName();
146           result += " exc:" + call.getException();
147           for (int i = 0; i < s.getNumberOfUses(); i++) {
148             result += " ";
149             result += s.getUse(i);
150           }
151           return result;
152         }
153         if (s instanceof SSAGetInstruction) {
154           SSAGetInstruction g = (SSAGetInstruction) s;
155           String fieldName = g.getDeclaredField().getName().toString();
156     
157           StringBuffer result = new StringBuffer();
158           result.append(g.getDef());
159           result.append(":=");
160           result.append(g.isStatic() ? "getstatic " : "getfield ");
161           result.append(fieldName);
162           if (!g.isStatic()) {
163             result.append(" ");
164             result.append(g.getUse(0));
165           }
166           return result.toString();
167         }
168         return s.toString();
169       }
170     
171       /*
172        * @see org.eclipse.jface.action.IAction#run()
173        */
174       @Override
175       public void run() {
176     
177         try {
178           final P proc = getProcedureForSelection();
179           Predicate<T> filter = new Predicate<T>() {
180             @Override public boolean test(T o) {
181               return supergraph.getProcOf(o).equals(proc);
182             }
183           };
184           Graph<T> localGraph = GraphSlicer.prune(supergraph, filter);
185     
186           // spawn the viewer
187           System.err.println("Spawn Viewer for " + proc);
188           DotUtil.dotify(localGraph, labels, dotFile, pdfFile, dotExe);
189     
190           if (DotUtil.getOutputType() == DotUtil.DotOutputType.PDF) {
191             PDFViewUtil.launchPDFView(pdfFile, pdfViewExe);
192           }
193         } catch (WalaException e) {
194           e.printStackTrace();
195           Assertions.UNREACHABLE();
196         }
197       }
198     
199       @SuppressWarnings("unchecked")
200       protected P getProcedureForSelection() {
201         // we assume the tree viewer's current selection is a P
202         IStructuredSelection selection = viewer.getSelection();
203         if (selection.size() != 1) {
204           throw new UnsupportedOperationException("did not expect selection of size " + selection.size());
205         }
206         P first = (P) selection.getFirstElement();
207     
208         return first;
209       }

232         return pdfFile;
233       }
