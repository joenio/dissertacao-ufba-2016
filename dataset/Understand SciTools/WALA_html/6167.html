11     package com.ibm.wala.util.io;
12     
13     import java.io.File;
14     import java.io.FileOutputStream;
15     import java.io.IOException;
16     import java.io.InputStream;
17     import java.net.URL;
18     
19     public class TemporaryFile {
20     
21       private final static String outputDir;
22     
23       static {
24         String dir = System.getProperty("java.io.tmpdir");
25     
26         while (dir.endsWith(File.separator))
27           dir = dir.substring(0, dir.length()-1);
28           
29         dir = dir + File.separator;
30         
31         outputDir = dir;
32       }
33     
34       public static File urlToFile(String fileName, URL input) throws IOException {
35         File F = new File(outputDir + File.separator + fileName);
36         return urlToFile(F , input);
37       }
38     
39       public static File urlToFile(File F, URL input) throws IOException {
40         return streamToFile(F, input.openStream());
41       }
42       
43       public static File streamToFile(File F, InputStream... inputs) throws IOException {
44         FileOutputStream output = new FileOutputStream(F);
45         
46         int read;
47         byte[] buffer = new byte[ 1024 ];
48         for(InputStream input : inputs) {
49           while ( (read = input.read(buffer)) != -1 ) {
50             output.write(buffer, 0, read);
51           }
52           input.close();
53         }
54         
55         output.close();
56         return F;
57       }
58     }
