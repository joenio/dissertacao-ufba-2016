30     /**
31      * A context selector that attempts to detect recursion beyond some depth in a
32      * base selector. If such recursion is detected, the base selector's context is
33      * replaced with {@link Everywhere#EVERYWHERE}.
34      */
35     public class RecursionBoundContextSelector implements ContextSelector {
36     
37       private final ContextSelector base;
38     
39       private final int recursionBound;
40     
41       /**
42        * the highest parameter index that we'll check . this is a HACK. ideally,
43        * given a context, we'd have some way to know all the {@link ContextKey}s
44        * that it knows about.
45        * 
46        * @see ContextKey#PARAMETERS
47        */
48       private static final int MAX_INTERESTING_PARAM = 5;
49     
50       /**
51        * @param base
52        * @param recursionBound
53        *          bound on recursion depth, with the top level of the context
54        *          returned by the base selector being depth 0. The
55        *          {@link Everywhere#EVERYWHERE} context is returned if the base
56        *          context <em>exceeds</em> this bound.
57        */
58       public RecursionBoundContextSelector(ContextSelector base, int recursionBound) {
59         this.base = base;
60         this.recursionBound = recursionBound;
61       }
62     
63       @Override
64       public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] actualParameters) {
65         Context baseContext = base.getCalleeTarget(caller, site, callee, actualParameters);
66         final boolean exceedsRecursionBound = exceedsRecursionBound(baseContext, 0);
67         if (!exceedsRecursionBound) {
68           return baseContext;
69         } else if (callee instanceof JavaScriptConstructor) {
70           // for constructors, we want to keep some basic context sensitivity to
71           // avoid horrible imprecision
72           return new CallStringContext(new CallString(site, caller.getMethod()));
73         } else {
74           // TODO somehow k-limit more smartly?
75           return Everywhere.EVERYWHERE;
76         }
77       }
78     
79       private boolean exceedsRecursionBound(Context baseContext, int curLevel) {
80         if (curLevel > recursionBound) {
81           return true;
82         }
83         // we just do a case analysis here. we might have to add cases later to
84         // account for new types of context / recursion.
85         CGNode callerNode = (CGNode) baseContext.get(ContextKey.CALLER);
86         if (callerNode != null && exceedsRecursionBound(callerNode.getContext(), curLevel + 1)) {
87           return true;
88         }
89         for (int i = 0; i < MAX_INTERESTING_PARAM; i++) {
90           FilteredPointerKey.SingleInstanceFilter filter = (SingleInstanceFilter) baseContext.get(ContextKey.PARAMETERS[i]);
91           if (filter != null) {
92             InstanceKey ik = filter.getInstance();
93             if (ik instanceof ScopeMappingInstanceKey) {
94               ik = ((ScopeMappingInstanceKey) ik).getBase();
95             }
96             if (ik instanceof InstanceKeyWithNode) {
97               if (exceedsRecursionBound(((InstanceKeyWithNode) ik).getNode().getContext(), curLevel + 1)) {
98                 return true;
99               }
100             }
101           }
102         }
103         return false;
104       }
105     
106       @Override
107       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
108         return base.getRelevantParameters(caller, site);
109       }
110     
111     }
