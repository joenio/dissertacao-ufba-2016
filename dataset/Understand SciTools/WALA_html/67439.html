34     /**
35      * A simple implementation of {@link CAstEntity} used by the {@link ClosureExtractor}.
36      * 
37      * @author mschaefer
38      *
39      */
40     class ExtractedFunction implements CAstEntity {
41     	private final String name;
42     	private String[] parms;
43     	private final ExtractionPos pos;
44     	private Rewrite r;
45     	private CAstNode root;
46     	private CAstControlFlowMap cfg;
47     	private CAstSourcePositionMap posmap;
48     	private CAstNodeTypeMap types;
49     	private Map<CAstNode, Collection<CAstEntity>> scopedEntities;
50     	
51     	public ExtractedFunction(String name, ExtractionPos pos) {
52     		this.name = name;
53     		this.pos = pos;
54     	}
55     	
56     	public void setRewrite(Rewrite r) {
57     		assert this.r == null : "Rewrite shouldn't be set more than once.";
58     		this.r = r;
59     		this.root = r.newRoot();
60     		this.cfg = r.newCfg();
61     		this.posmap = r.newPos();
62     		this.types = r.newTypes();
63     	}
64     	
65     	@Override
66       public int getKind() {
67     		return CAstEntity.FUNCTION_ENTITY;
68     	}
69     
70     	@Override
71       public String getName() {
72     		return name;
73     	}
74     
75     	@Override
76       public String getSignature() {
77     		return null;
78     	}
79     
80     	@Override
81       public String[] getArgumentNames() {
82     		computeParms();
83     		return parms;
84     	}
85     
86     	@Override
87       public CAstNode[] getArgumentDefaults() {
88     		return new CAstNode[0];
89     	}
90     
91     	@Override
92       public int getArgumentCount() {
93     		computeParms();
94     		return parms.length;
95     	}
96     	
97     	private void computeParms() {
98     		if(this.parms == null) {
99     		  ArrayList<String> parms = new ArrayList<String>();
100     		  parms.add(name);
101     		  parms.add("this");
102     		  parms.addAll(pos.getParameters());
103     			if(pos.containsThis())
104     			  parms.add(pos.getThisParmName());
105     			this.parms = parms.toArray(new String[0]);
106     		}
107     	}
108     
109     	@Override
110       public Map<CAstNode, Collection<CAstEntity>> getAllScopedEntities() {
111     		if(scopedEntities == null) {
112     			scopedEntities = HashMapFactory.make();
113     			// first, add all existing entities inside the body of this for loop
114     			for(Entry<CAstNode, Collection<CAstEntity>> e : r.newChildren().entrySet()) {
115     				if(NodePos.inSubtree(e.getKey(), root)) {
116     					Collection<CAstEntity> c = scopedEntities.get(e.getKey());
117     					if(c == null)
118     						scopedEntities.put(e.getKey(), c = HashSetFactory.make());
119     					c.addAll(e.getValue());					
120     				}
121     			}
122     			// now, add all new entities which arise from extracted nested for-in loops
123     			for(Iterator<ExtractionPos> iter=pos.getNestedLoops(); iter.hasNext();) {
124     				ExtractionPos nested_loop = iter.next();
125     				CAstNode callsite = nested_loop.getCallSite();
126     				CAstEntity scoped_entity = nested_loop.getExtractedEntity();
127     				Collection<CAstEntity> c = scopedEntities.get(callsite);
128     				if(c == null)
129     					scopedEntities.put(callsite, c = HashSetFactory.make());
130     				c.add(scoped_entity);
131     			}
132     		}
133     		return scopedEntities;
134     	}
135     
136     	@Override
137       public Iterator<CAstEntity> getScopedEntities(CAstNode construct) {
138     		if(getAllScopedEntities().containsKey(construct)) {
139     	        return getAllScopedEntities().get(construct).iterator();							
140     		} else {
141     			return EmptyIterator.instance();
142     		}
143     	}
144     
145     	@Override
146       public CAstNode getAST() {
147     		return root;
148     	}
149     
150     	@Override
151       public CAstControlFlowMap getControlFlow() {
152     		return cfg;
153     	}
154     
155     	@Override
156       public CAstSourcePositionMap getSourceMap() {
157     		return posmap;
158     	}
159     
160     	@Override
161       public Position getPosition() {
162     		return getSourceMap().getPosition(root);
163     	}
164     
165     	@Override
166       public CAstNodeTypeMap getNodeTypeMap() {
167     		return types;
168     	}
169     
170     	@Override
171       public Collection<CAstQualifier> getQualifiers() {
172     		return null;
173     	}
174     
175     	@Override
176       public CAstType getType() {
177     		return null;
178     	}
179     
180     	@Override
181       public Collection<CAstAnnotation> getAnnotations() {
182        return null;
183       }
184     
185       @Override
186     	public String toString() {
187     		return "<JS function " + name + ">";
188     	}
189     }
