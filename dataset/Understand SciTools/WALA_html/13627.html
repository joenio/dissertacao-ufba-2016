12     package com.ibm.wala.cast.js.callgraph.fieldbased.flowgraph;
13     
14     import com.ibm.wala.classLoader.IMethod;
15     import com.ibm.wala.ipa.callgraph.AnalysisCache;
16     import com.ibm.wala.ipa.cha.IClassHierarchy;
17     import com.ibm.wala.util.functions.Function;
18     
19     public class FilteredFlowGraphBuilder extends FlowGraphBuilder {
20     
21     	private final Function<IMethod, Boolean> filter;
22     	
23     	public FilteredFlowGraphBuilder(IClassHierarchy cha, AnalysisCache cache, boolean fullPointerAnalysis, Function<IMethod, Boolean> filter) {
24     		super(cha, cache, fullPointerAnalysis);
25     		this.filter = filter;
26     	}
27     
28     	@Override
29     	public void visitFunction(FlowGraph flowgraph, IMethod method) {
30     		if (filter.apply(method)) {
31     			super.visitFunction(flowgraph, method);
32     		}
33     	}
34     
35     	
36     }
