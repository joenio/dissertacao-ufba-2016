11     package com.ibm.wala.util.intset;
12     
13     
14     /**
15      * A factory for mutable shared bit vector int sets
16      */
17     public class MutableSharedBitVectorIntSetFactory implements MutableIntSetFactory<MutableSharedBitVectorIntSet> {
18     
19       private final MutableSparseIntSetFactory sparseFactory = new MutableSparseIntSetFactory();
20     
21       /*
22        * @see com.ibm.wala.util.intset.MutableIntSetFactory#make(int[])
23        */
24       @Override
25       public MutableSharedBitVectorIntSet make(int[] set) {
26         SparseIntSet s = sparseFactory.make(set);
27         return new MutableSharedBitVectorIntSet(s);
28       }
29     
30       /*
31        * @see com.ibm.wala.util.intset.MutableIntSetFactory#parse(java.lang.String)
32        */
33       @Override
34       public MutableSharedBitVectorIntSet parse(String string) throws NumberFormatException {
35         SparseIntSet s = sparseFactory.parse(string);
36         return new MutableSharedBitVectorIntSet(s);
37       }
38     
39       /*
40        * @see com.ibm.wala.util.intset.MutableIntSetFactory#makeCopy(com.ibm.wala.util.intset.IntSet)
41        */
42       @Override
43       public MutableSharedBitVectorIntSet makeCopy(IntSet x) throws IllegalArgumentException {
44         if (x == null) {
45           throw new IllegalArgumentException("x == null");
46         }
47         if (x instanceof MutableSharedBitVectorIntSet) {
48           return new MutableSharedBitVectorIntSet((MutableSharedBitVectorIntSet) x);
49         } else if (x instanceof SparseIntSet) {
50           return new MutableSharedBitVectorIntSet((SparseIntSet) x);
51         } else if (x instanceof BitVectorIntSet) {
52           return new MutableSharedBitVectorIntSet((BitVectorIntSet) x);
53         } else if (x instanceof DebuggingMutableIntSet) {
54           return new MutableSharedBitVectorIntSet(new SparseIntSet(x));
55         } else {
56           // really slow.  optimize as needed.
57           MutableSharedBitVectorIntSet result = new MutableSharedBitVectorIntSet();
58           for (IntIterator it = x.intIterator(); it.hasNext(); ) {
59             result.add(it.next());
60           }
61           return result;
62         }
63       }
64     
65       /*
66        * @see com.ibm.wala.util.intset.MutableIntSetFactory#make()
67        */
68       @Override
69       public MutableSharedBitVectorIntSet make() {
70         return new MutableSharedBitVectorIntSet();
71       }
72     
73     }
