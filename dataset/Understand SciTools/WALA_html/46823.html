37     public class JavaScriptScopeMappingInstanceKeys extends ScopeMappingInstanceKeys {
38     
39       private final IClassHierarchy cha;
40       private final IClass codeBody;
41     
42       public JavaScriptScopeMappingInstanceKeys(IClassHierarchy cha,
43     					    PropagationCallGraphBuilder builder, 
44     					    InstanceKeyFactory basic)
45       {
46         super(builder, basic);
47         this.cha = cha;
48     
49         this.codeBody = cha.lookupClass(JavaScriptTypes.CodeBody);
50       }
51     
52       protected LexicalParent[] getParents(InstanceKey base) {
53         JavaScriptMethodObject function = (JavaScriptMethodObject)
54           base.getConcreteType().getMethod(AstMethodReference.fnSelector);
55     
56         return function==null? new LexicalParent[0]: function.getParents();
57       }
58     
59       @Override
60       protected boolean needsScopeMappingKey(InstanceKey base) {
61         return 
62           cha.isSubclassOf(base.getConcreteType(), codeBody)
63                             &&
64           getParents(base).length > 0;
65       }
66     
67       @Override
68       protected Collection<CGNode> getConstructorCallers(ScopeMappingInstanceKey smik, Pair<String, String> name) {
69         // in JavaScript, the 'new' instruction is wrapped in a synthetic constructor method.  we want the 
70         // caller of that constructor method, which we obtain from the context for the constructor method
71         final Context creatorContext = smik.getCreator().getContext();
72         CGNode callerOfConstructor = (CGNode) creatorContext.get(ContextKey.CALLER);
73         Collection<CGNode> result = null;
74         if (callerOfConstructor != null) {
75           return Collections.singleton(callerOfConstructor);        
76         } else {
77           CallString cs = (CallString) creatorContext.get(CallStringContextSelector.CALL_STRING);
78           if (cs != null) {
79             IMethod[] methods = cs.getMethods();
80             assert methods.length == 1;
81             IMethod m = methods[0];
82             result = builder.getCallGraph().getNodes(m.getReference());        
83           }
84         }
85         if (result == null) {
86           IClassHierarchy cha = smik.getCreator().getClassHierarchy();
87           MethodReference ref = MethodReference.findOrCreate(JavaScriptLoader.JS, TypeReference.findOrCreate(cha.getLoaders()[0].getReference(), name.snd), AstMethodReference.fnAtomStr, AstMethodReference.fnDesc.toString());
88           final IMethod method = cha.resolveMethod(ref);
89           if (method != null) {
90             return builder.getCallGraph().getNodes(method.getReference());
91           }
92         }
93         return result;
94       }
