11     package com.ibm.wala.util.graph;
12     
13     import java.util.Collection;
14     import java.util.Iterator;
15     
16     /**
17      * An object which tracks graph nodes.
18      * 
19      * This is effectively a stripped-down collection interface. We choose to avoid using the full {@link Collection} interface, so that
20      * it takes less code to implement a new {@link NodeManager}.
21      * 
22      * @param <T> the type of nodes this {@link NodeManager} tracks.
23      */
24     public interface NodeManager<T> extends Iterable<T> {
25     
26       /**
27        * @return an {@link Iterator} of the nodes in this graph
28        */
29       @Override
30       public Iterator<T> iterator();
31     
32       /**
33        * @return the number of nodes in this graph
34        */
35       public int getNumberOfNodes();
36     
37       /**
38        * add a node to this graph
39        */
40       public void addNode(T n);
41     
42       /**
43        * remove a node from this graph
44        */
45       public void removeNode(T n) throws UnsupportedOperationException;
46     
47       /**
48        * @return true iff the graph contains the specified node
49        */
50       public boolean containsNode(T n);
51     }
