25     public class CompoundModule implements ModuleEntry, Module, SourceModule {
26       private final SourceModule[] constituents;
27       private final URL name;
28       
29       public CompoundModule(URL name, SourceModule[] constituents) {
30         this.name = name;
31         this.constituents = constituents;
32       }
33     
34       public SourceModule[] getConstituents() {
35         SourceModule[] stuff = new SourceModule[ constituents.length ];
36         System.arraycopy(constituents, 0, stuff, 0, constituents.length);
37         return stuff;
38       }
39       
40       @Override
41       public Iterator<ModuleEntry> getEntries() {
42         return new NonNullSingletonIterator<ModuleEntry>(this);
43       }
44       
45       @Override
46       public boolean isModuleFile() {
47         return false;
48       }
49     
50       @Override
51       public Module asModule() {
52         throw new UnsupportedOperationException();
53       }
54     
55       @Override
56       public String getClassName() {
57         throw new UnsupportedOperationException();
58       }
59     
60       @Override
61       public String getName() {
62         return name.toString();
63       }
64     
65       @Override
66       public URL getURL() {
67         return name;
68       }
69       
70       @Override
71       public boolean isClassFile() {
72         return false;
73       }
74     
75       @Override
76       public boolean isSourceFile() {
77         return true;
78       }
79     
80       @Override
81       public InputStream getInputStream() {
82         return new InputStream() {
83           private int index = 0;
84           private InputStream currentStream;
85           
86           @Override
87           public int read() throws IOException {
88             if (currentStream == null) {
89               if (index < constituents.length) {
90                 currentStream = constituents[index++].getInputStream();
91               } else {
92                 return -1;
93               }
94             }
95             int b = currentStream.read();
96             if (b == -1) {
97               currentStream = null;
98               return read();
99             }
100             return b;
101           }
102         };
103       }
104     
105       public class Reader extends java.io.Reader {
106         private final List<Pair<Integer,URL>> locations = new ArrayList<Pair<Integer,URL>>();
107         private int line = 0;
108         private int index = 0;
109         private LineNumberReader currentReader;
110         private URL currentName;
111         
112         @Override
113         public int read(char[] cbuf, int off, int len) throws IOException {
114           if (currentReader == null) {
115             if (index < constituents.length) {
116               currentName = constituents[index].getURL();
117               currentReader = new LineNumberReader(new InputStreamReader(constituents[index++].getInputStream()));
118             } else {
119               return -1;
120             }
121           }
122           
123           int x;
124           if ((x = currentReader.read(cbuf, off, len)) == -1) {
125             line += currentReader.getLineNumber();
126             locations.add(Pair.make(line, currentName));
127             
128             currentReader.close();
129             currentReader = null;
130             
131             return read(cbuf, off, len);
132           }
133     
134           return x;
135         }
136     
137         @Override
138         public void close() throws IOException {
139           if (currentReader!= null) {
140             currentReader.close();
141             currentReader = null;
142           }
143         }  
144         
145         public Pair<Integer,URL> getOriginalPosition(int lineNumber) {
146           int start = 0;
147           for(int i = 0; i < locations.size(); i++) {
148             if (locations.get(i).fst >= lineNumber) {
149               return Pair.make(lineNumber - start, locations.get(i).snd);
150             } else {
151               start = locations.get(i).fst;
152             }
153           }
154           throw new IllegalArgumentException("line number " + lineNumber + " too high");
155         }
156       }
157         
158       @Override
159       public Reader getInputReader() {
160         return new Reader();
161       }
162     
163       @Override
164       public Module getContainer() {
165         // stitched together module has no single container
166         return null;
167       }
