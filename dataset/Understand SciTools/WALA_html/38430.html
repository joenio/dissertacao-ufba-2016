11     package com.ibm.wala.ipa.callgraph.propagation;
12     
13     import com.ibm.wala.fixpoint.UnaryOperator;
14     
15     /**
16      * A SideEffect is a constraint which carries a points-to-set which is def'fed or used in created constraints.
17      * 
18      * The side effect doesn't actually def or use the fixedSet itself ... rather, the side effect creates <em>new</em> constraints that
19      * def or use the fixed set.
20      * 
21      * A "load" operator generates defs of the fixed set. A "store" operator generates uses of the fixed set.
22      */
23     public abstract class UnarySideEffect extends UnaryOperator<PointsToSetVariable> {
24       private PointsToSetVariable fixedSet;
25     
26       public UnarySideEffect(PointsToSetVariable fixedSet) {
27         this.fixedSet = fixedSet;
28       }
29     
30       @Override
31       public final byte evaluate(PointsToSetVariable lhs, PointsToSetVariable rhs) {
32         return evaluate(rhs);
33       }
34     
35       public abstract byte evaluate(PointsToSetVariable rhs);
36     
37       /**
38        * @return Returns the fixed points-to-set associated with this side effect.
39        */
40       PointsToSetVariable getFixedSet() {
41         return fixedSet;
42       }
43     
44       @Override
45       public boolean equals(Object o) {
46         if (o == null) {
47           return false;
48         }
49         if (getClass().equals(o.getClass())) {
50           UnarySideEffect other = (UnarySideEffect) o;
51           return fixedSet.equals(other.fixedSet);
52         } else {
53           return false;
54         }
55       }
56     
57       @Override
58       public int hashCode() {
59         return 8059 * fixedSet.hashCode();
60       }
61     
62       /**
63        * A "load" operator generates defs of the fixed set. A "store" operator generates uses of the fixed set.
64        */
65       abstract protected boolean isLoadOperator();
66     
67       /**
68        * Update the fixed points-to-set associated with this side effect.
69        */
70       public void replaceFixedSet(PointsToSetVariable p) {
71         fixedSet = p;
72       }
73     }
