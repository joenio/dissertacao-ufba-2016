14     import com.ibm.wala.ipa.callgraph.propagation.NodeKey;

16     public class JSSyntheticParameterKey extends NodeKey {
17       public static int lengthOffset = -1;
18     
19       private final int param;
20     
21       public JSSyntheticParameterKey(CGNode node, int param) {
22         super(node);
23         this.param = param;
24       }
25     
26       @Override
27       public boolean equals(Object o) {
28         return (o instanceof JSSyntheticParameterKey) &&
29           ((JSSyntheticParameterKey)o).param == param &&
30           internalEquals(o);
31       }
32     
33       @Override
34       public int hashCode() {
35         return param*getNode().hashCode();
36       }
37     
38       @Override
39       public String toString() {
40         return "p" + param + ":" +  getNode().getMethod().getName();
41       }

43     }
