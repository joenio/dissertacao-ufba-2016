11     package com.ibm.wala.shrikeBT;
12     
13     /**
14      * This class represents instructions that convert from one primitive type to another.
15      */
16     public final class ConversionInstruction extends Instruction implements IConversionInstruction {
17       final private String fromType;
18     
19       final private String toType;
20     
21       protected ConversionInstruction(short opcode) {
22         super(opcode);
23     
24         if (opcode < OP_i2b) {
25           int k = opcode - OP_i2l;
26           toType = indexedTypes[skip(k % 3, k / 3)];
27         } else {
28           toType = indexedTypes[(opcode - OP_i2b) + TYPE_byte_index];
29         }
30     
31         if (opcode < OP_i2b) {
32           fromType = indexedTypes[(opcode - OP_i2l) / 3];
33         } else {
34           fromType = TYPE_int;
35         }
36       }
37     
38       private final static ConversionInstruction[] preallocated = preallocate();
39     
40       private static ConversionInstruction[] preallocate() {
41         ConversionInstruction[] r = new ConversionInstruction[OP_i2s - OP_i2l + 1];
42         for (short i = OP_i2l; i <= OP_i2s; i++) {
43           r[i - OP_i2l] = new ConversionInstruction(i);
44         }
45         return r;
46       }
47     
48       public static ConversionInstruction make(String fromType, String toType) throws IllegalArgumentException {
49         int from = Util.getTypeIndex(fromType);
50         int to = Util.getTypeIndex(toType);
51         if (from < 0 || from > TYPE_double_index) {
52           throw new IllegalArgumentException("Cannot convert from type " + fromType);
53         }
54         if (from == TYPE_int_index && (to >= TYPE_byte_index && to <= TYPE_short_index)) {
55           return preallocated[(OP_i2b - OP_i2l) + (to - TYPE_byte_index)];
56         } else {
57           if (to < 0 || to > TYPE_double_index) {
58             throw new IllegalArgumentException("Cannot convert from type " + fromType + " to type " + toType);
59           }
60           if (to == from) {
61             throw new IllegalArgumentException("Cannot convert from type " + fromType + " to same type");
62           }
63           return preallocated[from * 3 + (to > from ? to - 1 : to)];
64         }
65       }
66     
67       @Override
68       public int getPoppedCount() {
69         return 1;
70       }
71     
72       @Override
73       public String getFromType() {
74         return fromType;
75       }
76     
77       private static int skip(int a, int b) {
78         return a < b ? a : a + 1;
79       }
80     
81       @Override
82       public String getToType() {
83         return toType;
84       }
85     
86       @Override
87       public String getPushedType(String[] types) {
88         return getToType();
89       }
90     
91       @Override
92       public byte getPushedWordSize() {
93         return Util.getWordSize(getToType());
94       }
95     
96       @Override
97       public boolean equals(Object o) {
98         if (o instanceof ConversionInstruction) {
99           ConversionInstruction i = (ConversionInstruction) o;
100           return i.opcode == opcode;
101         } else {
102           return false;
103         }
104       }
105     
106       @Override
107       public int hashCode() {
108         return opcode * 143111;
109       }
110     
111       @Override
112       public String toString() {
113         return "Conversion(" + getFromType() + "," + getToType() + ")";
114       }
115     
116       @Override
117       public void visit(IInstruction.Visitor v) throws NullPointerException {
118         v.visitConversion(this);
119       }
120     
121       @Override
122       public boolean isPEI() {
123         return false;
124       }
125     
126       @Override
127       public boolean throwsExceptionOnOverflow() {
128         return false;
129       }
130     }
