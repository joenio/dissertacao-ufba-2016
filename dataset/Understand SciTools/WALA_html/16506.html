31     /**
32      * Eliminate dead assignments (phis) from an SSA IR.
33      */
34     public class DeadAssignmentElimination {
35     
36       private static final boolean DEBUG = false;
37     
38       /**
39        * eliminate dead phis from an ir
40        * @throws IllegalArgumentException  if ir is null
41        */
42       public static void perform(IR ir) {
43         if (ir == null) {
44           throw new IllegalArgumentException("ir is null");
45         }
46         DefUse DU = new DefUse(ir);
47         DeadValueSystem system = new DeadValueSystem(ir, DU);
48         try {
49           system.solve(null);
50         } catch (CancelException e) {
51           throw new CancelRuntimeException(e);
52         }
53         doTransformation(ir,system);
54       }
55     
56       /**
57        * Perform the transformation
58        * @param ir IR to transform
59        * @param solution dataflow solution for dead assignment elimination
60        */
61       private static void doTransformation(IR ir, DeadValueSystem solution) {
62         ControlFlowGraph cfg = ir.getControlFlowGraph();
63         for (Iterator x = cfg.iterator(); x.hasNext();) {
64           BasicBlock b = (BasicBlock) x.next();
65           if (DEBUG) {
66             System.err.println("eliminateDeadPhis: " + b);
67           }
68           if (b.hasPhi()) {
69             HashSet<SSAPhiInstruction> toRemove = HashSetFactory.make(5);
70             for (Iterator it = b.iteratePhis(); it.hasNext();) {
71               SSAPhiInstruction phi = (SSAPhiInstruction) it.next();
72               if (phi != null) {
73                 int def = phi.getDef();
74                 if (solution.isDead(def)) {
75                   if (DEBUG) {
76                     System.err.println("Will remove phi: " + phi);
77                   }
78                   toRemove.add(phi);
79                 }
80               }
81             }
82             b.removePhis(toRemove);
83           }
84         }
85       }
86     
87       /**
88        * A dataflow system which computes whether or not a value is dead
89        */
90       private static class DeadValueSystem extends DefaultFixedPointSolver<BooleanVariable> {
91     
92         /**
93          * Map: value number -> BooleanVariable isLive
94          */
95         final private Map<Integer, BooleanVariable> vars = HashMapFactory.make();
96     
97         /**
98          * set of value numbers that are trivially dead
99          */
100         final private HashSet<Integer> trivialDead = HashSetFactory.make();
101     
102         /**
103          * @param ir the IR to analyze
104          * @param DU def-use information for the IR
105          */
106         DeadValueSystem(IR ir, DefUse DU) {
107           // create a variable for each potentially dead phi instruction.
108           for (Iterator it = ir.iteratePhis(); it.hasNext();) {
109             SSAPhiInstruction phi = (SSAPhiInstruction) it.next();
110             if (phi == null) {
111               continue;
112             }
113             int def = phi.getDef();
114             if (DU.getNumberOfUses(def) == 0) {
115               // the phi is certainly dead ... record this with a dataflow fact.
116               trivialDead.add(new Integer(def));
117             } else {
118               boolean maybeDead = true;
119               for (Iterator uses = DU.getUses(def); uses.hasNext();) {
120                 SSAInstruction u = (SSAInstruction) uses.next();
121                 if (!(u instanceof SSAPhiInstruction)) {
122                   // certainly not dead
123                   maybeDead = false;
124                   break;
125                 }
126               }
127               if (maybeDead) {
128                 // perhaps the phi is dead .. create a variable
129                 BooleanVariable B = new BooleanVariable(false);
130                 vars.put(new Integer(def), B);
131               }
132             }
133           }
134     
135           // Now create dataflow equations; v is live iff any phi that uses v is live
136           for (Iterator it = vars.entrySet().iterator(); it.hasNext();) {
137             Map.Entry E = (Map.Entry) it.next();
138             Integer def = (Integer) E.getKey();
139             BooleanVariable B = (BooleanVariable) E.getValue();
140             for (Iterator uses = DU.getUses(def.intValue()); uses.hasNext();) {
141               SSAPhiInstruction u = (SSAPhiInstruction) uses.next();
142               Integer ud = new Integer(u.getDef());
143               if (trivialDead.contains(ud)) {
144                 // do nothing ... u will not keep def live
145               } else {
146                 if (!vars.keySet().contains(ud)) {
147                   // u is not potentially dead ... certainly v is live.
148                   // record this.
149                   B.set(true);
150                 } else {
151                   // maybe u is dead?
152                   // add constraint v is live if u is live.
153                   BooleanVariable U = vars.get(ud);
154                   newStatement(B, UnaryOr.instance(), U, true, false);
155                 }
156               }
157             }
158           }
159         }
160     
161         @Override
162         protected void initializeVariables() {
163           //do nothing: all variables are initialized to false (TOP), meaning "not live"
164         }
165     
166         @Override
167         protected void initializeWorkList() {
168           addAllStatementsToWorkList();
169         }
170     
171         /**
172          * @param value
173          * @return true iff there are no uses of the given value number
174          */
175         private boolean isDead(int value) {
176           Integer V = Integer.valueOf(value);
177           if (trivialDead.contains(V)) {
178             return true;
179           } else {
180             BooleanVariable B = vars.get(V);
181             if (B == null) {
182               return false;
183             } else {
184               return !B.getValue();
185             }
186           }
187         }
188     
189         @Override
190         protected BooleanVariable[] makeStmtRHS(int size) {
191           return new BooleanVariable[size];
192         }
193     
194       }
