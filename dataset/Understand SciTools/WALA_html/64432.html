14     package com.ibm.wala.cast.tree.impl;
15     
16     import java.util.Collection;
17     import java.util.HashMap;
18     
19     import com.ibm.wala.cast.tree.CAstNode;
20     import com.ibm.wala.cast.tree.CAstNodeTypeMap;
21     import com.ibm.wala.cast.tree.CAstType;
22     
23     public class CAstNodeTypeMapRecorder 
24         extends HashMap <CAstNode,CAstType>
25         implements CAstNodeTypeMap
26     {
27       private static final long serialVersionUID= 7812144102027916961L;
28     
29       @Override
30       public CAstType getNodeType(CAstNode node) {
31         return get(node);
32       }
33     
34       public void add(CAstNode node, CAstType type) {
35         put(node, type);
36       }
37     
38       @Override
39       public Collection<CAstNode> getMappedNodes() {
40         return keySet();
41       }
42     
43       public void addAll(CAstNodeTypeMap other) {
44         for(CAstNode o : other.getMappedNodes()) {
45           put(o, other.getNodeType(o));
46         }
47       }
48     }
