47     /**
48      * Class responsible for mapping Polyglot type system objects representing types,
49      * methods and fields to the corresponding WALA TypeReferences, MethodReferences
50      * and FieldReferences. Used during translation and by clients to help correlate
51      * WALA analysis results to the various AST nodes.
52      * @author rfuhrer
53      */
54     public class PolyglotIdentityMapper implements IdentityMapper<Type, CodeInstance, FieldInstance> {
55       private final Map<Type, TypeReference> fTypeMap = HashMapFactory.make();
56     
57       private final Map<FieldInstance, FieldReference> fFieldMap = HashMapFactory.make();
58     
59       private final Map<CodeInstance, MethodReference> fMethodMap = HashMapFactory.make();
60     
61       /**
62        * Map from Polyglot local ClassTypes to their enclosing methods. Used by localTypeToTypeID().<br>
63        * Needed since Polyglot doesn't provide this information. (It doesn't need to, since it
64        * doesn't need to generate unambiguous names for such entities -- it hands the source
65        * off to javac to generate bytecode. It probably also wouldn't want to, since that would
66        * create back-pointers from Type objects in the TypeSystem to AST's.)
67        */
68       protected Map<ClassType, CodeInstance> fLocalTypeMap = new LinkedHashMap<ClassType, CodeInstance>();
69     
70       protected final ClassLoaderReference fClassLoaderRef;
71     
72       public PolyglotIdentityMapper(ClassLoaderReference clr) {
73         fClassLoaderRef= clr;
74       }
75     
76       @Override
77       public FieldReference getFieldRef(FieldInstance field) {
78         if (!fFieldMap.containsKey(field)) {
79           FieldReference ref= referenceForField(field);
80           fFieldMap.put(field, ref);
81           return ref;
82         }
83         return fFieldMap.get(field);
84       }
85     
86       @Override
87       public TypeReference getTypeRef(Type type) {
88         if (!fTypeMap.containsKey(type)) {
89           TypeReference ref= referenceForType(type);
90           fTypeMap.put(type, ref);
91           return ref;
92         }
93         return fTypeMap.get(type);
94       }
95     
96       @Override
97       public MethodReference getMethodRef(CodeInstance method) {
98         if (!fMethodMap.containsKey(method)) {
99           MethodReference sel= referenceForMethod(method);
100           fMethodMap.put(method, sel);
101           return sel;
102         }
103         return fMethodMap.get(method);
104       }
105     
106       public void mapLocalAnonTypeToMethod(ClassType anonLocalType, CodeInstance owningProc) {
107         fLocalTypeMap.put(anonLocalType, owningProc);
108       }
109     
110       /**
111        * Create a FieldReference for the given Polyglot FieldInstance.<br>
112        * N.B.: This method <b>does not canonicalize</b> the FieldReferences,
113        * but rather creates a new one for each call.
114        * You more likely want to call getFieldRef(). This method is exposed
115        * so that multiple Polyglot instances can use the translation services
116        * without having FieldInstances collide (producing the dreaded "we are
117        * TypeSystem_c but type Foo is from TypeSystem_c" exception).
118        */
119       public FieldReference referenceForField(FieldInstance field) {
120         Type targetType= field.container();
121         Type fieldType= field.type();
122         TypeReference targetTypeRef= TypeReference.findOrCreate(fClassLoaderRef, typeToTypeID(targetType));
123         TypeReference fieldTypeRef= TypeReference.findOrCreate(fClassLoaderRef, typeToTypeID(fieldType));
124         Atom fieldName= Atom.findOrCreateUnicodeAtom(field.name().toString());
125         FieldReference fieldRef= FieldReference.findOrCreate(targetTypeRef, fieldName, fieldTypeRef);
126     
127         return fieldRef;
128       }
129     
130       /**
131        * Create a TypeReference for the given Polyglot Type.<br>
132        * N.B.: This method <b>does not canonicalize</b> the TypeReferences,
133        * but rather creates a new one for each call.
134        * You more likely want to call getTypeRef(). This method is exposed
135        * so that multiple Polyglot instances can use the translation services
136        * without having Types collide (producing the dreaded "we are
137        * TypeSystem_c but type Foo is from TypeSystem_c" exception).
138        */
139       public TypeReference referenceForType(Type type) {
140         TypeName typeName= TypeName.string2TypeName(typeToTypeID(type));
141         TypeReference typeRef= TypeReference.findOrCreate(fClassLoaderRef, typeName);
142     
143         return typeRef;
144       }
145     
146       private Selector selectorForMethod(CodeInstance procInstance) {
147         Atom name= 
148           (procInstance instanceof ConstructorInstance) ?
149               MethodReference.initAtom : 
150                 (procInstance instanceof InitializerInstance) ?
151                     MethodReference.clinitName : 
152                       Atom.findOrCreateUnicodeAtom(((MethodInstance) procInstance).name().toString());
153     
154         TypeName[] argTypeNames = null;
155         if (! (procInstance instanceof InitializerInstance)) {
156           List formalTypes = ((ProcedureInstance)procInstance).formalTypes();
157           int numArgs = formalTypes.size();
158           // Descriptor prefers null to an empty array
159           if (numArgs > 0) {
160             argTypeNames = new TypeName[numArgs];
161           
162             int i = 0;
163             for(Iterator iter = formalTypes.iterator(); iter.hasNext(); i++)  {
164               Type argType= (Type) iter.next();
165               argTypeNames[i]= TypeName.string2TypeName(typeToTypeID(argType));
166             }
167           }
168         }
169     
170         Type retType= 
171           (procInstance instanceof MethodInstance) ? ((MethodInstance) procInstance).returnType() : procInstance.typeSystem().Void();
172     	TypeName retTypeName= TypeName.string2TypeName(typeToTypeID(retType));
173     
174     	Descriptor desc= Descriptor.findOrCreate(argTypeNames, retTypeName);
175     
176     	return new Selector(name, desc);
177       }
178     
179       /**
180        * Create a MethodReference for the given Polyglot MethodInstance.<br>
181        * N.B.: This method <b>does not canonicalize</b> the MethodReferences,
182        * but rather creates a new one for each call.
183        * You more likely want to call getMethodRef(). This method is exposed
184        * so that multiple Polyglot instances can use the translation services
185        * without having MethodInstances collide (producing the dreaded "we are
186        * TypeSystem_c but type Foo is from TypeSystem_c" exception).
187        */
188       public MethodReference referenceForMethod(CodeInstance procInstance) {
189         // Handles both ConstructorInstance's and MethodInstance's
190         TypeName ownerType= TypeName.string2TypeName(typeToTypeID(((MemberDef) procInstance.def()).container().get()));
191         TypeReference ownerTypeRef= TypeReference.findOrCreate(fClassLoaderRef, ownerType);
192         MethodReference methodRef= MethodReference.findOrCreate(ownerTypeRef, selectorForMethod(procInstance));
193     
194         return methodRef;
195       }
196     
197       /**
198        * Translates the given Polyglot type to a name suitable for use in a WALA TypeReference
199        * (i.e. a bytecode-compliant type name).
200        */
201       public String typeToTypeID(Type type) {
202         if (type.isPrimitive()) {
203           PrimitiveType ptype= (PrimitiveType) type;
204     
205           return JavaPrimitiveTypeMap.getShortName(ptype.name().toString());
206         } else if (type.isArray()) {
207           ArrayType atype= (ArrayType) type;
208           return "[" + typeToTypeID(atype.base());
209         } else if (type.isNull()) {
210           Assertions.UNREACHABLE("typeToTypeID() encountered a null type!");
211           return null;
212         }
213         Assertions.productionAssertion(type.isClass(), "typeToTypeID() encountered the type " + type + " that is neither primitive, array, nor class!");
214     
215         ClassType ctype= (ClassType) type;
216     
217         return (ctype.isLocal() || ctype.isAnonymous()) ? anonLocalTypeToTypeID(ctype) : composeWALATypeDescriptor(ctype);
218       }
219     
220       public String anonLocalTypeToTypeID(ClassType ctype) {
221         CodeInstance procInstance= fLocalTypeMap.get(ctype);
222     
223         String outerTypeID= typeToTypeID(ctype.outer());
224         String shortName= (ctype.isAnonymous()) ? PolyglotJava2CAstTranslator.anonTypeName(ctype) : ctype.fullName().name().toString();
225     
226         return outerTypeID + '/' + getMethodRef(procInstance).getSelector() + '/' + shortName;
227       }
228     
229       public String composeWALATypeDescriptor(ClassType ctype) {
230         return "L" + composeWALATypeName(ctype);
231       }
232     
233       public String composeWALATypeName(ClassType ctype) {
234         if (ctype.package_() != null) {
235           String packageName = ctype.package_().fullName().toString();
236     
237           Assertions.productionAssertion(ctype.fullName().toString().startsWith(packageName));
238           return packageName.replace('.','/') + "/" + ctype.fullName().toString().substring( packageName.length()+1 ).replace('.','$'); 
239         } else {
240           return ctype.fullName().toString().replace('.', '$');
241         }
242       }
