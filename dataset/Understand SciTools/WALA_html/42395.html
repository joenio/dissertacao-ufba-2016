34     /**
35      * A policy telling a {@link ClosureExtractor} to extract the body of every for-in loop.
36      * 
37      * NB: This policy matches on Rhino-specific encodings of for-in loops and hence is inherently non-portable.
38      * 
39      * @author mschaefer
40      *
41      */
42     public class ForInBodyExtractionPolicy extends ExtractionPolicy {
43       public static final ForInBodyExtractionPolicy INSTANCE = new ForInBodyExtractionPolicy();
44       
45       public static final ExtractionPolicyFactory FACTORY = new ExtractionPolicyFactory() {
46         @Override
47         public ExtractionPolicy createPolicy(CAstEntity entity) {
48           return INSTANCE;
49         }
50       };
51       
52       private ForInBodyExtractionPolicy() {}
53     
54       @Override
55       public List<ExtractionRegion> extract(CAstNode node) {
56         SomeConstant loopVar = new SomeConstant();
57         
58         /* 
59          * matches Rhino 1.7.3 encoding of for-in loop bodies:
60          * 
61          *   BLOCK_STMT
62          *     decl/assign of loop variable
63          *     optional SCOPE
64          *       <loopBody>
65          */
66         if(new NodeOfKind(BLOCK_STMT,
67             new Alt(new NodeOfKind(DECL_STMT,
68                                    loopVar,
69                                    new SubtreeOfKind(EACH_ELEMENT_GET)),
70                     new NodeOfKind(ASSIGN,
71                                    new NodeOfKind(VAR, loopVar),
72                                    new SubtreeOfKind(EACH_ELEMENT_GET))),
73             new AnyNode(),
74             new NodeOfKind(BLOCK_STMT,
75               new SubtreeOfKind(LABEL_STMT),
76               new SubtreeOfKind(EMPTY))).matches(node)) {
77           List<String> parms = Collections.singletonList((String)loopVar.getLastMatch());
78           if(node.getChild(1).getKind() == LOCAL_SCOPE) {
79             return Collections.<ExtractionRegion>singletonList(new TwoLevelExtractionRegion(1, 2, 0, -1, parms, Collections.<String>emptyList()));
80           } else {
81             return Collections.singletonList(new ExtractionRegion(1, 2, parms, Collections.<String>emptyList()));
82           }
83         }
84         
85         /* matches Rhino < 1.7.3 encoding of for-in loop bodies:
86          * 
87          *   BLOCK_STMT
88          *     ASSIGN
89          *       VAR <loopVar>
90          *       EACH_ELEMENT_GET
91          *         VAR <forin_tmp>
92          *     <loopBody>
93          */
94         if(new NodeOfKind(BLOCK_STMT,
95             new NodeOfKind(ASSIGN,
96                 new NodeOfKind(VAR, loopVar),
97                 new SubtreeOfKind(EACH_ELEMENT_GET)),
98                 new AnyNode()).matches(node)) {
99           List<String> parms = Collections.singletonList((String)loopVar.getLastMatch());
100           return Collections.singletonList(new ExtractionRegion(1, 2, parms, Collections.<String>emptyList()));
101         }
102         return null;
103       }
104     
105     }
