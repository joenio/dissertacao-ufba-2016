49     package com.ibm.wala.dalvik.classLoader;
50     
51     public abstract class Literal {
52         public int value;
53     
54         public static class IntLiteral extends Literal
55         {
56             public final int value;
57             public IntLiteral(int value)
58             {
59                 this.value = value;
60             }
61         }
62         public static class LongLiteral extends Literal
63         {
64             public final long value;
65             public LongLiteral(long value)
66             {
67                 this.value = value;
68             }
69         }
70         public static class FloatLiteral extends Literal
71         {
72             public final float value;
73             public FloatLiteral(float value)
74             {
75                 this.value = value;
76             }
77         }
78         public static class DoubleLiteral extends Literal
79         {
80             public final double value;
81             public DoubleLiteral(double value)
82             {
83                 this.value = value;
84             }
85         }
86     
87     }
