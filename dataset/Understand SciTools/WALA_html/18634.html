11     package com.ibm.wala.cast.js.html;
12     
13     import java.util.ArrayList;
14     import java.util.List;
15     
16     import com.ibm.wala.cast.tree.CAstSourcePositionMap.Position;
17     
18     public class CompositeFileMapping implements FileMapping {
19       private final List<FileMapping> mappings = new ArrayList<FileMapping>(2);
20         
21       public CompositeFileMapping(FileMapping a, FileMapping b) {
22         addMapping(a);
23         addMapping(b);    
24       }
25     
26       private void addMapping(FileMapping fm) {
27         if (fm instanceof CompositeFileMapping) {
28           mappings.addAll(((CompositeFileMapping)fm).mappings);
29         } else {
30           mappings.add(fm);
31         }
32       }
33       @Override
34       public IncludedPosition getIncludedPosition(Position line) {
35         for (FileMapping fm : mappings) {
36           IncludedPosition result = fm.getIncludedPosition(line);
37           if (result != null) {
38             return result;
39           }
40         }
41         return null;
42       }
43     
44       @Override
45       public String toString() {
46         return mappings.toString();
47       }
48     }
