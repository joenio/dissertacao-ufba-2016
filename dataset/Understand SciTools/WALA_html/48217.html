11     package com.ibm.wala.util.perf;
12     
13     /**
14      * Basic class to time events
15      */
16     public class Stopwatch {
17     
18       protected int count;
19     
20       /**
21        * elapsed time in nanoseconds
22        */
23       private long elapsedTime;
24     
25       private long startTime;
26     
27       public Stopwatch() {
28       }
29     
30       public void start() {
31         startTime = System.nanoTime();
32       }
33     
34       public void stop() {
35         long endTime = System.nanoTime();
36         count++;
37         elapsedTime += (endTime - startTime);
38       }
39     
40       /**
41        * @return elapsed time in ms
42        */
43       public long getElapsedMillis() {
44         return elapsedTime / 1000000;
45       }
46       
47       /**
48        * @return number of times this stopwatch was stopped
49        */
50       public int getCount() {
51         return count;
52       }
53       
54       @Override
55       public String toString() {
56         StringBuffer sb = new StringBuffer();
57         sb.append("total: " + getElapsedMillis());
58         if (count > 0){
59           sb.append(", inv: " + count);
60           sb.append(", avg: " + getElapsedMillis()/count);
61         }
62         return sb.toString();
63       }
64     }
