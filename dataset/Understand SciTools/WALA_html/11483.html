11     package com.ibm.wala.cast.ipa.callgraph;
12     
13     import java.util.Map;
14     
15     import com.ibm.wala.classLoader.CallSiteReference;
16     import com.ibm.wala.classLoader.IClass;
17     import com.ibm.wala.classLoader.IMethod;
18     import com.ibm.wala.classLoader.Language;
19     import com.ibm.wala.ipa.callgraph.CGNode;
20     import com.ibm.wala.ipa.callgraph.MethodTargetSelector;
21     import com.ibm.wala.types.MethodReference;
22     import com.ibm.wala.util.strings.Atom;
23     
24     /**
25      *  A MethodTargetSelector implementation that supports multiple languages.
26      * It works by delegating to a language-specific child selector based on
27      * the language associated with the MethodReference for which a target is
28      * being chosen.  
29      *
30      *  This provides a simple way to combine language-specific target
31      * selection policies---such as those used for constructor calls in
32      * JavaScript and for bean methods in J2EE.
33      *
34      * @author Julian Dolby (dolby@us.ibm.com)
35      */
36     public class CrossLanguageMethodTargetSelector
37         implements MethodTargetSelector 
38     {
39     
40       private final Map<Atom,MethodTargetSelector> languageSelectors;
41     
42       public CrossLanguageMethodTargetSelector(Map<Atom, MethodTargetSelector> languageSelectors) {
43         this.languageSelectors = languageSelectors;
44       }
45     
46       private static Atom getLanguage(MethodReference target) {
47         return target
48     	.getDeclaringClass()
49     	.getClassLoader()
50     	.getLanguage();
51       }
52     
53       private static Atom getLanguage(CallSiteReference site) {
54         return getLanguage(site.getDeclaredTarget());
55       }
56     
57       private MethodTargetSelector getSelector(CallSiteReference site) {
58         return languageSelectors.get(getLanguage(site));
59       }
60     
61       @Override
62       public IMethod getCalleeTarget(CGNode caller, CallSiteReference site, IClass receiver) {
63         assert getSelector(site) != null: "no selector for " + getLanguage(site) + " method " + site;
64         return getSelector(site).getCalleeTarget(caller, site, receiver);
65       }
66     
67     }
