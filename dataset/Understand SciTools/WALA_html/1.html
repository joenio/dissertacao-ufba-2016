213         BytecodeInfo result = new BytecodeInfo();
214         result.exceptionTypes = computeDeclaredExceptions();
215     
216         if (isNative()) {
217           return result;
218         }
219         if (verbose) {
220           methodsParsed += 1;
221           if (methodsParsed % 100 == 0) {
222             System.out.println(methodsParsed + " methods processed...");
223           }
224         }
225     
226         processBytecodesWithShrikeBT(result);
227         return result;

331       private MethodReference computeMethodReference() {
332         try {
333           Atom name = Atom.findOrCreateUnicodeAtom(getMethodName());
334           ImmutableByteArray desc = ImmutableByteArray.make(getMethodSignature());
335           Descriptor D = Descriptor.findOrCreate(declaringClass.getClassLoader().getLanguage(), desc);
336           return MethodReference.findOrCreate(declaringClass.getReference(), name, D);
337         } catch (InvalidClassFileException e) {
338           Assertions.UNREACHABLE();
339           return null;
340         }
341       }

432       private void processBytecodesWithShrikeBT(BytecodeInfo info) throws InvalidClassFileException {
433         info.decoder = makeDecoder();
434         if (!isAbstract() && info.decoder == null) {
435           throw new InvalidClassFileException(-1, "non-abstract method " + getReference() + " has no bytecodes");
436         }
437         if (info.decoder == null) {
438           return;
439         }
440         info.pcMap = info.decoder.getInstructionsToBytecodes();
441     
442         processDebugInfo(info);
443     
444         SimpleVisitor simpleVisitor = new SimpleVisitor(info);
445     
446         BytecodeLanguage lang = (BytecodeLanguage) getDeclaringClass().getClassLoader().getLanguage();
447         IInstruction[] instructions = info.decoder.getInstructions();
448         for (int i = 0; i < instructions.length; i++) {
449           simpleVisitor.setInstructionIndex(i);
450           instructions[i].visit(simpleVisitor);
451           if (instructions[i].isPEI()) {
452             Collection<TypeReference> t = lang.getImplicitExceptionTypes(instructions[i]);
453             if (t != null) {
454               simpleVisitor.implicitExceptions.addAll(t);
455             }
456           }
457         }
458     
459         // copy the Set results into arrays; will use less
460         // storage
461         copyVisitorSetsToArrays(simpleVisitor, info);
462       }

464       private void copyVisitorSetsToArrays(SimpleVisitor simpleVisitor, BytecodeInfo info) {
465         info.newSites = new NewSiteReference[simpleVisitor.newSites.size()];
466         int i = 0;
467         for (Iterator<NewSiteReference> it = simpleVisitor.newSites.iterator(); it.hasNext();) {
468           info.newSites[i++] = it.next();
469         }
470     
471         info.fieldsRead = new FieldReference[simpleVisitor.fieldsRead.size()];
472         i = 0;
473         for (Iterator<FieldReference> it = simpleVisitor.fieldsRead.iterator(); it.hasNext();) {
474           info.fieldsRead[i++] = it.next();
475         }
476     
477         info.fieldsRead = new FieldReference[simpleVisitor.fieldsRead.size()];
478         i = 0;
479         for (Iterator<FieldReference> it = simpleVisitor.fieldsRead.iterator(); it.hasNext();) {
480           info.fieldsRead[i++] = it.next();
481         }
482     
483         info.fieldsWritten = new FieldReference[simpleVisitor.fieldsWritten.size()];
484         i = 0;
485         for (Iterator<FieldReference> it = simpleVisitor.fieldsWritten.iterator(); it.hasNext();) {
486           info.fieldsWritten[i++] = it.next();
487         }
488     
489         info.callSites = new CallSiteReference[simpleVisitor.callSites.size()];
490         i = 0;
491         for (Iterator<CallSiteReference> it = simpleVisitor.callSites.iterator(); it.hasNext();) {
492           info.callSites[i++] = it.next();
493         }
494     
495         info.arraysRead = new TypeReference[simpleVisitor.arraysRead.size()];
496         i = 0;
497         for (Iterator<TypeReference> it = simpleVisitor.arraysRead.iterator(); it.hasNext();) {
498           info.arraysRead[i++] = it.next();
499         }
500     
501         info.arraysWritten = new TypeReference[simpleVisitor.arraysWritten.size()];
502         i = 0;
503         for (Iterator<TypeReference> it = simpleVisitor.arraysWritten.iterator(); it.hasNext();) {
504           info.arraysWritten[i++] = it.next();
505         }
506     
507         info.implicitExceptions = new TypeReference[simpleVisitor.implicitExceptions.size()];
508         i = 0;
509         for (Iterator it = simpleVisitor.implicitExceptions.iterator(); it.hasNext();) {
510           info.implicitExceptions[i++] = (TypeReference) it.next();
511         }
512     
513         info.castTypes = new TypeReference[simpleVisitor.castTypes.size()];
514         i = 0;
515         for (Iterator<TypeReference> it = simpleVisitor.castTypes.iterator(); it.hasNext();) {
516           info.castTypes[i++] = it.next();
517         }
518     
519         info.hasMonitorOp = simpleVisitor.hasMonitorOp;
520       }

576       private class SimpleVisitor extends IInstruction.Visitor {
577     
578         private final BytecodeInfo info;
579     
580         public SimpleVisitor(BytecodeInfo info) {
581           this.info = info;
582         }
583     
584         // TODO: make a better Set implementation for these.
585         final Set<CallSiteReference> callSites = HashSetFactory.make(5);
586     
587         final Set<FieldReference> fieldsWritten = HashSetFactory.make(5);
588     
589         final Set<FieldReference> fieldsRead = HashSetFactory.make(5);
590     
591         final Set<NewSiteReference> newSites = HashSetFactory.make(5);
592     
593         final Set<TypeReference> arraysRead = HashSetFactory.make(5);
594     
595         final Set<TypeReference> arraysWritten = HashSetFactory.make(5);
596     
597         final Set<TypeReference> implicitExceptions = HashSetFactory.make(5);
598     
599         final Set<TypeReference> castTypes = HashSetFactory.make(5);

601         boolean hasMonitorOp;

603         private int instructionIndex;
604     
605         public void setInstructionIndex(int i) {
606           instructionIndex = i;
607         }
608     
609         public int getProgramCounter() throws InvalidClassFileException {
610           return info.pcMap[instructionIndex];
611         }

613         @Override
614         public void visitMonitor(MonitorInstruction instruction) {
615           hasMonitorOp = true;
616         }
617     
618         @Override
619         public void visitNew(NewInstruction instruction) {
620           ClassLoaderReference loader = getReference().getDeclaringClass().getClassLoader();
621           TypeReference t = ShrikeUtil.makeTypeReference(loader, instruction.getType());
622           try {
623             newSites.add(NewSiteReference.make(getProgramCounter(), t));
624           } catch (InvalidClassFileException e) {
625             e.printStackTrace();
626             Assertions.UNREACHABLE();
627           }
628         }
629     
630         @Override
631         public void visitGet(IGetInstruction instruction) {
632           ClassLoaderReference loader = getReference().getDeclaringClass().getClassLoader();
633           FieldReference f = FieldReference.findOrCreate(loader, instruction.getClassType(), instruction.getFieldName(), instruction
634               .getFieldType());
635           fieldsRead.add(f);
636         }
637     
638         @Override
639         public void visitPut(IPutInstruction instruction) {
640           ClassLoaderReference loader = getReference().getDeclaringClass().getClassLoader();
641           FieldReference f = FieldReference.findOrCreate(loader, instruction.getClassType(), instruction.getFieldName(), instruction
642               .getFieldType());
643           fieldsWritten.add(f);
644         }
645     
646         @Override
647         public void visitInvoke(IInvokeInstruction instruction) {
648           IClassLoader loader = getDeclaringClass().getClassLoader();
649           MethodReference m = MethodReference.findOrCreate(loader.getLanguage(), loader.getReference(), instruction.getClassType(),
650               instruction.getMethodName(), instruction.getMethodSignature());
651           int programCounter = 0;
652           try {
653             programCounter = getProgramCounter();
654           } catch (InvalidClassFileException e) {
655             e.printStackTrace();
656             Assertions.UNREACHABLE();
657           }
658           CallSiteReference site = null;
659           site = CallSiteReference.make(programCounter, m, instruction.getInvocationCode());
660           callSites.add(site);
661         }
662     
663         /*
664          * @see com.ibm.wala.shrikeBT.Instruction.Visitor#visitArrayLoad(com.ibm.wala.shrikeBT.ArrayLoadInstruction)
665          */
666         @Override
667         public void visitArrayLoad(IArrayLoadInstruction instruction) {
668           arraysRead.add(ShrikeUtil.makeTypeReference(getDeclaringClass().getClassLoader().getReference(), instruction.getType()));
669         }

808        * @throws InvalidClassFileException
809        */
810       public Set<TypeReference> getCaughtExceptionTypes() throws InvalidClassFileException {
811     
812         ExceptionHandler[][] handlers = getHandlers();
813         if (handlers == null) {
814           return Collections.emptySet();
815         }
816         HashSet<TypeReference> result = HashSetFactory.make(10);
817         ClassLoaderReference loader = getReference().getDeclaringClass().getClassLoader();
818         for (int i = 0; i < handlers.length; i++) {
819           for (int j = 0; j < handlers[i].length; j++) {
820             TypeReference t = ShrikeUtil.makeTypeReference(loader, handlers[i][j].getCatchClass());
821             if (t == null) {
822               t = TypeReference.JavaLangThrowable;
823             }
824             result.add(t);
825           }
826         }
827         return result;
828       }
