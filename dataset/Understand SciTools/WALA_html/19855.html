11     package com.ibm.wala.ipa.callgraph.propagation;
12     
13     import com.ibm.wala.fixpoint.UnaryOperator;
14     import com.ibm.wala.fixpoint.UnaryStatement;
15     
16     /**
17      * Corresponds to: "is a superset of". Used for assignment.
18      * 
19      * Unary op: <lhs>:= Assign( <rhs>)
20      * 
21      * (Technically, it's a binary op, since it includes lhs as an implicit input; this allows it to compose with other ops that define
22      * the same lhs, so long as they're all Assign ops)
23      */
24     class AssignOperator extends UnaryOperator<PointsToSetVariable> implements IPointerOperator {
25     
26       @Override
27       public UnaryStatement<PointsToSetVariable> makeEquation(PointsToSetVariable lhs, PointsToSetVariable rhs) {
28         return new AssignEquation(lhs, rhs);
29       }
30     
31       @Override
32       public byte evaluate(PointsToSetVariable lhs, PointsToSetVariable rhs) {
33     
34         if (PropagationCallGraphBuilder.DEBUG_ASSIGN) {
35           String S = "EVAL Assign " + lhs.getPointerKey() + " " + rhs.getPointerKey();
36           S = S + "\nEVAL " + lhs + " " + rhs;
37           System.err.println(S);
38         }
39         boolean changed = lhs.addAll(rhs);
40         if (PropagationCallGraphBuilder.DEBUG_ASSIGN) {
41           System.err.println("RESULT " + lhs + (changed ? " (changed)" : ""));
42         }
43     
44         return changed ? CHANGED : NOT_CHANGED;
45       }
46     
47       @Override
48       public String toString() {
49         return "Assign";
50       }
51     
52       @Override
53       public int hashCode() {
54         return 9883;
55       }
56     
57       @Override
58       public final boolean equals(Object o) {
59         // this is a singleton
60         return (this == o);
61       }
62     
63       /*
64        * @see com.ibm.wala.ipa.callgraph.propagation.IPointerOperator#isComplex()
65        */
66       @Override
67       public boolean isComplex() {
68         return false;
69       }
70     }
