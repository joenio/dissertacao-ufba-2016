89        * create a CG builder for script.  Note that the script at dir/name is loaded via the classloader, not from the filesystem.

91       public static JSCFABuilder makeScriptCGBuilder(String dir, String name, CGBuilderType builderType, ClassLoader loader) throws IOException, WalaException {
92         URL script = getURLforFile(dir, name, loader);
93         CAstRewriterFactory preprocessor = builderType.extractCorrelatedPairs ? new CorrelatedPairExtractorFactory(translatorFactory, script) : null;
94         JavaScriptLoaderFactory loaders = JSCallGraphUtil.makeLoaders(preprocessor);
95     
96         AnalysisScope scope = makeScriptScope(dir, name, loaders, loader);
97     
98         return makeCG(loaders, scope, builderType, AstIRFactory.makeDefaultFactory());
99       }

101       public static URL getURLforFile(String dir, String name, ClassLoader loader) throws IOException {
102         File f = null;
103         FileProvider provider = new FileProvider();
104         try {
105           f = provider.getFile(dir + File.separator + name, loader);
106         } catch (FileNotFoundException e) {
107           // I guess we need to do this on Windows sometimes?  --MS
108           // if this fails, we won't catch the exception
109           f = provider.getFile(dir + "/" + name, JSCallGraphBuilderUtil.class.getClassLoader());
110         }
111         return f.toURI().toURL();
112       }

172       public static JSCFABuilder makeHTMLCGBuilder(URL url, CGBuilderType builderType) throws IOException, WalaException {
173         IRFactory<IMethod> irFactory = AstIRFactory.makeDefaultFactory();
174         CAstRewriterFactory preprocessor = builderType.extractCorrelatedPairs ? new CorrelatedPairExtractorFactory(translatorFactory, url) : null;
175         JavaScriptLoaderFactory loaders = new WebPageLoaderFactory(translatorFactory, preprocessor);
176         SourceModule[] scriptsArray = makeHtmlScope(url, loaders);
177         
178         JSCFABuilder builder = makeCGBuilder(loaders, scriptsArray, builderType, irFactory);
179         if(builderType.extractCorrelatedPairs)
180           builder.setContextSelector(new PropertyNameContextSelector(builder.getAnalysisCache(), 2, builder.getContextSelector()));
181         builder.setBaseURL(url);
182         return builder;
183       }

185       public static SourceModule[] makeHtmlScope(URL url, JavaScriptLoaderFactory loaders) {
186         Set<SourceModule> scripts = HashSetFactory.make();
187         
188         JavaScriptLoader.addBootstrapFile(WebUtil.preamble);
189         scripts.add(getPrologueFile("prologue.js"));
190         scripts.add(getPrologueFile("preamble.js"));
191     
192         try {
193           scripts.addAll(WebUtil.extractScriptFromHTML(url, true).fst);
194         } catch (Error e) {
195           SourceModule dummy = new SourceURLModule(url);
196           scripts.add(dummy);
197           ((CAstAbstractLoader)loaders.getTheLoader()).addMessage(dummy, e.warning);
198         }
199             
200         SourceModule[] scriptsArray = scripts.toArray(new SourceModule[ scripts.size() ]);
201         return scriptsArray;
202       }

218       public static JSCFABuilder makeCGBuilder(JavaScriptLoaderFactory loaders, SourceModule[] scripts, CGBuilderType builderType, IRFactory<IMethod> irFactory) throws IOException, WalaException {
219         AnalysisScope scope = makeScope(scripts, loaders, JavaScriptLoader.JS);
220         return makeCG(loaders, scope, builderType, irFactory);
221       }

223       protected static JSCFABuilder makeCG(JavaScriptLoaderFactory loaders, AnalysisScope scope, CGBuilderType builderType, IRFactory<IMethod> irFactory) throws IOException, WalaException {
224         try {
225           IClassHierarchy cha = makeHierarchy(scope, loaders);
226           com.ibm.wala.cast.js.util.Util.checkForFrontEndErrors(cha);
227           Iterable<Entrypoint> roots = makeScriptRoots(cha);
228           JSAnalysisOptions options = makeOptions(scope, cha, roots);
229           options.setHandleCallApply(builderType.handleCallApply());
230           AnalysisCache cache = makeCache(irFactory);
231           JSCFABuilder builder = new JSZeroOrOneXCFABuilder(cha, options, cache, null, null, ZeroXInstanceKeys.ALLOCATIONS,
232               builderType.useOneCFA());
233           if(builderType.extractCorrelatedPairs())
234             builder.setContextSelector(new PropertyNameContextSelector(builder.getAnalysisCache(), 2, builder.getContextSelector()));
235     
236           return builder;
237         } catch (ClassHierarchyException e) {
238           Assert.assertTrue("internal error building class hierarchy", false);
239           return null;
240         }
241       }
