11     package com.ibm.wala.util.shrike;
12     
13     import java.util.HashMap;
14     
15     import com.ibm.wala.shrikeBT.BytecodeConstants;
16     import com.ibm.wala.shrikeBT.Constants;
17     import com.ibm.wala.types.ClassLoaderReference;
18     import com.ibm.wala.types.TypeName;
19     import com.ibm.wala.types.TypeReference;
20     import com.ibm.wala.util.collections.HashMapFactory;
21     import com.ibm.wala.util.strings.ImmutableByteArray;
22     
23     /**
24      * Utilities to interface with the Shrike CT library.
25      */
26     public class ShrikeUtil implements BytecodeConstants {
27     
28       final private static HashMap<String, TypeReference> primitiveMap;
29     
30       static {
31         primitiveMap = HashMapFactory.make(10);
32         primitiveMap.put("I", TypeReference.Int);
33         primitiveMap.put("J", TypeReference.Long);
34         primitiveMap.put("S", TypeReference.Short);
35         primitiveMap.put("B", TypeReference.Byte);
36         primitiveMap.put("C", TypeReference.Char);
37         primitiveMap.put("D", TypeReference.Double);
38         primitiveMap.put("F", TypeReference.Float);
39         primitiveMap.put("Z", TypeReference.Boolean);
40         primitiveMap.put("V", TypeReference.Void);
41         primitiveMap.put(Constants.TYPE_null, TypeReference.Null);
42       }
43     
44       /**
45        * @param type a type as a String returned by Shrike
46        */
47       public static TypeReference makeTypeReference(ClassLoaderReference loader, String type) throws IllegalArgumentException {
48         if (type == null) {
49           throw new IllegalArgumentException("null type");
50         }
51         TypeReference p = primitiveMap.get(type);
52         if (p != null) {
53           return p;
54         }
55         ImmutableByteArray b = ImmutableByteArray.make(type);
56         TypeName T = null;
57         /*if (b.get(0) != '[') {
58           T = TypeName.findOrCreate(b, 0, b.length() - 1);
59         } else {*/
60           if (b.get(b.length() - 1) == ';') {
61             T = TypeName.findOrCreate(b, 0, b.length() - 1);
62           } else {
63             T = TypeName.findOrCreate(b);
64           }
65         //}
66         return TypeReference.findOrCreate(loader, T);
67       }
68     }
