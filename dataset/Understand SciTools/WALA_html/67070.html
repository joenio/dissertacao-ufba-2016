11     package com.ibm.wala.dataflow.graph;
12     
13     import com.ibm.wala.fixpoint.BitVectorVariable;
14     import com.ibm.wala.fixpoint.UnaryOperator;
15     
16     
17     /**
18      * Operator OUT = IN U c
19      */
20     public class BitVectorUnionConstant extends UnaryOperator<BitVectorVariable> {
21       
22       private final int c;
23       public BitVectorUnionConstant(int c) {
24         if (c < 0) {
25           throw new IllegalArgumentException("Invalid c: " + c);
26         }
27         this.c = c;
28       }
29       
30     
31       @Override
32       public byte evaluate(BitVectorVariable lhs, BitVectorVariable rhs) throws IllegalArgumentException {
33     
34         if (lhs == null) {
35           throw new IllegalArgumentException("lhs == null");
36         }
37         BitVectorVariable U = new BitVectorVariable();
38         U.copyState(lhs);
39         U.addAll(rhs);
40         U.set(c);
41         if (!lhs.sameValue(U)) {
42           lhs.copyState(U);
43           return CHANGED;
44         } else {
45           return NOT_CHANGED;
46         }
47       }
48     
49       @Override
50       public String toString() {
51         return "U " + c;
52       }
53     
54       @Override
55       public int hashCode() {
56         return 9901 * c;
57       }
58     
59       @Override
60       public boolean equals(Object o) {
61         if (o instanceof BitVectorUnionConstant) {
62           BitVectorUnionConstant other = (BitVectorUnionConstant)o;
63           return c == other.c;
64         } else {
65           return false;
66         }
67       }
68     }
