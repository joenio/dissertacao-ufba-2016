11     package com.ibm.wala.ipa.callgraph.propagation.cfa;
12     
13     import com.ibm.wala.cfg.ControlFlowGraph;
14     import com.ibm.wala.classLoader.IClass;
15     import com.ibm.wala.ipa.callgraph.CGNode;
16     import com.ibm.wala.ipa.callgraph.propagation.SSAContextInterpreter;
17     import com.ibm.wala.ipa.callgraph.propagation.rta.DelegatingRTAContextInterpreter;
18     import com.ibm.wala.ssa.DefUse;
19     import com.ibm.wala.ssa.IR;
20     import com.ibm.wala.ssa.ISSABasicBlock;
21     import com.ibm.wala.ssa.SSAInstruction;
22     
23     /**
24      * An {@link SSAContextInterpreter} that first checks with A, then defaults to B.
25      */
26     public class DelegatingSSAContextInterpreter extends DelegatingRTAContextInterpreter implements SSAContextInterpreter {
27     
28       private final SSAContextInterpreter A;
29     
30       private final SSAContextInterpreter B;
31     
32       /**
33        * neither A nor B should be null.
34        */
35       public DelegatingSSAContextInterpreter(SSAContextInterpreter A, SSAContextInterpreter B) {
36         super(A, B);
37         this.A = A;
38         this.B = B;
39         if (A == null) {
40           throw new IllegalArgumentException("A cannot be null");
41         }
42         if (B == null) {
43           throw new IllegalArgumentException("B cannot be null");
44         }
45       }
46     
47       @Override
48       public IR getIR(CGNode node) {
49         if (A != null) {
50           if (A.understands(node)) {
51             return A.getIR(node);
52           }
53         }
54         assert B.understands(node);
55         return B.getIR(node);
56       }
57     
58       @Override
59       public int getNumberOfStatements(CGNode node) {
60         if (A != null) {
61           if (A.understands(node)) {
62             return A.getNumberOfStatements(node);
63           }
64         }
65         assert B.understands(node);
66         return B.getNumberOfStatements(node);
67       }
68     
69       @Override
70       public boolean understands(CGNode node) {
71         return A.understands(node) || B.understands(node);
72       }
73     
74       @Override
75       public boolean recordFactoryType(CGNode node, IClass klass) {
76         boolean result = false;
77         if (A != null) {
78           result |= A.recordFactoryType(node, klass);
79         }
80         result |= B.recordFactoryType(node, klass);
81         return result;
82       }
83     
84       @Override
85       public ControlFlowGraph<SSAInstruction, ISSABasicBlock> getCFG(CGNode node) {
86         if (A != null) {
87           if (A.understands(node)) {
88             return A.getCFG(node);
89           }
90         }
91         assert B.understands(node);
92         return B.getCFG(node);
93       }
94     
95       @Override
96       public DefUse getDU(CGNode node) {
97         if (A != null) {
98           if (A.understands(node)) {
99             return A.getDU(node);
100           }
101         }
102         assert B.understands(node);
103         return B.getDU(node);
104       }
105     }
