11     package com.ibm.wala.ipa.callgraph.propagation.cfa;
12     
13     import com.ibm.wala.classLoader.CallSiteReference;
14     import com.ibm.wala.classLoader.IMethod;
15     import com.ibm.wala.ipa.callgraph.ContextItem;
16     
17     public class CallString implements ContextItem {
18       private final CallSiteReference sites[];
19     
20       private final IMethod methods[];
21     
22       public CallString(CallSiteReference site, IMethod method) {
23         if (site == null) {
24           throw new IllegalArgumentException("null site");
25         }
26         this.sites = new CallSiteReference[] { site };
27         this.methods = new IMethod[] { method };
28       }
29     
30       CallString(CallSiteReference site, IMethod method, int length, CallString base) {
31         int sitesLength = Math.min(length, base.sites.length + 1);
32         int methodsLength = Math.min(length, base.methods.length + 1);
33         sites = new CallSiteReference[sitesLength];
34         sites[0] = site;
35         System.arraycopy(base.sites, 0, sites, 1, Math.min(length - 1, base.sites.length));
36         methods = new IMethod[methodsLength];
37         methods[0] = method;
38         System.arraycopy(base.methods, 0, methods, 1, Math.min(length - 1, base.methods.length));
39       }
40     
41       @Override
42       public String toString() {
43         StringBuffer str = new StringBuffer("[");
44         for (int i = 0; i < sites.length; i++) {
45           str.append(" ").append(methods[i].getSignature()).append("@").append(sites[i].getProgramCounter());
46         }
47         str.append(" ]");
48         return str.toString();
49       }
50     
51       @Override
52       public int hashCode() {
53         int code = 1;
54         for (int i = 0; i < sites.length; i++) {
55           code *= sites[i].hashCode() * methods[i].hashCode();
56         }
57     
58         return code;
59       }
60     
61       @Override
62       public boolean equals(Object o) {
63         if (o instanceof CallString) {
64           CallString oc = (CallString) o;
65           if (oc.sites.length == sites.length) {
66             for (int i = 0; i < sites.length; i++) {
67               if (!(sites[i].equals(oc.sites[i]) && methods[i].equals(oc.methods[i]))) {
68                 return false;
69               }
70             }
71     
72             return true;
73           }
74         }
75     
76         return false;
77       }
78     
79       public CallSiteReference[] getCallSiteRefs() {
80         return this.sites;
81       }
82     
83       public IMethod[] getMethods() {
84         return this.methods;
85       }
86     
87     }
