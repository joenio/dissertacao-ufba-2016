34     
35     public class JavaScriptEclipseProjectPath extends EclipseProjectPath<IIncludePathEntry, IJavaScriptProject> {
36     
37     	public enum JSLoader implements ILoader {
38     		JAVASCRIPT(JavaScriptTypes.jsLoader);
39     		
40     	    private ClassLoaderReference ref;
41     
42     	    JSLoader(ClassLoaderReference ref) {
43     	      this.ref = ref;
44     	    }
45     
46     		@Override
47     		public ClassLoaderReference ref() {
48     			return ref;
49     		}
50     	}
51     	
52       private final Set<Pair<String, Plugin>> models = HashSetFactory.make();
53     
54     	protected JavaScriptEclipseProjectPath(Set<Pair<String, Plugin>> models) throws IOException,
55     			CoreException {
56     		super(AnalysisScopeType.SOURCE_FOR_PROJ_AND_LINKED_PROJS);
57     		this.models.addAll(models);
58     		this.models.add(Pair.make("prologue.js", (Plugin)Activator.getDefault()));
59     	}
60     
61     	public static JavaScriptEclipseProjectPath make(IJavaScriptProject p, Set<Pair<String, Plugin>> models) throws IOException, CoreException {
62     	  JavaScriptEclipseProjectPath path = new JavaScriptEclipseProjectPath(models);
63     	  path.create(p.getProject());  
64     	  return path;
65     	}
66     
67     	
68     	@Override
69       public EclipseProjectPath create(IProject project) throws CoreException, IOException {
70         EclipseProjectPath path = super.create(project);
71       
72         Collection<Module> s = modules.get(JSLoader.JAVASCRIPT);
73         for(Pair<String,Plugin> model : models) {
74           URL modelFile = JsdtUtil.getPrologueFile(model.fst, model.snd);
75           assert modelFile != null : "cannot find file for " + model;
76           s.add(new SourceURLModule(modelFile) {
77             @Override
78             public String getName() {
79               return super.getName().substring(1);
80             }
81           });
82         }
83     
84         return path;
85     }
86     
87       @Override
88     	protected IJavaScriptProject makeProject(IProject p) {
89     		try {
90     			if (p.hasNature(JavaScriptCore.NATURE_ID)) {
91     				return JavaScriptCore.create(p);
92     			} else {
93     				return null;
94     			}
95     		} catch (CoreException e) {
96     			return null;
97     		}
98     	}
99     
100     	@Override
101     	protected IIncludePathEntry resolve(IIncludePathEntry entry) {
102     		return JavaScriptCore.getResolvedIncludepathEntry(entry);
103     	}
104     
105     	@Override
106     	protected void resolveClasspathEntry(IJavaScriptProject project,
107     			IIncludePathEntry entry,
108     			ILoader loader,
109     			boolean includeSource, boolean cpeFromMainProject) {
110     		IIncludePathEntry e = JavaScriptCore.getResolvedIncludepathEntry(entry);
111     		switch (e.getEntryKind()) {
112     		case IIncludePathEntry.CPE_SOURCE:
113     			resolveSourcePathEntry(JSLoader.JAVASCRIPT, true, cpeFromMainProject, e.getPath(), null, e.getExclusionPatterns(), "js");
114     		}
115     	}
116     
117     	@Override
118     	protected void resolveProjectClasspathEntries(IJavaScriptProject project, boolean includeSource) {
119     	  try {
120     			resolveClasspathEntries(project, Arrays.asList(project.getRawIncludepath()), Loader.EXTENSION, includeSource, true);
121     		} catch (JavaScriptModelException e) {
122     			// TODO Auto-generated catch block
123     			e.printStackTrace();
124     		}
125     	}	
126     }
