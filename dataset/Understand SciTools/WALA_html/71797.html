11     package com.ibm.wala.shrikeCT;
12     
13     /**
14      * This class builds serializable Exceptions attributes.
15      */
16     public final class ExceptionsWriter extends ClassWriter.Element {
17       final private int attrID;
18     
19       private int[] table;
20     
21       /**
22        * Build an empty writer.
23        * 
24        * @throws IllegalArgumentException if w is null
25        */
26       public ExceptionsWriter(ClassWriter w) {
27         if (w == null) {
28           throw new IllegalArgumentException("w is null");
29         }
30         attrID = w.addCPUtf8("Exceptions");
31       }
32     
33       @Override
34       public int getSize() {
35         return table == null ? 8 : 8 + table.length * 2;
36       }
37     
38       @Override
39       public int copyInto(byte[] buf, int offset) throws IllegalArgumentException {
40         ClassWriter.setUShort(buf, offset, attrID);
41         ClassWriter.setInt(buf, offset + 2, getSize() - 6);
42         ClassWriter.setUShort(buf, offset + 6, table == null ? 0 : table.length);
43         offset += 8;
44         if (table != null) {
45           for (int i = 0; i < table.length; i++) {
46             ClassWriter.setUShort(buf, offset, table[i]);
47             offset += 2;
48           }
49         }
50         return offset;
51       }
52     
53       /**
54        * Set the list of exceptions that can be thrown.
55        * 
56        * @param exceptions an array of indices to constant pool Class entries
57        * @throws IllegalArgumentException if exceptions is null
58        */
59       public void setRawTable(int[] exceptions) {
60         if (exceptions == null) {
61           throw new IllegalArgumentException("exceptions is null");
62         }
63         for (int i = 0; i < exceptions.length; i++) {
64           if (exceptions[i] < 1 || exceptions[i] > 0xFFFF) {
65             throw new IllegalArgumentException("Invalid CP index: " + exceptions[i]);
66           }
67         }
68     
69         this.table = exceptions;
70       }
71     }
