16     import com.ibm.wala.cfg.exc.intra.NullPointerState;
17     import com.ibm.wala.util.CancelException;

21     /**
22      * This abstract class is used as interface for analysis that remove impossible
23      * control flow from a CFG. This is done by detecting exceptions that may always
24      * (or never) appear.
25      * 
26      * @author Juergen Graf <graf@kit.edu>
27      * 
28      */
29     public interface ExceptionPruningAnalysis<I, T extends IBasicBlock<I>> {
30     
31       /**
32        * Computes impossible control flow that is due to exceptions that definitely
33        * will not appear or that will always be thrown. You have to run this method
34        * before using getPruned() to extract the result of the analysis.
35        * @param progress
36        *          A progress monitor that is used to display the progress of the
37        *          analysis. It can also be used to detect a cancel request from the
38        *          user. The common behavior is to cancel the method if
39        *          progress.isCanceled() is true by throwing a CancelException.
40        * @return Number of edges that have been removed from the cfg.
41        * @throws UnsoundGraphException
42        *           Thrown if the original CFG contains inconsistencies.
43        * @throws CancelException
44        *           Thrown if the user requested cancellation through the progress
45        *           monitor.
46        */
47       int compute(IProgressMonitor progress) throws UnsoundGraphException, CancelException;
48       
49       /**
50        * Returns the result of the analysis: A control flow graph where impossible
51        * control flow has been removed. The way how and which impossible flow is
52        * detected may vary between different implementations of this class.
53        * Run compute(IProgressMonitor) first.
54        * 
55        * @return The improved CFG without edges that were detected as impossible
56        *         flow.
57        */
58       ControlFlowGraph<I, T> getCFG();
59     
60       /**
61        * Returns true if the corresponding method contains instructions that may
62        * throw an exception which is not caught in the same method.
63        * Run compute(IPrograssMonitor) first.
64        * @return true if the corresponding method contains instructions that may
65        * throw an exception which is not caught in the same method
66        */
67       boolean hasExceptions();
68     
69       /**
70        * Returns the state of a node. The node has to be part of the cfg.
71        * @param bb Node
72        * @return NullPointerState
73        */
74       NullPointerState getState(T bb);
75     
76     }
