11     package com.ibm.wala.cast.tree.impl;
12     
13     import java.util.NoSuchElementException;
14     
15     import com.ibm.wala.cast.tree.CAst;
16     import com.ibm.wala.cast.tree.CAstNode;
17     import com.ibm.wala.cast.util.CAstPrinter;
18     
19     /**
20      * An implementation of CAst, i.e. a simple factory for creating capa ast nodes. This class simply creates generic nodes with a kind
21      * field, and either an array of children or a constant values. Note that there is no easy way to mutate these trees; do not change
22      * this (see CAstNode for the rationale for this rule).
23      * 
24      * @author Julian Dolby (dolby@us.ibm.com)
25      * 
26      */
27     public class CAstImpl implements CAst {
28       private int nextID = 0;
29     
30       @Override
31       public String makeUnique() {
32         return "id" + (nextID++);
33       }
34     
35       protected static class CAstNodeImpl implements CAstNode {
36         protected final CAstNode[] cs;
37     
38         protected final int kind;
39     
40         protected CAstNodeImpl(int kind, CAstNode[] cs) {
41           this.kind = kind;
42           this.cs = cs;
43     
44           for (int i = 0; i < cs.length; i++)
45             assert cs[i] != null : "argument " + i + " is null for node kind " + kind + " [" + CAstPrinter.entityKindAsString(kind)
46                 + "]";
47         }
48     
49         @Override
50         public int getKind() {
51           return kind;
52         }
53     
54         @Override
55         public Object getValue() {
56           return null;
57         }
58     
59         @Override
60         public CAstNode getChild(int n) {
61           try {
62             return cs[n];
63           } catch (ArrayIndexOutOfBoundsException e) {
64             throw new NoSuchElementException(n + " of " + CAstPrinter.print(this));
65           }
66         }
67     
68         @Override
69         public int getChildCount() {
70           return cs.length;
71         }
72     
73         @Override
74         public String toString() {
75           return System.identityHashCode(this) + ":" + CAstPrinter.print(this);
76         }
77     
78         @Override
79         public int hashCode() {
80           int code = getKind() * (getChildCount() + 13);
81           for (int i = 0; i < getChildCount() && i < 15; i++) {
82             code *= getChild(i).getKind();
83           }
84     
85           return code;
86         }
87       }
88     
89       @Override
90       public CAstNode makeNode(final int kind, final CAstNode[] cs) {
91         return new CAstNodeImpl(kind, cs);
92       }
93     
94       @Override
95       public CAstNode makeNode(int kind, CAstNode c1, CAstNode[] cs) {
96         CAstNode[] children = new CAstNode[cs.length + 1];
97         children[0] = c1;
98         System.arraycopy(cs, 0, children, 1, cs.length);
99         return makeNode(kind, children);
100       }
101     
102       @Override
103       public CAstNode makeNode(int kind) {
104         return makeNode(kind, new CAstNode[0]);
105       }
106     
107       @Override
108       public CAstNode makeNode(int kind, CAstNode c1) {
109         return makeNode(kind, new CAstNode[] { c1 });
110       }
111     
112       @Override
113       public CAstNode makeNode(int kind, CAstNode c1, CAstNode c2) {
114         return makeNode(kind, new CAstNode[] { c1, c2 });
115       }
116     
117       @Override
118       public CAstNode makeNode(int kind, CAstNode c1, CAstNode c2, CAstNode c3) {
119         return makeNode(kind, new CAstNode[] { c1, c2, c3 });
120       }
121     
122       @Override
123       public CAstNode makeNode(int kind, CAstNode c1, CAstNode c2, CAstNode c3, CAstNode c4) {
124         return makeNode(kind, new CAstNode[] { c1, c2, c3, c4 });
125       }
126     
127       @Override
128       public CAstNode makeNode(int kind, CAstNode c1, CAstNode c2, CAstNode c3, CAstNode c4, CAstNode c5) {
129         return makeNode(kind, new CAstNode[] { c1, c2, c3, c4, c5 });
130       }
131     
132       @Override
133       public CAstNode makeNode(int kind, CAstNode c1, CAstNode c2, CAstNode c3, CAstNode c4, CAstNode c5, CAstNode c6) {
134         return makeNode(kind, new CAstNode[] { c1, c2, c3, c4, c5, c6 });
135       }
136     
137       protected static class CAstValueImpl implements CAstNode {
138         protected final Object value;
139     
140         protected CAstValueImpl(Object value) {
141           this.value = value;
142         }
143     
144         @Override
145         public int getKind() {
146           return CAstNode.CONSTANT;
147         }
148     
149         @Override
150         public Object getValue() {
151           return value;
152         }
153     
154         @Override
155         public CAstNode getChild(int n) {
156           throw new NoSuchElementException();
157         }
158     
159         @Override
160         public int getChildCount() {
161           return 0;
162         }
163     
164         @Override
165         public String toString() {
166           return "CAstValue: " + value;
167         }
168     
169         @Override
170         public int hashCode() {
171           return getKind() * toString().hashCode();
172         }
173       }
174     
175       @Override
176       public CAstNode makeConstant(final Object value) {
177         return new CAstValueImpl(value);
178       }
179     
180       @Override
181       public CAstNode makeConstant(boolean value) {
182         return makeConstant(value ? Boolean.TRUE : Boolean.FALSE);
183       }
184     
185       @Override
186       public CAstNode makeConstant(char value) {
187         return makeConstant(new Character(value));
188       }
189     
190       @Override
191       public CAstNode makeConstant(short value) {
192         return makeConstant(new Short(value));
193       }
194     
195       @Override
196       public CAstNode makeConstant(int value) {
197         return makeConstant(new Integer(value));
198       }
199     
200       @Override
201       public CAstNode makeConstant(long value) {
202         return makeConstant(new Long(value));
203       }
204     
205       @Override
206       public CAstNode makeConstant(float value) {
207         return makeConstant(new Float(value));
208       }
209     
210       @Override
211       public CAstNode makeConstant(double value) {
212         return makeConstant(new Double(value));
213       }
214     
215     }
