154          * @return the name of each class that's in the classpath
155          */
156         private static String[] generateRootClassesFromWorkspace() {
157     	String classpath = System.getProperty("java.class.path");
158     	Object[] binDirectories = extractBinDirectories(classpath);
159     	HashSet<String> classFileNames = HashSetFactory.make();
160     	for (int i = 0; i < binDirectories.length; i++) {
161     	    String dir = (String) binDirectories[i];
162     	    File fdir = new File(dir);
163     	    classFileNames.addAll(findClassNames(dir, fdir));
164     	}
165     	String[] result = new String[classFileNames.size()];
166     	Iterator<String> it = classFileNames.iterator();
167     	for (int i = 0; i < result.length; i++) {
168     	    result[i] = it.next();
169     	}
170     	return result;
171         }

174          * @param fdir
175          * @return Collection <String>representing the class names in a particular
176          *         file
177          */
178         /**
179          * @param rootDir
180          *            root of the classpath governing file f
181          * @param f
182          *            a File or directory
183          * @return Collection <String>representing the class names in f
184          */
185         private static Collection<String> findClassNames(String rootDir, File f) {
186     	HashSet<String> result = HashSetFactory.make();
187     	if (f.isDirectory()) {
188     	    File[] files = f.listFiles();
189     	    for (int i = 0; i < files.length; i++) {
190     		result.addAll(findClassNames(rootDir, files[i]));
191     	    }
192     	} else {
193     	    if (f.getName().indexOf(".class") > 0) {
194     		String p = f.getAbsolutePath();
195     		p = p.substring(rootDir.length() + 1);
196     		// trim the last 6 characters which hold ".class"
197     		p = p.substring(0, p.length() - 6);
198     		p = p.replace('\\', '.');
199     		return Collections.singleton(p);
200     	    }
201     	}
202     	return result;
203         }

205         /**
206          * @param classpath
207          * @return set of strings that are names of directories that contain "bin"
208          */
209         private static Object[] extractBinDirectories(String classpath) {
210     	StringTokenizer t = new StringTokenizer(classpath, ";");
211     	HashSet<String> result = HashSetFactory.make();
212     	while (t.hasMoreTokens()) {
213     	    String n = t.nextToken();
214     	    if (n.indexOf("bin") > 0) {
215     		result.add(n);
216     	    }
217     	}
218     	return result.toArray();

221         /**
222          * Trace the heap and return the results
223          */
224         public Result perform() throws ClassNotFoundException,
225     	    IllegalArgumentException, IllegalAccessException {
226     	Result result = new Result();
227     	IdentityHashMap<Object, Object> objectsVisited = new IdentityHashMap<Object, Object>();
228     	if (traceStatics) {
229     	    for (int i = 0; i < rootClasses.length; i++) {
230     		Class<?> c = Class.forName(rootClasses[i]);
231     		Field[] fields = c.getDeclaredFields();
232     		for (int j = 0; j < fields.length; j++) {
233     		    if (isStatic(fields[j])) {
234     			traverse(fields[j], result, objectsVisited);
235     		    }
236     		}
237     	    }
238     	}
239     	for (Iterator<?> it = rootInstances.iterator(); it.hasNext();) {
240     	    Object instance = it.next();
241     	    Class<?> c = instance.getClass();
242     	    Set<Field> fields = getAllInstanceFields(c);
243     	    for (Iterator<Field> it2 = fields.iterator(); it2.hasNext();) {
244     		Field f = it2.next();
245     		traverse(f, instance, result, objectsVisited);
246     	    }
247     	}
248     
249     	return result;
250         }

252         /**
253          * @param o
254          * @return the estimated size of the object
255          */
256         private static int computeSizeOf(Object o) {
257     	int result = BYTES_IN_HEADER;
258     	Class<?> c = o.getClass();
259     	if (c.isArray()) {
260     	    Class<?> elementType = c.getComponentType();
261     	    int length = Array.getLength(o);
262     	    result += length * sizeOfSlot(elementType);
263     	} else if (c.isPrimitive()) {
264     	    throw new Error();
265     	} else {
266     	    Collection<Field> fields = getAllInstanceFields(c);
267     	    for (Iterator<Field> it = fields.iterator(); it.hasNext();) {
268     		Field f = it.next();
269     		result += sizeOfSlot(f.getType());
270     	    }
271     	}
272     	return result;
273         }

275         /**
276          * @param o
277          * @return the estimated size of the object
278          */
279         private int sizeOf(Object o) {
280     	Class<?> c = o.getClass();
281     	if (c.isArray()) {
282     	    return computeSizeOf(o);
283     	} else {
284     	    Integer S = sizeMap.get(c);
285     	    if (S == null) {
286     		S = new Integer(computeSizeOf(o));
287     		sizeMap.put(c, S);
288     	    }
289     	    return S.intValue();
290     	}
291         }

293         /**
294          * @param c
295          * @return size of a field of type c, in bytes
296          */
297         private static int sizeOfSlot(Class<?> c) {
298     	if (!c.isPrimitive()) {
299     	    return 4;
300     	} else {
301     	    if (c.equals(Boolean.TYPE) || c.equals(Character.TYPE)
302     		    || c.equals(Byte.TYPE)) {
303     		return 1;
304     	    } else if (c.equals(Short.TYPE)) {
305     		return 2;
306     	    } else if (c.equals(Integer.TYPE) || c.equals(Float.TYPE)) {
307     		return 4;
308     	    } else if (c.equals(Long.TYPE) || c.equals(Double.TYPE)) {
309     		return 8;
310     	    } else {
311     		throw new Error();
312     	    }
313     	}
314         }

316         /**
317          * Traverse the heap starting at a static field
318          * 
319          * @param result
320          * @throws IllegalAccessException
321          * @throws IllegalArgumentException
322          */
323         private void traverse(Field root, Result result,
324     	    IdentityHashMap<Object, Object> objectsVisited)
325     	    throws IllegalArgumentException, IllegalAccessException {
326     	if (DEBUG) {
327     	    System.err.println(("traverse root " + root));
328     	}
329     	root.setAccessible(true);
330     	Object contents = root.get(null);
331     	if (contents != null && (objectsVisited.get(contents) == null)) {
332     	    objectsVisited.put(contents, DUMMY);
333     	    Class<?> c = contents.getClass();
334     	    if (c.isArray()) {
335     		result.registerReachedFrom(root, root, contents);
336     		arrayWorkList.push(Pair.make(root, contents));
337     	    } else {
338     		result.registerReachedFrom(root, root, contents);
339     		if (internalClasses.contains(c)) {
340     		    contents = Pair.make(root, contents);
341     		}
342     		scalarWorkList.push(contents);
343     	    }
344     	}
345     
346     	drainWorkLists(root, result, objectsVisited);
347         }
348     
349         private void traverse(Field root, Object instance, Result result,
350     	    IdentityHashMap<Object, Object> objectsVisited)
351     	    throws IllegalArgumentException, IllegalAccessException {
352     
353     	traverseFieldOfScalar(root, root, instance, null, objectsVisited,
354     		result);
355     	drainWorkLists(root, result, objectsVisited);
356         }

358         @SuppressWarnings("rawtypes")
359         private void drainWorkLists(Field root, Result result,
360     	    IdentityHashMap<Object, Object> objectsVisited)
361     	    throws IllegalAccessException {
362     	while (!scalarWorkList.isEmpty() || !arrayWorkList.isEmpty()) {
363     	    if (!scalarWorkList.isEmpty()) {
364     		Object scalar = scalarWorkList.pop();
365     		if (scalar instanceof Pair) {
366     		    Pair p = (Pair) scalar;
367     		    traverseScalar(root, p.snd, p.fst, result, objectsVisited);
368     		} else {
369     		    traverseScalar(root, scalar, null, result, objectsVisited);
370     		}
371     	    }
372     	    if (!arrayWorkList.isEmpty()) {
373     		Pair p = (Pair) arrayWorkList.pop();
374     		traverseArray(root, p.snd, p.fst, result, objectsVisited);
375     	    }
376     	}
377         }

379         private void traverseArray(Field root, Object array, Object container,
380     	    Result result, IdentityHashMap<Object, Object> objectsVisited)
381     	    throws IllegalArgumentException {
382     	if (DEBUG) {
383     	    System.err.println(("traverse array " + array.getClass()));
384     	}
385     	Class<?> elementKlass = array.getClass().getComponentType();
386     	if (elementKlass.isPrimitive()) {
387     	    return;
388     	}
389     	assert container != null;
390     	int length = Array.getLength(array);
391     	for (int i = 0; i < length; i++) {
392     	    Object contents = Array.get(array, i);
393     
394     	    if (contents != null && (objectsVisited.get(contents) == null)) {
395     		objectsVisited.put(contents, DUMMY);
396     		Class<?> klass = contents.getClass();
397     		if (klass.isArray()) {
398     		    result.registerReachedFrom(root, container, contents);
399     		    contents = Pair.make(container, contents);
400     		    arrayWorkList.push(contents);
401     		} else {
402     		    result.registerReachedFrom(root, container, contents);
403     		    contents = Pair.make(container, contents);
404     		    scalarWorkList.push(contents);
405     		}
406     	    }
407     	}
408         }

410         private void traverseScalar(Field root, Object scalar, Object container,
411     	    Result result, IdentityHashMap<Object, Object> objectsVisited)
412     	    throws IllegalArgumentException, IllegalAccessException {
413     
414     	Class<?> c = scalar.getClass();
415     	if (DEBUG) {
416     	    System.err.println(("traverse scalar " + c));
417     	}
418     	Field[] fields = getAllReferenceInstanceFields(c);
419     	for (int i = 0; i < fields.length; i++) {
420     	    traverseFieldOfScalar(root, fields[i], scalar, container,
421     		    objectsVisited, result);
422     	}
423         }

432         private final boolean isInBadPackage(Class<?> c) {
433     	Package p = c.getPackage();
434     	if (p == null) {
435     	    return false;
436     	}
437     	Object status = packageStatus.get(p);
438     	if (status == OK) {
439     	    return false;
440     	} else if (status == BAD) {
441     	    return true;
442     	} else {
443     	    if (p.getName() != null && p.getName().indexOf("sun.reflect") != -1) {
444     		if (DEBUG) {
445     		    System.err.println(("making " + p + " a BAD package"));
446     		}
447     		packageStatus.put(p, BAD);
448     		return true;
449     	    } else {
450     		if (DEBUG) {
451     		    System.err.println(("making " + p + " an OK package"));
452     		}
453     		packageStatus.put(p, OK);
454     		return false;
455     	    }
456     	}
457         }

459         /**
460          * @param root
461          * @param f
462          * @param scalar
463          * @param container
464          * @param objectsVisited
465          * @param result
466          * @throws IllegalArgumentException
467          * @throws IllegalAccessException
468          */
469         private void traverseFieldOfScalar(Field root, Field f, Object scalar,
470     	    Object container, IdentityHashMap<Object, Object> objectsVisited,
471     	    Result result) throws IllegalArgumentException,
472     	    IllegalAccessException {
473     	// The following atrocious hack is needed to prevent the IBM 141 VM
474     	// from crashing
475     	if (isInBadPackage(f.getType())) {
476     	    return;
477     	}
478     	if (container == null) {
479     	    container = f;
480     	}
481     	f.setAccessible(true);
482     	Object contents = f.get(scalar);
483     	if (contents != null && (objectsVisited.get(contents) == null)) {
484     	    try {
485     		objectsVisited.put(contents, DUMMY);
486     	    } catch (Exception e) {
487     		e.printStackTrace();
488     		return;
489     	    }
490     	    Class<?> klass = contents.getClass();
491     	    if (klass.isArray()) {
492     		result.registerReachedFrom(root, container, contents);
493     		contents = Pair.make(container, contents);
494     		arrayWorkList.push(contents);
495     	    } else {
496     		result.registerReachedFrom(root, container, contents);
497     		if (internalClasses.contains(klass)) {
498     		    contents = Pair.make(container, contents);
499     		}
500     		scalarWorkList.push(contents);
501     	    }
502     	}
503         }

505         private static HashSet<Field> getAllInstanceFields(Class<?> c) {
506     	HashSet<Field> result = HashSetFactory.make();
507     	Class<?> klass = c;
508     	while (klass != null) {
509     	    Field[] fields = klass.getDeclaredFields();
510     	    for (int i = 0; i < fields.length; i++) {
511     		if (!isStatic(fields[i])) {
512     		    result.add(fields[i]);
513     		}
514     	    }
515     	    klass = klass.getSuperclass();
516     	}
517     	return result;
518         }

520         final private HashMap<Class<?>, Field[]> allReferenceFieldsCache = HashMapFactory
521     	    .make();
522     
523         /**
524          * @return Field[] representing reference instance fields of a class
525          */
526         private Field[] getAllReferenceInstanceFields(Class<?> c) {
527     	if (allReferenceFieldsCache.containsKey(c))
528     	    return allReferenceFieldsCache.get(c);
529     	else {
530     	    HashSet<Field> s = HashSetFactory.make();
531     	    Class<?> klass = c;
532     	    while (klass != null) {
533     		Field[] fields = klass.getDeclaredFields();
534     		for (int i = 0; i < fields.length; i++) {
535     		    if (!isStatic(fields[i])) {
536     			Class<?> fc = fields[i].getType();
537     			if (!fc.isPrimitive()) {
538     			    s.add(fields[i]);
539     			}
540     		    }
541     		}
542     		klass = klass.getSuperclass();
543     	    }
544     	    Field[] result = new Field[s.size()];
545     	    Object[] temp = s.toArray();
546     	    for (int i = 0; i < result.length; i++) {
547     		result[i] = (Field) temp[i];
548     	    }
549     	    allReferenceFieldsCache.put(c, result);
550     	    return result;
551     	}
552         }

627          * @author sfink
628          * 
629          *         Statistics about objects reached from a single root
630          */
631         class Demographics {
632     	/**
633     	 * mapping: Object (key) -> Integer (number of instances in a partition)
634     	 */
635     	private final HashMap<Object, Integer> instanceCount = HashMapFactory
636     		.make();
637     
638     	/**
639     	 * mapping: Object (key) -> Integer (bytes)
640     	 */
641     	private final HashMap<Object, Integer> sizeCount = HashMapFactory
642     		.make();
643     
644     	/**
645     	 * Total number of instances discovered
646     	 */
647     	private int totalInstances = 0;
648     
649     	/**
650     	 * Total number of bytes discovered
651     	 */
652     	private int totalSize = 0;
653     
654     	/**
655     	 * @param key
656     	 *            a name for the heap partition to which o belongs
657     	 * @param o
658     	 *            the object to register
659     	 */
660     	public void registerObject(Object key, Object o) {
661     	    Integer I = instanceCount.get(key);
662     	    int newCount = (I == null) ? 1 : I.intValue() + 1;
663     	    instanceCount.put(key, new Integer(newCount));
664     	    totalInstances++;
665     
666     	    I = sizeCount.get(key);
667     	    int s = sizeOf(o);
668     	    int newSizeCount = (I == null) ? s : I.intValue() + s;
669     	    sizeCount.put(key, new Integer(newSizeCount));
670     	    totalSize += s;
671     	}
672     
673     	@Override
674     	public String toString() {
675     	    StringBuffer result = new StringBuffer();
676     	    result.append("Totals: " + totalInstances + " " + totalSize + "\n");
677     	    TreeSet<Object> sorted = new TreeSet<Object>(new SizeComparator());
678     	    sorted.addAll(instanceCount.keySet());
679     	    for (Iterator<Object> it = sorted.iterator(); it.hasNext();) {
680     		Object key = it.next();
681     		Integer I = instanceCount.get(key);
682     		Integer bytes = sizeCount.get(key);
683     		result.append("  ").append(I).append("   ").append(bytes)
684     			.append("   ");
685     		result.append(bytes.intValue() / I.intValue()).append("   ");
686     		result.append(key);
687     		result.append("\n");
688     	    }
689     	    return result.toString();
690     	}

693     	 * compares two keys based on the total size of the heap traced from
694     	 * that key
695     	 */
696     	private class SizeComparator implements Comparator<Object> {
697     	    /*

699     	     * java.lang.Object)

703     		Integer i1 = sizeCount.get(o1);
704     		Integer i2 = sizeCount.get(o2);
705     		return i2.intValue() - i1.intValue();
706     	    }
