48       /**
49        * @return the JVM "stack word size" for the given JVM type, looking at index 'index'
50        */
51       static byte getWordSize(String s, int index) {
52         switch (s.charAt(index)) {
53         case 'V':
54           return 0;
55         case 'J':
56         case 'D':
57           return 2;
58         default:
59           return 1;
60         }
61       }

63       /**
64        * Computes the character length of the internal JVM type given by s.substring(i).
65        */
66       private static int getTypeLength(String s, int i) {
67         switch (s.charAt(i)) {
68         case 'L':
69           return s.indexOf(';', i) - i + 1;
70         case '[':
71           return getTypeLength(s, i + 1) + 1;
72         default:
73           return 1;
74         }
75       }

77       /**
78        * Compute the total number of JVM "stack words" occupied by the method parameters for method signature "type". Any "this"
79        * parameter is not included.
80        * 
81        * @throws IllegalArgumentException if type is null
82        */
83       public static int getParamsWordSize(String type) throws IllegalArgumentException {
84         if (type == null) {
85           throw new IllegalArgumentException("type is null");
86         }
87         int index = 1;
88         int count = 0;
89     
90         if (type.indexOf(')', 1) < 0) {
91           throw new IllegalArgumentException("Invalid method descriptor (missing ')'): " + type);
92         }
93         while (type.charAt(index) != ')') {
94           count += getWordSize(type, index);
95           index += getTypeLength(type, index);
96         }
97         return count;
98       }

101        * Convert a fully-qualified Java class name ('.' separated) into an internal JVM type name ('/' separated, starting with 'L' and
102        * ending with ';').
103        * 
104        * @throws IllegalArgumentException if c is null
105        */
106       public static String makeType(String c) {
107         if (c == null) {
108           throw new IllegalArgumentException("c is null");
109         }
110         if (c.startsWith("[")) {
111           return c.replace('.', '/');
112         } else if (!c.endsWith(";")) {
113           return "L" + c.replace('.', '/') + ";";
114         } else {
115           return c;
116         }

204       /**
205        * Compute the number of parameters given by method signature "type". Any "this" parameter is not included.
206        * 
207        * @throws IllegalArgumentException if type == null
208        */
209       static int getParamsCount(String type) throws IllegalArgumentException {
210         if (type == null || type.length() < 2) {
211           throw new IllegalArgumentException("invalid type: " + type);
212         }
213         int index = 1;
214         int count = 0;
215     
216         try {
217           while (type.charAt(index) != ')') {
218             count++;
219             index += getTypeLength(type, index);
220           }
221           return count;
222         } catch (StringIndexOutOfBoundsException e) {
223           throw new IllegalArgumentException("invalid type: " + type);
224         }
225       }

227       /**
228        * Extract the types of the parameters given by method signature "type".
229        * 
230        * @param thisClassType null if the method is static, otherwise the type of "this"
231        * @return an array of the parameter types in order, including "this" as the first parameter if thisClassType was non-null
232        * @throws IllegalArgumentException if type == null
233        */
234       public static String[] getParamsTypes(String thisClassType, String type) throws IllegalArgumentException {
235         if (type == null) {
236           throw new IllegalArgumentException("type == null");
237         }
238         int count = thisClassType != null ? 1 : 0;
239         String[] r = new String[getParamsCount(type) + count];
240         int index = 1;
241     
242         if (thisClassType != null) {
243           r[0] = thisClassType;
244         }
245     
246         while (type.charAt(index) != ')') {
247           int len = getTypeLength(type, index);
248     
249           r[count] = type.substring(index, index + len);
250           count++;
251           index += len;
252         }
253         return r;
254       }

256       /**
257        * Compute the types of the local variables on entry to a method. Similar to "getParamsTypes" except null array entries are
258        * inserted to account for unused local variables because of 2-word parameter values.
259        * 
260        * @throws IllegalArgumentException if type == null
261        */
262       public static String[] getParamsTypesInLocals(String thisClassType, String type) throws IllegalArgumentException {
263         if (type == null) {
264           throw new IllegalArgumentException("type == null");
265         }
266         int count = thisClassType != null ? 1 : 0;
267         String[] r = new String[getParamsWordSize(type) + count];
268         int index = 1;
269     
270         if (thisClassType != null) {
271           r[0] = thisClassType;
272         }
273     
274         while (type.charAt(index) != ')') {
275           int len = getTypeLength(type, index);
276           String t = getStackType(type.substring(index, index + len));
277     
278           r[count] = t;
279           count += getWordSize(t);
280           index += len;
281         }
282         return r;
283       }

367        *

370       public static String computeSignature(Class<?>[] params, Class<?> result) throws IllegalArgumentException {
371         if (params == null) {
372           throw new IllegalArgumentException("params == null");
373         }
374         StringBuffer buf = new StringBuffer();
375         buf.append("(");
376         for (int i = 0; i < params.length; i++) {
377           buf.append(makeType(params[i]));
378         }
379         buf.append(")");
380         buf.append(makeType(result));
381         return buf.toString();
382       }

425         if (paramTypes == null) {
426           return name;
427         } else {
428           StringBuffer buf = new StringBuffer(name);
429           buf.append("(");
430           for (int i = 0; i < paramTypes.length; i++) {
431             if (i > 0) {
432               buf.append(",");

445       public static Method findMethod(Class<?> c, String name, Class<?>[] paramTypes) {
446         if (c == null) {
447           throw new IllegalArgumentException("c is null");
448         }
449         Method[] methods = c.getMethods();
450         Method result = null;
451         for (int i = 0; i < methods.length; i++) {
452           Method m = methods[i];
453           if (m.getName().equals(name) && (paramTypes == null || Arrays.equals(m.getParameterTypes(), paramTypes))) {
454             if (result != null) {
455               throw new IllegalArgumentException("Method " + makeName(name, paramTypes) + " is ambiguous in class " + c);
456             }
457             result = m;
458           }
459         }
460         return result;
461       }

463       /**
464        * Make an Instruction which calls a method, given its name, Java Class, and a list of parameter classes to use for overload
465        * resolution. Method information is obtained using reflection.
466        * 
467        * @throws IllegalArgumentException if name is null
468        */
469       public static InvokeInstruction makeInvoke(Class<?> c, String name, Class<?>[] paramTypes) {
470         if (name == null) {
471           throw new IllegalArgumentException("name is null");
472         }
473         InvokeInstruction result = null;
474     
475         if (name.equals("<init>")) {
476           Constructor<?>[] cs = c.getConstructors();
477           for (int i = 0; i < cs.length; i++) {
478             Constructor<?> con = cs[i];
479             if (paramTypes == null || Arrays.equals(con.getParameterTypes(), paramTypes)) {
480               if (result != null) {
481                 throw new IllegalArgumentException("Constructor " + makeName(name, paramTypes) + " is ambiguous in class " + c);
482               }
483               result = InvokeInstruction
484                   .make(computeSignature(con.getParameterTypes(), Void.TYPE), makeType(c), name, Dispatch.SPECIAL);
485             }
486           }
487         } else {
488           Method m = findMethod(c, name, paramTypes);
489           if (m != null) {
490             Dispatch opcode = Dispatch.VIRTUAL;
491             if ((m.getModifiers() & Constants.ACC_STATIC) != 0) {
492               opcode = Dispatch.STATIC;
493             } else if (m.getDeclaringClass().isInterface()) {
494               opcode = Dispatch.INTERFACE;
495             }
496             result = InvokeInstruction.make(computeSignature(m.getParameterTypes(), m.getReturnType()), makeType(c), name, opcode);
497           }
498         }
499     
500         if (result == null) {
501           throw new IllegalArgumentException("Method " + makeName(name, paramTypes) + " is not present in class " + c);
502         } else {
503           return result;
504         }
505       }

517       /**
518        * Compute the type index constant (Constants.TYPE_...) from the JVM type. Returns -1 if the type is not one of the predefined
519        * constants.
520        */
521       static int getTypeIndex(String t) {
522         if (t == null) {
523           return -1;
524         } else {
525           int len = t.length();
526           if (len < 1) {
527             return -1;
528           } else {
529             char ch = t.charAt(0);
530             if (ch < typeIndices.length) {
531               int r = typeIndices[ch];
532               if (r != 4) {
533                 if (len > 1) {
534                   return -1;
535                 } else {
536                   return r;
537                 }
538               } else {
539                 return r;
540               }
541             } else {
542               return -1;
543             }
544           }
545         }
546       }

569       public static void readFully(InputStream s, byte[] bytes) throws IllegalArgumentException, IllegalArgumentException, IOException {
570         if (s == null) {
571           throw new IllegalArgumentException("s == null");
572         }
573         if (bytes == null) {
574           throw new IllegalArgumentException("bytes == null");
575         }
576         int offset = 0;
577         while (offset < bytes.length) {
578           int r = s.read(bytes, offset, bytes.length - offset);
579           if (r < 0) {
580             throw new IOException("Class truncated");
581           }
582           offset += r;
583         }
584       }

586       public static byte[] readFully(InputStream s) throws IOException {
587         if (s == null) {
588           throw new IllegalArgumentException("s is null");
589         }
590         byte[] bytes = new byte[s.available()];
591         readFully(s, bytes);
592         int b = s.read();
593         if (b < 0) {
594           return bytes;
595         } else {
596           byte[] big = new byte[bytes.length * 2 + 1];
597           System.arraycopy(bytes, 0, big, 0, bytes.length);
598           big[bytes.length] = (byte) b;
599           int offset = bytes.length + 1;
600           do {
601             if (big.length == offset) {
602               // grow array by factor of 2
603               bytes = new byte[offset * 2];
604               System.arraycopy(big, 0, bytes, 0, offset);
605               big = bytes;
606             }
607             int r = s.read(big, offset, big.length - offset);
608             if (r < 0) {
609               bytes = new byte[offset];
610               System.arraycopy(big, 0, bytes, 0, offset);
611               return bytes;
612             }
613             offset += r;
614           } while (true);
615         }
616       }

618       public static Pair<boolean[], boolean[]> computeBasicBlocks(IInstruction[] instructions, ExceptionHandler[][] handlers) {
619     
620         // Compute r so r[i] == true iff instruction i begins a basic block.
621         boolean[] r = new boolean[instructions.length];
622         boolean[] catchers = new boolean[instructions.length];
623         
624         r[0] = true;
625         for (int i = 0; i < instructions.length; i++) {
626           int[] targets = instructions[i].getBranchTargets();
627     
628           // if there are any targets, then break the basic block here.
629           // also break the basic block after a return
630           if (targets.length > 0 || !instructions[i].isFallThrough()) {
631             if (i + 1 < instructions.length && !r[i + 1]) {
632               r[i + 1] = true;
633             }
634           }
635     
636           for (int j = 0; j < targets.length; j++) {
637             if (!r[targets[j]]) {
638               r[targets[j]] = true;
639             }
640           }
641           if (instructions[i].isPEI()) {
642             ExceptionHandler[] hs = handlers[i];
643             // break the basic block here.
644             if (i + 1 < instructions.length && !r[i + 1]) {
645               r[i + 1] = true;
646             }
647             if (hs != null && hs.length > 0) {
648               for (int j = 0; j < hs.length; j++) {
649                 // exceptionHandlers.add(hs[j]);
650                 if (!r[hs[j].getHandler()]) {
651                   // we have not discovered the catch block yet.
652                   // form a new basic block
653                   r[hs[j].getHandler()] = true;
654                 }
655                 catchers[hs[j].getHandler()] = true;
656               }
657             }
658           }
659         }
660     
661         return Pair.make(r, catchers);
662       }
