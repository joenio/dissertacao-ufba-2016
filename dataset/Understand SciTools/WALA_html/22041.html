93             ArrayList<MethodReference> entryPointMRs = new ArrayList<MethodReference>();

95             // onLocation
96             entryPointMRs.add(StringStuff.makeMethodReference("android.location.LocationListener.onLocationChanged(Landroid/location/Location;)V"));
97             for(MethodReference mr:entryPointMRs)
98             for(IMethod im:cha.getPossibleTargets(mr))
99             {
100                 
101     
102                 // limit to functions defined within the application
103                 if(im.getReference().getDeclaringClass().getClassLoader().
104                                     equals(ClassLoaderReference.Application)) {
105                     
106                     entries.add(new DefaultEntrypoint(im, cha));
107                 }
108             }
109         }

111         public static List<Entrypoint> defaultEntryPoints(ClassHierarchy cha) {
112         	List<Entrypoint> entries = new ArrayList<Entrypoint>();
113         	for (MethodNamePattern mnp:new AndroidSpecs().getEntrypointSpecs()) {
114         		for (IMethod im: mnp.getPossibleTargets(cha)) {
115         			
116         			// limit to functions defined within the application
117         			if(LoaderUtils.fromLoader(im, ClassLoaderReference.Application))
118         			{
119         				
120         				entries.add(new DefaultEntrypoint(im, cha));
121         			}
122         		}
123         	}
124         	return entries;
125         }

296         private void populateIntentList(String basePackage, Element eElement) {
297             ArrayList<String[]> IntentList;
298             NodeList actionList = eElement.getElementsByTagName("action");
299             Node parent = eElement.getParentNode();
300             IntentList = chooseIntentList(parent.getNodeName());
301     
302             String IntentClass = parent.getAttributes().getNamedItem("android:name").getTextContent();
303     
304             for (int i = 0; i < actionList.getLength(); i++)
305             {
306                 Node nNode = actionList.item(i);
307                 if (nNode.getNodeType() == Node.ELEMENT_NODE) {
308                     IntentList.add(new String[2]);
309                     IntentList.get(IntentList.size()-1)[0] = actionList.item(i).getAttributes().getNamedItem("android:name").getTextContent();
310     
311                     if (IntentClass.startsWith(basePackage))
312                         IntentList.get(IntentList.size()-1)[1] = IntentClass;
313                     else {
314                         if (IntentClass.startsWith("."))
315                             IntentList.get(IntentList.size()-1)[1] = basePackage + IntentClass;
316                         else {
317                             IntentList.get(IntentList.size()-1)[1] = basePackage + "." + IntentClass;
318                             IntentList.add(new String[2]);
319                             IntentList.get(IntentList.size()-1)[0] = actionList.item(i).getAttributes().getNamedItem("android:name").getTextContent();
320                             IntentList.get(IntentList.size()-1)[1] = IntentClass;
321                         }
322     
323                         //IntentList.get(IntentList.size()-1)[1] = basePackage + (IntentClass.startsWith(".") ? IntentClass : "." + IntentClass);
324                     }
325     
326                     //System.out.println(IntentList.get(IntentList.size()-1)[0] + " ~> " + IntentList.get(IntentList.size()-1)[1]);
327                 }
328             }
329         }

331         @SuppressWarnings("unused")
332     	private void populateEntryPoints(ClassHierarchy cha) {
333             String method = null;
334             IMethod im = null;
335             for (String[] intent: ActivityIntentList) {
336                 //method = IntentToMethod(intent[0]);
337                 method = "onCreate(Landroid/os/Bundle;)V";
338                 
339                 if (method != null)
340                     im = cha.resolveMethod(StringStuff.makeMethodReference(intent[1]+"."+method));
341                 if (im!=null)
342                     entries.add(new DefaultEntrypoint(im,cha));
343     
344             }
345             for (String[] intent: ReceiverIntentList) {
346                 //Seems that every broadcast receiver can be an entrypoints?
347     //          method = IntentToMethod(intent[0]);
348                 method = "onReceive(Landroid/content/Context;Landroid/content/Intent;)V";
349                 
350                 if (method != null)
351                     im = cha.resolveMethod(StringStuff.makeMethodReference(intent[1]+"."+method));
352                 if (im!=null)
353                     entries.add(new DefaultEntrypoint(im,cha));
354             }
355             //IMethod im = cha.resolveMethod(StringStuff.makeMethodReference("android.app.Activity.onCreate(Landroid/os/Bundle;)V"));
356             //entries.add(new DefaultEntrypoint(im, cha));
357         }

364                     intent.contentEquals("android.media.action.STILL_IMAGE_CAMERA") ||
365                     intent.contentEquals("android.intent.action.MUSIC_PLAYER") ||

378         private ArrayList<String[]> chooseIntentList(String name) {
379             if (name.equals("activity"))
380                 return ActivityIntentList;
381             else if (name.equals("receiver"))
382                 return ReceiverIntentList;
383             else if (name.equals("service"))
384                 return ServiceIntentList;
385             else {
386                 return ActivityIntentList;
387     //          throw new UnimplementedError("EntryPoints intent category not yet covered: " + name);
388             }
389         }
