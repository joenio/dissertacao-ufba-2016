25     /**
26      * Transforms property reads to make prototype chain operations explicit. Each
27      * read is converted to a do loop that walks up the prototype chain until the
28      * property is found or the chain has ended.
29      */
30     public class PropertyReadExpander extends CAstRewriter<PropertyReadExpander.RewriteContext, PropertyReadExpander.ExpanderKey> {
31     
32       static enum ExpanderKey implements CopyKey<ExpanderKey> {
33         EVERYWHERE, EXTRA {
34           @Override
35           public ExpanderKey parent() { return EVERYWHERE; }
36         };
37         @Override
38         public ExpanderKey parent() { return null; }
39       }
40       
41       private int readTempCounter = 0;
42     
43       private static final String TEMP_NAME = "readTemp";
44     
45       abstract static class RewriteContext implements CAstRewriter.RewriteContext<ExpanderKey> {
46     
47         @Override
48         public ExpanderKey key() {
49           return ExpanderKey.EVERYWHERE;
50         }
51     
52         /**
53          * are we in a context where a property access should be treated as a read?
54          * e.g., should return false if we are handling the LHS of an assignment
55          */
56         abstract boolean inRead();
57     
58         /**
59          * are we handling a sub-node of an assignment?
60          */
61         abstract boolean inAssignment();
62     
63         /**
64          * @see AssignPreOrPostOpContext
65          */
66         abstract void setAssign(CAstNode receiverTemp, CAstNode elementTemp);
67       }
68     
69       /**
70        * for handling property reads within assignments with pre or post-ops, e.g.,
71        * x.f++
72        */
73       private final class AssignPreOrPostOpContext extends RewriteContext {
74         private CAstNode receiverTemp;
75     
76         private CAstNode elementTemp;
77     
78         @Override
79         public boolean inAssignment() {
80           return true;
81         }
82     
83         @Override
84         public boolean inRead() {
85           return true;
86         }
87     
88         /**
89          * store the CAstNodes used to represent the loop variable for the
90          * prototype-chain traversal (receiverTemp) and the desired property
91          * (elementTemp)
92          */
93         @Override
94         public void setAssign(CAstNode receiverTemp, CAstNode elementTemp) {
95           this.receiverTemp = receiverTemp;
96           this.elementTemp = elementTemp;
97         }
98     
99       };
100     
101       private final static RewriteContext READ = new RewriteContext() {
102         @Override
103         public boolean inAssignment() {
104           return false;
105         }
106     
107         @Override
108         public boolean inRead() {
109           return true;
110         }
111     
112         @Override
113         public void setAssign(CAstNode receiverTemp, CAstNode elementTemp) {
114           Assertions.UNREACHABLE();
115         }
116       };
117     
118       private final static RewriteContext ASSIGN = new RewriteContext() {
119         @Override
120         public boolean inAssignment() {
121           return true;
122         }
123     
124         @Override
125         public boolean inRead() {
126           return false;
127         }
128     
129         @Override
130         public void setAssign(CAstNode receiverTemp, CAstNode elementTemp) {
131           Assertions.UNREACHABLE();
132         }
133       };
134     
135       public PropertyReadExpander(CAst Ast) {
136         super(Ast, true, READ);
137       }
138     
139       /**
140        * create a CAstNode l representing a loop that traverses the prototype chain
141        * from receiver searching for the constant property element. update nodeMap
142        * to map root to an expression that reads the property from the right node.
143        * 
144        * @param root
145        * @param receiver
146        * @param element
147        * @param context
148        * @param nodeMap
149        * @return
150        */
151       private CAstNode makeConstRead(CAstNode root, CAstNode receiver, CAstNode element, RewriteContext context,
152           Map<Pair<CAstNode, ExpanderKey>, CAstNode> nodeMap) {
153         CAstNode get, result;
154         String receiverTemp = TEMP_NAME + (readTempCounter++);
155         String elt = (String) element.getValue();
156         if (elt.equals("prototype") || elt.equals("__proto__")) {
157           result = Ast.makeNode(CAstNode.BLOCK_EXPR, get = Ast.makeNode(CAstNode.OBJECT_REF, receiver, Ast.makeConstant(elt)));
158         } else {
159     
160           if (context.inAssignment()) {
161             context.setAssign(Ast.makeNode(CAstNode.VAR, Ast.makeConstant(receiverTemp)), Ast.makeConstant(elt));
162           }
163     
164           result = Ast.makeNode(
165              CAstNode.BLOCK_EXPR,
166                   // declare loop variable and initialize to the receiver
167                   Ast.makeNode(CAstNode.DECL_STMT, Ast.makeConstant(new InternalCAstSymbol(receiverTemp, JSAstTranslator.Any, false, false)), receiver),
168                   Ast.makeNode(CAstNode.LOOP,
169                   // while the desired property of the loop variable is not
170                   // defined...
171                       Ast.makeNode(
172                           CAstNode.UNARY_EXPR,
173                           CAstOperator.OP_NOT,
174                           Ast.makeNode(CAstNode.IS_DEFINED_EXPR, Ast.makeNode(CAstNode.VAR, Ast.makeConstant(receiverTemp)),
175                               Ast.makeConstant(elt))),
176                       // set the loop variable to be its prototype
177                       Ast.makeNode(
178                           CAstNode.ASSIGN,
179                           Ast.makeNode(CAstNode.VAR, Ast.makeConstant(receiverTemp)),
180                           Ast.makeNode(CAstNode.OBJECT_REF, Ast.makeNode(CAstNode.VAR, Ast.makeConstant(receiverTemp)),
181                               Ast.makeConstant("__proto__")))),
182                   get = Ast.makeNode(CAstNode.OBJECT_REF, Ast.makeNode(CAstNode.VAR, Ast.makeConstant(receiverTemp)),
183                       Ast.makeConstant(elt)));
184         }
185     
186         nodeMap.put(Pair.make(root, context.key()), result);
187         nodeMap.put(Pair.make(root, ExpanderKey.EXTRA), get);
188     
189         return result;
190       }
191     
192       /**
193        * similar to makeConstRead(), but desired property is some expression instead
194        * of a constant
195        * 
196        * @see #makeConstRead(CAstNode, CAstNode, CAstNode, RewriteContext, Map)
197        */
198       private CAstNode makeVarRead(CAstNode root, CAstNode receiver, CAstNode element, RewriteContext context,
199           Map<Pair<CAstNode, ExpanderKey>, CAstNode> nodeMap) {
200         String receiverTemp = TEMP_NAME + (readTempCounter++);
201         String elementTemp = TEMP_NAME + (readTempCounter++);
202     
203         if (context.inAssignment()) {
204           context.setAssign(Ast.makeNode(CAstNode.VAR, Ast.makeConstant(receiverTemp)),
205               Ast.makeNode(CAstNode.VAR, Ast.makeConstant(elementTemp)));
206         }
207     
208         CAstNode get;
209         CAstNode result = Ast.makeNode(
210             CAstNode.BLOCK_EXPR,
211             Ast.makeNode(CAstNode.DECL_STMT, Ast.makeConstant(new InternalCAstSymbol(receiverTemp, JSAstTranslator.Any, false, false)), receiver),
212             Ast.makeNode(CAstNode.DECL_STMT, Ast.makeConstant(new InternalCAstSymbol(elementTemp, JSAstTranslator.Any, false, false)), element),
213             Ast.makeNode(
214                 CAstNode.LOOP,
215                 Ast.makeNode(
216                     CAstNode.UNARY_EXPR,
217                     CAstOperator.OP_NOT,
218                     Ast.makeNode(CAstNode.IS_DEFINED_EXPR, Ast.makeNode(CAstNode.VAR, Ast.makeConstant(receiverTemp)),
219                         Ast.makeNode(CAstNode.VAR, Ast.makeConstant(elementTemp)))),
220                 Ast.makeNode(
221                     CAstNode.ASSIGN,
222                     Ast.makeNode(CAstNode.VAR, Ast.makeConstant(receiverTemp)),
223                     Ast.makeNode(CAstNode.OBJECT_REF, Ast.makeNode(CAstNode.VAR, Ast.makeConstant(receiverTemp)),
224                         Ast.makeConstant("__proto__")))),
225             get = Ast.makeNode(CAstNode.OBJECT_REF, Ast.makeNode(CAstNode.VAR, Ast.makeConstant(receiverTemp)),
226                 Ast.makeNode(CAstNode.VAR, Ast.makeConstant(elementTemp))));
227     
228         nodeMap.put(Pair.make(root, context.key()), get);
229      
230         return result;
231       }

233       @Override
234       protected CAstNode copyNodes(CAstNode root, final CAstControlFlowMap cfg, RewriteContext context, Map<Pair<CAstNode, ExpanderKey>, CAstNode> nodeMap) {
235         int kind = root.getKind();
236     
237         if (kind == CAstNode.OBJECT_REF && context.inRead()) {
238           // if we see a property access (OBJECT_REF) in a read context, transform
239           // to a loop traversing the prototype chain
240           CAstNode readLoop;
241           CAstNode receiver = copyNodes(root.getChild(0), cfg, READ, nodeMap);
242           CAstNode element = copyNodes(root.getChild(1), cfg, READ, nodeMap);
243           if (element.getKind() == CAstNode.CONSTANT && element.getValue() instanceof String) {
244             readLoop = makeConstRead(root, receiver, element, context, nodeMap);
245           } else {
246             readLoop = makeVarRead(root, receiver, element, context, nodeMap);
247           }
248           return readLoop;
249     
250         } else if (kind == CAstNode.ASSIGN_PRE_OP || kind == CAstNode.ASSIGN_POST_OP) {
251           // handle cases like x.f++, represented as ASSIGN_POST_OP(x.f,1,+)
252           AssignPreOrPostOpContext ctxt = new AssignPreOrPostOpContext();
253           // generate loop for the first child (x.f for example), keeping the loop var and element var in ctxt
254           CAstNode lval = copyNodes(root.getChild(0), cfg, ctxt, nodeMap);
255           CAstNode rval = copyNodes(root.getChild(1), cfg, READ, nodeMap);
256           CAstNode op = copyNodes(root.getChild(2), cfg, READ, nodeMap);
257           if (ctxt.receiverTemp != null) {  // if we found a nested property access
258             String temp1 = TEMP_NAME + (readTempCounter++);
259             String temp2 = TEMP_NAME + (readTempCounter++);
260             CAstNode copy = Ast.makeNode(
261                 CAstNode.BLOCK_EXPR,
262                 // assign lval to temp1 (where lval is a block that includes the prototype chain loop)
263                 Ast.makeNode(CAstNode.DECL_STMT, Ast.makeConstant(new InternalCAstSymbol(temp1, JSAstTranslator.Any, true, false)), lval),
264                 // ? --MS
265                 //rval,
266                 // assign temp2 the new value to be assigned
267                 Ast.makeNode(CAstNode.DECL_STMT, Ast.makeConstant(new InternalCAstSymbol(temp2, JSAstTranslator.Any, true, false)),
268                     Ast.makeNode(CAstNode.BINARY_EXPR, op, Ast.makeNode(CAstNode.VAR, Ast.makeConstant(temp1)), rval)),
269                 // write temp2 into the property
270                 Ast.makeNode(CAstNode.ASSIGN, Ast.makeNode(CAstNode.OBJECT_REF, ctxt.receiverTemp, ctxt.elementTemp),
271                     Ast.makeNode(CAstNode.VAR, Ast.makeConstant(temp2))),
272                 // final value depends on whether we had a pre op or post op    
273                 Ast.makeNode(CAstNode.VAR, Ast.makeConstant((kind == CAstNode.ASSIGN_PRE_OP) ? temp2 : temp1)));
274             nodeMap.put(Pair.make(root, context.key()), copy);
275             return copy;
276           } else {
277             CAstNode copy = Ast.makeNode(kind, lval, rval, op);
278             nodeMap.put(Pair.make(root, context.key()), copy);
279             return copy;
280           }
281     
282         } else if (kind == CAstNode.ASSIGN) {
283           // use ASSIGN context for LHS so we don't translate property accesses there
284           CAstNode copy = Ast.makeNode(CAstNode.ASSIGN, copyNodes(root.getChild(0), cfg, ASSIGN, nodeMap),
285               copyNodes(root.getChild(1), cfg, READ, nodeMap));
286           nodeMap.put(Pair.make(root, context.key()), copy);
287           return copy;
288     
289         } else if (kind == CAstNode.BLOCK_EXPR) {
290           CAstNode children[] = new CAstNode[root.getChildCount()];
291           int last = (children.length - 1);
292           for (int i = 0; i < last; i++) {
293             children[i] = copyNodes(root.getChild(i), cfg, READ, nodeMap);
294           }
295           children[last] = copyNodes(root.getChild(last), cfg, context, nodeMap);
296     
297           CAstNode copy = Ast.makeNode(CAstNode.BLOCK_EXPR, children);
298           nodeMap.put(Pair.make(root, context.key()), copy);
299           return copy;
300     
301         } else if (root.getKind() == CAstNode.CONSTANT) {
302           CAstNode copy = Ast.makeConstant(root.getValue());
303           nodeMap.put(Pair.make(root, context.key()), copy);
304           return copy;
305     
306         } else if (root.getKind() == CAstNode.OPERATOR) {
307           nodeMap.put(Pair.make(root, context.key()), root);
308           return root;
309     
310         } else {
311           CAstNode children[] = new CAstNode[root.getChildCount()];
312           for (int i = 0; i < children.length; i++) {
313             children[i] = copyNodes(root.getChild(i), cfg, READ, nodeMap);
314           }
315           for(Object label: cfg.getTargetLabels(root)) {
316             if (label instanceof CAstNode) {
317               copyNodes((CAstNode)label, cfg, READ, nodeMap);
318             }
319           }
320           CAstNode copy = Ast.makeNode(kind, children);
321           nodeMap.put(Pair.make(root, context.key()), copy);
322           return copy;
323         }
324       }
325     }
