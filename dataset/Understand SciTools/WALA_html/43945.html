11     package com.ibm.wala.demandpa.alg.refinepolicy;
12     
13     import java.util.Collection;
14     
15     import com.ibm.wala.classLoader.IClass;
16     import com.ibm.wala.classLoader.IField;
17     import com.ibm.wala.demandpa.alg.statemachine.StateMachine;
18     import com.ibm.wala.demandpa.flowgraph.IFlowLabel;
19     import com.ibm.wala.demandpa.util.ArrayContents;
20     import com.ibm.wala.ipa.callgraph.propagation.PointerKey;
21     import com.ibm.wala.ipa.cha.IClassHierarchy;
22     import com.ibm.wala.types.ClassLoaderReference;
23     import com.ibm.wala.types.TypeReference;
24     import com.ibm.wala.util.collections.HashSetFactory;
25     
26     public class TunedFieldRefinementPolicy implements FieldRefinePolicy {
27     
28       private static final boolean DEBUG = false;
29     
30       private final IClassHierarchy cha;
31     
32       private final Collection<IClass> typesToRefine = HashSetFactory.make();
33     
34       private IClass firstSkippedClass = null;
35     
36       @Override
37       public boolean nextPass() {
38         if (firstSkippedClass != null) {
39           typesToRefine.add(firstSkippedClass);
40           if (DEBUG) {
41             System.err.println("now refining " + firstSkippedClass);
42           }
43           firstSkippedClass = null;
44           return true;
45         } else {
46           return false;
47         }
48       }
49     
50       @Override
51       public boolean shouldRefine(IField field, PointerKey basePtr, PointerKey val, IFlowLabel label, StateMachine.State state) {
52         if (field == null) {
53           throw new IllegalArgumentException("null field");
54         }
55         if (field == ArrayContents.v()) {
56           return true;
57         }
58         IClass classToCheck = removeInner(field.getDeclaringClass());
59         if (superOfAnyEncountered(classToCheck)) {
60           return true;
61         } else {
62           if (firstSkippedClass == null) {
63             firstSkippedClass = classToCheck;
64           }
65           return false;
66         }
67       }
68     
69       private boolean superOfAnyEncountered(IClass klass) {
70         for (IClass toRefine : typesToRefine) {
71           if (cha.isAssignableFrom(klass, toRefine) || cha.isAssignableFrom(toRefine, klass)) {
72             return true;
73           }
74         }
75         return false;
76       }
77     
78       /**
79        * 
80        * @param klass
81        * @return the top-level {@link IClass} where klass is declared, or klass itself if klass is top-level or if top-level
82        *         class not loaded
83        */
84       private IClass removeInner(IClass klass) {
85         ClassLoaderReference cl = klass.getClassLoader().getReference();
86         String klassStr = klass.getName().toString();
87         int dollarIndex = klassStr.indexOf('$');
88         if (dollarIndex == -1) {
89           return klass;
90         } else {
91           String topMostName = klassStr.substring(0, dollarIndex);
92           IClass topMostClass = cha.lookupClass(TypeReference.findOrCreate(cl, topMostName));
93           return (topMostClass != null) ? topMostClass : klass;
94         }
95       }
96     
97       public TunedFieldRefinementPolicy(IClassHierarchy cha) {
98         this.cha = cha;
99       }
100     
101     }
