16     /**
17      * A load from a pointer. 
18      * 
19      * v = *p
20      */
21     public class SSALoadIndirectInstruction extends SSAAbstractUnaryInstruction {
22     
23       private final TypeReference loadedType;
24       
25       /**
26        * @param lval the value number which is def'fed by this instruction.
27        * @param addressVal the value number holding the pointer p deferenced (*p)
28        */
29       public SSALoadIndirectInstruction(int iindex, int lval, TypeReference t, int addressVal) {
30         super(iindex, lval, addressVal);
31         this.loadedType = t;
32       }
33     
34       public TypeReference getLoadedType() {
35         return loadedType;
36       }
37       
38       @Override
39       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
40         Assertions.UNREACHABLE("not implemented");
41         return null;
42       }
43     
44       @Override
45       public String toString(SymbolTable symbolTable) {
46         return getValueString(symbolTable, getDef(0)) + " =  *"  + getValueString(symbolTable, getUse(0)) + ": " + loadedType;
47       }
48     
49       @Override
50       public void visit(IVisitor v) {
51         ((IVisitorWithAddresses)v).visitLoadIndirect(this);
52       }
53     
54     }
