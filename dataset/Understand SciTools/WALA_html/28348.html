11     package com.ibm.wala.shrikeBT.analysis;
12     
13     /**
14      * This interface provides information about the class hierarchy to some consumer, such as a bytecode verifier.
15      * 
16      * All class names are given in JVM type format, e.g., Ljava/lang/Object;.
17      */
18     public interface ClassHierarchyProvider {
19       /**
20        * @return the superclass of the given class, or null if the superclass is not known or cl is java.lang.Object
21        */
22       public String getSuperClass(String cl);
23     
24       /**
25        * @return the superinterfaces of the given class, or null if they are not known
26        */
27       public String[] getSuperInterfaces(String cl);
28     
29       /**
30        * @return the complete set of direct subclasses or implementors of cl, or null if the complete set is not known
31        */
32       public String[] getSubClasses(String cl);
33     
34       /**
35        * @return whether or not cl is an interface, or Constants.MAYBE if not known
36        */
37       public int isInterface(String cl);
38     }
