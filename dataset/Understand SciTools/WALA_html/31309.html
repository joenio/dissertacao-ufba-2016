26     /**
27      * 0-x-CFA Call graph builder, optimized to not disambiguate instances of
28      * "uninteresting" types
29      */
30     public class JSZeroOrOneXCFABuilder extends JSCFABuilder {
31     
32       private static final boolean USE_OBJECT_SENSITIVITY = false;
33       
34     
35       public JSZeroOrOneXCFABuilder(IClassHierarchy cha, JSAnalysisOptions options, AnalysisCache cache,
36           ContextSelector appContextSelector, SSAContextInterpreter appContextInterpreter, int instancePolicy, boolean doOneCFA) {
37         super(cha, options, cache);
38     
39         SSAContextInterpreter contextInterpreter = setupSSAContextInterpreter(cha, options, cache, appContextInterpreter);
40     
41         setupMethodTargetSelector(cha, options);
42     
43         setupContextSelector(options, appContextSelector, doOneCFA);
44     
45         setInstanceKeys(new JavaScriptScopeMappingInstanceKeys(cha, this, new JavaScriptConstructorInstanceKeys(new ZeroXInstanceKeys(
46             options, cha, contextInterpreter, instancePolicy))));
47       }
48     
49       private void setupContextSelector(JSAnalysisOptions options, ContextSelector appContextSelector, boolean doOneCFA) {
50         // baseline selector
51         ContextSelector def = new ContextInsensitiveSelector();
52         ContextSelector contextSelector = appContextSelector == null ? def : new DelegatingContextSelector(appContextSelector, def);
53         
54         // JavaScriptConstructorContextSelector ensures at least a 0-1-CFA (i.e.,
55         // Andersen's-style) heap abstraction. This level of heap abstraction is
56         // _necessary_ for correctness (we rely on it when handling lexical scoping)
57         contextSelector = new JavaScriptConstructorContextSelector(contextSelector);
58         
59         //contextSelector = new OneLevelForLexicalAccessFunctions(contextSelector);
60         
61         if (USE_OBJECT_SENSITIVITY) {
62           contextSelector = new ObjectSensitivityContextSelector(contextSelector);
63         }
64         if (options.handleCallApply()) {
65           contextSelector = new JavaScriptFunctionApplyContextSelector(contextSelector);
66         }
67         if (doOneCFA) {
68           contextSelector = new nCFAContextSelector(1, contextSelector);
69         }
70         setContextSelector(contextSelector);
71       }
72     
73       
74       private void setupMethodTargetSelector(IClassHierarchy cha, JSAnalysisOptions options) {
75         MethodTargetSelector targetSelector = new JavaScriptConstructTargetSelector(cha, options
76             .getMethodTargetSelector());
77         if (options.handleCallApply()) {
78           targetSelector = new JavaScriptFunctionApplyTargetSelector(new JavaScriptFunctionDotCallTargetSelector(targetSelector));
79         }
80         if (options.useLoadFileTargetSelector()) {
81           targetSelector = new LoadFileTargetSelector(targetSelector, this);
82         }
83         options.setSelector(targetSelector);
84       }
85     
86       private SSAContextInterpreter setupSSAContextInterpreter(IClassHierarchy cha, JSAnalysisOptions options, AnalysisCache cache,
87           SSAContextInterpreter appContextInterpreter) {
88         SSAContextInterpreter contextInterpreter = makeDefaultContextInterpreters(appContextInterpreter, options, cha);
89         if (options.handleCallApply()) {
90           contextInterpreter = new DelegatingSSAContextInterpreter(new JavaScriptFunctionApplyContextInterpreter(options, cache),
91               contextInterpreter);
92         }
93         setContextInterpreter(contextInterpreter);
94         return contextInterpreter;
95       }
96     
97       /**
98        * @param options
99        *          options that govern call graph construction
100        * @param cha
101        *          governing class hierarchy
102        * @param cl
103        *          classloader that can find DOMO resources
104        * @param scope
105        *          representation of the analysis scope
106        * @param xmlFiles
107        *          set of Strings that are names of XML files holding bypass logic
108        *          specifications.
109        * @return a 0-1-Opt-CFA Call Graph Builder.
110        */
111       public static JSCFABuilder make(JSAnalysisOptions options, AnalysisCache cache, IClassHierarchy cha, ClassLoader cl,
112           AnalysisScope scope, String[] xmlFiles, byte instancePolicy, boolean doOneCFA) {
113         com.ibm.wala.ipa.callgraph.impl.Util.addDefaultSelectors(options, cha);
114         for (int i = 0; i < xmlFiles.length; i++) {
115           com.ibm.wala.ipa.callgraph.impl.Util.addBypassLogic(options, scope, cl, xmlFiles[i], cha);
116         }
117     
118         return new JSZeroOrOneXCFABuilder(cha, options, cache, null, null, instancePolicy, doOneCFA);
119       }
120     
121     }
