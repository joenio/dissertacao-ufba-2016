11     package com.ibm.wala.viz.viewer;
12     
13     import java.awt.event.WindowAdapter;
14     import java.awt.event.WindowEvent;
15     
16     import javax.swing.JFrame;
17     import javax.swing.JTabbedPane;
18     import javax.swing.UIManager;
19     
20     import com.ibm.wala.ipa.callgraph.CallGraph;
21     import com.ibm.wala.ipa.callgraph.propagation.InstanceKey;
22     import com.ibm.wala.ipa.callgraph.propagation.PointerAnalysis;
23     
24     /**
25      * Viewer for ClassHeirarcy, CallGraph and Pointer Analysis results.
26      * A driver for example can be found in com.ibm.wala.js.rhino.vis.JsViewer.
27      * @author yinnonh
28      *
29      */
30     public class WalaViewer extends JFrame {
31     
32       protected static final String DefaultMutableTreeNode = null;
33     
34     
35       public WalaViewer(CallGraph cg, PointerAnalysis<InstanceKey> pa) {
36         setNativeLookAndFeel();
37         
38         JTabbedPane tabbedPane = new JTabbedPane();
39         tabbedPane.add("Call Graph", new CgPanel(cg));
40         tabbedPane.add("Class Hierarchy", new ChaPanel(cg.getClassHierarchy()));
41         PaPanel paPanel = createPaPanel(cg, pa);
42         paPanel.init();
43         tabbedPane.add("Pointer Analysis", paPanel);
44     
45         setSize(600, 800);
46         setExtendedState(MAXIMIZED_BOTH);
47         addWindowListener(new ExitListener());
48     
49         this.setTitle("Wala viewer");
50         
51         add(tabbedPane);
52         setVisible(true);
53       }
54     
55       protected PaPanel createPaPanel(CallGraph cg, PointerAnalysis<InstanceKey> pa) {
56         return new PaPanel(cg, pa);
57       }
58       
59       public static void setNativeLookAndFeel() {
60         try {
61           UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
62         } catch(Exception e) {
63           e.printStackTrace();
64         }
65       }
66       
67       private static class ExitListener extends WindowAdapter {
68         @Override
69         public void windowClosing(WindowEvent event) {
70           System.exit(0);
71         }
72       }
73     }
