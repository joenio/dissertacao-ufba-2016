14     package com.ibm.wala.cast.tree.impl;
15     
16     import java.util.Iterator;
17     import java.util.Map;
18     
19     import com.ibm.wala.cast.tree.CAstReference;
20     import com.ibm.wala.cast.tree.CAstType;
21     import com.ibm.wala.cast.tree.CAstTypeDictionary;
22     import com.ibm.wala.util.collections.HashMapFactory;
23     
24     public class CAstTypeDictionaryImpl implements CAstTypeDictionary {
25       protected final Map fMap = HashMapFactory.make();
26     
27       @Override
28       public CAstType getCAstTypeFor(Object/*ASTType*/ astType) {
29           return (CAstType) fMap.get(astType);
30       }
31     
32       @SuppressWarnings("unchecked")
33       public void map(Object/*ASTType*/ astType, CAstType castType) {
34         fMap.put(astType, castType);
35       }
36     
37       @Override
38       public Iterator iterator() {
39         return fMap.values().iterator();
40       }
41     
42       @Override
43       public CAstReference resolveReference(CAstReference ref) {
44         return ref;
45       }
46     }
