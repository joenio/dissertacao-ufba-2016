11     package com.ibm.wala.cast.js.translator;

13     import com.ibm.wala.cast.ir.translator.ConstantFoldingRewriter;
14     import com.ibm.wala.cast.tree.CAst;
15     import com.ibm.wala.cast.tree.impl.CAstOperator;
16     
17     public class JSConstantFoldingRewriter extends ConstantFoldingRewriter {
18     
19       public JSConstantFoldingRewriter(CAst Ast) {
20         super(Ast);
21       }
22     
23       @Override
24       protected Object eval(CAstOperator op, Object lhs, Object rhs) {
25         if (op == CAstOperator.OP_ADD) {
26           if (lhs instanceof String || rhs instanceof String) {
27             return "" + lhs + rhs;
28           } else if (lhs instanceof Number && rhs instanceof Number) {
29             return new Double(((Number)lhs).doubleValue() + ((Number)rhs).doubleValue());
30           } 
31         } else if (op == CAstOperator.OP_BIT_AND) {
32     
33         } else if (op == CAstOperator.OP_BIT_OR) {
34     
35         } else if (op == CAstOperator.OP_BIT_XOR) {
36     
37         } else if (op == CAstOperator.OP_BITNOT) {
38     
39         } else if (op == CAstOperator.OP_CONCAT) {
40           if (lhs instanceof String || rhs instanceof String) {
41             return "" + lhs + rhs;
42           }
43           
44         } else if (op == CAstOperator.OP_DIV) {
45           if (lhs instanceof Number && rhs instanceof Number) {
46             return new Double(((Number)lhs).doubleValue() / ((Number)rhs).doubleValue());
47           } 
48           
49         } else if (op == CAstOperator.OP_EQ) {
50     
51         } else if (op == CAstOperator.OP_GE) {
52           if (lhs instanceof Number && rhs instanceof Number) {
53             return Boolean.valueOf(((Number)lhs).doubleValue() >= ((Number)rhs).doubleValue());
54           } 
55            
56         } else if (op == CAstOperator.OP_GT) {
57           if (lhs instanceof Number && rhs instanceof Number) {
58             return Boolean.valueOf(((Number)lhs).doubleValue() > ((Number)rhs).doubleValue());
59           } 
60     
61         } else if (op == CAstOperator.OP_LE) {
62           if (lhs instanceof Number && rhs instanceof Number) {
63             return Boolean.valueOf(((Number)lhs).doubleValue() <= ((Number)rhs).doubleValue());
64           } 
65     
66         } else if (op == CAstOperator.OP_LSH) {
67     
68         } else if (op == CAstOperator.OP_LT) {
69           if (lhs instanceof Number && rhs instanceof Number) {
70             return Boolean.valueOf(((Number)lhs).doubleValue() < ((Number)rhs).doubleValue());
71           } 
72     
73         } else if (op == CAstOperator.OP_MOD) {
74           if (lhs instanceof Number && rhs instanceof Number) {
75             return new Double(((Number)lhs).doubleValue() % ((Number)rhs).doubleValue());
76           } 
77           
78         } else if (op == CAstOperator.OP_MUL) {
79           if (lhs instanceof Number && rhs instanceof Number) {
80             return new Double(((Number)lhs).doubleValue() * ((Number)rhs).doubleValue());
81           } 
82           
83         } else if (op == CAstOperator.OP_NE) {
84     
85         } else if (op == CAstOperator.OP_NOT) {
86     
87         } else if (op == CAstOperator.OP_REL_AND) {
88     
89         } else if (op == CAstOperator.OP_REL_OR) {
90     
91         } else if (op == CAstOperator.OP_REL_XOR) {
92     
93         } else if (op == CAstOperator.OP_RSH) {
94     
95         } else if (op == CAstOperator.OP_STRICT_EQ) {
96     
97         } else if (op == CAstOperator.OP_STRICT_NE) {
98     
99         } else if (op == CAstOperator.OP_SUB) {
100           if (lhs instanceof Number && rhs instanceof Number) {
101             return new Double(((Number)lhs).doubleValue() - ((Number)rhs).doubleValue());
102           } 
103           
104         } else if (op == CAstOperator.OP_URSH) {
105     
106         }
107         
108         // no constant value
109         return null;
110       }
111     
112     }
