27     /**
28      * This is a demo class.
29      * 
30      * Class files are taken as input arguments (or if there are none, from standard input). The methods in those files are
31      * instrumented: we insert a System.err.println() at ever method call, and a System.err.println() at every method entry.
32      * 
33      * In Unix, I run it like this: java -cp ~/dev/shrike/shrike com.ibm.wala.shrikeBT.shrikeCT.tools.Bench test.jar -o output.jar
34      * 
35      * The instrumented classes are placed in the directory "output" under the current directory. Disassembled code is written to the
36      * file "report" under the current directory.
37      * 
38      * @author Rob O' Callahan
39      */
40     public class AddBytecodeDebug {
41       private static OfflineInstrumenter instrumenter;
42     
43       public static void main(String[] args) throws Exception {
44         for (int i = 0; i < 1; i++) {
45           instrumenter = new OfflineInstrumenter(true);
46     
47           Writer w = new BufferedWriter(new FileWriter("report", false));
48     
49           args = instrumenter.parseStandardArgs(args);
50           instrumenter.setPassUnmodifiedClasses(true);
51           instrumenter.beginTraversal();
52           ClassInstrumenter ci;
53           while ((ci = instrumenter.nextClass()) != null) {
54             doClass(ci, w);
55           }
56           instrumenter.close();
57         }
58       }
59     
60       private static void doClass(final ClassInstrumenter ci, Writer w) throws Exception {
61         final String className = ci.getReader().getName();
62         w.write("Class: " + className + "\n");
63         w.flush();
64     
65         ci.enableFakeLineNumbers(10000);
66     
67         for (int m = 0; m < ci.getReader().getMethodCount(); m++) {
68           MethodData d = ci.visitMethod(m);
69           if (d != null) {
70             d.setHasChanged();
71     
72             MethodEditor me = new MethodEditor(d);
73             me.beginPass();
74             ExceptionHandler[][] handlers = me.getHandlers();
75             boolean[] putDumperAt = new boolean[handlers.length];
76             for (int i = 0; i < handlers.length; i++) {
77               for (int j = 0; j < handlers[i].length; j++) {
78                 int offset = handlers[i][j].getHandler();
79                 if (!putDumperAt[offset]) {
80                   putDumperAt[offset] = true;
81                   me.insertBefore(offset, new MethodEditor.Patch() {
82                     @Override
83                     public void emitTo(Output w) {
84                       w.emit(DupInstruction.make(0));
85                       w.emit(Util.makeInvoke(Throwable.class, "printStackTrace", new Class[0]));
86                     }
87                   });
88                 }
89               }
90             }
91             me.applyPatches();
92           }
93         }
94     
95         if (ci.isChanged()) {
96           ClassWriter cw = ci.emitClass();
97           instrumenter.outputModifiedClass(ci, cw);
98         }
99       }
100     }
