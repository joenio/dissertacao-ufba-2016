202       @Override
203       public IField getField(Atom name) {
204         if (fieldMap.containsKey(name)) {
205           return fieldMap.get(name);
206         } else {
207           List<IField> fields = findDeclaredField(name);
208           if (!fields.isEmpty()) {
209             if (fields.size() == 1) {
210               IField f = fields.iterator().next();
211               fieldMap.put(name, f);
212               return f;
213             } else {
214               throw new IllegalStateException("multiple fields with name " + name);
215             }
216           } else if ((superClass = getSuperclass()) != null) {
217             IField f = superClass.getField(name);
218             if (f != null) {
219               fieldMap.put(name, f);
220               return f;
221             }
222           }
223           // try superinterfaces
224           for (IClass i : getAllImplementedInterfaces()) {
225             IField f = i.getField(name);
226             if (f != null) {
227               fieldMap.put(name, f);
228               return f;
229             }
230           }
231         }
232     
233         return null;
234       }

237       @Override
238       public IField getField(Atom name, TypeName type) {
239         boolean unresolved = false;
240         try {
241           // typically, there will be at most one field with the name
242           IField field = getField(name);
243           if (field != null && field.getFieldTypeReference().getName().equals(type)) {
244             return field;
245           } else {
246             unresolved = true;
247           }
248         } catch (IllegalStateException e) {
249           assert e.getMessage().startsWith("multiple fields with");
250           unresolved = true;
251         }
252         
253         if(unresolved){
254           // multiple fields.  look through all of them and see if any have the appropriate type
255           List<IField> fields = findDeclaredField(name);
256           for (IField f : fields) {
257             if (f.getFieldTypeReference().getName().equals(type)) {
258               return f;
259             }
260           }
261           // check superclass
262           if (getSuperclass() != null) {
263             IField f = superClass.getField(name, type);
264             if (f != null) {
265               return f;
266             }
267           }
268           // try superinterfaces
269           for (IClass i : getAllImplementedInterfaces()) {
270             IField f = i.getField(name, type);
271             if (f != null) {
272               return f;
273             }
274           }
275         }
276         return null;
277       }

279       private void computeSuperclass() {
280         superclassComputed = true;
281     
282         if (superName == null) {
283           if (!getReference().equals(loader.getLanguage().getRootType())) {
284             superClass = loader.lookupClass(loader.getLanguage().getRootType().getName());
285           }
286           return;
287         }
288     
289         superClass = loader.lookupClass(TypeName.findOrCreate(superName));
290       }

292       @Override
293       public IClass getSuperclass() {
294         if (!superclassComputed) {
295           computeSuperclass();
296         }
297         if (superClass == null && !getReference().equals(TypeReference.JavaLangObject)) {
298           throw new IllegalStateException("No superclass found for " + this + " Superclass name " + superName);
299         }
300         return superClass;
301       }

387       @Override
388       public Collection<IMethod> getAllMethods() {
389         Collection<IMethod> result = new LinkedList<IMethod>();
390         Iterator<IMethod> declaredMethods = getDeclaredMethods().iterator();
391         while (declaredMethods.hasNext()) {
392           result.add(declaredMethods.next());
393         }
394         if (isInterface()) {
395           for (IClass i : getDirectInterfaces()) {
396             result.addAll(i.getAllMethods());
397           }
398         }
399         IClass s = getSuperclass();
400         while (s != null) {
401           Iterator<IMethod> superDeclaredMethods = s.getDeclaredMethods().iterator();
402           while (superDeclaredMethods.hasNext()) {
403             result.add(superDeclaredMethods.next());
404           }
405           s = s.getSuperclass();
406         }
407         return result;
408       }

424       /*
425        * @see com.ibm.wala.classLoader.IClass#getMethod(com.ibm.wala.types.Selector)
426        */
427       @Override
428       public IMethod getMethod(Selector selector) {
429         try {
430           computeMethodMapIfNeeded();
431         } catch (InvalidClassFileException e1) {
432           e1.printStackTrace();
433           Assertions.UNREACHABLE();
434         }
435     
436         // my methods + cached parent stuff
437         IMethod result = methodMap.get(selector);
438         if (result != null) {
439           return result;
440         }
441         if (inheritCache != null) {
442           result = inheritCache.get(selector);
443           if (result != null) {
444             return result;
445           }
446         }
447     
448         // check parent, caching if found
449         if (!selector.equals(MethodReference.clinitSelector) && !selector.equals(MethodReference.initSelector)) {
450           IClass superclass = getSuperclass();
451           if (superclass != null) {
452             IMethod inherit = superclass.getMethod(selector);
453             if (inherit != null) {
454               if (inheritCache == null) {
455                 inheritCache = new BimodalMap<Selector, IMethod>(5);
456               }
457               inheritCache.put(selector, inherit);
458               return inherit;
459             }
460           }
461         }
462         
463         // check interfaces for Java 8 default implementation
464         // Java seems to require a single default implementation, so take that on faith here
465         for(IClass iface : getAllImplementedInterfaces()) {
466           for(IMethod m : iface.getDeclaredMethods()) {
467             if (!m.isAbstract() && m.getSelector().equals(selector)) {          
468               if (inheritCache == null) {
469                 inheritCache = new BimodalMap<Selector, IMethod>(5);
470               }
471               inheritCache.put(selector, m);
472     
473               return m;
474             }
475           }
476         }
477         
478         // no method found
479         if (inheritCache == null) {
480           inheritCache = new BimodalMap<Selector, IMethod>(5);
481         }
482         inheritCache.put(selector, null);
483         return null;
484       }

493       /**
494        * @return Collection of IClasses, representing the interfaces this class implements.
495        */
496       protected Collection<IClass> computeAllInterfacesAsCollection() {
497         Collection<? extends IClass> c = getDirectInterfaces();
498         Set<IClass> result = HashSetFactory.make();
499         for (Iterator<? extends IClass> it = c.iterator(); it.hasNext();) {
500           IClass klass = it.next();
501           if (klass.isInterface()) {
502             result.add(klass);
503           } else {
504             Warnings.add(ClassHierarchyWarning.create("expected an interface " + klass));
505           }
506         }
507     
508         // at this point result holds all interfaces the class directly extends.
509         // now expand to a fixed point.
510         Set<IClass> last = null;
511         do {
512           last = HashSetFactory.make(result);
513           for (IClass i : last) {
514             result.addAll(i.getDirectInterfaces());
515           }
516         } while (last.size() < result.size());
517     
518         // now add any interfaces implemented by the super class
519         IClass sup = null;
520         sup = getSuperclass();
521         if (sup != null) {
522           result.addAll(sup.getAllImplementedInterfaces());
523         }
524         return result;
525       }

527       /**
528        * @param interfaces a set of class names
529        * @return Set of all IClasses that can be loaded corresponding to the class names in the interfaces array; raise warnings if
530        *         classes can not be loaded
531        */
532       private Collection<IClass> array2IClassSet(ImmutableByteArray[] interfaces) {
533         ArrayList<IClass> result = new ArrayList<IClass>(interfaces.length);
534         for (int i = 0; i < interfaces.length; i++) {
535           ImmutableByteArray name = interfaces[i];
536           IClass klass = null;
537           klass = loader.lookupClass(TypeName.findOrCreate(name));
538           if (klass == null) {
539             Warnings.add(ClassNotFoundWarning.create(name));
540           } else {
541             result.add(klass);
542           }
543         }
544         return result;
545       }

547       protected List<IField> findDeclaredField(Atom name) {
548         
549         List<IField> result = new ArrayList<IField>(1);
550         
551         if (instanceFields != null) {
552           for (int i = 0; i < instanceFields.length; i++) {
553             if (instanceFields[i].getName() == name) {
554               result.add(instanceFields[i]);
555             }
556           }
557         }
558     
559         if (staticFields != null) {
560           for (int i = 0; i < staticFields.length; i++) {
561             if (staticFields[i].getName() == name) {
562               result.add(staticFields[i]);
563             }
564           }
565         }
566     
567         return result;
568       }

570       protected void addFieldToList(List<FieldImpl> L, Atom name, ImmutableByteArray fieldType, int accessFlags,
571           Collection<Annotation> annotations, TypeSignature sig) {
572         TypeName T = null;
573         if (fieldType.get(fieldType.length() - 1) == ';') {
574           T = TypeName.findOrCreate(fieldType, 0, fieldType.length() - 1);
575         } else {
576           T = TypeName.findOrCreate(fieldType);
577         }
578         TypeReference type = TypeReference.findOrCreate(getClassLoader().getReference(), T);
579         FieldReference fr = FieldReference.findOrCreate(getReference(), name, type);
580         FieldImpl f = new FieldImpl(this, fr, accessFlags, annotations, sig);
581         L.add(f);
582       }

584       /**
585        * set up the methodMap mapping
586        */
587       protected void computeMethodMapIfNeeded() throws InvalidClassFileException {
588         if (methodMap == null) {
589           synchronized (this) {
590             if (methodMap == null) {
591               IMethod[] methods = computeDeclaredMethods();
592               
593               final Map<Selector, IMethod> tmpMethodMap;
594               if (methods.length > 5) {
595                 tmpMethodMap = HashMapFactory.make(methods.length);
596               } else {
597                 tmpMethodMap= new SmallMap<Selector, IMethod>();
598               }
599               for (int i = 0; i < methods.length; i++) {
600                 IMethod m = methods[i];
601                 tmpMethodMap.put(m.getReference().getSelector(), m);
602               }
603               
604               methodMap = tmpMethodMap;
605             }
606           }
607         }
608       }
