465         /**
466          * Evaluate a meet of locals of machine states at the entry to a basic block.
467          *
468          * TODO: add some efficiency shortcuts. TODO: clean up and refactor.
469          *
470          * @param bb the basic block at whose entry the meet occurs
471          * @return true if the lhs value changes. false otherwise.
472          */
473         private boolean meetLocals(IVariable lhs, IVariable[] rhs, BasicBlock bb, Meeter meeter) {
474     
475             boolean changed = false;
476             MachineState L = (MachineState) lhs;
477             // need we allocate lhs.locals?
478             int nLocals = computeMeetNLocals(rhs);
479             if (nLocals > -1 && L.locals == null) {
480                 L.allocateLocals();
481                 changed = true;
482             }
483     
484             // evaluate the element-wise meet over the locals.
485             for (int i = 0; i < nLocals; i++) {
486                 int[] R = new int[rhs.length];
487                 for (int j = 0; j < rhs.length; j++) {
488                     R[j] = ((MachineState) rhs[j]).getLocal(i);
489                 }
490                 int meet = meeter.meetLocal(i, R, bb);
491                 if (L.locals[i] == TOP) {
492                     if (meet != TOP) {
493                         changed = true;
494                         L.locals[i] = meet;
495                     }
496                 } else if (meet != L.locals[i]) {
497                     changed = true;
498                     L.locals[i] = meet;
499                 }
500             }
501             return changed;
502         }

504         /**
505          * @return the number of locals to meet. Return -1 if there is no local meet necessary.
506          * @param operands The operands for this operator. operands[0] is the left-hand side.
507          */
508         private static int computeMeetNLocals(IVariable[] operands) {
509             MachineState lhs = (MachineState) operands[0];
510             int nLocals = -1;
511             if (lhs.locals != null) {
512                 nLocals = lhs.locals.length;
513             } else {
514                 for (int i = 1; i < operands.length; i++) {
515                     MachineState rhs = (MachineState) operands[i];
516                     if (rhs.locals != null) {
517                         nLocals = rhs.locals.length;
518                         break;
519                     }
520                 }
521             }
522             return nLocals;
523         }

525         /**
526          * @return the height of stacks that are being meeted. Return -1 if there is no stack meet necessary.
527          * @param operands The operands for this operator. operands[0] is the left-hand side.
528          */
529         @SuppressWarnings("unused")
530     	private static int computeMeetStackHeight(IVariable[] operands) {
531             MachineState lhs = (MachineState) operands[0];
532             int height = -1;
533             if (lhs.stack != null) {
534                 height = lhs.stackHeight;
535             } else {
536                 for (int i = 1; i < operands.length; i++) {
537                     MachineState rhs = (MachineState) operands[i];
538                     if (rhs.stack != null) {
539                         height = rhs.stackHeight;
540                         break;
541                     }
542                 }
543             }
544             return height;
545         }

712             public void copyState(MachineState other) {
713                 if (other.stack == null) {
714                     stack = null;
715                 } else {
716                     stack = new int[other.stack.length];
717                     System.arraycopy(other.stack, 0, stack, 0, other.stack.length);
718                 }
719                 if (other.locals == null) {
720                     locals = null;
721                 } else {
722                     locals = new int[other.locals.length];
723                     System.arraycopy(other.locals, 0, locals, 0, other.locals.length);
724                 }
725                 stackHeight = other.stackHeight;
726             }
727     
728             boolean stateEquals(MachineState exit) {
729                 if (stackHeight != exit.stackHeight)
730                     return false;
731                 if (locals == null) {
732                     if (exit.locals != null)
733                         return false;
734                 } else {
735                     if (exit.locals == null)
736                         return false;
737                     else if (locals.length != exit.locals.length)
738                         return false;
739                 }
740     
741                 for (int i = 0; i < stackHeight; i++) {
742                     if (stack[i] != exit.stack[i])
743                         return false;
744                 }
745                 if (locals != null) {
746                     for (int i = 0; i < locals.length; i++) {
747                         if (locals[i] == TOP) {
748                             if (exit.locals[i] != TOP)
749                                 return false;
750                         }
751                         if (locals[i] != exit.locals[i])
752                             return false;
753                     }
754                 }
755                 return true;
756             }

764                  return stackHeight;
