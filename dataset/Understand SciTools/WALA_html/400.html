15     /**
16      * Special case: if supportsMerge(), then the problem is not really IFDS anymore. (TODO: rename it?). Instead, we perform a merge
17      * operation before propagating at every program point. This way, we can implement standard interprocedural dataflow and ESP-style
18      * property simulation, and various other things.
19      */
20     public interface IMergeFunction {
21     
22       /**
23        * @param x set of factoid numbers that previously have been established to hold at a program point
24        * @param j a new factoid number which has been discovered to hold at a program point
25        * @return the factoid number z which should actually be propagated, based on a merge of the new fact j into the old state
26        *         represented by x. return -1 if no fact should be propagated.
27        */
28       int merge(IntSet x, int j);
29     
30     }
