15     import org.eclipse.core.runtime.CoreException;
16

25     import com.ibm.wala.util.debug.Assertions;
26     import com.ibm.wala.util.io.FileProvider;
27     
28     /**
29      * An {@link EclipseProjectAnalysisEngine} specialized for source code analysis
30      */
31     abstract public class EclipseProjectSourceAnalysisEngine<P> extends EclipseProjectAnalysisEngine<P> {
32     
33       public static final String defaultFileExt = "java";
34     
35       /**
36        * file extension for source files in this Eclipse project
37        */
38       final String fileExt;
39     
40       public EclipseProjectSourceAnalysisEngine(P project) throws IOException, CoreException {
41         this(project, defaultFileExt);
42       }
43     
44       public EclipseProjectSourceAnalysisEngine(P project, String fileExt) throws IOException, CoreException {
45         super(project);
46         this.fileExt = fileExt;
47         try {
48           setExclusionsFile((new EclipseFileProvider()).getFileFromPlugin(CorePlugin.getDefault(), "J2SEClassHierarchyExclusions.txt")
49               .getAbsolutePath());
50         } catch (IOException e) {
51           try {
52             setExclusionsFile((new FileProvider()).getFile("J2SEClassHierarchyExclusions.txt", getClass().getClassLoader())
53               .getAbsolutePath());
54           } catch (IOException f) {
55             Assertions.UNREACHABLE("Cannot find exclusions file");
56           }
57         }
58       }
59     
60       /**
61        * we don't provide a default implementation of this method to avoid
62        * introducing a dependence on com.ibm.wala.cast from this project
63        */
64       @Override
65       public abstract AnalysisCache makeDefaultCache();
66       
67       protected abstract ClassLoaderReference getSourceLoader();
68     
69       @Override
70       public AnalysisOptions getDefaultOptions(Iterable<Entrypoint> entrypoints) {
71         AnalysisOptions options = new AnalysisOptions(getScope(), entrypoints);
72     
73         SSAOptions ssaOptions = new SSAOptions();
74         ssaOptions.setDefaultValues(new SSAOptions.DefaultValues() {
75           @Override
76           public int getDefaultValue(SymbolTable symtab, int valueNumber) {
77             return symtab.getDefaultValue(valueNumber);
78           }
79         });
80     
81         options.setSSAOptions(ssaOptions);
82     
83         return options;
84       }
85     }
