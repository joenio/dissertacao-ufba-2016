15     package com.ibm.wala.cast.ir.translator;
16     
17     import java.util.Collection;
18     import java.util.Iterator;
19     import java.util.Map;
20     
21     import com.ibm.wala.cast.tree.CAstAnnotation;
22     import com.ibm.wala.cast.tree.CAstEntity;
23     import com.ibm.wala.cast.tree.CAstNode;
24     import com.ibm.wala.cast.tree.CAstSourcePositionMap.Position;
25     import com.ibm.wala.util.collections.EmptyIterator;
26     import com.ibm.wala.util.collections.HashMapFactory;
27     import com.ibm.wala.util.collections.HashSetFactory;
28     import com.ibm.wala.util.debug.Assertions;
29     
30     public abstract class AbstractEntity implements CAstEntity {
31       private Position sourcePosition;
32     
33       private final Map<CAstNode, Collection<CAstEntity>> scopedEntities = HashMapFactory.make();
34     
35       @Override
36       public Map<CAstNode, Collection<CAstEntity>> getAllScopedEntities() {
37         return scopedEntities;
38       }
39     
40       @Override
41       public String getSignature() {
42         Assertions.UNREACHABLE();
43         return null;
44       }
45     
46       
47       @Override
48       public Collection<CAstAnnotation> getAnnotations() {
49         return null;
50       }
51     
52       public void setPosition(Position pos) {
53         sourcePosition = pos;
54       }
55     
56       @Override
57       public Position getPosition() {
58         return sourcePosition;
59       }
60     
61       @Override
62       public Iterator<CAstEntity> getScopedEntities(CAstNode construct) {
63         if (scopedEntities.containsKey(construct)) {
64           return scopedEntities.get(construct).iterator();
65         } else {
66           return EmptyIterator.instance();
67         }
68       }
69     
70       public void addScopedEntity(CAstNode construct, CAstEntity child) {
71         if (!scopedEntities.containsKey(construct)) {
72           Collection<CAstEntity> set = HashSetFactory.make(1);
73           scopedEntities.put(construct, set);
74         }
75         scopedEntities.get(construct).add(child);
76       }
77     }
