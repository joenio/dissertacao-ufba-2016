11     package com.ibm.wala.classLoader;
12     
13     import com.ibm.wala.ipa.cha.IClassHierarchy;
14     import com.ibm.wala.shrikeBT.Constants;
15     import com.ibm.wala.shrikeCT.InvalidClassFileException;
16     import com.ibm.wala.types.MethodReference;
17     import com.ibm.wala.util.debug.Assertions;
18     
19     /**
20      * Note that classes from JVML have some features that are not present in all "bytecode" languages currently supported.
21      * 
22      * @param <T> type of classloader which loads this format of class.
23      */
24     public abstract class JVMClass<T extends IClassLoader> extends BytecodeClass<T> {
25     
26       protected JVMClass(T loader, IClassHierarchy cha) {
27         super(loader, cha);
28       }
29     
30       /**
31        * JVM-level modifiers; cached here for efficiency
32        */
33       protected int modifiers;
34     
35       @Override
36       public int getModifiers() {
37         return modifiers;
38       }
39     
40       @Override
41       public boolean isPublic() {
42         boolean result = ((modifiers & Constants.ACC_PUBLIC) != 0);
43         return result;
44       }
45     
46       @Override
47       public boolean isPrivate() {
48         boolean result = ((modifiers & Constants.ACC_PRIVATE) != 0);
49         return result;
50       }
51     
52       @Override
53       public boolean isInterface() {
54         boolean result = ((modifiers & Constants.ACC_INTERFACE) != 0);
55         return result;
56     
57       }
58     
59       /*
60        * @see com.ibm.wala.classLoader.IClass#isAbstract()
61        */
62       @Override
63       public boolean isAbstract() {
64         boolean result = ((modifiers & Constants.ACC_ABSTRACT) != 0);
65         return result;
66       }
67     
68       /**
69        * @see com.ibm.wala.classLoader.IClass#getClassInitializer()
70        */
71       @Override
72       public IMethod getClassInitializer() {
73         try {
74           computeMethodMapIfNeeded();
75         } catch (InvalidClassFileException e) {
76           e.printStackTrace();
77           Assertions.UNREACHABLE();
78         }
79         return methodMap.get(MethodReference.clinitSelector);
80       }
81     
82     }
