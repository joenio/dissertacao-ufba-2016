11     package com.ibm.wala.cast.tree;
12     
13     /**
14      *  This interface represents nodes of CAPA Abstract Syntax Trees.  It
15      * is a deliberately minimal interface, simply assuming that the nodes
16      * form a tree and have some minimal state at each node.  In
17      * particular, a node has a kind---which should be one of the symbolic
18      * constants in this file---and potentially has child nodes, a
19      * constant values, or possibly both.
20      *
21      *  Note that there is no support for mutating these trees.  This is
22      * deliberate, and should not be changed.  We do not want to force all
23      * clients of the capa ast to handle mutating programs.  In
24      * particular, the DOMO infrastructure has many forms of caching and
25      * other operations that rely on the underlying program being
26      * immutable.  If you need to mutate these trees for some reason---and
27      * think carefully if you really need to, since this is meant to be
28      * essentially a wire format between components---make specialized
29      * implementations that understand how to do that.
30      *
31      *  Also note that this interface does not assume that you need some
32      * great big class hierarchy to structure types of nodes in an ast.
33      * Some people prefer such hierarchies as a matter of taste, but this
34      * interface is designed to not inflict this design choice on others.
35      *
36      *  Finally note that the set of node types in this file is not meant
37      * to be exhaustive.  As new languages are added, feel free to add new
38      * nodes types as needed.
39      *
40      * @author Julian Dolby (dolby@us.ibm.com)
41      * @author Robert M. Fuhrer (rfuhrer@watson.ibm.com)
42      *
43      */
44     public interface CAstNode {
45     
46       // statement kinds
47       /**
48        * Represents a standard case statement. Children:
49        * </ul>
50        *   <li>condition expression
51        *   <li>BLOCK_STMT containing all the cases
52        * </ul>
53        */
54       public static final int SWITCH = 1;
55     
56       /**
57        * Represents a standard while loop. Children:
58        * <ul>
59        *   <li>expression denoting the loop condition
60        *   <li>statement denoting the loop body
61        * </ul>
62        */
63       public static final int LOOP = 2;
64     
65       /**
66        * Represents a block of sequential statements. Children:
67        * <ul>
68        *   <li>statement #1
69        *   <li>statement #2
70        *   <li>...
71        * </ul>
72        */
73       public static final int BLOCK_STMT = 3;
74     
75       /**
76        *  Represents a standard try/catch statement. Note that while some
77        * languages choose to bundle together the notion of try/catch and
78        * the notion of unwind-protect (aka 'finally'), the CAst does not.
79        * There is a separate UNWIND node type.  Children: 
80        * <ul> 
81        *  <li>the code of the try block.  
82        *  <li>the code of the catch block <li>...
83        * </ul>
84        */
85       public static final int TRY = 4;
86     
87       /**
88        * Represents an expression statement (e.g. "foo();"). Children:
89        * <ul>
90        *   <li>the expression
91        * </ul>
92        */
93       public static final int EXPR_STMT = 5;
94       public static final int DECL_STMT = 6;
95       public static final int RETURN = 7;
96       public static final int GOTO = 8;
97       public static final int BREAK = 9;
98       public static final int CONTINUE = 10;
99       public static final int IF_STMT = 11;
100       public static final int THROW = 12;
101       public static final int FUNCTION_STMT = 13;
102       public static final int ASSIGN = 14;
103       public static final int ASSIGN_PRE_OP = 15;
104       public static final int ASSIGN_POST_OP = 16;
105       public static final int LABEL_STMT = 17;
106       public static final int IFGOTO = 18;
107       public static final int EMPTY = 19;
108       public static final int RETURN_WITHOUT_BRANCH = 20;
109       public static final int CATCH = 21;
110       public static final int UNWIND = 22;
111       public static final int MONITOR_ENTER = 23;
112       public static final int MONITOR_EXIT = 24;
113       public static final int ECHO = 25;
114       public static final int YIELD_STMT = 26;
115       public static final int FORIN_LOOP = 27;
116     
117       // expression kinds
118       public static final int FUNCTION_EXPR = 100;
119       public static final int EXPR_LIST = 101;
120       public static final int CALL = 102;
121       public static final int GET_CAUGHT_EXCEPTION = 103;
122     
123       /**
124        * Represents a block of sequentially-executed nodes, the last of which produces
125        * the value for the entire block (like progn from lisp).
126        * Children:
127        * <ul>
128        *   <li>node 1
129        *   <li>node 2
130        *   <li>...
131        *   <li>block value expression
132        * </ul>
133        */
134       public static final int BLOCK_EXPR = 104;
135       public static final int BINARY_EXPR = 105;
136       public static final int UNARY_EXPR = 106;
137       public static final int IF_EXPR = 107;
138       public static final int ANDOR_EXPR = 108; // TODO blow away?
139       public static final int NEW = 109;
140       public static final int OBJECT_LITERAL = 110;
141       public static final int VAR = 111;
142       public static final int OBJECT_REF = 112;
143       public static final int CHOICE_EXPR = 113;
144       public static final int CHOICE_CASE = 114;
145       public static final int SUPER = 115;
146       public static final int THIS = 116;
147       public static final int ARRAY_LITERAL = 117;
148       public static final int CAST = 118;
149       public static final int INSTANCEOF = 119;
150       public static final int ARRAY_REF = 120;
151       public static final int ARRAY_LENGTH = 121;
152       public static final int TYPE_OF = 122;
153       public static final int EACH_ELEMENT_HAS_NEXT = 123;
154       public static final int EACH_ELEMENT_GET = 124;
155       public static final int LIST_EXPR = 125;
156       public static final int EMPTY_LIST_EXPR = 126;
157       public static final int TYPE_LITERAL_EXPR = 127;
158       public static final int IS_DEFINED_EXPR = 128;
159       public static final int MACRO_VAR = 129;
160     
161       // new nodes with an explicit enclosing argument, e.g. "outer.new Inner()". They are mostly treated the same, except in JavaCAst2IRTranslator.doNewObject
162       public static final int NEW_ENCLOSING = 130;
163       
164       // explicit lexical scopes
165       public static final int LOCAL_SCOPE = 200;
166       public static final int SPECIAL_PARENT_SCOPE = 201;
167     
168       // literal expression kinds
169       public static final int CONSTANT = 300;
170       public static final int OPERATOR = 301;
171     
172       // special stuff
173       public static final int PRIMITIVE = 400;
174       public static final int ERROR = 401;
175       public static final int VOID = 402;
176       public static final int ASSERT = 403;
177       public static final int INCLUDE = 404;
178       public static final int NAMED_ENTITY_REF = 405;
179     
180       public static final int SUB_LANGUAGE_BASE = 1000;
181     
182       /** 
183        * What kind of node is this?  Should return some constant from this file.
184        */ 
185       int getKind();
186     
187       /**
188        *  Returns the constant value represented by this node, if
189        * appropriate, and null otherwise.
190        */
191       Object getValue();
192     
193       /**
194        *  Return the nth child of this node.  If there is no such child,
195        * this method should throw a NoSuchElementException.
196        */
197       CAstNode getChild(int n);
198     
199       /**
200        * How many children does this node have?
201        */
202       int getChildCount();
203     
204     }
