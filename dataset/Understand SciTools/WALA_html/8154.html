49     package org.scandroid.flow;
50     
51     import java.util.Collections;
52     import java.util.Set;
53     
54     import org.scandroid.domain.CodeElement;
55     import org.scandroid.domain.DomainElement;
56     import org.scandroid.domain.IFDSTaintDomain;
57     import org.scandroid.domain.LocalElement;
58     import org.scandroid.flow.types.FlowType;
59     import org.scandroid.util.CGAnalysisContext;
60     
61     import com.ibm.wala.dataflow.IFDS.TabulationResult;
62     import com.ibm.wala.ipa.callgraph.CGNode;
63     import com.ibm.wala.ipa.callgraph.propagation.InstanceKey;
64     import com.ibm.wala.ipa.callgraph.propagation.PointerKey;
65     import com.ibm.wala.ipa.cfg.BasicBlockInContext;
66     import com.ibm.wala.ssa.analysis.IExplodedBasicBlock;
67     import com.ibm.wala.util.collections.HashSetFactory;
68     import com.ibm.wala.util.intset.OrdinalSet;
69     
70     public class LocalSinkPoint implements ISinkPoint {
71     	
72     	private final BasicBlockInContext<IExplodedBasicBlock> block;
73     	private final int ssaVal;
74     	private final FlowType<IExplodedBasicBlock> sinkFlow;
75     
76     	public LocalSinkPoint(BasicBlockInContext<IExplodedBasicBlock> block,
77     			int ssaVal, FlowType<IExplodedBasicBlock> sinkFlow) {
78     		this.block = block;
79     		this.ssaVal = ssaVal;
80     		this.sinkFlow = sinkFlow;
81     	}
82     	
83     	@SuppressWarnings("unchecked")
84     	@Override
85     	public Set<FlowType<IExplodedBasicBlock>> findSources(CGAnalysisContext<IExplodedBasicBlock> ctx,
86     			TabulationResult<BasicBlockInContext<IExplodedBasicBlock>, CGNode, DomainElement> flowResult,
87     			IFDSTaintDomain<IExplodedBasicBlock> domain) {
88     		Set<FlowType<IExplodedBasicBlock>> sources = HashSetFactory.make();
89     
90     		final CodeElement localElt = new LocalElement(ssaVal);
91     		Set<CodeElement> elts = HashSetFactory.make(Collections.singleton(localElt));
92     
93     		final CGNode node = block.getNode();
94     		PointerKey pk = ctx.pa.getHeapModel().getPointerKeyForLocal(node,
95     				ssaVal);
96     		OrdinalSet<InstanceKey> iks = ctx.pa.getPointsToSet(pk);
97     
98     		for (InstanceKey ik : iks) {
99     			elts.addAll(ctx.codeElementsForInstanceKey(ik));
100     		}
101     
102     		for (CodeElement elt : elts) {
103     			for (DomainElement de : domain.getPossibleElements(elt)) {
104     				if (flowResult.getResult(block).contains(
105     						domain.getMappedIndex(de))) {
106     					sources.add(de.taintSource);
107     				}
108     			}
109     		}
110     		return sources;
111     	}
112     	
113     	@Override
114     	public FlowType<IExplodedBasicBlock> getFlow() {
115     		return sinkFlow;
116     	}
117     
118     	@Override
119     	public int hashCode() {
120     		final int prime = 31;
121     		int result = 1;
122     		result = prime * result + ((block == null) ? 0 : block.hashCode());
123     		result = prime * result
124     				+ ((sinkFlow == null) ? 0 : sinkFlow.hashCode());
125     		result = prime * result + ssaVal;
126     		return result;
127     	}
128     
129     	@Override
130     	public boolean equals(Object obj) {
131     		if (this == obj)
132     			return true;
133     		if (obj == null)
134     			return false;
135     		if (getClass() != obj.getClass())
136     			return false;
137     		LocalSinkPoint other = (LocalSinkPoint) obj;
138     		if (block == null) {
139     			if (other.block != null)
140     				return false;
141     		} else if (!block.equals(other.block))
142     			return false;
143     		if (sinkFlow == null) {
144     			if (other.sinkFlow != null)
145     				return false;
146     		} else if (!sinkFlow.equals(other.sinkFlow))
147     			return false;
148     		if (ssaVal != other.ssaVal)
149     			return false;
150     		return true;
151     	}
152     
153     	@Override
154     	public String toString() {
155     		return "SinkPoint [block=" + block + ", ssaVal=" + ssaVal
156     				+ ", sinkFlow=" + sinkFlow + "]";
157     	}
158     
159     }
