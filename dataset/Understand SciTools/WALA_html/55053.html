36             System.err.println(("Cannot use int set factory " + System.getProperty(INT_SET_FACTORY_CONFIG_PROPERTY_NAME)));

40           setDefaultIntSetFactory(defaultFactory);

53       public static IntSet make(Set<Integer> x) {
54         int[] vals = new int[ x.size() ];
55         Iterator<Integer> vs = x.iterator();
56         for(int i = 0; i < vals.length; i++) {
57           vals[i] = vs.next();
58         }
59         return make(vals);
60       }

68       /**
69        * This method constructs an appropriate mutable copy of set.
70        * 
71        * @return a new {@link MutableIntSet} object with the same value as set
72        * @throws UnimplementedError if we haven't supported the set type yet.
73        * @throws IllegalArgumentException if set == null
74        */
75       public static MutableIntSet makeMutableCopy(IntSet set) throws IllegalArgumentException, UnimplementedError {
76         if (set == null) {
77           throw new IllegalArgumentException("set == null");
78         }
79         if (set instanceof SparseIntSet) {
80           return MutableSparseIntSet.make(set);
81         } else if (set instanceof BitVectorIntSet) {
82           return new BitVectorIntSet(set);
83         } else if (set instanceof BimodalMutableIntSet) {
84           return BimodalMutableIntSet.makeCopy(set);
85         } else if (set instanceof MutableSharedBitVectorIntSet) {
86           return new MutableSharedBitVectorIntSet((MutableSharedBitVectorIntSet) set);
87         } else if (set instanceof SemiSparseMutableIntSet) {
88           return new SemiSparseMutableIntSet((SemiSparseMutableIntSet) set);
89         } else if (set instanceof DebuggingMutableIntSet) {
90           MutableIntSet pCopy = makeMutableCopy(((DebuggingMutableIntSet) set).primaryImpl);
91           MutableIntSet sCopy = makeMutableCopy(((DebuggingMutableIntSet) set).secondaryImpl);
92           return new DebuggingMutableIntSet(pCopy, sCopy);
93         } else if (set instanceof EmptyIntSet) {
94           return IntSetUtil.make();
95         } else {
96           Assertions.UNREACHABLE(set.getClass().toString());
97           return null;
98         }
99       }

102        * Compute the asymmetric difference of two sets, a \ b.
103        */
104       public static IntSet diff(IntSet A, IntSet B) {
105         if (A == null) {
106           throw new IllegalArgumentException("null A");
107         }
108         if (B == null) {
109           throw new IllegalArgumentException("null B");

133       /**
134        * Compute the asymmetric difference of two sets, a \ b.
135        */
136       public static IntSet diff(IntSet A, IntSet B, MutableIntSetFactory<?> factory) {
137         if (factory == null) {
138           throw new IllegalArgumentException("null factory");
139         }
140         if (A == null) {
141           throw new IllegalArgumentException("null A");
142         }
143         if (B == null) {
144           throw new IllegalArgumentException("null B");
145         }
146         if (A instanceof SparseIntSet && B instanceof SparseIntSet) {
147           return SparseIntSet.diff((SparseIntSet) A, (SparseIntSet) B);
148         } else if (A instanceof SemiSparseMutableIntSet && B instanceof SemiSparseMutableIntSet) {
149           IntSet d = SemiSparseMutableIntSet.diff((SemiSparseMutableIntSet) A, (SemiSparseMutableIntSet) B);
150           return d;
151         } else {
152           return defaultSlowDiff(A, B, factory);
153         }
154       }

159        * @throws IllegalArgumentException if B == null

161       public static MutableIntSet removeAll(MutableIntSet A, IntSet B) throws IllegalArgumentException {
162         if (A == null) {
163           throw new IllegalArgumentException("A == null");
164         }
165         if (B == null) {
166           throw new IllegalArgumentException("B == null");
167         }
168         if (A instanceof SemiSparseMutableIntSet && B instanceof SemiSparseMutableIntSet) {
169           if (DEBUG) {
170             System.err.println("call SemiSparseMutableIntSet.removeAll");
171           }
172           return ((SemiSparseMutableIntSet) A).removeAll((SemiSparseMutableIntSet) B);
173         } else {
174           for (IntIterator it = B.intIterator(); it.hasNext();) {
175             int I = it.next();
176             A.remove(I);
177             if (DEBUG) {
178               System.err.println(("removed " + I + " now is " + A));
179             }
180           }
181           if (DEBUG) {
182             System.err.println(("return " + A));
183           }
184           return A;
185         }
186       }

188       /**
189        * @return index \in [low,high] s.t. data[index] = key, or -1 if not found
190        */
191       public static int binarySearch(int[] data, int key, int low, int high) throws IllegalArgumentException {
192         if (data == null) {
193           throw new IllegalArgumentException("null array");
194         }
195         if (data.length == 0) {
196           return -1;
197         }
198         if (low <= high && (low < 0 || high < 0)) {
199           throw new IllegalArgumentException("can't search negative indices " + low + " " + high);
200         }
201         if (high > data.length - 1) {
202           high = data.length - 1;
203         }
204         if (low <= high) {
205           int mid = (low + high) / 2;
206           int midValue = data[mid];
207           if (midValue == key) {
208             return mid;
209           } else if (midValue > key) {
210             return binarySearch(data, key, low, mid - 1);
211           } else {
212             return binarySearch(data, key, mid + 1, high);
213           }
214         } else {
215           return -1;
216         }
217       }

236       /**
237        * @return a new sparse int set which adds j to s
238        * @throws IllegalArgumentException if s == null
239        */
240       public static IntSet add(IntSet s, int j) throws IllegalArgumentException {
241         if (s == null) {
242           throw new IllegalArgumentException("s == null");
243         }
244         if (s instanceof SparseIntSet) {
245           SparseIntSet sis = (SparseIntSet) s;
246           return SparseIntSet.add(sis, j);
247         } else {
248           // really slow. optimize as needed.
249           MutableSparseIntSet result = MutableSparseIntSet.make(s);
250           result.add(j);
251           return result;
252         }
253       }

255       public static int[] toArray(IntSet s) {
256         int i = 0;
257         int[] result = new int[ s.size() ];
258         IntIterator x = s.intIterator();
259         while (x.hasNext()) {
260           result[i++] = x.next();
261         }
262         assert ! x.hasNext();
263         return result;
264       }
