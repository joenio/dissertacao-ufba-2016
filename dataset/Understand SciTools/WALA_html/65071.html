11     package com.ibm.wala.util.collections;
12     
13     import java.util.Comparator;
14     
15     /**
16      * A comparator based on hash codes
17      */
18     public class HashCodeComparator<T> implements Comparator<T> {
19     
20       @SuppressWarnings("rawtypes")
21       private static final HashCodeComparator INSTANCE = new HashCodeComparator();
22     
23       /*
24        * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
25        */
26       @Override
27       public int compare(T o1, T o2) throws NullPointerException {
28         // by convention null is less than non-null
29         if (o1 == null) {
30           return o2 == null ? 0 : -1;
31         } else if (o2 == null) {
32           return 1;
33         }
34         return o1.hashCode() - o2.hashCode();
35       }
36     
37       @SuppressWarnings("unchecked")
38       public static <T> HashCodeComparator<T> instance() {
39         return INSTANCE;
40       }
41     
42     }
