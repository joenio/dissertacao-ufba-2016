16     import java.io.InputStream;

18     import org.eclipse.core.resources.ResourcesPlugin;
19     import org.eclipse.core.runtime.CoreException;
20     import org.eclipse.core.runtime.IPath;
21     
22     import com.ibm.wala.client.AbstractAnalysisEngine;
23     import com.ibm.wala.ide.util.EclipseProjectPath;
24     import com.ibm.wala.ipa.callgraph.AnalysisCache;
25     import com.ibm.wala.ipa.callgraph.AnalysisOptions;
26     import com.ibm.wala.ipa.callgraph.AnalysisScope;
27     import com.ibm.wala.ipa.callgraph.CallGraphBuilder;
28     import com.ibm.wala.ipa.cha.IClassHierarchy;
29     import com.ibm.wala.util.config.FileOfClasses;
30     import com.ibm.wala.util.io.FileProvider;
31     
32     abstract public class EclipseProjectAnalysisEngine<P> extends AbstractAnalysisEngine {
33     
34       protected final P project;
35       
36       protected final IPath workspaceRootPath;
37     
38       protected EclipseProjectPath<?,P> ePath;
39     
40       public EclipseProjectAnalysisEngine(P project) throws IOException, CoreException {
41         super();
42         this.project = project;
43         this.workspaceRootPath = ResourcesPlugin.getWorkspace().getRoot().getLocation();
44         assert project != null;
45         assert workspaceRootPath != null;
46       }
47     
48       abstract protected EclipseProjectPath<?,P> createProjectPath(P project) throws IOException, CoreException;
49     
50       @Override
51       abstract protected CallGraphBuilder getCallGraphBuilder(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache);
52     
53       abstract protected AnalysisScope makeAnalysisScope();
54       
55       @Override
56       public void buildAnalysisScope() throws IOException {
57         try {
58           ePath = createProjectPath(project);
59           super.scope = ePath.toAnalysisScope(makeAnalysisScope());
60           if (getExclusionsFile() != null) {
61             InputStream is = new File(getExclusionsFile()).exists()? new FileInputStream(getExclusionsFile()): FileProvider.class.getClassLoader().getResourceAsStream(getExclusionsFile());
62             scope.setExclusions(new FileOfClasses(is));
63           }
64         } catch (CoreException e) {
65           assert false : e.getMessage();
66         }
67       }
68     
69       public EclipseProjectPath<?,P> getEclipseProjectPath() {
70         return ePath;
71       }
72     
73       @Override
74       public IClassHierarchy getClassHierarchy() {
75         if (super.getClassHierarchy() == null) {
76           setClassHierarchy( buildClassHierarchy() );
77         }
78     
79         return super.getClassHierarchy();
80       }
81     
82     }
