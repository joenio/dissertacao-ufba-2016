31     public class SyntheticIR extends IR {
32     
33       private final static boolean PARANOID = true;
34     
35       /**
36        * Create an SSA form, induced over a list of instructions provided externally. This entrypoint is often used for, e.g., native
37        * method models
38        * 
39        * @param method the method to construct SSA form for
40        * @param context the governing context
41        * @param instructions the SSA instructions which define the body of the method
42        * @param constants a Map giving information on constant values for the symbol table
43        * @throws AssertionError if method is null
44        */
45       public SyntheticIR(IMethod method, Context context, AbstractCFG cfg, SSAInstruction[] instructions, SSAOptions options,
46           Map<Integer, ConstantValue> constants) throws AssertionError {
47         super(method, instructions, makeSymbolTable(method, instructions, constants, cfg), new SSACFG(method, cfg, instructions),
48             options);
49         if (PARANOID) {
50           repOK(instructions);
51         }
52     
53         setupLocationMap();
54       }
55     
56       /**
57        * throw an assertion if the instruction array contains a phi instruction
58        */
59       private void repOK(SSAInstruction[] instructions) {
60         for (SSAInstruction s : instructions) {
61           if (s instanceof SSAPhiInstruction) {
62             Assertions.UNREACHABLE();
63           }
64           if (s instanceof SSAPiInstruction) {
65             Assertions.UNREACHABLE();
66           }
67         }
68       }
69     
70       /**
71        * Set up the symbol table according to statements in the IR
72        * 
73        * @param constants Map: value number (Integer) -> ConstantValue
74        */
75       private static SymbolTable makeSymbolTable(IMethod method, SSAInstruction[] instructions, Map<Integer, ConstantValue> constants,
76           AbstractCFG cfg) {
77         if (method == null) {
78           throw new IllegalArgumentException("null method");
79         }
80         SymbolTable symbolTable = new SymbolTable(method.getNumberOfParameters());
81     
82         // simulate allocation of value numbers
83         for (int i = 0; i < instructions.length; i++) {
84           if (instructions[i] != null) {
85             SSAInstruction s = instructions[i];
86             updateForInstruction(constants, symbolTable, s);
87           }
88         }
89     
90         /**
91          * In InducedCFGs, we have nulled out phi instructions from the instruction array ... so go back and retrieve them now.
92          */
93         if (cfg instanceof InducedCFG) {
94           InducedCFG icfg = (InducedCFG) cfg;
95           for (SSAPhiInstruction phi : icfg.getAllPhiInstructions()) {
96             updateForInstruction(constants, symbolTable, phi);
97           }
98         }
99     
100         return symbolTable;
101       }
102     
103       private static void updateForInstruction(Map<Integer, ConstantValue> constants, SymbolTable symbolTable, SSAInstruction s) {
104         for (int j = 0; j < s.getNumberOfDefs(); j++) {
105           symbolTable.ensureSymbol(s.getDef(j));
106         }
107         for (int j = 0; j < s.getNumberOfUses(); j++) {
108           int vn = s.getUse(j);
109           symbolTable.ensureSymbol(vn);
110           if (constants != null && constants.containsKey(new Integer(vn)))
111             symbolTable.setConstantValue(vn, constants.get(new Integer(vn)));
112         }
113       }

115       /**
116        * This returns "", as synthetic IRs have no line numbers right now.
117        */
118       @Override
119       protected String instructionPosition(int instructionIndex) {
120         return "";
121       }

123       /**
124        * This returns null, as synthetic IRs have no local names right now.
125        */
126       @Override
127       public SSA2LocalMap getLocalMap() {
128         return null;
129       }
130     
131       @Override
132       protected SSAIndirectionData<Name> getIndirectionData() {
133          return null;
134       }
135     }
