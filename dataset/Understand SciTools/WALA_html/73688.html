12     package com.ibm.wala.shrikeCT;
13     
14     import java.lang.invoke.MethodHandles;
15     import java.lang.invoke.MethodHandles.Lookup;
16     import java.lang.invoke.MethodType;
17     import java.lang.reflect.Method;
18     
19     import com.ibm.wala.shrikeCT.ClassReader.AttrIterator;
20     
21     public class BootstrapMethodsReader extends AttributeReader {
22     
23       public interface BootstrapMethod {
24         int invokeType();
25         String methodClass();
26         String methodName();
27         String methodType();
28         int callArgumentCount();
29         Object callArgument(ClassLoader cl, int i);
30         int callArgumentIndex(int i);
31         int callArgumentKind(int i);
32         ConstantPoolParser getCP();
33         int getIndexInClassFile();
34       }
35       
36       private BootstrapMethod entries[];
37       
38       protected BootstrapMethodsReader(AttrIterator attr) throws InvalidClassFileException {
39         super(attr, "BootstrapMethods");
40         readBootstrapEntries();
41       }
42     
43       private void readBootstrapEntries() throws InvalidClassFileException {
44         final ConstantPoolParser cp = cr.getCP();
45     
46         entries = new BootstrapMethod[cr.getUShort(attr + 6)];
47         int base = 8;
48         for(int i = 0; i < entries.length; i++) {
49           final int methodHandleOffset = cr.getUShort(attr + base);
50           final int argsBase = attr + base + 4;
51           final int index = i;
52           
53           final int argumentCount = cr.getUShort(attr + base + 2);
54           entries[i] = new BootstrapMethod() {
55             private final int invokeType = cp.getCPHandleKind(methodHandleOffset);
56             private final String methodClass = cp.getCPHandleClass(methodHandleOffset);
57             private final String methodName = cp.getCPHandleName(methodHandleOffset);
58             private final String methodType = cp.getCPHandleType(methodHandleOffset);
59             
60             @Override
61             public String toString() {
62               return methodClass + ":" + methodName + methodType;
63             }
64             
65             @Override
66             public int invokeType() {
67               return invokeType;
68             }
69     
70             @Override
71             public String methodClass() {
72               return methodClass;
73             }
74     
75             @Override
76             public String methodName() {
77               return methodName;
78             }
79     
80             @Override
81             public String methodType() {
82               return methodType;
83             }
84     
85             
86             @Override
87             public int callArgumentCount() {
88               return argumentCount;
89             }
90     
91             @Override
92             public int callArgumentKind(int i) {
93               return cp.getItemType(callArgumentIndex(i));
94             }
95     
96             @Override
97             public int callArgumentIndex(int i) {
98               assert 0 <= i && i < argumentCount;
99               int index = argsBase + (2*i);
100               return cr.getUShort(index);
101             }
102     
103             @Override
104             public Object callArgument(ClassLoader cl, int i) {
105               try {
106                 int index = callArgumentIndex(i);
107                 int t = callArgumentKind(i);
108                 switch (t) {
109                 case ClassConstants.CONSTANT_Utf8:
110                   return cp.getCPUtf8(index);
111                 case ClassConstants.CONSTANT_Class:
112                   return cp.getCPClass(index);
113                 case ClassConstants.CONSTANT_String:
114                   return cp.getCPString(index);
115                 case ClassConstants.CONSTANT_Integer:
116                   return cp.getCPInt(index);
117                 case ClassConstants.CONSTANT_Float:
118                   return cp.getCPFloat(index);
119                 case ClassConstants.CONSTANT_Double:
120                   return cp.getCPDouble(index);
121                 case ClassConstants.CONSTANT_Long:
122                   return cp.getCPLong(index);
123                 case ClassConstants.CONSTANT_MethodHandle:
124                   String className = cp.getCPHandleClass(index);
125                   String eltName = cp.getCPHandleName(index);
126                   String eltDesc = cp.getCPHandleType(index);
127                   MethodType type = MethodType.fromMethodDescriptorString(eltDesc, cl);
128                   Class<?> cls = Class.forName(className.replace('/', '.'), false, cl);
129                   Method m = cls.getDeclaredMethod(eltName, type.parameterList().toArray(new Class[type.parameterCount()]));
130                   Lookup lk = MethodHandles.lookup().in(cls);
131                   m.setAccessible(true);
132                   return lk.unreflect(m);
133                 case ClassConstants.CONSTANT_MethodType:
134                   return MethodType.fromMethodDescriptorString(cp.getCPMethodType(index), cl);
135                 default:
136                   assert false : "invalid type " + t;
137                 }            
138               } catch (IllegalArgumentException e) {
139                 assert false : e;
140               } catch (InvalidClassFileException e) {
141                 assert false : e;
142               } catch (ClassNotFoundException e) {
143                 assert false : e;
144               } catch (NoSuchMethodException e) {
145                 assert false : e;
146               } catch (SecurityException e) {
147                 assert false : e;
148               } catch (IllegalAccessException e) {
149                 assert false : e;
150               }
151               return null;
152             }
153     
154             @Override
155             public ConstantPoolParser getCP() {
156               return cp;
157             }
158     
159             @Override
160             public int getIndexInClassFile() {
161               return index;
162             }
163           };
164           
165           base += (argumentCount*2) + 4;
166         }
167       }
168     
169       public int count() {
170         return entries.length;
171       }
172       
173       public BootstrapMethod getEntry(int i) {
174         return entries[i];
175       }
176     }
