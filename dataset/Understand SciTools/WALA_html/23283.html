31     /**
32      * This is a demo class.
33      * 
34      * Class files are taken as input arguments (or if there are none, from standard input). The methods in those files are
35      * instrumented: we insert a System.err.println() at ever method call, and a System.err.println() at every method entry.
36      * 
37      * In Unix, I run it like this: java -cp ~/dev/shrike/shrike com.ibm.wala.shrikeBT.shrikeCT.tools.BatchVerifier test.jar -o
38      * output.jar
39      * 
40      * The instrumented classes are placed in the directory "output" under the current directory. Disassembled code is written to the
41      * file "report" under the current directory.
42      */
43     public class BatchVerifier {
44       private static boolean disasm = false;
45     
46       final private static ClassHierarchyStore store = new ClassHierarchyStore();
47     
48       private static int errors = 0;
49     
50       public static void main(String[] args) throws Exception {
51         OfflineInstrumenter oi = new OfflineInstrumenter(true);
52         args = oi.parseStandardArgs(args);
53     
54         for (int i = 0; i < args.length; i++) {
55           if (args[i].equals("-d")) {
56             disasm = true;
57           }
58         }
59     
60         PrintWriter w = new PrintWriter(new BufferedWriter(new FileWriter("report", false)));
61     
62         oi.beginTraversal();
63         ClassInstrumenter ci;
64         while ((ci = oi.nextClass()) != null) {
65           ClassReader cr = ci.getReader();
66           CTUtils.addClassToHierarchy(store, cr);
67         }
68     
69         oi.beginTraversal();
70         while ((ci = oi.nextClass()) != null) {
71           doClass(ci.getReader(), w);
72         }
73     
74         oi.close();
75     
76         if (errors > 0) {
77           System.err.println(errors + " error" + (errors > 1 ? "s" : "") + " detected");
78         }
79       }
80     
81       private static void doClass(final ClassReader cr, PrintWriter w) throws Exception {
82         int methodCount = cr.getMethodCount();
83         w.write("Verifying " + cr.getName() + "\n");
84         w.flush();
85     
86         ClassReader.AttrIterator iter = new ClassReader.AttrIterator();
87     
88         for (int i = 0; i < methodCount; i++) {
89           cr.initMethodAttributeIterator(i, iter);
90           for (; iter.isValid(); iter.advance()) {
91             if (iter.getName().equals("Code")) {
92               w.write("Verifying " + cr.getName() + "." + cr.getMethodName(i) + " " + cr.getMethodType(i) + ":\n");
93               w.flush();
94     
95               CodeReader code = new CodeReader(iter);
96               CTDecoder d = new CTDecoder(code);
97               try {
98                 d.decode();
99               } catch (Decoder.InvalidBytecodeException e) {
100                 throw new InvalidClassFileException(code.getRawOffset(), e.getMessage());
101               }
102               MethodData md = new MethodData(d, cr.getMethodAccessFlags(i), CTDecoder.convertClassToType(cr.getName()), cr
103                   .getMethodName(i), cr.getMethodType(i));
104     
105               if (disasm) {
106                 w.write("ShrikeBT code:\n");
107                 (new Disassembler(md)).disassembleTo(w);
108                 w.flush();
109               }
110     
111               Verifier v = new Verifier(md);
112               try {
113                 v.verify();
114               } catch (FailureException e) {
115                 w.println("ERROR: VERIFICATION FAILED");
116                 e.printStackTrace(w);
117                 e.printPath(w);
118                 errors++;
119                 w.flush();
120               }
121     
122               break;
123             }
124           }
125         }
126       }
127     }
