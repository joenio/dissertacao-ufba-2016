12     package com.ibm.wala.classLoader;
13     
14     import java.io.Reader;
15     import java.util.Collection;
16     import java.util.NoSuchElementException;
17     
18     import com.ibm.wala.ipa.cha.IClassHierarchyDweller;
19     import com.ibm.wala.types.Selector;
20     import com.ibm.wala.types.TypeName;
21     import com.ibm.wala.types.TypeReference;
22     import com.ibm.wala.types.annotations.Annotation;
23     import com.ibm.wala.util.strings.Atom;
24     
25     /**
26      * Basic interface for an object that represents a single Java class for analysis purposes, including array classes.
27      */
28     public interface IClass extends IClassHierarchyDweller {
29     
30       /**
31        * Return the object that represents the defining class loader for this class.
32        * 
33        * @return the object that represents the defining class loader for this class.
34        */
35       IClassLoader getClassLoader();
36     
37       /**
38        * Is this class a Java interface?
39        */
40       boolean isInterface();
41     
42       /**
43        * @return true iff this class is abstract
44        */
45       boolean isAbstract();
46     
47       /**
48        * @return true iff this class is public
49        */
50       boolean isPublic();
51       
52       /**
53        * @return true iff this class is private
54        */
55       boolean isPrivate();
56     
57       /**
58        * Return the integer that encodes the class's modifiers, as defined by the JVM specification
59        * 
60        * @return the integer that encodes the class's modifiers, as defined by the JVM specification
61        */
62       int getModifiers() throws UnsupportedOperationException;
63     
64       /**
65        * @return the superclass, or null if java.lang.Object
66        * @throws IllegalStateException if there's some problem determining the superclass
67        */
68       IClass getSuperclass();
69     
70       /**
71        * @return Collection of (IClass) interfaces this class directly implements. If this class is an interface, returns the interfaces
72        *         it immediately extends.
73        */
74       Collection<? extends IClass> getDirectInterfaces();
75     
76       /**
77        * @return Collection of (IClass) interfaces this class implements, including all ancestors of interfaces immediately implemented.
78        *         If this class is an interface, it returns all super-interfaces.
79        */
80       Collection<IClass> getAllImplementedInterfaces();
81     
82       /**
83        * Finds method matching signature. Delegates to superclass if not found.
84        * 
85        * @param selector a method signature
86        * @return IMethod from this class matching the signature; null if not found in this class or any superclass.
87        */
88       IMethod getMethod(Selector selector);
89     
90       /**
91        * Finds a field.
92        * 
93        * @throws IllegalStateException if the class contains multiple fields with name <code>name</code>.
94        */
95       IField getField(Atom name);
96     
97       /**
98        * Finds a field, given a name and a type.  Returns <code>null</code> if not found.
99        */
100       IField getField(Atom name, TypeName type);
101       
102       /**
103        * @return canonical TypeReference corresponding to this class
104        */
105       TypeReference getReference();
106     
107       /**
108        * @return String holding the name of the source file that defined this class, or null if none found
109        * @throws NoSuchElementException if this class was generated from more than one source file
110        * The assumption that a class is generated from a single source file is java
111        * specific, and will change in the future.  In place of this API, use the APIs in IClassLoader.  
112        * SJF .. we should think about this deprecation.  postponing deprecation for now.  
113        */
114       String getSourceFileName() throws NoSuchElementException;
115     
116       /**
117        * @return String representing the source file holding this class, or null if not found
118        * @throws NoSuchElementException if this class was generated from more than one source file
119        * The assumption that a class is generated from a single source file is java
120        * specific, and will change in the future.   In place of this API, use the APIs in IClassLoader.  
121        * SJF .. we should think about this deprecation.  postponing deprecation for now.    
122        */
123       Reader getSource() throws NoSuchElementException;
124     
125       /**
126        * @return the method that is this class's initializer, or null if none
127        */
128       IMethod getClassInitializer();
129     
130       /**
131        * @return true iff the class is an array class.
132        */
133       boolean isArrayClass();
134     
135       /**
136        * @return an Iterator of the IMethods declared by this class.
137        */
138       Collection<IMethod> getDeclaredMethods();
139     
140       /**
141        * Compute the instance fields declared by this class or any of its superclasses.
142        */
143       Collection<IField> getAllInstanceFields();
144     
145       /**
146        * Compute the static fields declared by this class or any of its superclasses.
147        */
148       Collection<IField> getAllStaticFields();
149     
150       /**
151        * Compute the instance and static fields declared by this class or any of its superclasses.
152        */
153       Collection<IField> getAllFields();
154     
155       /**
156        * Compute the methods declared by this class or any of its superclasses.
157        */
158       Collection<IMethod> getAllMethods();
159     
160       /**
161        * Compute the instance fields declared by this class.
162        * 
163        * @return Collection of IFields
164        */
165       Collection<IField> getDeclaredInstanceFields();
166     
167       /**
168        * @return Collection of IField
169        */
170       Collection<IField> getDeclaredStaticFields();
171     
172       /**
173        * @return the TypeName for this class
174        */
175       TypeName getName();
176     
177       /**
178        * Does 'this' refer to a reference type? If not, then it refers to a primitive type.
179        */
180       boolean isReferenceType();
181     
182       /**
183        * get annotations, if any
184        */
185       Collection<Annotation> getAnnotations();
186     
187     }
