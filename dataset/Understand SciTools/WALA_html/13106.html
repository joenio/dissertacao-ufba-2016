12     
13     import java.lang.management.ManagementFactory;
14     import java.lang.management.MemoryMXBean;
15     import java.lang.management.MemoryNotificationInfo;
16     import java.lang.management.MemoryPoolMXBean;
17     import java.lang.management.MemoryType;
18     
19     import javax.management.InstanceNotFoundException;
20     import javax.management.ListenerNotFoundException;
21     import javax.management.Notification;
22     import javax.management.NotificationListener;
23     import javax.management.openmbean.CompositeData;
24     
25     import com.ibm.wala.util.MonitorUtil.IProgressMonitor;

29     /**
30      * A class to control execution through the {@link IProgressMonitor} interface.
31      * 
32      * This class bounds each work item with a time in milliseconds. If there is no
33      * apparent progress within the specified bound, this class cancels itself.
34      */
35     public class ProgressMaster implements IProgressMonitor {
36     
37       private final IProgressMonitor delegate;
38     
39       private volatile boolean timedOut = false;
40     
41       private volatile boolean tooMuchMemory = false;
42     
43       /**
44        * If -1, work items can run forever.
45        */
46       private final int msPerWorkItem;
47     
48       private final boolean checkMemory;
49       
50       private Timeout currentNanny;
51     
52       private ProgressMaster(IProgressMonitor monitor, int msPerWorkItem, boolean checkMemory) {
53         this.delegate = monitor;
54         this.msPerWorkItem = msPerWorkItem;
55         this.checkMemory = checkMemory;
56       }
57     
58       public static ProgressMaster make(IProgressMonitor monitor, int msPerWorkItem, boolean checkMemory) {
59         if (monitor == null) {
60           throw new IllegalArgumentException("null monitor");
61         }
62         return new ProgressMaster(monitor, msPerWorkItem, checkMemory);
63       }
64     
65       @Override
66       public synchronized void beginTask(String name, int totalWork) {
67         delegate.beginTask(name, totalWork);
68         startNanny();
69       }
70     
71       private synchronized void startNanny() {
72         killNanny();
73         if (msPerWorkItem >= 1 || checkMemory) {
74           currentNanny = new Timeout();
75           // don't let nanny thread prevent JVM exit
76           currentNanny.setDaemon(true);
77           currentNanny.start();
78         }
79       }
80     
81       public synchronized void reset() {
82         killNanny();
83         setCanceled(false);
84         timedOut = false;
85         tooMuchMemory = false;
86       }
87     
88       /**
89        * Was the last cancel state due to a timeout?
90        */
91       public boolean lastItemTimedOut() {
92         return timedOut;
93       }
94     
95       public boolean lastItemTooMuchMemory() {
96         return tooMuchMemory;
97       }
98       
99       @Override
100       public synchronized void done() {
101         killNanny();
102         delegate.done();
103       }
104     
105       private synchronized void killNanny() {
106         if (currentNanny != null) {
107           currentNanny.interrupt();
108           try {
109             currentNanny.join();
110           } catch (InterruptedException e) {
111           }
112           currentNanny = null;
113         }
114       }
115     
116       @Override
117       public boolean isCanceled() {
118         return delegate.isCanceled() || timedOut || tooMuchMemory;
119       }
120     
121       public void setCanceled(boolean value) {
122         killNanny();
123       }
124     
125       /** BEGIN Custom change: subtasks and canceling */
126     
127       @Override
128       public void subTask(String subTask) {
129         delegate.subTask(subTask);
130       }
131     
132       @Override
133       public void cancel() {
134         setCanceled(true);
135       }
136     /** END Custom change: subtasks and canceling */
137       @Override
138       public synchronized void worked(int work) {
139         killNanny();
140         delegate.worked(work);
141         startNanny();
142       }
143     
144       public int getMillisPerWorkItem() {
145         return msPerWorkItem;
146       }
147     
148       public static class TooMuchMemoryUsed extends Exception {
149     
150       }
151     
152       private class Timeout extends Thread {
153         
154         @Override
155         public void run() {
156           try {
157             MemoryMXBean gcbean = null;
158             NotificationListener listener = null;
159             
160             if (checkMemory) {
161               for(MemoryPoolMXBean pool : ManagementFactory.getMemoryPoolMXBeans()) {
162                 if (pool.getType().equals(MemoryType.HEAP)) {
163                   pool.setCollectionUsageThreshold((long) (pool.getUsage().getMax() * MAX_USED_MEM_BEFORE_BACKING_OUT));
164                 }
165               }
166               
167               final Thread nannyThread = this;
168               gcbean = ManagementFactory.getMemoryMXBean();
169     
170               listener = new NotificationListener() {
171                 @Override
172                 public void handleNotification(Notification notification, Object arg1) {
173                   MemoryNotificationInfo info = MemoryNotificationInfo.from((CompositeData) notification.getUserData());
174                   long used = info.getUsage().getUsed();
175                   long max = Runtime.getRuntime().maxMemory();
176     
177                   if (((double)used/(double)max) > MAX_USED_MEM_BEFORE_BACKING_OUT) {
178                     System.err.println("used " + used + " of " + max);
179                     tooMuchMemory = true;
180                     nannyThread.interrupt();
181                   }
182                 }
183               };
184               try {
185                 ManagementFactory.getPlatformMBeanServer().addNotificationListener(gcbean.getObjectName(), listener, null, null);
186               } catch (InstanceNotFoundException e) {
187                 throw new Error("cannot find existing bean");
188               }
189             }
190     
191             Thread.sleep(msPerWorkItem);
192     
193             if (checkMemory) {
194               try {
195               ManagementFactory.getPlatformMBeanServer().removeNotificationListener(gcbean.getObjectName(), listener);
196               } catch (InstanceNotFoundException | ListenerNotFoundException e) {
197                 throw new Error("cannot find existing bean");
198               }
199             }
200             
201             if (isInterrupted()) {
202               return;
203             }
204             
205             timedOut = true;
206           } catch (InterruptedException e) {
207             return;
208           }
209         }
210         
211         private static final double MAX_USED_MEM_BEFORE_BACKING_OUT = .7;
212         
213       }
214     
215       @Override
216       public String getCancelMessage() {
217         return tooMuchMemory? "too much memory": timedOut? "timed out" : "unknown";
218       }
219     
220     }
