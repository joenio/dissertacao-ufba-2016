24     /**
25      * A {@link ContextSelector} to handle default reflection logic.
26      */
27     public class ReflectionContextSelector {
28     
29       public static ContextSelector createReflectionContextSelector(AnalysisOptions options) {
30       
31         if (options == null) {
32           throw new IllegalArgumentException("null options");
33         }
34         
35         // start with a dummy
36         ContextSelector result = new ContextSelector() {
37           @Override
38           public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] receiver) {
39             return null;
40           }
41           @Override
42           public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
43             return EmptyIntSet.instance;
44           }
45         };
46         if (options.getReflectionOptions().getNumFlowToCastIterations() > 0) {
47           result = new DelegatingContextSelector(new FactoryContextSelector(), result);
48         }
49         if (!options.getReflectionOptions().isIgnoreStringConstants()) {
50           result = new DelegatingContextSelector(new DelegatingContextSelector(new DelegatingContextSelector(
51               new ClassFactoryContextSelector(), new GetClassContextSelector()), new ClassNewInstanceContextSelector()), result);
52         }
53         if (!options.getReflectionOptions().isIgnoreMethodInvoke()) {
54           result = new DelegatingContextSelector(new ReflectiveInvocationSelector(), new DelegatingContextSelector(
55               new JavaLangClassContextSelector(), result));
56         }
57         // if NEITHER string constants NOR method invocations are ignored
58         if (!options.getReflectionOptions().isIgnoreStringConstants() && !options.getReflectionOptions().isIgnoreMethodInvoke()) {
59           result = new DelegatingContextSelector(new GetMethodContextSelector(options.getReflectionOptions().isApplicationClassesOnly()),result);
60         }
61         return result;
62       }
63     }
