31     /**
32      * Common utilities for CFA-style call graph builders.
33      */
34     public abstract class JSCFABuilder extends JSSSAPropagationCallGraphBuilder {
35     
36       public JSCFABuilder(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache) {
37         super(cha, options, cache, new AstCFAPointerKeys() {
38     
39           private boolean isBogusKey(InstanceKey K) {
40             TypeReference t = K.getConcreteType().getReference();
41             return t == JavaScriptTypes.Null || t == JavaScriptTypes.Undefined;
42           }
43           
44           @Override
45           public PointerKey getPointerKeyForObjectCatalog(InstanceKey I) {
46             if (isBogusKey(I)) {
47               return null;
48             } else {
49               return super.getPointerKeyForObjectCatalog(I);
50             }
51           }
52     
53           @Override
54           public PointerKey getPointerKeyForInstanceField(InstanceKey I, IField f) {
55             if (isBogusKey(I)) {
56               return null;
57             } else {
58               return super.getPointerKeyForInstanceField(I, f);
59             }
60           }
61     
62           @Override
63           public PointerKey getPointerKeyForArrayContents(InstanceKey I) {
64             if (isBogusKey(I)) {
65               return null;
66             } else {
67               return super.getPointerKeyForArrayContents(I);
68             }
69           }
70     
71           @Override
72           public Iterator<PointerKey> getPointerKeysForReflectedFieldRead(InstanceKey I, InstanceKey F) {
73             if (isBogusKey(I)) {
74               return EmptyIterator.instance();
75             } else {
76               return super.getPointerKeysForReflectedFieldRead(I, F);
77             }
78           }
79     
80           @Override
81           public Iterator<PointerKey> getPointerKeysForReflectedFieldWrite(InstanceKey I, InstanceKey F) {
82              if (isBogusKey(I)) {
83               return EmptyIterator.instance();
84             } else {
85               return super.getPointerKeysForReflectedFieldWrite(I, F);
86             }
87           }
88     
89           @Override
90           protected PointerKey getInstanceFieldPointerKeyForConstant(InstanceKey I, ConstantKey F) {
91             Object v = F.getValue();
92             String strVal = JSCallGraphUtil.simulateToStringForPropertyNames(v);
93             // if we know the string representation of the constant, use it...
94             if (strVal != null) {
95               IField f = I.getConcreteType().getField(Atom.findOrCreateUnicodeAtom(strVal));
96               return getPointerKeyForInstanceField(I, f);
97               
98             // ...otherwise it is some unknown string
99             } else {
100               return ReflectedFieldPointerKey.mapped(new ConcreteTypeKey(getFieldNameType(F)), I);
101             }
102           }
103     
104           /**
105            * All values used as property names get implicitly converted to strings in JavaScript.
106            * @see com.ibm.wala.cast.ipa.callgraph.DelegatingAstPointerKeys#getFieldNameType(com.ibm.wala.ipa.callgraph.propagation.InstanceKey)
107            */
108           @Override
109           protected IClass getFieldNameType(InstanceKey F) {
110             return F.getConcreteType().getClassHierarchy().lookupClass(JavaScriptTypes.String);
111           }      
112         });
113       }
114     
115     }
