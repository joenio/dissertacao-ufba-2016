36       public static AnalysisOptions makeAnalysisOptions(AnalysisScope scope, Iterable<Entrypoint> entrypoints) {
37         AnalysisOptions options = new AnalysisOptions(scope, entrypoints);
38         return options;
39       }

43       /**
44        * should we check the heap footprint before and after CG construction?
45        */
46       private static final boolean CHECK_FOOTPRINT = false;
47     
48       public static AnalysisScope makeJ2SEAnalysisScope(String scopeFile, String exclusionsFile) throws IOException {
49         AnalysisScope scope = AnalysisScopeReader.readJavaScope(scopeFile, (new FileProvider()).getFile(exclusionsFile), MY_CLASSLOADER);
50         return scope;
51       }
52     
53       public static CallGraph buildRTA(AnalysisOptions options, AnalysisCache cache, IClassHierarchy cha, AnalysisScope scope)
54           throws IllegalArgumentException, CancelException {
55         StopwatchGC S = null;
56         if (CHECK_FOOTPRINT) {
57           S = new StopwatchGC("build RTA graph");
58           S.start();
59         }
60     
61         CallGraphBuilder builder = Util.makeRTABuilder(options, cache, cha, scope);
62         CallGraph cg = builder.makeCallGraph(options, null);
63     
64         if (CHECK_FOOTPRINT) {
65           S.stop();
66           System.err.println(S.report());
67         }
68         return cg;
69       }
70     
71       public static CallGraph buildZeroCFA(AnalysisOptions options, AnalysisCache cache, IClassHierarchy cha, AnalysisScope scope,
72           boolean testPAtoString) throws IllegalArgumentException, CancelException {
73         StopwatchGC S = null;

75           S = new StopwatchGC("build RTA graph");
76           S.start();
77         }
78     
79         SSAPropagationCallGraphBuilder builder = Util.makeZeroCFABuilder(options, cache, cha, scope);
80         CallGraph cg = builder.makeCallGraph(options, null);

82           builder.getPointerAnalysis().toString();
83         }
84     
85         if (CHECK_FOOTPRINT) {

88         }
89         return cg;
