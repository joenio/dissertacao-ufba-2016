40     /**
41      * An edge label in a flow graph
42      * 
43      * @author manu
44      * 
45      */
46     public interface IFlowLabel {
47     
48       public void visit(IFlowLabelVisitor v, Object dst);
49     
50       /**
51        * 
52        * @return the bar (inverse) edge corresponding to this edge
53        */
54       public IFlowLabel bar();
55     
56       public interface IFlowLabelVisitor {
57     
58         void visitAssignGlobal(AssignGlobalLabel label, Object dst);
59     
60         void visitAssign(AssignLabel label, Object dst);
61     
62         void visitGetField(GetFieldLabel label, Object dst);
63     
64         void visitMatch(MatchLabel label, Object dst);
65     
66         void visitNew(NewLabel label, Object dst);
67     
68         void visitPutField(PutFieldLabel label, Object dst);
69     
70         void visitParam(ParamLabel label, Object dst);
71     
72         void visitReturn(ReturnLabel label, Object dst);
73     
74         void visitAssignGlobalBar(AssignGlobalBarLabel label, Object dst);
75     
76         void visitAssignBar(AssignBarLabel label, Object dst);
77     
78         void visitGetFieldBar(GetFieldBarLabel label, Object dst);
79     
80         void visitMatchBar(MatchBarLabel label, Object dst);
81     
82         void visitNewBar(NewBarLabel label, Object dst);
83     
84         void visitPutFieldBar(PutFieldBarLabel label, Object dst);
85     
86         void visitReturnBar(ReturnBarLabel label, Object dst);
87     
88         void visitParamBar(ParamBarLabel label, Object dst);
89     
90       }
91     
92       /**
93        * 
94        * @return true if this is a "barred" edge
95        */
96       public boolean isBarred();
97     
98     }
