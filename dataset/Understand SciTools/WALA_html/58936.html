37       public static void addDefaultDispatchLogic(AnalysisOptions options, AnalysisScope scope, IClassHierarchy cha) {
38         com.ibm.wala.ipa.callgraph.impl.Util.addDefaultSelectors(options, cha);
39     
40         Map<Atom,MethodTargetSelector> methodTargetSelectors = HashMapFactory.make();
41         methodTargetSelectors.put(JavaScriptLoader.JS.getName(), new JavaScriptConstructTargetSelector(cha,
42             new StandardFunctionTargetSelector(cha, options.getMethodTargetSelector())));
43         methodTargetSelectors.put(Language.JAVA.getName(), options.getMethodTargetSelector());
44     
45         options.setSelector(new CrossLanguageMethodTargetSelector(methodTargetSelectors));
46       }
