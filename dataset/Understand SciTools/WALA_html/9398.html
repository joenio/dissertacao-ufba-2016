65     /**
66      *  Builds an Android Model incorporating two loops.
67      *
68      *  Functions are inserted in sequence until ExecutionOrder.START_OF_LOOP is reached.
69      *  This loop is closed later when AFTER_LOOP gets stepped over.
70      *
71      *  Functions in MULTIPLE_TIMES_IN_LOOP are in a single inner loop.
72      *
73      *  This structure may be used to model an Application where no state is kept over the 
74      *  restart of the Application (instance-state) or when the potential restart of the App
75      *  shall be ignored.
76      *
77      *  {@inheritDoc}
78      *
79      *  @author     Tobias Blaschke <code@tobiasblaschke.de>
80      */
81     public class LoopAndroidModel extends SingleStartAndroidModel {
82         private static final Logger logger = LoggerFactory.getLogger(LoopAndroidModel.class);
83         
84         //protected VolatileMethodSummary body;
85         //protected JavaInstructionFactory insts;
86         //protected DexFakeRootMethod.ReuseParameters paramTypes;
87         /**
88          *
89          *  @param  body    The MethodSummary to add instructions to
90          *  @param  insts   Will be used to generate the instructions
91          */
92         public LoopAndroidModel(VolatileMethodSummary body, TypeSafeInstructionFactory insts,
93                 SSAValueManager paramManager, Iterable<? extends Entrypoint> entryPoints) {
94             super(body, insts, paramManager, entryPoints);
95         }
96     
97         private int outerLoopPC = -1;
98         Map<TypeReference, SSAValue> outerStartingPhis;
99         /**
100          * Prepares the PC to get looped to.
101          *
102          * Thus it tries to assure a new basic block starts here. Additionally it reserves some
103          * space for the insertion of Phi-Functions.
104          *
105          * {@inheritDoc}
106          */
107         protected int enterSTART_OF_LOOP (int PC) {
108             logger.info("PC {} is the jump target of START_OF_LOOP", PC);
109             
110             this.outerLoopPC = PC;
111             PC = body.getNextProgramCounter();
112             paramManager.scopeDown(true);
113     
114             // Top-Half of Phi-Handling
115             outerStartingPhis = new HashMap<TypeReference, SSAValue>();
116             List<TypeReference> outerPhisNeeded = returnTypesBetween(ExecutionOrder.START_OF_LOOP,
117                     ExecutionOrder.AFTER_LOOP);
118             
119             for (TypeReference phiType: outerPhisNeeded) {
120                 final TypeKey phiKey = new TypeKey(phiType.getName());
121                 if (paramManager.isSeen(phiKey, false)) {
122                     final SSAValue newValue = paramManager.getFree(phiType, phiKey);
123                     outerStartingPhis.put(phiType, newValue);
124                 }
125             }
126     
127             body.reserveProgramCounters (outerPhisNeeded.size());
128             // Actual Phis will be placed by the bottom-half handler...
129     
130             PC = body.getNextProgramCounter();  // Needed if no calls
131             return PC;
132         }
133     
134         /**
135          *  Loops to START_OF_LOOP.
136          *
137          *  It inserts a gotoInstruction and fills the space reserved before with actual PhiInstructions
138          *
139          *  {@inheritDoc}
140          */
141         protected int enterAFTER_LOOP (int PC) {
142             assert(outerLoopPC > 0) : "Somehow you managed to get the loop-target negative. This is wierd!";
143     
144             // Insert the Phis at the beginning of the Block
145             int phiPC = outerLoopPC + 1;
146             boolean oldAllowReserved = body.allowReserved(true);
147             logger.info("Setting block-inner Phis");
148             for (TypeReference phiType : outerStartingPhis.keySet()) {
149                 final SSAValue oldPhi = outerStartingPhis.get(phiType);
150                 final List<SSAValue> forPhi = new ArrayList<SSAValue>(2);
151                 forPhi.add(paramManager.getSuper(oldPhi.key));
152                 forPhi.add(paramManager.getCurrent(oldPhi.key));
153                 
154                 SSAPhiInstruction phi = insts.PhiInstruction(phiPC, oldPhi, forPhi); 
155                 phiPC++;
156                 body.addStatement(phi);
157                 paramManager.setPhi(oldPhi, phi);
158             }
159             body.allowReserved(oldAllowReserved);
160            
161             // Close the Loop
162             logger.info("Closing Loop");
163             logger.info("PC {}: Goto {}", PC, outerLoopPC);
164             NamedKey trueKey = new SSAValue.NamedKey(TypeReference.BooleanName, "true");
165             SSAValue trueVal = paramManager.getFree(TypeReference.Boolean, trueKey);
166             paramManager.setPhi(trueVal, null);
167             body.addConstant(trueVal.getNumber(), new ConstantValue(true));
168             body.addStatement(insts.ConditionalBranchInstruction(PC, IConditionalBranchInstruction.Operator.EQ, TypeReference.Boolean, trueVal.getNumber(), trueVal.getNumber(), outerLoopPC));
169             paramManager.scopeUp();
170             
171             // Add Phi-Statements at the beginning of this block...
172             logger.info("Setting outer-block Phis");
173             for (TypeReference phiType : outerStartingPhis.keySet()) {
174                 final VariableKey  phiKey = outerStartingPhis.get(phiType).key;
175                 PC = body.getNextProgramCounter();
176     
177                 List<SSAValue> all = paramManager.getAllForPhi(phiKey);
178                 logger.debug("Into phi {} for {}", all, phiType.getName());
179                 // Narf ... unpacking...
180     
181                 paramManager.invalidate(phiKey);
182                 final SSAValue newValue = paramManager.getFree(phiType, phiKey);
183                 SSAPhiInstruction phi = insts.PhiInstruction(PC, newValue, all);
184                 body.addStatement(phi);
185     
186                 paramManager.setPhi(newValue, phi);
187             }
188             PC = body.getNextProgramCounter();
189             return PC;
190         }
191     
192         /**
193          *  Does not insert any special handling.
194          *
195          *  {@inheritDoc}
196          */
197         protected int leaveAT_LAST (int PC) {
198             logger.info("Leaving Model with PC = {}", PC);
199             return PC;
200         }

202     }
