91         @SuppressWarnings("unchecked")
92     	public static <E extends ISSABasicBlock>
93         void addDomainElements(
94                 Map<BasicBlockInContext<E>, Map<FlowType<E>,Set<CodeElement>>> taintMap, 
95                 BasicBlockInContext<E> block, 
96                 FlowType taintType, 
97                 Set<CodeElement> newElements) {
98             Map<FlowType<E>,Set<CodeElement>> blockMap = taintMap.get(block);
99             if(blockMap == null) {
100                 blockMap = new HashMap<FlowType<E>,Set<CodeElement>>();
101                 taintMap.put(block, blockMap);
102             }
103             
104             Set<CodeElement> elements = blockMap.get(taintType);
105             if(elements == null) {
106                 elements = new HashSet<CodeElement>();
107                 blockMap.put(taintType, elements);
108             }
109             elements.addAll(newElements);
110         }
111     
112         public static <E extends ISSABasicBlock>
113         void addDomainElement(
114                 Map<BasicBlockInContext<E>, Map<FlowType<E>,Set<CodeElement>>> taintMap,
115                 BasicBlockInContext<E> block,
116                 FlowType taintType,
117                 CodeElement element) {
118             
119             Set<CodeElement> elements = new HashSet<CodeElement>();
120             elements.add(element);
121             addDomainElements(taintMap, block, taintType, elements);
122         }

125         private static<E extends ISSABasicBlock> 
126         void processInputSource(CGAnalysisContext<E> ctx,
127         						Map<BasicBlockInContext<E>,
128                                 Map<FlowType<E>,Set<CodeElement>>> taintMap, 
129                                 SourceSpec ss, 
130                                 CallGraph cg, 
131                                 ISupergraph<BasicBlockInContext<E>, CGNode> graph,
132                                 ClassHierarchy cha,
133                                 PointerAnalysis<InstanceKey> pa) {
134         	int[] newArgNums;
135         	for (IMethod im:ss.getNamePattern().getPossibleTargets(cha)) {
136                 newArgNums = (ss.getArgNums() == null) ? SourceSpec.getNewArgNums((im.isStatic())?im.getNumberOfParameters():im.getNumberOfParameters()-1) : ss.getArgNums();
137                 for (CGNode node: cg.getNodes(im.getReference())) {
138                     BasicBlockInContext<E>[] entriesForProcedure = graph.getEntriesForProcedure(node);
139                     if (entriesForProcedure == null || 0 == entriesForProcedure.length) {
140                         continue;
141                     }
142                 
143                     for (BasicBlockInContext<E> bb:entriesForProcedure) {
144                         ss.addDomainElements(ctx, taintMap, im, bb, null, newArgNums, graph, pa, cg);
145                     }
146                 }
147         	}
148         }

150         private static<E extends ISSABasicBlock> 
151         void processStaticFieldSource(CGAnalysisContext<E> ctx, 
152         						Map<BasicBlockInContext<E>,
153                                 Map<FlowType<E>,Set<CodeElement>>> taintMap, 
154                                 StaticFieldSourceSpec ss, 
155                                 CallGraph cg, 
156                                 ISupergraph<BasicBlockInContext<E>, CGNode> graph,
157                                 ClassHierarchy cha,
158                                 PointerAnalysis<InstanceKey> pa) {
159         	// get the first block:
160         	BasicBlockInContext<E> bb = null;
161         	for (CGNode n : cg.getEntrypointNodes() ){
162         		 bb = graph.getEntriesForProcedure(n)[0];
163         	}
164         	
165         	assert bb != null : "Could not find entry basic block.";
166         	    	
167         	ss.addDomainElements(ctx, taintMap, bb.getMethod(), bb, null, null, graph, pa, cg);
168         }

170         private static<E extends ISSABasicBlock> 
171         void processFunctionCalls(CGAnalysisContext<E> ctx,
172         						  Map<BasicBlockInContext<E>,
173                                   Map<FlowType<E>,Set<CodeElement>>> taintMap, 
174                                   ArrayList<SourceSpec> ssAL, ISupergraph<BasicBlockInContext<E>, CGNode> graph, 
175                                   PointerAnalysis<InstanceKey> pa, 
176                                   ClassHierarchy cha, CallGraph cg) {
177         	Collection<IMethod> targets = new HashSet<IMethod>();
178         	ArrayList<Collection<IMethod>> targetList = new ArrayList<Collection<IMethod>>();
179         	
180         	for (int i = 0; i < ssAL.size(); i++) {
181         		Collection<IMethod> tempList = ssAL.get(i).getNamePattern().getPossibleTargets(cha);
182         		targets.addAll(tempList);
183         		targetList.add(tempList);
184         	}
185         	
186     		Iterator<BasicBlockInContext<E>> graphIt = graph.iterator();
187     		while (graphIt.hasNext()) {
188     			BasicBlockInContext<E> block = graphIt.next();
189     			Iterator<SSAInstruction> instructions = block.iterator();
190     
191     			while (instructions.hasNext()) {
192     				SSAInstruction inst = instructions.next();
193     
194     				if (!(inst instanceof SSAInvokeInstruction)) {
195     					continue;
196     				}
197     
198     				SSAInvokeInstruction invInst = (SSAInvokeInstruction) inst;
199     				for (IMethod target : cha.getPossibleTargets(invInst.getDeclaredTarget())) {
200     					if (targets.contains(target)) {
201     						for (int i = 0; i < targetList.size(); i++) {
202     							if (targetList.get(i).contains(target)) {
203     
204     								int[] argNums = ssAL.get(i).getArgNums();
205     								argNums = (argNums == null) ? SourceSpec.getNewArgNums((target.isStatic())?target.getNumberOfParameters():target.getNumberOfParameters()-1) : argNums;
206     
207     								ssAL.get(i).addDomainElements(ctx, taintMap, target, block, invInst, argNums, graph, pa, cg);
208     
209     							}
210     						}
211     					}
212     				}
213     			}
214     		}
215         }

217         public static <E extends ISSABasicBlock>
218           Map<BasicBlockInContext<E>,Map<FlowType<E>,Set<CodeElement>>> analyze(
219                 CGAnalysisContext<E> analysisContext, Map<InstanceKey, String> prefixes,
220                 ISpecs s) {
221             return analyze(analysisContext, analysisContext.cg, analysisContext.getClassHierarchy(), analysisContext.graph, analysisContext.pa, prefixes, s);
222         }
223     
224         public static <E extends ISSABasicBlock>
225           Map<BasicBlockInContext<E>,Map<FlowType<E>,Set<CodeElement>>> 
226         analyze(CGAnalysisContext<E> ctx,
227         	  CallGraph cg, 
228               ClassHierarchy cha, 
229               ISupergraph<BasicBlockInContext<E>, CGNode> graph,
230               PointerAnalysis<InstanceKey> pa, 
231               Map<InstanceKey, String> prefixes,
232               ISpecs s) {
233     
234             Map<BasicBlockInContext<E>, Map<FlowType<E>,Set<CodeElement>>> taintMap = HashMapFactory.make();
235     
236             SourceSpec[] ss = s.getSourceSpecs();
237             
238             ArrayList<SourceSpec> ssAL = new ArrayList<SourceSpec>();
239             for (int i = 0; i < ss.length; i++) {
240             	if (ss[i] instanceof EntryArgSourceSpec)
241             		processInputSource(ctx, taintMap, ss[i], cg, graph, cha, pa);
242             	else if (ss[i] instanceof CallRetSourceSpec || ss[i] instanceof CallArgSourceSpec)
243             		ssAL.add(ss[i]);
244             	else if (ss[i] instanceof StaticFieldSourceSpec) {
245             		processStaticFieldSource(ctx, taintMap, (StaticFieldSourceSpec)ss[i], cg, graph, cha, pa);
246             	} else 
247             		throw new UnsupportedOperationException("Unrecognized SourceSpec");
248             } 
249             if (!ssAL.isEmpty())
250             	processFunctionCalls(ctx, taintMap, ssAL, graph, pa, cha, cg);
251     
252             return taintMap;
253         }
