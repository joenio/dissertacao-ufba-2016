14     package com.ibm.wala.cast.java.translator.polyglot;
15     
16     import java.util.ArrayList;
17     import java.util.Collection;
18     import java.util.Collections;
19     import java.util.Iterator;
20     
21     import polyglot.types.ArrayType;
22     import polyglot.types.ClassType;
23     import polyglot.types.ObjectType;
24     import polyglot.types.PrimitiveType;
25     import polyglot.types.Type;
26     import polyglot.types.TypeSystem;
27     
28     import com.ibm.wala.cast.java.types.JavaPrimitiveTypeMap;
29     import com.ibm.wala.cast.tree.CAstType;
30     import com.ibm.wala.cast.tree.impl.CAstTypeDictionaryImpl;
31     import com.ibm.wala.util.debug.Assertions;
32     
33     public class PolyglotTypeDictionary extends CAstTypeDictionaryImpl {
34       private final class PolyglotJavaArrayType implements CAstType.Array {
35         private final Type fEltPolyglotType;
36     
37         private final CAstType fEltCAstType;
38     
39         private PolyglotJavaArrayType(ArrayType arrayType) {
40           super();
41           fEltPolyglotType = arrayType.base();
42           fEltCAstType = getCAstTypeFor(fEltPolyglotType);
43         }
44     
45         @Override
46         public int getNumDimensions() {
47           return 1; // always 1 for Java
48         }
49     
50         @Override
51         public CAstType getElementType() {
52           return fEltCAstType;
53         }
54     
55         @Override
56         public String getName() {
57           return "[" + fEltCAstType.getName();
58         }
59     
60         @Override
61         public Collection<CAstType> getSupertypes() {
62           if (fEltPolyglotType.isPrimitive())
63             return Collections.singleton(getCAstTypeFor(fTypeSystem.Object()));
64           Assertions.productionAssertion(fEltPolyglotType.isReference(), "Non-primitive, non-reference array element type!");
65           ObjectType baseRefType = (ObjectType) fEltPolyglotType;
66           Collection<CAstType> supers = new ArrayList<CAstType>();
67           for (Iterator<Type> superIter = baseRefType.interfaces().iterator(); superIter.hasNext(); ) {
68             supers.add(getCAstTypeFor(superIter.next()));
69           }
70           if (baseRefType instanceof ClassType) {
71               ClassType baseClassType = (ClassType) baseRefType;
72             if (baseClassType.superClass() != null)
73                   supers.add(getCAstTypeFor(baseRefType.superClass()));
74           }
75           return supers;
76         }
77       }
78     
79       protected final TypeSystem fTypeSystem;
80     
81       protected final PolyglotJava2CAstTranslator fTranslator;
82     
83       public PolyglotTypeDictionary(TypeSystem typeSystem, PolyglotJava2CAstTranslator translator) {
84         fTypeSystem = typeSystem;
85         fTranslator = translator;
86       }
87     
88       @Override
89       public CAstType getCAstTypeFor(Object astType) {
90         CAstType type = super.getCAstTypeFor(astType);
91         // Handle the case where we haven't seen an AST decl for some type before
92         // processing a reference. This can certainly happen with classes in byte-
93         // code libraries, for which we never see an AST decl.
94         // In this case, just create a new type and return that.
95         if (type == null) {
96           final Type polyglotType = (Type) astType;
97     
98           if (polyglotType.isClass())
99             type = fTranslator.new PolyglotJavaType((ClassType) astType, this, fTypeSystem);
100           else if (polyglotType.isPrimitive()) {
101             type = JavaPrimitiveTypeMap.lookupType(((PrimitiveType) polyglotType).name().toString());
102           } else if (polyglotType.isArray()) {
103             type = new PolyglotJavaArrayType((ArrayType) polyglotType);
104           } else
105             Assertions.UNREACHABLE("getCAstTypeFor() passed type that is not primitive, array, or class?");
106           super.map(astType, type);
107         }
108         return type;
109       }
110     }
