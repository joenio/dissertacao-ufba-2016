45     /**
46      * A trivial field-based heap model, which only uses the information of which types (classes) are live.
47      * 
48      * Note that this heap model is based on ssa value numbers for locals, since we will build a pointer flow graph based on this heap
49      * model when resolving reflection.
50      * 
51      * This is an inefficient prototype.
52      */
53     public class TypeBasedHeapModel implements HeapModel {
54     
55       private final static boolean DEBUG = false;
56     
57       final DefaultPointerKeyFactory pointerKeys = new DefaultPointerKeyFactory();
58     
59       private final ClassBasedInstanceKeys iKeyFactory;
60     
61       private final Collection<IClass> klasses;
62     
63       private final CallGraph cg;
64     
65       private final Collection<CGNode> nodesHandled = HashSetFactory.make();
66     
67       /**
68        * Map: <PointerKey> -> thing, where thing is a FilteredPointerKey or an InstanceKey representing a constant.
69        * 
70        * computed lazily
71        */
72       private Map<PointerKey, Object> pKeys;
73     
74       /**
75        * @param klasses Collection<IClass>
76        * @throws IllegalArgumentException if cg is null
77        */
78       public TypeBasedHeapModel(AnalysisOptions options, Collection<IClass> klasses, CallGraph cg) {
79         if (cg == null) {
80           throw new IllegalArgumentException("cg is null");
81         }
82         iKeyFactory = new ClassBasedInstanceKeys(options, cg.getClassHierarchy());
83         this.klasses = klasses;
84         this.cg = cg;
85       }
86     
87       private void initAllPKeys() {
88         if (pKeys == null) {
89           pKeys = HashMapFactory.make();
90         }
91         for (Iterator<IClass> it = klasses.iterator(); it.hasNext();) {
92           IClass klass = it.next();
93           pKeys.putAll(computePointerKeys(klass));
94         }
95         for (Iterator it = cg.iterator(); it.hasNext();) {
96           CGNode node = (CGNode) it.next();
97           initPKeysForNode(node);
98         }
99       }
100     
101       private void initPKeysForNode(CGNode node) {
102         if (pKeys == null) {
103           pKeys = HashMapFactory.make();
104         }
105         if (!nodesHandled.contains(node)) {
106           nodesHandled.add(node);
107           pKeys.putAll(computePointerKeys(node));
108         }
109       }
110     
111       private Map<PointerKey, Object> computePointerKeys(CGNode node) {
112     
113         if (DEBUG) {
114           System.err.println("computePointerKeys " + node);
115         }
116     
117         IR ir = node.getIR();
118         if (ir == null) {
119           return Collections.emptyMap();
120         }
121         Map<PointerKey, Object> result = HashMapFactory.make();
122         SymbolTable s = ir.getSymbolTable();
123         if (s == null) {
124           return Collections.emptyMap();
125         }
126         TypeInference ti = TypeInference.make(ir, false);
127     
128         for (int i = 1; i <= s.getMaxValueNumber(); i++) {
129           if (DEBUG) {
130             System.err.print(i);
131           }
132           if (s.isConstant(i)) {
133             if (s.isStringConstant(i)) {
134               TypeReference type = node.getMethod().getDeclaringClass().getClassLoader().getLanguage().getConstantType(
135                   s.getStringValue(i));
136               result.put(pointerKeys.getPointerKeyForLocal(node, i), getInstanceKeyForConstant(type, s.getConstantValue(i)));
137             }
138           } else {
139             TypeAbstraction t = ti.getType(i);
140             if (DEBUG) {
141               System.err.println(" type " + t);
142             }
143             if (t.getType() != null && t.getType().isReferenceType()) {
144               result.put(pointerKeys.getPointerKeyForLocal(node, i), pointerKeys.getFilteredPointerKeyForLocal(node, i,
145                   new FilteredPointerKey.SingleClassFilter(t.getType())));
146             }
147           }
148         }
149         return result;
150       }
151     
152       private Map<PointerKey, Object> computePointerKeys(IClass klass) {
153         Map<PointerKey, Object> result = HashMapFactory.make();
154         if (klass.isArrayClass()) {
155           ArrayClass a = (ArrayClass) klass;
156           if (a.getElementClass() != null && a.getElementClass().isReferenceType()) {
157             PointerKey p = pointerKeys.getPointerKeyForArrayContents(new ConcreteTypeKey(a));
158             result.put(p, p);
159           }
160         } else {
161           for (Iterator<IField> it = klass.getAllFields().iterator(); it.hasNext();) {
162             IField f = it.next();
163             if (!f.getFieldTypeReference().isPrimitiveType()) {
164               if (f.isStatic()) {
165                 PointerKey p = pointerKeys.getPointerKeyForStaticField(f);
166                 result.put(p, p);
167               } else {
168                 PointerKey p = pointerKeys.getPointerKeyForInstanceField(new ConcreteTypeKey(klass), f);
169                 result.put(p, p);
170               }
171             }
172           }
173         }
174         return result;
175       }

177       @Override
178       @SuppressWarnings("unchecked")
179       public Iterator<PointerKey> iteratePointerKeys() {
180         initAllPKeys();
181         return Predicate.filter(pKeys.values().iterator(), new Predicate() {
182           @Override
183           public boolean test(Object o) {
184             return o instanceof PointerKey;
185           }
186         }).iterator();
187       }

226       /**
227        * Note that this always returns a {@link FilteredPointerKey}, since the {@link TypeBasedPointerAnalysis} relies on the type
228        * filter to compute points to sets.
229        * 
230        * @see com.ibm.wala.ipa.callgraph.propagation.PointerKeyFactory#getPointerKeyForLocal(com.ibm.wala.ipa.callgraph.CGNode, int)
231        */
232       @Override
233       public FilteredPointerKey getPointerKeyForLocal(CGNode node, int valueNumber) {
234         initPKeysForNode(node);
235         PointerKey p = pointerKeys.getPointerKeyForLocal(node, valueNumber);
236         Object result = pKeys.get(p);
237         if (result == null) {
238           // a null constant
239           return null;
240         }
241         if (result instanceof FilteredPointerKey) {
242           return (FilteredPointerKey) result;
243         } else {
244           if (result instanceof ConcreteTypeKey) {
245             ConcreteTypeKey c = (ConcreteTypeKey) result;
246             if (c.getConcreteType().getReference().equals(TypeReference.JavaLangString)) {
247               // a string constant;
248               return pointerKeys.getFilteredPointerKeyForLocal(node, valueNumber, new FilteredPointerKey.SingleClassFilter(c
249                   .getConcreteType()));
250             } else {
251               Assertions.UNREACHABLE("need to handle " + result.getClass());
252               return null;
253             }
254           } else {
255             Assertions.UNREACHABLE("need to handle " + result.getClass());
256             return null;
257           }
258         }
259       }
