11     package com.ibm.wala.util.intset;
12     
13     import java.io.Serializable;
14     import java.util.Arrays;
15     
16     /**
17      * simple implementation of IntVector
18      */
19     public class SimpleIntVector implements IntVector, Serializable {
20       
21       private final static int MAX_SIZE = Integer.MAX_VALUE / 4;
22     
23       private final static float GROWTH_FACTOR = 1.5f;
24     
25       private final static int INITIAL_SIZE = 1;
26       
27       int maxIndex = -1;
28     
29       int[] store;
30     
31       final int defaultValue;
32       
33       public SimpleIntVector(int defaultValue) {
34         this.defaultValue = defaultValue;
35         store = new int[getInitialSize()];
36         store[0] = defaultValue;
37       }
38     
39       public SimpleIntVector(int defaultValue, int initialSize) {
40         if (initialSize <= 0) {
41           throw new IllegalArgumentException("Illegal initialSize: " + initialSize);
42         }
43         this.defaultValue = defaultValue;
44         store = new int[initialSize];
45         store[0] = defaultValue;
46       }
47     
48       int getInitialSize() {
49         return INITIAL_SIZE;
50       }
51     
52       float getGrowthFactor() {
53         return GROWTH_FACTOR;
54       }
55     
56       /*
57        * @see com.ibm.wala.util.intset.IntVector#get(int)
58        */
59       @Override
60       public int get(int x) {
61         if (x < 0) {
62           throw new IllegalArgumentException("illegal x: " + x);
63         }
64         if (x < store.length) {
65           return store[x];
66         } else {
67           return defaultValue;
68         }
69       }
70     
71       /*
72        * @see com.ibm.wala.util.intset.IntVector#set(int, int)
73        */
74       @Override
75       public void set(int x, int value) {
76         if (x < 0) {
77           throw new IllegalArgumentException("illegal x: " + x);
78         }
79         if (x > MAX_SIZE) {
80           throw new IllegalArgumentException("x is too big: " + x);
81         }
82         maxIndex = Math.max(maxIndex,x);
83         if (value == defaultValue) {
84           if (x >= store.length) {
85             return;
86           } else {
87             store[x] = value;
88           }
89         } else {
90           ensureCapacity(x);
91           store[x] = value;
92         }
93       }
94     
95       /**
96        * make sure we can store to a particular index
97        * 
98        * @param capacity
99        */
100       private void ensureCapacity(int capacity) {
101         if (capacity >= store.length) {
102           int[] old = store;
103           store = new int[1 + (int) (getGrowthFactor() * capacity)];
104           Arrays.fill(store, defaultValue);
105           System.arraycopy(old, 0, store, 0, old.length);
106         }
107       }
108     
109       /*
110        * @see com.ibm.wala.util.intset.IntVector#reportStats()
111        */
112       public void performVerboseAction() {
113         System.err.println(("size:       " + store.length));
114         System.err.println(("occupancy:  " + computeOccupancy()));
115       }
116     
117       /**
118        * @return the percentage of entries in delegateStore that are non-null
119        */
120       private double computeOccupancy() {
121         int count1 = 0;
122         for (int i = 0; i < store.length; i++) {
123           if (store[i] != -1) {
124             count1++;
125           }
126         }
127         int count = count1;
128         return (double) count / (double) store.length;
129       }
130       
131       @Override
132       public int getMaxIndex() {
133         return maxIndex;
134       }
135     
136     }
