21     /**
22      * A method descriptor; something like: (Ljava/langString;)Ljava/lang/Class;
23      * 
24      * Descriptors are canonical
25      */
26     public final class Descriptor {
27     
28       /**
29        * A mapping from Key -> Descriptor
30        */
31       private static final Map<Key, Descriptor> map = HashMapFactory.make();
32     
33       /**
34        * key holds the logical value of this descriptor
35        */
36       private final Key key;
37     
38       /**
39        * @param parameters the parameters for a descriptor
40        * @param returnType the return type
41        * @return the canonical representative for this descriptor value
42        */
43       public static Descriptor findOrCreate(TypeName[] parameters, TypeName returnType) {
44         if (returnType == null) {
45           throw new IllegalArgumentException("null returnType");
46         }
47         if (parameters != null && parameters.length == 0) {
48           parameters = null;
49         }
50         Key k = new Key(returnType, parameters);
51         Descriptor result = map.get(k);
52         if (result == null) {
53           result = new Descriptor(k);
54           map.put(k, result);
55         }
56         return result;
57       }
58     
59       /**
60        * @param b a byte array holding the string representation of this descriptor
61        * @return the canonical representative for this descriptor value
62        */
63       public static Descriptor findOrCreate(Language l, ImmutableByteArray b) throws IllegalArgumentException {
64         TypeName returnType = StringStuff.parseForReturnTypeName(l, b);
65         TypeName[] parameters = StringStuff.parseForParameterNames(l, b);
66         Key k = new Key(returnType, parameters);
67         Descriptor result = map.get(k);
68         if (result == null) {
69           result = new Descriptor(k);
70           map.put(k, result);
71         }
72         return result;
73       }
74     
75       public static Descriptor findOrCreate(ImmutableByteArray b) throws IllegalArgumentException {
76         return findOrCreate(Language.JAVA, b);
77       }
78     
79       /**
80        * @param s string representation of this descriptor
81        * @return the canonical representative for this descriptor value
82        */
83       public static Descriptor findOrCreateUTF8(String s) throws IllegalArgumentException {
84         return findOrCreateUTF8(Language.JAVA, s);
85       }
86     
87       /**
88        * @param s string representation of this descriptor
89        * @return the canonical representative for this descriptor value
90        */
91       public static Descriptor findOrCreateUTF8(Language l, String s) throws IllegalArgumentException {
92         byte[] b = UTF8Convert.toUTF8(s);
93         return findOrCreate(l, new ImmutableByteArray(b));
94       }

96       /**
97        * @param key "value" of this descriptor
98        */
99       private Descriptor(Key key) {
100         this.key = key;
101       }
102     
103       @Override
104       public boolean equals(Object obj) {
105         return this == obj;
106       }
107     
108       @Override
109       public int hashCode() {
110         return key.hashCode();
111       }
112     
113       @Override
114       public String toString() {
115         return key.toString();
116       }
117     
118       /**
119        * @return a unicode string representation of this descriptor
120        */
121       public String toUnicodeString() {
122         return key.toUnicodeString();
123       }
124     
125       /**
126        * @return the name of the return type of this descriptor
127        */
128       public TypeName getReturnType() {
129         return key.returnType;
130       }
131     
132       /**
133        * @return the type names for the parameters in this descriptor
134        */
135       public TypeName[] getParameters() {
136         return key.parameters;
137       }
138     
139       /**
140        * @return number of parameters in this descriptor
141        */
142       public int getNumberOfParameters() {
143         return key.parameters == null ? 0 : key.parameters.length;
144       }
145     
146       /**
147        * value that defines a descriptor: used to canonicalize instances
148        */
149       private static class Key {
150         final private TypeName returnType;
151     
152         final private TypeName[] parameters;
153     
154         final private int hashCode; // cached for efficiency
155     
156         Key(TypeName returnType, TypeName[] parameters) {
157           this.returnType = returnType;
158           this.parameters = parameters;
159           if (parameters != null) {
160             assert parameters.length > 0;
161           }
162           hashCode = computeHashCode();
163         }
164     
165         @Override
166         public int hashCode() {
167           return hashCode;
168         }
169     
170         public int computeHashCode() {
171           int result = returnType.hashCode() * 5309;
172           if (parameters != null) {
173             for (int i = 0; i < parameters.length; i++) {
174               result += parameters[i].hashCode() * (5323 ^ i);
175             }
176           }
177           return result;
178         }
179     
180         @Override
181         public boolean equals(Object obj) {
182           assert obj instanceof Key;
183           Key other = (Key) obj;
184           if (!returnType.equals(other.returnType)) {
185             return false;
186           }
187           if (parameters == null) {
188             return (other.parameters == null) ? true : false;
189           }
190           if (other.parameters == null) {
191             return false;
192           }
193           if (parameters.length != other.parameters.length) {
194             return false;
195           }
196           for (int i = 0; i < parameters.length; i++) {
197             if (!parameters[i].equals(other.parameters[i])) {
198               return false;
199             }
200           }
201           return true;
202         }
203     
204         @Override
205         public String toString() {
206           StringBuffer result = new StringBuffer();
207           result.append("(");
208           if (parameters != null) {
209             for (int i = 0; i < parameters.length; i++) {
210               TypeName p = parameters[i];
211               result.append(p);
212               appendSemicolonIfNeeded(result, p);
213             }
214           }
215           result.append(")");
216           result.append(returnType);
217           appendSemicolonIfNeeded(result, returnType);
218           return result.toString();
219         }
220     
221         public String toUnicodeString() {
222           StringBuffer result = new StringBuffer();
223           result.append("(");
224           if (parameters != null) {
225             for (int i = 0; i < parameters.length; i++) {
226               TypeName p = parameters[i];
227               result.append(p.toUnicodeString());
228               appendSemicolonIfNeeded(result, p);
229             }
230           }
231           result.append(")");
232           result.append(returnType);
233           appendSemicolonIfNeeded(result, returnType);
234           return result.toString();
235         }
236     
237         private void appendSemicolonIfNeeded(StringBuffer result, TypeName p) {
238           if (p.isArrayType()) {
239             TypeName e = p.getInnermostElementType();
240             String x = e.toUnicodeString();
241             if (x.startsWith("L") || x.startsWith("P")) {
242               result.append(";");
243             }
244           } else {
245             String x = p.toUnicodeString();
246             if (x.startsWith("L") || x.startsWith("P")) {
247               result.append(";");
248             }
249           }
250         }
251       }
