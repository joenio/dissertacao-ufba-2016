341       /**
342        * Evaluate a meet of machine states.
343        * 
344        * TODO: add some efficiency shortcuts. TODO: clean up and refactor.
345        * 
346        * @param bb the basic block at whose entry the meet occurs
347        * @return true if the lhs value changes. false otherwise.

349       private boolean meet(IVariable lhs, IVariable[] rhs, BasicBlock bb, Meeter meeter) {
350     
351         boolean changed = meetStacks(lhs, rhs, bb, meeter);
352     
353         changed |= meetLocals(lhs, rhs, bb, meeter);
354         return changed;
355       }

358        * Evaluate a meet of machine states at a catch block.
359        *

363        * @return true if the lhs value changes. false otherwise.

365       private boolean meetForCatchBlock(IVariable lhs, IVariable[] rhs, BasicBlock bb, Meeter meeter) {
366     
367         boolean changed = meetStacksAtCatchBlock(lhs, bb, meeter);
368         changed |= meetLocals(lhs, rhs, bb, meeter);

372       /**
373        * Evaluate a meet of the stacks of machine states at the entry of a catch block.
374        * 
375        * TODO: add some efficiency shortcuts. TODO: clean up and refactor.
376        * 
377        * @param bb the basic block at whose entry the meet occurs
378        * @return true if the lhs value changes. false otherwise.
379        */
380       private boolean meetStacksAtCatchBlock(IVariable lhs, BasicBlock bb, Meeter meeter) {
381         boolean changed = false;
382         MachineState L = (MachineState) lhs;
383     
384         // evaluate the meet of the stack of height 1, which holds the exception
385         // object.
386     
387         // allocate lhs.stack if it's
388         // not already allocated.
389         if (L.stack == null) {
390           L.allocateStack(1);
391           L.stackHeight = 1;
392         }
393     
394         int meet = meeter.meetStackAtCatchBlock(bb);
395         if (L.stack[0] == TOP) {
396           if (meet != TOP) {
397             changed = true;
398             L.stack[0] = meet;
399           }
400         } else if (meet != L.stack[0]) {
401           changed = true;
402           L.stack[0] = meet;
403         }
404         return changed;
405       }

407       /**
408        * Evaluate a meet of the stacks of machine states at the entry of a basic block.
409        * 
410        * TODO: add some efficiency shortcuts. TODO: clean up and refactor.
411        * 
412        * @param bb the basic block at whose entry the meet occurs
413        * @return true if the lhs value changes. false otherwise.
414        */
415       private boolean meetStacks(IVariable lhs, IVariable[] rhs, BasicBlock bb, Meeter meeter) {
416         boolean changed = false;
417         MachineState L = (MachineState) lhs;
418     
419         // evaluate the element-wise meet over the stacks
420     
421         // first ... how high are the stacks?
422         int height = computeMeetStackHeight(rhs);
423     
424         // if there's any stack height to meet, allocate lhs.stack if it's
425         // not already allocated.
426         if (height > -1 && (L.stack == null || L.stack.length < height)) {
427           L.allocateStack(height);
428           L.stackHeight = height;
429           changed = true;
430         }
431     
432         // now do the element-wise meet.
433         for (int i = 0; i < height; i++) {
434           int[] R = new int[rhs.length];
435           for (int j = 0; j < R.length; j++) {
436             MachineState m = (MachineState) rhs[j];
437             if (m.stack == null || m.stack.length < i+1) {
438               R[j] = TOP;
439             } else {
440               R[j] = m.stack[i];
441               if (R[j] == 0) {
442                 R[j] = TOP;
443               }
444             }
445           }
446           int meet = meeter.meetStack(i, R, bb);
447           if (L.stack[i] == TOP) {
448             if (meet != TOP) {
449               changed = true;
450               L.stack[i] = meet;
451             }
452           } else if (meet != L.stack[i]) {
453             changed = true;
454             L.stack[i] = meet;
455           }
456         }
457         return changed;
458       }

460       /**
461        * Evaluate a meet of locals of machine states at the entry to a basic block.
462        * 
463        * TODO: add some efficiency shortcuts. TODO: clean up and refactor.
464        * 
465        * @param bb the basic block at whose entry the meet occurs
466        * @return true if the lhs value changes. false otherwise.
467        */
468       private boolean meetLocals(IVariable lhs, IVariable[] rhs, BasicBlock bb, Meeter meeter) {
469     
470         boolean changed = false;
471         MachineState L = (MachineState) lhs;
472         // need we allocate lhs.locals?
473         int nLocals = computeMeetNLocals(rhs);
474         if (nLocals > -1 && (L.locals == null || L.locals.length < nLocals)) {
475           L.allocateLocals(nLocals);
476         }
477     
478         // evaluate the element-wise meet over the locals.
479         for (int i = 0; i < nLocals; i++) {
480           int[] R = new int[rhs.length];
481           for (int j = 0; j < rhs.length; j++) {
482             R[j] = ((MachineState) rhs[j]).getLocal(i);
483           }
484           int meet = meeter.meetLocal(i, R, bb);
485           if (L.locals[i] == TOP) {
486             if (meet != TOP) {
487               changed = true;
488               L.locals[i] = meet;
489             }
490           } else if (meet != L.locals[i]) {
491             changed = true;
492             L.locals[i] = meet;
493           }
494         }
495         return changed;
496       }

498       /**
499        * @return the number of locals to meet. Return -1 if there is no local meet necessary.
500        * @param operands The operands for this operator. operands[0] is the left-hand side.
501        */
502       private static int computeMeetNLocals(IVariable[] operands) {
503         MachineState lhs = (MachineState) operands[0];
504         int nLocals = -1;
505         if (lhs.locals != null) {
506           nLocals = lhs.locals.length;
507         } else {
508           for (int i = 1; i < operands.length; i++) {
509             MachineState rhs = (MachineState) operands[i];
510             if (rhs.locals != null) {
511               nLocals = rhs.locals.length;
512               break;
513             }
514           }
515         }
516         return nLocals;
517       }
518     
519       /**
520        * @return the height of stacks that are being meeted. Return -1 if there is no stack meet necessary.
521        * @param operands The operands for this operator. operands[0] is the left-hand side.
522        */
523       private static int computeMeetStackHeight(IVariable[] operands) {
524         MachineState lhs = (MachineState) operands[0];
525         int height = -1;
526         if (lhs.stack != null) {
527           height = lhs.stackHeight;
528         } else {
529           for (int i = 1; i < operands.length; i++) {
530             MachineState rhs = (MachineState) operands[i];
531             if (rhs.stack != null) {
532               height = rhs.stackHeight;
533               break;
534             }
535           }
536         }
537         return height;
538       }

540       /**
541        * Representation of the state of the JVM stack machine at some program point.
542        */
543       public class MachineState extends AbstractVariable<MachineState> {
544         private int[] stack;
545     
546         private int[] locals;
547     
548         // NOTE: stackHeight == -1 is a special code meaning "this variable is TOP"
549         private int stackHeight;
550     
551         private final BasicBlock bb;
552     
553         /**
554          * I'm not using clone because I don't want to necessarily inherit the AbstractVariable state from the superclass
555          */
556         public MachineState duplicate() {
557           MachineState result = new MachineState(bb);
558           result.copyState(this);
559           return result;
560         }
561     
562         public MachineState(BasicBlock bb) {
563           setTOP();
564           this.bb = bb;
565         }
566     
567         public BasicBlock getBasicBlock() {
568           return bb;
569         }
570     
571         void setTOP() {
572           stackHeight = -1;
573           stack = null;
574         }
575     
576         boolean isTOP() {
577           return stackHeight == -1;
578         }
579     
580         public void push(int i) {
581           if (stack == null || stackHeight >= stack.length)
582             allocateStack(stackHeight+1);
583           stack[stackHeight++] = i;
584         }
585     
586         public int pop() {
587           if (stackHeight <= 0) {
588             assert stackHeight > 0 : "can't pop stack of height " + stackHeight;
589           }
590           stackHeight -= 1;
591           return stack[stackHeight];
592         }
593     
594         public int peek() {
595           return stack[stackHeight - 1];
596         }
597     
598         public void swap() {
599           int temp = stack[stackHeight - 1];
600           stack[stackHeight - 1] = stack[stackHeight - 2];
601           stack[stackHeight - 2] = temp;
602         }
603     
604         private void allocateStack(int stackHeight) {
605           if (stack == null) {
606             stack = new int[stackHeight + 1 ];
607             this.stackHeight = 0;
608           } else {
609             int[] newStack = new int[ Math.max(stack.length, stackHeight) * 2 + 1 ];
610             System.arraycopy(stack, 0, newStack, 0, stack.length);
611             stack = newStack;
612           }
613         }
614     
615         private void allocateLocals(int maxLocals) {
616           int[] result = new int[maxLocals];
617           int start = 0;
618           if (locals != null) {
619             System.arraycopy(locals, 0, result, 0, locals.length);
620             start = locals.length;
621           } 
622           
623           for (int i = start; i < maxLocals; i++) {
624             result[i] = OPTIMISTIC ? TOP : BOTTOM;
625           }
626           
627           locals = result;
628         }
629     
630         public void clearStack() {
631           stackHeight = 0;
632         }
633     
634         /**
635          * set the value of local i to symbol j
636          * 
637          * @param i
638          * @param j
639          */
640         public void setLocal(int i, int j) {
641           if (locals == null || locals.length < i+1) {
642             if (OPTIMISTIC && (j == TOP)) {
643               return;
644             } else {
645               allocateLocals(i+1);
646             }
647           }
648           locals[i] = j;
649         }
650     
651         /**
652          * @param i
653          * @return the number of the symbol corresponding to local i
654          */
655         public int getLocal(int i) {
656           if (locals == null || locals.length < i+1) {
657             if (OPTIMISTIC) {
658               return TOP;
659             } else {
660               return BOTTOM;
661             }
662           } else {
663             return locals[i];
664           }
665         }
666     
667         public void replaceValue(int from, int to) {
668           if (stack != null)
669             for (int i = 0; i < stackHeight; i++)
670               if (stack[i] == from)
671                 stack[i] = to;
672     
673           if (locals != null)
674             for (int i = 0; i < locals.length; i++)
675               if (locals[i] == from)
676                 locals[i] = to;
677         }
678     
679         public boolean hasValue(int val) {
680           if (stack != null)
681             for (int i = 0; i < stackHeight; i++)
682               if (stack[i] == val)
683                 return true;
684     
685           if (locals != null)
686             for (int i = 0; i < locals.length; i++)
687               if (locals[i] == val)
688                 return true;
689     
690           return false;
691         }
692     
693         @Override
694         public String toString() {
695           if (isTOP()) {
696             return "<TOP>@" + System.identityHashCode(this);
697           }
698           StringBuffer result = new StringBuffer("<");
699           result.append("S");
700           if (stackHeight == 0) {
701             result.append("[empty]");
702           } else {
703             result.append(array2StringBuffer(stack, stackHeight));
704           }
705           result.append("L");
706           result.append(array2StringBuffer(locals, locals.length));
707           result.append(">");
708           return result.toString();
709         }
710     
711         private StringBuffer array2StringBuffer(int[] array, int n) {
712           StringBuffer result = new StringBuffer("[");
713           if (array == null) {
714             result.append(OPTIMISTIC ? "TOP" : "BOTTOM");
715           } else {
716             for (int i = 0; i < n - 1; i++) {
717               result.append(array[i]).append(",");
718             }
719             result.append(array[n - 1]);
720           }
721           result.append("]");
722           return result;
723         }

725         @Override
726         public void copyState(MachineState other) {
727           if (other.stack == null) {
728             stack = null;
729           } else {
730             stack = new int[other.stack.length];
731             System.arraycopy(other.stack, 0, stack, 0, other.stack.length);
732           }
733           if (other.locals == null) {
734             locals = null;
735           } else {
736             locals = new int[other.locals.length];
737             System.arraycopy(other.locals, 0, locals, 0, other.locals.length);
738           }
739           stackHeight = other.stackHeight;
740         }
741     
742         boolean stateEquals(MachineState exit) {
743           if (stackHeight != exit.stackHeight)
744             return false;
745           if (locals == null) {
746             if (exit.locals != null)
747               return false;
748           } else {
749             if (exit.locals == null)
750               return false;
751             else if (locals.length != exit.locals.length)
752               return false;
753           }
754     
755           for (int i = 0; i < stackHeight; i++) {
756             if (stack[i] != exit.stack[i])
757               return false;
758           }
759           if (locals != null) {
760             for (int i = 0; i < locals.length; i++) {
761               if (locals[i] == TOP) {
762                 if (exit.locals[i] != TOP)
763                   return false;
764               }
765               if (locals[i] != exit.locals[i])
766                 return false;
767             }
768           }
769           return true;
770         }
771     
772         /**
773          * Returns the stackHeight.
774          * 
775          * @return int
776          */
777         public int getStackHeight() {
778           return stackHeight;
779         }
780     
781         /**
782          * Use with care.
783          */
784         public int[] getLocals() {
785           return locals;
786         }
787     
788       }

810       /**
811        * This gives some basic facilities for shoving things around on the stack. Client analyses should subclass this as needed.
812        */
813       protected static abstract class BasicStackFlowProvider implements FlowProvider, Constants {
814         private final ShrikeCFG cfg;
815     
816         protected MachineState workingState;
817     
818         private BasicStackMachineVisitor visitor;
819     
820         private com.ibm.wala.shrikeBT.IInstruction.Visitor edgeVisitor;
821     
822         private int currentInstructionIndex = 0;
823     
824         private BasicBlock currentBlock;
825     
826         private BasicBlock currentSuccessorBlock;
827     
828         /**
829          * Only subclasses can instantiate
830          */
831         protected BasicStackFlowProvider(ShrikeCFG cfg) {
832           this.cfg = cfg;
833         }
834     
835         /**
836          * Initialize the visitors used to perform the flow functions
837          */
838         protected void init(BasicStackMachineVisitor v, com.ibm.wala.shrikeBT.IInstruction.Visitor ev) {
839           this.visitor = v;
840           this.edgeVisitor = ev;
841         }
842     
843         @Override
844         public boolean needsNodeFlow() {
845           return true;
846         }
847     
848         @Override
849         public boolean needsEdgeFlow() {
850           return false;
851         }

853         @Override
854         public MachineState flow(MachineState entry, BasicBlock basicBlock) {
855           workingState = entry.duplicate();
856           currentBlock = basicBlock;
857           currentSuccessorBlock = null;
858           IInstruction[] instructions = getInstructions();
859           if (DEBUG) {
860             System.err.println(("Entry to BB" + cfg.getNumber(basicBlock) + " " + workingState));
861           }
862           for (int i = basicBlock.getFirstInstructionIndex(); i <= basicBlock.getLastInstructionIndex(); i++) {
863             currentInstructionIndex = i;
864             instructions[i].visit(visitor);
865     
866             if (DEBUG) {
867               System.err.println(("After " + instructions[i] + " " + workingState));
868             }
869           }
870           return workingState;
871         }
872     
873         @Override
874         public MachineState flow(MachineState entry, BasicBlock from, BasicBlock to) {
875           workingState = entry.duplicate();
876           currentBlock = from;
877           currentSuccessorBlock = to;
878           IInstruction[] instructions = getInstructions();
879           if (DEBUG) {
880             System.err.println(("Entry to BB" + cfg.getNumber(from) + " " + workingState));
881           }
882           for (int i = from.getFirstInstructionIndex(); i <= from.getLastInstructionIndex(); i++) {
883             currentInstructionIndex = i;
884             instructions[i].visit(edgeVisitor);
885             if (DEBUG) {
886               System.err.println(("After " + instructions[i] + " " + workingState));
887             }
888           }
889           return workingState;
890         }

893           return currentInstructionIndex;
894         }
895

897           return cfg.getProgramCounter(currentInstructionIndex);
898         }
899     
900         protected BasicBlock getCurrentBlock() {
901           return currentBlock;
902         }
903     
904         protected BasicBlock getCurrentSuccessor() {
905           return currentSuccessorBlock;
906         }

994     
995             int size = instruction.getSize();
996             int delta = instruction.getDelta();
997             assert size == 1 || size == 2;
998             assert delta == 0 || delta == 1 || delta == 2;
999             int toPop = size + delta;
1000             int v1 = workingState.pop();
1001             int v2 = (toPop > 1) ? workingState.pop() : IGNORE;
1002             int v3 = (toPop > 2) ? workingState.pop() : IGNORE;
1003             int v4 = (toPop > 3) ? workingState.pop() : IGNORE;
1004     
1005             if (size > 1) {
1006               workingState.push(v2);
1007             }
1008             workingState.push(v1);
1009             if (v4 != IGNORE) {
1010               workingState.push(v4);
1011             }
1012             if (v3 != IGNORE) {
1013               workingState.push(v3);
1014             }
1015             if (v2 != IGNORE) {
1016               workingState.push(v2);
1017             }
1018             workingState.push(v1);
1019     
1020           }
