11     package com.ibm.wala.ipa.slicer;
12     
13     import com.ibm.wala.ipa.callgraph.CGNode;
14     import com.ibm.wala.ssa.SSAPhiInstruction;
15     
16     /**
17      * identifier of a phi instruction
18      */
19     public class PhiStatement extends Statement {
20       private final SSAPhiInstruction phi;
21     
22       public PhiStatement(CGNode node, SSAPhiInstruction phi) {
23         super(node);
24         this.phi = phi;
25       }
26     
27       @Override
28       public Kind getKind() {
29         return Kind.PHI;
30       }
31       
32       @Override
33       public boolean equals(Object obj) {
34         if (obj == null) {
35           return false;
36         }
37         if (getClass().equals(obj.getClass())) {
38           PhiStatement other = (PhiStatement) obj;
39           return getNode().equals(other.getNode()) && phi.getDef() == other.phi.getDef();
40         } else {
41           return false;
42         }
43       }
44     
45       @Override
46       public int hashCode() {
47         return 3691 * phi.getDef() + getNode().hashCode();
48       }
49     
50       @Override
51       public String toString() {
52         return "PHI " + getNode() + ":" + phi;
53       }
54     
55       public SSAPhiInstruction getPhi() {
56         return phi;
57       }
58     
59     }
