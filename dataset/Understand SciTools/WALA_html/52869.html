295         @Override
296         public CAstNode visit(ConstructorDecl cd, MethodContext mc) {
297           // Needs to examine the initializers in the ClassContext
298           // and glue that code into the right place relative to the
299           // constructor method body ("wherever that may turn out to be").
300           List/* <FieldDecl|Initializer> */inits = mc.getInitializers();
301           Block body = cd.body();
302     
303           if (hasSuperCall(body)) {
304             // Split at call to super:
305             // super();
306             // field initializer code
307             // remainder of ctor body
308             CAstNode[] bodyNodes = new CAstNode[inits.size() + body.statements().size()];
309     
310             int idx = 0;
311             for (Iterator iter = body.statements().iterator(); iter.hasNext();) {
312               Stmt s = (Stmt) iter.next();
313     
314               bodyNodes[idx++] = walkNodes(s, mc);
315               if (idx == 1) {
316                 Assertions.productionAssertion(isSpecialCallStmt(s, ConstructorCall.SUPER));
317                 idx = insertInitializers(mc, bodyNodes, false, idx);
318               }
319             }
320             return makeNode(mc, fFactory, body, CAstNode.BLOCK_STMT, bodyNodes);
321           } else if (hasThisCall(body)) {
322             return walkNodes(body, mc);
323           } else {
324             // add explicit call to default super()
325             // RMF 4/17/2009- The following search for a superClass default ctor
326             // won't work if we process Object in source. In particular, the
327             // superClass might be null. In that case, simply omit the explicit
328             // super() call.
329             ClassType superClass = (ClassType) ((ObjectType) cd.constructorDef().asInstance().container()).superClass();
330             CAstNode[] bodyNodes;
331             int idx= 0;
332             int bodyInitsSize= inits.size() + body.statements().size();
333             if (superClass != null) {
334               ProcedureInstance defaultSuperCtor = findDefaultCtor(superClass);
335               CallSiteReference callSiteRef = CallSiteReference.make(0, fIdentityMapper.getMethodRef(defaultSuperCtor),
336                   IInvokeInstruction.Dispatch.SPECIAL);
337               CAstNode superCall = makeNode(mc, fFactory, cd, CAstNode.CALL,
338                   makeNode(mc, fFactory, cd, CAstNode.SUPER),
339                   fFactory.makeConstant(callSiteRef));
340               bodyNodes = new CAstNode[bodyInitsSize + 1];
341               bodyNodes[idx++] = superCall;
342             } else { // no super class, so no super call
343               bodyNodes = new CAstNode[bodyInitsSize];
344             }
345             idx= insertInitializers(mc, bodyNodes, false, idx);
346     
347             for (Iterator iter = body.statements().iterator(); iter.hasNext(); idx++) {
348               Stmt s = (Stmt) iter.next();
349               bodyNodes[idx] = walkNodes(s, mc);
350             }
351             return makeNode(mc, fFactory, body, CAstNode.BLOCK_STMT, bodyNodes);
352           }
353         }

355         private ProcedureInstance findDefaultCtor(ClassType superClass) {
356           List/* <ProcedureInstance> */ctors = superClass.constructors();
357           for (Iterator iter = ctors.iterator(); iter.hasNext();) {
358             ConstructorInstance ctor = (ConstructorInstance) iter.next();
359             if (ctor.formalTypes().isEmpty())
360               return ctor;
361           }
362           Assertions.UNREACHABLE("Couldn't find default ctor");
363           return null;
364         }

386             rhsNode = walkNodes(init, ctorContext);
387           }
388           CAstNode assNode = makeNode(ctorContext, fFactory, f, CAstNode.ASSIGN, lhsNode, rhsNode);
389     
390           return assNode;

491         protected CAstNode processAssign(Node assign, CAstNode lhsCAstNode, Assign.Operator operator, Expr rhs, WalkContext wc) {
492           if (operator == Assign.ASSIGN) {
493             return makeNode(wc, fFactory, assign, CAstNode.ASSIGN, lhsCAstNode, walkNodes(rhs, wc));
494           } else {
495             CAstNode op =
496               makeNode(wc, 
497                   fFactory, 
498                   assign, 
499                   CAstNode.ASSIGN_PRE_OP,
500                   lhsCAstNode,
501                   walkNodes(rhs, wc),
502                   mapAssignOperator(operator));
503     
504             if (rhs.type().isLongOrLess() &&
505                 (mapAssignOperator(operator)==CAstOperator.OP_DIV ||
506                  mapAssignOperator(operator)==CAstOperator.OP_MOD))
507             {
508               Collection excTargets = wc.getCatchTargets(fDivByZeroType);
509               if (!excTargets.isEmpty()) {
510                 for (Iterator iterator = excTargets.iterator(); iterator.hasNext();) {
511                   Pair catchPair = (Pair) iterator.next();
512                   wc.cfg().add(op, catchPair.snd, fDivByZeroType);
513                 }
514               } else {
515                 wc.cfg().add(op, 
516                     CAstControlFlowMap.EXCEPTION_TO_EXIT, 
517                     fDivByZeroType);
518               }
519             }
520     
521             return op;
522           }
523         }

532         protected CAstOperator mapAssignOperator(Assign.Operator op) {
533           if (op == Assign.ADD_ASSIGN)
534             return CAstOperator.OP_ADD;
535           else if (op == Assign.BIT_AND_ASSIGN)
536             return CAstOperator.OP_BIT_AND;
537           else if (op == Assign.BIT_OR_ASSIGN)
538             return CAstOperator.OP_BIT_OR;
539           else if (op == Assign.BIT_XOR_ASSIGN)
540             return CAstOperator.OP_BIT_XOR;
541           else if (op == Assign.DIV_ASSIGN)
542             return CAstOperator.OP_DIV;
543           else if (op == Assign.MOD_ASSIGN)
544             return CAstOperator.OP_MOD;
545           else if (op == Assign.MUL_ASSIGN)
546             return CAstOperator.OP_MUL;
547           else if (op == Assign.SHL_ASSIGN)
548             return CAstOperator.OP_LSH;
549           else if (op == Assign.SHR_ASSIGN)
550             return CAstOperator.OP_RSH;
551           else if (op == Assign.SUB_ASSIGN)
552             return CAstOperator.OP_SUB;
553           else if (op == Assign.USHR_ASSIGN)
554             return CAstOperator.OP_URSH;
555           Assertions.UNREACHABLE("Unknown assignment operator");
556           return null;
557         }

559         @Override
560         public CAstNode visit(Binary b, WalkContext wc) {
561           Expr left = b.left();
562           Expr right = b.right();
563           Operator operator = b.operator();
564     
565           if (operator.equals(Binary.COND_AND))
566             return makeNode(wc, fFactory, b, CAstNode.IF_EXPR, walkNodes(left, wc), walkNodes(right, wc), fFactory.makeConstant(false));
567           else if (operator.equals(Binary.COND_OR))
568             return makeNode(wc, fFactory, b, CAstNode.IF_EXPR, walkNodes(left, wc), fFactory.makeConstant(true), walkNodes(right, wc));
569           else {
570     	Type leftType = left.type();
571     	Type rightType = right.type();
572     	if (leftType.isPrimitive() && rightType.isPrimitive()) {
573     	  CAstNode leftNode = walkNodes(left, wc);
574     	  CAstNode rightNode = walkNodes(right, wc);
575     	  
576     	  try {

579     	    if (! result.typeEquals(leftType, fTypeSystem.emptyContext())) {

583     	    if (! result.typeEquals(rightType, fTypeSystem.emptyContext())) {
584     	      rightNode = makeNode(wc, fFactory, b, CAstNode.CAST, fFactory.makeConstant(getTypeDict().getCAstTypeFor(result)), rightNode, fFactory.makeConstant(getTypeDict().getCAstTypeFor(rightType)));
585     	    }
586     	  } catch (SemanticException e) {
587     
588     	  }
589     
590     	  CAstNode op = makeNode(wc, fFactory, b, CAstNode.BINARY_EXPR, mapBinaryOpcode(operator), leftNode, rightNode);
591     
592     	  if (leftType.isLongOrLess() &&
593     	      rightType.isLongOrLess() &&
594     	      (mapBinaryOpcode(operator)==CAstOperator.OP_DIV ||
595     	       mapBinaryOpcode(operator)==CAstOperator.OP_MOD))
596     	  {
597     	    Collection excTargets = wc.getCatchTargets(fDivByZeroType);
598     	    if (!excTargets.isEmpty()) {
599     	      for (Iterator iterator = excTargets.iterator(); 
600     		   iterator.hasNext();) 
601     	      {
602     		Pair catchPair = (Pair) iterator.next();
603     		wc.cfg().add(op, catchPair.snd, fDivByZeroType);
604     	      }
605     	    } else {
606     	      wc.cfg().add(op, 
607     			   CAstControlFlowMap.EXCEPTION_TO_EXIT, 
608     			   fDivByZeroType);
609     	    }
610     	  }
611     
612     	  return op;
613     	  
614     	} else {
615     	  return makeNode(wc, fFactory, b, CAstNode.BINARY_EXPR, mapBinaryOpcode(operator), walkNodes(left, wc), walkNodes(right, wc));
616     	}
617           }

637           MethodInstance methodInstance = c.methodInstance();
638           boolean isStatic = methodInstance.flags().isStatic();
639           StructType methodOwner = methodInstance.container();
640     
641           if (methodOwner.isArray()) {
642             List realOne = methodInstance.overrides(fTypeSystem.emptyContext());
643             Assertions.productionAssertion(realOne.size() == 2, "bad array method");
644             methodInstance = (MethodInstance) realOne.get(1);
645             methodOwner = methodInstance.container();
646           }
647     
648           if (!methodOwner.isClass()) {
649             Assertions.productionAssertion(false, "owner " + methodOwner + " of " + methodInstance + " is not a class");
650           }
651     
652           boolean isIntf = ((ClassType) methodOwner).flags().isInterface();
653           Receiver target = c.target();
654           boolean isSpecial = methodInstance.flags().isPrivate()
655               || (target instanceof Special && ((Special) target).kind() == Special.SUPER);

974           Receiver target = f.target();
975           Type targetType = target.type();
976     
977           if (targetType.isArray()) {

989             // evaluation

993             if (fi.isConstant()) {

997               return makeNode(wc, fFactory, f, CAstNode.BLOCK_EXPR, targetNode, makeNode(wc, fFactory, f, CAstNode.OBJECT_REF,
998                   makeNode(wc, fFactory, null, CAstNode.VOID), fFactory.makeConstant(fieldRef)));

1000           } else {
1001             Collection excTargets = wc.getCatchTargets(fNPEType);

1003               // connect NPE exception edge to relevant catch targets
1004               // (presumably only one)
1005               for (Iterator iterator = excTargets.iterator(); iterator.hasNext();) {
1006                 Pair catchPair = (Pair) iterator.next();
1007                 wc.cfg().add(f, catchPair.snd, fNPEType);
1008               }
1009             } else {
1010               // connect exception edge to exit
1011               wc.cfg().add(f, CAstControlFlowMap.EXCEPTION_TO_EXIT, fNPEType);
1012             }

1022             }

2688       protected void processClassMembers(Node n, ClassType classType, List<ClassMember> members, DelegatingContext classContext,
2689           List<CAstEntity> memberEntities) {
2690         // Collect all initializer-related gorp
2691         for (Iterator<ClassMember> memberIter = members.iterator(); memberIter.hasNext();) {
2692           ClassMember member = memberIter.next();
2693     
2694           if (member instanceof Initializer) {
2695             Initializer initializer = (Initializer) member;
2696     
2697             if (initializer.flags().flags().isStatic())
2698               classContext.getStaticInitializers().add(initializer);
2699             else
2700               classContext.getInitializers().add(initializer);
2701           } else if (member instanceof FieldDecl) {
2702             FieldDecl fd = (FieldDecl) member;
2703     
2704             if (fd.init() != null) {
2705               if (fd.flags().flags().isStatic())
2706                 classContext.getStaticInitializers().add(fd);
2707               else
2708                 classContext.getInitializers().add(fd);
2709             }
2710           }
2711         }
2712     
2713         // Now process
2714         for (Iterator<ClassMember> memberIter = members.iterator(); memberIter.hasNext();) {
2715           ClassMember member = memberIter.next();
2716     
2717           if (!(member instanceof Initializer)) {
2718             CAstEntity memberEntity = walkEntity(member, classContext);
2719     
2720             memberEntities.add(memberEntity);
2721           }
2722         }
2723     
2724         // add class initializer, if needed
2725         if (!classContext.getStaticInitializers().isEmpty()) {
2726             InitializerDef initDef = fTypeSystem.initializerDef(n.position(), Types.ref(classType), Flags.STATIC);
2727           InitializerInstance initInstance = fTypeSystem.createInitializerInstance(n.position(), Types.ref(initDef));
2728     
2729           Map<CAstNode, Collection<CAstEntity>> childEntities = HashMapFactory.make();
2730           final MethodContext mc = new MethodContext(initInstance, childEntities, classContext);
2731     
2732           List inits = classContext.getStaticInitializers();
2733           CAstNode[] bodyNodes = new CAstNode[inits.size()];
2734           insertInitializers(mc, bodyNodes, true, 0);
2735     
2736           CAstNode ast = makeNode(mc, fFactory, n, CAstNode.BLOCK_STMT, bodyNodes);
2737     
2738           memberEntities.add(new ProcedureEntity(ast, fTypeSystem, initInstance, new String[0], childEntities, mc));
2739     
2740         }
2741       }

2743       protected void addConstructorsToAnonymousClass(New n, ClassType anonType, ClassContext classContext,
2744           List<CAstEntity> memberEntities) {
2745         List superConstructors = ((ClassType) anonType.superClass()).constructors();
2746     
2747         for (Iterator iter = superConstructors.iterator(); iter.hasNext();) {
2748           ConstructorInstance superCtor = (ConstructorInstance) iter.next();
2749     
2750           Map<CAstNode, Collection<CAstEntity>> childEntities = HashMapFactory.make();
2751           final MethodContext mc = new MethodContext(superCtor, childEntities, classContext);
2752     
2753           String[] fakeArguments = new String[superCtor.formalTypes().size() + 1];
2754           for (int i = 0; i < fakeArguments.length; i++) {
2755             fakeArguments[i] = (i == 0) ? "this" : ("argument" + i);
2756           }
2757     
2758           List inits = classContext.getInitializers();
2759     
2760           CAstNode[] bodyNodes = new CAstNode[inits.size() + 1];
2761     
2762           CallSiteReference callSiteRef = CallSiteReference.make(0, fIdentityMapper.getMethodRef(superCtor),
2763               IInvokeInstruction.Dispatch.SPECIAL);
2764           CAstNode[] children = new CAstNode[fakeArguments.length + 1];
2765           children[0] = makeNode(mc, fFactory, n, CAstNode.SUPER);
2766           children[1] = fFactory.makeConstant(callSiteRef);
2767           for (int i = 1; i < fakeArguments.length; i++) {
2768             children[i + 1] = makeNode(mc, fFactory, n, CAstNode.VAR, fFactory.makeConstant(fakeArguments[i]));
2769           }
2770           bodyNodes[0] = makeNode(mc, fFactory, n, CAstNode.CALL, children);
2771     
2772           insertInitializers(mc, bodyNodes, false, 1);
2773     
2774           CAstNode ast = makeNode(mc, fFactory, n, CAstNode.BLOCK_STMT, bodyNodes);
2775     
2776           memberEntities.add(new ProcedureEntity(ast, fTypeSystem, superCtor, anonType, fakeArguments, childEntities, mc));
2777         }
2778       }

2786       protected CAstEntity walkEntity(Node rootNode, final WalkContext context) {
2787         if (rootNode instanceof SourceFile) {
2788           SourceFile file = (SourceFile) rootNode;
2789           List<CAstEntity> declEntities = new ArrayList<CAstEntity>();
2790     
2791           for (Iterator iter = file.decls().iterator(); iter.hasNext();) {
2792             TopLevelDecl decl = (TopLevelDecl) iter.next();
2793     
2794             declEntities.add(walkEntity(decl, context));
2795           }
2796           return new CompilationUnitEntity(file, declEntities);
2797         } else if (rootNode instanceof ClassDecl) {
2798           final ClassDecl cd = (ClassDecl) rootNode;
2799           final List<CAstEntity> memberEntities = new ArrayList<CAstEntity>();
2800           final ClassContext classContext = new ClassContext(cd.classDef().asType(), memberEntities, context);
2801     
2802           processClassMembers(rootNode, cd.classDef().asType(), cd.body().members(), classContext, memberEntities);
2803     
2804           return new ClassEntity(classContext, memberEntities, cd, cd.position());
2805         } else if (rootNode instanceof New) {
2806           final New n = (New) rootNode;
2807           final List<CAstEntity> memberEntities = new ArrayList<CAstEntity>();
2808           ClassType anonType = n.anonType().asType();
2809           String anonTypeName = anonTypeName(anonType);
2810           final ClassContext classContext = new ClassContext(anonType, memberEntities, context);
2811     
2812           processClassMembers(rootNode, anonType, n.body().members(), classContext, memberEntities);
2813           addConstructorsToAnonymousClass(n, anonType, classContext, memberEntities);
2814     
2815           return new ClassEntity(classContext, memberEntities, anonType, anonTypeName, n.position());
2816         } else if (rootNode instanceof ProcedureDecl) {
2817           final ProcedureDecl pd = (ProcedureDecl) rootNode;
2818           final Map<CAstNode, Collection<CAstEntity>> memberEntities = HashMapFactory.make();
2819           final MethodContext mc = new MethodContext(pd.procedureInstance().asInstance(), memberEntities, context);
2820     
2821           CAstNode pdAST = null;
2822     
2823           if (!pd.flags().flags().isAbstract()) {
2824             // Presumably the MethodContext's parent is a ClassContext,
2825             // and he has the list of initializers. Hopefully the following
2826             // will glue that stuff in the right place in any constructor body.
2827             pdAST = walkNodes(pd, mc);
2828           }
2829     
2830           List/* <Formal> */formals = pd.formals();
2831           String[] argNames;
2832           int i = 0;
2833           if (!pd.flags().flags().isStatic()) {
2834             argNames = new String[formals.size() + 1];
2835             argNames[i++] = "this";
2836           } else {
2837             argNames = new String[formals.size()];
2838           }
2839           for (Iterator iter = formals.iterator(); iter.hasNext(); i++) {
2840             Formal formal = (Formal) iter.next();
2841             argNames[i] = formal.name().toString();
2842           }
2843     
2844           return new ProcedureEntity(pdAST, fTypeSystem, pd.procedureInstance().asInstance(), argNames, memberEntities, mc);
2845         } else if (rootNode instanceof FieldDecl) {
2846           final FieldDecl fd = (FieldDecl) rootNode;
2847     
2848           return new FieldEntity(fd, context);
2849         } else {
2850           Assertions.UNREACHABLE("Unknown node type for walkEntity():" + rootNode.getClass().getName());
2851           return null;
2852         }
2853       }
