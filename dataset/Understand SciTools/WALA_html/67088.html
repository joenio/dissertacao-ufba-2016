27     public class PolyglotSyncDuplicatorTest extends SyncDuplicatorTest {
28     
29       @Override
30       protected JavaSourceAnalysisEngine getAnalysisEngine(final String[] mainClassDescriptors, Collection<String> sources, List<String> libs) {
31         JavaSourceAnalysisEngine engine = new PolyglotJavaSourceAnalysisEngine() {
32           @Override
33           protected Iterable<Entrypoint> makeDefaultEntrypoints(AnalysisScope scope, IClassHierarchy cha) {
34             return Util.makeMainEntrypoints(JavaSourceAnalysisScope.SOURCE, cha, mainClassDescriptors);
35           }
36     
37           @Override
38           public IRTranslatorExtension getTranslatorExtension() {
39             JavaIRTranslatorExtension ext = new JavaIRTranslatorExtension();
40             ext.setCAstRewriterFactory(new CAstRewriterFactory() {
41               @Override
42               public CAstRewriter createCAstRewriter(CAst ast) {
43                 return new SynchronizedBlockDuplicator(ast, true, testMethod);
44               }
45             });
46             return ext;
47           }
48     
49         };
50         engine.setExclusionsFile(CallGraphTestUtil.REGRESSION_EXCLUSIONS);
51         populateScope(engine, sources, libs);
52         return engine;
53       }

55     }
