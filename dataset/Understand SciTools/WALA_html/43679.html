12     package com.ibm.wala.cfg;
13     
14     import com.ibm.wala.classLoader.IMethod;
15     import com.ibm.wala.util.graph.INodeWithNumber;
16     
17     /**
18      * An interface for a basic block in a control flow graph.
19      */
20     public interface IBasicBlock<InstType> extends INodeWithNumber, Iterable<InstType> {
21     
22       /**
23        * Get the index of the first instruction in the basic block.  The value
24        * is an index into the instruction array that contains all the instructions
25        * for the method.
26        * 
27        * If the result is < 0, the block has no instructions
28        * 
29        * @return the instruction index for the first instruction in the basic block.
30        */
31       public int getFirstInstructionIndex();
32     
33       /**
34        * Get the index of the last instruction in the basic block.  The value
35        * is an index into the instruction array that contains all the instructions
36        * for the method.
37        * 
38        * If the result is < 0, the block has no instructions
39        * 
40        * @return the instruction index for the last instruction in the basic block
41        */
42       public int getLastInstructionIndex();
43       
44     
45       /**
46        * Return true if the basic block represents a catch block.
47        * @return true if the basic block represents a catch block.
48        */
49       public boolean isCatchBlock();
50       
51       /**
52        * Return true if the basic block represents the unique exit block.
53        * @return true if the basic block represents the unique exit block.
54        */
55       public boolean isExitBlock();
56       
57       /**
58        * Return true if the basic block represents the unique entry block.
59        * @return true if the basic block represents the unique entry block.
60        */
61       public boolean isEntryBlock();
62       
63       /**
64        * @return governing method for this block
65        */
66       public IMethod getMethod();
67     
68       /**
69        * Each basic block should have a unique number in its cfg
70        * @return the basic block's number
71        */
72       public int getNumber();
73     
74     }
