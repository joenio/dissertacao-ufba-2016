36       /**
37        * This method computes the serialVersionUID for class r (if there isn't one already) and adds the field to the classwriter w.
38        * 
39        * When run as a program, just takes a list of class files as command line arguments and computes their serialVersionUIDs.
40        * 
41        * @throws IllegalArgumentException if r is null
42        */
43       public static void addSerialVersionUID(ClassReader r, ClassWriter w) throws InvalidClassFileException {
44         if (r == null) {
45           throw new IllegalArgumentException("r is null");
46         }
47         int numFields = r.getFieldCount();
48         for (int i = 0; i < numFields; i++) {
49           if (r.getFieldName(i).equals("serialVersionUID")) {
50             return; // already has a serialVersionUID
51           }
52         }
53     
54         long UID = computeSerialVersionUID(r);
55         w.addField(ClassReader.ACC_PUBLIC | ClassReader.ACC_STATIC | ClassReader.ACC_FINAL, "serialVersionUID", "J",
56             new ClassWriter.Element[] { new ConstantValueWriter(w, UID) });
57       }

59       /**
60        * This class implements a stream that just discards everything written to it.
61        */
62       public static final class SinkOutputStream extends OutputStream {
63         @Override
64         public void write(int b) {
65         }
66     
67         @Override
68         public void write(byte[] b) {
69         }
70     
71         @Override
72         public void write(byte[] b, int off, int len) {
73         }
74       }

76       /**
77        * This method computes the serialVersionUID for class r. See the specification at
78        * http://java.sun.com/j2se/1.4.2/docs/guide/serialization/spec/class.html
79        * 
80        * @throws IllegalArgumentException if r is null
81        */
82       public static long computeSerialVersionUID(final ClassReader r) throws InvalidClassFileException {
83         if (r == null) {
84           throw new IllegalArgumentException("r is null");
85         }
86         MessageDigest digest;
87         try {
88           digest = MessageDigest.getInstance("SHA");
89         } catch (NoSuchAlgorithmException e) {
90           throw new Error("SHA algorithm not supported: " + e.getMessage());
91         }
92         SinkOutputStream sink = new SinkOutputStream();
93         DataOutputStream out = new DataOutputStream(new DigestOutputStream(sink, digest));
94     
95         try {
96           // step 1
97           out.writeUTF(r.getName());
98           // step 2
99           out.writeInt(r.getAccessFlags());
100     
101           // step 3
102           String[] interfaces = r.getInterfaceNames();
103           Arrays.sort(interfaces);
104           for (int i = 0; i < interfaces.length; i++) {
105             out.writeUTF(interfaces[i]);
106           }
107     
108           // step 4
109           Integer[] fields = new Integer[r.getFieldCount()];
110           final String[] fieldNames = new String[fields.length];
111           int fieldCount = 0;
112           for (int f = 0; f < fields.length; f++) {
113             int flags = r.getFieldAccessFlags(f);
114             if ((flags & ClassReader.ACC_PRIVATE) == 0 || (flags & (ClassReader.ACC_STATIC | ClassReader.ACC_TRANSIENT)) == 0) {
115               fields[fieldCount] = new Integer(f);
116               fieldNames[f] = r.getFieldName(f);
117               fieldCount++;
118             }
119           }
120           Arrays.sort(fields, 0, fieldCount, new Comparator<Integer>() {
121             @Override
122             public int compare(Integer o1, Integer o2) {
123               String name1 = fieldNames[o1.intValue()];
124               String name2 = fieldNames[o2.intValue()];
125               return name1.compareTo(name2);
126             }
127           });
128           for (int i = 0; i < fieldCount; i++) {
129             int f = fields[i].intValue();
130             out.writeUTF(fieldNames[f]);
131             out.writeInt(r.getFieldAccessFlags(f));
132             out.writeUTF(r.getFieldType(f));
133           }
134     
135           // steps 5, 6 and 7
136           Integer[] methods = new Integer[r.getMethodCount()];
137           final int[] methodKinds = new int[methods.length];
138           final String[] methodSigs = new String[methods.length];
139           int methodCount = 0;
140           for (int m = 0; m < methodSigs.length; m++) {
141             String name = r.getMethodName(m);
142             int flags = r.getMethodAccessFlags(m);
143             if (name.equals("<clinit>") || (flags & ClassReader.ACC_PRIVATE) == 0) {
144               methods[methodCount] = new Integer(m);
145               methodSigs[m] = name + r.getMethodType(m);
146               if (name.equals("<clinit>")) {
147                 methodKinds[m] = 0;
148               } else if (name.equals("<init>")) {
149                 methodKinds[m] = 1;
150               } else {
151                 methodKinds[m] = 2;
152               }
153               methodCount++;
154             }
155           }
156           Arrays.sort(methods, 0, methodCount, new Comparator<Integer>() {
157             @Override
158             public int compare(Integer o1, Integer o2) {
159               int m1 = o1.intValue();
160               int m2 = o2.intValue();
161               if (methodKinds[m1] != methodKinds[m2]) {
162                 return methodKinds[m1] - methodKinds[m2];
163               }
164               String name1 = methodSigs[m1];
165               String name2 = methodSigs[m2];
166               return name1.compareTo(name2);
167             }
168           });
169           for (int i = 0; i < methodCount; i++) {
170             int m = methods[i].intValue();
171             out.writeUTF(r.getMethodName(m));
172             out.writeInt(r.getMethodAccessFlags(m));
173             out.writeUTF(r.getMethodType(m));
174           }
175         } catch (IOException e1) {
176           throw new Error("Unexpected IOException: " + e1.getMessage());
177         } finally {
178           try {
179             out.close();
180           } catch (IOException e2) {
181           }
182         }
183     
184         byte[] hash = digest.digest();
185         return (hash[0] & 0xFF) | (hash[1] & 0xFF) << 8 | (hash[2] & 0xFF) << 16 | hash[3] << 24 | (hash[4] & 0xFF) << 32
186             | (hash[5] & 0xFF) << 40 | (hash[6] & 0xFF) << 48 | (hash[7] & 0xFF) << 56;
187       }

189       public static void main(String[] args) {
190         for (int i = 0; i < args.length; i++) {
191           if (args[i] == null) {
192             throw new IllegalArgumentException("args[" + i + "] is null");
193           }
194           try {
195             byte[] data = Util.readFully(new FileInputStream(args[i]));
196             ClassReader r = new ClassReader(data);
197             System.out.println(Util.makeClass(r.getName()) + ": serialVersionUID = " + computeSerialVersionUID(r));
198           } catch (FileNotFoundException e) {
199             System.err.println("File not found: " + args[i]);
200           } catch (IOException e) {
201             System.err.println("Error reading file: " + args[i]);
202           } catch (InvalidClassFileException e) {
203             System.err.println("Invalid class file: " + args[i]);
204           }
205         }
206       }
