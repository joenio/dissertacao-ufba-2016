11     package com.ibm.wala.cast.ipa.callgraph;
12     
13     import com.ibm.wala.ipa.callgraph.propagation.AbstractFieldPointerKey;
14     import com.ibm.wala.ipa.callgraph.propagation.InstanceKey;
15     
16     public abstract class ReflectedFieldPointerKey extends AbstractFieldPointerKey {
17     
18       ReflectedFieldPointerKey(InstanceKey container) {
19         super(container);
20       }
21     
22       public abstract Object getFieldIdentifier();
23     
24       private static final Object arrayStateKey = new Object() {
25         @Override
26         public String toString() { return "ArrayStateKey"; }
27       };
28     
29       @Override
30       public boolean equals(Object obj) {
31         if (obj == this) {
32           return true;
33         }
34         if (obj instanceof ReflectedFieldPointerKey) {
35           ReflectedFieldPointerKey other = (ReflectedFieldPointerKey) obj;
36           return
37     	getFieldIdentifier().equals(other.getFieldIdentifier()) &&
38     	getInstanceKey().equals(other.getInstanceKey());
39         } else {
40           return false;
41         }
42       }
43     
44       @Override
45       public int hashCode() {
46         return getFieldIdentifier().hashCode() ^ getInstanceKey().hashCode();
47       }
48     
49       @Override
50       public String toString() { return "[" + getInstanceKey() + "; " + getFieldIdentifier() + "]"; }
51       
52       public static ReflectedFieldPointerKey literal(final String lit, InstanceKey instance) {
53         return new ReflectedFieldPointerKey(instance) {
54           @Override
55           public Object getFieldIdentifier() {
56     	return lit;
57           }
58         };   
59       }
60     
61       public static ReflectedFieldPointerKey mapped(final InstanceKey mapFrom, InstanceKey instance) {
62         return new ReflectedFieldPointerKey(instance) {
63           @Override
64           public Object getFieldIdentifier() {
65     	return mapFrom;
66           }
67         };   
68       }
69     
70       public static ReflectedFieldPointerKey index(InstanceKey instance) {
71         return new ReflectedFieldPointerKey(instance) {
72           @Override
73           public Object getFieldIdentifier() {
74     	return arrayStateKey;
75           }
76         };   
77       }
78     
79     }
