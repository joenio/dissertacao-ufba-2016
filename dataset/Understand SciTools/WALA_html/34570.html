11     package com.ibm.wala.shrikeBT;
12     
13     /**
14      * This class represents dup instructions. There are two kinds of dup instructions, dup and dup_x1:
15      * 
16      * dup: a::rest => a::a::rest dup_x1: a::b::rest => a::b::a::rest
17      */
18     public final class DupInstruction extends Instruction {
19       final private int size;
20     
21       final private byte delta;
22     
23       protected DupInstruction(byte size, byte delta) {
24         super((short) -1);
25         this.size = size;
26         this.delta = delta;
27       }
28     
29       private final static DupInstruction[] preallocated = preallocate();
30     
31       private static DupInstruction[] preallocate() {
32         DupInstruction[] r = new DupInstruction[9];
33     
34         for (int i = 0; i < r.length; i++) {
35           r[i] = new DupInstruction((byte) (i / 3), (byte) (i % 3));
36         }
37         return r;
38       }
39     
40       /**
41        * DupInstructions with size or delta 2 might cause code generation failures when the working stack contains long or double
42        * values, when these DupInstructions cannot be easily translated into Java bytecode instructions. For safety, avoid using
43        * DupInstructions with size or delta 2.
44        * 
45        * Don't create these outside the shrikeBT decoder.
46        */
47       static DupInstruction make(int size, int delta) {
48         if (size < 0 || size > 2) {
49           throw new IllegalArgumentException("Invalid dup size: " + size);
50         }
51         if (delta < 0 || delta > 2) {
52           throw new IllegalArgumentException("Invalid dup delta: " + delta);
53         }
54         return preallocated[size * 3 + delta];
55       }
56     
57       /**
58        * @param delta 0 for dup, 1 for dup_x1
59        */
60       public static DupInstruction make(int delta) {
61         if (delta < 0 || delta > 1) {
62           throw new IllegalArgumentException("Invalid dup delta: " + delta);
63         }
64         return make(1, delta);
65       }
66     
67       @Override
68       public boolean equals(Object o) {
69         if (o instanceof DupInstruction) {
70           DupInstruction i = (DupInstruction) o;
71           return i.size == size && i.delta == delta;
72         } else {
73           return false;
74         }
75       }
76     
77       public int getSize() {
78         return size;
79       }
80     
81       public int getDelta() {
82         return delta;
83       }
84     
85       @Override
86       public int hashCode() {
87         return size + 8431890 + 10 * delta;
88       }
89     
90       @Override
91       public int getPoppedCount() {
92         return size + delta;
93       }
94     
95       @Override
96       public String toString() {
97         return "Dup(" + size + "," + delta + ")";
98       }
99     
100       @Override
101       public void visit(IInstruction.Visitor v) {
102         if (v == null) {
103           throw new IllegalArgumentException("illegal null visitor");
104         }
105         v.visitDup(this);
106       }
107     
108       @Override
109       public boolean isPEI() {
110         return false;
111       }
112     }
