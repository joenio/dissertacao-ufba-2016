31     /**
32      * A {@link ContextSelector} to intercept calls to reflective method invocations such as Constructor.newInstance and Method.invoke
33      */
34     class ReflectiveInvocationSelector implements ContextSelector {
35     
36       public ReflectiveInvocationSelector() {
37       }
38     
39       /**
40        * Creates a callee target based on the following criteria:
41        * <ol>
42        * <li>If the method being invoked through reflection is definitely static, then do not create a callee target for any
43        * <code>callee</code> method that is not static. In this case, return <code>null</code>.
44        * <li>If the method being invoked through reflection takes a constant number of parameters, <code>n</code>, then do not create a
45        * callee target for any <code>callee</code> method that takes a number of parameters different from <code>n</code>. In this case,
46        * return <code>null</code>.
47        * <li>Otherwise, return a new {@link ReceiverInstanceContext} for <code>receiver</code>.
48        * </ol>
49        */
50       @Override
51       public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] receiver) {
52         if (receiver == null || receiver.length == 0 || !mayUnderstand(caller, site, callee, receiver[0])) {
53           return null;
54         }
55         IR ir = caller.getIR();
56         SSAAbstractInvokeInstruction[] invokeInstructions = ir.getCalls(site);
57         if (invokeInstructions.length != 1) {
58           return new ReceiverInstanceContext(receiver[0]);
59         }
60         SymbolTable st = ir.getSymbolTable();
61         ConstantKey receiverConstantKey = (ConstantKey) receiver[0];
62         IMethod m = (IMethod) receiverConstantKey.getValue();
63         boolean isStatic = m.isStatic();
64         boolean isConstructor = isConstructorConstant(receiver[0]);
65     
66         // If the method being invoked through reflection is not a constructor and is definitely static, then
67         // we should not create a callee target for any method that is not static
68         if (!isConstructor) {
69           int recvUse = invokeInstructions[0].getUse(1);
70           if (st.isNullConstant(recvUse) && !isStatic) {
71             return null;
72           }
73         }
74     
75         // If the method being invoked through reflection is being passed n parameters,
76         // then we should not create a callee target for any method that takes a number
77         // of parameters different from n
78         int numberOfParams = isStatic ? m.getNumberOfParameters() : m.getNumberOfParameters() - 1;
79         // instruction[0] is a call to Method.invoke(), where the receiver is a specific method,
80         // the first parameter is the receiver of the method invocation, and the second parameter
81         // is an array of objects corresponding to the parameters passed to the method.
82         int paramIndex = isConstructor ? 1 : 2;
83         int paramUse = invokeInstructions[0].getUse(paramIndex);
84         SSAInstruction instr = caller.getDU().getDef(paramUse);
85         if (!(instr instanceof SSANewInstruction)) {
86           return new ReceiverInstanceContext(receiver[0]);
87         }
88         SSANewInstruction newInstr = (SSANewInstruction) instr;
89         if (!newInstr.getConcreteType().isArrayType()) {
90           return null;
91         }
92         int vn = newInstr.getUse(0);
93         try {
94           int arrayLength = st.getIntValue(vn);
95           if (arrayLength == numberOfParams) {
96             return new ReceiverInstanceContext(receiver[0]);
97           } else {
98             return new IllegalArgumentExceptionContext();
99           }
100         } catch (IllegalArgumentException e) {
101           return new ReceiverInstanceContext(receiver[0]);
102         }
103       }
104     
105       /**
106        * This object may understand a dispatch to Constructor.newInstance().
107        */
108       private boolean mayUnderstand(CGNode caller, CallSiteReference site, IMethod targetMethod, InstanceKey instance) {
109         if (instance instanceof ConstantKey) {
110           if (targetMethod.getReference().equals(ReflectiveInvocationInterpreter.METHOD_INVOKE) || isConstructorConstant(instance)
111               && targetMethod.getReference().equals(ReflectiveInvocationInterpreter.CTOR_NEW_INSTANCE)) {
112             return true;
113           }
114         }
115         return false;
116       }
117     
118       private boolean isConstructorConstant(InstanceKey instance) {
119         if (instance instanceof ConstantKey) {
120           ConstantKey c = (ConstantKey) instance;
121           if (c.getConcreteType().getReference().equals(TypeReference.JavaLangReflectConstructor)) {
122             return true;
123           }
124         }
125         return false;
126       }
127     
128       private static final IntSet thisParameter = IntSetUtil.make(new int[]{0});
129     
130       @Override
131       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
132         if (site.isDispatch() || site.getDeclaredTarget().getNumberOfParameters() > 0) {
133           return thisParameter;
134         } else {
135           return EmptyIntSet.instance;
136         }
137       }
138     }
