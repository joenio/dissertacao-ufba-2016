42     /**
43      * Optimistic call graph builder that propagates inter-procedural data flow iteratively as
44      * call edges are discovered. Slower, but potentially more sound than {@link PessimisticCallGraphBuilder}.
45      * 
46      * This variant uses a worklist algorithm, generally making it scale better than
47      * {@link OptimisticCallgraphBuilder}, which repeatedly runs the pessimistic algorithm.
48      * 
49      * @author mschaefer
50      *
51      */
52     public class WorklistBasedOptimisticCallgraphBuilder extends FieldBasedCallGraphBuilder {
53     	/** The maximum number of iterations to perform. */
54     	public int ITERATION_CUTOFF = Integer.MAX_VALUE;
55     	
56     	private final boolean handleCallApply;
57     	
58     	private FlowGraphBuilder builder;

60     	public WorklistBasedOptimisticCallgraphBuilder(IClassHierarchy cha, AnalysisOptions options, AnalysisCache cache, boolean supportFullPointerAnalysis) {
61     		super(cha, options, cache, supportFullPointerAnalysis);
62     		handleCallApply = options instanceof JSAnalysisOptions && ((JSAnalysisOptions)options).handleCallApply();
63     	}

67     	  builder = new FlowGraphBuilder(cha, cache, false);
68     	  return builder.buildFlowGraph();

71     	@Override
72       protected Set<Pair<CallVertex,FuncVertex>> extractCallGraphEdges(FlowGraph flowgraph, IProgressMonitor monitor) throws CancelException {
73     	  VertexFactory factory = flowgraph.getVertexFactory();
74     	  Set<Vertex> worklist = HashSetFactory.make();
75     	  Map<Vertex, Set<FuncVertex>> reachingFunctions = HashMapFactory.make();
76     	  Map<VarVertex, JavaScriptInvoke> reflectiveCalleeVertices = HashMapFactory.make();
77     	  
78     	  for(Vertex v : flowgraph) {
79     	    if(v instanceof FuncVertex) {
80     	      FuncVertex fv = (FuncVertex)v;
81     	      worklist.add(fv);
82     	      MapUtil.findOrCreateSet(reachingFunctions, fv).add(fv);
83     	    }
84     	  }
85     	  
86     	  while(!worklist.isEmpty()) {
87           MonitorUtil.throwExceptionIfCanceled(monitor);
88     
89           Vertex v = worklist.iterator().next();
90           worklist.remove(v);
91           Set<FuncVertex> vReach = MapUtil.findOrCreateSet(reachingFunctions, v);
92     	    for(Vertex w : Iterator2Iterable.make(flowgraph.getSucc(v))) {
93     	      MonitorUtil.throwExceptionIfCanceled(monitor);
94     
95     	      Set<FuncVertex> wReach = MapUtil.findOrCreateSet(reachingFunctions, w);
96     	      boolean changed = false;
97     	      if(w instanceof CallVertex) {
98     	        for(FuncVertex fv : vReach) {
99     	          if(wReach.add(fv)) {
100     	            changed = true;
101     	            addCallEdge(flowgraph, (CallVertex)w, fv, worklist);
102     
103     	            // special handling of invocations of Function.prototype.call
104     	            if(handleCallApply && changed && fv.getFullName().equals("Lprologue.js/Function_prototype_call")) {
105     	              JavaScriptInvoke invk = ((CallVertex)w).getInstruction();
106     	              VarVertex reflectiveCalleeVertex = factory.makeVarVertex(((CallVertex)w).getCaller(), invk.getUse(1));
107                     reflectiveCalleeVertices.put(reflectiveCalleeVertex, invk);
108                     for(FuncVertex fw : MapUtil.findOrCreateSet(reachingFunctions, reflectiveCalleeVertex))
109                       addReflectiveCallEdge(flowgraph, reflectiveCalleeVertex, invk, fw, worklist);
110     	            }
111     	          }
112     	        }
113     	      } else if(handleCallApply && reflectiveCalleeVertices.containsKey(w)) {
114     	        JavaScriptInvoke invk = reflectiveCalleeVertices.get(w);
115     	        for(FuncVertex fv : vReach) {
116     	          if(wReach.add(fv)) {
117     	            changed = true;
118     	            addReflectiveCallEdge(flowgraph, (VarVertex)w, invk, fv, worklist);
119     	          }
120     	        }
121     	      } else {
122     
123     	        changed = wReach.addAll(vReach);
124     	      }
125     	      if(changed)
126     	        worklist.add(w);
127     	    }
128     	  }
129     	  
130     	  Set<Pair<CallVertex, FuncVertex>> res = HashSetFactory.make();
131     	  for(Vertex v : reachingFunctions.keySet())
132     	    if(v instanceof CallVertex)
133     	      for(FuncVertex fv : reachingFunctions.get(v))
134     	        res.add(Pair.make((CallVertex)v, fv));
135     	  return res;
136     	}

138     	// add flow corresponding to a new call edge
139     	private void addCallEdge(FlowGraph flowgraph, CallVertex c, FuncVertex callee, Set<Vertex> worklist) throws CancelException {
140     		VertexFactory factory = flowgraph.getVertexFactory();
141     		FuncVertex caller = c.getCaller();
142     		JavaScriptInvoke invk = c.getInstruction();
143     	
144     		int offset = 0;
145         if (invk.getDeclaredTarget().getSelector().equals(JavaScriptMethods.ctorReference.getSelector())) {
146           offset = 1;
147         }
148         
149     		for(int i=0;i<invk.getNumberOfParameters();++i) {
150     		  // only flow receiver into 'this' if invk is, in fact, a method call
151           flowgraph.addEdge(factory.makeVarVertex(caller, invk.getUse(i)), factory.makeArgVertex(callee));
152     		  if(i != 1 || !invk.getDeclaredTarget().getSelector().equals(AstMethodReference.fnSelector))
153     		    addFlowEdge(flowgraph, factory.makeVarVertex(caller, invk.getUse(i)), factory.makeParamVertex(callee, i+offset), worklist);
154     		}
155     
156     		// flow from return vertex to result vertex
157     		addFlowEdge(flowgraph, factory.makeRetVertex(callee), factory.makeVarVertex(caller, invk.getDef()), worklist);			
158     	}
159     
160       public void addFlowEdge(FlowGraph flowgraph, Vertex from, Vertex to, Set<Vertex> worklist) {
161         flowgraph.addEdge(from, to);
162         worklist.add(from);
163       }
164     	
165     	// add data flow corresponding to a reflective invocation via Function.prototype.call
166     	// NB: for f.call(...), f will _not_ appear as a call target, but the appropriate argument and return data flow will be set up
167       private void addReflectiveCallEdge(FlowGraph flowgraph, VarVertex reflectiveCallee, JavaScriptInvoke invk, FuncVertex realCallee, Set<Vertex> worklist) throws CancelException {
168         VertexFactory factory = flowgraph.getVertexFactory();
169         FuncVertex caller = reflectiveCallee.getFunction();
170     
171         // flow from arguments to parameters
172         for(int i=2;i<invk.getNumberOfParameters();++i) {
173           addFlowEdge(flowgraph, factory.makeVarVertex(caller, invk.getUse(i)), factory.makeParamVertex(realCallee, i-1), worklist);
174     
175           // flow from return vertex to result vertex
176           addFlowEdge(flowgraph, factory.makeRetVertex(realCallee), factory.makeVarVertex(caller, invk.getDef()), worklist);
177         }
178       }
