11     package com.ibm.wala.analysis.typeInference;
12     
13     import java.util.HashMap;
14     
15     import com.ibm.wala.types.TypeReference;
16     import com.ibm.wala.util.collections.HashMapFactory;
17     
18     /**
19      * Abstraction of a primitive type in Java.
20      */
21     public class JavaPrimitiveType extends PrimitiveType {
22     
23       public static final PrimitiveType BOOLEAN = makePrimitive(TypeReference.Boolean, 1);
24     
25       public static final PrimitiveType CHAR = makePrimitive(TypeReference.Char, 16);
26     
27       public static final PrimitiveType BYTE = makePrimitive(TypeReference.Byte, 8);
28     
29       public static final PrimitiveType SHORT = makePrimitive(TypeReference.Short, 16);
30     
31       public static final PrimitiveType INT = makePrimitive(TypeReference.Int, 32);
32     
33       public static final PrimitiveType LONG = makePrimitive(TypeReference.Long, 64);
34     
35       public static final PrimitiveType FLOAT = makePrimitive(TypeReference.Float, 32);
36     
37       public static final PrimitiveType DOUBLE = makePrimitive(TypeReference.Double, 64);
38     
39       public static void init() {
40         
41       }
42       
43       private JavaPrimitiveType(TypeReference reference, int size) {
44         super(reference, size);
45       }
46       
47       private static PrimitiveType makePrimitive(TypeReference reference, int size) {
48         return new JavaPrimitiveType(reference, size);
49       }
50     
51       final private static HashMap<String, String> primitiveNameMap;
52       static {
53         primitiveNameMap = HashMapFactory.make(9);
54         primitiveNameMap.put("I", "int");
55         primitiveNameMap.put("J", "long");
56         primitiveNameMap.put("S", "short");
57         primitiveNameMap.put("B", "byte");
58         primitiveNameMap.put("C", "char");
59         primitiveNameMap.put("D", "double");
60         primitiveNameMap.put("F", "float");
61         primitiveNameMap.put("Z", "boolean");
62         primitiveNameMap.put("V", "void");
63       }
64     
65       @Override
66       public String toString() {
67         String result = primitiveNameMap.get(reference.getName().toString());
68         return (result != null) ? result : reference.getName().toString();
69       }
70     
71     }
