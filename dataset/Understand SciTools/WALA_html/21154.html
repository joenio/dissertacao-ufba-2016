494       @Override
495       public TypeReference getConstantType(Object o) {
496         if (o == null) {
497           // TODO: do we really want null here instead of TypeReference.Null?
498           // lots of code seems to depend on this being null.
499           return null;
500         } else if (o instanceof Boolean) {
501           return TypeReference.Boolean;
502         } else if (o instanceof Long) {
503           return TypeReference.Long;
504         } else if (o instanceof Double) {
505           return TypeReference.Double;
506         } else if (o instanceof Float) {
507           return TypeReference.Float;
508         } else if (o instanceof Number) {
509           return TypeReference.Int;
510         } else if (o instanceof String) {
511           return TypeReference.JavaLangString;
512         } else if (o instanceof ClassToken || o instanceof TypeReference) {
513           return TypeReference.JavaLangClass;
514         } else if (o instanceof IMethod) {
515           IMethod m = (IMethod) o;
516           return m.isInit() ? TypeReference.JavaLangReflectConstructor : TypeReference.JavaLangReflectMethod;
517         } else if (o instanceof MethodHandle || o instanceof ReferenceToken) {
518           return TypeReference.JavaLangInvokeMethodHandle;
519         } else if (o instanceof MethodType) {
520           return TypeReference.JavaLangInvokeMethodType;
521         } else {
522           assert false : "unknown constant " + o + ": " + o.getClass();
523           return null;
524         }
525       }

542       /**
543        * @return Collection<TypeReference>, set of exception types a call to a declared target might throw.
544        * @throws InvalidClassFileException
545        * @throws IllegalArgumentException if target is null
546        * @throws IllegalArgumentException if cha is null
547        */
548       @Override
549       public Collection<TypeReference> inferInvokeExceptions(MethodReference target, IClassHierarchy cha)
550           throws InvalidClassFileException {
551     
552         if (cha == null) {
553           throw new IllegalArgumentException("cha is null");
554         }
555         if (target == null) {
556           throw new IllegalArgumentException("target is null");
557         }
558         ArrayList<TypeReference> set = new ArrayList<TypeReference>(cha.getJavaLangRuntimeExceptionTypes());
559         set.addAll(cha.getJavaLangErrorTypes());
560     
561         IClass klass = cha.lookupClass(target.getDeclaringClass());
562         if (klass == null) {
563           Warnings.add(MethodResolutionFailure.moderate(target));
564         }
565         if (klass != null) {
566           IMethod M = klass.getMethod(target.getSelector());
567           if (M == null) {
568             Warnings.add(MethodResolutionFailure.severe(target));
569           } else {
570             TypeReference[] exceptionTypes = M.getDeclaredExceptions();
571             if (exceptionTypes != null) {
572               set.addAll(Arrays.asList(exceptionTypes));
573             }
574           }
575         }
576         return set;
577       }

579       /**
580        * @param pei a potentially-excepting instruction
581        * @return the exception types that pei may throw, independent of the class hierarchy. null if none.
582        * 
583        *         Notes
584        *         <ul>
585        *         <li>this method will <em>NOT</em> return the exception type explicitly thrown by an athrow
586        *         <li>this method will <em>NOT</em> return the exception types that a called method may throw
587        *         <li>this method ignores OutOfMemoryError
588        *         <li>this method ignores linkage errors
589        *         <li>this method ignores IllegalMonitorState exceptions
590        *         </ul>
591        * 
592        * @throws IllegalArgumentException if pei is null
593        */
594       @Override
595       public Collection<TypeReference> getImplicitExceptionTypes(IInstruction pei) {
596         if (pei == null) {
597           throw new IllegalArgumentException("pei is null");
598         }
599         switch (((Instruction) pei).getOpcode()) {
600         case OP_iaload:
601         case OP_laload:
602         case OP_faload:
603         case OP_daload:
604         case OP_aaload:
605         case OP_baload:
606         case OP_caload:
607         case OP_saload:
608         case OP_iastore:
609         case OP_lastore:
610         case OP_fastore:
611         case OP_dastore:
612         case OP_bastore:
613         case OP_castore:
614         case OP_sastore:
615           return getArrayAccessExceptions();
616         case OP_aastore:
617           return getAaStoreExceptions();
618         case OP_getfield:
619         case OP_putfield:
620         case OP_invokevirtual:
621         case OP_invokespecial:
622         case OP_invokeinterface:
623           return getNullPointerException();
624         case OP_idiv:
625         case OP_irem:
626         case OP_ldiv:
627         case OP_lrem:
628           return getArithmeticException();
629         case OP_new:
630           return newScalarExceptions;
631         case OP_newarray:
632         case OP_anewarray:
633         case OP_multianewarray:
634           return newArrayExceptions;
635         case OP_arraylength:
636           return getNullPointerException();
637         case OP_athrow:
638           // N.B: the caller must handle the explicitly-thrown exception
639           return getNullPointerException();
640         case OP_checkcast:
641           return getClassCastException();
642         case OP_monitorenter:
643         case OP_monitorexit:
644           // we're currently ignoring MonitorStateExceptions, since J2EE stuff
645           // should be
646           // logically single-threaded
647           return getNullPointerException();
648         case OP_ldc_w:
649           if (((ConstantInstruction) pei).getType().equals(TYPE_Class))
650             return getClassNotFoundException();
651           else
652             return null;
653         case OP_getstatic:
654         case OP_putstatic:
655           return getExceptionInInitializerError();
656         default:
657           return Collections.emptySet();
658         }
659       }

715       @Override
716       public Object getMetadataToken(Object value) {
717         if (value instanceof ClassToken) {
718           return ShrikeUtil.makeTypeReference(ClassLoaderReference.Application, ((ClassToken) value).getTypeName());
719         } else if (value instanceof ReferenceToken) {
720           ReferenceToken tok = (ReferenceToken)value;
721           TypeReference cls = ShrikeUtil.makeTypeReference(ClassLoaderReference.Application, "L" + tok.getClassName());
722           return MethodReference.findOrCreate(cls, new Selector(Atom.findOrCreateUnicodeAtom(tok.getElementName()), Descriptor.findOrCreateUTF8(tok.getDescriptor())));
723         } else if (value instanceof MethodHandle || value instanceof MethodType) {
724           return value;
725         } else {
726           assert value instanceof TypeReference;
727           return value;
728         }
729       }
