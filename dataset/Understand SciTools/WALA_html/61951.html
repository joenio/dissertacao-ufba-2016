11     package com.ibm.wala.ipa.cfg;
12     
13     import com.ibm.wala.cfg.IBasicBlock;
14     
15     /**
16      * This class is used by the PrunedCFG to determine which edges in a given CFG should be kept in the pruned version.
17      */
18     public interface EdgeFilter<T extends IBasicBlock> {
19     
20       /**
21        * This method must return true if and only if a normal edge from src to dst exists in the original CFG and should be kept for the
22        * pruned version of that CFG. Note that this must _must_ return false for any normal edge that is not in the original CFG.
23        */
24       boolean hasNormalEdge(T src, T dst);
25     
26       /**
27        * This method must return true if and only if an exceptional edge from src to dst exists in the original CFG and should be kept
28        * for the pruned version of that CFG. Note that this must _must_ return false for any exceptional edge that is not in the
29        * original CFG.
30        */
31       boolean hasExceptionalEdge(T src, T dst);
32     
33     }
