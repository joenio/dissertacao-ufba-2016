85         /**
86          *  Perform all checks defined in this class.
87          *
88          *  @return if the checks passed
89          */
90         public boolean all() {
91             return allBut(EnumSet.noneOf(Test.class));
92         }
93     
94         /**
95          *  Perform all checks defined in this class but the listed ones.
96          *
97          *  @param skip checks not to perform
98          *  @return if the checks passed
99          */
100         public boolean allBut(Set<Test> skip) {
101             boolean pass = true;
102     
103             if (! skip.contains(Test.OVERRIDES)) {
104                 pass &= checkOverridesInPlace();
105             }
106             if (! skip.contains(Test.INTENTS)) {
107                 pass &= checkIntentSpecs();
108             }
109             if (! skip.contains(Test.REUSE)) {
110                 pass &= checkAllComponentsReuse();
111             }
112             if (! skip.contains(Test.STUBS_VERSION)) {
113                 pass &= checkStubsVersion();
114             }
115             if (! skip.contains(Test.OBJECT_IN_EP)) {
116                     pass &= checkNoObjectInEntryPoints();
117             }
118             return pass;
119         }

144         /**
145          *  Checks whether stubs are recent enough to contain some used functions.
146          *
147          *  If the stubs are to old some parts that rely on these functions may be 
148          *  skipped entirely.
149          *
150          *  @return if check passed
151          */
152         public boolean checkStubsVersion() {
153             boolean pass = true;
154     
155             if (this.cha.lookupClass(AndroidTypes.Fragment) == null) {
156                 pass = false;
157             }
158     
159             if (this.cha.lookupClass(AndroidTypes.UserHandle) == null) {
160                 pass = false;
161             }
162     
163             if (this.cha.resolveMethod(
164                             this.cha.lookupClass(AndroidTypes.Activity),
165                             Selector.make("getLoaderManager()Landroid/app/LoaderManager;")) == null) {
166                 pass = false;
167             }
168     
169             if (this.cha.resolveMethod(
170                             this.cha.lookupClass(AndroidTypes.Activity),
171                             Selector.make("getSystemService(Ljava/lang/String;)Ljava/lang/Object;")) == null) {
172                  pass = false;
173             }
174     
175             if (this.cha.resolveMethod(
176                             this.cha.lookupClass(AndroidTypes.Context),
177                             Selector.make("getSystemService(Ljava/lang/String;)Ljava/lang/Object;")) == null) {
178                 pass = false;
179             }
180     
181     
182             return pass;
183         }

185         /**
186          *  Is enough info present to resolve Intents.
187          *
188          *  This information is needed by the startComponent-functions in order to resolve the 
189          *  target of the call (if enough context is present).
190          *
191          *  If this information is unavailable the call will be resolved to the function
192          *  AndroidModel.Class.startUNKNOWNComponent which will call all Components of the
193          *  specific type (Activity, Service, ..) present in the application.
194          *
195          *  @return if check passed
196          */
197         public boolean checkIntentSpecs() {
198             boolean pass = true;
199     
200             final List <AndroidEntryPoint> entrypoits = AndroidEntryPointManager.ENTRIES;
201     
202             for (AndroidEntryPoint ep : entrypoits) {
203                 final TypeName test = ep.getMethod().getDeclaringClass().getName();
204                 if (! this.manager.existsIntentFor(test)) {
205                     if (test.toString().startsWith("Landroid/")) {
206                         continue;
207                     }
208                      pass = false;
209                 }
210             }
211     
212             return pass;
213         }
214     
215         /**
216          *  In order to for the startComponent-calls to work components should be set reuse.
217          *
218          *  If components are _not_ set reuse the following will happen:
219          *
220          *      * The caller context can not be set 
221          *      * No result will be transmitted back to onActivityResult
222          *
223          *  @return if the test passed
224          */
225         public boolean checkAllComponentsReuse() {
226             boolean pass = true;
227     
228             final IInstantiationBehavior behaviour = this.manager.getInstantiationBehavior(cha); // XXX: This generates false positives without cha!
229             final List <AndroidEntryPoint> entrypoits = AndroidEntryPointManager.ENTRIES;
230     
231             for (AndroidEntryPoint ep : entrypoits) {
232                 final TypeName test = ep.getMethod().getDeclaringClass().getName();
233                 IInstantiationBehavior.InstanceBehavior behave = behaviour.getBehavior(test,
234                         /* asParameterTo=  */ null, /* inCall= */ null, /* withName= */ null);
235                 if (behave != IInstantiationBehavior.InstanceBehavior.REUSE) {
236                      pass = false;
237                 }
238             }
239     
240             return pass;
241         }
242     
243         /**
244          *  Check if an Entrypoint takes an object.
245          */
246         public boolean checkNoObjectInEntryPoints() {
247             boolean pass = true;
248     
249             final List <AndroidEntryPoint> entrypoits = AndroidEntryPointManager.ENTRIES;
250             for (AndroidEntryPoint ep : entrypoits) {
251                 final TypeName[] params =  ep.getMethod().getDescriptor().getParameters();
252                 if (params == null) continue;
253                 for (final TypeName type : params) {
254                     if (type.equals(TypeReference.JavaLangObject.getName())) {  // Why if JavaLangObjectName private? .. narf
255                         pass = false;
256                     }
257                 }
258             }
259             return pass;
260         }
