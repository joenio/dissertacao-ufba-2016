11     package com.ibm.wala.util.graph;
12     
13     import com.ibm.wala.util.collections.SimpleVector;
14     
15     /**
16      * A simple, extremely inefficient tree implementation
17      */
18     public class BasicTree<T> {
19     
20       private final T value;
21       final private SimpleVector<BasicTree<T>> children = new SimpleVector<BasicTree<T>>();
22     
23       protected BasicTree(T value) {
24         this.value = value;
25       }
26       
27       public static <T> BasicTree<T> make(T value) {
28         if (value == null) {
29           throw new IllegalArgumentException("null value");
30         }
31         return new BasicTree<T>(value);
32       }
33       
34       public T getRootValue() {
35         return value;
36       }
37     
38       public T getChildValue(int i) {
39         if (children.get(i) == null) {
40           return null;
41         } else {
42           return children.get(i).getRootValue();
43         }
44       }
45       
46       public BasicTree<T> getChild(int i) {
47         return children.get(i);
48       }
49     
50       public void setChild(int i, BasicTree<T> tree) {
51         children.set(i, tree);
52       }
53       
54       public int getMaxChildIndex() {
55         return children.getMaxIndex();
56       }
57       
58     }
