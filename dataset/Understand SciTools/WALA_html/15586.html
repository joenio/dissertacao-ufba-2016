112     class SuperEnclosing {
113     	int seVar = 6;
114     	int seVar2 = 10;
115     	class SubEnclosed extends SuperEnclosing {
116     		int seVar2 = 11;
117     		SubEnclosed() {
118     			this.seVar = 5;
119     			SuperEnclosing.this.seVar = 7;
120     			this.seVar2 = 12;
121     			SuperEnclosing.this.seVar2 = 13;
122     		}
123     		int hello() {
124     			return seVar; // this is ours from SuperEnclosing, not the enlosing SuperEnclosing's
125     			// so even though the variable is defined in SuperEnclosing, this is NOT the same as
126     			// SuperEnclosing.this.x
127     		}
128     		int helloAgain() {
129     			return SubEnclosed.this.seVar; // this is ours from SuprEnclosing, not the enclosing SuperEnclosing's
130     		}
131     		int hello2() {
132     			return seVar2;
133     		}
134     	}
135     }

137     class SuperEnclosing2 {

139     	public void setSEVar() {
140     		seVar= 1001;
141     	}
142     	class SubEnclosed2 extends SuperEnclosing2 {
143     		SubEnclosed2() {
144     			// this.seVar = 5; // illegal 
145     			SuperEnclosing2.this.seVar += -(-(+7)); // makes 13
146     		}
147     		int hello() {
148     			return seVar; // since seVar is private, this now refers to the enclosing one
