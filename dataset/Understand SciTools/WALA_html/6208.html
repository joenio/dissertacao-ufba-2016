59       @Test public void testMultiDim() throws ClassHierarchyException, IllegalArgumentException, CancelException, IOException {
60         AnalysisScope scope = CallGraphTestUtil.makeJ2SEAnalysisScope(TestConstants.WALA_TESTDATA, CallGraphTestUtil.REGRESSION_EXCLUSIONS);
61         ClassHierarchy cha = ClassHierarchy.make(scope);
62         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util
63             .makeMainEntrypoints(scope, cha, TestConstants.MULTI_DIM_MAIN);
64         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
65     
66         CallGraphBuilder builder = Util.makeVanillaZeroOneCFABuilder(options, new AnalysisCache(),cha, scope);
67         CallGraph cg = builder.makeCallGraph(options, null);
68         PointerAnalysis<InstanceKey> pa = builder.getPointerAnalysis();
69         
70         CGNode node = findDoNothingNode(cg);
71         PointerKey pk = pa.getHeapModel().getPointerKeyForLocal(node, 1);
72         OrdinalSet<InstanceKey> ptsTo = pa.getPointsToSet(pk);
73         Assert.assertEquals(1, ptsTo.size());
74       }
75       
76       private final static CGNode findDoNothingNode(CallGraph cg) {
77         for (Iterator<? extends CGNode> it = cg.iterator(); it.hasNext(); ) {
78           CGNode n = it.next();
79           if (n.getMethod().getName().toString().equals("doNothing")) {
80             return n;
81           }
82         }
83         Assertions.UNREACHABLE("Unexpected: failed to find doNothing node");
84         return null;
85       }
