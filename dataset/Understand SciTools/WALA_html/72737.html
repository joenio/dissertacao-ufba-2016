94       @Test
95       public void testSlice1() throws ClassHierarchyException, IllegalArgumentException, CancelException, IOException {
96         AnalysisScope scope = findOrCreateAnalysisScope();
97         IClassHierarchy cha = findOrCreateCHA(scope);
98         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha,
99             TestConstants.SLICE1_MAIN);
100         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
101     
102         CallGraphBuilder builder = Util.makeZeroOneCFABuilder(options, new AnalysisCache(), cha, scope);
103         CallGraph cg = builder.makeCallGraph(options, null);
104     
105         CGNode main = findMainMethod(cg);
106     
107         Statement s = findCallTo(main, "println");
108         System.err.println("Statement: " + s);
109         // compute a data slice
110         Collection<Statement> computeBackwardSlice = Slicer.computeBackwardSlice(s, cg, builder.getPointerAnalysis(),
111             DataDependenceOptions.FULL, ControlDependenceOptions.NONE);
112         Collection<Statement> slice = computeBackwardSlice;
113         dumpSlice(slice);
114     
115         int i = 0;
116         for (Statement st : slice) {
117           if (st.getNode().getMethod().getDeclaringClass().getClassLoader().getReference().equals(ClassLoaderReference.Application)) {
118             i++;
119           }
120         }
121         Assert.assertEquals(16, i);
122       }

894         Descriptor d = Descriptor.findOrCreateUTF8("([Ljava/lang/String;)V");

899       /**
900        * @param cg
901        * @param d
902        * @param name
903        * @return
904        */
905       private static CGNode findMethod(CallGraph cg, Descriptor d, Atom name) {
906         for (Iterator<? extends CGNode> it = cg.getSuccNodes(cg.getFakeRootNode()); it.hasNext();) {
907           CGNode n = it.next();
908           if (n.getMethod().getName().equals(name) && n.getMethod().getDescriptor().equals(d)) {
909             return n;
910           }
911         }
912         // if it's not a successor of fake root, just iterate over everything
913         for (CGNode n : cg) {
914           if (n.getMethod().getName().equals(name) && n.getMethod().getDescriptor().equals(d)) {
915             return n;
916           }
917         }
918         Assertions.UNREACHABLE("failed to find method " + name);
919         return null;
920       }

922       public static CGNode findMethod(CallGraph cg, String name) {
923         Atom a = Atom.findOrCreateUnicodeAtom(name);
924         for (Iterator<? extends CGNode> it = cg.iterator(); it.hasNext();) {
925           CGNode n = it.next();
926           if (n.getMethod().getName().equals(a)) {
927             return n;
928           }
929         }
930         System.err.println("call graph " + cg);
931         Assertions.UNREACHABLE("failed to find method " + name);
932         return null;
933       }

935       public static Statement findCallTo(CGNode n, String methodName) {
936         IR ir = n.getIR();
937         for (Iterator<SSAInstruction> it = ir.iterateAllInstructions(); it.hasNext();) {
938           SSAInstruction s = it.next();
939           if (s instanceof SSAInvokeInstruction) {
940             SSAInvokeInstruction call = (SSAInvokeInstruction) s;
941             if (call.getCallSite().getDeclaredTarget().getName().toString().equals(methodName)) {
942               IntSet indices = ir.getCallInstructionIndices(((SSAInvokeInstruction) s).getCallSite());
943               Assertions.productionAssertion(indices.size() == 1, "expected 1 but got " + indices.size());
944               return new NormalStatement(n, indices.intIterator().next());
945             }
946           }
947         }
948         Assertions.UNREACHABLE("failed to find call to " + methodName + " in " + n);
949         return null;
950       }

952       public static Statement findFirstAllocation(CGNode n) {
953         IR ir = n.getIR();
954         for (int i = 0; i < ir.getInstructions().length; i++) {
955           SSAInstruction s = ir.getInstructions()[i];
956           if (s instanceof SSANewInstruction) {
957             return new NormalStatement(n, i);
958           }
959         }
960         Assertions.UNREACHABLE("failed to find allocation in " + n);
961         return null;
962       }
