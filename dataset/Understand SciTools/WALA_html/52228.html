43     /**
44      * An {@link SSAContextInterpreter} specialized to interpret reflective invocations such as Constructor.newInstance and
45      * Method.invoke on an {@link IMethod} constant.
46      */
47     public class ReflectiveInvocationInterpreter extends AbstractReflectionInterpreter {

49       public final static MethodReference CTOR_NEW_INSTANCE = MethodReference.findOrCreate(TypeReference.JavaLangReflectConstructor,
50           "newInstance", "([Ljava/lang/Object;)Ljava/lang/Object;");
51     
52       public final static MethodReference METHOD_INVOKE = MethodReference.findOrCreate(TypeReference.JavaLangReflectMethod, "invoke",
53           "(Ljava/lang/Object;[Ljava/lang/Object;)Ljava/lang/Object;");

56       private final Map<String, IR> cache = HashMapFactory.make();

59       /*
60        * @see com.ibm.wala.ipa.callgraph.propagation.SSAContextInterpreter#getIR(com.ibm.wala.ipa.callgraph.CGNode)
61        */
62       @Override
63       public IR getIR(CGNode node) {
64         if (node == null) {
65           throw new IllegalArgumentException("node is null");
66         }
67         assert understands(node);
68         if (DEBUG) {
69           System.err.println("generating IR for " + node);
70         }
71         ReceiverInstanceContext recv = (ReceiverInstanceContext) node.getContext();
72         ConstantKey c = (ConstantKey) recv.getReceiver();
73         IMethod m = (IMethod) c.getValue();
74     /** BEGIN Custom change: caching */
75         final IMethod method = node.getMethod();
76         final String hashKey = method.toString() + "@" + recv.toString();
77         
78         IR result = cache.get(hashKey);
79         
80         if (result == null) {
81           result = makeIR(method, m, recv);
82           cache.put(hashKey, result);
83         }
84         
85     /** END Custom change: caching */
86         return result;
87       }

98       /*
99        * @see com.ibm.wala.ipa.callgraph.propagation.rta.RTAContextInterpreter#understands(com.ibm.wala.ipa.callgraph.CGNode)
100        */
101       @Override
102       public boolean understands(CGNode node) {
103         if (node == null) {
104           throw new IllegalArgumentException("node is null");
105         }
106         if (!(node.getContext() instanceof ReceiverInstanceContext)) {
107           return false;
108         }
109         ReceiverInstanceContext r = (ReceiverInstanceContext) node.getContext();
110         if (!(r.getReceiver() instanceof ConstantKey)) {
111           return false;
112         }
113         return node.getMethod().getReference().equals(METHOD_INVOKE) || node.getMethod().getReference().equals(CTOR_NEW_INSTANCE);
114       }

137       /**
138        * TODO: clean this up. Create the IR for the synthetic method (e.g. Method.invoke)
139        * 
140        * @param method is something like Method.invoke or Construction.newInstance
141        * @param target is the method being called reflectively
142        */
143       private IR makeIR(IMethod method, IMethod target, ReceiverInstanceContext context) {
144         SSAInstructionFactory insts = method.getDeclaringClass().getClassLoader().getInstructionFactory();
145     
146         SpecializedMethod m = new SpecializedMethod(method, method.getDeclaringClass(), method.isStatic(), false);
147         Map<Integer, ConstantValue> constants = HashMapFactory.make();
148     
149         int nextLocal = method.getNumberOfParameters() + 1; // nextLocal = first free value number
150     
151         int nargs = target.getNumberOfParameters(); // nargs := number of parameters to target, including "this" pointer
152         int args[] = new int[nargs];
153         int pc = 0;
154         int parametersVn = -1; // parametersVn will hold the value number of parameters array
155     
156         if (method.getReference().equals(CTOR_NEW_INSTANCE)) {
157           // allocate the new object constructed
158           TypeReference allocatedType = target.getDeclaringClass().getReference();
159           m
160               .addInstruction(allocatedType, insts.NewInstruction(m.allInstructions.size(), args[0] = nextLocal++, NewSiteReference.make(pc++, allocatedType)),
161                   true);
162           parametersVn = 2;
163         } else {
164           // for Method.invoke, v3 is the parameter to the method being called
165           parametersVn = 3;
166           if (target.isStatic()) {
167             // do nothing
168           } else {
169             // set up args[0] == the receiver for method.invoke, held in v2.
170             // insert a cast for v2 to filter out bogus types
171             args[0] = nextLocal++;
172             TypeReference type = target.getParameterType(0);
173             SSACheckCastInstruction cast = insts.CheckCastInstruction(m.allInstructions.size(), args[0], 2, type, true);
174             m.addInstruction(null, cast, false);
175           }
176         }
177         int nextArg = target.isStatic() ? 0 : 1; // nextArg := next index in args[] array that needs to be initialized
178         int nextParameter = 0; // nextParameter := next index in the parameters[] array that needs to be copied into the args[] array.
179     
180         // load each of the parameters into a local variable, args[something]
181         for (int j = nextArg; j < nargs; j++) {
182           // load the next parameter into v_temp.
183           int indexConst = nextLocal++;
184           constants.put(new Integer(indexConst), new ConstantValue(nextParameter++));
185           int temp = nextLocal++;
186           m.addInstruction(null, insts.ArrayLoadInstruction(m.allInstructions.size(), temp, parametersVn, indexConst, TypeReference.JavaLangObject), false);
187           pc++;
188     
189           // cast v_temp to the appropriate type and store it in args[j]
190           args[j] = nextLocal++;
191           TypeReference type = target.getParameterType(j);
192           // we insert a cast to filter out bogus types
193           SSACheckCastInstruction cast = insts.CheckCastInstruction(m.allInstructions.size(), args[j], temp, type, true);
194           m.addInstruction(null, cast, false);
195           pc++;
196         }
197     
198         int exceptions = nextLocal++;
199         int result = -1;
200     
201         // emit the dispatch and return instructions
202         if (method.getReference().equals(CTOR_NEW_INSTANCE)) {
203           m.addInstruction(null, insts.InvokeInstruction(m.allInstructions.size(), args, exceptions, CallSiteReference.make(pc++, target.getReference(),
204               IInvokeInstruction.Dispatch.SPECIAL), null), false);
205           m.addInstruction(null, insts.ReturnInstruction(m.allInstructions.size(), args[0], false), false);
206         } else {
207           Dispatch d = target.isStatic() ? Dispatch.STATIC : Dispatch.VIRTUAL;
208           if (target.getReturnType().equals(TypeReference.Void)) {
209             m.addInstruction(null, insts.InvokeInstruction(m.allInstructions.size(), args, exceptions, CallSiteReference.make(pc++, target.getReference(), d), null),
210                 false);
211           } else {
212             result = nextLocal++;
213             m.addInstruction(null, insts.InvokeInstruction(m.allInstructions.size(), result, args, exceptions, CallSiteReference.make(pc++,
214                 target.getReference(), d), null), false);
215             m.addInstruction(null, insts.ReturnInstruction(m.allInstructions.size(), result, false), false);
216           }
217         }
218     
219         SSAInstruction[] instrs = new SSAInstruction[m.allInstructions.size()];
220         m.allInstructions.<SSAInstruction> toArray(instrs);
221     
222         return new SyntheticIR(method, context, new InducedCFG(instrs, method, context), instrs, SSAOptions.defaultOptions(), constants);
223       }
