37     /**
38      * Check properties of a call to clone() in RTA
39      */
40     public class CloneTest extends WalaTestCase {
41     
42       @Test public void testClone() throws ClassHierarchyException, IllegalArgumentException, CancelException, IOException {
43     
44         AnalysisScope scope = CallGraphTestUtil.makeJ2SEAnalysisScope(TestConstants.WALA_TESTDATA, CallGraphTestUtil.REGRESSION_EXCLUSIONS);
45         ClassHierarchy cha = ClassHierarchy.make(scope);
46         Iterable<Entrypoint> entrypoints = new AllApplicationEntrypoints(scope, cha);
47         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
48     
49         CallGraph cg = CallGraphTestUtil.buildRTA(options, new AnalysisCache(),cha, scope);
50     
51         // Find node corresponding to java.text.MessageFormat.clone()
52         TypeReference t = TypeReference.findOrCreate(ClassLoaderReference.Primordial, "Ljava/text/MessageFormat");
53         MethodReference m = MethodReference.findOrCreate(t, "clone", "()Ljava/lang/Object;");
54         CGNode node = cg.getNodes(m).iterator().next();
55     
56         // Check there's exactly one target for each super call in
57         // MessageFormat.clone()
58         for (Iterator<CallSiteReference> i = node.iterateCallSites(); i.hasNext();) {
59           CallSiteReference site = i.next();
60           if (site.isSpecial()) {
61             if (site.getDeclaredTarget().getDeclaringClass().equals(TypeReference.JavaLangObject)) {
62               Set<CGNode> targets = cg.getPossibleTargets(node, site);
63               if (targets.size() != 1) {
64                 System.err.println(targets.size() + " targets found for " + site);
65                 for (Iterator<CGNode> k = targets.iterator(); k.hasNext();) {
66                   System.err.println("  " + k.next());
67                 }
68                 Assert.fail("found " + targets.size() + " targets for " + site + " in " + node);
69               }
70             }
71           }
72         }
73       }
74     }
