47         throws ClassHierarchyException, IllegalArgumentException, CancelException, IOException
48       {
49         AnalysisScope scope = CallGraphTestUtil.makeJ2SEAnalysisScope(scopeFile, exclusionsFile);

52         CHACallGraph CG = new CHACallGraph(cha);

55         return CG;
56       }
