11     package com.ibm.wala.util.collections;
12     
13     import java.util.Iterator;
14     
15     import com.ibm.wala.util.functions.IntFunction;
16     import com.ibm.wala.util.intset.IntIterator;
17     
18     /**
19      * An <code>IntMapIterator</code> maps an <code>Iterator</code> contents to produce a new Iterator
20      */
21     public class IntMapIterator<T> implements Iterator<T> {
22       final IntIterator i;
23     
24       final IntFunction<T> f;
25     
26       public IntMapIterator(IntIterator i, IntFunction<T> f) {
27         if (i == null) {
28           throw new IllegalArgumentException("null i");
29         }
30         if (f == null) {
31           throw new IllegalArgumentException("null f");
32         }
33         this.i = i;
34         this.f = f;
35       }
36     
37       @Override
38       public T next() {
39         return f.apply(i.next());
40       }
41     
42       @Override
43       public boolean hasNext() {
44         return i.hasNext();
45       }
46     
47       @Override
48       public void remove() throws UnsupportedOperationException {
49         throw new java.lang.UnsupportedOperationException();
50       }
51     
52       @Override
53       public String toString() {
54         return "map: " + f + " of " + i;
55       }
56     
57     }
