227         for (IClass clazz : cha) {
228

230             for (IMethod method : clazz.getDeclaredMethods()) {
231               if (method.isPublic() && !method.isAbstract()) {
232                 System.out.println("Entry:" + method.getReference());

237         }
238         return new Iterable<Entrypoint>() {
239           @Override

284       public static Iterable<Entrypoint> makePrimordialMainEntrypoints(AnalysisScope scope, ClassHierarchy cha) {
285         final Atom mainMethod = Atom.findOrCreateAsciiAtom("main");
286         final HashSet<Entrypoint> result = HashSetFactory.make();
287         for (IClass klass : cha) {
288           MethodReference mainRef = MethodReference.findOrCreate(klass.getReference(), mainMethod, Descriptor

290           IMethod m = klass.getMethod(mainRef.getSelector());

292             result.add(new DefaultEntrypoint(m, cha));

294         }
295         return new Iterable<Entrypoint>() {
296           @Override

452       /**
453        * @param name
454        * @param cg
455        * @return a graph whose nodes are MethodReferences, and whose edges represent calls between MethodReferences
456        * @throws IllegalArgumentException if cg is null
457        */
458       public static Graph<MethodReference> squashCallGraph(final String name, final CallGraph cg) {
459         if (cg == null) {
460           throw new IllegalArgumentException("cg is null");
461         }
462         final Set<MethodReference> nodes = HashSetFactory.make();
463         for (Iterator<CGNode> nodesI = cg.iterator(); nodesI.hasNext();) {
464           nodes.add((nodesI.next()).getMethod().getReference());
465         }
466     
467         return new Graph<MethodReference>() {
468           @Override
469           public String toString() {
470             StringBuffer result = new StringBuffer();
471             result.append("squashed " + name + " call graph\n");
472             result.append("Original graph:");
473             result.append(cg.toString());
474             return result.toString();
475           }
