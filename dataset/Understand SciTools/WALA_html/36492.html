132       @Test
133       public void testReflect3() throws IOException, ClassHierarchyException, IllegalArgumentException, CancelException {
134         AnalysisScope scope = findOrCreateAnalysisScope();
135         IClassHierarchy cha = findOrCreateCHA(scope);
136         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha,
137             TestConstants.REFLECT3_MAIN);
138         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
139         CallGraph cg = CallGraphTestUtil.buildZeroOneCFA(options, new AnalysisCache(), cha, scope, false);
140     
141         TypeReference tr = TypeReference.findOrCreate(ClassLoaderReference.Application, "Ljava/lang/Class");
142         MethodReference mr = MethodReference.findOrCreate(tr, "newInstance", "()Ljava/lang/Object;");
143         Set<CGNode> newInstanceNodes = cg.getNodes(mr);
144         Set<CGNode> succNodes = HashSetFactory.make();
145         for (CGNode newInstanceNode : newInstanceNodes) {
146           Iterator<? extends CGNode> succNodesIter = cg.getSuccNodes(newInstanceNode);
147           while (succNodesIter.hasNext()) {
148             succNodes.add(succNodesIter.next());
149           }
150         }
151         TypeReference extraneousTR = TypeReference.findOrCreate(ClassLoaderReference.Application, "Lreflection/Reflect3$Hash");
152         IClass hashClass = cha.lookupClass(extraneousTR);
153         assert hashClass != null;
154         MethodReference extraneousMR = MethodReference.findOrCreate(extraneousTR, "<init>", "()V");
155         Set<CGNode> extraneousNodes = cg.getNodes(extraneousMR);
156         succNodes.retainAll(extraneousNodes);
157         Assert.assertTrue(succNodes.isEmpty());
158       }

250       @SuppressWarnings("unchecked")
251       @Test
252       public void testReflect7() throws Exception {
253         AnalysisScope scope = findOrCreateAnalysisScope();
254         IClassHierarchy cha = findOrCreateCHA(scope);
255         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha,
256             TestConstants.REFLECT7_MAIN);
257         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
258         CallGraph cg = CallGraphTestUtil.buildZeroOneCFA(options, new AnalysisCache(), cha, scope, false);
259     
260         final String mainClass = "Lreflection/Reflect7";
261         TypeReference mainTr = TypeReference.findOrCreate(ClassLoaderReference.Application, mainClass);
262         MethodReference mainMr = MethodReference.findOrCreate(mainTr, "main", "([Ljava/lang/String;)V");
263     
264         TypeReference constrTr = TypeReference.findOrCreate(ClassLoaderReference.Primordial, "Ljava/lang/reflect/Constructor");
265         MethodReference newInstanceMr = MethodReference
266             .findOrCreate(constrTr, "newInstance", "([Ljava/lang/Object;)Ljava/lang/Object;");
267     
268         String fpInitSig = "java.io.FilePermission.<init>(Ljava/lang/String;Ljava/lang/String;)V";
269         String fpToStringSig = "java.security.Permission.toString()Ljava/lang/String;";
270     
271         Set<CGNode> mainNodes = cg.getNodes(mainMr);
272     
273         // Get all the children of the main node(s)
274         Collection<CGNode> mainChildren = getSuccNodes(cg, mainNodes);
275     
276         // Verify that one of those children is Constructor.newInstance, where
277         // Constructor is a FilePermission constructor
278         CGNode filePermConstrNewInstanceNode = null;
279     
280         for (CGNode node : mainChildren) {
281           Context context = node.getContext();
282           if (context instanceof ReceiverInstanceContext && node.getMethod().getReference().equals(newInstanceMr)) {
283             ReceiverInstanceContext r = (ReceiverInstanceContext) context;
284             ConstantKey<IMethod> c = (ConstantKey<IMethod>) r.getReceiver();
285             IMethod ctor = c.getValue();
286             if (ctor.getSignature().equals(fpInitSig)) {
287               filePermConstrNewInstanceNode = node;
288               break;
289             }
290           }
291         }
292         Assert.assertTrue(filePermConstrNewInstanceNode != null);
293     
294         // Now verify that this node has FilePermission.<init> children
295         CGNode filePermInitNode = null;
296     
297         Iterator<? extends CGNode> filePermConstrNewInstanceChildren = cg.getSuccNodes(filePermConstrNewInstanceNode);
298         while (filePermConstrNewInstanceChildren.hasNext()) {
299           CGNode node = filePermConstrNewInstanceChildren.next();
300           if (node.getMethod().getSignature().equals(fpInitSig)) {
301             filePermInitNode = node;
302             break;
303           }
304         }
305         Assert.assertTrue(filePermInitNode != null);
306     
307         // Furthermore, verify that main has a FilePermission.toString child
308         CGNode filePermToStringNode = null;
309         for (CGNode node : mainChildren) {
310           if (node.getMethod().getSignature().equals(fpToStringSig)) {
311             filePermToStringNode = node;
312             break;
313           }
314         }
315     
316         Assert.assertTrue(filePermToStringNode != null);
317       }
318     
319       private Collection<CGNode> getSuccNodes(CallGraph cg, Collection<CGNode> nodes) {
320         Set<CGNode> succNodes = HashSetFactory.make();
321         for (CGNode newInstanceNode : nodes) {
322           Iterator<? extends CGNode> succNodesIter = cg.getSuccNodes(newInstanceNode);
323           while (succNodesIter.hasNext()) {
324             succNodes.add(succNodesIter.next());
325           }
326         }
327         return succNodes;
328       }

645        * <ul>
646        *  <li>GetMethodContext$B#foo()</li>
647        *  <li>GetMethodContext$C#bar()</li>
648        * </ul>
649        * and must not contain
650        * <ul>
651        *  <li>GetMethodContext$A#bar()</li>

654        *  <li>GetMethodContext$B#bar()</li>
655        *  <li>GetMethodContext$B#baz()</li>
656        *  <li>GetMethodContext$C#baz()</li>
657        *  <li>GetMethodContext$C#foo()</li>
658        * </ul>
659        */
660       @Test
661       public void testGetMethodContext() throws WalaException, IllegalArgumentException, CancelException, IOException {
662         TypeReference ta = TypeReference.findOrCreate(ClassLoaderReference.Application, "Lreflection/GetMethodContext$A");
663         TypeReference tb = TypeReference.findOrCreate(ClassLoaderReference.Application, "Lreflection/GetMethodContext$B");
664         TypeReference tc = TypeReference.findOrCreate(ClassLoaderReference.Application, "Lreflection/GetMethodContext$C");

666                  sbar = Selector.make("bar()V"),
667                  sbaz = Selector.make("baz()V");
668         MethodReference mafoo = MethodReference.findOrCreate(ta,sfoo),
669                         mbfoo = MethodReference.findOrCreate(tb,sfoo),
670                         mcfoo = MethodReference.findOrCreate(tc,sfoo),
671                         mabar = MethodReference.findOrCreate(ta,sbar),
672                         mbbar = MethodReference.findOrCreate(tb,sbar),
673                         mcbar = MethodReference.findOrCreate(tc,sbar),
674                         mabaz = MethodReference.findOrCreate(ta,sbaz),
675                         mbbaz = MethodReference.findOrCreate(tb,sbaz),
676                         mcbaz = MethodReference.findOrCreate(tc,sbaz);
677         AnalysisScope scope = findOrCreateAnalysisScope();
678         IClassHierarchy cha = findOrCreateCHA(scope);
679         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha,
680             TestConstants.REFLECTGETMETHODCONTEXT_MAIN);
681         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
682         CallGraph cg = CallGraphTestUtil.buildZeroOneCFA(options, new AnalysisCache(), cha, scope, false);
683         Set<CGNode> cgn;
684         cgn = cg.getNodes(mabar); Assert.assertTrue(cgn.isEmpty());
685         cgn = cg.getNodes(mabaz); Assert.assertTrue(cgn.isEmpty());
686         cgn = cg.getNodes(mafoo); Assert.assertTrue(cgn.isEmpty());
687         
688         cgn = cg.getNodes(mbbar); Assert.assertTrue(cgn.isEmpty());
689         cgn = cg.getNodes(mbbaz); Assert.assertTrue(cgn.isEmpty());
690         
691         cgn = cg.getNodes(mcbaz); Assert.assertTrue(cgn.isEmpty());
692         cgn = cg.getNodes(mcfoo); Assert.assertTrue(cgn.isEmpty());
693         
694         
695         cgn = cg.getNodes(mbfoo); Assert.assertTrue(1 == cgn.size());
696     
697         cgn = cg.getNodes(mcbar); Assert.assertTrue(1 == cgn.size());
