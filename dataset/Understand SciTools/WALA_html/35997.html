14     package com.ibm.wala.cast.java.translator.polyglot;
15     
16     import java.io.IOException;
17     import java.util.HashMap;
18     import java.util.Map;
19     
20     import com.ibm.wala.cast.java.ipa.callgraph.JavaSourceAnalysisScope;
21     import com.ibm.wala.classLoader.ClassLoaderFactoryImpl;
22     import com.ibm.wala.classLoader.ClassLoaderImpl;
23     import com.ibm.wala.classLoader.IClassLoader;
24     import com.ibm.wala.ipa.callgraph.AnalysisScope;
25     import com.ibm.wala.ipa.cha.IClassHierarchy;
26     import com.ibm.wala.types.ClassLoaderReference;
27     import com.ibm.wala.util.config.SetOfClasses;
28     
29     public class PolyglotClassLoaderFactory extends ClassLoaderFactoryImpl {
30     
31       /**
32        * A map from ClassLoaderReference to IRTranslatorExtension, so that source files in different languages are processed by the
33        * right kind of IRTranslatorExtension.
34        */
35       final protected Map<ClassLoaderReference, IRTranslatorExtension> fExtensionMap = new HashMap<ClassLoaderReference, IRTranslatorExtension>();
36     
37       public PolyglotClassLoaderFactory(SetOfClasses exclusions, IRTranslatorExtension javaExtInfo) {
38         super(exclusions);
39         fExtensionMap.put(JavaSourceAnalysisScope.SOURCE, javaExtInfo);
40       }
41     
42       protected IRTranslatorExtension getExtensionFor(ClassLoaderReference clr) {
43         return fExtensionMap.get(clr);
44       }
45     
46       @Override
47       protected IClassLoader makeNewClassLoader(ClassLoaderReference classLoaderReference, IClassHierarchy cha, IClassLoader parent,
48           AnalysisScope scope) throws IOException {
49         if (classLoaderReference.equals(JavaSourceAnalysisScope.SOURCE)) {
50           ClassLoaderImpl cl = new PolyglotSourceLoaderImpl(classLoaderReference, parent, getExclusions(), cha,
51               getExtensionFor(classLoaderReference));
52           cl.init(scope.getModules(classLoaderReference));
53           return cl;
54         } else {
55           return super.makeNewClassLoader(classLoaderReference, cha, parent, scope);
56         }
57       }
58     }
