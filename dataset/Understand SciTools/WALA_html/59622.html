11     package com.ibm.wala.util.graph.traverse;
12     
13     import java.util.ArrayList;
14     import java.util.HashSet;
15     import java.util.Iterator;
16     import java.util.NoSuchElementException;
17     
18     import com.ibm.wala.util.collections.HashSetFactory;
19     import com.ibm.wala.util.collections.NonNullSingletonIterator;
20     import com.ibm.wala.util.graph.Graph;
21     
22     /**
23      * This class implements breadth-first search over a Graph, returning an Iterator of the nodes of the graph in order of discovery.
24      * This class follows the outNodes of the graph nodes to define the graph, but this behavior can be changed by overriding the
25      * getConnected method.
26      * 
27      * This traversal only visits nodes within k hops of a root.
28      */
29     public class BoundedBFSIterator<T> implements Iterator<T> {
30     
31       /**
32        * List of nodes as discovered
33        */
34       final ArrayList<T> Q = new ArrayList<T>();
35     
36       /**
37        * Set of nodes that have been visited
38        */
39       final HashSet<T> visited = HashSetFactory.make();
40     
41       /**
42        * index of the node currently being searched
43        */
44       private int index = 0;
45     
46       /**
47        * Governing Graph
48        */
49       protected Graph<T> G;
50     
51       /**
52        * limit on number of hops
53        */
54       private final int k;
55     
56       /**
57        * boundary[i] is the first index which represents a child that is > i hops away.
58        */
59       private final int[] boundary;
60     
61       /**
62        * how many hops away is the next element.
63        */
64       private int currentHops = 0;
65     
66       /**
67        * Construct a breadth-first iterator starting with a particular node in a directed graph.
68        * 
69        * @param G the graph whose nodes to enumerate
70        * @throws IllegalArgumentException if G is null
71        */
72       public BoundedBFSIterator(Graph<T> G, T N, int k) {
73         if (G == null) {
74           throw new IllegalArgumentException("G is null");
75         }
76         if (k < 0) {
77           throw new IllegalArgumentException("invalid k : " + k);
78         }
79         this.k = k;
80         boundary = new int[k];
81         init(G, new NonNullSingletonIterator<T>(N));
82       }
83     
84       /**
85        * Construct a breadth-first enumerator across the (possibly improper) subset of nodes reachable from the nodes in the given
86        * enumeration.
87        * 
88        * @param G the graph whose nodes to enumerate
89        * @param nodes the set of nodes from which to start searching
90        * @throws IllegalArgumentException if G is null
91        */
92       public BoundedBFSIterator(Graph<T> G, Iterator<? extends T> nodes, int k) {
93         if (G == null) {
94           throw new IllegalArgumentException("G is null");
95         }
96         if (k < 0) {
97           throw new IllegalArgumentException("invalid k: " + k);
98         }
99         this.k = k;
100         boundary = new int[k];
101         init(G, nodes);
102       }
103     
104       private void init(Graph<T> G, Iterator<? extends T> nodes) {
105         this.G = G;
106         if (G.getNumberOfNodes() == 0) {
107           return;
108         }
109         while (nodes.hasNext()) {
110           T o = nodes.next();
111           if (!visited.contains(o)) {
112             Q.add(o);
113             visited.add(o);
114           }
115         }
116         index = 0;
117         if (Q.size() > 0) {
118           T current = Q.get(0);
119           visitChildren(current);
120         }
121       }
122     
123       private void visitChildren(T N) {
124         if (currentHops == k) {
125           return;
126         }
127         if (boundary[currentHops] == 0) {
128           boundary[currentHops] = Q.size();
129         }
130         for (Iterator<? extends T> children = getConnected(N); children.hasNext();) {
131           T child = children.next();
132           if (!visited.contains(child)) {
133             Q.add(child);
134             visited.add(child);
135           }
136         }
137       }
138     
139       /**
140        * Return whether there are any more nodes left to enumerate.
141        * 
142        * @return true if there nodes left to enumerate.
143        */
144       @Override
145       public boolean hasNext() {
146         return (Q.size() > index);
147       }
148     
149       /**
150        * Find the next graph node in discover time order.
151        * 
152        * @return the next graph node in discover time order.
153        */
154       @Override
155       public T next() throws NoSuchElementException {
156         if (!hasNext()) {
157           throw new NoSuchElementException();
158         }
159         T result = Q.get(index);
160         index++;
161         if (currentHops < k && index == boundary[currentHops]) {
162           currentHops++;
163         }
164         if (hasNext()) {
165           T N = Q.get(index);
166           visitChildren(N);
167         }
168         return result;
169       }
170     
171       /**
172        * get the out edges of a given node
173        * 
174        * @param n the node of which to get the out edges
175        * @return the out edges
176        * 
177        */
178       protected Iterator<? extends T> getConnected(T n) {
179         return G.getSuccNodes(n);
180       }
181     
182       /**
183        * @see java.util.Iterator#remove()
184        */
185       @Override
186       public void remove() throws UnsupportedOperationException {
187         throw new UnsupportedOperationException();
188       }
189     
190       /**
191        * @return the currentHops
192        */
193       public int getCurrentHops() {
194         return currentHops;
195       }
196     }
