244       /**
245        * compute a points-to set for a pointer key, aiming to satisfy some predicate
246        * 
247        * @param pk the pointer key
248        * @param ikeyPred the desired predicate that each instance key in the points-to set should ideally satisfy
249        * @return a pair consisting of (1) a {@link PointsToResult} indicating whether a points-to set satisfying the predicate was
250        *         computed, and (2) the last computed points-to set for the variable (possibly <code>null</code> if no points-to set
251        *         could be computed in the budget)
252        * @throws IllegalArgumentException if <code>pk</code> is not a {@link LocalPointerKey}; to eventually be fixed
253        */
254       public Pair<PointsToResult, Collection<InstanceKey>> getPointsTo(PointerKey pk, Predicate<InstanceKey> ikeyPred)
255           throws IllegalArgumentException {
256         Pair<PointsToResult, Collection<InstanceKeyAndState>> p = getPointsToWithStates(pk, ikeyPred);
257         final Collection<InstanceKeyAndState> p2SetWithStates = p.snd;
258         Collection<InstanceKey> finalP2Set = p2SetWithStates != null ? removeStates(p2SetWithStates) : Collections.<InstanceKey>emptySet();
259         return Pair.make(p.fst, finalP2Set);
260       }

281       /**
282        * Unwrap a Collection of WithState<T> objects, returning a Collection containing the wrapped objects
283        */
284       private static <T> Collection<T> removeStates(final Collection<? extends WithState<T>> p2SetWithStates) {
285         if (p2SetWithStates == null) {
286           throw new IllegalArgumentException("p2SetWithStates == null");
287         }
288         Collection<T> finalP2Set = Iterator2Collection.toSet(new MapIterator<WithState<T>, T>(p2SetWithStates.iterator(),
289             new Function<WithState<T>, T>() {
290     
291               @Override
292               public T apply(WithState<T> object) {
293                 return object.getWrapped();
294               }
295             }));
296         return finalP2Set;
297       }

316       private Pair<PointsToResult, Collection<InstanceKeyAndState>> outerRefinementLoop(PointerKeyAndState queried,
317           Predicate<InstanceKey> ikeyPred) {
318         Collection<InstanceKeyAndState> lastP2Set = null;
319         boolean succeeded = false;
320         int numPasses = refinementPolicy.getNumPasses();
321         int passNum = 0;
322         for (; passNum < numPasses; passNum++) {
323           setNumNodesTraversed(0);
324           setTraversalBudget(refinementPolicy.getBudgetForPass(passNum));
325           Collection<InstanceKeyAndState> curP2Set = null;
326           PointsToComputer computer = null;
327           boolean completedPassInBudget = false;
328           try {
329             while (true) {
330               try {
331                 computer = new PointsToComputer(queried);
332                 computer.compute();
333                 curP2Set = computer.getComputedP2Set(queried);
334                 // System.err.println("completed pass");
335                 if (DEBUG) {
336                   System.err.println("traversed " + getNumNodesTraversed() + " nodes");
337                   System.err.println("POINTS-TO SET " + curP2Set);
338                 }
339                 completedPassInBudget = true;
340                 break;
341               } catch (StatesMergedException e) {
342                 if (DEBUG) {
343                   System.err.println("restarting...");
344                 }
345               }
346             }
347           } catch (BudgetExceededException e) {
348     
349           }
350           if (curP2Set != null) {
351             if (lastP2Set == null) {
352               lastP2Set = curP2Set;
353             } else if (lastP2Set.size() > curP2Set.size()) {
354               // got a more precise set
355               assert removeStates(lastP2Set).containsAll(removeStates(curP2Set));
356               lastP2Set = curP2Set;
357             } else {
358               // new set size is >= lastP2Set, so don't update
359               assert removeStates(curP2Set).containsAll(removeStates(lastP2Set));
360             }
361             if (curP2Set.isEmpty() || passesPred(curP2Set, ikeyPred)) {
362               // we did it!
363               // if (curP2Set.isEmpty()) {
364               // System.err.println("EMPTY PTO SET");
365               // }
366               succeeded = true;
367               break;
368             } else if (completedPassInBudget) {
369             }
370           }
371           // if we get here, means either budget for pass was exceeded,
372           // or points-to set wasn't good enough
373           // so, start new pass, if more refinement to do
374           if (!refinementPolicy.nextPass()) {
375             break;
376           }
377         }
378         PointsToResult result = null;
379         if (succeeded) {
380           result = PointsToResult.SUCCESS;
381         } else if (passNum == numPasses) {
382           // we ran all the passes without succeeding and
383           // without the refinement policy giving up
384           result = PointsToResult.BUDGETEXCEEDED;
385         } else {
386           if (lastP2Set != null) {
387             result = PointsToResult.NOMOREREFINE;
388           } else {
389             // we stopped before the maximum number of passes, but we never
390             // actually finished a pass, so we count this as BUDGETEXCEEDED
391             result = PointsToResult.BUDGETEXCEEDED;
392           }
393         }
394         return Pair.make(result, lastP2Set);
395       }

402       /**
403        * check if the points-to set of a variable passes some predicate, without necessarily computing the whole points-to set
404        * 
405        * @param pk the pointer key
406        * @param ikeyPred the desired predicate that each instance key in the points-to set should ideally satisfy
407        * @param pa a pre-computed points-to analysis
408        * @return a {@link PointsToResult} indicating whether a points-to set satisfying the predicate was computed
409        * @throws IllegalArgumentException if <code>pk</code> is not a {@link LocalPointerKey}; to eventually be fixed
410        */
411       public PointsToResult pointsToPassesPred(PointerKey pk, Predicate<InstanceKey> ikeyPred, PointerAnalysis<InstanceKey> pa)
412           throws IllegalArgumentException {
413         if (!(pk instanceof com.ibm.wala.ipa.callgraph.propagation.LocalPointerKey)) {
414           throw new IllegalArgumentException("only locals for now");
415         }
416         LocalPointerKey queriedPk = (LocalPointerKey) pk;
417         if (DEBUG) {
418           System.err.println("answering query for " + pk);
419         }
420         boolean succeeded = false;
421         startNewQuery();
422         int numPasses = refinementPolicy.getNumPasses();
423         int passNum = 0;
424         boolean completedSomePass = false;
425         if (MEASURE_MEMORY_USAGE) {
426           lastQueryMemoryUse = -1;
427         }
428         for (; passNum < numPasses; passNum++) {
429           setNumNodesTraversed(0);
430           setTraversalBudget(refinementPolicy.getBudgetForPass(passNum));
431           boolean completedPassInBudget = false;
432           boolean passed = false;
433           long initialMemory = 0;
434           try {
435             while (true) {
436               try {
437                 if (MEASURE_MEMORY_USAGE) {
438                   initialMemory = Util.getUsedMemory();
439                 }
440                 PointsToComputer computer = new PointsToComputer(queriedPk);
441                 passed = doTopLevelTraversal(queriedPk, ikeyPred, computer, pa);
442                 // System.err.println("completed pass");
443                 if (DEBUG) {
444                   System.err.println("traversed " + getNumNodesTraversed() + " nodes");
445                 }
446                 completedPassInBudget = true;
447                 completedSomePass = true;
448                 break;
449               } catch (StatesMergedException e) {
450                 if (DEBUG) {
451                   System.err.println("restarting...");
452                 }
453               } finally {
454                 if (MEASURE_MEMORY_USAGE) {
455                   long memoryAfterPass = Util.getUsedMemory();
456                   assert initialMemory != 0;
457                   long usedByPass = memoryAfterPass - initialMemory;
458                   if (usedByPass > lastQueryMemoryUse) {
459                     lastQueryMemoryUse = usedByPass;
460                     if (usedByPass > 20000000) {
461                       System.err.println("DOH!");
462                       System.exit(1);
463                     }
464                   }
465                 }
466     
467               }
468             }
469           } catch (BudgetExceededException e) {
470     
471           }
472           if (completedPassInBudget) {
473             if (passed) {
474               succeeded = true;
475               break;
476             }
477           }
478           // if we get here, means either budget for pass was exceeded,
479           // or points-to set wasn't good enough
480           // so, start new pass, if more refinement to do
481           if (!refinementPolicy.nextPass()) {
482             break;
483           }
484         }
485         PointsToResult result = null;
486         if (succeeded) {
487           result = PointsToResult.SUCCESS;
488         } else if (passNum == numPasses) {
489           // we ran all the passes without succeeding and
490           // without the refinement policy giving up
491           result = PointsToResult.BUDGETEXCEEDED;
492         } else {
493           result = completedSomePass ? PointsToResult.NOMOREREFINE : PointsToResult.BUDGETEXCEEDED;
494         }
495         if (MEASURE_MEMORY_USAGE) {
496           System.err.println("memory " + lastQueryMemoryUse);
497         }
498         return result;
499       }

559       private Pair<PointsToResult, Collection<PointerKey>> getFlowsToInternal(InstanceKeyAndState ikAndState) {
560         InstanceKey ik = ikAndState.getInstanceKey();
561         if (!(ik instanceof InstanceKeyWithNode)) {
562           assert false : "TODO: handle " + ik.getClass();
563         }
564         if (DEBUG) {
565           System.err.println("answering flows-to query for " + ikAndState);
566         }
567         Collection<PointerKeyAndState> lastFlowsToSet = null;
568         boolean succeeded = false;
569         int numPasses = refinementPolicy.getNumPasses();
570         int passNum = 0;
571         for (; passNum < numPasses; passNum++) {
572           setNumNodesTraversed(0);
573           setTraversalBudget(refinementPolicy.getBudgetForPass(passNum));
574           Collection<PointerKeyAndState> curFlowsToSet = null;
575           FlowsToComputer computer = null;
576           try {
577             while (true) {
578               try {
579                 computer = new FlowsToComputer(ikAndState);
580                 computer.compute();
581                 curFlowsToSet = computer.getComputedFlowsToSet();
582                 // System.err.println("completed pass");
583                 if (DEBUG) {
584                   System.err.println("traversed " + getNumNodesTraversed() + " nodes");
585                   System.err.println("FLOWS-TO SET " + curFlowsToSet);
586                 }
587                 break;
588               } catch (StatesMergedException e) {
589                 if (DEBUG) {
590                   System.err.println("restarting...");
591                 }
592               }
593             }
594           } catch (BudgetExceededException e) {
595     
596           }
597           if (curFlowsToSet != null) {
598             if (lastFlowsToSet == null) {
599               lastFlowsToSet = curFlowsToSet;
600             } else if (lastFlowsToSet.size() > curFlowsToSet.size()) {
601               // got a more precise set
602               assert removeStates(lastFlowsToSet).containsAll(removeStates(curFlowsToSet));
603               lastFlowsToSet = curFlowsToSet;
604             } else {
605               // new set size is >= lastP2Set, so don't update
606               // TODO what is wrong with this assertion?!? --MS
607               // assert removeStates(curFlowsToSet).containsAll(removeStates(lastFlowsToSet));
608             }
609             // TODO add predicate support
610             if (curFlowsToSet.isEmpty() /* || passesPred(curFlowsToSet, ikeyPred) */) {
611               succeeded = true;
612               break;
613             }
614           }
615           // if we get here, means either budget for pass was exceeded,
616           // or points-to set wasn't good enough
617           // so, start new pass, if more refinement to do
618           if (!refinementPolicy.nextPass()) {
619             break;
620           }
621         }
622         PointsToResult result = null;
623         if (succeeded) {
624           result = PointsToResult.SUCCESS;
625         } else if (passNum == numPasses) {
626           // we ran all the passes without succeeding and
627           // without the refinement policy giving up
628           result = PointsToResult.BUDGETEXCEEDED;
629         } else {
630           if (lastFlowsToSet != null) {
631             result = PointsToResult.NOMOREREFINE;
632           } else {
633             // we stopped before the maximum number of passes, but we never
634             // actually finished a pass, so we count this as BUDGETEXCEEDED
635             result = PointsToResult.BUDGETEXCEEDED;
636           }
637         }
638         return Pair.make(result, lastFlowsToSet == null ? null : removeStates(lastFlowsToSet));
639       }

1109         /**
1110          * Adds new targets for a virtual call, based on the points-to set of the receiver, and propagates values for the parameters /
1111          * return value of the new targets. NOTE: this method will <em>not</em> do any propagation for virtual call targets that have
1112          * already been discovered.
1113          * 
1114          * @param receiverAndState the receiver
1115          * @param callSiteAndCGNode the call
1116          */
1117         void propTargets(PointerKeyAndState receiverAndState, CallerSiteContext callSiteAndCGNode) {
1118           final CGNode caller = callSiteAndCGNode.getCaller();
1119           CallSiteReference call = callSiteAndCGNode.getCallSite();
1120           final State receiverState = receiverAndState.getState();
1121           OrdinalSet<InstanceKeyAndState> p2set = makeOrdinalSet(find(pkToP2Set, receiverAndState));
1122           for (InstanceKeyAndState ikAndState : p2set) {
1123             InstanceKey ik = ikAndState.getInstanceKey();
1124             IMethod targetMethod = options.getMethodTargetSelector().getCalleeTarget(caller, call, ik.getConcreteType());
1125             if (targetMethod == null) {
1126               // NOTE: target method can be null because we don't
1127               // always have type filters
1128               continue;
1129             }
1130             // if we've already handled this target, we can stop
1131             if (callToOTFTargets.get(callSiteAndCGNode).contains(targetMethod)) {
1132               continue;
1133             }
1134             callToOTFTargets.put(callSiteAndCGNode, targetMethod);
1135             // TODO can we just pick one of these, rather than all of them?
1136             // TODO handle clone() properly
1137             Set<CGNode> targetCGNodes = cg.getNodes(targetMethod.getReference());
1138             for (final CGNode targetForCall : targetCGNodes) {
1139               if (DEBUG) {
1140                 System.err.println("adding target " + targetForCall + " for call " + call);
1141               }
1142               if (hasNullIR(targetForCall)) {
1143                 continue;
1144               }
1145               g.addSubgraphForNode(targetForCall);
1146               // need to check flows through parameters and returns,
1147               // in direction of value flow and reverse
1148               SSAAbstractInvokeInstruction[] calls = getCallInstrs(caller, call);
1149               for (final SSAAbstractInvokeInstruction invokeInstr : calls) {
1150                 final ReturnLabel returnLabel = ReturnLabel.make(new CallerSiteContext(caller, call));
1151                 if (invokeInstr.hasDef()) {
1152                   final PointerKeyAndState defAndState = new PointerKeyAndState(heapModel.getPointerKeyForLocal(caller, invokeInstr
1153                       .getDef()), receiverState);
1154                   final PointerKey ret = heapModel.getPointerKeyForReturnValue(targetForCall);
1155                   doTransition(receiverState, returnLabel, new Function<State, Object>() {
1156     
1157                     @Override
1158                     public Object apply(State retState) {
1159                       repropCallArg(defAndState, new PointerKeyAndState(ret, retState), returnLabel.bar());
1160                       return null;
1161                     }
1162     
1163                   });
1164                 }
1165                 final PointerKeyAndState exc = new PointerKeyAndState(heapModel.getPointerKeyForLocal(caller, invokeInstr
1166                     .getException()), receiverState);
1167                 final PointerKey excRet = heapModel.getPointerKeyForExceptionalReturnValue(targetForCall);
1168                 doTransition(receiverState, returnLabel, new Function<State, Object>() {
1169     
1170                   @Override
1171                   public Object apply(State excRetState) {
1172                     repropCallArg(exc, new PointerKeyAndState(excRet, excRetState), returnLabel.bar());
1173                     return null;
1174                   }
1175     
1176                 });
1177                 for (Iterator<Integer> iter = new PointerParamValueNumIterator(targetForCall); iter.hasNext();) {
