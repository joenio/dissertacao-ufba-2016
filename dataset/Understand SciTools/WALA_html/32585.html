12     package com.ibm.wala.ipa.callgraph.propagation;
13     
14     import java.util.HashMap;
15     import java.util.Map;
16     
17     import com.ibm.wala.ipa.callgraph.Context;
18     import com.ibm.wala.ipa.callgraph.ContextItem;
19     import com.ibm.wala.ipa.callgraph.ContextKey;
20     
21     /**
22      * A selective Cartesian product context that enforces object sensitivity on some set
23      * of parameter positions.
24      */
25     public class SelectiveCPAContext implements Context {
26       // base context
27       protected final Context base;
28       
29       // maps parameters to their abstract objects
30       private final Map<ContextKey, InstanceKey> parameterObjs;
31       
32       // cached hash code
33       private final int hashCode;
34     
35       // helper method for constructing the parameterObjs map
36       private static Map<ContextKey, InstanceKey> makeMap(InstanceKey[] x) {
37         Map<ContextKey, InstanceKey> result = new HashMap<ContextKey, InstanceKey>();
38         for(int i = 0; i < x.length; i++) {
39           if (x[i] != null) {
40             result.put(ContextKey.PARAMETERS[i], x[i]);
41           }
42         }
43         return result;
44       }
45     
46       public SelectiveCPAContext(Context base, InstanceKey[] x) {
47         this(base, makeMap(x));
48       }
49     
50       public SelectiveCPAContext(Context base, Map<ContextKey, InstanceKey> parameterObjs) {
51         this.base = base;
52         this.parameterObjs = parameterObjs;
53         hashCode = base.hashCode() ^ parameterObjs.hashCode();
54       }
55     
56       @Override
57       public ContextItem get(ContextKey name) {
58         if (parameterObjs.containsKey(name)) {
59           return new FilteredPointerKey.SingleInstanceFilter(parameterObjs.get(name));
60         } else {
61           return base.get(name);
62         }
63       }
64     
65       @Override
66       public int hashCode() {
67         return hashCode;
68       }
69     
70       @Override
71       public boolean equals(Object other) {
72         if (this == other) {
73           return true;
74         }
75         return other != null &&
76             getClass().equals(other.getClass()) &&
77             base.equals(((SelectiveCPAContext)other).base) &&
78             parameterObjs.equals(((SelectiveCPAContext)other).parameterObjs);
79       }     
80     
81     
82     }
