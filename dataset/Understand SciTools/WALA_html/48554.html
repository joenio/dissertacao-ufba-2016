154             final ClassDataItem classData = classDef.getClassData();

268           Collection<Annotation> getAnnotations(Set<AnnotationVisibility> types) {
269         	  Set<Annotation> result = HashSetFactory.make();
270         	  AnnotationDirectoryItem d = dexModuleEntry.getClassDefItem().getAnnotations();
271         	  if (d.getClassAnnotations() != null) {
272         		  for(AnnotationItem a : d.getClassAnnotations().getAnnotations()) {
273         			  if (types == null || types.contains(a.getVisibility())) {
274         				  result.add(DexUtil.getAnnotation(a, getClassLoader().getReference()));
275         			  }
276         		  }
277         	  }
278         	  return result;
279           }

289     	static Set<AnnotationVisibility> getTypes(boolean runtimeInvisible) {
290     		Set<AnnotationVisibility> types = HashSetFactory.make();
291       		types.add(AnnotationVisibility.SYSTEM);
292       		if (runtimeInvisible) {
293       			types.add(AnnotationVisibility.BUILD);
294       		} else {
295       			types.add(AnnotationVisibility.RUNTIME);
296       		}
297     		return types;
298     	}

300           List<AnnotationItem> getAnnotations(MethodIdItem m, Set<AnnotationVisibility> types) {
301         	  List<AnnotationItem> result = new ArrayList<AnnotationItem>();
302         	  AnnotationDirectoryItem d = dexModuleEntry.getClassDefItem().getAnnotations();
303         	  if (d != null) {
304         		  for(AnnotationItem a : d.getMethodAnnotations(m).getAnnotations()) {
305             		  if (types == null || types.contains(a.getVisibility())) {
306             			  result.add(a);
307             		  }
308         		  }
309         	  }
310         	  return result;
311           }

324           Map<Integer,List<AnnotationItem>> getParameterAnnotations(MethodIdItem m) {
325         	  Map<Integer,List<AnnotationItem>> result = HashMapFactory.make();
326         	  AnnotationDirectoryItem d = dexModuleEntry.getClassDefItem().getAnnotations();
327         	  if (d != null) {
328         		  int i = 0;
329         		  for(AnnotationSetItem as : d.getParameterAnnotations(m).getAnnotationSets()) {
330         			  for(AnnotationItem a : as.getAnnotations()) {
331         				  if (! result.containsKey(i)) {
332         					  result.put(i, new ArrayList<AnnotationItem>());
333         				  }
334         				  result.get(i).add(a);
335         			  }
336         			  i++;
337         		  }
338         	  }
339         	  return result;
340           }

343          * (non-Javadoc)
344          * @see com.ibm.wala.classLoader.BytecodeClass#computeDeclaredMethods()
345          */
346         @Override
347         protected IMethod[] computeDeclaredMethods() throws InvalidClassFileException {
348         	ArrayList<IMethod> methodsAL = new ArrayList<IMethod>();
349         	    	
350             if (methods == null && classDef.getClassData() == null)
351                 methods = new IMethod[0];
352     
353             if (methods == null && classDef.getClassData() != null){
354     //            final EncodedMethod[] directMethods = classDef.getClassData().getDirectMethods();
355     //            final EncodedMethod[] virtualMethods = classDef.getClassData().getVirtualMethods();
356                 final List<EncodedMethod> directMethods = classDef.getClassData().getDirectMethods();
357                 final List<EncodedMethod> virtualMethods = classDef.getClassData().getVirtualMethods();
358     
359                 //methods = new IMethod[dSize+vSize];
360     
361                 // Create Direct methods (static, private, constructor)
362                 for (int i = 0; i < directMethods.size(); i++) {
363                     EncodedMethod dMethod = directMethods.get(i);
364                     //methods[i] = new DexIMethod(dMethod,this);
365                     methodsAL.add(new DexIMethod(dMethod,this));
366     
367                     //Set construcorId
368                     //if ( (dMethod.accessFlags & CONSTRUCTOR.getValue()) != 0){
369                     //    construcorId = i;
370                     //}
371                     //Set clinitId             
372                     //if (methods[i].isClinit())
373                     if (methodsAL.get(i).isClinit()) {
374                         clinitId = i;
375                     }
376                 }
377     
378                 // Create virtual methods (other methods)
379                 for (int i = 0; i < virtualMethods.size(); i++) {
380                     //methods[dSize+i] = new DexIMethod(virtualMethods[i],this);
381                     methodsAL.add(new DexIMethod(virtualMethods.get(i),this));
382                     //is this enough to determine if the class is an activity?
383                     //maybe check superclass?  -- but that may also not be enough
384                     //may need to keep checking superclass of superclass, etc.
385                     
386                 }
387             }
388             
389             if (methods == null)
390             	methods = methodsAL.toArray(new IMethod[methodsAL.size()]);
391             
392             return methods;
393         }
