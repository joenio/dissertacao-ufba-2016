48       /**
49        * This class uses reflection to access classes and methods that are only
50        * available when Eclipse is running as an IDE environment. The choice to use
51        * reflection is related to builds: with this design the build doesn't need to
52        * provide IDE bundles during compilation and hence can spot invalid uses of
53        * such classes through this bundle.
54        * 
55        * Because of this class, this bundle must OPTIONALY require
56        * 'org.eclipse.core.resources'.
57        */
58       private static final class EclipseUtil {
59         private static Object workspaceRoot = null;
60         private static Method workspaceRoot_getFile = null;
61     
62         public static Module getJarFileModule(String fileName) {
63           // Using reflection to enable this code to be built without the
64           // org.eclipse.core.resources bundle
65           //
66           try {
67             if (workspaceRoot_getFile == null) {
68               Class<?> cls = Class.forName("org.eclipse.core.resources.ResourcesPlugin");
69               Method getWorkspaceMethod = cls.getDeclaredMethod("getWorkspace");
70               Object workspace = getWorkspaceMethod.invoke(null);
71               Method getRoot = workspace.getClass().getDeclaredMethod("getRoot");
72               workspaceRoot = getRoot.invoke(workspace);
73               workspaceRoot_getFile = workspaceRoot.getClass().getMethod("getFile", IPath.class);
74             }
75     
76             IPath path = new Path(fileName);
77             if (workspaceRoot_getFile.invoke(workspaceRoot, path) != null) {
78               return new JarFileModule(new JarFile(fileName, false));
79             }
80           } catch (Exception e) {
81           }
82           return null;
83         }
84       }
85       
86       @Override
87       public Module getJarFileModule(String fileName, ClassLoader loader) throws IOException {
88         if (CorePlugin.getDefault() == null) {
89           return getJarFileFromClassLoader(fileName, loader);
90         } else if (plugIn != null) {
91           return getFromPlugin(plugIn, fileName);
92         } else if (CorePlugin.IS_RESOURCES_BUNDLE_AVAILABLE) {
93           Module module = EclipseUtil.getJarFileModule(fileName);
94           if (module != null) {
95             return module;
96           }
97         }
98         return getFromPlugin(CorePlugin.getDefault(), fileName);
99       }

101       /**
102        * @param fileName
103        * @return the jar file packaged with this plug-in of the given name, or null
104        *         if not found.
105        */
106       private JarFileModule getFromPlugin(Plugin p, String fileName) throws IOException {
107         URL url = getFileURLFromPlugin(p, fileName);
108         return (url == null) ? null : new JarFileModule(new JarFile(filePathFromURL(url)));
109       }
110       
111       /**
112        * get a file URL for a file from a plugin
113        * 
114        * @param fileName
115        *          the file name
116        * @return the URL, or <code>null</code> if the file is not found
117        * @throws IOException
118        */
119       private  URL getFileURLFromPlugin(Plugin p, String fileName) throws IOException {
120         try {
121           URL url = FileLocator.find(p.getBundle(), new Path(fileName), null);
122           if (url == null) {
123             // try lib/fileName
124             String libFileName = "lib/" + fileName;
125             url = FileLocator.find(p.getBundle(), new Path(libFileName), null);
126             if (url == null) {
127               // try bin/fileName
128               String binFileName = "bin/" + fileName;
129               url = FileLocator.find(p.getBundle(), new Path(binFileName), null);
130               if (url == null) {
131                 // try it as an absolute path?
132                 File f = new File(fileName);
133                 if (!f.exists()) {
134                   // give up
135                   return null;
136                 } else {
137                   url = f.toURI().toURL();
138                 }
139               }
140             }
141           }
142           url = FileLocator.toFileURL(url);
143           url = fixupFileURLSpaces(url);
144           return url;
145         } catch (ExceptionInInitializerError e) {
146           throw new IOException("failure to get file URL for " + fileName);
147         }
148       }
149     
150       /**
151        * escape spaces in a URL, primarily to work around a bug in
152        * {@link File#toURL()}
153        * 
154        * @param url
155        * @return an escaped version of the URL
156        */
157       private URL fixupFileURLSpaces(URL url) {
158         String urlString = url.toExternalForm();
159         StringBuffer fixedUpUrl = new StringBuffer();
160         int lastIndex = 0;
161         while (true) {
162           int spaceIndex = urlString.indexOf(' ', lastIndex);
163     
164           if (spaceIndex < 0) {
165             fixedUpUrl.append(urlString.substring(lastIndex));
166             break;
167           }
168     
169           fixedUpUrl.append(urlString.substring(lastIndex, spaceIndex));
170           fixedUpUrl.append("%20");
171           lastIndex = spaceIndex + 1;
172         }
173         try {
174           return new URL(fixedUpUrl.toString());
175         } catch (MalformedURLException e) {
176           e.printStackTrace();
177           Assertions.UNREACHABLE();
178         }
179         return null;
180       }

182       @Override
183       public URL getResource(String fileName, ClassLoader loader) throws IOException {
184         if (fileName == null) {
185           throw new IllegalArgumentException("null fileName");
186         }
187         Plugin p = plugIn == null ? CorePlugin.getDefault() : plugIn;
188         if (p == null && loader == null) {
189           throw new IllegalArgumentException("null loader");
190         }
191         return (p == null) ? loader.getResource(fileName) : FileLocator.find(p.getBundle(),
192             new Path(fileName), null);
193       }
194       
195       @Override
196       public File getFile(String fileName, ClassLoader loader) throws IOException {
197         Plugin p = plugIn == null ? CorePlugin.getDefault() : plugIn;
198         if (p == null) {
199           return getFileFromClassLoader(fileName, loader); 
200         } else {
201             try {
202               return getFileFromPlugin(p, fileName);
203             } catch (IOException e) {
204               return getFileFromClassLoader(fileName, loader); 
205             }
206         }
207       }
208       
209       /**
210        * @param fileName
211        * @return the jar file packaged with this plug-in of the given name, or null
212        *         if not found.
213        * @throws IllegalArgumentException
214        *           if p is null
215        */
216       public File getFileFromPlugin(Plugin p, String fileName) throws IOException {
217     
218         if (p == null) {
219           throw new IllegalArgumentException("p is null");
220         }
221         if (fileName == null) {
222           throw new IllegalArgumentException("null fileName");
223         }
224         URL url = getFileURLFromPlugin(p, fileName);
225         if (url == null) {
226           throw new FileNotFoundException(fileName);
227         }
228         return new File(filePathFromURL(url));
229       }

231       /**
232        * This is fragile.  Use with care.
233        * @return a String representing the path to the wala.core plugin installation
234        */
235       public static String getWalaCorePluginHome() {
236         if (CorePlugin.getDefault() == null) {
237            return null;
238         }
239         String install = Platform.getInstallLocation().getURL().getPath();
240         Bundle b = Platform.getBundle("com.ibm.wala.core");
241         String l = b.getLocation();
242         if (l.startsWith("update@")) {
243           l = l.replace("update@", "");
244         }
245         if (l.startsWith("reference:file:")) {
246           return l.replace("reference:file:","");
247         } else {
248           return install + File.separator + l;
249         }
250       }
