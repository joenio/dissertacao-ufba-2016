108       @Test
109       public void testIntraproc1() {
110         AnalysisCache cache = new AnalysisCache();
111         final MethodReference ref = MethodReference.findOrCreate(ClassLoaderReference.Application, "Ldataflow/StaticDataflow", "test1",
112             "()V");
113         IMethod method = cha.resolveMethod(ref);
114         IR ir = cache.getIRFactory().makeIR(method, Everywhere.EVERYWHERE, SSAOptions.defaultOptions());
115         ExplodedControlFlowGraph ecfg = ExplodedControlFlowGraph.make(ir);
116         IntraprocReachingDefs reachingDefs = new IntraprocReachingDefs(ecfg, cha);
117         BitVectorSolver<IExplodedBasicBlock> solver = reachingDefs.analyze();
118         for (IExplodedBasicBlock ebb : ecfg) {
119           if (ebb.getNumber() == 4) {
120             IntSet out = solver.getOut(ebb).getValue();
121             Assert.assertEquals(1, out.size());
122             Assert.assertTrue(out.contains(1));
123           }
124         }
125       }

147       @Test
148       public void testContextInsensitive() throws IllegalArgumentException, CallGraphBuilderCancelException {
149         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha,
150             "Ldataflow/StaticDataflow");
151         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
152     
153         CallGraphBuilder builder = Util.makeZeroOneCFABuilder(options, new AnalysisCache(), cha, scope);
154         CallGraph cg = builder.makeCallGraph(options, null);
155         ExplodedInterproceduralCFG icfg = ExplodedInterproceduralCFG.make(cg);
156         ContextInsensitiveReachingDefs reachingDefs = new ContextInsensitiveReachingDefs(icfg, cha);
157         BitVectorSolver<BasicBlockInContext<IExplodedBasicBlock>> solver = reachingDefs.analyze();
158         for (BasicBlockInContext<IExplodedBasicBlock> bb : icfg) {
159           if (bb.getNode().toString().contains("testInterproc")) {
160             IExplodedBasicBlock delegate = bb.getDelegate();
161             if (delegate.getNumber() == 4) {
162               IntSet solution = solver.getOut(bb).getValue();
163               IntIterator intIterator = solution.intIterator();
164               List<Pair<CGNode, Integer>> applicationDefs = new ArrayList<Pair<CGNode,Integer>>();
165               while (intIterator.hasNext()) {
166                 int next = intIterator.next();
167                 final Pair<CGNode, Integer> def = reachingDefs.getNodeAndInstrForNumber(next);
168                 if (def.fst.getMethod().getDeclaringClass().getClassLoader().getReference().equals(ClassLoaderReference.Application)) {
169                   System.out.println(def);
170                   applicationDefs.add(def);
171                 }
172               }
173               Assert.assertEquals(2, applicationDefs.size());
174             }
175           }
176         }
177       }

179       @Test
180       public void testContextSensitive() throws IllegalArgumentException, CancelException {
181         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha,
182             "Ldataflow/StaticDataflow");
183         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
184     
185         CallGraphBuilder builder = Util.makeZeroOneCFABuilder(options, new AnalysisCache(), cha, scope);
186         CallGraph cg = builder.makeCallGraph(options, null);
187         AnalysisCache cache = new AnalysisCache();
188         ContextSensitiveReachingDefs reachingDefs = new ContextSensitiveReachingDefs(cg, cache);
189         TabulationResult<BasicBlockInContext<IExplodedBasicBlock>, CGNode, Pair<CGNode, Integer>> result = reachingDefs.analyze();
190         ISupergraph<BasicBlockInContext<IExplodedBasicBlock>, CGNode> supergraph = reachingDefs.getSupergraph();
191         for (BasicBlockInContext<IExplodedBasicBlock> bb : supergraph) {
192           if (bb.getNode().toString().contains("testInterproc")) {
193             IExplodedBasicBlock delegate = bb.getDelegate();
194             if (delegate.getNumber() == 4) {
195               IntSet solution = result.getResult(bb);
196               IntIterator intIterator = solution.intIterator();
197               List<Pair<CGNode, Integer>> applicationDefs = new ArrayList<Pair<CGNode,Integer>>();
198               while (intIterator.hasNext()) {
199                 int next = intIterator.next();
200                 final Pair<CGNode, Integer> def = reachingDefs.getDomain().getMappedObject(next);
201                 if (def.fst.getMethod().getDeclaringClass().getClassLoader().getReference().equals(ClassLoaderReference.Application)) {
202                   System.out.println(def);
203                   applicationDefs.add(def);
204                 }
205               }
206               Assert.assertEquals(1, applicationDefs.size());
207             }
208           }
209         }
210       }
