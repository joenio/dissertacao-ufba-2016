11     package com.ibm.wala.ipa.callgraph.propagation;
12     
13     import com.ibm.wala.fixpoint.UnaryOperator;
14     import com.ibm.wala.fixpoint.UnaryStatement;
15     
16     /**
17      * A specialized equation class introduced for efficiency.
18      */
19     public final class AssignEquation extends UnaryStatement<PointsToSetVariable> {
20     
21       AssignEquation(PointsToSetVariable lhs, PointsToSetVariable rhs) {
22         super(lhs, rhs);
23       }
24     
25       @Override
26       public UnaryOperator<PointsToSetVariable> getOperator() {
27         return PropagationCallGraphBuilder.assignOperator;
28       }
29     
30       @Override
31       public boolean equals(Object o) {
32         if (o instanceof AssignEquation) {
33           AssignEquation other = (AssignEquation) o;
34           return getLHS().equals(other.getLHS()) && getRightHandSide().equals(other.getRightHandSide());
35         } else {
36           return false;
37         }
38       }
39     }
