29       public static void addClassToHierarchy(ClassHierarchyStore store, ClassReader cr) throws InvalidClassFileException,
30           IllegalArgumentException {
31         if (store == null) {
32           throw new IllegalArgumentException("store is null");
33         }
34         if (cr == null) {
35           throw new IllegalArgumentException();
36         }
37         String[] superInterfaces = new String[cr.getInterfaceCount()];
38         for (int i = 0; i < superInterfaces.length; i++) {
39           superInterfaces[i] = CTDecoder.convertClassToType(cr.getInterfaceName(i));
40         }
41         String superName = cr.getSuperName();
42         if ("java/io/File".equals(cr.getName()) || "java/lang/Throwable".equals(cr.getName())) {
43           System.err.println(superName);
44         }
45         store.setClassInfo(CTDecoder.convertClassToType(cr.getName()), (cr.getAccessFlags() & Constants.ACC_INTERFACE) != 0, (cr
46             .getAccessFlags() & Constants.ACC_FINAL) != 0, superName != null ? CTDecoder.convertClassToType(superName) : null,
47             superInterfaces);
48       }

50       /**
51        * Compile and add a method to a {@link ClassWriter}.
52        * 
53        * @param md the method data
54        * @param classWriter the target class writer
55        * @param rawLines line number information if available, otherwise <code>null</code>
56        */
57       public static void compileAndAddMethodToClassWriter(MethodData md, ClassWriter classWriter, ClassWriter.Element rawLines) {
58         if (classWriter == null) {
59           throw new IllegalArgumentException("classWriter is null");
60         }
61         if (md == null) {
62           throw new IllegalArgumentException("md is null");
63         }
64         CTCompiler compiler = CTCompiler.make(classWriter, md);
65         compiler.compile();
66         CTCompiler.Output output = compiler.getOutput();
67         CodeWriter code = new CodeWriter(classWriter);
68         code.setMaxStack(output.getMaxStack());
69         code.setMaxLocals(output.getMaxLocals());
70         code.setCode(output.getCode());
71         code.setRawHandlers(output.getRawHandlers());
72       
73         LineNumberTableWriter lines = null;
74         // I guess it is the line numbers in the java files.
75         if (rawLines == null) {
76           // add fake line numbers: just map each bytecode instruction to its own
77           // 'line'
78       
79           // NOTE:Should not use md.getInstructions().length, because the
80           // the length of the created code can be smaller than the md's instruction
81           // length
82       
83           // WRONG: int[] newLineMap = new int[md.getInstructions().length];
84           int[] newLineMap = new int[code.getCodeLength()];
85           for (int i = 0; i < newLineMap.length; i++) {
86             newLineMap[i] = i;
87           }
88           int[] rawTable = LineNumberTableWriter.makeRawTable(newLineMap);
89           lines = new LineNumberTableWriter(classWriter);
90           lines.setRawTable(rawTable);
91         }
92         code.setAttributes(new ClassWriter.Element[] { rawLines == null ? lines : rawLines });
93         Element[] elements = { code };
94         // System.out.println("Name:"+md.getName()+" Sig:"+md.getSignature());
95         classWriter.addMethod(md.getAccess(), md.getName(), md.getSignature(), elements);
96       }
