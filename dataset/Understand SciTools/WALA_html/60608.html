59        * Build an IR, then check integrity on two flavors of CFG

61       private void doMethod(String methodSig) {
62         try {
63           MethodReference mr = StringStuff.makeMethodReference(Language.JAVA, methodSig);
64     
65           IMethod m = cha.resolveMethod(mr);
66           if (m == null) {
67             Assertions.UNREACHABLE("could not resolve " + mr);
68           }
69           AnalysisOptions options = new AnalysisOptions();
70           AnalysisCache cache = makeAnalysisCache();
71           options.getSSAOptions().setPiNodePolicy(SSAOptions.getAllBuiltInPiNodes());
72           IR ir = cache.getSSACache().findOrCreateIR(m, Everywhere.EVERYWHERE, options.getSSAOptions());
73     
74           ControlFlowGraph<SSAInstruction, ISSABasicBlock> cfg = ir.getControlFlowGraph();
75           try {
76             GraphIntegrity.check(cfg);
77           } catch (UnsoundGraphException e) {
78             e.printStackTrace();
79             System.err.println(ir);
80             Assert.assertTrue(" failed cfg integrity check for " + methodSig, false);
81           }
82     
83           try {
84             GraphIntegrity.check(cfg);
85           } catch (UnsoundGraphException e) {
86             e.printStackTrace();
87             System.err.println(ir);
88             System.err.println(cfg);
89             Assert.assertTrue(" failed 2-exit cfg integrity check for " + methodSig, false);
90           }
91         } catch (Exception e) {
92           e.printStackTrace();
93           Assertions.UNREACHABLE();
94         }
95       }

109       @Test
110       public void testIRCacheIdempotence() {
111         MethodReference mr = StringStuff.makeMethodReference("hello.Hello.main([Ljava/lang/String;)V");
112     
113         IMethod m = cha.resolveMethod(mr);
114         AnalysisCache cache = makeAnalysisCache();
115         IR irBefore = cache.getIR(m);
116         cache.getSSACache().wipe();
117         IR irAfter = cache.getIR(m);
118         for (int i = 0; i < irBefore.getInstructions().length; i++) {
119           System.out.println(irBefore.getInstructions()[i]);
120           System.out.println(irAfter.getInstructions()[i]);
121           Assert.assertEquals(irAfter.getInstructions()[i], irBefore.getInstructions()[i]);
122         }
123       }

163       public static void testCFG(SSACFG cfg, int[][] assertions) {
164       	for(int i = 0; i < assertions.length; i++) {
165       		SSACFG.BasicBlock bb= cfg.getNode(i);
166       		Assert.assertEquals("basic block " + i, assertions[i].length, cfg.getSuccNodeCount(bb));
167       		for(int j = 0; j < assertions[i].length; j++) {
168       			Assert.assertTrue(cfg.hasEdge(bb, cfg.getNode(assertions[i][j])));
169       		}
170       	}
171       }
