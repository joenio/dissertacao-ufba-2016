11     package com.ibm.wala.util.collections;
12     
13     import java.util.ArrayList;
14     import java.util.Collection;
15     import java.util.Iterator;
16     import java.util.LinkedHashSet;
17     import java.util.List;

20     /**
21      * Converts an {@link Iterator} to a {@link Collection}. Note that if you just want to use Java 5's for-each loop with an
22      * {@link Iterator}, use {@link Iterator2Iterable}.
23      * 
24      * @see Iterator2Iterable
25      */
26     public abstract class Iterator2Collection<T> implements Collection<T> {
27     
28       protected abstract Collection<T> getDelegate();
29       /**
30        * Returns a {@link Set} containing all elements in i. Note that duplicates will be removed.
31        */
32       public static <T> Iterator2Set<T> toSet(Iterator<? extends T> i) throws IllegalArgumentException {
33         if (i == null) {
34           throw new IllegalArgumentException("i == null");
35         }
36         return new Iterator2Set<T>(i, new LinkedHashSet<T>(5));
37       }
38     
39       /**
40        * Returns a {@link List} containing all elements in i, preserving duplicates.
41        */
42       public static <T> Iterator2List<T> toList(Iterator<? extends T> i) throws IllegalArgumentException {
43         if (i == null) {
44           throw new IllegalArgumentException("i == null");
45         }
46         return new Iterator2List<T>(i, new ArrayList<T>(5));
47       }
48     
49       @Override
50       public String toString() {
51         return getDelegate().toString();
52       }
53     
54       /*
55        * @see java.util.Collection#size()
56        */
57       @Override
58       public int size() {
59         return getDelegate().size();
60       }
61     
62       /*
63        * @see java.util.Collection#clear()
64        */
65       @Override
66       public void clear() {
67         getDelegate().clear();
68       }
69     
70       /*
71        * @see java.util.Collection#isEmpty()
72        */
73       @Override
74       public boolean isEmpty() {
75         return getDelegate().isEmpty();
76       }
77     
78       /*
79        * @see java.util.Collection#toArray()
80        */
81       @Override
82       public Object[] toArray() {
83         return getDelegate().toArray();
84       }
85     
86       /*
87        * @see java.util.Collection#add(java.lang.Object)
88        */
89       @Override
90       public boolean add(T arg0) {
91         return getDelegate().add(arg0);
92       }
93     
94       /*
95        * @see java.util.Collection#contains(java.lang.Object)
96        */
97       @Override
98       public boolean contains(Object arg0) {
99         return getDelegate().contains(arg0);
100       }
101     
102       /*
103        * @see java.util.Collection#remove(java.lang.Object)
104        */
105       @Override
106       public boolean remove(Object arg0) {
107         return getDelegate().remove(arg0);
108       }
109     
110       /*
111        * @see java.util.Collection#addAll(java.util.Collection)
112        */
113       @Override
114       public boolean addAll(Collection<? extends T> arg0) {
115         return getDelegate().addAll(arg0);
116       }
117     
118       /*
119        * @see java.util.Collection#containsAll(java.util.Collection)
120        */
121       @Override
122       public boolean containsAll(Collection<?> arg0) {
123         return getDelegate().containsAll(arg0);
124       }
125     
126       /*
127        * @see java.util.Collection#removeAll(java.util.Collection)
128        */
129       @Override
130       public boolean removeAll(Collection<?> arg0) {
131         return getDelegate().removeAll(arg0);
132       }
133     
134       /*
135        * @see java.util.Collection#retainAll(java.util.Collection)
136        */
137       @Override
138       public boolean retainAll(Collection<?> arg0) {
139         return getDelegate().retainAll(arg0);
140       }
141     
142       /*
143        * @see java.util.Collection#iterator()
144        */
145       @Override
146       public Iterator<T> iterator() {
147         return getDelegate().iterator();
148       }
149     
150       @Override
151       @SuppressWarnings("hiding")
152       public <T> T[] toArray(T[] a) {
153         return getDelegate().toArray(a);
154       }
155     
156       @Override
157       public boolean equals(Object o) {
158         return getDelegate().equals(o);
159       }
160     
161       @Override
162       public int hashCode() {
163         return getDelegate().hashCode();
164       }
165     }
