11     package com.ibm.wala.cast.js.callgraph.fieldbased.flowgraph.vertices;
12     
13     import com.ibm.wala.cast.js.ssa.JavaScriptInvoke;
14     import com.ibm.wala.cast.js.types.JavaScriptMethods;
15     import com.ibm.wala.classLoader.CallSiteReference;
16     
17     /**
18      * A call vertex represents the possible callees of a function call or <code>new</code> expression.
19      * 
20      * @author mschaefer
21      *
22      */
23     public class CallVertex extends Vertex {
24     	// method containing the call
25     	private final FuncVertex func;
26     	
27     	// PC of the call site
28     	private final CallSiteReference site;
29     	
30     	// the call instruction itself
31     	private final JavaScriptInvoke invk;
32     
33     	CallVertex(FuncVertex func, CallSiteReference site, JavaScriptInvoke invk) {
34     		this.func = func;
35     		this.site = site;
36     		this.invk = invk;
37     	}
38     
39     	public FuncVertex getCaller() {
40     		return func;
41     	}
42     
43     	public CallSiteReference getSite() {
44     		return site;
45     	}
46     	
47     	public JavaScriptInvoke getInstruction() {
48     		return invk;
49     	}
50     	
51     	/**
52     	 * Does this call vertex correspond to a <code>new</code> instruction? 
53     	 */
54     	public boolean isNew() {
55     		return site.getDeclaredTarget() == JavaScriptMethods.ctorReference;
56     	}
57     
58     	@Override
59     	public <T> T accept(VertexVisitor<T> visitor) {
60     		return visitor.visitCalleeVertex(this);
61     	}
62     	
63     	@Override
64     	public String toString() {
65     		return "Callee(" + func + ", " + site + ")";
66     	}
67     }
