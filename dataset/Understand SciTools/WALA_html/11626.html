51     import org.jf.dexlib.Code.Opcode;

53     import com.ibm.wala.dalvik.classLoader.DexIMethod;

57     public abstract class Invoke extends Instruction {
58     
59         public final int [] args;
60         public final String clazzName;
61         public final String methodName;
62         public final String descriptor;
63     
64         protected Invoke(int instLoc, String clazzName, String methodName, String descriptor, int [] args, Opcode opcode, DexIMethod method)
65         {
66             super(instLoc, opcode, method);
67             this.clazzName = clazzName;
68             this.methodName = methodName;
69             this.descriptor = descriptor;
70             this.args = args;
71         }
72     
73         public static class InvokeVirtual extends Invoke
74         {
75     
76             public InvokeVirtual(int instLoc,
77                     String clazzName, String methodName, String descriptor,
78                     int[] args, Opcode opcode, DexIMethod method) {
79                 super(instLoc, clazzName, methodName, descriptor, args, opcode, method);
80             }
81             @Override
82             public IDispatch getInvocationCode() {
83                 return IInvokeInstruction.Dispatch.VIRTUAL;
84             }
85     
86             @Override
87             public String toString()
88             {
89                 StringBuilder argString = new StringBuilder();
90                 argString.append("(");
91                 String sep = "";
92                 for(int r:args)
93                 {
94                     argString.append(sep+r);
95                     sep=",";
96                 }
97                 argString.append(")");
98                 return "InvokeVirtual "+clazzName+" "+methodName+" "+descriptor+" "+argString.toString()+" "+pc;
99             }
100     
101         }
