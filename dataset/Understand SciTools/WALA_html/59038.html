63     /**
64      * This class constructs an SSA {@link IR} from a backing ShrikeBT instruction stream.
65      * 
66      * The basic algorithm here is an abstract interpretation over the Java bytecode to determine types of stack locations and local
67      * variables. As a side effect, the flow functions of the abstract interpretation emit instructions, eliminating the stack
68      * abstraction and moving to a register-transfer language in SSA form.
69      */
70     public class SSABuilder extends AbstractIntStackMachine {

72       public static SSABuilder make(IBytecodeMethod method, SSACFG cfg, ShrikeCFG scfg, SSAInstruction[] instructions,
73           SymbolTable symbolTable, boolean buildLocalMap, SSAPiNodePolicy piNodePolicy) throws IllegalArgumentException {
74         if (scfg == null) {
75           throw new IllegalArgumentException("scfg == null");
76         }
77         return new SSABuilder(method, cfg, scfg, instructions, symbolTable, buildLocalMap, piNodePolicy);
78       }

136         @Override
137         public int meetStack(int slot, int[] rhs, BasicBlock bb) {
138     
139           assert bb != null : "null basic block";
140     
141           if (bb.isExitBlock()) {
142             return TOP;
143           }
144     
145           if (allTheSame(rhs)) {
146             for (int i = 0; i < rhs.length; i++) {
147               if (rhs[i] != TOP) {
148                 return rhs[i];
149               }
150             }
151             // didn't find anything but TOP
152             return TOP;
153           } else {
154             SSACFG.BasicBlock newBB = cfg.getNode(shrikeCFG.getNumber(bb));
155             // if we already have a phi for this stack location
156             SSAPhiInstruction phi = newBB.getPhiForStackSlot(slot);
157             int result;
158             if (phi == null) {
159               // no phi already exists. create one.
160               result = symbolTable.newPhi(rhs);
161               PhiValue v = symbolTable.getPhiValue(result);
162               phi = v.getPhiInstruction();
163               newBB.addPhiForStackSlot(slot, phi);
164             } else {
165               // already created a phi. update it to account for the
166               // new merge.
167               result = phi.getDef();
168               phi.setValues(rhs.clone());
169             }
170             return result;
171           }
172         }

174         /**
175          * @see com.ibm.wala.analysis.stackMachine.AbstractIntStackMachine.Meeter#meetLocal(int, int[], BasicBlock)
176          */
177         @Override
178         public int meetLocal(int n, int[] rhs, BasicBlock bb) {
179           if (allTheSame(rhs)) {
180             for (int i = 0; i < rhs.length; i++) {
181               if (rhs[i] != TOP) {
182                 return rhs[i];
183               }
184             }
185             // didn't find anything but TOP
186             return TOP;
187           } else {
188             SSACFG.BasicBlock newBB = cfg.getNode(shrikeCFG.getNumber(bb));
189             if (bb.isExitBlock()) {
190               // no phis in exit block please
191               return TOP;
192             }
193             // if we already have a phi for this local
194             SSAPhiInstruction phi = newBB.getPhiForLocal(n);
195             int result;
196             if (phi == null) {
197               // no phi already exists. create one.
198               result = symbolTable.newPhi(rhs);
199               PhiValue v = symbolTable.getPhiValue(result);
200               phi = v.getPhiInstruction();
201               newBB.addPhiForLocal(n, phi);
202             } else {
203               // already created a phi. update it to account for the
204               // new merge.
205               result = phi.getDef();
206               phi.setValues(rhs.clone());
207             }
208             return result;
209           }
210         }

212         /**
213          * Are all rhs values all the same? Note, we consider TOP (-1) to be same as everything else.
214          * 
215          * @param rhs
216          * @return boolean
217          */
218         private boolean allTheSame(int[] rhs) {
219           int x = -1;
220           // set x := the first non-TOP value
221           int i = 0;
222           for (i = 0; i < rhs.length; i++) {
223             if (rhs[i] != TOP) {
224               x = rhs[i];
225               break;
226             }
227           }
228           // check the remaining values
229           for (i = i + 1; i < rhs.length; i++) {
230             if (rhs[i] != x && rhs[i] != TOP)
231               return false;
232           }
233           return true;
234         }

236         /**
237          * @see com.ibm.wala.analysis.stackMachine.AbstractIntStackMachine.Meeter#meetStackAtCatchBlock(BasicBlock)
238          */
239         @Override
240         public int meetStackAtCatchBlock(BasicBlock bb) {
241           int bbNumber = shrikeCFG.getNumber(bb);
242           SSACFG.ExceptionHandlerBasicBlock newBB = (SSACFG.ExceptionHandlerBasicBlock) cfg.getNode(bbNumber);
243           SSAGetCaughtExceptionInstruction s = newBB.getCatchInstruction();
244           int exceptionValue;
245           if (s == null) {
246             exceptionValue = symbolTable.newSymbol();
247             s = insts.GetCaughtExceptionInstruction(SSAInstruction.NO_INDEX, bbNumber, exceptionValue);
248             newBB.setCatchInstruction(s);
249           } else {
250             exceptionValue = s.getException();
251           }
252           return exceptionValue;
253     
254         }

257       @Override
258       protected void initializeVariables() {
259         MachineState entryState = getEntryState();
260         int parameterNumber = 0;
261         int local = -1;
262         for (int i = 0; i < method.getNumberOfParameters(); i++) {
263           local++;
264           TypeReference t = method.getParameterType(i);
265           if (t != null) {
266             int symbol = symbolTable.getParameter(parameterNumber++);
267             entryState.setLocal(local, symbol);
268             if (t.equals(TypeReference.Double) || t.equals(TypeReference.Long)) {
269               local++;
270             }
271           }
272         }
273     
274         // This useless value ensures that the state cannot be empty, even
275         // for a static method with no arguments in blocks with an empty stack
276         // and no locals being used. This ensures that propagation of the
277         // state thru the CFGSystem will always show changes the first time
278         // it reaches a block, and thus no piece of the CFG will be skipped.
279         //
280         // (note that this bizarre state really happened, in java_cup)
281         //
282         entryState.push(symbolTable.newSymbol());
283       }

309         public SymbolicPropagator(ShrikeCFG shrikeCFG, SSAInstruction[] instructions, SymbolTable symbolTable, SSA2LocalMap localMap,
310             SSACFG cfg, SSAPiNodePolicy piNodePolicy) {
311           super(shrikeCFG);
312           this.piNodePolicy = piNodePolicy;
313           this.cfg = cfg;
314           this.creators = new SSAInstruction[0];
315           this.shrikeCFG = shrikeCFG;
316           this.instructions = instructions;
317           this.symbolTable = symbolTable;
318           this.loader = shrikeCFG.getMethod().getDeclaringClass().getClassLoader().getReference();
319           this.localMap = localMap;

323         @Override
324         public boolean needsEdgeFlow() {
325           return piNodePolicy != null;
326         }

328         private void emitInstruction(SSAInstruction s) {
329           if (s != null) {
330             instructions[getCurrentInstructionIndex()] = s;
331             for (int i = 0; i < s.getNumberOfDefs(); i++) {
332               if (creators.length < (s.getDef(i) + 1)) {
333                 SSAInstruction[] arr = new SSAInstruction[2 * s.getDef(i)];
334                 System.arraycopy(creators, 0, arr, 0, creators.length);
335                 creators = arr;
336               }
337     
338               assert s.getDef(i) != -1 : "invalid def " + i + " for " + s;
339             
340               creators[s.getDef(i)] = s;
341             }
342           }
343         }
344     
345         private SSAInstruction getCurrentInstruction() {
346           return instructions[getCurrentInstructionIndex()];
347         }
348     
349         /**
350          * If we've already created the current instruction, return the value number def'ed by the current instruction. Else, create a
351          * new symbol.
352          */
353         private int reuseOrCreateDef() {
354           if (getCurrentInstruction() == null) {
355             return symbolTable.newSymbol();
356           } else {
357             return getCurrentInstruction().getDef();
358           }
359         }
360     
361         /**
362          * If we've already created the current instruction, return the value number representing the exception the instruction may
363          * throw. Else, create a new symbol
364          */
365         private int reuseOrCreateException() {
366     
367           if (getCurrentInstruction() != null) {
368             assert getCurrentInstruction() instanceof SSAInvokeInstruction;
369           }
370           if (getCurrentInstruction() == null) {
371             return symbolTable.newSymbol();
372           } else {
373             SSAInvokeInstruction s = (SSAInvokeInstruction) getCurrentInstruction();
374             return s.getException();
375           }
376         }

483           /**
484            * @see com.ibm.wala.shrikeBT.IInstruction.Visitor#visitConstant(ConstantInstruction)
485            */
486           @Override
487           public void visitConstant(com.ibm.wala.shrikeBT.ConstantInstruction instruction) {
488             Language l = cfg.getMethod().getDeclaringClass().getClassLoader().getLanguage();
489             
490             TypeReference type = l.getConstantType(instruction.getValue());
491             int symbol = 0;
492             if (l.isNullType(type)) {
493               symbol = symbolTable.getNullConstant();
494             } else if (l.isIntType(type)) {
495               Integer value = (Integer) instruction.getValue();
496               symbol = symbolTable.getConstant(value.intValue());
497             } else if (l.isLongType(type)) {
498               Long value = (Long) instruction.getValue();
499               symbol = symbolTable.getConstant(value.longValue());
500             } else if (l.isFloatType(type)) {
501               Float value = (Float) instruction.getValue();
502               symbol = symbolTable.getConstant(value.floatValue());
503             } else if (l.isDoubleType(type)) {
504               Double value = (Double) instruction.getValue();
505               symbol = symbolTable.getConstant(value.doubleValue());
506             } else if (l.isStringType(type)) {
507               String value = (String) instruction.getValue();
508               symbol = symbolTable.getConstant(value);
509             } else if (l.isMetadataType(type)) {
510               Object rval = l.getMetadataToken(instruction.getValue());
511               symbol = reuseOrCreateDef();
512               emitInstruction(insts.LoadMetadataInstruction(getCurrentInstructionIndex(), symbol, type, rval));
513             } else {
514               Assertions.UNREACHABLE("unexpected " + type);
515             }
516             workingState.push(symbol);
517           }

577            * @see com.ibm.wala.shrikeBT.IInstruction.Visitor#visitInvoke(IInvokeInstruction)
578            */
579           @Override
580           public void visitInvoke(IInvokeInstruction instruction) {
581             doIndirectReads(bytecodeIndirections.indirectlyReadLocals(getCurrentInstructionIndex()));
582             int n = instruction.getPoppedCount();
583             int[] params = new int[n];
584             for (int i = n - 1; i >= 0; i--) {
585               params[i] = workingState.pop();
586             }
587             Language lang = shrikeCFG.getMethod().getDeclaringClass().getClassLoader().getLanguage();
588             MethodReference m = MethodReference.findOrCreate(lang, loader, instruction.getClassType(), instruction.getMethodName(),
589                 instruction.getMethodSignature());
590             IInvokeInstruction.IDispatch code = instruction.getInvocationCode();
591             CallSiteReference site = CallSiteReference.make(getCurrentProgramCounter(), m, code);
592             int exc = reuseOrCreateException();
593             
594             BootstrapMethod bootstrap = null;
595             if (instruction instanceof InvokeDynamicInstruction) {
596               bootstrap = ((InvokeDynamicInstruction)instruction).getBootstrap();
597             }
598             
599             if (instruction.getPushedWordSize() > 0) {
600               int result = reuseOrCreateDef();
601               workingState.push(result);
602               emitInstruction(insts.InvokeInstruction(getCurrentInstructionIndex(), result, params, exc, site, bootstrap));
603             } else {
604               emitInstruction(insts.InvokeInstruction(getCurrentInstructionIndex(), params, exc, site, bootstrap));
605             }
606             doIndirectWrites(bytecodeIndirections.indirectlyWrittenLocals(getCurrentInstructionIndex()), -1);
607           }

609           @Override
610           public void visitLocalLoad(ILoadInstruction instruction) {
611             if (instruction.isAddressOf()) {

621               emitInstruction(insts.AddressOfInstruction(getCurrentInstructionIndex(), result, t, type));
622               workingState.push(result);
623             } else {
624               super.visitLocalLoad(instruction);
625             }
626           }

726           private int findRethrowException() {
727             int index = getCurrentInstructionIndex();
728             SSACFG.BasicBlock bb = cfg.getBlockForInstruction(index);
729             if (bb.isCatchBlock()) {
730               SSACFG.ExceptionHandlerBasicBlock newBB = (SSACFG.ExceptionHandlerBasicBlock) bb;
731               SSAGetCaughtExceptionInstruction s = newBB.getCatchInstruction();
732               return s.getDef();
733             } else {
734               // TODO: should we really use dominators here? maybe it would be cleaner to propagate
735               // the notion of 'current exception to rethrow' using the abstract interpreter.
736               if (dom == null) {
737                 dom = Dominators.make(cfg, cfg.entry());
738               }
739     
740               ISSABasicBlock x = bb;
741               while (x != null) {
742                 if (x.isCatchBlock()) {
743                   SSACFG.ExceptionHandlerBasicBlock newBB = (SSACFG.ExceptionHandlerBasicBlock) x;
744                   SSAGetCaughtExceptionInstruction s = newBB.getCatchInstruction();
745                   return s.getDef();
746                 } else {
747                   x = dom.getIdom(x);
748                 }
749               }
750     
751               // assert false;
752               return -1;
753             }
754           }

825         private void reuseOrCreatePi(SSAInstruction piCause, int ref) {
826           int n = getCurrentInstructionIndex();
827           SSACFG.BasicBlock bb = cfg.getBlockForInstruction(n);
828     
829           BasicBlock path = getCurrentSuccessor();
830           int outNum = shrikeCFG.getNumber(path);

839         }

924       /**
925        * A logical mapping from <pc, valueNumber> -> local number Note: make sure this class remains static: this persists as part of
926        * the IR!!
927        */
928       private static class SSA2LocalMap implements com.ibm.wala.ssa.IR.SSA2LocalMap {
929     
930         private final ShrikeCFG shrikeCFG;
931     
932         /**
933          * Mapping Integer -> IntPair where p maps to (vn,L) iff we've started a range at pc p where value number vn corresponds to
934          * local L
935          */
936         private final IntPair[] localStoreMap;
937     
938         /**
939          * For each basic block i and local j, block2LocalState[i][j] gives the contents of local j at the start of block i
940          */
941         private final int[][] block2LocalState;
942     
943         /**
944          * @param nInstructions number of instructions in the bytecode for this method
945          * @param nBlocks number of basic blocks in the CFG
946          */
947         SSA2LocalMap(ShrikeCFG shrikeCfg, int nInstructions, int nBlocks) {
948           shrikeCFG = shrikeCfg;
949           localStoreMap = new IntPair[nInstructions];
950           block2LocalState = new int[nBlocks][];
951         }
952     
953         /**
954          * Record the beginning of a new range, starting at the given program counter, in which a particular value number corresponds to
955          * a particular local number
956          */
957         void startRange(int pc, int localNumber, int valueNumber) {
958           localStoreMap[pc] = new IntPair(valueNumber, localNumber);
959         }
960     
961         /**
962          * Finish populating the map of local variable information
963          */
964         private void finishLocalMap(SSABuilder builder) {
965           for (Iterator it = shrikeCFG.iterator(); it.hasNext();) {
966             ShrikeCFG.BasicBlock bb = (ShrikeCFG.BasicBlock) it.next();
967             MachineState S = builder.getIn(bb);
968             int number = bb.getNumber();
969             block2LocalState[number] = S.getLocals();
970           }
971         }
972     
973         /**
974          * @param index - index into IR instruction array
975          * @param vn - value number
976          */
977         @Override
978         public String[] getLocalNames(int index, int vn) {
979           try {
980             if (!shrikeCFG.getMethod().hasLocalVariableTable()) {
981               return null;
982             } else {
983               int[] localNumbers = findLocalsForValueNumber(index, vn);
984               if (localNumbers == null) {
985                 return null;
986               } else {
987                 IBytecodeMethod m = shrikeCFG.getMethod();
988                 String[] result = new String[localNumbers.length];
989                 for (int i = 0; i < localNumbers.length; i++) {
990                   result[i] = m.getLocalVariableName(m.getBytecodeIndex(index), localNumbers[i]);
991                 }
992                 return result;
993               }
994             }
995           } catch (Exception e) {
996             e.printStackTrace();
997             Assertions.UNREACHABLE();
998             return null;
999           }
1000         }

1002         public int[] allocateNewLocalsArray(int maxLocals) {
1003           int[] result = new int[maxLocals];
1004           for (int i = 0; i < maxLocals; i++) {
1005             result[i] = OPTIMISTIC ? TOP : BOTTOM;
1006           }
1007           return result;
1008         }
1009         
1010         private int[] setLocal(int[] locals, int localNumber, int valueNumber) {
1011           if (locals == null) {
1012             locals = allocateNewLocalsArray(localNumber + 1);
1013           } else if (locals.length <= localNumber) {
1014             int[] newLocals = allocateNewLocalsArray(2 * Math.max(locals.length, localNumber) + 1);
1015             System.arraycopy(locals, 0, newLocals, 0, locals.length);
1016             locals = newLocals;
1017           }
1018           
1019           locals[localNumber] = valueNumber;
1020           
1021           return locals;
1022         }
1023         /**
1024          * @param pc a program counter (index into ShrikeBT instruction array)
1025          * @param vn a value number
1026          * @return if we know that immediately after the given program counter, v_vn corresponds to some set of locals, then return an
1027          *         array of the local numbers. else return null.
1028          */
1029         private int[] findLocalsForValueNumber(int pc, int vn) {
1030           if (vn < 0) {
1031             return null;
1032           }
1033           IBasicBlock bb = shrikeCFG.getBlockForInstruction(pc);
1034           int firstInstruction = bb.getFirstInstructionIndex();
1035           // walk forward from the first instruction to reconstruct the
1036           // state of the locals at this pc
1037           int[] locals = block2LocalState[bb.getNumber()];
1038           for (int i = firstInstruction; i <= pc; i++) {
1039             if (localStoreMap[i] != null) {
1040               IntPair p = localStoreMap[i];
1041               locals = setLocal(locals, p.getY(), p.getX());
1042             }
1043           }
1044           return locals == null ? null : extractIndices(locals, vn);
1045         }
1046     
1047         /**
1048          * @return the indices i s.t. x[i] == y, or null if none found.
1049          */
1050         private int[] extractIndices(int[] x, int y) {
1051           assert x != null;
1052           int count = 0;
1053           for (int i = 0; i < x.length; i++) {
1054             if (x[i] == y) {
1055               count++;
1056             }
1057           }
1058           if (count == 0) {
1059             return null;
1060           } else {
1061             int[] result = new int[count];
1062             int j = 0;
1063             for (int i = 0; i < x.length; i++) {
1064               if (x[i] == y) {
1065                 result[j++] = i;
1066               }
1067             }
1068             return result;
1069           }
1070         }
