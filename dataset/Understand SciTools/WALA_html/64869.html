11     package com.ibm.wala.cast.util;
12     
13     import javax.xml.parsers.DocumentBuilder;
14     import javax.xml.parsers.DocumentBuilderFactory;
15     import javax.xml.parsers.ParserConfigurationException;
16     
17     import org.w3c.dom.DOMImplementation;
18     import org.w3c.dom.Document;
19     import org.w3c.dom.Element;
20     import org.w3c.dom.Text;
21     
22     import com.ibm.wala.cast.tree.CAstNode;
23     
24     public class CAstToDOM extends CAstPrinter {
25     
26       private static final String VALUE_TAG = "value";
27       private static final String TYPE_TAG = "type";
28     
29       public static Document toDOM(CAstNode astRoot) {
30         try {
31           DocumentBuilderFactory documentBuilderFactory =
32     	DocumentBuilderFactory.newInstance();
33           DocumentBuilder documentBuilder = 
34     	documentBuilderFactory.newDocumentBuilder();
35           DOMImplementation domImplementation = 
36     	documentBuilder.getDOMImplementation();
37           Document document =
38     	domImplementation.createDocument("CAst", "CAst", null);
39           Element rootNode =
40     	document.getDocumentElement();
41     
42           nodeToDOM(document, rootNode, astRoot);
43     
44           return document;
45         } catch (ParserConfigurationException e) {
46           e.printStackTrace();
47           throw new RuntimeException("DOM builder error.");
48         }
49       }
50     
51       private static void nodeToDOM(Document doc, Element root, CAstNode astNode) {
52         Element nodeElt = doc.createElement(kindAsString(astNode.getKind()));
53     
54         if (astNode.getValue() == null) {
55           for(int i = 0; i < astNode.getChildCount(); i++) {
56     	nodeToDOM(doc, nodeElt, astNode.getChild(i));
57           }
58     
59         } else {
60           Element typeTag = doc.createElement(TYPE_TAG);
61           Text type = doc.createTextNode(astNode.getValue().getClass().toString());
62           typeTag.appendChild( type );
63           nodeElt.appendChild(typeTag);
64     
65           Element valueTag = doc.createElement(VALUE_TAG);
66           Text value = doc.createTextNode( astNode.getValue().toString() );
67           valueTag.appendChild( value );
68           nodeElt.appendChild(valueTag);
69         }
70     
71         root.appendChild(nodeElt);
72       }
73     
74     }
