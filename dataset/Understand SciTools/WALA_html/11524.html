98       private static void runAnalysisForTestCase(String mainClass) throws IllegalArgumentException, CancelException, IOException {
99         System.err.println("=======---------------=============");
100         System.err.println(("ANALYZING " + mainClass + "\n\n"));
101         // describe the "scope", what is the program we're analyzing
102         AnalysisScope scope = CallGraphTestUtil.makeJ2SEAnalysisScope(TestInfo.SCOPE_FILE, CallGraphTestUtil.REGRESSION_EXCLUSIONS);
103     
104         // build a type hierarchy
105         ClassHierarchy cha = null;
106         try {
107           cha = ClassHierarchy.make(scope);
108         } catch (ClassHierarchyException e) {
109           // TODO Auto-generated catch block
110           e.printStackTrace();
111         }
112     
113         // set up call graph construction options; mainly what should be considered
114         // entrypoints?
115         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha, mainClass);
116         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
117     
118         // build an RTA call graph
119         CallGraphBuilder rtaBuilder = Util.makeRTABuilder(options, new AnalysisCache(), cha, scope);
120         final CallGraph cg = rtaBuilder.makeCallGraph(options, null);
121         // System.err.println(cg.toString());
122     
123         MemoryAccessMap fam = new SimpleMemoryAccessMap(cg, rtaBuilder.getPointerAnalysis().getHeapModel(), false);
124         // System.err.println(fam.toString());
125     
126         IDemandPointerAnalysis dmp = makeDemandPointerAnalysis(options, cha, scope, cg, fam);
127     
128         IDemandPointerAnalysis simpleDmp = new SimpleDemandPointsTo(cg, dmp.getHeapModel(), fam, cha, options);
129         CGNode main = cg.getEntrypointNodes().iterator().next();
130     
131         IR ir = main.getIR();
132         TypeInference ti = TypeInference.make(ir, false);
133         for (int i = 1; i <= ir.getSymbolTable().getMaxValueNumber(); i++) {
134           TypeAbstraction t = ti.getType(i);
135           if (t != null) {
136             LocalPointerKey v = (LocalPointerKey) dmp.getHeapModel().getPointerKeyForLocal(main, i);
137             Collection<InstanceKey> p = dmp.getPointsTo(v);
138             Collection<InstanceKey> oldP = simpleDmp.getPointsTo(v);
139             if (!sameContents(p, oldP)) {
140               System.err.println(("different result for " + v));
141               System.err.println(("old " + oldP + "\n\nnew " + p));
142             }
143             printResult(v, p);
144           }
145         }
146       }
147     
148       private static <T> boolean sameContents(Collection<T> c1, Collection<T> c2) {
149         return c1.containsAll(c2) && c2.containsAll(c1);
150       }
151     
152       private static void printResult(PointerKey pk, Collection<InstanceKey> result) {
153         if (VERBOSE) {
154           System.err.println("points-to for " + pk);
155           if (result.isEmpty()) {
156             System.err.println("  EMPTY!");
157           }
158           for (Iterator<InstanceKey> it = result.iterator(); it.hasNext();) {
159             System.err.println("  " + it.next());
160           }
161         }
162       }
163     
164       private static IDemandPointerAnalysis makeDemandPointerAnalysis(AnalysisOptions options, ClassHierarchy cha, AnalysisScope scope,
165           CallGraph cg, MemoryAccessMap fam) {
166         SSAPropagationCallGraphBuilder builder = Util.makeVanillaZeroOneCFABuilder(options, new AnalysisCache(), cha, scope);
167         // return new TestNewGraphPointsTo(cg, builder, fam, cha, warnings);
168         DemandRefinementPointsTo fullDemandPointsTo = DemandRefinementPointsTo.makeWithDefaultFlowGraph(cg, builder, fam, cha, options, new DummyStateMachine.Factory<IFlowLabel>());
169         // fullDemandPointsTo.setCGRefinePolicy(new AlwaysRefineCGPolicy());
170         // fullDemandPointsTo.setFieldRefinePolicy(new AlwaysRefineFieldsPolicy());
171         fullDemandPointsTo.setRefinementPolicyFactory(new SinglePassRefinementPolicy.Factory(new AlwaysRefineFieldsPolicy(),
172             new AlwaysRefineCGPolicy()));
173         return fullDemandPointsTo;
174       }
