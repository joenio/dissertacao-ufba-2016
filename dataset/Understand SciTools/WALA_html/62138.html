11     package com.ibm.wala.util.intset;
12     
13     import java.util.Iterator;
14     import java.util.TreeSet;
15     
16     /**
17      * An object that creates mutable sparse int sets.
18      */
19     public class MutableSparseIntSetFactory implements MutableIntSetFactory<MutableSparseIntSet> {
20     
21       /**
22        * @throws IllegalArgumentException  if set is null
23        */
24       @Override
25       public MutableSparseIntSet make(int[] set) {
26         if (set == null) {
27           throw new IllegalArgumentException("set is null");
28         }
29         if (set.length == 0) {
30           return MutableSparseIntSet.makeEmpty();
31         } else {
32           // XXX not very efficient.
33           TreeSet<Integer> T = new TreeSet<Integer>();
34           for (int i = 0; i < set.length; i++) {
35             T.add(new Integer(set[i]));
36           }
37           int[] copy = new int[T.size()];
38           int i = 0;
39           for (Iterator<Integer> it = T.iterator(); it.hasNext();) {
40             Integer I = it.next();
41             copy[i++] = I.intValue();
42           }
43           MutableSparseIntSet result = new MutableSparseIntSet(copy);
44           return result;
45         }
46       }
47     
48       /**
49        * @param string
50        */
51       @Override
52       public MutableSparseIntSet parse(String string) throws NumberFormatException {
53         int[] backingStore = SparseIntSet.parseIntArray(string);
54         return new MutableSparseIntSet(backingStore);
55       }
56     
57       /*
58        * @see com.ibm.wala.util.intset.MutableIntSetFactory#make(com.ibm.wala.util.intset.IntSet)
59        */
60       @Override
61       public MutableSparseIntSet makeCopy(IntSet x) throws IllegalArgumentException {
62         if (x == null) {
63           throw new IllegalArgumentException("x == null");
64         }
65         return MutableSparseIntSet.make(x);
66       }
67     
68       /*
69        * @see com.ibm.wala.util.intset.MutableIntSetFactory#make()
70        */
71       @Override
72       public MutableSparseIntSet make() {
73         return MutableSparseIntSet.makeEmpty();
74       }
75     
76     }
