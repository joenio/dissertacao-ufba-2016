48       public interface WalkContext<C extends WalkContext<C, T>, T> {
49     
50         /**
51          * get a mapping from CAstNodes to the scoped entities (e.g. functions or
52          * local classes) introduced by those nodes. Also maps <code>null</code> to
53          * those entities not corresponding to any node (e.g nested classes)
54          */
55         Map<CAstNode, Collection<CAstEntity>> getScopedEntities();
56     
57         /**
58          *   associate a child entity with a given CAstNode, e.g. for a function declaration
59          */
60         void addScopedEntity(CAstNode newNode, CAstEntity visit);
61     
62         /**
63          * for recording control-flow relationships among the CAst nodes
64          */
65         CAstControlFlowRecorder cfg();
66     
67         /**
68          * for recording source positions
69          */
70         CAstSourcePositionRecorder pos();
71     
72         /**
73          * for recording types of nodes
74          */
75         CAstNodeTypeMapRecorder getNodeTypeMap();
76     
77         /**
78          * for a 'continue' style goto, return the control flow target
79          */
80         T getContinueFor(String label);
81     
82         /**
83          * for a 'break' style goto, return the control flow target
84          */
85         T getBreakFor(String label);
86     
87       }

177       public static class DoLoopTranslator {
178         private final boolean replicateForDoLoops;
179         
180         private final CAst Ast;
181         
182         public DoLoopTranslator(boolean replicateForDoLoops, CAst ast) {
183           this.replicateForDoLoops = replicateForDoLoops;
184           Ast = ast;
185         }
186     
187         public CAstNode translateDoLoop(CAstNode loopTest, CAstNode loopBody, CAstNode continueNode, CAstNode breakNode, WalkContext<?,?> wc) {      
188           if (replicateForDoLoops) {
189             loopBody = Ast.makeNode(CAstNode.BLOCK_STMT, loopBody, continueNode);
190             
191             CAstRewriter.Rewrite x = (new CAstCloner(Ast, false)).copy(loopBody, wc.cfg(), wc.pos(), wc.getNodeTypeMap(), null);
192             CAstNode otherBody = x.newRoot();
193             
194             wc.cfg().addAll(x.newCfg());
195             wc.pos().addAll(x.newPos());
196             wc.getNodeTypeMap().addAll(x.newTypes());
197      
198             return Ast.makeNode(CAstNode.BLOCK_STMT, 
199                 loopBody,
200                 Ast.makeNode(CAstNode.LOOP, loopTest, otherBody),
201                 breakNode);
202             
203           } else {
204             CAstNode header = Ast.makeNode(CAstNode.LABEL_STMT, Ast.makeConstant("_do_label"), Ast.makeNode(CAstNode.EMPTY));
205             CAstNode loopGoto = Ast.makeNode(CAstNode.IFGOTO, loopTest);
206     
207             wc.cfg().map(header, header);
208             wc.cfg().map(loopGoto, loopGoto);
209             wc.cfg().add(loopGoto, header, Boolean.TRUE);
210     
211             return Ast.makeNode(CAstNode.BLOCK_STMT, 
212                 header, 
213                 Ast.makeNode(CAstNode.BLOCK_STMT, loopBody, continueNode), 
214                 loopGoto, 
215                 breakNode);
216           }
217         }
218         };
