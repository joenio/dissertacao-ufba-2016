173       private void populate() {
174         if (!isPopulated) {
175           // ensure that we keep the single, canonical IR live throughout initialization, while the instructionIndices map
176           // is live.
177           IR ir = node.getIR();
178           isPopulated = true;
179     
180           Map<SSAInstruction, Integer> instructionIndices = computeInstructionIndices(ir);
181           createNodes(ref, cOptions, ir);
182           createScalarEdges(cOptions, ir, instructionIndices);
183         }
184       }

191       /**
192        * return the set of all PARAM_CALLER and HEAP_PARAM_CALLER statements associated with a given call
193        */
194       public Set<Statement> getCallerParamStatements(SSAAbstractInvokeInstruction call) throws IllegalArgumentException {
195         if (call == null) {
196           throw new IllegalArgumentException("call == null");
197         }
198         populate();
199         return callerParamStatements.get(call.getCallSite());
200       }
201     
202       /**
203        * return the set of all PARAM_CALLER, HEAP_PARAM_CALLER, and NORMAL statements (i.e., the actual call statement) associated with
204        * a given call
205        */
206       public Set<Statement> getCallStatements(SSAAbstractInvokeInstruction call) throws IllegalArgumentException {
207         Set<Statement> callerParamStatements = getCallerParamStatements(call);
208         Set<Statement> result = HashSetFactory.make(callerParamStatements.size() + 1);
209         result.addAll(callerParamStatements);
210         result.add(callSite2Statement.get(call.getCallSite()));
211         return result;
212       }

225       /**
226        * Create all control dependence edges in this PDG.
227        */
228       private void createControlDependenceEdges(ControlDependenceOptions cOptions, IR ir,
229           Map<SSAInstruction, Integer> instructionIndices) {
230         if (cOptions.equals(ControlDependenceOptions.NONE)) {
231           return;
232         }
233         if (ir == null) {
234           return;
235         }
236         ControlFlowGraph<SSAInstruction, ISSABasicBlock> controlFlowGraph = ir.getControlFlowGraph();
237         if (cOptions.equals(ControlDependenceOptions.NO_EXCEPTIONAL_EDGES)) {
238           PrunedCFG<SSAInstruction, ISSABasicBlock> prunedCFG = ExceptionPrunedCFG.make(controlFlowGraph);
239           // In case the CFG has only the entry and exit nodes left 
240           // and no edges because the only control dependencies
241           // were exceptional, simply return because at this point there are no nodes.
242           // Otherwise, later this may raise an Exception.
243           if (prunedCFG.getNumberOfNodes() == 2 
244               && prunedCFG.containsNode(controlFlowGraph.entry()) 
245               && prunedCFG.containsNode(controlFlowGraph.exit())
246               && GraphUtil.countEdges(prunedCFG) == 0) {
247             return;
248           }
249           controlFlowGraph = prunedCFG;
250         } else {
251           Assertions.productionAssertion(cOptions.equals(ControlDependenceOptions.FULL));
252         }
253     
254         ControlDependenceGraph<SSAInstruction, ISSABasicBlock> cdg = new ControlDependenceGraph<SSAInstruction, ISSABasicBlock>(
255             controlFlowGraph);
256         for (ISSABasicBlock bb : cdg) {
257           if (bb.isExitBlock()) {
258             // nothing should be control-dependent on the exit block.
259             continue;
260           }
261     
262           Statement src = null;
263           if (bb.isEntryBlock()) {
264             src = new MethodEntryStatement(node);
265           } else {
266             SSAInstruction s = ir.getInstructions()[bb.getLastInstructionIndex()];
267             if (s == null) {
268               // should have no control dependent successors.
269               // leave src null.
270             } else {
271               src = ssaInstruction2Statement(s, ir, instructionIndices);
272               // add edges from call statements to parameter passing and return
273               // SJF: Alexey and I think that we should just define ParamStatements
274               // as
275               // being control dependent on nothing ... they only represent pure
276               // data dependence. So, I'm commenting out the following.
277               // if (s instanceof SSAAbstractInvokeInstruction) {
278               // SSAAbstractInvokeInstruction call = (SSAAbstractInvokeInstruction)
279               // s;
280               // for (Statement st : callerParamStatements.get(call.getCallSite()))
281               // {
282               // addEdge(src, st);
283               // }
284               // for (Statement st : callerReturnStatements.get(call.getCallSite()))
285               // {
286               // addEdge(src, st);
287               // }
288               // }
289             }
290           }
291           // add edges for every control-dependent statement in the IR, if there are
292           // any
293           // control-dependent successors
294           if (src != null) {
295             for (Iterator<? extends ISSABasicBlock> succ = cdg.getSuccNodes(bb); succ.hasNext();) {
296               ISSABasicBlock bb2 = succ.next();
297               for (Iterator<SSAInstruction> it2 = bb2.iterator(); it2.hasNext();) {
298                 SSAInstruction st = it2.next();
299                 if (st != null) {
300                   Statement dest = ssaInstruction2Statement(st, ir, instructionIndices);
301                   assert src != null;
302                   delegate.addEdge(src, dest);
303     /** BEGIN Custom change: control deps */                
304                   delegate.addEdge(src, dest, Dependency.CONTROL_DEP);
305     /** END Custom change: control deps */                
306                 }
307               }
308             }
309           }
310         }
311     
312         // the CDG does not represent control dependences from the entry node.
313         // add these manually
314         // We add control dependences to all instructions in all basic blocks B that _must_ execute.
315         // B is the set of blocks that dominate the exit basic block
316         Statement methodEntry = new MethodEntryStatement(node);
317         Dominators<ISSABasicBlock> dom = Dominators.make(controlFlowGraph, controlFlowGraph.entry());
318         for (ISSABasicBlock exitDom : Iterator2Iterable.make(dom.dominators(controlFlowGraph.exit()))) {
319           for (SSAInstruction st : exitDom) {
320             Statement dest = ssaInstruction2Statement(st, ir, instructionIndices);
321             delegate.addEdge(methodEntry, dest);
322     /** BEGIN Custom change: control deps */                
323             delegate.addEdge(methodEntry, dest, Dependency.CONTROL_DEP);
324     /** END Custom change: control deps */                
325           }
326         }
327         // add CD from method entry to all callee parameter assignments
328         // SJF: Alexey and I think that we should just define ParamStatements as
329         // being control dependent on nothing ... they only represent pure
330         // data dependence. So, I'm commenting out the following.
331         // for (int i = 0; i < paramCalleeStatements.length; i++) {
332         // addEdge(methodEntry, paramCalleeStatements[i]);
333         // }
334     
335         /**
336          * JTD: While phi nodes live in a particular basic block, they represent a meet of values from multiple blocks. Hence, they are
337          * really like multiple statements that are control dependent in the manner of the predecessor blocks. When the slicer is
338          * following both data and control dependences, it therefore seems right to add control dependence edges to represent how a phi
339          * node depends on predecessor blocks.
340          */
341         if (!dOptions.equals(DataDependenceOptions.NONE)) {
342           for (ISSABasicBlock bb : cdg) {
343             for (Iterator<SSAPhiInstruction> ps = bb.iteratePhis(); ps.hasNext();) {
344               SSAPhiInstruction phi = ps.next();
345               Statement phiSt = ssaInstruction2Statement(phi, ir, instructionIndices);
346               int phiUseIndex = 0;
347               for (Iterator<? extends ISSABasicBlock> preds = controlFlowGraph.getPredNodes(bb); preds.hasNext();) {
348                 ISSABasicBlock pb = preds.next();
349                 int use = phi.getUse(phiUseIndex);
350                 if (use == AbstractIntStackMachine.TOP) {
351                   // the predecessor is part of some infeasible bytecode. we probably don't want slices to include such code, so ignore.
352                   continue;
353                 }
354                 if (controlFlowGraph.getSuccNodeCount(pb) > 1) {
355                   // in this case, there is more than one edge from the
356                   // predecessor block, hence the phi node actually
357                   // depends on the last instruction in the previous
358                   // block, rather than having the same dependences as
359                   // statements in that block.
360                   SSAInstruction pss = ir.getInstructions()[pb.getLastInstructionIndex()];
361                   assert pss != null;
362                   Statement pst = ssaInstruction2Statement(pss, ir, instructionIndices);
363                   delegate.addEdge(pst, phiSt);
364     /** BEGIN Custom change: control deps */                
365                   delegate.addEdge(pst, phiSt, Dependency.CONTROL_DEP);
366     /** END Custom change: control deps */                
367                 } else {
368                   for (Iterator<? extends ISSABasicBlock> cdps = cdg.getPredNodes(pb); cdps.hasNext();) {
369                     ISSABasicBlock cpb = cdps.next();
370     /** BEGIN Custom change: control deps */                
371                     if (cpb.getLastInstructionIndex() < 0) {
372                       continue;
373                     }
374     /** END Custom change: control deps */                
375                     SSAInstruction cps = ir.getInstructions()[cpb.getLastInstructionIndex()];
376                     assert cps != null : "unexpected null final instruction for CDG predecessor " + cpb + " in node " + node;
377                     Statement cpst = ssaInstruction2Statement(cps, ir, instructionIndices);
378                     delegate.addEdge(cpst, phiSt);
379     /** BEGIN Custom change: control deps */                
380                     delegate.addEdge(cpst, phiSt, Dependency.CONTROL_DEP);
381     /** END Custom change: control deps */                
382                   }
383                 }
384                 phiUseIndex++;
385               }
386             }
387           }
388         }
389       }

391       /**
392        * Create all data dependence edges in this PDG.
393        * 
394        * Scalar dependences are taken from SSA def-use information.
395        * 
396        * Heap dependences are computed by a reaching defs analysis.
397        * 
398        * @param pa
399        * @param mod
400        */
401       private void createScalarDataDependenceEdges(IR ir, Map<SSAInstruction, Integer> instructionIndices) {
402         if (dOptions.equals(DataDependenceOptions.NONE)) {
403           return;
404         }
405     
406         if (ir == null) {
407           return;
408         }
409     
410         // this is tricky .. I'm explicitly creating a new DefUse to make sure it refers to the instructions we need from
411         // the "one true" ir of the moment.
412         DefUse DU = new DefUse(ir);
413         SSAInstruction[] instructions = ir.getInstructions();
414     
415         //
416         // TODO: teach some other bit of code about the uses of
417         // GetCaughtException, and then delete this code.
418         //
419         if (!dOptions.isIgnoreExceptions()) {
420           for (ISSABasicBlock bb : ir.getControlFlowGraph()) {
421             if (bb.isCatchBlock()) {
422               SSACFG.ExceptionHandlerBasicBlock ehbb = (SSACFG.ExceptionHandlerBasicBlock) bb;
423     
424               if (ehbb.getCatchInstruction() != null) {
425                 Statement c = ssaInstruction2Statement(ehbb.getCatchInstruction(), ir, instructionIndices);
426     
427                 for (ISSABasicBlock pb : ir.getControlFlowGraph().getExceptionalPredecessors(ehbb)) {
428                   SSAInstruction st = instructions[pb.getLastInstructionIndex()];
429     
430                   if (st instanceof SSAAbstractInvokeInstruction) {
431                     delegate.addEdge(new ExceptionalReturnCaller(node, pb.getLastInstructionIndex()), c);
432                   } else if (st instanceof SSAAbstractThrowInstruction) {
433                     delegate.addEdge(ssaInstruction2Statement(st, ir, instructionIndices), c);
434                   }
435                 }
436               }
437             }
438           }
439         }
440     
441         for (Iterator<? extends Statement> it = iterator(); it.hasNext();) {
442           Statement s = it.next();
443           switch (s.getKind()) {
444           case NORMAL:
445           case CATCH:
446           case PI:
447           case PHI: {
448             SSAInstruction statement = statement2SSAInstruction(instructions, s);
449             // note that data dependencies from invoke instructions will pass
450             // interprocedurally
451             if (!(statement instanceof SSAAbstractInvokeInstruction)) {
452               if (dOptions.isTerminateAtCast() && (statement instanceof SSACheckCastInstruction)) {
453                 break;
454               }
455               if (dOptions.isTerminateAtCast() && (statement instanceof SSAInstanceofInstruction)) {
456                 break;
457               }
458               // add edges from this statement to every use of the def of this
459               // statement
460               for (int i = 0; i < statement.getNumberOfDefs(); i++) {
461                 int def = statement.getDef(i);
462                 for (Iterator<SSAInstruction> it2 = DU.getUses(def); it2.hasNext();) {
463                   SSAInstruction use = it2.next();
464                   if (dOptions.isIgnoreBasePtrs()) {
465                     if (use instanceof SSANewInstruction) {
466                       // cut out array length parameters
467                       continue;
468                     }
469                     if (hasBasePointer(use)) {
470                       int base = getBasePointer(use);
471                       if (def == base) {
472                         // skip the edge to the base pointer
473                         continue;
474                       }
475                       if (use instanceof SSAArrayReferenceInstruction) {
476                         SSAArrayReferenceInstruction arr = (SSAArrayReferenceInstruction) use;
477                         if (def == arr.getIndex()) {
478                           // skip the edge to the array index
479                           continue;
480                         }
481                       }
482                     }
483                   }
484                   Statement u = ssaInstruction2Statement(use, ir, instructionIndices);
485                   delegate.addEdge(s, u);
486                 }
487               }
488             }
489             break;
490           }
491           case EXC_RET_CALLER:
492           case NORMAL_RET_CALLER:
493           case PARAM_CALLEE: {
494             if (dOptions.isIgnoreExceptions()) {
495               assert !s.getKind().equals(Kind.EXC_RET_CALLER);
496             }
497     
498             ValueNumberCarrier a = (ValueNumberCarrier) s;
499             for (Iterator<SSAInstruction> it2 = DU.getUses(a.getValueNumber()); it2.hasNext();) {
500               SSAInstruction use = it2.next();
501               if (dOptions.isIgnoreBasePtrs()) {
502                 if (use instanceof SSANewInstruction) {
503                   // cut out array length parameters
504                   continue;
505                 }
506                 if (hasBasePointer(use)) {
507                   int base = getBasePointer(use);
508                   if (a.getValueNumber() == base) {
509                     // skip the edge to the base pointer
510                     continue;
511                   }
512                   if (use instanceof SSAArrayReferenceInstruction) {
513                     SSAArrayReferenceInstruction arr = (SSAArrayReferenceInstruction) use;
514                     if (a.getValueNumber() == arr.getIndex()) {
515                       // skip the edge to the array index
516                       continue;
517                     }
518                   }
519                 }
520               }
521               Statement u = ssaInstruction2Statement(use, ir, instructionIndices);
522               delegate.addEdge(s, u);
523             }
524             break;
525           }
526           case NORMAL_RET_CALLEE:
527             for (NormalStatement ret : computeReturnStatements(ir)) {
528               delegate.addEdge(ret, s);
529             }
530             break;
531           case EXC_RET_CALLEE:
532             if (dOptions.isIgnoreExceptions()) {
533               Assertions.UNREACHABLE();
534             }
535             // TODO: this is overly conservative. deal with catch blocks?
536             for (IntIterator ii = getPEIs(ir).intIterator(); ii.hasNext();) {
537               int index = ii.next();
538               SSAInstruction pei = ir.getInstructions()[index];
539               if (dOptions.isTerminateAtCast() && (pei instanceof SSACheckCastInstruction)) {
540                 continue;
541               }
542               if (pei instanceof SSAAbstractInvokeInstruction) {
543                 if (! dOptions.isIgnoreExceptions()) {
544                   Statement st = new ExceptionalReturnCaller(node, index);
545                   delegate.addEdge(st, s);
546                 }
547               } else {
548                 delegate.addEdge(new NormalStatement(node, index), s);
549               }
550             }
551             break;
552           case PARAM_CALLER: {
553             ParamCaller pac = (ParamCaller) s;
554             int vn = pac.getValueNumber();
555             // note that if the caller is the fake root method and the parameter
556             // type is primitive,
557             // it's possible to have a value number of -1. If so, just ignore it.
558             if (vn > -1) {
559               if (ir.getSymbolTable().isParameter(vn)) {
560                 Statement a = new ParamCallee(node, vn);
561                 delegate.addEdge(a, pac);
562               } else {
563                 SSAInstruction d = DU.getDef(vn);
564                 if (dOptions.isTerminateAtCast() && (d instanceof SSACheckCastInstruction)) {
565                   break;
566                 }
567                 if (d != null) {
568                   if (d instanceof SSAAbstractInvokeInstruction) {
569                     SSAAbstractInvokeInstruction call = (SSAAbstractInvokeInstruction) d;
570                     if (vn == call.getException()) {
571                       if (! dOptions.isIgnoreExceptions()) {
572                         Statement st = new ExceptionalReturnCaller(node, instructionIndices.get(d));
573                         delegate.addEdge(st, pac);
574                       }
575                     } else {
576                       Statement st = new NormalReturnCaller(node, instructionIndices.get(d));
577                       delegate.addEdge(st, pac);
578                     }
579                   } else {
580                     Statement ds = ssaInstruction2Statement(d, ir, instructionIndices);
581                     delegate.addEdge(ds, pac);
582                   }
583                 }
584               }
585             }
586           }
587             break;
588     
589           case HEAP_RET_CALLEE:
590           case HEAP_RET_CALLER:
591           case HEAP_PARAM_CALLER:
592           case HEAP_PARAM_CALLEE:
593           case METHOD_ENTRY:
594           case METHOD_EXIT:
595             // do nothing
596             break;
597           default:
598             Assertions.UNREACHABLE(s.toString());
599             break;
600           }
601         }
602       }

652       /**
653        * Create heap data dependence edges in this PDG relevant to a particular {@link PointerKey}.
654        */
655       private void createHeapDataDependenceEdges(final PointerKey pk) {
656     
657         if (locationsHandled.contains(pk)) {
658           return;
659         } else {
660           locationsHandled.add(pk);
661         }
662         if (dOptions.isIgnoreHeap() || (exclusions != null && exclusions.excludes(pk))) {
663           return;
664         }
665     
666         TypeReference t = HeapExclusions.getType(pk);
667         if (t == null) {
668           return;
669         }
670     
671         // It's OK to create a new IR here; we're not keeping any hashing live up to this point
672         IR ir = node.getIR();
673         if (ir == null) {
674           return;
675         }
676     
677         if (VERBOSE) {
678           System.err.println("Location " + pk);
679         }
680     
681         // in reaching defs calculation, exclude heap statements that are
682         // irrelevant.
683         Predicate f = new Predicate() {
684           @Override public boolean test(Object o) {
685             if (o instanceof HeapStatement) {
686               HeapStatement h = (HeapStatement) o;
687               return h.getLocation().equals(pk);
688             } else {
689               return true;
690             }
691           }
692         };
693         Collection<Statement> relevantStatements = Iterator2Collection.toSet(new FilterIterator<Statement>(iterator(), f));
694     
695         Map<Statement, OrdinalSet<Statement>> heapReachingDefs = new HeapReachingDefs(modRef, heapModel).computeReachingDefs(node, ir, pa, mod,
696             relevantStatements, new HeapExclusions(SetComplement.complement(new SingletonSet(t))), cg);
697     
698         for (Statement st : heapReachingDefs.keySet()) {
699           switch (st.getKind()) {
700           case NORMAL:
701           case CATCH:
702           case PHI:
703           case PI: {
704             OrdinalSet<Statement> defs = heapReachingDefs.get(st);
705             if (defs != null) {
706               for (Statement def : defs) {
707                 delegate.addEdge(def, st);
708               }
709             }
710           }
711             break;
712           case EXC_RET_CALLER:
713           case NORMAL_RET_CALLER:
714           case PARAM_CALLEE:
715           case NORMAL_RET_CALLEE:
716           case PARAM_CALLER:
717           case EXC_RET_CALLEE:
718             break;
719           case HEAP_RET_CALLEE:
720           case HEAP_RET_CALLER:
721           case HEAP_PARAM_CALLER: {
722             OrdinalSet<Statement> defs = heapReachingDefs.get(st);
723             if (defs != null) {
724               for (Statement def : defs) {
725                 delegate.addEdge(def, st);
726               }
727             }
728             break;
729           }
730           case HEAP_PARAM_CALLEE:
731           case METHOD_ENTRY:
732           case METHOD_EXIT:
733             // do nothing .. there are no incoming edges
734             break;
735           default:
736             Assertions.UNREACHABLE(st.toString());
737             break;
738           }
739         }
740       }

746         } else if (use instanceof SSAArrayReferenceInstruction) {

811       public static synchronized Statement ssaInstruction2Statement(CGNode node, SSAInstruction s,
812           Map<SSAInstruction, Integer> instructionIndices, IR ir) {
813         if (node == null) {
814           throw new IllegalArgumentException("null node");
815         }
816         if (s == null) {
817           throw new IllegalArgumentException("null s");
818         }
819         if (s instanceof SSAPhiInstruction) {
820           SSAPhiInstruction phi = (SSAPhiInstruction) s;
821           return new PhiStatement(node, phi);
822         } else if (s instanceof SSAPiInstruction) {
823           SSAPiInstruction pi = (SSAPiInstruction) s;
824           return new PiStatement(node, pi);
825         } else if (s instanceof SSAGetCaughtExceptionInstruction) {
826           return new GetCaughtExceptionStatement(node, ((SSAGetCaughtExceptionInstruction) s));
827         } else {
828           Integer x = instructionIndices.get(s);
829           if (x == null) {
830             Assertions.UNREACHABLE(s.toString() + "\nnot found in map of\n" + ir);
831           }
832           return new NormalStatement(node, x.intValue());
833         }
834       }

853       /**
854        * Convert a NORMAL or PHI Statement to an SSAInstruction
855        */
856       private SSAInstruction statement2SSAInstruction(SSAInstruction[] instructions, Statement s) {
857         SSAInstruction statement = null;
858         switch (s.getKind()) {
859         case NORMAL:
860           NormalStatement n = (NormalStatement) s;
861           statement = instructions[n.getInstructionIndex()];
862           break;
863         case PHI:
864           PhiStatement p = (PhiStatement) s;
865           statement = p.getPhi();
866           break;
867         case PI:
868           PiStatement ps = (PiStatement) s;
869           statement = ps.getPi();
870           break;
871         case CATCH:
872           GetCaughtExceptionStatement g = (GetCaughtExceptionStatement) s;
873           statement = g.getInstruction();
874           break;
875         default:
876           Assertions.UNREACHABLE(s.toString());
877         }
878         return statement;
879       }

899        * create nodes representing defs of the return values
900        * 
901        * @param mod the set of heap locations which may be written (transitively) by this node. These are logically parameters in the
902        *          SDG.
903        * @param dOptions
904        */
905       private void createReturnStatements() {
906         ArrayList<Statement> list = new ArrayList<Statement>();
907         if (!node.getMethod().getReturnType().equals(TypeReference.Void)) {
908           NormalReturnCallee n = new NormalReturnCallee(node);
909           delegate.addNode(n);
910           list.add(n);
911         }
912         if (!dOptions.isIgnoreExceptions()) {
913           ExceptionalReturnCallee e = new ExceptionalReturnCallee(node);
914           delegate.addNode(e);
915           list.add(e);
916         }
917         if (!dOptions.isIgnoreHeap()) {
918           for (PointerKey p : mod.get(node)) {
919             Statement h = new HeapStatement.HeapReturnCallee(node, p);
920             delegate.addNode(h);
921             list.add(h);
922           }
923         }
924         returnStatements = new Statement[list.size()];
925         list.toArray(returnStatements);
926       }

960       private void createSpecialStatements(IR ir) {

962         for (Iterator<SSAInstruction> it = ir.iterateAllInstructions(); it.hasNext();) {
963           SSAInstruction s = it.next();

966           } else if (s instanceof SSAGetCaughtExceptionInstruction) {

968           } else if (s instanceof SSAPiInstruction) {
969             delegate.addNode(new PiStatement(node, (SSAPiInstruction) s));
970           }
971         }

974       /**
975        * Create nodes in the graph corresponding to "normal" (bytecode) instructions
976        */
977       private void createNormalStatements(IR ir, Map<CGNode, OrdinalSet<PointerKey>> ref) {
978         // create a node for every normal instruction in the IR
979         SSAInstruction[] instructions = ir.getInstructions();
980         for (int i = 0; i < instructions.length; i++) {
981           SSAInstruction s = instructions[i];
982     
983           if (s instanceof SSAGetCaughtExceptionInstruction) {
984             continue;
985           }
986     
987           if (s != null) {
988             final NormalStatement statement = new NormalStatement(node, i);
989             delegate.addNode(statement);
990             if (s instanceof SSAAbstractInvokeInstruction) {
991               callSite2Statement.put(((SSAAbstractInvokeInstruction) s).getCallSite(), statement);
992               addParamPassingStatements(i, ref, ir);
993             }
994           }
995         }
996       }

998       /**
999        * Create nodes in the graph corresponding to in/out parameter passing for a call instruction
1000        */
1001       private void addParamPassingStatements(int callIndex, Map<CGNode, OrdinalSet<PointerKey>> ref, IR ir) {
1002         SSAAbstractInvokeInstruction call = (SSAAbstractInvokeInstruction) ir.getInstructions()[callIndex];
1003         Collection<Statement> params = MapUtil.findOrCreateSet(callerParamStatements, call.getCallSite());
1004         Collection<Statement> rets = MapUtil.findOrCreateSet(callerReturnStatements, call.getCallSite());
1005         for (int j = 0; j < call.getNumberOfUses(); j++) {
1006           Statement st = new ParamCaller(node, callIndex, call.getUse(j));
1007           delegate.addNode(st);
1008           params.add(st);
1009         }
1010         if (!call.getDeclaredResultType().equals(TypeReference.Void)) {
1011           Statement st = new NormalReturnCaller(node, callIndex);
1012           delegate.addNode(st);
1013           rets.add(st);
1014         }
1015         {
1016           if (!dOptions.isIgnoreExceptions()) {
1017             Statement st = new ExceptionalReturnCaller(node, callIndex);
1018             delegate.addNode(st);
1019             rets.add(st);
1020           }
1021         }
1022     
1023         if (!dOptions.isIgnoreHeap()) {
1024           OrdinalSet<PointerKey> uref = unionHeapLocations(cg, node, call, ref);
1025           for (PointerKey p : uref) {
1026             Statement st = new HeapStatement.HeapParamCaller(node, callIndex, p);
1027             delegate.addNode(st);
1028             params.add(st);
1029           }
1030           OrdinalSet<PointerKey> umod = unionHeapLocations(cg, node, call, mod);
1031           for (PointerKey p : umod) {
1032             Statement st = new HeapStatement.HeapReturnCaller(node, callIndex, p);
1033             delegate.addNode(st);
1034             rets.add(st);
1035           }
1036         }
1037       }

1041        */
1042       private OrdinalSet<PointerKey> unionHeapLocations(CallGraph cg, CGNode n, SSAAbstractInvokeInstruction call,
1043           Map<CGNode, OrdinalSet<PointerKey>> loc) {

1047         }
1048         return new OrdinalSet<PointerKey>(bv, loc.get(n).getMapping());
1049       }
