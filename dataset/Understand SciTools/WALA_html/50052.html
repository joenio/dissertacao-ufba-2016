97       /**
98        * Build an IR, then check getLocalNames
99        */
100       @Test
101       public void testAliasNames() {
102         try {
103           AnalysisScope scope = AnalysisScopeReader.readJavaScope(TestConstants.WALA_TESTDATA, (new FileProvider())
104               .getFile("J2SEClassHierarchyExclusions.txt"), MY_CLASSLOADER);
105           ClassHierarchy cha = ClassHierarchy.make(scope);
106           TypeReference t = TypeReference.findOrCreateClass(scope.getApplicationLoader(), "cornerCases", "AliasNames");
107           IClass klass = cha.lookupClass(t);
108           Assert.assertTrue(klass != null);
109           IMethod m = klass.getMethod(new Selector(Atom.findOrCreateAsciiAtom("foo"), Descriptor
110               .findOrCreateUTF8("([Ljava/lang/String;)V")));
111     
112           AnalysisOptions options = new AnalysisOptions();
113           options.getSSAOptions().setPiNodePolicy(SSAOptions.getAllBuiltInPiNodes());
114           IR ir = cache.getSSACache().findOrCreateIR(m, Everywhere.EVERYWHERE, options.getSSAOptions());
115     
116           for (int offsetIndex = 0; offsetIndex < ir.getInstructions().length; offsetIndex++) {
117             SSAInstruction instr = ir.getInstructions()[offsetIndex];
118             if (instr != null) {
119               String[] localNames = ir.getLocalNames(offsetIndex, instr.getDef());
120               if (localNames != null && localNames.length > 0 && localNames[0] == null) {
121                 System.err.println(ir);
122                 Assert.assertTrue(" getLocalNames() returned [null,...] for the def of instruction at offset " + offsetIndex
123                     + "\n\tinstr", false);
124               }
125             }
126           }
127         } catch (Exception e) {
128           e.printStackTrace();
129           Assertions.UNREACHABLE();
130         }
131       }

133       @Test
134       public void testLocalNamesWithoutPiNodes() {
135         SSAPiNodePolicy save = options.getSSAOptions().getPiNodePolicy();
136         options.getSSAOptions().setPiNodePolicy(null);
137         MethodReference mref = scope.findMethod(AnalysisScope.APPLICATION, "LcornerCases/Locals", Atom.findOrCreateUnicodeAtom("foo"),
138             new ImmutableByteArray(UTF8Convert.toUTF8("([Ljava/lang/String;)V")));
139         Assert.assertNotNull("method not found", mref);
140         IMethod imethod = cha.resolveMethod(mref);
141         Assert.assertNotNull("imethod not found", imethod);
142         IR ir = cache.getIRFactory().makeIR(imethod, Everywhere.EVERYWHERE, options.getSSAOptions());
143         options.getSSAOptions().setPiNodePolicy(save);
144     
145         // v1 should be the parameter "a" at pc 0
146         String[] names = ir.getLocalNames(0, 1);
147         Assert.assertTrue("failed local name resolution for v1@0", names != null);
148         Assert.assertTrue("incorrect number of local names for v1@0: " + names.length, names.length == 1);
149         Assert.assertTrue("incorrect local name resolution for v1@0: " + names[0], names[0].equals("a"));
150     
151         // v2 is a compiler-induced temporary
152         Assert.assertTrue("didn't expect name for v2 at pc 2", ir.getLocalNames(2, 2) == null);
153     
154         // at pc 5, v1 should represent the locals "a" and "b"
155         names = ir.getLocalNames(5, 1);
156         Assert.assertTrue("failed local name resolution for v1@5", names != null);
157         Assert.assertTrue("incorrect number of local names for v1@5: " + names.length, names.length == 2);
158         Assert.assertTrue("incorrect local name resolution #0 for v1@5: " + names[0], names[0].equals("a"));
159         Assert.assertTrue("incorrect local name resolution #1 for v1@5: " + names[1], names[1].equals("b"));
160       }
