27     /**
28      * We need to generate synthetic methods for Function.apply() in the target
29      * selector, so that the AstMethod for Function_prototype_apply() in the prologue doesn't
30      * actually get used in the CGNodes used for calls to Function.prototype.apply(). The
31      * generated dummy methods should <em>never</em> actually be used except as a
32      * stub.
33      */
34     public class JavaScriptFunctionApplyTargetSelector implements MethodTargetSelector {
35     
36       private final MethodTargetSelector base;
37     
38       private static final TypeName APPLY_TYPE_NAME = TypeName.findOrCreate("Lprologue.js/Function_prototype_apply");
39     
40       private IMethod applyMethod;
41       public JavaScriptFunctionApplyTargetSelector(MethodTargetSelector base) {
42         this.base = base;
43       }
44     
45       private IMethod createApplyDummyMethod(IClass declaringClass) {
46         final MethodReference ref = genSyntheticMethodRef(declaringClass);
47         // number of args doesn't matter
48         JavaScriptSummary S = new JavaScriptSummary(ref, 1);
49         return new JavaScriptSummarizedFunction(ref, S, declaringClass);  
50       }
51     
52       public static final String SYNTHETIC_APPLY_METHOD_PREFIX = "$$ apply_dummy";
53     
54       private MethodReference genSyntheticMethodRef(IClass receiver) {
55         Atom atom = Atom.findOrCreateUnicodeAtom(SYNTHETIC_APPLY_METHOD_PREFIX);
56         Descriptor desc = Descriptor.findOrCreateUTF8(JavaScriptLoader.JS, "()LRoot;");
57         MethodReference ref = MethodReference.findOrCreate(receiver.getReference(), atom, desc);
58         return ref;
59       }
60     
61       @Override
62       public IMethod getCalleeTarget(CGNode caller, CallSiteReference site, IClass receiver) {
63         IMethod method = receiver.getMethod(AstMethodReference.fnSelector);
64         if (method != null) {
65           TypeName tn = method.getReference().getDeclaringClass().getName();
66           if (tn.equals(APPLY_TYPE_NAME)) {
67             if (applyMethod == null) {
68               applyMethod = createApplyDummyMethod(receiver);
69             }
70             return applyMethod;
71           }
72         }
73         return base.getCalleeTarget(caller, site, receiver);
74       }
75     
76     }
