44       @Test
45       public void testZeroLengthArray() throws ClassHierarchyException, IllegalArgumentException, CancelException, IOException {
46         AnalysisScope scope = CallGraphTestUtil.makeJ2SEAnalysisScope(TestConstants.WALA_TESTDATA,
47             CallGraphTestUtil.REGRESSION_EXCLUSIONS);
48         ClassHierarchy cha = ClassHierarchy.make(scope);
49         Iterable<Entrypoint> entrypoints = com.ibm.wala.ipa.callgraph.impl.Util.makeMainEntrypoints(scope, cha,
50             TestConstants.ZERO_LENGTH_ARRAY_MAIN);
51         AnalysisOptions options = CallGraphTestUtil.makeAnalysisOptions(scope, entrypoints);
52     
53         CallGraphBuilder builder = Util.makeVanillaZeroOneCFABuilder(options, new AnalysisCache(), cha, scope);
54         CallGraph cg = builder.makeCallGraph(options, null);
55         PointerAnalysis<InstanceKey> pa = builder.getPointerAnalysis();
56     //    System.err.println(pa);
57     
58         HeapModel heapModel = pa.getHeapModel();
59         CGNode mainNode = cg.getNode(
60             cha.resolveMethod(MethodReference.findOrCreate(
61                 TypeReference.findOrCreate(ClassLoaderReference.Application, TestConstants.ZERO_LENGTH_ARRAY_MAIN),
62                 Selector.make("main([Ljava/lang/String;)V"))), Everywhere.EVERYWHERE);
63         OrdinalSet<InstanceKey> pointsToSet = pa.getPointsToSet(heapModel.getPointerKeyForLocal(mainNode, 4));
64         Assert.assertEquals(1, pointsToSet.size());
65         InstanceKey arrayKey = pointsToSet.iterator().next();
66         OrdinalSet<InstanceKey> arrayContents = pa.getPointsToSet(heapModel.getPointerKeyForArrayContents(arrayKey));
67         System.err.println(arrayContents);
68         Assert.assertEquals(0, arrayContents.size());
69         
70       }
