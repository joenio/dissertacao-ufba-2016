201             public int meetLocal(int n, int[] rhs, DexCFG.BasicBlock bb) {
202                 if (allTheSame(rhs)) {
203                     for (int i = 0; i < rhs.length; i++) {
204                         if (rhs[i] != TOP) {
205                             return rhs[i];
206                         }
207                     }
208                     // didn't find anything but TOP
209                     return TOP;
210                 } else {
211                     SSACFG.BasicBlock newBB = cfg.getNode(dexCFG.getNumber(bb));
212                     if (bb.isExitBlock()) {
213                         // no phis in exit block please
214                         return TOP;
215                     }
216                     // if we already have a phi for this local
217                     SSAPhiInstruction phi = newBB.getPhiForLocal(n);
218                     int result;
219                     if (phi == null) {
220                         // no phi already exists. create one.
221                         result = symbolTable.newPhi(rhs);
222                         PhiValue v = symbolTable.getPhiValue(result);
223                         phi = v.getPhiInstruction();
224                         newBB.addPhiForLocal(n, phi);
225                     } else {
226                         // already created a phi. update it to account for the
227                         // new merge.
228                         result = phi.getDef();
229                         phi.setValues(rhs.clone());
230                     }
231                     return result;
232                 }

235             /**
236              * Are all rhs values all the same? Note, we consider TOP (-1) to be same as everything else.
237              *
238              * @param rhs
239              * @return boolean
240              */
241             private boolean allTheSame(int[] rhs) {
242                 int x = -1;
243                 // set x := the first non-TOP value
244                 int i = 0;
245                 for (i = 0; i < rhs.length; i++) {
246                     if (rhs[i] != TOP) {
247                         x = rhs[i];
248                         break;
249                     }
250                 }
251                 // check the remaining values
252                 for (i = i + 1; i < rhs.length; i++) {
253                     if (rhs[i] != x && rhs[i] != TOP)
254                         return false;
255                 }
256                 return true;
257             }

369             private void emitInstruction(SSAInstruction s) {
370                 instructions[getCurrentInstructionIndex()] = s;
371                 for (int i = 0; i < s.getNumberOfDefs(); i++) {
372                     if (creators.length < (s.getDef(i) + 1)) {
373                         SSAInstruction[] arr = new SSAInstruction[2 * s.getDef(i)];
374                         System.arraycopy(creators, 0, arr, 0, creators.length);
375                         creators = arr;
376                     }
377     
378                     assert s.getDef(i) != -1 : "invalid def " + i + " for " + s;
379     
380                     creators[s.getDef(i)] = s;
381                 }
382             }

385                 return instructions[getCurrentInstructionIndex()];

388             /**
389              * If we've already created the current instruction, return the value number def'ed by the current instruction. Else, create a
390              * new symbol.
391              */
392     
393             private int reuseOrCreateDef() {
394                 if (getCurrentInstruction() == null || !getCurrentInstruction().hasDef()) {
395                     return symbolTable.newSymbol();
396                 } else {
397                     return getCurrentInstruction().getDef();
398                 }
399             }
400     
401             /**
402              * If we've already created the current instruction, return the value number representing the exception the instruction may
403              * throw. Else, create a new symbol
404              */
405             private int reuseOrCreateException() {
406     
407                 if (getCurrentInstruction() != null) {
408                     assert getCurrentInstruction() instanceof SSAInvokeInstruction;
409                 }
410                 if (getCurrentInstruction() == null) {
411                     return symbolTable.newSymbol();
412                 } else {
413                     SSAInvokeInstruction s = (SSAInvokeInstruction) getCurrentInstruction();
414                     return s.getException();
415                 }
416             }

1420              * @param nBlocks number of basic blocks in the CFG
1421              */

1483             /**
1484              * @param pc a program counter (index into ShrikeBT instruction array)
1485              * @param vn a value number
1486              * @return if we know that immediately after the given program counter, v_vn corresponds to some set of locals, then return an
1487              *         array of the local numbers. else return null.
1488              */
1489             private int[] findLocalsForValueNumber(int pc, int vn) {
1490     
1491                 IBasicBlock<Instruction> bb = dexCFG.getBlockForInstruction(pc);
1492                 int firstInstruction = bb.getFirstInstructionIndex();
1493                 // walk forward from the first instruction to reconstruct the
1494                 // state of the locals at this pc
1495                 int[] locals = block2LocalState[bb.getNumber()];
1496                 if (locals == null) {
1497                     locals = allocateNewLocalsArray();
1498                 }
1499                 for (int i = firstInstruction; i <= pc; i++) {
1500                     if (localStoreMap[i] != null) {
1501                         IntPair p = localStoreMap[i];
1502                         locals[p.getY()] = p.getX();
1503                     }
1504                 }
1505                 return extractIndices(locals, vn);
1506             }

1509                 int[] result = new int[maxLocals];
1510                 for (int i = 0; i < maxLocals; i++) {
1511                     result[i] = OPTIMISTIC ? TOP : BOTTOM;
1512                 }
1513                 return result;
1514             }

1516             /**
1517              * @return the indices i s.t. x[i] == y, or null if none found.
1518              */
1519             private int[] extractIndices(int[] x, int y) {
1520                 int count = 0;
1521                 for (int i = 0; i < x.length; i++) {
1522                     if (x[i] == y) {
1523                         count++;
1524                     }
1525                 }
1526                 if (count == 0) {
1527                     return null;
1528                 } else {
1529                     int[] result = new int[count];
1530                     int j = 0;
1531                     for (int i = 0; i < x.length; i++) {
1532                         if (x[i] == y) {
1533                             result[j++] = i;
1534                         }
1535                     }
1536                     return result;
1537                 }
1538             }
