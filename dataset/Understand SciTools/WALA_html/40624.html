74         JavaScriptLoaderFactory loaders = new JavaScriptLoaderFactory(translatorFactory);

94     	public Pair<JSCallGraph, PointerAnalysis<ObjectVertex>> buildCG(JavaScriptLoaderFactory loaders, SourceModule[] scripts, BuilderType builderType, IProgressMonitor monitor, boolean supportFullPointerAnalysis) throws IOException, WalaException, CancelException  {
95     		CAstAnalysisScope scope = new CAstAnalysisScope(scripts, loaders, Collections.singleton(JavaScriptLoader.JS));
96     		IClassHierarchy cha = ClassHierarchy.make(scope, loaders, JavaScriptLoader.JS);
97     		Util.checkForFrontEndErrors(cha);
98     		Iterable<Entrypoint> roots = JSCallGraphUtil.makeScriptRoots(cha);
99     		FieldBasedCallGraphBuilder builder = null;
100     		
101     		AnalysisCache cache = new AnalysisCache(AstIRFactory.makeDefaultFactory());
102     		switch(builderType) {
103     		case PESSIMISTIC:
104     			builder = new PessimisticCallGraphBuilder(cha, JSCallGraphUtil.makeOptions(scope, cha, roots), cache, supportFullPointerAnalysis);
105     			break;
106     		case OPTIMISTIC:
107     			builder = new OptimisticCallgraphBuilder(cha, JSCallGraphUtil.makeOptions(scope, cha, roots), cache, supportFullPointerAnalysis);
108     			break;
109     		case OPTIMISTIC_WORKLIST:
110     		  builder = new WorklistBasedOptimisticCallgraphBuilder(cha, JSCallGraphUtil.makeOptions(scope, cha, roots), cache, supportFullPointerAnalysis);
111     		  break;
112     		}
113     		
114     		return builder.buildCallGraph(roots, monitor);
115     	}

123     	@SuppressWarnings("unused")
124       private static void compareCGs(Map<String, Set<String>> cg1, Map<String, Set<String>> cg2) {
125     	  boolean diff = false;
126     	  for(String key : cg1.keySet()) {
127     	    Set<String> targets1 = cg1.get(key), targets2 = cg2.get(key);
128     	    if(targets2 == null) {
129     	      diff = true;
130     	      System.err.println("CG2 doesn't have call site" + key);
131     	    } else {
132     	      for(String target : targets1)
133     	        if(!targets2.contains(target)) {
134     	          diff = true;
135     	          System.err.println("CG2 doesn't have edge " + key + " -> " + target);
136     	        }
137     	      for(String target : targets2)
138     	        if(!targets1.contains(target)) {
139     	          diff = true;
140     	          System.err.println("CG1 doesn't have edge " + key + " -> " + target);
141     	        }
142     	    }
143     	  }
144     	  for(String key : cg2.keySet()) {
145     	    if(!cg1.containsKey(key)) {
146             diff = true;
147     	      System.err.println("CG1 doesn't have call site " + key);
148     	    }
149     	  }
150     	  if(!diff)
151     	    System.err.println("call graphs are identical");
152     	}
