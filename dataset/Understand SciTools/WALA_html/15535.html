36     /**
37      * Produces {@link com.ibm.wala.analysis.reflection.GetMethodContext} if appropriate.
38      * @author Michael Heilmann
39      * @see com.ibm.wala.analysis.reflection.GetMethodContext
40      * @see com.ibm.wala.analysis.reflection.GetMethodContextInterpreter
41      */
42     public class GetMethodContextSelector implements ContextSelector {
43       
44       /**
45        * If <tt>true</tt>, debug information is emitted.
46        */
47       protected static final boolean DEBUG = false;
48     
49       /**
50        * whether to only follow get method calls on application classes, ignoring system ones
51        */
52       private final boolean applicationClassesOnly;
53       
54       public GetMethodContextSelector(boolean applicationClassesOnly) {
55         this.applicationClassesOnly = applicationClassesOnly;
56       }
57     
58       /**
59        *  If
60        *  <ul>
61        *    <li>the {@link CallSiteReference} invokes either {@link java.lang.Class#getMethod} or {@link java.lang.Class#getDeclaredMethod},</li> 
62        *    <li>and the receiver is a type constant and</li>
63        *    <li>the first argument is a constant,</li>
64        *  </ul>
65        *  then return a {@link GetMethodContextSelector}.
66        */
67       @Override
68       public Context getCalleeTarget(CGNode caller,CallSiteReference site,IMethod callee,InstanceKey[] receiver) {
69         if (receiver != null && receiver.length > 0 && mayUnderstand(caller, site, callee, receiver[0])) {
70           if (DEBUG) {
71             System.out.print("site := " + site + ", receiver := " + receiver[0]);
72           }
73           // If the first argument is a constant ...
74           IR ir = caller.getIR();
75           SymbolTable symbolTable = ir.getSymbolTable();
76           SSAAbstractInvokeInstruction[] invokeInstructions = caller.getIR().getCalls(site);
77           if (invokeInstructions.length != 1) {
78             return null;
79           }
80           int use = invokeInstructions[0].getUse(1);
81           if (symbolTable.isStringConstant(invokeInstructions[0].getUse(1))) {
82             String sym = symbolTable.getStringValue(use);
83             if (DEBUG) {
84               System.out.println(invokeInstructions);
85               System.out.println(", with constant := `" + sym + "`");
86               for (InstanceKey instanceKey:receiver) {
87                 System.out.println(" " + instanceKey);
88               }
89             }
90             // ... return an GetMethodContext.
91             ConstantKey ck = makeConstantKey(caller.getClassHierarchy(),sym);
92             if (DEBUG) {
93               System.out.println(ck);
94             }
95             
96             IClass type = getTypeConstant(receiver[0]);
97             if (!applicationClassesOnly || 
98                 !(type.getClassLoader().getReference().equals(ClassLoaderReference.Primordial) ||
99                   type.getClassLoader().getReference().equals(ClassLoaderReference.Extension))) {
100               return new GetMethodContext(new PointType(type),ck);
101             }
102           }
103           if (DEBUG) {
104             System.out.println(", with constant := no");
105           }
106           // Otherwise, return null.
107           // TODO Remove this, just fall-through.
108           return null;
109         }
110         return null;
111       }
112     
113       /**
114        * If <tt>instance</tt> is a {@link ConstantKey} and its value is an instance of {@link IClass},
115        * return that value. Otherwise, return <tt>null</tt>.
116        */
117       private IClass getTypeConstant(InstanceKey instance) {
118         if (instance instanceof ConstantKey) {
119           ConstantKey c = (ConstantKey) instance;
120           if (c.getValue() instanceof IClass) {
121             return (IClass) c.getValue();
122           }
123         }
124         return null;
125       }
126       
127       /**
128        * Create a constant key for a string.
129        * @param cha the class hierarchy
130        * @param str the string
131        * @return the constant key
132        */
133       protected static ConstantKey<String> makeConstantKey(IClassHierarchy cha,String str) {
134         IClass cls = cha.lookupClass(TypeReference.JavaLangString);
135         ConstantKey<String> ck = new ConstantKey<String>(str,cls);
136         return ck;
137       }
138     
139       private static final Collection<MethodReference> UNDERSTOOD_METHOD_REFS = HashSetFactory.make();
140     
141       static {
142         UNDERSTOOD_METHOD_REFS.add(GetMethodContextInterpreter.GET_METHOD);
143         UNDERSTOOD_METHOD_REFS.add(GetMethodContextInterpreter.GET_DECLARED_METHOD);
144       }
145     
146       /**
147        * This object understands a dispatch to {@link java.lang.Class#getMethod(String, Class...)}
148        * or {@link java.lang.Class#getDeclaredMethod} when the receiver is a type constant.
149        */
150       private boolean mayUnderstand(CGNode caller,CallSiteReference site,IMethod targetMethod,InstanceKey instance) {
151         return UNDERSTOOD_METHOD_REFS.contains(targetMethod.getReference())
152             && getTypeConstant(instance) != null;
153       }
154     
155       /**
156        * TODO
157        *  MH: Shouldn't be the first TWO parameters be relevant?
158        *      Documentation is not too helpful about the implications.
159        */
160       private static final IntSet thisParameter = IntSetUtil.make(new int[]{0});
161     
162       @Override
163       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
164         if (site.isDispatch() || site.getDeclaredTarget().getNumberOfParameters() > 0) {
165           return thisParameter;
166         } else {
167           return EmptyIntSet.instance;
168         }
169       } 
170     }
