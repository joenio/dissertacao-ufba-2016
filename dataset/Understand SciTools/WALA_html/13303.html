173       private static String escapeForXML(String x, char from, String to) {
174         return (x.indexOf(from) != -1) ? x.replaceAll(Character.toString(from), to) : x;
175       }
176     
177       public static String escapeForXML(String x) {
178         return 
179           escapeForXML(
180             escapeForXML(
181     	  escapeForXML(
182     	    escapeForXML(x, '&', "&amp;"),
183     	    '"', "&quot;"),
184     	  '<', "&lt;"),
185     	'>', "&gt;");
186       }

192       public void doPrintTo(CAstNode top, CAstSourcePositionMap pos, Writer w, int depth, boolean uglyBrackets) {
193         try {
194           CAstSourcePositionMap.Position p = (pos!=null)? pos.getPosition( top ): null;
195           for(int i = 0; i < depth; i++) w.write("  ");
196           if (top == null) {
197     	  w.write("(null)\n");
198           } else if (top.getValue() != null) {
199     	if (uglyBrackets) {
200     	  w.write("<constant value=\"");
201     	  w.write( escapeForXML( top.getValue().toString() ) );
202     	  w.write("\" type=\"");
203     	  w.write( top.getValue().getClass().toString() );
204     	  w.write("\"");
205     	} else {
206     	  w.write( "\"" );
207     	  w.write( top.getValue().toString() );
208     	  w.write( "\"" );
209     	}
210     	if (p != null) {
211     	  if (uglyBrackets) 
212     	    w.write(" lineNumber=\"" + p + "\"");
213     	  else
214     	    w.write( " at " + p );
215     	}
216     	if (uglyBrackets) w.write("/>");
217     	w.write("\n");
218           } else {
219     	if (uglyBrackets) w.write("<");
220     	w.write( kindAsString( top.getKind() ) );
221     	if (p != null)
222     	  if (uglyBrackets)
223     	    w.write( " position=\"" + p + "\"");
224     	  else
225     	    w.write( " at " + p );
226     	if (uglyBrackets) w.write(">");
227     	w.write("\n");
228     	for(int i = 0; i < top.getChildCount(); i++) {
229     	  doPrintTo( top.getChild(i), pos, w, depth+1, uglyBrackets );
230     	}
231     	if (uglyBrackets) {
232     	  for(int i = 0; i < depth; i++) w.write("  ");
233     	  w.write("</" + kindAsString(top.getKind()) + ">\n");
234     	}
235           }
236         } catch (java.io.IOException e) {
237     	
238         }
239       }

257       public static void printTo(CAstEntity e, Writer w) {
258         //anca: check if the writer is null
259         if(w != null)
260           instance.doPrintTo(e, w);
261       }
262     
263       protected void doPrintTo(CAstEntity e, Writer w) {
264           try {
265     	w.write(getEntityKindAsString(e.getKind()));
266     	w.write(": ");
267     	w.write(e.getName());
268     	w.write('\n');
269     	if (e.getArgumentNames().length > 0) {
270     	  w.write("(");
271     	  String[] names = e.getArgumentNames();
272     	  for(int i = 0; i < names.length; i++) {
273     	    w.write("  " + names[i]);
274     	  }
275     	  w.write("  )\n");
276     	}
277     	if (e.getAST() != null) {
278     	    doPrintTo(e.getAST(), e.getSourceMap(), w);
279     	    w.write('\n');
280     	}
281     	for(Iterator i= e.getAllScopedEntities().values().iterator();
282     	    i.hasNext(); ) 
283     	{
284     	  for(Iterator j = ((Collection) i.next()).iterator(); j.hasNext(); ) {
285     	    doPrintTo((CAstEntity) j.next(), w);
286     	  }
287     	}
288     	w.flush();
289         } catch (IOException e1) {
290           System.err.println("unexpected I/O exception " + e1);
291         }
292       }
