11     package com.ibm.wala.cast.js.loader;
12     
13     import com.ibm.wala.cast.js.translator.JavaScriptTranslatorFactory;
14     import com.ibm.wala.cast.js.types.JavaScriptTypes;
15     import com.ibm.wala.cast.loader.SingleClassLoaderFactory;
16     import com.ibm.wala.cast.tree.rewrite.CAstRewriterFactory;
17     import com.ibm.wala.classLoader.IClassLoader;
18     import com.ibm.wala.ipa.cha.IClassHierarchy;
19     import com.ibm.wala.types.ClassLoaderReference;
20     
21     /**
22      * Creates the single {@link IClassLoader class loader} used for JavaScript.  
23      */
24     public class JavaScriptLoaderFactory extends SingleClassLoaderFactory {
25       protected final JavaScriptTranslatorFactory translatorFactory;
26       protected final CAstRewriterFactory preprocessor;
27       
28       public JavaScriptLoaderFactory(JavaScriptTranslatorFactory factory) {
29         this(factory, null);
30       }
31     
32       public JavaScriptLoaderFactory(JavaScriptTranslatorFactory factory, CAstRewriterFactory preprocessor) {
33         this.translatorFactory = factory;
34         this.preprocessor = preprocessor;
35       }
36     
37       @Override
38       protected IClassLoader makeTheLoader(IClassHierarchy cha) {
39         return new JavaScriptLoader( cha, translatorFactory, preprocessor );
40       }
41     
42       @Override
43       public ClassLoaderReference getTheReference() {
44         return JavaScriptTypes.jsLoader;
45       }
46     }
