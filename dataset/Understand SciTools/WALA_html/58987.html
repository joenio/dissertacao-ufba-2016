12     package com.ibm.wala.cast.js.ipa.callgraph.correlations.extraction;
13     
14     import java.util.Collections;
15     import java.util.Iterator;
16     import java.util.List;
17     import java.util.Set;
18     
19     import com.ibm.wala.cast.tree.CAstEntity;
20     import com.ibm.wala.cast.tree.CAstNode;
21     import com.ibm.wala.util.collections.HashSetFactory;
22     import com.ibm.wala.util.collections.Pair;
23     
24     /**
25      * A special {@link ChildPos} representing the position of a node which is the body of a for-in loop.
26      * 
27      * <p>This also stores some additional data obtained while rewriting the loop body, such as whether
28      * <code>return</code> statements were encountered.</p>
29      * 
30      * @author mschaefer
31      *
32      */
33     public class ExtractionPos extends NodePos {
34       private final CAstNode parent;
35     	private final ExtractionRegion region;
36     	private final NodePos parent_pos;
37     	private boolean contains_return;
38     	private boolean contains_this;
39     	private final Set<Pair<String, CAstNode>> goto_targets = HashSetFactory.make();
40     	private boolean contains_outer_goto;
41     	private final Set<ExtractionPos> nested_loops = HashSetFactory.make();
42     	private CAstEntity extracted_entity;
43     	private CAstNode callsite;
44     
45     	public ExtractionPos(CAstNode parent, ExtractionRegion region, NodePos parent_pos) {
46     	  this.parent = parent;
47     	  this.region = region;
48     	  this.parent_pos = parent_pos;
49     	}
50     	
51     	public CAstNode getParent() {
52     	  return parent;
53     	}
54     	
55     	public int getStart() {
56     	  return region.getStart();
57     	}
58     	
59     	public int getEnd() {
60     	  return region.getEnd();
61     	}
62     	
63     	public ExtractionRegion getRegion() {
64     	  return region;
65     	}
66     	
67     	public boolean contains(CAstNode node) {
68     	  for(int i=getStart();i<getEnd();++i)
69     	    if(NodePos.inSubtree(node, parent.getChild(i)))
70     	      return true;
71     	  return false;
72     	}
73     	
74     	public List<String> getParameters() {
75     	  return region.getParameters();
76     	}
77     	
78     	public void addGotoTarget(String label, CAstNode node) {
79     		// check whether this target lies beyond an enclosing for-in loop
80     		ExtractionPos outer = getEnclosingExtractionPos(parent_pos);
81     		if(outer != null && !outer.contains(node)) {
82     			// the goto needs to be handled by the outer loop
83     			outer.addGotoTarget(label, node);
84     			// but we need to remember to pass it on
85     			contains_outer_goto = true;
86     		} else {
87     			// this goto is our responsibility
88     			goto_targets.add(Pair.make(label, node));
89     		}
90     	}
91     
92       public boolean containsReturn() {
93     		return contains_return;
94     	}
95     	
96     	public void addReturn() {
97     		this.contains_return = true;
98     	}
99     	
100     	public Set<Pair<String, CAstNode>> getGotoTargets() {
101     		return Collections.unmodifiableSet(goto_targets);
102     	}
103     	
104     	public void addThis() {
105     		contains_this = true;
106     	}
107     	
108     	public boolean containsThis() {
109     		return contains_this;
110     	}
111     	
112     	public boolean containsGoto() {
113     		return !getGotoTargets().isEmpty();
114     	}
115     	
116     	public boolean containsOuterGoto() {
117     		return contains_outer_goto;
118     	}
119     	
120     	public boolean containsJump() {
121     		return containsGoto() || containsReturn() || containsOuterGoto();
122     	}
123     	
124     	public String getThisParmName() {
125     		return "thi$";
126     	}
127     	
128     	public void addNestedPos(ExtractionPos loop) {
129     		nested_loops.add(loop);
130     	}
131     	
132     	public Iterator<ExtractionPos> getNestedLoops() {
133     		return nested_loops.iterator();
134     	}
135     	
136     	public void setExtractedEntity(CAstEntity entity) {
137     		assert this.extracted_entity == null : "Cannot reset extracted entity.";
138     		extracted_entity = entity;
139     	}
140     	
141     	public CAstEntity getExtractedEntity() {
142     		assert extracted_entity != null : "Extracted entity not set.";
143     		return extracted_entity;
144     	}
145     	
146     	public void setCallSite(CAstNode callsite) {
147     		assert this.callsite == null : "Cannot reset call site.";
148     		this.callsite = callsite;
149     	}
150     	
151     	public CAstNode getCallSite() {
152     		assert callsite != null : "Call site not set.";
153     		return callsite;
154     	}
155     	
156     	@Override
157     	public <A> A accept(PosSwitch<A> ps) {
158     		return ps.caseForInLoopBodyPos(this);
159     	}
160     
161     	// return the outermost enclosing extraction position around 'pos' within the same function; "null" if there is none
162     	public static ExtractionPos getOutermostEnclosingExtractionPos(NodePos pos) {
163     		return pos.accept(new PosSwitch<ExtractionPos>() {
164     			@Override 
165     			public ExtractionPos caseRootPos(RootPos pos) { 
166     				return null; 
167     			}
168     			
169     			@Override 
170     			public ExtractionPos caseChildPos(ChildPos pos) {
171     				int kind = pos.getParent().getKind();
172     				if(kind == CAstNode.FUNCTION_STMT || kind == CAstNode.FUNCTION_EXPR)
173     					return null;
174     				return getOutermostEnclosingExtractionPos(pos.getParentPos()); 
175     			}
176     			
177     			@Override 
178     			public ExtractionPos caseForInLoopBodyPos(ExtractionPos pos) {
179     				ExtractionPos outer = getEnclosingExtractionPos(pos.getParentPos());
180     				return outer == null ? pos : outer; 
181     			}
182     
183           @Override
184           public ExtractionPos caseLabelPos(LabelPos pos) {
185             return getOutermostEnclosingExtractionPos(pos.getParentPos());
186           }			
187     		});
188     	}
189     
190     	// return the innermost enclosing extraction position around 'pos' within the same function; "null" if there is none
191     	public static ExtractionPos getEnclosingExtractionPos(NodePos pos) {
192     		return pos.accept(new PosSwitch<ExtractionPos>() {
193     			@Override
194     			public ExtractionPos caseRootPos(RootPos pos) { 
195     				return null; 
196     			}
197     			
198     			@Override 
199     			public ExtractionPos caseChildPos(ChildPos pos) { 
200     				int kind = pos.getParent().getKind();
201     				if(kind == CAstNode.FUNCTION_STMT || kind == CAstNode.FUNCTION_EXPR)
202     					return null;
203     				return getEnclosingExtractionPos(pos.getParentPos()); 
204     			}
205     			
206     			@Override 
207     			public ExtractionPos caseForInLoopBodyPos(ExtractionPos pos) { 
208     				return pos; 
209     			}
210     
211           @Override
212           public ExtractionPos caseLabelPos(LabelPos pos) {
213             return getEnclosingExtractionPos(pos.getParentPos());
214           }			
215     		});
216     	}
217     
218     	// is this the outermost for-in loop within its enclosing function?
219     	public boolean isOutermost() {
220     		return getEnclosingExtractionPos(parent_pos) == null;
221     	}
222     
223       public NodePos getParentPos() {
224         return parent_pos;
225       }
226     }
