38     package com.ibm.wala.util.collections;
39     
40     import java.util.AbstractSet;
41     import java.util.Collection;
42     import java.util.Iterator;
43     import java.util.NoSuchElementException;
44     
45     /**
46      * A set implementation backed by an array. This implementation is space-efficient for small sets, but several operations like
47      * {@link #contains(Object)} are linear time.
48      * 
49      * @param <T>
50      */
51     
52     public class ArraySet<T> extends AbstractSet<T> {
53     
54       @SuppressWarnings("rawtypes")
55       private static final ArraySet EMPTY = new ArraySet<Object>(0, true) {
56         @Override
57         /*
58          * @throws UnsupportedOperationException unconditionally
59          */
60         public boolean add(Object obj_) {
61           throw new UnsupportedOperationException();
62         }
63       };
64     
65       @SuppressWarnings("all")
66       public static final <T> ArraySet<T> empty() {
67         return (ArraySet<T>) EMPTY;
68       }
69     
70       private T[] _elems;
71     
72       private int _curIndex = 0;
73     
74       private final boolean checkDupes;
75     
76       @SuppressWarnings("all")
77       public ArraySet(int n, boolean checkDupes) {
78         if (n < 0) {
79           throw new IllegalArgumentException("invalid n: " + n);
80         }
81         _elems = (T[]) new Object[n];
82         this.checkDupes = checkDupes;
83       }
84     
85       public ArraySet() {
86         this(1, true);
87       }
88     
89       @SuppressWarnings("all")
90       public ArraySet(ArraySet<T> other) throws IllegalArgumentException {
91         if (other == null) {
92           throw new IllegalArgumentException("other == null");
93         }
94         int size = other._curIndex;
95         this._elems = (T[]) new Object[size];
96         this.checkDupes = other.checkDupes;
97         this._curIndex = size;
98         System.arraycopy(other._elems, 0, _elems, 0, size);
99       }
100     
101       private ArraySet(Collection<T> other) {
102         this(other.size(), true);
103         addAll(other);
104       }
105     
106       /**
107        * @throws UnsupportedOperationException if this {@link ArraySet} is immutable (optional)
108        */
109       @SuppressWarnings("all")
110       public boolean add(T o) {
111         if (o == null) {
112           throw new IllegalArgumentException("null o");
113         }
114         if (checkDupes && this.contains(o)) {
115           return false;
116         }
117         if (_curIndex == _elems.length) {
118           // lengthen array
119           Object[] tmp = _elems;
120           _elems = (T[]) new Object[tmp.length * 2];
121           System.arraycopy(tmp, 0, _elems, 0, tmp.length);
122         }
123         _elems[_curIndex] = o;
124         _curIndex++;
125         return true;
126       }
127     
128       public boolean addAll(ArraySet<T> other) throws IllegalArgumentException {
129         if (other == null) {
130           throw new IllegalArgumentException("other == null");
131         }
132         boolean ret = false;
133         for (int i = 0; i < other.size(); i++) {
134           boolean added = add(other.get(i));
135           ret = ret || added;
136         }
137         return ret;
138       }
139     
140       /*
141        * @see AAA.util.AAASet#contains(java.lang.Object)
142        */
143       @Override
144       public boolean contains(Object obj_) {
145         for (int i = 0; i < _curIndex; i++) {
146           if (_elems[i].equals(obj_))
147             return true;
148         }
149         return false;
150       }
151     
152       public boolean intersects(ArraySet<T> other) throws IllegalArgumentException {
153         if (other == null) {
154           throw new IllegalArgumentException("other == null");
155         }
156         for (int i = 0; i < other.size(); i++) {
157           if (contains(other.get(i)))
158             return true;
159         }
160         return false;
161       }
162     
163       public void forall(ObjectVisitor<T> visitor) {
164         if (visitor == null) {
165           throw new IllegalArgumentException("null visitor");
166         }
167         for (int i = 0; i < _curIndex; i++) {
168           visitor.visit(_elems[i]);
169         }
170       }
171     
172       @Override
173       public int size() {
174         return _curIndex;
175       }
176     
177       /**
178        * @throws IndexOutOfBoundsException if the index is out of range (index &lt; 0 || index &gt;= size()).
179        */
180       public T get(int i) {
181         return _elems[i];
182       }
183     
184       @Override
185       public boolean remove(Object obj_) {
186         int ind;
187         for (ind = 0; ind < _curIndex && !_elems[ind].equals(obj_); ind++) {
188         }
189         // check if object was never there
190         if (ind == _curIndex)
191           return false;
192         return remove(ind);
193       }
194     
195       /**
196        * @return <code>true</code> (SJF: So why return a value?)
197        */
198       public boolean remove(int ind) {
199         try {
200           // hope i got this right...
201           System.arraycopy(_elems, ind + 1, _elems, ind, _curIndex - (ind + 1));
202           _curIndex--;
203           return true;
204         } catch (ArrayIndexOutOfBoundsException e) {
205           throw new IllegalArgumentException("invalid ind: " + ind);
206         }
207       }
208     
209       @Override
210       public void clear() {
211         _curIndex = 0;
212       }
213     
214       /*
215        * @see java.util.Set#iterator()
216        */
217       @Override
218       public Iterator<T> iterator() {
219         return new ArraySetIterator();
220       }
221     
222       public class ArraySetIterator implements Iterator<T> {
223     
224         int ind = 0;
225     
226         final int setSize = size();
227     
228         public ArraySetIterator() {
229         }
230     
231         @Override
232         public void remove() {
233           throw new UnsupportedOperationException();
234         }
235     
236         @Override
237         public boolean hasNext() {
238           return ind < setSize;
239         }
240     
241         @Override
242         public T next() {
243           if (ind >= setSize) {
244             throw new NoSuchElementException();
245           }
246           return get(ind++);
247         }
248     
249       }
250     
251       public static <T> ArraySet<T> make() {
252         return new ArraySet<T>();
253       }
254     
255       public static <T> ArraySet<T> make(Collection<T> other) throws IllegalArgumentException {
256         if (other == null) {
257           throw new IllegalArgumentException("other == null");
258         }
259         return new ArraySet<T>(other);
260       }
261     
262     }
