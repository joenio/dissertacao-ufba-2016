35     /**
36      * ensures that no contexts returned by a base context selector are recursive
37      * (assertion failure otherwise)
38      */
39     public class RecursionCheckContextSelector implements ContextSelector {
40     
41       private final ContextSelector base;
42      
43         
44       /**
45        * the highest parameter index that we'll check . this is a HACK. ideally,
46        * given a context, we'd have some way to know all the {@link ContextKey}s
47        * that it knows about.
48        * 
49        * @see ContextKey#PARAMETERS
50        */
51       private static final int MAX_INTERESTING_PARAM = 5;
52     
53       public RecursionCheckContextSelector(ContextSelector base) {
54         this.base = base;
55       }
56     
57       @Override
58       public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] actualParameters) {
59         Context baseContext = base.getCalleeTarget(caller, site, callee, actualParameters);
60         assert !recursiveContext(baseContext, callee);
61         return baseContext;
62       }
63     
64       private boolean recursiveContext(Context baseContext, IMethod callee) {
65         if (!recursionPossible(callee)) {
66           return false;
67         }
68         LinkedList<Pair<Context,Collection<IMethod>>> worklist = new LinkedList<Pair<Context,Collection<IMethod>>>();
69         worklist.push(Pair.make(baseContext, (Collection<IMethod>)Collections.singleton(callee)));
70         while (!worklist.isEmpty()) {
71           Pair<Context, Collection<IMethod>> p = worklist.removeFirst();
72           Context curContext = p.fst;
73           Collection<IMethod> curEncountered = p.snd;
74           // we just do a case analysis here. we might have to add cases later to
75           // account for new types of context / recursion.
76           CGNode callerNode = (CGNode) curContext.get(ContextKey.CALLER);
77           if (callerNode != null) {
78             if (!updateForNode(baseContext, curEncountered, worklist, callerNode)) {
79               System.err.println("callee " + callee);
80               return true;
81             }
82           }
83           for (int i = 0; i < MAX_INTERESTING_PARAM; i++) {
84             FilteredPointerKey.SingleInstanceFilter filter = (SingleInstanceFilter) curContext.get(ContextKey.PARAMETERS[i]);
85             if (filter != null) {
86               InstanceKey ik = filter.getInstance();
87               if (ik instanceof ScopeMappingInstanceKey) {
88                 ik = ((ScopeMappingInstanceKey) ik).getBase();
89               }
90               if (ik instanceof InstanceKeyWithNode) {
91                 CGNode node = ((InstanceKeyWithNode)ik).getNode();
92                 if (!updateForNode(baseContext, curEncountered, worklist, node)) {
93                   System.err.println("callee " + callee);
94                   return true;
95                 }
96               }
97             }
98           }
99         }
100         return false;
101       }
102     
103       private boolean updateForNode(Context baseContext, Collection<IMethod> curEncountered, LinkedList<Pair<Context, Collection<IMethod>>> worklist, CGNode callerNode) {
104         final IMethod method = callerNode.getMethod();
105         if (!recursionPossible(method)) {
106           assert !curEncountered.contains(method);
107           return true;
108         }
109         if (curEncountered.contains(method)) {
110           System.err.println("recursion in context on method " + method);
111           System.err.println("encountered methods: ");
112           for (IMethod m : curEncountered) {
113             System.err.println("  " + m);
114           }
115           System.err.println("context " + baseContext);
116           return false;
117         }    
118         Collection<IMethod> newEncountered = new ArrayList<IMethod>(curEncountered);
119         newEncountered.add(method);
120         worklist.add(Pair.make(callerNode.getContext(),newEncountered));
121         return true;
122       }
123     
124     
125       @Override
126       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
127         return base.getRelevantParameters(caller, site);
128       }
129     
130       /**
131        * is it possible for m to be involved in a recursive cycle?
132        * @param m
133        * @return
134        */
135       private boolean recursionPossible(IMethod m) {
136         // object or array constructors cannot be involved
137         if (m.getReference().getName().equals(JavaScriptMethods.ctorAtom)) {
138           TypeReference declaringClass = m.getReference().getDeclaringClass();
139           if (declaringClass.equals(JavaScriptTypes.Object) || declaringClass.equals(JavaScriptTypes.Array)) {
140             return false;
141           }
142         }
143         return true;
144       }
145     }
