12     package com.ibm.wala.dataflow.ssa;
13     
14     import java.util.Iterator;
15     
16     import com.ibm.wala.analysis.typeInference.TypeInference;
17     import com.ibm.wala.fixedpoint.impl.DefaultFixedPointSolver;
18     import com.ibm.wala.fixedpoint.impl.NullaryOperator;
19     import com.ibm.wala.fixpoint.AbstractOperator;
20     import com.ibm.wala.fixpoint.IVariable;
21     import com.ibm.wala.ssa.IR;
22     import com.ibm.wala.ssa.SSAInstruction;
23     import com.ibm.wala.ssa.SymbolTable;
24     
25     /**
26      * This class performs intra-procedural propagation over an SSA form.
27      * 
28      * A client will subclass an {@link SSAInference} by providing factories that generate {@link IVariable}s corresponding to SSA value
29      * numbers, and {@link AbstractOperator}s corresponding to SSA instructions. This class will set up a dataflow system induced by the
30      * SSA def-use graph, and solve the system by iterating to a fixed point.
31      * 
32      * @see TypeInference for the canonical client of this machinery.
33      */
34     public abstract class SSAInference<T extends IVariable<?>> extends DefaultFixedPointSolver<T> {
35       static final boolean DEBUG = false;
36     
37       /**
38        * The governing SSA form
39        */
40       private IR ir;
41     
42       /**
43        * The governing symbol table
44        */
45       private SymbolTable symbolTable;
46     
47       /**
48        * Dataflow variables, one for each value in the symbol table.
49        */
50       private IVariable[] vars;
51     
52       public interface OperatorFactory<T extends IVariable> {
53         /**
54          * Get the dataflow operator induced by an instruction in SSA form.
55          * 
56          * @param instruction
57          * @return dataflow operator for the instruction, or null if the instruction is not applicable to the dataflow system.
58          */
59         AbstractOperator<T> get(SSAInstruction instruction);
60       }
61     
62       public interface VariableFactory {
63         /**
64          * Make the variable for a given value number.
65          * 
66          * @return a newly created dataflow variable, or null if not applicable.
67          */
68         public IVariable makeVariable(int valueNumber);
69       }
70     
71       /**
72        * initializer for SSA Inference equations.
73        */
74       protected void init(IR ir, VariableFactory varFactory, OperatorFactory<T> opFactory) {
75     
76         this.ir = ir;
77         this.symbolTable = ir.getSymbolTable();
78     
79         createVariables(varFactory);
80         createEquations(opFactory);
81       }
82     
83       private void createEquations(OperatorFactory<T> opFactory) {
84         SSAInstruction[] instructions = ir.getInstructions();
85         for (int i = 0; i < instructions.length; i++) {
86           SSAInstruction s = instructions[i];
87           makeEquationForInstruction(opFactory, s);
88         }
89         for (Iterator it = ir.iteratePhis(); it.hasNext();) {
90           SSAInstruction s = (SSAInstruction) it.next();
91           makeEquationForInstruction(opFactory, s);
92         }
93         for (Iterator it = ir.iteratePis(); it.hasNext();) {
94           SSAInstruction s = (SSAInstruction) it.next();
95           makeEquationForInstruction(opFactory, s);
96         }
97         for (Iterator it = ir.iterateCatchInstructions(); it.hasNext();) {
98           SSAInstruction s = (SSAInstruction) it.next();
99           makeEquationForInstruction(opFactory, s);
100         }
101       }
102     
103       /**
104        * Create a dataflow equation induced by a given instruction
105        */
106       private void makeEquationForInstruction(OperatorFactory<T> opFactory, SSAInstruction s) {
107         if (s != null && s.hasDef()) {
108           AbstractOperator<T> op = opFactory.get(s);
109           if (op != null) {
110             T def = getVariable(s.getDef());
111             if (op instanceof NullaryOperator) {
112               newStatement(def, (NullaryOperator<T>) op, false, false);
113             } else {
114               int n = s.getNumberOfUses();
115               T[] uses = makeStmtRHS(n);
116               for (int j = 0; j < n; j++) {
117                 if (s.getUse(j) > -1) {
118                   uses[j] = getVariable(s.getUse(j));
119                   assert uses[j] != null;
120                 }
121               }
122               newStatement(def, op, uses, false, false);
123             }
124           }
125         }
126       }
127     
128       /**
129        * Create a dataflow variable for each value number
130        */
131       private void createVariables(VariableFactory factory) {
132         vars = new IVariable[symbolTable.getMaxValueNumber() + 1];
133         for (int i = 1; i < vars.length; i++) {
134           vars[i] = factory.makeVariable(i);
135         }
136     
137       }
138     
139       /**
140        * @return the dataflow variable representing the value number, or null if none found.
141        */
142       @SuppressWarnings("unchecked")
143       protected T getVariable(int valueNumber) {
144         if (valueNumber < 0) {
145           throw new IllegalArgumentException("Illegal valueNumber " + valueNumber);
146         }
147         if (DEBUG) {
148           System.err.println(("getVariable for " + valueNumber + " returns " + vars[valueNumber]));
149         }
150         assert vars != null : "null vars array";
151         return (T) vars[valueNumber];
152       }
153     
154       /**
155        * Return a string representation of the system
156        * 
157        * @return a string representation of the system
158        */
159       @Override
160       public String toString() {
161         StringBuffer result = new StringBuffer("Type inference : \n");
162         for (int i = 0; i < vars.length; i++) {
163           result.append("v").append(i).append("  ").append(vars[i]).append("\n");
164         }
165         return result.toString();
166       }
167     }
