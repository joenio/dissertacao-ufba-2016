11     package com.ibm.wala.client;
12     
13     import java.util.Collection;
14     import java.util.jar.JarFile;
15     
16     import com.ibm.wala.classLoader.Module;
17     import com.ibm.wala.ipa.callgraph.AnalysisOptions;
18     import com.ibm.wala.ipa.callgraph.Entrypoint;
19     
20     public interface AnalysisEngine {
21     
22       /**
23        * Specify the list of modules that should be analyzed. If an EARFile is included in the list, all of its contained modules should
24        * be examined. Multiple ear files can be specified for cross-app invocations, which will become increasingly common in the 5.1
25        * release.
26        * 
27        * @param moduleFiles A non-null Collection of module files: (EARFile, WARFile, ApplicationClientFile, EJBJarFile).
28        */
29       void setModuleFiles(Collection moduleFiles);
30     
31       /**
32        * Specify the jar files that represent the standard J2SE libraries
33        * 
34        * @param libs an array of jar files; usually rt.jar for vanilla JDK core.jar, server.jar, and xml.jar for some WAS runtimes
35        */
36       void setJ2SELibraries(JarFile[] libs);
37     
38       /**
39        * Specify the mdoules that represent the standard J2SE libraries
40        * 
41        * @param libs an array of Modules; usually rt.jar for vanilla JDK core.jar, server.jar, and xml.jar for some WAS runtimes
42        */
43       void setJ2SELibraries(Module[] libs);
44     
45       /**
46        * Specify whether the engine should or should not employ "closed-world" analysis.
47        * 
48        * In a closed-world analysis, the engine considers only application client main methods and servlet entrypoints to the
49        * application.
50        * 
51        * In an open-world analysis, the engine additionally considers all EJB local and remote interface methods as entrypoints.
52        * 
53        * By default, this property is false; the default analysis is open-world
54        * 
55        * @param b whether to use closed-world analysis
56        */
57       void setClosedWorld(boolean b);
58     
59       /**
60        * Get the default analysis options appropriate for this engine
61        */
62       AnalysisOptions getDefaultOptions(Iterable<Entrypoint> entrypoints);
63     }
