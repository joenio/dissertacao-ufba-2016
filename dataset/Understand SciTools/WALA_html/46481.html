113         public Set<TypeReference> getComponents() {
114             if (ENTRIES.isEmpty()) {
115                 throw new IllegalStateException("No entrypoints loaded yet.");
116             }
117             
118             final Set<TypeReference> ret = new HashSet<TypeReference>();
119             for (final AndroidEntryPoint ep : ENTRIES) {
120                 final TypeReference epClass = ep.getMethod().getDeclaringClass().getReference();
121                 if (AndroidComponent.isAndroidComponent(epClass , ep.getClassHierarchy())) {
122                     ret.add(epClass);
123                 }
124             }
125             return ret;
126         }

255         private Class abstractAndroidModel = LoopAndroidModel.class;
256         /**
257          *  What special handling to insert into the model.
258          *
259          *  At given points in the model (called labels) special code is inserted into it (like loops).
260          *  This setting controls what code is inserted there.
261          *
262          *  @see    com.ibm.wala.dalvik.ipa.callgraph.androidModel.structure.AbstractAndroidModel
263          *  @see    com.ibm.wala.dalvik.ipa.callgraph.androidModel.structure.SequentialAndroidModel
264          *  @see    com.ibm.wala.dalvik.ipa.callgraph.androidModel.structure.LoopAndroidModel
265          *  @return An object that handles "events" that occur while generating the model.
266          *  @throws IllegalStateException if initialization fails
267          */
268         public AbstractAndroidModel makeModelBehavior(VolatileMethodSummary body, TypeSafeInstructionFactory insts,
269                 SSAValueManager paramManager, Iterable<? extends Entrypoint> entryPoints) {
270             if (abstractAndroidModel == null) {
271                 return new LoopAndroidModel(body, insts, paramManager, entryPoints);
272             } else {
273                 try {
274                     final Constructor<AbstractAndroidModel> ctor = this.abstractAndroidModel.getDeclaredConstructor(
275                         VolatileMethodSummary.class, TypeSafeInstructionFactory.class, SSAValueManager.class,
276                         Iterable.class);
277                     if (ctor == null) {
278                         throw new IllegalStateException("Canot find the constructor of " + this.abstractAndroidModel);
279                     }
280                     return (AbstractAndroidModel) ctor.newInstance(body, insts, paramManager, entryPoints);
281                 } catch (java.lang.InstantiationException e) {
282                     throw new IllegalStateException(e);
283                 } catch (java.lang.IllegalAccessException e) {
284                     throw new IllegalStateException(e);
285                 } catch (java.lang.reflect.InvocationTargetException e) {
286                     throw new IllegalStateException(e);
287                 } catch (java.lang.NoSuchMethodException e) {
288                     throw new IllegalStateException(e);
289                 }
290             }
291         }

329         /**
330          *  Set the package of the analyzed application.
331          *
332          *  Setting the package of the application is completely optional. However if you do it it helps
333          *  determining whether an Intent has an internal target.
334          *
335          *  If a AndroidManifest.xml is read this getts set automaticly.
336          *
337          *  @param  pack    The package of the analyzed application
338          *  @throws IllegalArgumentException if the package has already been set and the value of the
339          *      packages differ. Or if the given package is null.
340          */
341         public void setPackage(String pack) {
342             if (pack == null) {
343                 throw new IllegalArgumentException("Setting the package to null is disallowed.");
344             }
345             if ((! pack.startsWith("L") || pack.contains("."))) {
346                 pack = StringStuff.deployment2CanonicalTypeString(pack);
347             }
348             if (this.pack == null) {
349                 logger.info("Setting the package to {}", pack);
350                 this.pack = pack;
351             } else if (!(this.pack.equals(pack))) {
352                 throw new IllegalArgumentException("The already set package " + this.pack + " and " + pack +
353                         " differ. You can only set pack once.");
354             }
355         }

415         /**
416          *  Set more information to an Intent.
417          *
418          *  You can call this method multiple times on the same Intent as long as you don't lower the associated
419          *  information. So if you only want to change a specific value of it it is more safe to retrieve the Intent
420          *  first and union it yourself before registering it.
421          *
422          *  @param  intent  An Intent with more or the same information as known to the system before.
423          *  @throws IllegalArgumentException if you lower the information on an already registered Intent or the 
424          *      information is incompatible.
425          *  @see    registerIntentForce()
426          */
427         public void registerIntent(Intent intent) {
428             if (overrideIntents.containsKey(intent)) {
429                 final Intent original = overrideIntents.get(intent);
430                 final Intent.IntentType oriType = original.getType();
431                 final Intent.IntentType newType = intent.getType();
432     
433                 if ((newType == Intent.IntentType.UNKNOWN_TARGET) && (oriType != Intent.IntentType.UNKNOWN_TARGET)) {
434                     throw new IllegalArgumentException("You are lowering information on the Intent-Target of the " +
435                             "Intent " + original + " from " + oriType + " to " + newType + ". Use registerIntentForce()" +
436                             "If you are sure you want to do this!");
437                 } else if (oriType != newType) {
438                     throw new IllegalArgumentException("You are changing the Intents target to a contradicting one! " +
439                             newType + "(new) is incompatible to " + oriType + "(before). On Intent " + intent +
440                             ". Use registerIntentForce() if you are sure you want to do this!");
441                 }
442     
443                 // TODO: Add actual target to the Intent and compare these?
444                 registerIntentForce(intent);
445             } else {
446                 registerIntentForce(intent);
447             }
448         }

466         /**
467          *  Override target of an Intent (or add an alias).
468          *
469          *  Use this for example to add an internal target to an implicit Intent, add an alias to an Intent, 
470          *  resolve a System-name to an internal Intent, do weird stuff...
471          *
472          *  None of the Intents have to be registered before. However if the source is registered you may not 
473          *  lower information on it.
474          *
475          *  Currently only one target to an Intent is supported! If you want to emulate multiple Targets you
476          *  may have to add a synthetic class and register it as an Intent. If the target is not set to Internal
477          *  multiple targets may implicitly emulated. See the Documentation for these targets for detail.
478          *
479          *  If you only intend to make an Intent known see {@link #registerIntent(Intent)}.
480          *
481          *  @param  from    the Intent to override
482          *  @param  to      the new Intent to resolve once 'from' is seen
483          *  @see    setOverrideForce()
484          *  @throws IllegalArgumentException if you override an Intent with itself
485          */
486         public void setOverride(Intent from, Intent to) {
487             if (from == null) {
488                 throw new IllegalArgumentException("The Intent given as 'from' is null");
489             }
490             if (to == null) {
491                 throw new IllegalArgumentException("The Intent given as 'to' is null");
492             }
493             if (from.equals(to)) {
494                 throw new IllegalArgumentException("You cannot override an Intent with itself! If you want to " +
495                         "alter Information on an Intent use registerIntent (you may register it multiple times).");
496             }
497     
498             if (overrideIntents.containsKey(from)) {
499                 final Intent ori = overrideIntents.get(from);
500                 final Intent source;
501                 if (ori == from) {
502                     // The Intent has been registered before. Set the registered variant as source so Information
503                     // that may have been altered is not lost. Not that it would matter now...
504                     final Intent.IntentType oriType = ori.getType();
505                     final Intent.IntentType newType = from.getType();
506                     if ((newType == Intent.IntentType.UNKNOWN_TARGET) && (oriType != Intent.IntentType.UNKNOWN_TARGET)) {
507                         // TODO: Test target resolvability
508                         source = ori;
509                     } else {
510                         source = from;
511                     }
512                 } else {
513                     source = from;
514                 }
515     
516                 // Make sure the new target is not less specific than a known override
517                 final Intent original = overrideIntents.get(to);
518                 final Intent.IntentType oriType = original.getType();
519                 final Intent.IntentType newType = to.getType();
520     
521                 if ((newType == Intent.IntentType.UNKNOWN_TARGET) && (oriType != Intent.IntentType.UNKNOWN_TARGET)) {
522                     throw new IllegalArgumentException("You are lowering information on the Intent-Target of the " +
523                             "Intent " + original + " from " + oriType + " to " + newType + ". Use setOverrideForce()" +
524                             "If you are sure you want to do this!");
525                 } else if (oriType != newType) {
526                     throw new IllegalArgumentException("You are changing the Intents target to a contradicting one! " +
527                             newType + "(new) is incompatible to " + oriType + "(before). On Intent " + to +
528                             ". Use setOverrideForce() if you are sure you want to do this!");
529                 }
530     
531                 // TODO: Check resolvable Target is not overridden with unresolvable one
532     
533                 setOverrideForce(source, to);
534             } else {
535                 setOverrideForce(from, to);
536             }
537         }

579         /**
580          *  Get Intent with applied overrides.
581          *
582          *  If there are no overrides or the Intent is not registered return it as is.
583          *  
584          *  See {@link #setOverride(Intent, Intent)}.
585          *  See {@link #registerIntent(Intent)}.
586          *
587          *  @param  intent  The intent to resolve
588          *  @return where to resolve it to or the given intent if no information is available
589          *
590          *  @todo TODO: Malicious Intent-Table could cause endless loops
591          */
592         public Intent getIntent(Intent intent) {
593             if (overrideIntents.containsKey(intent)) {
594                 Intent ret = overrideIntents.get(intent);
595                 while (!(ret.equals(intent))) {
596                     // Follow the chain of overrides
597                     if (!overrideIntents.containsKey(intent)) {
598                         logger.info("Resolved {} to {}", intent, ret);
599                         return ret;
600                     } else {
601                         logger.debug("Resolving {} hop over {}", intent, ret);
602                         final Intent old = ret;
603                         ret = overrideIntents.get(ret);
604     
605                         if (ret == old) { // Yes, ==
606                             // This is an evil hack(tm). I should fix the Intent-Table!
607                             logger.warn("Malformend Intent-Table, staying with " + ret + " for " + intent);
608                             return ret;
609                         }
610                     }
611                 }
612                 ret = overrideIntents.get(ret); // Once again to get Info set in register
613                 logger.info("Resolved {} to {}", intent, ret);
614                 return ret;
615             } else {
616                 logger.info("No information on {} hash: {}", intent, intent.hashCode());
617                 for (Intent known : overrideIntents.keySet()) {
618                     logger.debug("Known Intents: {} hash: {}", known, known.hashCode());
619                 }
620                 return intent;
621             }
622         }

665         private boolean allowIntentRerouting = true;
666     
667         /**
668          *  Controll modification of an Intents target after construction.
669          *
670          *  After an Intent has been constructed its target may be changed using functions like
671          *  setAction or setComponent. 
672          *  This setting controlls the behavior of the model on occurrence of such a function:
673          *
674          *  If set to false the Intent will be marked as unresolvable.
675          *  
676          *  If set to true the first occurrence of such a function changes the target of the
677          *  Intent unless:
678          *
679          *  * The Intent was explicit and the new action is not: The call gets ignored
680          *  * The Intent was explicit and the new target is explicit: It becomes unresolvable
681          *  * It's the second occurrence of such a function: It becomes unresolvable
682          *  * It was resolvable: It becomes unresolvable
683          *
684          *  The default is to activate this behavior.
685          *
686          *  @param  allow   Allow rerouting as described
687          *  @return previous setting
688          */
689         public boolean setAllowIntentRerouting(boolean allow) {
690             boolean prev = allowIntentRerouting;
691             allowIntentRerouting = allow;
692             return prev;
693         }
694     
695         /**
696          *  Controll modification of an Intents target after construction.
697          *
698          *  See: {@link #setAllowIntentRerouting(boolean)}. 
699          *
700          *  @return the set behavior or true, the default
701          */
702         public boolean isAllowIntentRerouting() {
703             return allowIntentRerouting;
704         }
