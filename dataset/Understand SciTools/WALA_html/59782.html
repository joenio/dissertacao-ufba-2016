84     	public static Set<IMethod> assertUserCodeReachable(CallGraph cg, Set<MethodReference> uncalled) throws InvalidClassFileException {
85     		Set<IMethod> result = HashSetFactory.make();
86     	  for(Iterator<IClass> clss = cg.getClassHierarchy().getLoader(ClassLoaderReference.Application).iterateAllClasses();
87     			clss.hasNext(); ) 
88     		{
89     			IClass cls = clss.next();
90     			if (cls.isInterface()) {
91     				continue;
92     			}
93     			if (! cls.getName().toString().startsWith("Landroid") && ! cls.getName().toString().equals("Lde/ecspride/R$styleable")) {
94     				for(IMethod m : cls.getDeclaredMethods()) {
95     					if (!m.isInit() && !m.isAbstract() && !uncalled.contains(m.getReference())) {
96     					  if (! cg.getNodes(m.getReference()).isEmpty()) {
97     					    System.err.println("found " + m);
98     					  } else {
99     					    result.add(m);
100     					  }
101     					}
102     				}
103     			}
104     		}
105         return result;
106     	}

151       public static Collection<Object[]> generateData(final URI[] androidLibs, final File androidJavaJar, final String filter) {
152         String f = walaProperties.getProperty("droidbench.root");
153         if (f == null || !new File(f).exists()) {
154           f = "/tmp/DroidBench";       
155         }
156       
157         System.err.println("Use " + f + " as droid bench root");
158         assert new File(f).exists() : "Use " + f + " as droid bench root";
159         assert new File(f + "/apk/").exists() : "Use " + f + " as droid bench root";
160         return generateData(f, androidLibs, androidJavaJar, filter);
161       }

163     	public static Collection<Object[]> generateData(String droidBenchRoot, final URI[] androidLibs, final File androidJavaJar, final String filter) {
164     	  final List<Object[]> files = new LinkedList<Object[]>();
165     	  FileUtil.recurseFiles(new VoidFunction<File>() {
166     	    @Override
167     	    public void apply(File f) {
168     	      Set<MethodReference> uncalled = uncalledFunctions.get(f.getName());
169     	      if (uncalled == null) {
170     	        uncalled = Collections.emptySet();
171     	      }
172     	      files.add(new Object[]{ androidLibs, androidJavaJar, f.getAbsolutePath(), uncalled }); 
173     	    }
174     	  }, new Predicate<File>() {
175     	    @Override
176     	    public boolean test(File t) {
177     	      return (filter == null || t.getAbsolutePath().contains(filter)) && t.getName().endsWith("apk") && ! skipTests.contains(t.getName().toString());
178     	    } }, new File(droidBenchRoot + "/apk/"));
179     	  return files;
180     	}
