11     package com.ibm.wala.cast.js.html;

13     import java.io.File;

15     import java.io.InputStreamReader;
16     import java.io.Reader;

18     import java.net.URL;

21     
22     import com.ibm.wala.cast.ir.translator.TranslatorToCAst.Error;
23     import com.ibm.wala.cast.js.html.jericho.JerichoHtmlParser;
24     import com.ibm.wala.util.collections.Pair;
25     
26     public class WebUtil { 
27     
28       public static final String preamble = "preamble.js";
29     
30       private static IHtmlParserFactory factory = new IHtmlParserFactory() {
31         @Override
32         public IHtmlParser getParser() {
33           return new JerichoHtmlParser();
34         }
35       };
36       
37       public static void setFactory(IHtmlParserFactory factory) {
38         WebUtil.factory = factory;
39       }
40     
41       /**
42        * 
43        * @param url
44        * @return a pair (S,F), where S is a set of extracted sources, and F is the
45        *         temp file holding the combined sources (or <code>null</code> if no
46        *         such file exists)
47        * @throws Error
48        */
49       public static Pair<Set<MappedSourceModule>,File> extractScriptFromHTML(URL url, boolean useDOMModel) throws Error {
50         try {
51           JSSourceExtractor extractor = useDOMModel? new DefaultSourceExtractor(): new DomLessSourceExtractor();
52           Set<MappedSourceModule> sources = extractor.extractSources(url, factory.getParser(), new IdentityUrlResolver());
53           return Pair.make(sources, extractor.getTempFile());
54         } catch (IOException e) {
55           throw new RuntimeException("trouble with " + url, e);
56         }
57       }
58       
59       public static void main(String[] args) throws MalformedURLException, Error {
60         System.err.println(extractScriptFromHTML(new URL(args[0]), Boolean.parseBoolean(args[1])));
61       }
62     
63       public static Reader getStream(URL url) throws IOException {
64         URLConnection conn = url.openConnection();
65         conn.setDefaultUseCaches(false);
66         conn.setUseCaches(false);
67       
68         return new InputStreamReader(conn.getInputStream());
69       }
70     }
