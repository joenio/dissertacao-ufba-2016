11     package com.ibm.wala.cast.ir.ssa;
12     
13     import java.util.Collection;
14     
15     import com.ibm.wala.ssa.SSAInstruction;
16     import com.ibm.wala.ssa.SSAInstructionFactory;
17     import com.ibm.wala.ssa.SymbolTable;
18     import com.ibm.wala.types.TypeReference;
19     
20     /**
21      *  An assert statement, as found in a variety of languages.  It has a
22      * use which is the value being asserted to be true.  Additionally,
23      * there is flag which denotes whether the assertion is from a
24      * specification (the usual case) or is an assertion introduced by
25      * "compilation" of whatever sort (e.g. to add assertions regarding
26      * loop conditions needed by bounded model checking).
27      *
28      * @author Julian Dolby (dolby@us.ibm.com)
29      */
30     public class AstAssertInstruction extends SSAInstruction {
31       private final int value;
32     
33       private final boolean fromSpecification;
34     
35       public AstAssertInstruction(int iindex, int value, boolean fromSpecification) {
36         super(iindex);
37         this.value = value;
38         this.fromSpecification = fromSpecification;
39       }
40     
41       @Override
42       public int getNumberOfUses() {
43         return 1;
44       }
45     
46       @Override
47       public int getUse(int i) {
48         assert i == 0;
49         return value;
50       }
51     
52       @Override
53       public SSAInstruction copyForSSA(SSAInstructionFactory insts, int[] defs, int[] uses) {
54         return ((AstInstructionFactory)insts).AssertInstruction(iindex, uses == null ? value : uses[0], fromSpecification);
55       }
56     
57       @Override
58       public String toString(SymbolTable symbolTable) {
59         return "assert " + getValueString(symbolTable, value) + " (fromSpec: " + fromSpecification + ")";
60       }
61     
62       @Override
63       public void visit(IVisitor v) {
64         ((AstInstructionVisitor) v).visitAssert(this);
65       }
66     
67       @Override
68       public int hashCode() {
69         return 2177 * value;
70       }
71     
72       @Override
73       public Collection<TypeReference> getExceptionTypes() {
74         return null;
75       }
76     
77       @Override
78       public boolean isFallThrough() {
79         return true;
80       }
81     
82       public boolean isFromSpecification() {
83         return fromSpecification;
84       }
85     }
