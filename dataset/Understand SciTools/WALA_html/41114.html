124         @Override
125         public IClass getSuperclass() {
126           boolean excludedSupertype=false;
127           for (Iterator iter = superTypeNames.iterator(); iter.hasNext();) {
128             TypeName name = (TypeName) iter.next();
129             IClass domoType = lookupClass(name);
130             if (domoType != null && !domoType.isInterface()) {
131               return domoType;
132             }
133             if (domoType == null && getClassHierarchy().getScope().getExclusions().contains(name.toString().substring(1))){
134               excludedSupertype = true;
135             }
136           }
137     
138           // The following test allows the root class to reside in source; without
139           // it, the assertion requires all classes represented by a JavaClass to
140           // have a superclass.
141     /** BEGIN Custom change: Common superclass is optional */
142           // Is no longer true in new X10 - no common object super class
143           if (existsCommonSuperclass && !getName().equals(JavaSourceLoaderImpl.this.getLanguage().getRootType().getName()) && !excludedSupertype) {
144     /** END Custom change: Common superclass is optional */
145             Assertions.UNREACHABLE("Cannot find super class for " + this + " in " + superTypeNames);
146           }
147           
148           if (excludedSupertype){
149             System.err.println("Not tracking calls through excluded superclass of " + getName() + " extends " + superTypeNames);
150           }
151           
152           return null;
153         }

155         @Override
156         public Collection<IClass> getDirectInterfaces() {
157           List<IClass> result = new ArrayList<IClass>();
158           for (Iterator iter = superTypeNames.iterator(); iter.hasNext();) {
159             TypeName name = (TypeName) iter.next();
160             IClass domoType = lookupClass(name);
161             if (domoType != null && domoType.isInterface()) {
162                 result.add(domoType);
163             }
164             if (domoType == null && !getClassHierarchy().getScope().getExclusions().contains(name.toString().substring(1))){
165               assert false : "Failed to find non-excluded interface: " + name;
166             }
167           }
168     
169           return result;
170         }

206       private Collection<Annotation> getAnnotations(CAstEntity e) {
207         Collection<CAstAnnotation> annotations = e.getAnnotations();
208         if (annotations == null || annotations.isEmpty()) {
209           return null;
210         } else {
211           Collection<Annotation> result = HashSetFactory.make();
212           for(CAstAnnotation ca : annotations) {
213             TypeName walaTypeName = toWALATypeName(ca.getType());
214             TypeReference ref = TypeReference.findOrCreate(getReference(), walaTypeName);
215             if (ca.getArguments() == null || ca.getArguments().isEmpty()) {
216               result.add(Annotation.make(ref));
217             } else {
218               Map<String,ElementValue> args = HashMapFactory.make();
219               for(Map.Entry<String, Object> a : ca.getArguments().entrySet()) {
220                 args.put(a.getKey(), new ConstantElementValue(a.getValue()));
221               }
222               result.add(Annotation.makeWithNamed(ref, args));
223             }
224           }
225           return result;
226         }

283         private TypeReference[] computeParameterTypes(CAstEntity methodEntity) {
284           TypeReference[] types;
285           CAstType.Function type = (Function) methodEntity.getType();
286           int argCount = type.getArgumentTypes().size();
287           if (isStatic()) {
288             types = new TypeReference[argCount];
289             for (int i = 0; i < argCount; i++) {
290               types[i] = TypeReference.findOrCreate(JavaSourceLoaderImpl.this.getReference(), ((CAstType) type.getArgumentTypes()
291                   .get(i)).getName());
292             }
293           } else {
294             types = new TypeReference[argCount + 1];
295             types[0] = cls.getReference();
296             for (int i = 0; i < argCount; i++) {
297               types[i + 1] = TypeReference.findOrCreate(JavaSourceLoaderImpl.this.getReference(), ((CAstType) type.getArgumentTypes()
298                   .get(i)).getName());
299             }
300           }
301     
302           return types;
303         }

310         private TypeReference[] computeExceptionTypes(CAstEntity methodEntity) {
311           CAstType.Function fType = (Function) methodEntity.getType();
312           Collection exceptionTypes = fType.getExceptionTypes();
313     
314           TypeReference[] result = new TypeReference[exceptionTypes.size()];
315           int i = 0;

317             CAstType type = (CAstType) iter.next();
318             result[i] = TypeReference.findOrCreate(JavaSourceLoaderImpl.this.getReference(), type.getName());
319           }
320     
321           return result;
322         }

391         @Override
392         public LexicalParent[] getParents() {
393           if (AstTranslator.DEBUG_LEXICAL) {
394             System.err.println(("resolving parents of " + this));
395           }
396     
397           if (lexicalInfo() == null) {
398             if (AstTranslator.DEBUG_LEXICAL)
399               System.err.println("no info");
400             return new LexicalParent[0];
401           }
402     
403           final String[] parents = lexicalInfo().getScopingParents();
404     
405           if (parents == null) {
406             if (AstTranslator.DEBUG_LEXICAL)
407               System.err.println("no parents");
408             return new LexicalParent[0];
409           }
410     
411           LexicalParent result[] = new LexicalParent[parents.length];
412     
413           for (int i = 0; i < parents.length; i++) {
414             int lastLeftParen = parents[i].lastIndexOf('(');
415             int lastQ = parents[i].lastIndexOf('/', lastLeftParen);
416             String typeName = parents[i].substring(0, lastQ);
417             final IClass cls = lookupClass(TypeName.string2TypeName(typeName));
418     
419             String sig = parents[i].substring(lastQ);
420             int nameEnd = sig.indexOf('(');
421             String nameStr = sig.substring(1, nameEnd);
422             Atom name = Atom.findOrCreateUnicodeAtom(nameStr);
423     
424             String descStr = sig.substring(nameEnd);
425             Descriptor desc = Descriptor.findOrCreateUTF8(Language.JAVA, descStr);
426     
427             final Selector sel = new Selector(name, desc);
428     
429             if (AstTranslator.DEBUG_LEXICAL)
430               System.err.println(("get " + typeName + ", " + nameStr + ", " + descStr));
431     
432             final int hack = i;
433             result[i] = new LexicalParent() {
434               @Override
435               public String getName() {
436                 return parents[hack];
437               }
438     
439               @Override
440               public AstMethod getMethod() {
441                 return (AstMethod) cls.getMethod(sel);
442               }
443             };
444     
445             if (AstTranslator.DEBUG_LEXICAL)
446               System.err.println(("parent " + result[i].getName() + " is " + result[i].getMethod()));
447           }
448     
449           return result;
450         }

453       public static int mapToInt(Collection/* <CAstQualifier> */qualifiers) {
454         int result = 0;
455         for (Iterator iter = qualifiers.iterator(); iter.hasNext();) {
456           CAstQualifier q = (CAstQualifier) iter.next();
457     
458           if (q == CAstQualifier.PUBLIC)
459             result |= ClassConstants.ACC_PUBLIC;
460           if (q == CAstQualifier.PROTECTED)
461             result |= ClassConstants.ACC_PROTECTED;
462           if (q == CAstQualifier.PRIVATE)
463             result |= ClassConstants.ACC_PRIVATE;
464           if (q == CAstQualifier.STATIC)
465             result |= ClassConstants.ACC_STATIC;
466           if (q == CAstQualifier.FINAL)
467             result |= ClassConstants.ACC_FINAL;
468           if (q == CAstQualifier.SYNCHRONIZED)
469             result |= ClassConstants.ACC_SYNCHRONIZED;
470           if (q == CAstQualifier.TRANSIENT)
471             result |= ClassConstants.ACC_TRANSIENT;
472           if (q == CAstQualifier.NATIVE)
473             result |= ClassConstants.ACC_NATIVE;
474           if (q == CAstQualifier.INTERFACE)
475             result |= ClassConstants.ACC_INTERFACE;
476           if (q == CAstQualifier.ABSTRACT)
477             result |= ClassConstants.ACC_ABSTRACT;
478           if (q == CAstQualifier.VOLATILE)
479             result |= ClassConstants.ACC_VOLATILE;
480           if (q == CAstQualifier.STRICTFP)
481             result |= ClassConstants.ACC_STRICT;
482         }
483         return result;
484       }

541       public IClass defineType(CAstEntity type, String typeName, CAstEntity owner) {
542         Collection<TypeName> superTypeNames = new ArrayList<TypeName>();
543         for (Iterator superTypes = type.getType().getSupertypes().iterator(); superTypes.hasNext();) {
544           superTypeNames.add(toWALATypeName(((CAstType) superTypes.next())));
545         }
546     
547         JavaClass javaClass = new JavaClass(typeName, superTypeNames, type.getPosition(), type.getQualifiers(), this,
548             (owner != null) ? (JavaClass) fTypeMap.get(owner) : (JavaClass) null, getAnnotations(type));
549     
550         if (getParent().lookupClass(javaClass.getName()) != null) {
551           return null;
552         }
553         
554         fTypeMap.put(type, javaClass);
555         loadedClasses.put(javaClass.getName(), javaClass);
556         return javaClass;
557       }
