79         public TranslatorAssertions(Object[][] data) {
80           for (int dataIndex = 0; dataIndex < data.length; dataIndex++) {
81             Object[] entry = data[dataIndex];
82             String clsName = (String) entry[0];
83             this.classes.add(clsName);
84     
85             String superName = (String) entry[1];
86             this.supers.put(clsName, superName);
87     
88             String[] instanceFields = (String[]) entry[2];
89             if (instanceFields != null) {
90               for (int i = 0; i < instanceFields.length; i++) {
91                 this.instanceFields.add(Pair.make(clsName, instanceFields[i]));
92               }
93             }
94     
95             String[] staticFields = (String[]) entry[3];
96             if (staticFields != null) {
97               for (int i = 0; i < staticFields.length; i++) {
98                 this.staticFields.add(Pair.make(clsName, staticFields[i]));
99               }
100             }
101     
102             Pair[] instanceMethods = (Pair[]) entry[4];
103             if (instanceMethods != null) {
104               for (int i = 0; i < instanceMethods.length; i++) {
105                 this.instanceMethods.put(Pair.make(clsName, instanceMethods[i].fst), instanceMethods[i].snd);
106               }
107             }
108     
109             Pair[] staticMethods = (Pair[]) entry[5];
110             if (staticMethods != null) {
111               for (int i = 0; i < staticMethods.length; i++) {
112                 this.staticMethods.put(Pair.make(clsName, staticMethods[i].fst), staticMethods[i].snd);
113               }
114             }
115           }
116         }

139       protected void dump(ClassHierarchy cha) {
140         for (Iterator<?> clss = cha.iterator(); clss.hasNext();) {
141           IClass cls = (IClass) clss.next();
142           System.err.println(("class " + cls));
143           for (Iterator<?> flds = cls.getDeclaredInstanceFields().iterator(); flds.hasNext();) {
144             IField fld = (IField) flds.next();
145             System.err.println(("instance field " + fld));
146           }
147           for (Iterator<?> flds = cls.getDeclaredStaticFields().iterator(); flds.hasNext();) {
148             IField fld = (IField) flds.next();
149             System.err.println(("static field " + fld));
150           }
151           for (Iterator<?> mths = cls.getDeclaredMethods().iterator(); mths.hasNext();) {
152             IMethod mth = (IMethod) mths.next();
153             if (mth.isStatic())
154               System.err.print("static ");
155             System.err.println(("method " + mth + " with " + mth.getNumberOfParameters() + " parameters"));
156             for (int i = 0; i < mth.getNumberOfParameters(); i++) {
157               System.err.println(("param " + i + ": " + mth.getParameterType(i)));
158             }
159             System.err.println(factory.makeIR(mth, Everywhere.EVERYWHERE, options));
160           }
161         }
162       }

164       public void checkAssertions(ClassHierarchy cha, TranslatorAssertions assertions) {
165         Set<String> classes = assertions.getClasses();
166         Map<String, String> supers = assertions.getSuperClasses();
167         Set<Pair<String, String>> instanceFields = assertions.getInstanceFields();
168         Set<Pair<String, String>> staticFields = assertions.getStaticFields();
169         Map<Pair<String, Object>, Object> instanceMethods = assertions.getInstanceMethods();
170         Map<Pair<String, Object>, Object> staticMethods = assertions.getStaticMethods();
171     
172         int clsCount = 0;
173         for (Iterator<?> clss = cha.iterator(); clss.hasNext();) {
174           IClass cls = (IClass) clss.next();
175           clsCount++;
176           Assert.assertTrue("found class " + cls.getName().toString(), classes.contains(cls.getName().toString()));
177     
178           if (cls.getSuperclass() == null) {
179             Assert.assertTrue(cls.getName() + " has no superclass", supers.get(cls.getName()) == null);
180           } else {
181             Assert.assertTrue("super of " + cls.getName() + " is " + cls.getSuperclass().getName(), supers
182                 .get(cls.getName().toString()).equals(cls.getSuperclass().getName().toString()));
183           }
184     
185           for (Iterator<?> flds = cls.getDeclaredInstanceFields().iterator(); flds.hasNext();) {
186             IField fld = (IField) flds.next();
187             Assert.assertTrue(cls.getName() + " has field " + fld.getName(), instanceFields.contains(Pair.make(
188                 cls.getName().toString(), fld.getName().toString())));
189           }
190     
191           for (Iterator<?> flds = cls.getDeclaredStaticFields().iterator(); flds.hasNext();) {
192             IField fld = (IField) flds.next();
193             Assert.assertTrue(cls.getName() + " has static field " + fld.getName(), staticFields.contains(Pair.make(cls.getName()
194                 .toString(), fld.getName().toString())));
195           }
196     
197           for (Iterator<?> mths = cls.getDeclaredMethods().iterator(); mths.hasNext();) {
198             IMethod mth = (IMethod) mths.next();
199             Integer np = new Integer(mth.getNumberOfParameters());
200             Pair<String, String> key = Pair.make(cls.getName().toString(), mth.getName().toString());
201     
202             if (mth.isStatic()) {
203               Assert.assertTrue(cls.getName() + " has static method " + mth.getName(), staticMethods.containsKey(key));
204               Assert.assertTrue(cls.getName() + "::" + mth.getName() + " has " + np + " parameters", staticMethods.get(key).equals(np));
205             } else {
206               Assert.assertTrue(cls.getName() + " has method " + mth.getName(), instanceMethods.containsKey(key));
207               Assert.assertTrue(cls.getName() + "::" + mth.getName() + " has " + np + " parameters", instanceMethods.get(key)
208                   .equals(np));
209             }
210           }
211         }
212     
213         Assert.assertTrue("want " + classes.size() + " classes", clsCount == classes.size());
214       }

216       protected void testInternal(String[] args, TranslatorAssertions assertions) throws Exception {
217         String testPath = getTestPath();
218         SourceFileModule[] fileNames = new SourceFileModule[args.length];
219         for (int i = 0; i < args.length; i++) {
220           if (new File(args[i]).exists()) {
221             fileNames[i] = CAstCallGraphUtil.makeSourceModule(new File(args[i]).toURI().toURL(), args[i]);
222           } else if (new File(testPath + args[i]).exists()) {
223             fileNames[i] = CAstCallGraphUtil.makeSourceModule(new File(testPath + args[i]).toURI().toURL(), args[i]);
224           } else {
225             URL url = getClass().getClassLoader().getResource(args[i]);
226             fileNames[i] = CAstCallGraphUtil.makeSourceModule(url, args[i]);
227           }
228           Assert.assertTrue(args[i], fileNames[i] != null);
229         }
230     
231         ClassHierarchy cha = runTranslator(fileNames);
232         dump(cha);
233         if (assertions != null) {
234           checkAssertions(cha, assertions);
235         } else {
236           System.err.println(("WARNING: no assertions for " + getClass()));
237         }
238       }
239     
240       protected void testInternal(String arg, TranslatorAssertions assertions) {
241         try {
242           testInternal(new String[] { arg }, assertions);
243         } catch (Exception e) {
244           e.printStackTrace();
245           Assert.assertTrue(e.toString(), false);
246         }
