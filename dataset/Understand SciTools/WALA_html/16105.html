11     package com.ibm.wala.ipa.summaries;
12     
13     import java.util.HashMap;
14     import java.util.HashSet;
15     import java.util.Map;
16     import java.util.Set;

18     import com.ibm.wala.classLoader.IClass;
19     import com.ibm.wala.classLoader.IMethod;
20     import com.ibm.wala.classLoader.SyntheticMethod;
21     import com.ibm.wala.ipa.cha.IClassHierarchy;
22     import com.ibm.wala.types.MemberReference;
23     import com.ibm.wala.types.MethodReference;
24     import com.ibm.wala.types.TypeReference;
25     import com.ibm.wala.util.collections.HashMapFactory;
26     import com.ibm.wala.util.collections.HashSetFactory;

29     /**
30      * "Non-standard" bypass rules to use during call graph construction.
31      * 
32      * Normally, the method bypass rules replace the IMethod that is resolved by other means, via the getBypass() method. However, the
33      * bypass rules can be invoked even before resolving the target of a call, by checking the intercept rules.
34      */
35     public class MethodBypass {
36     
37       static final boolean DEBUG = false;
38     
39       /**
40        * Method summaries collected for methods. Mapping Object -> MethodSummary where Object is either a
41        * <ul>
42        * <li>MethodReference
43        * <li>TypeReference
44        * <li>Atom (package name)
45        * </ul>
46        */
47       private final Map methodSummaries;
48     
49       /**
50        * Set of TypeReferences which are marked "allocatable"
51        */
52       private final Set allocatable;
53     
54       /**
55        * Governing class hierarchy.
56        */
57       private final IClassHierarchy cha;
58     
59       /**
60        * Mapping from MethodReference -> SyntheticMethod
61        */
62       final private HashMap<MethodReference, SummarizedMethod> syntheticMethods = HashMapFactory.make();
63     
64       /**
65        * Set of method references that have been considered already.
66        */
67       final private HashSet<MethodReference> considered = HashSetFactory.make();
68     
69       public MethodBypass(Map methodSummaries, Set allocatable, IClassHierarchy cha) {
70         this.methodSummaries = methodSummaries;
71         this.allocatable = allocatable;
72         this.cha = cha;
73       }
74     
75       /**
76        * Lookup bypass rules based on a method reference only.
77        * 
78        * Method getBypass.
79        * 
80        * @param m
81        * @return Object
82        */
83       private SyntheticMethod getBypass(MethodReference m) {
84         if (DEBUG) {
85           System.err.println(("MethodBypass.getBypass? " + m));
86         }
87         SyntheticMethod result = findOrCreateSyntheticMethod(m);
88         if (result != null) {
89           return result;
90         }
91         // first lookup failed ... try resolving target via CHA and try again.
92         m = resolveTarget(m);
93         return findOrCreateSyntheticMethod(m);
94       }
95     
96       /**
97        * @param m a method reference
98        * @return a SyntheticMethod corresponding to m; or null if none is available.
99        */
100       private SyntheticMethod findOrCreateSyntheticMethod(MethodReference m) {
101         if (considered.contains(m)) {
102           return syntheticMethods.get(m);
103         } else {
104           considered.add(m);
105           MethodSummary summ = findSummary(m);
106           if (summ != null) {
107             TypeReference T = m.getDeclaringClass();
108             IClass c = cha.lookupClass(T);
109             assert c != null : "null class for " + T;
110             SummarizedMethod n = new SummarizedMethod(m, summ, c);
111             syntheticMethods.put(m, n);
112             return n;
113           }
114           return null;
115         }
116       }
117     
118       private MethodSummary findSummary(MemberReference m) {
119         MethodSummary result = (MethodSummary) methodSummaries.get(m);
120         if (result != null) {
121           if (DEBUG) {
122             System.err.println(("findSummary succeeded: " + m));
123           }
124           return result;
125         }
126     
127         // try the class instead.
128         TypeReference t = m.getDeclaringClass();
129         result = (MethodSummary) methodSummaries.get(t);
130         if (result != null) {
131           if (DEBUG) {
132             System.err.println(("findSummary succeeded: " + t));
133           }
134           return result;
135         }
136         if (t.isArrayType())
137           return null;
138     
139         // finally try the package.
140         Atom p = extractPackage(t);
141         result = (MethodSummary) methodSummaries.get(p);
142         if (result != null) {
143           if (DEBUG) {
144             System.err.println(("findSummary succeeded: " + p));
145           }
146           return result;
147         } else {
148           if (DEBUG) {
149             System.err.println(("findSummary failed: " + m));
150           }
151           return result;
152         }
153       }
154     
155       /**
156        * Method getBypass. check to see if a call to the receiver 'target' should be redirected to a different receiver.
157        * 
158        * @param target
159        * @return Object
160        * @throws IllegalArgumentException if target is null
161        */
162       public SyntheticMethod getBypass(IMethod target) {
163         if (target == null) {
164           throw new IllegalArgumentException("target is null");
165         }
166         return getBypass(target.getReference());
167       }
168     
169       /**
170        * Method extractPackage.
171        * 
172        * @param type
173        * @return Atom that represents the package name, or null if this is the unnamed package.
174        */
175       private Atom extractPackage(TypeReference type) {
176         String s = type.getName().toString();
177         int index = s.lastIndexOf('/');
178         if (index == -1) {
179           return null;
180         } else {
181           s = s.substring(0, index);
182           return Atom.findOrCreateAsciiAtom(s);
183         }
184       }
185     
186       protected IClassHierarchy getClassHierarchy() {
187         return cha;
188       }
189     
190       protected MethodReference resolveTarget(MethodReference target) {
191         IMethod m = getClassHierarchy().resolveMethod(target);
192         if (m != null) {
193           if (DEBUG) {
194             System.err.println(("resolveTarget: resolved to " + m));
195           }
196           target = m.getReference();
197         }
198         return target;
199       }
200     
201       /**
202        * Are we allowed to allocate (for analysis purposes) an instance of a given type? By default, the answer is yes iff T is not
203        * abstract. However, subclasses and summaries can override this to allow "special" abstract classes to be allocatable as well.
204        * 
205        * @throws IllegalArgumentException if klass is null
206        * 
207        */
208       public boolean isAllocatable(IClass klass) {
209         if (klass == null) {
210           throw new IllegalArgumentException("klass is null");
211         }
212         if (!klass.isAbstract() && !klass.isInterface()) {
213           return true;
214         } else {
215           return allocatable.contains(klass.getReference());
216         }
217       }
218     }
