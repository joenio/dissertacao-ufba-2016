38           List<Module> s = MapUtil.findOrCreateList(modules, loader);
39           Iterator<FileModule> htmlPages = new EclipseSourceDirectoryTreeModule(p, excludePaths, "html").getEntries();
40           while (htmlPages.hasNext()) {
41             FileModule htmlPage = htmlPages.next();
42             Set<MappedSourceModule> scripts;
43             String urlString = "file://" + htmlPage.getAbsolutePath();
44             try {
45               scripts = WebUtil.extractScriptFromHTML(new URL(urlString), true).fst;
46               s.addAll(scripts);
47             } catch (MalformedURLException e1) {
48               assert false : "internal error constructing URL " + urlString;
49             } catch (Error e1) {
50               System.err.print("skipping " + htmlPage.getAbsolutePath() + ": " + e1.warning);
51             } 
52           }
