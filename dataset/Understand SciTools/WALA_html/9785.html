18     package com.ibm.wala.sourcepos;
19     
20     import java.util.LinkedList;
21     
22     /**
23      * This class represents the flags which a entry in the CharacterRangeTable can
24      * have. The flags are bitwise ORed.
25      * 
26      * @see CRTData
27      * @see CRTable
28      * @author Siegfried Weber
29      * @author Juergen Graf <juergen.graf@gmail.com>
30      */
31     public final class CRTFlags {
32     
33       /** Stores the value of the {@code CRT_SOURCE_INFO} flag. */
34       static final short CRT_SOURCE_INFO = 0x0200;
35     
36       /** Stores the names of the flags. */
37       private static final String[] flagNames = { "CRT_STATEMENT", // 0x0001
38           "CRT_BLOCK", // 0x0002
39           "CRT_ASSIGNMENT", // 0x0004
40           "CRT_FLOW_CONTROLLER", // 0x0008
41           "CRT_FLOW_TARGET", // 0x0010
42           "CRT_INVOKE", // 0x0020
43           "CRT_CREATE", // 0x0040
44           "CRT_BRANCH_TRUE", // 0x0080
45           "CRT_BRANCH_FALSE", // 0x0100
46           "CRT_SOURCE_INFO" }; // 0x0200
47     
48       private static final String WARN_INVALID_FLAG = "Error at CRT entry %1$s: invalid flag %2$s";
49     
50       /** Stores the flags. */
51       private final short flags;
52     
53       /**
54        * Creates a new instance of CRTFlags.
55        * 
56        * @param flags
57        *          the flags
58        * @throws InvalidCRTDataException
59        *           An InvalidCRTDataException is thrown if the flags are not valid.
60        */
61       CRTFlags(short flags) throws InvalidCRTDataException {
62         this.flags = flags;
63         if (!isFlagValid())
64           throw new InvalidCRTDataException(WARN_INVALID_FLAG, Integer.toHexString(flags));
65       }
66     
67       /**
68        * Returns the flag names of this instance.
69        * 
70        * @return An array of Strings containing the flag names.
71        */
72       public final String[] getFlagNames() {
73         LinkedList<String> names = new LinkedList<String>();
74         int index = 0;
75         short tFlags = flags;
76         while (tFlags > 0) {
77           if (tFlags % 2 == 1) {
78             if (index < flagNames.length)
79               names.add(flagNames[index]);
80             else {
81               // assert false
82               // because exception was thrown in the constructor.
83               names.add("UNKNOWN (" + Integer.toHexString(2 << index) + ")");
84             }
85           }
86           tFlags >>= 1;
87           ++index;
88         }
89         return names.toArray(new String[names.size()]);
90       }
91     
92       /**
93        * Tests whether the flags are valid.
94        * 
95        * @return whether the flags are valid.
96        */
97       private boolean isFlagValid() {
98         return 0 < flags && flags < 2 << flagNames.length - 1;
99       }
100     
101     }
