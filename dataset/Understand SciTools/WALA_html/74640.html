39         @Override
40         @SuppressWarnings("unchecked")
41         public CAstNode makeNode(int kind, CAstNode children[]) {
42           if (kind == NAME_ASSERTION_SINGLE || kind == NAME_ASSERTION_MULTI) {
43             assert children.length == 2;
44             assert children[0].getValue() instanceof String;
45             @SuppressWarnings("unused")
46             CAstNode result = children[1];
47             if (kind == NAME_ASSERTION_SINGLE) {
48               testNameMap.put(children[0].getValue(), children[1]);
49             } else {
50               if (!testNameMap.containsKey(children[0].getValue())) {
51                 testNameMap.put(children[0].getValue(), new ArrayList());
52               }
53     
54               ((List) testNameMap.get(children[0].getValue())).add(children[1]);
55             }
56             return children[1];
57           } else {
58             return super.makeNode(kind, children);
59           }
60         }
61     
62       }

64       private void test(CAstPattern p, CAstNode n, Map names) {
65         System.err.println(("testing pattern " + p));
66         System.err.println(("testing with input " + CAstPrinter.print(n)));
67     
68         if (names == null) {
69           Assert.assertFalse(p.match(n, null));
70         } else {
71           Segments s = CAstPattern.match(p, n);
72           Assert.assertTrue(s != null);
73           for (Iterator ns = names.keySet().iterator(); ns.hasNext();) {
74             String nm = (String) ns.next();
75             Object o = names.get(nm);
76             if (o instanceof CAstNode) {
77               System.err.println(("found " + CAstPrinter.print(s.getSingle(nm)) + " for " + nm));
78               Assert.assertTrue("for name " + nm + ": expected " + names.get(nm) + " but got " + s.getSingle(nm), names.get(nm).equals(
79                   s.getSingle(nm)));
80             } else {
81               for (Iterator cs = s.getMultiple(nm).iterator(); cs.hasNext();) {
82                 System.err.println(("found " + CAstPrinter.print((CAstNode) cs.next()) + " for " + nm));
83               }
84               Assert.assertTrue("for name " + nm + ": expected " + names.get(nm) + " but got " + s.getMultiple(nm), names.get(nm)
85                   .equals(s.getMultiple(nm)));
86             }
87           }
88         }
89       }
