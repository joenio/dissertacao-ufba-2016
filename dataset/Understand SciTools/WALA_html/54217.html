15     package com.ibm.wala.cast.ir.translator;
16     
17     import java.util.Collection;
18     import java.util.HashSet;
19     import java.util.Set;
20     
21     import com.ibm.wala.cast.tree.CAstEntity;
22     import com.ibm.wala.cast.tree.CAstQualifier;
23     import com.ibm.wala.cast.tree.CAstType;
24     import com.ibm.wala.util.debug.Assertions;
25     
26     class AbstractFieldEntity extends AbstractDataEntity {
27       private final String name;
28     
29       private final Set<CAstQualifier> modifiers;
30     
31       private final CAstEntity declaringClass;
32     
33       public AbstractFieldEntity(String name, Set<CAstQualifier> modifiers, boolean isStatic, CAstEntity declaringClass) {
34         this.name = name;
35         this.declaringClass = declaringClass;
36     
37         this.modifiers = new HashSet<CAstQualifier>();
38         if (modifiers != null) {
39           this.modifiers.addAll(modifiers);
40         }
41         if (isStatic) {
42           this.modifiers.add(CAstQualifier.STATIC);
43         }
44       }
45     
46       @Override
47       public String toString() {
48         return "field " + name + " of " + declaringClass.getName();
49       }
50     
51       @Override
52       public int getKind() {
53         return FIELD_ENTITY;
54       }
55     
56       @Override
57       public String getName() {
58         return name;
59       }
60     
61       @Override
62       public CAstType getType() {
63         Assertions.UNREACHABLE();
64         return null;
65       }
66     
67       @Override
68       public Collection<CAstQualifier> getQualifiers() {
69         return modifiers;
70       }
71     }
