13     import java.lang.reflect.Constructor;
14     import java.lang.reflect.InvocationTargetException;
15     
16     
17     /**
18      * Test of Constructor.newInstance
19      */
20     public class Reflect22 {
21       public static void main(String[] args) throws ClassNotFoundException, SecurityException,
22           NoSuchMethodException, IllegalAccessException, InstantiationException,
23           IllegalArgumentException, InvocationTargetException {
24         Class helperClass = Class.forName("reflection.Helper");
25         Constructor[] constrs = helperClass.getDeclaredConstructors();
26         for (Constructor constr : constrs) {
27           if (constr.getParameterTypes().length == 1) {
28             Class paramType = constr.getParameterTypes()[0];
29             if (paramType.getName().equals("java.lang.Integer")) {
30               Integer i = new Integer(1);
31               Object[] initArgs = new Object[]{i};
32               constr.newInstance(initArgs);
33               break;
34             }
35           }
36         }
37       }
38     }
