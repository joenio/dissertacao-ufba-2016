47       protected static String getClasspathEntry(String elt) {
48         for (String s : System.getProperty("java.class.path").split(File.pathSeparator)) {
49           if (s.indexOf(elt) >= 0) {
50             File e = new File(s);
51              Assert.assertTrue(elt + " expected to exist", e.exists());
52             if (e.isDirectory() && !s.endsWith("/")) {
53               s = s + "/";
54             }
55             return s;
56           }
57         }
58         Assert.assertFalse("cannot find " + elt, true);
59         return null;
60       }

68       protected void instrument(String testJarLocation) throws IOException, ClassNotFoundException, InvalidClassFileException, FailureException {
69         if (! instrumentedJarBuilt) {
70           System.err.println("core data jar to instrument: " + testJarLocation);
71           
72           if (new File(instrumentedJarLocation).exists()) {
73             assert new File(instrumentedJarLocation).delete();
74           }
75           
76           String rtJar = null;
77           for(String jar : WalaProperties.getJ2SEJarFiles()) {
78             if (jar.endsWith("rt.jar") || jar.endsWith("classes.jar")) {
79               rtJar = jar;
80             }
81           }
82           
83           OfflineDynamicCallGraph.main(
84               rtJar == null?
85                 new String[]{testJarLocation, "-o", instrumentedJarLocation}:
86                 new String[]{testJarLocation, "-o", instrumentedJarLocation, "--rt-jar", rtJar});
87           Assert.assertTrue("expected to create /tmp/test.jar", new File(instrumentedJarLocation).exists());   
88           instrumentedJarBuilt = true;
89         }
90       }

92       protected void run(String mainClass, String exclusionsFile, String... args) throws IOException, ClassNotFoundException, SecurityException, NoSuchMethodException, IllegalArgumentException, IllegalAccessException, InvocationTargetException, InterruptedException {
93         Project p = new Project();
94         p.setBaseDir(new File(System.getProperty("java.io.tmpdir")));
95         p.init();
96         p.fireBuildStarted();
97     
98         Java childJvm = new Java();
99         childJvm.setTaskName("test_" + mainClass.replace('.', '_'));
100         childJvm.setClasspath(new Path(p, getClasspathEntry("com.ibm.wala.shrike") + ":" +  getClasspathEntry("com.ibm.wala.util") + ":" +  instrumentedJarLocation));
101         childJvm.setClassname(mainClass);
102     
103         String jvmArgs = "-noverify -Xmx500M -DdynamicCGFile=" + cgLocation + " -DdynamicCGHandleMissing=true";
104         if (exclusionsFile != null) {
105           File tmpFile = TemporaryFile.urlToFile("exclusions.txt", getClass().getClassLoader().getResource(exclusionsFile));
106           jvmArgs += " -DdynamicCGFilter=" + tmpFile.getCanonicalPath();
107         }
108         childJvm.setJvmargs(jvmArgs);
109         
110         StringBuffer argsStr = new StringBuffer();
111         for(String a : args) {
112           argsStr.append(a).append(" ");
113         }
114         childJvm.setArgs(argsStr.toString());
115         
116         childJvm.setFailonerror(true);
117         childJvm.setFork(true);
118         
119         if (new File(cgLocation).exists()) {
120           new File(cgLocation).delete();
121         }
122         
123         childJvm.init();
124         String commandLine = childJvm.getCommandLine().toString();
125         System.err.println(commandLine);
126         Process x = Runtime.getRuntime().exec(commandLine);
127         x.waitFor();
128         
129         Assert.assertTrue("expected to create call graph", new File(cgLocation).exists());
130       }

137           return MethodReference.findOrCreate(TypeReference.findOrCreate(ClassLoaderReference.Application, "L" + calleeClass), Selector.make(calleeMethod));

144       protected void checkEdges(CallGraph staticCG, Predicate<MethodReference> filter) throws IOException {
145         final Set<Pair<CGNode,CGNode>> edges = HashSetFactory.make();
146         check(staticCG, new EdgesTest() {
147           @Override
148           public void edgesTest(CallGraph staticCG, CGNode caller, MethodReference calleeRef) {
149               Set<CGNode> nodes = staticCG.getNodes(calleeRef);
150               Assert.assertEquals("expected one node for " + calleeRef, 1, nodes.size());
151               CGNode callee = nodes.iterator().next();
152             
153               Assert.assertTrue("no edge for " + caller + " --> " + callee, staticCG.getPossibleSites(caller, callee).hasNext());
154               Pair<CGNode,CGNode> x = Pair.make(caller, callee);
155               if (! edges.contains(x)) {
156                 edges.add(x);
157                 System.err.println("found expected edge " + caller + " --> " + callee);
158               }
159           }
160         }, filter);
161       }

167       protected void checkNodes(CallGraph staticCG, Predicate<MethodReference> filter) throws IOException {
168         final Set<MethodReference> notFound = HashSetFactory.make();
169         check(staticCG, new EdgesTest() {
170           @Override
171           public void edgesTest(CallGraph staticCG, CGNode caller, MethodReference callee) {
172             boolean checkForCallee = !staticCG.getNodes(callee).isEmpty();
173             if (!checkForCallee) {
174               notFound.add(callee);
175             } else {
176               System.err.println("found expected node " + callee);
177             }
178           }
179         }, filter);
180         
181         Assert.assertTrue("could not find " + notFound, notFound.isEmpty());
182       }

184       protected void check(CallGraph staticCG, EdgesTest test, Predicate<MethodReference> filter) throws IOException {
185         BufferedReader dynamicEdgesFile = new BufferedReader(new InputStreamReader(new GZIPInputStream(new FileInputStream(cgLocation))));
186         String line;
187         int lines = 0;
188         loop: while ((line = dynamicEdgesFile.readLine()) != null) {
189           lines++;
190           StringTokenizer edge = new StringTokenizer(line, "\t");
191           
192           CGNode caller;
193           String callerClass = edge.nextToken();
194           if ("root".equals(callerClass)) {
195             caller = staticCG.getFakeRootNode();
196           } else if ("clinit".equals(callerClass)) {
197               caller = staticCG.getFakeWorldClinitNode();
198           } else if ("callbacks".equals(callerClass)) {
199               continue loop;
200           } else {
201             String callerMethod = edge.nextToken();
202             MethodReference callerRef = MethodReference.findOrCreate(TypeReference.findOrCreate(ClassLoaderReference.Application, "L" + callerClass), Selector.make(callerMethod));
203             Set<CGNode> nodes = staticCG.getNodes(callerRef);
204             if (! filter.test(callerRef)) {
205               continue loop;
206             }
207             Assert.assertEquals(1, nodes.size());
208             caller = nodes.iterator().next();
209           }
210           
211           String calleeClass = edge.nextToken();
212           String calleeMethod = edge.nextToken();
213           MethodReference callee = callee(calleeClass, calleeMethod);
214           if (! filter.test(callee)) {
215             continue loop;
216           }
217           test.edgesTest(staticCG, caller, callee);
218         }
219         
220         dynamicEdgesFile.close();
221         
222         Assert.assertTrue("more than one edge", lines > 0);
223       }
