23     
24       public static <T> Collection<T> inferRoots(Graph<T> g){
25         if (g == null) {
26           throw new IllegalArgumentException("g is null");
27         }
28         HashSet<T> s = HashSetFactory.make();
29         for (Iterator<? extends T> it = g.iterator(); it.hasNext();) {
30           T node = it.next();
31           if (g.getPredNodeCount(node) == 0) {
32             s.add(node);
33           }
34         }
35         return s;
36       }
37     }
