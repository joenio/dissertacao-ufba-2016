31     /**
32      * a {@link HeapModel} that delegates to another except for pointer keys representing <code>this</code> parameters of methods, for
33      * which it returns a {@link FilteredPointerKey} for the type of the parameter
34      * 
35      * @see {@link DemandRefinementPointsTo}
36      * @author manu
37      * 
38      */
39     class ThisFilteringHeapModel implements HeapModel {
40     
41       private final HeapModel delegate;
42     
43       private final IClassHierarchy cha;
44     
45       @Override
46       public IClassHierarchy getClassHierarchy() {
47         return delegate.getClassHierarchy();
48       }
49     
50       @Override
51       public FilteredPointerKey getFilteredPointerKeyForLocal(CGNode node, int valueNumber, TypeFilter filter) {
52         return delegate.getFilteredPointerKeyForLocal(node, valueNumber, filter);
53       }
54     
55       @Override
56       public InstanceKey getInstanceKeyForAllocation(CGNode node, NewSiteReference allocation) {
57         return delegate.getInstanceKeyForAllocation(node, allocation);
58       }
59     
60       @Override
61       public InstanceKey getInstanceKeyForMetadataObject(Object obj, TypeReference objType) {
62         return delegate.getInstanceKeyForMetadataObject(obj, objType);
63       }
64     
65       @Override
66       public InstanceKey getInstanceKeyForConstant(TypeReference type, Object S) {
67         return delegate.getInstanceKeyForConstant(type, S);
68       }
69     
70       @Override
71       public InstanceKey getInstanceKeyForMultiNewArray(CGNode node, NewSiteReference allocation, int dim) {
72         return delegate.getInstanceKeyForMultiNewArray(node, allocation, dim);
73       }
74     
75       @Override
76       public InstanceKey getInstanceKeyForPEI(CGNode node, ProgramCounter instr, TypeReference type) {
77         return delegate.getInstanceKeyForPEI(node, instr, type);
78       }
79     
80       @Override
81       public PointerKey getPointerKeyForArrayContents(InstanceKey I) {
82         return delegate.getPointerKeyForArrayContents(I);
83       }
84     
85       @Override
86       public PointerKey getPointerKeyForExceptionalReturnValue(CGNode node) {
87         return delegate.getPointerKeyForExceptionalReturnValue(node);
88       }
89     
90       @Override
91       public PointerKey getPointerKeyForInstanceField(InstanceKey I, IField field) {
92         return delegate.getPointerKeyForInstanceField(I, field);
93       }
94     
95       @Override
96       public PointerKey getPointerKeyForLocal(CGNode node, int valueNumber) {
97         if (!node.getMethod().isStatic() && valueNumber == 1) {
98           return delegate.getFilteredPointerKeyForLocal(node, valueNumber, getFilter(node));
99         } else {
100           return delegate.getPointerKeyForLocal(node, valueNumber);
101         }
102       }
103     
104       private FilteredPointerKey.TypeFilter getFilter(CGNode target) {
105         FilteredPointerKey.TypeFilter filter = (FilteredPointerKey.TypeFilter) target.getContext().get(ContextKey.PARAMETERS[0]);
106     
107         if (filter != null) {
108           return filter;
109         } else {
110           // the context does not select a particular concrete type for the
111           // receiver.
112           IClass C = getReceiverClass(target.getMethod());
113           return new FilteredPointerKey.SingleClassFilter(C);
114         }
115       }
116     
117       /**
118        * @param method
119        * @return the receiver class for this method.
120        */
121       private IClass getReceiverClass(IMethod method) {
122         TypeReference formalType = method.getParameterType(0);
123         IClass C = cha.lookupClass(formalType);
124         if (method.isStatic()) {
125           Assertions.UNREACHABLE("asked for receiver of static method " + method);
126         }
127         if (C == null) {
128           Assertions.UNREACHABLE("no class found for " + formalType + " recv of " + method);
129         }
130         return C;
131       }
132     
133       @Override
134       public PointerKey getPointerKeyForReturnValue(CGNode node) {
135         return delegate.getPointerKeyForReturnValue(node);
136       }
137     
138       @Override
139       public PointerKey getPointerKeyForStaticField(IField f) {
140         return delegate.getPointerKeyForStaticField(f);
141       }
142     
143       @Override
144       public Iterator<PointerKey> iteratePointerKeys() {
145         return delegate.iteratePointerKeys();
146       }
147     
148       public ThisFilteringHeapModel(HeapModel delegate, IClassHierarchy cha) {
149         if (delegate == null) {
150           throw new IllegalArgumentException("delegate null");
151         }
152         this.delegate = delegate;
153         this.cha = cha;
154       }
155     
156     }
