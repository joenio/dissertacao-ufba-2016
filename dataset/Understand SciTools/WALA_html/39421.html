15     /**
16      * Abstract base class for implementations of bitvectors
17      */
18     @SuppressWarnings("rawtypes")
19     abstract public class BitVectorBase<T extends BitVectorBase> implements Cloneable, Serializable {
20     
21       protected final static boolean DEBUG = false;
22     
23       protected final static int LOG_BITS_PER_UNIT = 5;
24     
25       protected final static int BITS_PER_UNIT = 32;
26     
27       protected final static int MASK = 0xffffffff;
28     
29       protected final static int LOW_MASK = 0x1f;
30     
31       protected int bits[];

55       /**
56        * Convert bitIndex to a subscript into the bits[] array.
57        */
58       public static int subscript(int bitIndex) {
59         return bitIndex >> LOG_BITS_PER_UNIT;
60       }
61     
62       /**
63        * Clears all bits.
64        */
65       public final void clearAll() {
66         for (int i = 0; i < bits.length; i++) {
67           bits[i] = 0;
68         }
69       }

71       @Override
72       public int hashCode() {
73         int h = 1234;
74         for (int i = bits.length - 1; i >= 0;) {
75           h ^= bits[i] * (i + 1);
76           i--;
77         }
78         return h;
79       }
80     
81       /**
82        * How many bits are set?
83        */
84       public final int populationCount() {
85         int count = 0;
86         for (int i = 0; i < bits.length; i++) {
87           count += Bits.populationCount(bits[i]);
88         }
89         return count;
90       }
91     
92       public boolean isZero() {
93         int setLength = bits.length;
94         for (int i = setLength - 1; i >= 0;) {
95           if (bits[i] != 0)
96             return false;
97           i--;
98         }
99         return true;
100       }
101     
102       @Override
103       @SuppressWarnings("unchecked")
104       public Object clone() {
105         BitVectorBase<T> result = null;
106         try {
107           result = (BitVectorBase<T>) super.clone();
108         } catch (CloneNotSupportedException e) {
109           // this shouldn't happen, since we are Cloneable
110           throw new InternalError();
111         }
112         result.bits = new int[bits.length];
113         System.arraycopy(bits, 0, result.bits, 0, result.bits.length);
114         return result;
115       }

117       @Override
118       public String toString() {
119         StringBuffer buffer = new StringBuffer();
120         boolean needSeparator = false;
121         buffer.append('{');
122         int limit = length();
123         for (int i = 0; i < limit; i++) {
124           if (get(i)) {
125             if (needSeparator) {
126               buffer.append(", ");
127             } else {
128               needSeparator = true;
129             }
130             buffer.append(i);
131           }
132         }
133         buffer.append('}');
134         return buffer.toString();
135       }

137       /*
138        * @see com.ibm.wala.util.intset.IntSet#contains(int)
139        */
140       public boolean contains(int i) {
141         return get(i);
142       }

144       private static final int[][] masks = new int[][] {
145           { 0xFFFF0000 },
146           { 0xFF000000, 0x0000FF00 },
147           { 0xF0000000, 0x00F00000, 0x0000F000, 0x000000F0 },
148           { 0xC0000000, 0x0C000000, 0x00C00000, 0x000C0000, 0x0000C000, 0x00000C00, 0x000000C0, 0x0000000C },
149           { 0x80000000, 0x20000000, 0x08000000, 0x02000000, 0x00800000, 0x00200000, 0x00080000, 0x00020000, 0x00008000, 0x00002000,
150               0x00000800, 0x00000200, 0x00000080, 0x00000020, 0x00000008, 0x00000002 } };

152       public int max() {
153         int lastWord = bits.length - 1;
154     
155         while (lastWord >= 0 && bits[lastWord] == 0)
156           lastWord--;
157     
158         if (lastWord < 0)
159           return -1;
160     
161         int count = lastWord * BITS_PER_UNIT;
162     
163         int top = bits[lastWord];
164     
165         int j = 0;
166         for (int i = 0; i < masks.length; i++) {
167           if ((top & masks[i][j]) != 0) {
168             j <<= 1;
169           } else {
170             j <<= 1;
171             j++;
172           }
173         }
174     
175         return count + (31 - j);
176       }
177     
178       /**
179        * @return min j >= start s.t get(j)
180        */
181       public int nextSetBit(int start) {
182         if (start < 0) {
183           throw new IllegalArgumentException("illegal start: " + start);
184         }
185         int word = subscript(start);
186         int bit = (1 << (start & LOW_MASK));
187         while (word < bits.length) {
188           if (bits[word] != 0) {
189             do {
190               if ((bits[word] & bit) != 0)
191                 return start;
192               bit <<= 1;
193               start++;
194             } while (bit != 0);
195           } else {
196             start += (BITS_PER_UNIT - (start & LOW_MASK));
197           }
198     
199           word++;
200           bit = 1;
201         }
202     
203         return -1;
204       }

206       /**
207        * Copies the values of the bits in the specified set into this set.
208        * 
209        * @param set the bit set to copy the bits from
210        * @throws IllegalArgumentException if set is null
211        */
212       public void copyBits(BitVectorBase set) {
213         if (set == null) {
214           throw new IllegalArgumentException("set is null");
215         }
216         int setLength = set.bits.length;
217         bits = new int[setLength];
218         for (int i = setLength - 1; i >= 0;) {
219           bits[i] = set.bits[i];
220           i--;
221         }
222       }
