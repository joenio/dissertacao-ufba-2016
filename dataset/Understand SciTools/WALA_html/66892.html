36     public class MiscellaneousHacksContextSelector implements ContextSelector {
37     
38       private final Set<MethodReference> methodsToSpecialize;
39     
40       private final ContextSelector specialPolicy;
41     
42       private final ContextSelector basePolicy;
43     
44       public MiscellaneousHacksContextSelector(ContextSelector special, ContextSelector base, IClassHierarchy cha,
45           String[][] descriptors) {
46         basePolicy = base;
47         specialPolicy = special;
48         methodsToSpecialize = HashSetFactory.make();
49         for (int i = 0; i < descriptors.length; i++) {
50           String[] descr = descriptors[i];
51           switch (descr.length) {
52     
53           // loader name, loader language, classname, method name, method descr
54           case 5: {
55             ClassLoaderReference clr = cha.getScope().getLoader(Atom.findOrCreateUnicodeAtom(descr[0]));
56             IClassLoader loader = cha.getLoader(clr);
57             MethodReference ref = 
58               MethodReference.findOrCreate(
59                   TypeReference.findOrCreate(clr, TypeName.string2TypeName(descr[2])), 
60                   Atom.findOrCreateUnicodeAtom(descr[3]), 
61                   Descriptor.findOrCreateUTF8(loader.getLanguage(), descr[4]));
62     
63             if (cha.resolveMethod(ref) != null) {
64               methodsToSpecialize.add(cha.resolveMethod(ref).getReference());
65             } else {
66               methodsToSpecialize.add(ref);
67             }
68             break;
69           }
70     
71             // classname, method name, method descr
72           case 3: {
73             MethodReference ref = MethodReference.findOrCreate(TypeReference.findOrCreate(ClassLoaderReference.Application, TypeName.string2TypeName(descr[0])), Atom
74                 .findOrCreateUnicodeAtom(descr[1]), Descriptor.findOrCreateUTF8(Language.JAVA, descr[2]));
75     
76             methodsToSpecialize.add(cha.resolveMethod(ref).getReference());
77             break;
78           }
79     
80             // loader name, classname, meaning all methods of that class
81           case 2: {
82             IClass klass = cha.lookupClass(TypeReference.findOrCreate(new ClassLoaderReference(Atom.findOrCreateUnicodeAtom(descr[0]),
83                 ClassLoaderReference.Java, null), TypeName.string2TypeName(descr[1])));
84     
85             for (Iterator M = klass.getDeclaredMethods().iterator(); M.hasNext();) {
86               methodsToSpecialize.add(((IMethod) M.next()).getReference());
87             }
88     
89             break;
90           }
91     
92             // classname, meaning all methods of that class
93           case 1: {
94             IClass klass = cha.lookupClass(TypeReference.findOrCreate(ClassLoaderReference.Application, TypeName.string2TypeName(descr[0])));
95     
96             for (Iterator M = klass.getDeclaredMethods().iterator(); M.hasNext();) {
97               methodsToSpecialize.add(((IMethod) M.next()).getReference());
98             }
99     
100             break;
101           }
102     
103           default:
104             Assertions.UNREACHABLE();
105           }
106         }
107     
108         System.err.println(("hacking context selector for methods " + methodsToSpecialize));
109       }
110     
111       @Override
112       public Context getCalleeTarget(CGNode caller, CallSiteReference site, IMethod callee, InstanceKey[] receiver) {
113         if (methodsToSpecialize.contains(site.getDeclaredTarget()) || methodsToSpecialize.contains(callee.getReference())) {
114           return specialPolicy.getCalleeTarget(caller, site, callee, receiver);
115         } else {
116           return basePolicy.getCalleeTarget(caller, site, callee, receiver);
117         }
118       }
119     
120       @Override
121       public IntSet getRelevantParameters(CGNode caller, CallSiteReference site) {
122         return specialPolicy.getRelevantParameters(caller, site).union(basePolicy.getRelevantParameters(caller, site));
123       }
