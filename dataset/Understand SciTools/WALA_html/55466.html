98         CodeReader reader = getCodeReader();
99         if (reader == null) {
100           return null;
101         }
102         final Decoder d = new CTDecoder(reader);
103         try {
104           d.decode();

117       @Override
118       public int getMaxStackHeight() {
119         CodeReader reader = getCodeReader();
120         // note that Shrike returns the maximum index in the zero-indexed stack
121         // array.
122         // Instead, we want the max number of entries on the stack.
123         // So we add 1.
124         // Additionally, ShrikeBT may add additional stack entries with
125         // Constant instructions. We add an additional 1 to account for this,
126         // which seems to handle all ShrikeBT code generation patterns.
127         // TODO: ShrikeBT should have a getMaxStack method on Decoder, I think.
128         return reader.getMaxStack() + 2;
129       }

196         @Override
197         public int compareTo(Object o) {
198           if (o instanceof SourcePosition) {
199             SourcePosition p = (SourcePosition) o;
200             if (firstLine != p.getFirstLine()) {
201               return firstLine - p.getFirstLine();
202             } else if (firstCol != p.getFirstCol()) {
203               return firstCol - p.getFirstCol();
204             } else if (lastLine != p.getLastLine()) {
205               return lastLine - p.getLastLine();
206             } else if (lastCol != p.getLastCol()) {
207               return lastCol - p.getLastCol();
208             } else {
209               return 0;
210             }
211           } else {
212             return -1;
213           }
214         }
215

218           return "(" + firstLine + "," + firstCol + "-" + lastLine + "," + lastCol + ")";

224       @Override
225       protected void processDebugInfo(BytecodeInfo bcInfo) throws InvalidClassFileException {
226         CodeReader cr = getCodeReader();
227         bcInfo.lineNumberMap = LineNumberTableReader.makeBytecodeToSourceMap(cr);
228         bcInfo.localVariableMap = LocalVariableTableReader.makeVarMap(cr);
229     /** BEGIN Custom change: precise bytecode positions */
230         
231         Position param = null;
232         try {
233             param = SourcePositionTableReader.findParameterPosition(shrikeMethodIndex, cr);
234         } catch (IOException e) {
235             e.printStackTrace();
236         }
237         
238         bcInfo.paramPositionMap = new SPos[getNumberOfParameters()];
239         if (param != null) {
240           SPos paramPos = new SPos(param.firstLine, param.lastLine, param.firstCol, param.lastCol);
241           for (int i = 0; i < getNumberOfParameters(); i++) {
242             bcInfo.paramPositionMap[i] = paramPos;
243           }
244         }
245     
246         Position pos[] = null;
247         try {
248           pos = SourcePositionTableReader.makeBytecodeToPositionMap(cr);
249         } catch (IOException e) {
250           e.printStackTrace();
251         }
252         
253         if (pos == null && bcInfo.lineNumberMap != null) {
254           pos = SourcePositionTableReader.makeLineNumberToPositionMap(bcInfo.lineNumberMap);
255         }
256         
257         if (pos != null) {
258           bcInfo.positionMap = new SPos[pos.length];
259           for (int i = 0; i < pos.length; i++) {
260             Position p = pos[i];
261             bcInfo.positionMap[i] = new SPos(p.firstLine, p.lastLine, p.firstCol, p.lastCol);
262           }
263         }
264     /** END Custom change: : precise bytecode positions */
265       }

267       @Override
268       public String getLocalVariableName(int bcIndex, int localNumber){
269         int[][] map = null;
270         try {
271           map = getBCInfo().localVariableMap;
272         } catch (InvalidClassFileException e1) {
273           return null;
274         }
275     
276         if (localNumber > getMaxLocals()) {
277           throw new IllegalArgumentException("illegal local number: " + localNumber + ", method " + getDeclaringClass().getName() + 
278                   "." + getName() + " uses at most " + getMaxLocals());
279         }
280     
281         if (map == null) {
282           return null;
283         } else {
284           int[] localPairs = map[bcIndex];
285           int localIndex = localNumber * 2;
286           if (localPairs == null || localIndex >= localPairs.length) {
287             // no information about the specified local at this program point
288             return null;
289           }
290           int nameIndex = localPairs[localIndex];
291           if (nameIndex == 0) {
292             return null;
293           } else {
294             try {
295               return getClassReader().getCP().getCPUtf8(nameIndex);
296             } catch (InvalidClassFileException e) {
297               e.printStackTrace();
298               Assertions.UNREACHABLE();
299               return null;
300             }
301           }
302         }
303       }

312         try {
313           ClassReader.AttrIterator iter = new ClassReader.AttrIterator();
314           getCodeReader().initAttributeIterator(iter);
315           for (; iter.isValid(); iter.advance()) {
316             if (iter.getName().equals("LocalVariableTable")) {
317               return true;
318             }
319           }
320           return false;
321         } catch (InvalidClassFileException e) {
322           e.printStackTrace();
323           Assertions.UNREACHABLE();
324           return false;

332       private CodeReader getCodeReader() {
333         ClassReader.AttrIterator iter = new AttrIterator();
334         getClassReader().initMethodAttributeIterator(shrikeMethodIndex, iter);
335     
336         // search for the code attribute
337         CodeReader code = null;
338         try {
339           for (; iter.isValid(); iter.advance()) {
340             if (iter.getName().equals("Code")) {
341               code = new CodeReader(iter);
342               break;
343             }
344           }
345         } catch (InvalidClassFileException e) {
346           Assertions.UNREACHABLE();
347         }
348         return code;
349       }

468       /**
469        * get annotations on parameters as an array of Collections, where each array
470        * element gives the annotations on the corresponding parameter. Note that the
471        * 'this' parameter for an instance method cannot have annotations.
472        */
473       public Collection<Annotation>[] getParameterAnnotations() {
474         int numAnnotatedParams = isStatic() ? getNumberOfParameters() : getNumberOfParameters() - 1;
475         @SuppressWarnings("unchecked")
476         Collection<Annotation>[] result = new Collection[numAnnotatedParams];
477         for (int i = 0; i < result.length; i++) {
478           result[i] = HashSetFactory.make();
479         }
480         try {
481           ClassLoaderReference reference = getDeclaringClass().getClassLoader().getReference();
482           AnnotationsReader r = getAnnotationsReader(AnnotationType.RuntimeInvisibleParameterAnnotations);
483           Collection<Annotation>[] paramAnnots = Annotation.getParameterAnnotationsFromReader(r, reference);
484           if (paramAnnots != null) {
485             assert paramAnnots.length == result.length : paramAnnots.length + " != " + result.length;
486             for (int i = 0; i < result.length; i++) {
487               result[i].addAll(paramAnnots[i]);
488             }
489           }
490           r = getAnnotationsReader(AnnotationType.RuntimeVisibleParameterAnnotations);
491           paramAnnots = Annotation.getParameterAnnotationsFromReader(r, reference);
492           if (paramAnnots != null) {
493             assert paramAnnots.length == result.length;
494             for (int i = 0; i < result.length; i++) {
495               result[i].addAll(paramAnnots[i]);
496             }
497           }
498         } catch (InvalidClassFileException e) {
499           e.printStackTrace();
500         }
501         return result;
502       }
