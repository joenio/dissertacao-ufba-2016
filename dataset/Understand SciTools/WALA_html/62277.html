11     package com.ibm.wala.demandpa.alg.refinepolicy;
12     
13     import com.ibm.wala.classLoader.IField;
14     import com.ibm.wala.demandpa.alg.statemachine.StateMachine;
15     import com.ibm.wala.demandpa.flowgraph.IFlowLabel;
16     import com.ibm.wala.ipa.callgraph.propagation.PointerKey;
17     
18     /**
19      * A field refine policy that first checks with A, then delegates to B
20      */
21     public class DelegatingFieldRefinePolicy implements FieldRefinePolicy {
22     
23       private final FieldRefinePolicy A;
24     
25       private final FieldRefinePolicy B;
26     
27       public DelegatingFieldRefinePolicy(FieldRefinePolicy a, FieldRefinePolicy b) {
28         if (a == null) {
29           throw new IllegalArgumentException("null A");
30         }
31         if (b == null) {
32           throw new IllegalArgumentException("null B");
33         }
34         A = a;
35         B = b;
36       }
37     
38       @Override
39       public boolean nextPass() {
40         // careful not to short-circuit here, since nextPass() can have side-effects
41         boolean AnextPass = A.nextPass();
42         boolean BnextPass = B.nextPass();
43         return AnextPass || BnextPass;
44       }
45     
46       /**
47        * returns <code>true</code> if
48        * <code>A.shouldRefine(field) || B.shouldRefine(field)</code>. Note that
49        * if <code>A.shouldRefine(field)</code> is <code>true</code>,
50        * <code>B.shouldRefine(field)</code> is <em>not</em> called.
51        */
52       @Override
53       public boolean shouldRefine(IField field, PointerKey basePtr, PointerKey val, IFlowLabel label, StateMachine.State state) {
54         // make code explicit to avoid subtle reliance on short-circuiting
55         boolean AshouldRefine = A.shouldRefine(field, basePtr, val, label, state);
56         if (AshouldRefine) {
57           return true;
58         } else {
59           return B.shouldRefine(field, basePtr, val, label, state);
60         }
61       }
62     
63     }
