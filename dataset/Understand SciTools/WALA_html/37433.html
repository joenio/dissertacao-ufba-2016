11     package com.ibm.wala.util.intset;
12     
13     
14     /**
15      * Set of longs; not necessary mutable TODO: extract a smaller interface?
16      */
17     public interface LongSet {
18     
19       /**
20        * @param i
21        * @return true iff this set contains long i
22        */
23       public boolean contains(long i);
24     
25       /**
26        * @return true iff this set contains integer i
27        */
28       public boolean containsAny(LongSet set);
29     
30       /**
31        * This implementation must not despoil the original value of "this"
32        * 
33        * @return a new IntSet which is the intersection of this and that
34        */
35       public LongSet intersection(LongSet that);
36     
37       /**
38        * @return true iff this set is empty
39        */
40       public boolean isEmpty();
41     
42       /**
43        * @return the number of elements in this set
44        */
45       public int size();
46       
47     
48       /**
49        * @return maximum integer in this set.
50        */
51       public long max();
52     
53       /**
54        * @return true iff <code>this</code> has the same value as
55        *         <code>that</code>.
56        */
57       public boolean sameValue(LongSet that);
58     
59       /**
60        * @return true iff <code>this</code> is a subset of <code>that</code>.
61        */
62       public boolean isSubset(LongSet that);
63     
64       /**
65        * @return a perhaps more efficient iterator
66        */
67       public LongIterator longIterator();
68     
69       /**
70        * Invoke an action on each element of the Set
71        * 
72        * @param action
73        */
74       public void foreach(LongSetAction action);
75     
76       /**
77        * Invoke an action on each element of the Set, excluding elements of Set X
78        * 
79        * @param action
80        */
81       public void foreachExcluding(LongSet X, LongSetAction action);
82     
83     }
