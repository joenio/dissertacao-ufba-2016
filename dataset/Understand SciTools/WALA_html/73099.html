11     package com.ibm.wala.fixedpoint.impl;
12     
13     import com.ibm.wala.fixpoint.AbstractOperator;
14     import com.ibm.wala.fixpoint.AbstractStatement;
15     import com.ibm.wala.fixpoint.IVariable;
16     
17     /**
18      * Represents a single step in an iterative solver
19      */
20     public abstract class GeneralStatement<T extends IVariable<?>> extends AbstractStatement<T, AbstractOperator<T>> {
21     
22       protected final T lhs;
23     
24       protected final T[] rhs;
25     
26       private final int hashCode;
27     
28       private final AbstractOperator<T> operator;
29     
30       /**
31        * Evaluate this equation, setting a new value for the left-hand side.
32        * 
33        * @return true if the lhs value changed. false otherwise
34        */
35       @Override
36       public byte evaluate() {
37         return operator.evaluate(lhs, rhs);
38       }
39     
40       /**
41        * Return the left-hand side of this equation.
42        * 
43        * @return the lattice cell this equation computes
44        */
45       @Override
46       public T getLHS() {
47         return lhs;
48       }
49     
50       /**
51        * Does this equation contain an appearance of a given cell?
52        * 
53        * Note: this uses reference equality, assuming that the variables are canonical! This is fragile. TODO: Address it perhaps, but
54        * be careful not to sacrifice efficiency.
55        * 
56        * @param cell the cell in question
57        * @return true or false
58        */
59       @Override
60       public boolean hasVariable(T cell) {
61         if (lhs == cell) {
62           return true;
63         }
64         for (int i = 0; i < rhs.length; i++) {
65           if (rhs[i] == cell)
66             return true;
67         }
68         return false;
69       }
70     
71       /**
72        * Constructor for case of zero operands on the right-hand side.
73        * 
74        * @param lhs the lattice cell set by this equation
75        * @param operator the equation operator
76        */
77       public GeneralStatement(T lhs, AbstractOperator<T> operator) {
78         super();
79         if (operator == null) {
80           throw new IllegalArgumentException("null operator");
81         }
82         this.operator = operator;
83         this.lhs = lhs;
84         this.rhs = null;
85         this.hashCode = makeHashCode();
86       }
87     
88       /**
89        * Constructor for case of two operands on the right-hand side.
90        * 
91        * @param lhs the lattice cell set by this equation
92        * @param operator the equation operator
93        * @param op1 the first operand on the rhs
94        * @param op2 the second operand on the rhs
95        */
96       public GeneralStatement(T lhs, AbstractOperator<T> operator, T op1, T op2) {
97         super();
98         if (operator == null) {
99           throw new IllegalArgumentException("null operator");
100         }
101         this.operator = operator;
102         this.lhs = lhs;
103         rhs = makeRHS(2);
104         rhs[0] = op1;
105         rhs[1] = op2;
106         this.hashCode = makeHashCode();
107       }
108     
109       /**
110        * Constructor for case of three operands on the right-hand side.
111        * 
112        * @param lhs the lattice cell set by this equation
113        * @param operator the equation operator
114        * @param op1 the first operand on the rhs
115        * @param op2 the second operand on the rhs
116        * @param op3 the third operand on the rhs
117        */
118       public GeneralStatement(T lhs, AbstractOperator<T> operator, T op1, T op2, T op3) {
119         super();
120         if (operator == null) {
121           throw new IllegalArgumentException("null operator");
122         }
123         this.operator = operator;
124         rhs = makeRHS(3);
125         this.lhs = lhs;
126         rhs[0] = op1;
127         rhs[1] = op2;
128         rhs[2] = op3;
129         this.hashCode = makeHashCode();
130       }
131     
132       /**
133        * Constructor for case of more than three operands on the right-hand side.
134        * 
135        * @param lhs the lattice cell set by this equation
136        * @param operator the equation operator
137        * @param rhs the operands of the right-hand side in order
138        * @throws IllegalArgumentException if rhs is null
139        */
140       public GeneralStatement(T lhs, AbstractOperator<T> operator, T[] rhs) {
141         super();
142         if (operator ==  null) {
143           throw new IllegalArgumentException("null operator");
144         }
145         if (rhs == null) {
146           throw new IllegalArgumentException("rhs is null");
147         }
148         this.operator = operator;
149         this.lhs = lhs;
150         this.rhs = rhs.clone();
151         this.hashCode = makeHashCode();
152       }
153     
154       /**
155        * TODO: use a better hash code?
156        */
157       private final static int[] primes = { 331, 337, 347, 1277 };
158     
159       private int makeHashCode() {
160         int result = operator.hashCode();
161         if (lhs != null)
162           result += lhs.hashCode() * primes[0];
163         for (int i = 0; i < Math.min(rhs.length, 2); i++) {
164           if (rhs[i] != null) {
165             result += primes[i + 1] * rhs[i].hashCode();
166           }
167         }
168         return result;
169       }
170     
171       protected abstract T[] makeRHS(int size);
172       
173       @Override
174       public int hashCode() {
175         return hashCode;
176       }
177     
178       @Override
179       public boolean equals(Object o) {
180         if (o == null) {
181           return false;
182         }
183         if (getClass().equals(o.getClass())) {
184           GeneralStatement<?> other = (GeneralStatement<?>) o;
185           if (hashCode == other.hashCode) {
186             if (lhs == null || other.lhs == null) {
187               if (other.lhs != lhs) {
188                 return false;
189               }
190             } else if (!lhs.equals(other.lhs)) {
191               return false;
192             }
193             if (operator.equals(other.operator) && rhs.length == other.rhs.length) {
194               for (int i = 0; i < rhs.length; i++) {
195                 if (rhs[i] == null || other.rhs[i] == null) {
196                   if (other.rhs[i] != rhs[i]) {
197                     return false;
198                   }
199                 } else if (!rhs[i].equals(other.rhs[i])) {
200                   return false;
201                 }
202               }
203               return true;
204             }
205           }
206         }
207         return false;
208       }
209     
210       @Override
211       public AbstractOperator<T> getOperator() {
212         return operator;
213       }
214     
215       @Override
216       public T[] getRHS() {
217         return rhs;
218       }
219     }
