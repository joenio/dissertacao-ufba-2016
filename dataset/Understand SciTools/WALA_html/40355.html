43     /**
44      * Computes interprocedural reaching definitions for static fields in a context-insensitive manner.
45      */
46     public class ContextInsensitiveReachingDefs {
47     
48       /**
49        * the exploded interprocedural control-flow graph on which to compute the analysis
50        */
51       private final ExplodedInterproceduralCFG icfg;
52     
53       /**
54        * maps call graph node and instruction index of putstatic instructions to more compact numbering for bitvectors
55        */
56       private final OrdinalSetMapping<Pair<CGNode, Integer>> putInstrNumbering;
57     
58       /**
59        * for resolving field references in putstatic instructions
60        */
61       private final IClassHierarchy cha;
62     
63       /**
64        * maps each static field to the numbers of the statements (in {@link #putInstrNumbering}) that define it; used for kills in flow
65        * functions
66        */
67       private final Map<IField, BitVector> staticField2DefStatements = HashMapFactory.make();
68     
69       private static final boolean VERBOSE = true;
70     
71       public ContextInsensitiveReachingDefs(ExplodedInterproceduralCFG icfg, IClassHierarchy cha) {
72         this.icfg = icfg;
73         this.cha = cha;
74         this.putInstrNumbering = numberPutStatics();
75       }
76     
77       /**
78        * generate a numbering of the putstatic instructions
79        */
80       @SuppressWarnings("unchecked")
81       private OrdinalSetMapping<Pair<CGNode, Integer>> numberPutStatics() {
82         ArrayList<Pair<CGNode, Integer>> putInstrs = new ArrayList<Pair<CGNode, Integer>>();
83         for (CGNode node : icfg.getCallGraph()) {
84           IR ir = node.getIR();
85           if (ir == null) {
86             continue;
87           }
88           SSAInstruction[] instructions = ir.getInstructions();
89           for (int i = 0; i < instructions.length; i++) {
90             SSAInstruction instruction = instructions[i];
91             if (instruction instanceof SSAPutInstruction && ((SSAPutInstruction) instruction).isStatic()) {
92               SSAPutInstruction putInstr = (SSAPutInstruction) instruction;
93               // instrNum is the number that will be assigned to this putstatic
94               int instrNum = putInstrs.size();
95               putInstrs.add(Pair.make(node, i));
96               // also update the mapping of static fields to def'ing statements
97               IField field = cha.resolveField(putInstr.getDeclaredField());
98               assert field != null;
99               BitVector bv = staticField2DefStatements.get(field);
100               if (bv == null) {
101                 bv = new BitVector();
102                 staticField2DefStatements.put(field, bv);
103               }
104               bv.set(instrNum);
105             }
106           }
107         }
108         return new ObjectArrayMapping<Pair<CGNode, Integer>>(putInstrs.toArray(new Pair[putInstrs.size()]));
109       }
110     
111       private class TransferFunctions implements ITransferFunctionProvider<BasicBlockInContext<IExplodedBasicBlock>, BitVectorVariable> {
112     
113         /**
114          * our meet operator is set union
115          */
116         @Override
117         public AbstractMeetOperator<BitVectorVariable> getMeetOperator() {
118           return BitVectorUnion.instance();
119         }
120     
121         @Override
122         public UnaryOperator<BitVectorVariable> getNodeTransferFunction(BasicBlockInContext<IExplodedBasicBlock> node) {
123           IExplodedBasicBlock ebb = node.getDelegate();
124           SSAInstruction instruction = ebb.getInstruction();
125           int instructionIndex = ebb.getFirstInstructionIndex();
126           CGNode cgNode = node.getNode();
127           if (instruction instanceof SSAPutInstruction && ((SSAPutInstruction) instruction).isStatic()) {
128             // kill all defs of the same static field, and gen this instruction
129             final SSAPutInstruction putInstr = (SSAPutInstruction) instruction;
130             final IField field = cha.resolveField(putInstr.getDeclaredField());
131             assert field != null;
132             BitVector kill = staticField2DefStatements.get(field);
133             BitVector gen = new BitVector();
134             gen.set(putInstrNumbering.getMappedIndex(Pair.make(cgNode, instructionIndex)));
135             return new BitVectorKillGen(kill, gen);
136           } else {
137             // identity function for non-putstatic instructions
138             return BitVectorIdentity.instance();
139           }
140         }
141     
142         /**
143          * here we need an edge transfer function for call-to-return edges (see
144          * {@link #getEdgeTransferFunction(BasicBlockInContext, BasicBlockInContext)})
145          */
146         @Override
147         public boolean hasEdgeTransferFunctions() {
148           return true;
149         }
150     
151         @Override
152         public boolean hasNodeTransferFunctions() {
153           return true;
154         }
155     
156         /**
157          * for direct call-to-return edges at a call site, the edge transfer function will kill all facts, since we only want to
158          * consider facts that arise from going through the callee
159          */
160         @Override
161         public UnaryOperator<BitVectorVariable> getEdgeTransferFunction(BasicBlockInContext<IExplodedBasicBlock> src,
162             BasicBlockInContext<IExplodedBasicBlock> dst) {
163           if (isCallToReturnEdge(src, dst)) {
164             return BitVectorKillAll.instance();
165           } else {
166             return BitVectorIdentity.instance();
167           }
168         }
169     
170         private boolean isCallToReturnEdge(BasicBlockInContext<IExplodedBasicBlock> src, BasicBlockInContext<IExplodedBasicBlock> dst) {
171           SSAInstruction srcInst = src.getDelegate().getInstruction();
172           return srcInst instanceof SSAAbstractInvokeInstruction && src.getNode().equals(dst.getNode());
173         }
174     
175       }
176     
177       /**
178        * run the analysis
179        * 
180        * @return the solver used for the analysis, which contains the analysis result
181        */
182       public BitVectorSolver<BasicBlockInContext<IExplodedBasicBlock>> analyze() {
183         // the framework describes the dataflow problem, in particular the underlying graph and the transfer functions
184         BitVectorFramework<BasicBlockInContext<IExplodedBasicBlock>, Pair<CGNode, Integer>> framework = new BitVectorFramework<BasicBlockInContext<IExplodedBasicBlock>, Pair<CGNode, Integer>>(
185             icfg, new TransferFunctions(), putInstrNumbering);
186         BitVectorSolver<BasicBlockInContext<IExplodedBasicBlock>> solver = new BitVectorSolver<BasicBlockInContext<IExplodedBasicBlock>>(
187             framework);
188         try {
189           solver.solve(null);
190         } catch (CancelException e) {
191           // this shouldn't happen
192           assert false;
193         }
194         if (VERBOSE) {
195           for (BasicBlockInContext<IExplodedBasicBlock> ebb : icfg) {
196             System.out.println(ebb);
197             System.out.println(ebb.getDelegate().getInstruction());
198             System.out.println(solver.getIn(ebb));
199             System.out.println(solver.getOut(ebb));
200           }
201         }
202         return solver;
203       }
204     
205       /**
206        * gets putstatic instruction corresponding to some fact number from a bitvector in the analysis result
207        */
208       public Pair<CGNode, Integer> getNodeAndInstrForNumber(int num) {
209         return putInstrNumbering.getMappedObject(num);
210       }
