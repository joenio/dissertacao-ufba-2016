11     package com.ibm.wala.util.graph;
12     
13     import java.util.Iterator;
14     
15     import com.ibm.wala.util.graph.impl.NumberedNodeIterator;
16     import com.ibm.wala.util.intset.IntSet;
17     
18     /**
19      * Basic functionality for a graph that delegates node and edge management, and tracks node numbers
20      */
21     public abstract class AbstractNumberedGraph<T> extends AbstractGraph<T> implements NumberedGraph<T> {
22     
23       /**
24        * @return the object which manages nodes in the graph
25        */
26       @Override
27       protected abstract NumberedNodeManager<T> getNodeManager();
28     
29       /**
30        * @return the object which manages edges in the graph
31        */
32       @Override
33       protected abstract NumberedEdgeManager<T> getEdgeManager();
34     
35       /*
36        * @see com.ibm.wala.util.graph.NumberedNodeManager#getMaxNumber()
37        */
38       @Override
39       public int getMaxNumber() {
40         return getNodeManager().getMaxNumber();
41       }
42     
43       /*
44        * @see com.ibm.wala.util.graph.NumberedNodeManager#getNode(int)
45        */
46       @Override
47       public T getNode(int number) {
48         return getNodeManager().getNode(number);
49       }
50     
51       /*
52        * @see com.ibm.wala.util.graph.NumberedNodeManager#getNumber(com.ibm.wala.util.graph.Node)
53        */
54       @Override
55       public int getNumber(T N) {
56         if (N == null) {
57           throw new IllegalArgumentException("N cannot be null");
58         }
59         return getNodeManager().getNumber(N);
60       }
61     
62       /*
63        * @see com.ibm.wala.util.graph.NumberedNodeManager#iterateNodes(com.ibm.wala.util.intset.IntSet)
64        */
65       @Override
66       public Iterator<T> iterateNodes(final IntSet s) {
67         return new NumberedNodeIterator<T>(s, this);
68       }
69     
70       /* 
71        * @see com.ibm.wala.util.graph.NumberedEdgeManager#getPredNodeNumbers(java.lang.Object)
72        */
73       @Override
74       public IntSet getPredNodeNumbers(T node) throws IllegalArgumentException {
75         assert getEdgeManager() != null;
76         return getEdgeManager().getPredNodeNumbers(node);
77       }
78     
79       /* 
80        * @see com.ibm.wala.util.graph.NumberedEdgeManager#getSuccNodeNumbers(java.lang.Object)
81        */
82       @Override
83       public IntSet getSuccNodeNumbers(T node) throws IllegalArgumentException {
84         return getEdgeManager().getSuccNodeNumbers(node);
85       }
86     }
