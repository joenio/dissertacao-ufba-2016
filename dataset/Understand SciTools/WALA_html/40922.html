11     package com.ibm.wala.shrikeCT;
12     
13     /**
14      * This class helps emit LineNumberTable attributes.
15      */
16     public final class LineNumberTableWriter extends ClassWriter.Element {
17       final private int attrID;
18     
19       private int[] rawTable = emptyTable;
20     
21       private static final int[] emptyTable = new int[0];
22     
23       /**
24        * Build an empty LineNumberTable.
25        * 
26        * @throws IllegalArgumentException if w is null
27        */
28       public LineNumberTableWriter(ClassWriter w) {
29         if (w == null) {
30           throw new IllegalArgumentException("w is null");
31         }
32         attrID = w.addCPUtf8("LineNumberTable");
33       }
34     
35       /**
36        * Set the raw table entries. Consider calling LineNumberTableWriter.makeRawTable to build the raw entries.
37        * 
38        * @param table a flattened sequence of (startPC, lineNumber) pairs
39        */
40       public void setRawTable(int[] table) {
41         if (table == null) {
42           table = emptyTable;
43         }
44     
45         if (table.length % 2 != 0) {
46           throw new IllegalArgumentException("Line number table has bad length: " + table.length);
47         }
48         if (table.length / 2 > 0xFFFF) {
49           throw new IllegalArgumentException("Too many line number table entries: " + table.length / 2);
50         }
51         for (int i = 0; i < table.length; i++) {
52           int v = table[i];
53           if (v < 0 || v > 0xFFFF) {
54             throw new IllegalArgumentException("Bad line number table entry at " + i + ": " + v);
55           }
56         }
57     
58         rawTable = table;
59       }
60     
61       @Override
62       public int getSize() {
63         return 8 + rawTable.length * 2;
64       }
65     
66       @Override
67       public int copyInto(byte[] buf, int offset) throws IllegalArgumentException {
68         ClassWriter.setUShort(buf, offset, attrID);
69         ClassWriter.setInt(buf, offset + 2, 2 + rawTable.length * 2);
70         ClassWriter.setUShort(buf, offset + 6, rawTable.length / 2);
71         offset += 8;
72         for (int i = 0; i < rawTable.length; i++) {
73           ClassWriter.setUShort(buf, offset, rawTable[i]);
74           offset += 2;
75         }
76     
77         return offset;
78       }
79     
80       /**
81        * @param newLineMap an array indexed by bytecode offset, mapping each bytecode offset to its line number (or 0 if there is no
82        *          line or it's not known)
83        * @return the line numbers in "raw" format, a flattened sequence of (startPC, lineNumber) pairs
84        * @throws IllegalArgumentException if newLineMap == null
85        */
86       public static int[] makeRawTable(int[] newLineMap) throws IllegalArgumentException {
87         if (newLineMap == null) {
88           throw new IllegalArgumentException("newLineMap == null");
89         }
90         int rawCount = 0;
91         int last = -1;
92         for (int i = 0; i < newLineMap.length; i++) {
93           int next = newLineMap[i];
94           if (next != last) {
95             rawCount++;
96           }
97         }
98         int[] rawTable = new int[rawCount * 2];
99         last = -1;
100         int index = 0;
101         for (int i = 0; i < newLineMap.length; i++) {
102           int next = newLineMap[i];
103           if (next != last) {
104             rawTable[index] = i;
105             rawTable[index + 1] = next;
106             index += 2;
107           }
108         }
109         return rawTable;
110       }
111     }
