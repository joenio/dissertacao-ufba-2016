41     public class BootstrapInstrumentor {
42       final private PrintWriter w;
43     
44       private int idx;
45       
46       /**
47        * Get ready to print a class to the given output stream.
48        */
49       public BootstrapInstrumentor(PrintWriter w) {
50         this.w = w;
51       }
52     
53       public static void main(String[] args) throws Exception {
54         PrintWriter w = new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out)));
55         BootstrapInstrumentor p = new BootstrapInstrumentor(w);
56         p.doit(args);
57       }
58     
59       public void doit(String[] args) throws Exception {
60         OfflineInstrumenter oi = new OfflineInstrumenter(true);
61         oi.parseStandardArgs(args);
62        
63         oi.setPassUnmodifiedClasses(true);
64         
65         ClassInstrumenter ci;
66         oi.beginTraversal();
67         while ((ci = oi.nextClass()) != null) {
68           try {
69             idx = 0;
70             Set<MethodData> bss = doClass(ci);
71             ClassWriter cw = ci.emitClass();
72             for(MethodData md : bss) {
73               CTUtils.compileAndAddMethodToClassWriter(md, cw, null);
74             }
75             oi.outputModifiedClass(ci, cw);
76           } finally {
77             w.flush();
78           }
79         }
80     
81         oi.close();
82       }

84       private Set<MethodData> dumpAttributes(ClassInstrumenter ci, int i, ClassReader.AttrIterator attrs) throws InvalidClassFileException,
85           InvalidBytecodeException, IOException, ClassNotFoundException, NoSuchMethodException, SecurityException, IllegalAccessException, IllegalArgumentException, InvocationTargetException, NoSuchFieldException {
86         Set<MethodData> result = HashSetFactory.make();
87         ClassReader cr = ci.getReader();
88         for (; attrs.isValid(); attrs.advance()) {
89           String name = attrs.getName();
90           if (name.equals("Code")) {
91             CodeReader code = new CodeReader(attrs);
92     
93             CTDecoder decoder = new CTDecoder(code);
94             decoder.decode();
95             ConstantPoolReader cpr = decoder.getConstantPool();
96             IInstruction[] origInsts = decoder.getInstructions();
97             for(IInstruction inst : origInsts) {
98               if (inst instanceof InvokeDynamicInstruction) {
99                 InvokeDynamicInstruction x = (InvokeDynamicInstruction) inst;
100                 BootstrapMethod m = x.getBootstrap();
101     
102                 IInstruction insts[] = new IInstruction[ m.callArgumentCount() + 8];
103                 int arg = 0;
104     
105                 insts[arg++] = InvokeInstruction.make("()Ljava/lang/invoke/MethodHandles$Lookup;", "java/lang/invoke/MethodHandles", "lookup", Dispatch.STATIC);
106     
107                 insts[arg++] = ConstantInstruction.makeString(x.getMethodName());
108                 
109                 insts[arg++] = ConstantInstruction.makeString(x.getMethodSignature());
110                 insts[arg++] = ConstantInstruction.makeClass(cr.getName());
111                 insts[arg++] = InvokeInstruction.make("()Ljava/lang/ClassLoader;", "java/lang/Class", "getClassLoader", Dispatch.VIRTUAL);
112                 insts[arg++] = InvokeInstruction.make("(Ljava/lang/String;Ljava/lang/ClassLoader;)Ljava/lang/invoke/MethodType;", "java/lang/invoke/MethodType", "fromMethodDescriptorString", Dispatch.STATIC);
113                 
114                 for(int an = 0; an < m.callArgumentCount(); an++) {
115                   insts[arg++] = ConstantInstruction.make(cpr, m.callArgumentIndex(an));
116                 }
117                 
118                 insts[arg++] = InvokeInstruction.make(m.methodType(), m.methodClass(), m.methodName(), ((InvokeDynamicInstruction) inst).getInvocationCode());
119                 insts[arg++] = ReturnInstruction.make("Ljava/lang/invoke/CallSite;");
120                 
121                 result.add(MethodData.makeWithDefaultHandlersAndInstToBytecodes(Constants.ACC_PUBLIC|Constants.ACC_STATIC, cr.getName(), "bs" + (idx++), "()Ljava/lang/invoke/CallSite;", insts));
122               }
123             }
124           }
125         }
126         
127         return result;
128       }

132       /**
133        * Print a class.
134        * @throws InvocationTargetException 
135        * @throws IllegalAccessException 
136        * @throws SecurityException 
137        * @throws NoSuchMethodException 
138        * @throws ClassNotFoundException 
139        * 
140        * @throws IllegalArgumentException if cr is null
141        * @throws NoSuchFieldException 
142        */
143       public Set<MethodData> doClass(final ClassInstrumenter ci) throws InvalidClassFileException, InvalidBytecodeException, IOException, ClassNotFoundException, NoSuchMethodException, SecurityException, IllegalAccessException, IllegalArgumentException, InvocationTargetException, NoSuchFieldException {
144         ClassReader cr = ci.getReader();
145         ClassReader.AttrIterator attrs = new ClassReader.AttrIterator();
146         cr.initClassAttributeIterator(attrs);
147         int methodCount = cr.getMethodCount();
148         
149         Set<MethodData> result = HashSetFactory.make();
150         for (int i = 0; i < methodCount; i++) {
151           cr.initMethodAttributeIterator(i, attrs);
152           result.addAll(dumpAttributes(ci, i, attrs));
153         }
154         
155         return result;
156       }
